88void Oscilloscope::setActive( bool _active )
89{
90 m_active = _active;
91 if( m_active )
92 {
93 connect( getGUI()->mainWindow(),
94 SIGNAL(periodicUpdate()),
95 this, SLOT(update()));
96 connect( Engine::audioEngine(),
97 SIGNAL(nextAudioBuffer(const lmms::SampleFrame*)),
98 this, SLOT(updateAudioBuffer(const lmms::SampleFrame*)));
99 }
100 else
101 {
102 disconnect( getGUI()->mainWindow(),
103 SIGNAL(periodicUpdate()),
104 this, SLOT(update()));
105 disconnect( Engine::audioEngine(),
106 SIGNAL(nextAudioBuffer(const lmms::SampleFrame*)),
107 this, SLOT(updateAudioBuffer(const lmms::SampleFrame*)));
108 // we have to update (remove last waves),
109 // because timer doesn't do that anymore
110 update();
111 }
112}

156void Oscilloscope::paintEvent( QPaintEvent * )
157{
158 QPainter p( this );
159
160 p.drawPixmap( 0, 0, m_background );
161
162 if( m_active && !Engine::getSong()->isExporting() )
163 {
164 AudioEngine const * audioEngine = Engine::audioEngine();
165
166 float masterOutput = audioEngine->masterGain();
167
168 const fpp_t frames = audioEngine->framesPerPeriod();
169 SampleFrame peakValues = getAbsPeakValues(m_buffer, frames);
170
171 auto const leftChannelClips = clips(peakValues.left() * masterOutput);
172 auto const rightChannelClips = clips(peakValues.right() * masterOutput);
173
174 p.setRenderHint( QPainter::Antialiasing );
175
176 // now draw all that stuff
177 int w = width() - 4;
178 const qreal xd = static_cast<qreal>(w) / frames;
179 const qreal half_h = -(height() - 6) / 3.0 * static_cast<qreal>(masterOutput) - 1;
180 int x_base = 2;
181 const qreal y_base = height() / 2 - 0.5;
182
183 qreal const width = 0.7;
184 for( ch_cnt_t ch = 0; ch < DEFAULT_CHANNELS; ++ch )
185 {
186 QColor color = ch == 0 ? (leftChannelClips ? clippingColor() : leftChannelColor()) : // Check left channel
187 ch == 1 ? (rightChannelClips ? clippingColor() : rightChannelColor()) : // Check right channel
188 otherChannelsColor(); // Any other channel
189 p.setPen(QPen(color, width));
190
191 for (auto frame = std::size_t{0}; frame < frames; ++frame)
192 {
193 sample_t const clippedSample = AudioEngine::clip(m_buffer[frame][ch]);
194 m_points[frame] = QPointF(
195 x_base + static_cast<qreal>(frame) * xd,
196 y_base + ( static_cast<qreal>(clippedSample) * half_h ) );
197 }
198 p.drawPolyline( m_points, frames );
199 }
200 }
201 else
202 {
203 p.setPen( QColor( 192, 192, 192 ) );
204 p.setFont(adjustedToPixelSize(p.font(), DEFAULT_FONT_SIZE));
205 p.drawText( 6, height()-5, tr( "Click to enable" ) );
206 }
207}
