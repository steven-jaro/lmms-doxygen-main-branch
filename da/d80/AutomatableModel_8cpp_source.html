106void AutomatableModel::saveSettings( QDomDocument& doc, QDomElement& element, const QString& name )
107{
108 bool mustQuote = mustQuoteName(name);
109
110 if( isAutomated() || m_scaleType != ScaleType::Linear )
111 {
112 // automation needs tuple of data (name, id, value)
113 // scale type also needs an extra value
114 // => it must be appended as a node
115
116 QDomElement me = doc.createElement( mustQuote ? QString("automatablemodel") : name );
117 me.setAttribute( "id", ProjectJournal::idToSave( id() ) );
118 me.setAttribute( "value", m_value );
119 me.setAttribute( "scale_type", m_scaleType == ScaleType::Logarithmic ? "log" : "linear" );
120 if(mustQuote) {
121 me.setAttribute( "nodename", name );
122 }
123 element.appendChild( me );
124 }
125 else
126 {
127 if(mustQuote)
128 {
129 QDomElement me = doc.createElement( "automatablemodel" );
130 me.setAttribute( "nodename", name );
131 me.setAttribute( "value", m_value );
132 element.appendChild( me );
133 }
134 else
135 {
136 // non automation, linear scale (default), can be saved as attribute
137 element.setAttribute( name, m_value );
138 }
139 }
140
141 // Skip saving MIDI connections if we're saving project and
142 // the discardMIDIConnections option is true.
143 auto controllerType = m_controllerConnection
144 ? m_controllerConnection->getController()->type()
145 : Controller::ControllerType::Dummy;
146 bool skipMidiController = Engine::getSong()->isSavingProject()
147 && Engine::getSong()->getSaveOptions().discardMIDIConnections.value();
148 if (m_controllerConnection && controllerType != Controller::ControllerType::Dummy
149 && !(skipMidiController && controllerType == Controller::ControllerType::Midi))
150 {
151 QDomElement controllerElement;
152
153 // get "connection" element (and create it if needed)
154 QDomNode node = element.namedItem( "connection" );
155 if( node.isElement() )
156 {
157 controllerElement = node.toElement();
158 }
159 else
160 {
161 controllerElement = doc.createElement( "connection" );
162 element.appendChild( controllerElement );
163 }
164
165 bool mustQuote = mustQuoteName(name);
166 QString elementName = mustQuote ? "controllerconnection"
167 : name;
168
169 QDomElement element = doc.createElement( elementName );
170 if(mustQuote)
171 element.setAttribute( "nodename", name );
172 m_controllerConnection->saveSettings( doc, element );
173
174 controllerElement.appendChild( element );
175 }
176}

181void AutomatableModel::loadSettings( const QDomElement& element, const QString& name )
182{
183 // compat code
184 QDomNode node = element.namedItem( AutomationClip::classNodeName() );
185 if( node.isElement() )
186 {
187 node = node.namedItem( name );
188 if( node.isElement() )
189 {
190 AutomationClip * p = AutomationClip::globalAutomationClip( this );
191 p->loadSettings( node.toElement() );
192 setValue( p->valueAt( 0 ) );
193 // in older projects we sometimes have odd automations
194 // with just one value in - eliminate if necessary
195 if( !p->hasAutomation() )
196 {
197 delete p;
198 }
199 return;
200 }
201 // logscales were not existing at this point of time
202 // so they can be ignored
203 }
204
205 QDomNode connectionNode = element.namedItem( "connection" );
206 // reads controller connection
207 if( connectionNode.isElement() )
208 {
209 QDomNode thisConnection = connectionNode.toElement().namedItem( name );
210 if( !thisConnection.isElement() )
211 {
212 thisConnection = connectionNode.toElement().namedItem( "controllerconnection" );
213 QDomElement tcElement = thisConnection.toElement();
214 // sanity check
215 if( tcElement.isNull() || tcElement.attribute( "nodename" ) != name )
216 {
217 // no, that wasn't it, act as if we never found one
218 thisConnection.clear();
219 }
220 }
221 if( thisConnection.isElement() )
222 {
223 setControllerConnection(new ControllerConnection(nullptr));
224 m_controllerConnection->loadSettings( thisConnection.toElement() );
225 //m_controllerConnection->setTargetName( displayName() );
226 }
227 }
228
229 // models can be stored as elements (port00) or attributes (port10):
230 // <ladspacontrols port10="4.41">
231 // <port00 value="4.41" id="4249278"/>
232 // </ladspacontrols>
233 // element => there is automation data, or scaletype information
234
235 node = element.namedItem( name ); // maybe we have luck?
236
237 // either: no node with name "name" found
238 // => look for nodes with attribute name="nodename"
239 // or: element with namedItem() "name" was found, but it's real nodename
240 // is given as attribute and does not match
241 // => look for the right node
242 if(node.isNull() ||
243 ( node.isElement() &&
244 node.toElement().hasAttribute("nodename") &&
245 node.toElement().attribute("nodename") != name))
246 {
247 for(QDomElement othernode = element.firstChildElement();
248 !othernode.isNull();
249 othernode = othernode.nextSiblingElement())
250 {
251 if((!othernode.hasAttribute("nodename") &&
252 othernode.nodeName() == name) ||
253 othernode.attribute("nodename") == name)
254 {
255 node = othernode;
256 break;
257 }
258 }
259 }
260 if( node.isElement() )
261 {
262 QDomElement nodeElement = node.toElement();
263 changeID( nodeElement.attribute( "id" ).toInt() );
264 setValue( LocaleHelper::toFloat( nodeElement.attribute( "value" ) ) );
265 if( nodeElement.hasAttribute( "scale_type" ) )
266 {
267 if( nodeElement.attribute( "scale_type" ) == "linear" )
268 {
269 setScaleType( ScaleType::Linear );
270 }
271 else if( nodeElement.attribute( "scale_type" ) == "log" )
272 {
273 setScaleType( ScaleType::Logarithmic );
274 }
275 }
276 }
277 else
278 {
279
280 setScaleType( ScaleType::Linear );
281
282 if( element.hasAttribute( name ) )
283 // attribute => read the element's value from the attribute list
284 {
285 setInitValue( LocaleHelper::toFloat( element.attribute( name ) ) );
286 }
287 else
288 {
289 reset();
290 }
291 }
292}

297void AutomatableModel::setValue( const float value )
298{
299 m_oldValue = m_value;
300 ++m_setValueDepth;
301 const float old_val = m_value;
302
303 m_value = fittedValue( value );
304 if( old_val != m_value )
305 {
306 // add changes to history so user can undo it
307 addJournalCheckPoint();
308
309 // notify linked models
310 for (const auto& linkedModel : m_linkedModels)
311 {
312 if (linkedModel->m_setValueDepth < 1 && linkedModel->fittedValue(value) != linkedModel->m_value)
313 {
314 bool journalling = linkedModel->testAndSetJournalling(isJournalling());
315 linkedModel->setValue(value);
316 linkedModel->setJournalling(journalling);
317 }
318 }
319 m_valueChanged = true;
320 emit dataChanged();
321 }
322 else
323 {
324 emit dataUnchanged();
325 }
326 --m_setValueDepth;
327}

399{

401 {

434float AutomatableModel::fittedValue( float value ) const
435{
436 value = std::clamp(value, m_minValue, m_maxValue);
437
438 if( m_step != 0 && m_hasStrictStepSize )
439 {
440 value = nearbyintf( value / m_step ) * m_step;
441 }
442
443 roundAt( value, m_maxValue );
444 roundAt( value, m_minValue );
445 roundAt( value, 0.0f );
446
447 if( value < m_minValue )
448 {
449 return m_minValue;
450 }
451 else if( value > m_maxValue )
452 {
453 return m_maxValue;
454 }
455
456 return value;
457}

495void AutomatableModel::linkModels( AutomatableModel* model1, AutomatableModel* model2 )
496{
497 auto model1ContainsModel2 = std::find(model1->m_linkedModels.begin(), model1->m_linkedModels.end(), model2) != model1->m_linkedModels.end();
498 if (!model1ContainsModel2 && model1 != model2)
499 {
500 // copy data
501 model1->m_value = model2->m_value;
502 if (model1->valueBuffer() && model2->valueBuffer())
503 {
504 std::copy_n(model2->valueBuffer()->data(),
505 model1->valueBuffer()->length(),
506 model1->valueBuffer()->data());
507 }
508 // send dataChanged() before linking (because linking will
509 // connect the two dataChanged() signals)
510 emit model1->dataChanged();
511 // finally: link the models
512 model1->linkModel( model2 );
513 model2->linkModel( model1 );
514 }

556float AutomatableModel::controllerValue( int frameOffset ) const
557{
558 if( m_controllerConnection )
559 {
560 float v = 0;
561 switch(m_scaleType)
562 {
563 case ScaleType::Linear:
564 v = minValue<float>() + ( range() * controllerConnection()->currentValue( frameOffset ) );
565 break;
566 case ScaleType::Logarithmic:
567 v = logToLinearScale(
568 controllerConnection()->currentValue( frameOffset ));
569 break;
570 default:
571 qFatal("AutomatableModel::controllerValue(int)"
572 "lacks implementation for a scale type");
573 break;
574 }
575 if (approximatelyEqual(m_step, 1) && m_hasStrictStepSize)
576 {
577 return std::round(v);
578 }
579 return v;
580 }
581
582 AutomatableModel* lm = m_linkedModels.front();
583 if (lm->controllerConnection() && lm->useControllerValue())
584 {
585 return fittedValue( lm->controllerValue( frameOffset ) );
586 }
587
588 return fittedValue( lm->m_value );
589}

592ValueBuffer * AutomatableModel::valueBuffer()
593{
594 QMutexLocker m( &m_valueBufferMutex );
595 // if we've already calculated the valuebuffer this period, return the cached buffer
596 if( m_lastUpdatedPeriod == s_periodCounter )
597 {
598 return m_hasSampleExactData
599 ? &m_valueBuffer
600 : nullptr;
601 }
602
603 float val = m_value; // make sure our m_value doesn't change midway
604
605 if (m_controllerConnection && m_useControllerValue && m_controllerConnection->getController()->isSampleExact())
606 {
607 auto vb = m_controllerConnection->valueBuffer();
608 if( vb )
609 {
610 float * values = vb->values();
611 float * nvalues = m_valueBuffer.values();
612 switch( m_scaleType )
613 {
614 case ScaleType::Linear:
615 for( int i = 0; i < m_valueBuffer.length(); i++ )
616 {
617 nvalues[i] = minValue<float>() + ( range() * values[i] );
618 }
619 break;
620 case ScaleType::Logarithmic:
621 for( int i = 0; i < m_valueBuffer.length(); i++ )
622 {
623 nvalues[i] = logToLinearScale( values[i] );
624 }
625 break;
626 default:
627 qFatal("AutomatableModel::valueBuffer() "
628 "lacks implementation for a scale type");
629 break;
630 }
631 m_lastUpdatedPeriod = s_periodCounter;
632 m_hasSampleExactData = true;
633 return &m_valueBuffer;
634 }
635 }
636
637 if (!m_controllerConnection)
638 {
639 AutomatableModel* lm = nullptr;
640 if (hasLinkedModels())
641 {
642 lm = m_linkedModels.front();
643 }
644 if (lm && lm->controllerConnection() && lm->useControllerValue() &&
645 lm->controllerConnection()->getController()->isSampleExact())
646 {
647 auto vb = lm->valueBuffer();
648 float * values = vb->values();
649 float * nvalues = m_valueBuffer.values();
650 for (int i = 0; i < vb->length(); i++)
651 {
652 nvalues[i] = fittedValue(values[i]);
653 }
654 m_lastUpdatedPeriod = s_periodCounter;
655 m_hasSampleExactData = true;
656 return &m_valueBuffer;
657 }
658 }
659
660 if( m_oldValue != val )
661 {
662 m_valueBuffer.interpolate( m_oldValue, val );
663 m_oldValue = val;
664 m_lastUpdatedPeriod = s_periodCounter;
665 m_hasSampleExactData = true;
666 return &m_valueBuffer;
667 }
668
669 // if we have no sample-exact source for a ValueBuffer, return NULL to signify that no data is available at the moment
670 // in which case the recipient knows to use the static value() instead
671 m_lastUpdatedPeriod = s_periodCounter;
672 m_hasSampleExactData = false;
673 return nullptr;
674}

711float AutomatableModel::globalAutomationValueAt( const TimePos& time )
712{
713 // get clips that connect to this model
714 auto clips = AutomationClip::clipsForModel(this);
715 if (clips.empty())
716 {
717 // if no such clips exist, return current value
718 return m_value;
719 }
720 else
721 {
722 // of those clips:
723 // find the clips which overlap with the time position
724 std::vector<AutomationClip*> clipsInRange;
725 for (const auto& clip : clips)
726 {
727 int s = clip->startPosition();
728 int e = clip->endPosition();
729 if (s <= time && e >= time) { clipsInRange.push_back(clip); }
730 }
731
732 AutomationClip * latestClip = nullptr;
733
734 if (!clipsInRange.empty())
735 {
736 // if there are more than one overlapping clips, just use the first one because
737 // multiple clip behaviour is undefined anyway
738 latestClip = clipsInRange[0];
739 }
740 else
741 // if we find no clips at the exact time, we need to search for the last clip before time and use that
742 {
743 int latestPosition = 0;
744
745 for (const auto& clip : clips)
746 {
747 int e = clip->endPosition();
748 if (e <= time && e > latestPosition)
749 {
750 latestPosition = e;
751 latestClip = clip;
752 }
753 }
754 }
755
756 if( latestClip )
757 {
758 // scale/fit the value appropriately and return it
759 const float value = latestClip->valueAt(time - latestClip->startPosition() + latestClip->startTimeOffset());
760 const float scaled_value = scaledValue( value );
761 return fittedValue( scaled_value );
762 }
763 // if we still find no clip, the value at that time is undefined so
764 // just return current value as the best we can do
765 else return m_value;
766 }
767}
