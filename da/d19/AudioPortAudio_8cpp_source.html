62AudioPortAudio::AudioPortAudio( bool & _success_ful, AudioEngine * _audioEngine ) :
63 AudioDevice(std::clamp<ch_cnt_t>(
64 ConfigManager::inst()->value("audioportaudio", "channels").toInt(),
65 DEFAULT_CHANNELS,
66 DEFAULT_CHANNELS), _audioEngine),
67 m_paStream( nullptr ),
68 m_wasPAInitError( false ),
69 m_outBuf(new SampleFrame[audioEngine()->framesPerPeriod()]),
70 m_outBufPos( 0 )
71{
72 _success_ful = false;
73
74 m_outBufSize = audioEngine()->framesPerPeriod();
75
76 PaError err = Pa_Initialize();
77
78 if( err != paNoError ) {
79 printf( "Couldn't initialize PortAudio: %s\n", Pa_GetErrorText( err ) );
80 m_wasPAInitError = true;
81 return;
82 }
83
84 if( Pa_GetDeviceCount() <= 0 )
85 {
86 return;
87 }
88
89 const QString& backend = ConfigManager::inst()->value( "audioportaudio", "backend" );
90 const QString& device = ConfigManager::inst()->value( "audioportaudio", "device" );
91
92 PaDeviceIndex inDevIdx = -1;
93 PaDeviceIndex outDevIdx = -1;
94 for( int i = 0; i < Pa_GetDeviceCount(); ++i )
95 {
96 const auto di = Pa_GetDeviceInfo(i);
97 if( di->name == device &&
98 Pa_GetHostApiInfo( di->hostApi )->name == backend )
99 {
100 inDevIdx = i;
101 outDevIdx = i;
102 }
103 }
104
105 if( inDevIdx < 0 )
106 {
107 inDevIdx = Pa_GetDefaultInputDevice();
108 }
109
110 if( outDevIdx < 0 )
111 {
112 outDevIdx = Pa_GetDefaultOutputDevice();
113 }
114
115 if( inDevIdx < 0 || outDevIdx < 0 )
116 {
117 return;
118 }
119
120 double inLatency = 0;//(double)audioEngine()->framesPerPeriod() / (double)sampleRate();
121 double outLatency = 0;//(double)audioEngine()->framesPerPeriod() / (double)sampleRate();
122
123 //inLatency = Pa_GetDeviceInfo( inDevIdx )->defaultLowInputLatency;
124 //outLatency = Pa_GetDeviceInfo( outDevIdx )->defaultLowOutputLatency;
125 const int samples = audioEngine()->framesPerPeriod();
126
127 // Configure output parameters.
128 m_outputParameters.device = outDevIdx;
129 m_outputParameters.channelCount = channels();
130 m_outputParameters.sampleFormat = paFloat32; // 32 bit floating point output
131 m_outputParameters.suggestedLatency = outLatency;
132 m_outputParameters.hostApiSpecificStreamInfo = nullptr;
133
134 // Configure input parameters.
135 m_inputParameters.device = inDevIdx;
136 m_inputParameters.channelCount = DEFAULT_CHANNELS;
137 m_inputParameters.sampleFormat = paFloat32; // 32 bit floating point input
138 m_inputParameters.suggestedLatency = inLatency;
139 m_inputParameters.hostApiSpecificStreamInfo = nullptr;
140
141 // Open an audio I/O stream.
142 err = Pa_OpenStream(
143 &m_paStream,
144 supportsCapture() ? &m_inputParameters : nullptr, // The input parameter
145 &m_outputParameters, // The outputparameter
146 sampleRate(),
147 samples,
148 paNoFlag, // Don't use any flags
149 _process_callback, // our callback function
150 this );
151
152 if( err == paInvalidDevice && sampleRate() < 48000 )
153 {
154 printf("Pa_OpenStream() failed with 44,1 KHz, trying again with 48 KHz\n");
155 // some backends or drivers do not allow 32 bit floating point data
156 // with a samplerate of 44100 Hz
157 setSampleRate( 48000 );
158 err = Pa_OpenStream(
159 &m_paStream,
160 supportsCapture() ? &m_inputParameters : nullptr, // The input parameter
161 &m_outputParameters, // The outputparameter
162 sampleRate(),
163 samples,
164 paNoFlag, // Don't use any flags
165 _process_callback, // our callback function
166 this );
167 }
168
169 if( err != paNoError )
170 {
171 printf( "Couldn't open PortAudio: %s\n", Pa_GetErrorText( err ) );
172 return;
173 }
174
175 printf( "Input device: '%s' backend: '%s'\n", Pa_GetDeviceInfo( inDevIdx )->name, Pa_GetHostApiInfo( Pa_GetDeviceInfo( inDevIdx )->hostApi )->name );
176 printf( "Output device: '%s' backend: '%s'\n", Pa_GetDeviceInfo( outDevIdx )->name, Pa_GetHostApiInfo( Pa_GetDeviceInfo( outDevIdx )->hostApi )->name );
177
178 // TODO: debug AudioEngine::pushInputFrames()
179 //m_supportsCapture = true;
180
181 _success_ful = true;
182}

231int AudioPortAudio::process_callback(const float* _inputBuffer, float* _outputBuffer, f_cnt_t _framesPerBuffer)
232{
233 if( supportsCapture() )
234 {
235 audioEngine()->pushInputFrames( (SampleFrame*)_inputBuffer, _framesPerBuffer );
236 }
237
238 if( m_stopped )
239 {
240 memset( _outputBuffer, 0, _framesPerBuffer *
241 channels() * sizeof(float) );
242 return paComplete;
243 }
244
245 while( _framesPerBuffer )
246 {
247 if( m_outBufPos == 0 )
248 {
249 // frames depend on the sample rate
250 const fpp_t frames = getNextBuffer( m_outBuf );
251 if( !frames )
252 {
253 m_stopped = true;
254 memset( _outputBuffer, 0, _framesPerBuffer *
255 channels() * sizeof(float) );
256 return paComplete;
257 }
258 m_outBufSize = frames;
259 }
260 const auto min_len = std::min(_framesPerBuffer, m_outBufSize - m_outBufPos);
261
262 for( fpp_t frame = 0; frame < min_len; ++frame )
263 {
264 for( ch_cnt_t chnl = 0; chnl < channels(); ++chnl )
265 {
266 (_outputBuffer + frame * channels())[chnl] = AudioEngine::clip(m_outBuf[frame][chnl]);
267 }
268 }
269
270 _outputBuffer += min_len * channels();
271 _framesPerBuffer -= min_len;
272 m_outBufPos += min_len;
273 m_outBufPos %= m_outBufSize;
274 }
275
276 return paContinue;
277}

320void AudioPortAudioSetupUtil::updateDevices()
321{
322 PaError err = Pa_Initialize();
323 if( err != paNoError ) {
324 printf( "Couldn't initialize PortAudio: %s\n", Pa_GetErrorText( err ) );
325 return;
326 }
327
328 // get active backend
329 const QString& backend = m_backendModel.currentText();
330 int hostApi = 0;
331 for( int i = 0; i < Pa_GetHostApiCount(); ++i )
332 {
333 const auto hi = Pa_GetHostApiInfo(i);
334 if( backend == hi->name )
335 {
336 hostApi = i;
337 break;
338 }
339 }
340
341 // get devices for selected backend
342 m_deviceModel.clear();
343 for( int i = 0; i < Pa_GetDeviceCount(); ++i )
344 {
345 const auto di = Pa_GetDeviceInfo(i);
346 if( di->hostApi == hostApi )
347 {
348 m_deviceModel.addItem( di->name );
349 }
350 }
351 Pa_Terminate();
352}
