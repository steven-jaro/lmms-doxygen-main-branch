61// Called to render each new track when rendering tracks individually.
62void RenderManager::renderNextTrack()
63{
64 m_activeRenderer.reset();
65
66 if (m_tracksToRender.empty())
67 {
68 // nothing left to render
69 restoreMutedState();
70 emit finished();
71 }
72 else
73 {
74 // pop the next track from our rendering queue
75 Track* renderTrack = m_tracksToRender.back();
76 m_tracksToRender.pop_back();
77
78 // mute everything but the track we are about to render
79 for (auto track : m_unmuted)
80 {
81 track->setMuted(track != renderTrack);
82 }
83
84 // for multi-render, prefix each output file with a different number
85 int trackNum = m_tracksToRender.size() + 1;
86
87 render( pathForTrack(renderTrack, trackNum) );
88 }
89}

91// Render the song into individual tracks
92void RenderManager::renderTracks()
93{
94 const TrackContainer::TrackList& tl = Engine::getSong()->tracks();
95
96 // find all currently unnmuted tracks -- we want to render these.
97 for (const auto& tk : tl)
98 {
99 Track::Type type = tk->type();
100
101 // Don't render automation tracks
102 if ( tk->isMuted() == false &&
103 ( type == Track::Type::Instrument || type == Track::Type::Sample ) )
104 {
105 m_unmuted.push_back(tk);
106 }
107 }
108
109 const TrackContainer::TrackList& t2 = Engine::patternStore()->tracks();
110 for (const auto& tk : t2)
111 {
112 Track::Type type = tk->type();
113
114 // Don't render automation tracks
115 if ( tk->isMuted() == false &&
116 ( type == Track::Type::Instrument || type == Track::Type::Sample ) )
117 {
118 m_unmuted.push_back(tk);
119 }
120 }
121
122 // copy the list of unmuted tracks into our rendering queue.
123 // we need to remember which tracks were unmuted to restore state at the end.
124 m_tracksToRender = m_unmuted;
125
126 renderNextTrack();
127}

135void RenderManager::render(QString outputPath)
136{
137 m_activeRenderer = std::make_unique<ProjectRenderer>(m_outputSettings, m_format, outputPath);
138
139 if( m_activeRenderer->isReady() )
140 {
141 // pass progress signals through
142 connect( m_activeRenderer.get(), SIGNAL(progressChanged(int)),
143 this, SIGNAL(progressChanged(int)));
144
145 // when it is finished, render the next track.
146 // if we have not queued any tracks, renderNextTrack will just clean up
147 connect( m_activeRenderer.get(), SIGNAL(finished()),
148 this, SLOT(renderNextTrack()));
149
150 m_activeRenderer->startProcessing();
151 }
152 else
153 {
154 qDebug( "Renderer failed to acquire a file device!" );
155 renderNextTrack();
156 }
157}

159// Unmute all tracks that were muted while rendering tracks
160void RenderManager::restoreMutedState()
161{
162 while (!m_unmuted.empty())
163 {
164 Track* restoreTrack = m_unmuted.back();
165 m_unmuted.pop_back();
166 restoreTrack->setMuted( false );
167 }
168}

170// Determine the output path for a track when rendering tracks individually
171QString RenderManager::pathForTrack(const Track *track, int num)
172{
173 QString extension = ProjectRenderer::getFileExtensionFromFormat( m_format );
174 QString name = track->name();
175 name = name.remove(QRegularExpression(FILENAME_FILTER));
176 name = QString( "%1_%2%3" ).arg( num ).arg( name ).arg( extension );
177 return QDir(m_outputPath).filePath(name);
178}
