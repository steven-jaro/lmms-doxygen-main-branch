47bool AudioFileFlac::startEncoding()
48{
49 m_sfinfo.samplerate=sampleRate();
50 m_sfinfo.channels=channels();
51 m_sfinfo.frames = audioEngine()->framesPerPeriod();
52 m_sfinfo.sections=1;
53 m_sfinfo.seekable=0;
54
55 m_sfinfo.format = SF_FORMAT_FLAC;
56
57 switch (getOutputSettings().getBitDepth())
58 {
59 case OutputSettings::BitDepth::Depth24Bit:
60 case OutputSettings::BitDepth::Depth32Bit:
61 // FLAC does not support 32bit sampling, so take it as 24.
62 m_sfinfo.format |= SF_FORMAT_PCM_24;
63 break;
64 default:
65 m_sfinfo.format |= SF_FORMAT_PCM_16;
66 }
67
68#ifdef LMMS_HAVE_SF_COMPLEVEL
69 double compression = getOutputSettings().getCompressionLevel();
70 sf_command(m_sf, SFC_SET_COMPRESSION_LEVEL, &compression, sizeof(double));
71#endif
72
73 m_sf = sf_open(
74#ifdef LMMS_BUILD_WIN32
75 outputFile().toLocal8Bit().constData(),
76#else
77 outputFile().toUtf8().constData(),
78#endif
79 SFM_WRITE,
80 &m_sfinfo
81 );
82
83 sf_command(m_sf, SFC_SET_CLIPPING, nullptr, SF_TRUE);
84
85 sf_set_string(m_sf, SF_STR_SOFTWARE, "LMMS");
86
87 return true;
88}

90void AudioFileFlac::writeBuffer(const SampleFrame* _ab, fpp_t const frames)
91{
92 OutputSettings::BitDepth depth = getOutputSettings().getBitDepth();
93 float clipvalue = std::nextafterf( -1.0f, 0.0f );
94
95 if (depth == OutputSettings::BitDepth::Depth24Bit || depth == OutputSettings::BitDepth::Depth32Bit) // Float encoding
96 {
97 auto buf = std::vector<sample_t>(frames * channels());
98 for(fpp_t frame = 0; frame < frames; ++frame)
99 {
100 for(ch_cnt_t channel=0; channel<channels(); ++channel)
101 {
102 // Clip the negative side to just above -1.0 in order to prevent it from changing sign
103 // Upstream issue: https://github.com/erikd/libsndfile/issues/309
104 // When this commit is reverted libsndfile-1.0.29 must be made a requirement for FLAC
105 buf[frame*channels() + channel] = std::max(clipvalue, _ab[frame][channel]);
106 }
107 }
108 sf_writef_float(m_sf, static_cast<float*>(buf.data()), frames);
109 }
110 else // integer PCM encoding
111 {
112 auto buf = std::vector<int_sample_t>(frames * channels());
113 convertToS16(_ab, frames, buf.data(), !isLittleEndian());
114 sf_writef_short(m_sf, static_cast<short*>(buf.data()), frames);
115 }
116
117}
