46JackServer::JackServer(bool sync, bool temporary, int timeout, bool rt, int priority, int port_max, bool verbose, jack_timer_type_t clock, char self_connect_mode, const char* server_name)
47{
48 if (rt) {
49 jack_info("JACK server starting in realtime mode with priority %ld", priority);
50 } else {
51 jack_info("JACK server starting in non-realtime mode");
52 }
53
54 jack_info("self-connect-mode is \"%s\"", jack_get_self_connect_mode_description(self_connect_mode));
55
56 fGraphManager = JackGraphManager::Allocate(port_max);
57 fMetadata = new JackMetadata(true);
58 fEngineControl = new JackEngineControl(sync, temporary, timeout, rt, priority, verbose, clock, server_name);
59 fEngine = new JackLockedEngine(fGraphManager, GetSynchroTable(), fEngineControl, self_connect_mode);
60
61 // A distinction is made between the threaded freewheel driver and the
62 // regular freewheel driver because the freewheel driver needs to run in
63 // threaded mode when freewheel mode is active and needs to run as a slave
64 // when freewheel mode isn't active.
65 JackFreewheelDriver* freewheelDriver = new JackFreewheelDriver(fEngine, GetSynchroTable());
66 fThreadedFreewheelDriver = new JackThreadedDriver(freewheelDriver);
67
68 fFreewheelDriver = freewheelDriver;
69 fDriverInfo = new JackDriverInfo();
70 fAudioDriver = NULL;
71 fFreewheel = false;
72 JackServerGlobals::fInstance = this; // Unique instance
73 JackServerGlobals::fUserCount = 1; // One user
74 JackGlobals::fVerbose = verbose;
75}

87int JackServer::Open(jack_driver_desc_t* driver_desc, JSList* driver_params)
88{
89 // TODO: move that in reworked JackServerGlobals::Init()
90 if (!JackMessageBuffer::Create()) {
91 jack_error("Cannot create message buffer");
92 }
93
94 if ((fAudioDriver = fDriverInfo->Open(driver_desc, fEngine, GetSynchroTable(), driver_params)) == NULL) {
95 jack_error("Cannot initialize driver");
96 goto fail_close1;
97 }
98
99 if (fRequestChannel.Open(fEngineControl->fServerName, this) < 0) {
100 jack_error("Server channel open error");
101 goto fail_close2;
102 }
103
104 if (fEngine->Open() < 0) {
105 jack_error("Cannot open engine");
106 goto fail_close3;
107 }
108
109 if (fFreewheelDriver->Open() < 0) {
110 jack_error("Cannot open freewheel driver");
111 goto fail_close4;
112 }
113
114 if (fAudioDriver->Attach() < 0) {
115 jack_error("Cannot attach audio driver");
116 goto fail_close5;
117 }
118
119 fFreewheelDriver->SetMaster(false);
120 fAudioDriver->SetMaster(true);
121 fAudioDriver->AddSlave(fFreewheelDriver);
122 InitTime();
123 SetClockSource(fEngineControl->fClockSource);
124 return 0;
125
126fail_close5:
127 fFreewheelDriver->Close();
128
129fail_close4:
130 fEngine->Close();
131
132fail_close3:
133 fRequestChannel.Close();
134
135fail_close2:
136 fAudioDriver->Close();
137
138fail_close1:
139 JackMessageBuffer::Destroy();
140 return -1;
141}

170
171 if (fFreewheel) {
172 if (fThreadedFreewheelDriver) {
173 res = fThreadedFreewheelDriver->Stop();
174 }
175 } else {
176 if (fAudioDriver) {
177 res = fAudioDriver->Stop();
178 }
179 }
180

245int JackServer::SetBufferSize(jack_nframes_t buffer_size)
246{
247 jack_log("JackServer::SetBufferSize nframes = %ld", buffer_size);
248 jack_nframes_t current_buffer_size = fEngineControl->fBufferSize;
249
250 if (current_buffer_size == buffer_size) {
251 jack_log("SetBufferSize: requirement for new buffer size equals current value");
252 return 0;
253 }
254
255 if (fAudioDriver->IsFixedBufferSize()) {
256 jack_log("SetBufferSize: driver only supports a fixed buffer size");
257 return -1;
258 }
259
260 if (fAudioDriver->Stop() != 0) {
261 jack_error("Cannot stop audio driver");
262 return -1;
263 }
264
265 if (fAudioDriver->SetBufferSize(buffer_size) == 0) {
266 fEngine->NotifyBufferSize(buffer_size);
267 return fAudioDriver->Start();
268 } else { // Failure: try to restore current value
269 jack_error("Cannot SetBufferSize for audio driver, restore current value %ld", current_buffer_size);
270 fAudioDriver->SetBufferSize(current_buffer_size);
271 fAudioDriver->Start();
272 // SetBufferSize actually failed, so return an error...
273 return -1;
274 }
275}

277/*
278Freewheel mode is implemented by switching from the (audio [slaves] + freewheel) driver to the freewheel driver only:
279
280 - "global" connection state is saved
281 - all audio driver and slaves ports are deconnected, thus there is no more dependencies with the audio driver and slaves
282 - the freewheel driver will be synchronized with the end of graph execution : all clients are connected to the freewheel driver
283 - the freewheel driver becomes the "master"
284
285Normal mode is restored with the connections state valid before freewheel mode was done. Thus one consider that
286no graph state change can be done during freewheel mode.
287*/
288
289int JackServer::SetFreewheel(bool onoff)
290{
291 jack_log("JackServer::SetFreewheel is = %ld want = %ld", fFreewheel, onoff);
292
293 if (fFreewheel) {
294 if (onoff) {
295 return -1;
296 } else {
297 fFreewheel = false;
298 fThreadedFreewheelDriver->Stop();
299 fGraphManager->Restore(&fConnectionState); // Restore connection state
300 fEngine->NotifyFreewheel(onoff);
301 fFreewheelDriver->SetMaster(false);
302 fAudioDriver->SetMaster(true);
303 return fAudioDriver->Start();
304 }
305 } else {
306 if (onoff) {
307 fFreewheel = true;
308 fAudioDriver->Stop();
309 fGraphManager->Save(&fConnectionState); // Save connection state
310 // Disconnect all slaves
311 std::list<JackDriverInterface*> slave_list = fAudioDriver->GetSlaves();
312 std::list<JackDriverInterface*>::const_iterator it;
313 for (it = slave_list.begin(); it != slave_list.end(); it++) {
314 JackDriver* slave = dynamic_cast<JackDriver*>(*it);
315 assert(slave);
316 fGraphManager->DisconnectAllPorts(slave->GetClientControl()->fRefNum);
317 }
318 // Disconnect master
319 fGraphManager->DisconnectAllPorts(fAudioDriver->GetClientControl()->fRefNum);
320 fEngine->NotifyFreewheel(onoff);
321 fAudioDriver->SetMaster(false);
322 fFreewheelDriver->SetMaster(true);
323 return fThreadedFreewheelDriver->Start();
324 } else {
325 return -1;
326 }
327 }
328}

352JackDriverInfo* JackServer::AddSlave(jack_driver_desc_t* driver_desc, JSList* driver_params)
353{
354 JackDriverInfo* info = new JackDriverInfo();
355 JackDriverClientInterface* slave = info->Open(driver_desc, fEngine, GetSynchroTable(), driver_params);
356
357 if (!slave) {
358 goto error1;
359 }
360 if (slave->Attach() < 0) {
361 goto error2;
362 }
363
364 slave->SetMaster(false);
365 fAudioDriver->AddSlave(slave);
366 return info;
367
368error2:
369 slave->Close();
370
371error1:
372 delete info;
373 return NULL;
374}

384int JackServer::SwitchMaster(jack_driver_desc_t* driver_desc, JSList* driver_params)
385{
386 std::list<JackDriverInterface*> slave_list;
387 std::list<JackDriverInterface*>::const_iterator it;
388
389 // Remove current master
390 fAudioDriver->Stop();
391 fAudioDriver->Detach();
392 fAudioDriver->Close();
393
394 // Open new master
395 JackDriverInfo* info = new JackDriverInfo();
396 JackDriverClientInterface* master = info->Open(driver_desc, fEngine, GetSynchroTable(), driver_params);
397
398 if (!master) {
399 goto error;
400 }
401
402 // Get slaves list
403 slave_list = fAudioDriver->GetSlaves();
404
405 // Move slaves in new master
406 for (it = slave_list.begin(); it != slave_list.end(); it++) {
407 JackDriverInterface* slave = *it;
408 master->AddSlave(slave);
409 }
410
411 // Delete old master
412 delete fDriverInfo;
413
414 // Activate master
415 fAudioDriver = master;
416 fDriverInfo = info;
417
418 if (fAudioDriver->Attach() < 0) {
419 goto error;
420 }
421
422 // Notify clients of new values
423 fEngine->NotifyBufferSize(fEngineControl->fBufferSize);
424 fEngine->NotifySampleRate(fEngineControl->fSampleRate);
425
426 // And finally start
427 fAudioDriver->SetMaster(true);
428 return fAudioDriver->Start();
429
430error:
431 delete info;
432 return -1;
433}
