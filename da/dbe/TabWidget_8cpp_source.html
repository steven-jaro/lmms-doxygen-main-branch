71void TabWidget::addTab(QWidget* w, const QString& name, const char* pixmap, int idx)
72{
73 // Append tab when position is not given
74 if (idx < 0/* || m_widgets.contains(idx) == true*/)
75 {
76 while(m_widgets.contains(++idx) == true)
77 {
78 }
79 }
80
81 // Tab's width when it is a text tab. This isn't correct for artwork tabs, but it's fixed later during the PaintEvent
82 int tab_width = horizontalAdvance(fontMetrics(), name) + 10;
83
84 // Register new tab
85 widgetDesc d = {w, pixmap, name, tab_width};
86 m_widgets[idx] = d;
87
88 // Position tab's window
89 if (!m_resizable)
90 {
91 w->setFixedSize(width() - 4, height() - m_tabbarHeight);
92 }
93 w->move(2, m_tabbarHeight - 1);
94 w->hide();
95
96 // Show tab's window if it's active
97 if (m_widgets.contains(m_activeTab))
98 {
99 // make sure new tab doesn't overlap current widget
100 m_widgets[m_activeTab].w->show();
101 m_widgets[m_activeTab].w->raise();
102 }
103}

108void TabWidget::setActiveTab(int idx)
109{
110 if (m_widgets.contains(idx))
111 {
112 int old_active = m_activeTab;
113 m_activeTab = idx;
114 m_widgets[m_activeTab].w->raise();
115 m_widgets[m_activeTab].w->show();
116 if (old_active != idx && m_widgets.contains(old_active))
117 {
118 m_widgets[old_active].w->hide();
119 }
120 update();
121 }
122}

125// Return the index of the tab at position "pos"
126int TabWidget::findTabAtPos(const QPoint* pos)
127{
128
129 if (pos->y() > 1 && pos->y() < m_tabbarHeight - 1)
130 {
131 int cx = ((m_caption == "") ? 4 : 14) + horizontalAdvance(fontMetrics(), m_caption);
132
133 for (widgetStack::iterator it = m_widgets.begin(); it != m_widgets.end(); ++it)
134 {
135 int const currentWidgetWidth = it->nwidth;
136
137 if (pos->x() >= cx && pos->x() <= cx + currentWidgetWidth)
138 {
139 return(it.key());
140 }
141 cx += currentWidgetWidth;
142 }
143 }
144
145 // Haven't found any tab at position "pos"
146 return(-1);
147}

150// Overload the QWidget::event handler to display tooltips (from https://doc.qt.io/qt-4.8/qt-widgets-tooltips-example.html)
151bool TabWidget::event(QEvent* event)
152{
153
154 if (event->type() == QEvent::ToolTip)
155 {
156 auto helpEvent = static_cast<QHelpEvent*>(event);
157
158 int idx = findTabAtPos(& helpEvent->pos());
159
160 if (idx != -1)
161 {
162 // Display tab's tooltip
163 QToolTip::showText(helpEvent->globalPos(), m_widgets[idx].name);
164 }
165 else
166 {
167 // The tooltip event doesn't relate to any tab, let's ignore it
168 QToolTip::hideText();
169 event->ignore();
170 }
171
172 return true;
173 }
174
175 // not a Tooltip event, let's propagate it to the other event handlers
176 return QWidget::event(event);
177}

182{
183
184 // Find index of tab that has been clicked

214void TabWidget::paintEvent(QPaintEvent* pe)
215{
216 QPainter p(this);
217 p.setFont(adjustedToPixelSize(font(), DEFAULT_FONT_SIZE));
218
219 // Draw background
220 QBrush bg_color = p.background();
221 p.fillRect(0, 0, width() - 1, height() - 1, bg_color);
222
223 // Draw external borders
224 p.setPen(tabBorder());
225 p.drawRect(0, 0, width() - 1, height() - 1);
226
227 // Draw tabs' bar background
228 p.fillRect(1, 1, width() - 2, m_tabheight + 2, tabBackground());
229
230 // Draw title, if any
231 if (!m_caption.isEmpty())
232 {
233 p.setPen(tabTitleText());
234 p.drawText(5, 11, m_caption);
235 }
236
237 // Calculate the tabs' x (tabs are painted next to the caption)
238 int tab_x_offset = m_caption.isEmpty() ? 4 : 14 + horizontalAdvance(fontMetrics(), m_caption);
239
240 // Compute tabs' width depending on the number of tabs (only applicable for artwork tabs)
241 widgetStack::iterator first = m_widgets.begin();
242 widgetStack::iterator last = m_widgets.end();
243 int tab_width = width();
244 if (first != last)
245 {
246 tab_width = (width() - tab_x_offset) / std::distance(first, last);
247 }
248
249 // Draw all tabs
250 p.setPen(tabText());
251 for (widgetStack::iterator it = first ; it != last ; ++it)
252 {
253 auto & currentWidgetDesc = *it;
254
255 // Draw a text tab or a artwork tab.
256 if (m_usePixmap)
257 {
258 // Fixes tab's width, because original size is only correct for text tabs
259 currentWidgetDesc.nwidth = tab_width;
260
261 // Get artwork
262 QPixmap artwork(embed::getIconPixmap(currentWidgetDesc.pixmap));
263
264 // Highlight active tab
265 if (it.key() == m_activeTab)
266 {
267 p.fillRect(tab_x_offset, 0, currentWidgetDesc.nwidth, m_tabbarHeight - 1, tabSelected());
268 }
269
270 // Draw artwork
271 p.drawPixmap(tab_x_offset + (currentWidgetDesc.nwidth - artwork.width()) / 2, 1, artwork);
272 }
273 else
274 {
275 // Highlight tab when active
276 if (it.key() == m_activeTab)
277 {
278 p.fillRect(tab_x_offset, 2, currentWidgetDesc.nwidth - 6, m_tabbarHeight - 4, tabSelected());
279 p.setPen(tabTextSelected());
280 p.drawText(tab_x_offset + 3, m_tabheight + 1, currentWidgetDesc.name);
281 }
282 else
283 {
284 // Draw text
285 p.setPen(tabText());
286 p.drawText(tab_x_offset + 3, m_tabheight + 1, currentWidgetDesc.name);
287 }
288 }
289
290 // Next tab's horizontal position
291 tab_x_offset += currentWidgetDesc.nwidth;
292 }
293}

298// Switch between tabs with mouse wheel
299void TabWidget::wheelEvent(QWheelEvent* we)
300{
301 if (position(we).y() > m_tabheight)
302 {
303 return;
304 }
305
306 we->accept();
307 int dir = (we->angleDelta().y() < 0) ? 1 : -1;
308 int tab = m_activeTab;
309 while(tab > -1 && static_cast<int>(tab) < m_widgets.count())
310 {
311 tab += dir;
312 if (m_widgets.contains(tab))
313 {
314 break;
315 }
316 }
317 setActiveTab(tab);
318}

323// Let parent widgets know how much space this tab widget needs
324QSize TabWidget::minimumSizeHint() const
325{
326 if (m_resizable)
327 {
328 int maxWidth = 0, maxHeight = 0;
329 for (const auto& widget : m_widgets)
330 {
331 maxWidth = std::max(maxWidth, widget.w->minimumSizeHint().width());
332 maxHeight = std::max(maxHeight, widget.w->minimumSizeHint().height());
333 }
334 // "-1" :
335 // in "addTab", under "Position tab's window", the widget is
336 // moved up by 1 pixel
337 return QSize(maxWidth + 4, maxHeight + m_tabbarHeight - 1);
338 }
339 else { return QWidget::minimumSizeHint(); }
340}
