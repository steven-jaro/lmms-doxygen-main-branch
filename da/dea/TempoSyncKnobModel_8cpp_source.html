73void TempoSyncKnobModel::calculateTempoSyncTime( bpm_t _bpm )
74{
75 float conversionFactor = 1.0;
76
77 if( m_tempoSyncMode != SyncMode::None )
78 {
79 switch( m_tempoSyncMode )
80 {
81 case SyncMode::Custom:
82 conversionFactor =
83 static_cast<float>( m_custom.getDenominator() ) /
84 static_cast<float>( m_custom.getNumerator() );
85 break;
86 case SyncMode::DoubleWholeNote:
87 conversionFactor = 0.125;
88 break;
89 case SyncMode::WholeNote:
90 conversionFactor = 0.25;
91 break;
92 case SyncMode::HalfNote:
93 conversionFactor = 0.5;
94 break;
95 case SyncMode::QuarterNote:
96 conversionFactor = 1.0;
97 break;
98 case SyncMode::EighthNote:
99 conversionFactor = 2.0;
100 break;
101 case SyncMode::SixteenthNote:
102 conversionFactor = 4.0;
103 break;
104 case SyncMode::ThirtysecondNote:
105 conversionFactor = 8.0;
106 break;
107 default: ;
108 }
109 bool journalling = testAndSetJournalling( false );
110 float oneUnit = 60000.0 / ( _bpm * conversionFactor * m_scale );
111 setValue( oneUnit * maxValue() );
112 setJournalling( journalling );
113 }
114
115 if( m_tempoSyncMode != m_tempoLastSyncMode )
116 {
117 emit syncModeChanged( m_tempoSyncMode );
118 m_tempoLastSyncMode = m_tempoSyncMode;
119 }
120 else if (m_tempoSyncMode == SyncMode::Custom)
121 {
122 emit syncModeChanged(m_tempoSyncMode);
123 }
124}

151void TempoSyncKnobModel::setSyncMode( SyncMode _new_mode )
152{
153 if( m_tempoSyncMode != _new_mode )
154 {
155 m_tempoSyncMode = _new_mode;
156 if( _new_mode == SyncMode::Custom )
157 {
158 connect( &m_custom, SIGNAL(dataChanged()),
159 this, SLOT(updateCustom()),
160 Qt::DirectConnection );
161 }
162 }
163 calculateTempoSyncTime( Engine::getSong()->getTempo() );
164}
