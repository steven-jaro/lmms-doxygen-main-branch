152 SERVER_EXPORT Jack::JackDriverClientInterface* driver_initialize(Jack::JackLockedEngine* engine, Jack::JackSynchro* table, const JSList* params) {
153 jack_nframes_t sample_rate = 44100;
154 jack_nframes_t buffer_size = JACK_GOLDFISH_BUFFER_SIZE;
155 unsigned int capture_ports = 2;
156 unsigned int playback_ports = 2;
157 int wait_time = 0;
158 const JSList * node;
159 const jack_driver_param_t * param;
160 bool monitor = false;
161
162 for (node = params; node; node = jack_slist_next (node)) {
163 param = (const jack_driver_param_t *) node->data;
164
165 switch (param->character) {
166
167 case 'C':
168 capture_ports = param->value.ui;
169 break;
170
171 case 'P':
172 playback_ports = param->value.ui;
173 break;
174
175 case 'r':
176 sample_rate = param->value.ui;
177 break;
178
179 case 'p':
180 buffer_size = param->value.ui;
181 break;
182
183 case 'w':
184 wait_time = param->value.ui;
185 break;
186
187 case 'm':
188 monitor = param->value.i;
189 break;
190 }
191 }
192
193 if (wait_time > 0) {
194 buffer_size = lroundf((wait_time * sample_rate) / 1000000.0f);
195 if (buffer_size > BUFFER_SIZE_MAX) {
196 buffer_size = BUFFER_SIZE_MAX;
197 jack_error("Buffer size set to %d", BUFFER_SIZE_MAX);
198 }
199 }
200
201 Jack::JackDriverClientInterface* driver = new Jack::JackThreadedDriver(new Jack::JackGoldfishDriver("system", "goldfish_pcm", engine, table));
202 if (driver->Open(buffer_size, sample_rate, 1, 1, capture_ports, playback_ports, monitor, "goldfish", "goldfish", 0, 0) == 0) {
203 return driver;
204 } else {
205 delete driver;
206 return NULL;
207 }
208 }
