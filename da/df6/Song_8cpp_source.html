146{
147 Engine::audioEngine()->requestChangeInModel();
148 const auto tempo = (bpm_t)m_tempoModel.value();
149 PlayHandleList & playHandles = Engine::audioEngine()->playHandles();
150 for (const auto& playHandle : playHandles)

152 auto nph = dynamic_cast<NotePlayHandle*>(playHandle);
153 if( nph && !nph->isReleased() )

155 nph->lock();

159 }
160 Engine::audioEngine()->doneChangeInModel();
161
162 Engine::updateFramesPerTick();
163
164 m_vstSyncController.setTempo( tempo );
165
166 emit tempoChanged( tempo );
167}

194void Song::processNextBuffer()
195{
196 m_vstSyncController.setPlaybackJumped(false);
197
198 // If nothing is playing, there is nothing to do
199 if (!m_playing) { return; }
200
201 // At the beginning of the song, we have to reset the LFOs
202 if (m_playMode == PlayMode::Song && getPlayPos() == 0)
203 {
204 EnvelopeAndLfoParameters::instances()->reset();
205 }
206
207 TrackList trackList;
208 int clipNum = -1; // The number of the clip that will be played
209
210 // Determine the list of tracks to play and the clip number
211 switch (m_playMode)
212 {
213 case PlayMode::Song:
214 trackList = tracks();
215 break;
216
217 case PlayMode::Pattern:
218 if (Engine::patternStore()->numOfPatterns() > 0)
219 {
220 clipNum = Engine::patternStore()->currentPattern();
221 trackList.push_back(PatternTrack::findPatternTrack(clipNum));
222 }
223 break;
224
225 case PlayMode::MidiClip:
226 if (m_midiClipToPlay)
227 {
228 clipNum = m_midiClipToPlay->getTrack()->getClipNum(m_midiClipToPlay);
229 trackList.push_back(m_midiClipToPlay->getTrack());
230 }
231 break;
232
233 default:
234 return;
235 }
236
237 // If the playback position is outside of the range [begin, end), move it to
238 // begin and inform interested parties.
239 // Returns true if the playback position was moved, else false.
240 const auto enforceLoop = [this](const TimePos& begin, const TimePos& end)
241 {
242 if (getPlayPos() < begin || getPlayPos() >= end)
243 {
244 setToTime(begin);
245 m_vstSyncController.setPlaybackJumped(true);
246 emit updateSampleTracks();
247 return true;
248 }
249 return false;
250 };
251
252 const auto& timeline = getTimeline();
253 const auto loopEnabled = !m_exporting && timeline.loopEnabled();
254
255 // Ensure playback begins within the loop if it is enabled
256 if (loopEnabled) { enforceLoop(timeline.loopBegin(), timeline.loopEnd()); }
257
258 // Inform VST plugins and sample tracks if the user moved the play head
259 if (getPlayPos().jumped())
260 {
261 m_vstSyncController.setPlaybackJumped(true);
262 emit updateSampleTracks();
263 getPlayPos().setJumped(false);
264 }
265
266 const auto framesPerTick = Engine::framesPerTick();
267 const auto framesPerPeriod = Engine::audioEngine()->framesPerPeriod();
268
269 f_cnt_t frameOffsetInPeriod = 0;
270
271 while (frameOffsetInPeriod < framesPerPeriod)
272 {
273 auto frameOffsetInTick = getPlayPos().currentFrame();
274
275 // If a whole tick has elapsed, update the frame and tick count, and check any loops
276 if (frameOffsetInTick >= framesPerTick)
277 {
278 // Transfer any whole ticks from the frame count to the tick count
279 const auto elapsedTicks = static_cast<int>(frameOffsetInTick / framesPerTick);
280 getPlayPos().setTicks(getPlayPos().getTicks() + elapsedTicks);
281 frameOffsetInTick -= elapsedTicks * framesPerTick;
282 getPlayPos().setCurrentFrame(frameOffsetInTick);
283
284 // If we are playing a pattern track, or a MIDI clip with no loop enabled,
285 // loop back to the beginning when we reach the end
286 if (m_playMode == PlayMode::Pattern)
287 {
288 enforceLoop(TimePos{0}, TimePos{Engine::patternStore()->lengthOfCurrentPattern(), 0});
289 }
290 else if (m_playMode == PlayMode::MidiClip && m_loopMidiClip && !loopEnabled)
291 {
292 enforceLoop(-m_midiClipToPlay->startTimeOffset(), m_midiClipToPlay->length() - m_midiClipToPlay->startTimeOffset());
293 }
294
295 // Handle loop points, and inform VST plugins of the loop status
296 if (loopEnabled || (m_loopRenderRemaining > 1 && getPlayPos() >= timeline.loopBegin()))
297 {
298 m_vstSyncController.startCycle(
299 timeline.loopBegin().getTicks(), timeline.loopEnd().getTicks());
300
301 // Loop if necessary, and decrement the remaining loops if we did
302 if (enforceLoop(timeline.loopBegin(), timeline.loopEnd())
303 && m_loopRenderRemaining > 1)
304 {
305 m_loopRenderRemaining--;
306 }
307 }
308 else
309 {
310 m_vstSyncController.stopCycle();
311 }
312 }
313
314 const f_cnt_t framesUntilNextPeriod = framesPerPeriod - frameOffsetInPeriod;
315 const auto framesUntilNextTick = static_cast<f_cnt_t>(std::ceil(framesPerTick - frameOffsetInTick));
316
317 // We want to proceed to the next buffer or tick, whichever is closer
318 const auto framesToPlay = std::min(framesUntilNextPeriod, framesUntilNextTick);
319
320 if (frameOffsetInPeriod == 0)
321 {
322 // First frame of buffer: update VST sync position.
323 // This must be done after we've corrected the frame/tick count,
324 // but before actually playing any frames.
325 m_vstSyncController.setAbsolutePosition(getPlayPos().getTicks()
326 + getPlayPos().currentFrame() / static_cast<double>(framesPerTick));
327 m_vstSyncController.update();
328 }
329
330 if (static_cast<f_cnt_t>(frameOffsetInTick) == 0)
331 {
332 // First frame of tick: process automation and play tracks
333 processAutomations(trackList, getPlayPos(), framesToPlay);
334 processMetronome(frameOffsetInPeriod);
335
336 for (const auto track : trackList)
337 {
338 track->play(getPlayPos(), framesToPlay, frameOffsetInPeriod, clipNum);
339 }
340 }
341
342 // Update frame counters
343 frameOffsetInPeriod += framesToPlay;
344 frameOffsetInTick += framesToPlay;
345 getPlayPos().setCurrentFrame(frameOffsetInTick);
346 m_elapsedMilliSeconds[static_cast<std::size_t>(m_playMode)] += TimePos::ticksToMilliseconds(framesToPlay / framesPerTick, getTempo());
347 m_elapsedBars = getPlayPos(PlayMode::Song).getBar();
348 m_elapsedTicks = (getPlayPos(PlayMode::Song).getTicks() % ticksPerBar()) / 48;
349 }

353void Song::processAutomations(const TrackList &tracklist, TimePos timeStart, fpp_t)
354{
355 AutomatedValueMap values;
356
357 QSet<const AutomatableModel*> recordedModels;
358
359 TrackContainer* container = this;
360 int clipNum = -1;
361
362 switch (m_playMode)
363 {
364 case PlayMode::Song:
365 break;
366 case PlayMode::Pattern:
367 {
368 if (tracklist.empty()) { return; }
369 Q_ASSERT(tracklist.at(0)->type() == Track::Type::Pattern);
370 auto patternTrack = dynamic_cast<PatternTrack*>(tracklist.at(0));
371 container = Engine::patternStore();
372 clipNum = patternTrack->patternIndex();
373 }
374 break;
375 default:
376 return;
377 }
378
379 values = container->automatedValuesAt(timeStart, clipNum);
380 const TrackList& tracks = container->tracks();
381
382 Track::clipVector clips;
383 for (Track* track : tracks)
384 {
385 if (track->type() == Track::Type::Automation) {
386 track->getClipsInRange(clips, 0, timeStart);
387 }
388 }
389
390 // Process recording
391 for (Clip* clip : clips)
392 {
393 auto p = dynamic_cast<AutomationClip *>(clip);
394 TimePos relTime = timeStart - p->startPosition();
395 if (p->isRecording() && relTime >= 0 && relTime < p->length())
396 {
397 const AutomatableModel* recordedModel = p->firstObject();
398 // The automation system really needs to be reworked.
399 // For whatever reason, the values in an automation clip are stored in un-un-scaled format, so if you
400 // are automating a log knob, when you draw an curve, the values being stored are not the actual values the
401 // knob will take, but instead the unscaled version of the unscaled numbers. The tooltip shows the number you expect, but if you double-click,
402 // you can see that the true values are stored by their inverse scaled value....which is wrong, since they weren't scaled in the first place...?
403 // Anyhow, in the meantime before we redo the automation system, when recording automations, we have to get the inverseScaledValue
404 // and store that so that when playing it back, it scales the value correctly.
405 p->recordValue(relTime, recordedModel->inverseScaledValue(recordedModel->value<float>()));
406
407 recordedModels << recordedModel;
408 }
409 }
410
411 // Checks if an automated model stopped being automated by automation clip
412 // so we can move the control back to any connected controller again
413 for (auto it = m_oldAutomatedValues.begin(); it != m_oldAutomatedValues.end(); it++)
414 {
415 AutomatableModel * am = it.key();
416 if (am->controllerConnection() && !values.contains(am))
417 {
418 am->setUseControllerValue(true);
419 }
420 }
421 m_oldAutomatedValues = values;
422
423 // Apply values
424 for (auto it = values.begin(); it != values.end(); it++)
425 {
426 if (! recordedModels.contains(it.key()))
427 {
428 it.key()->setAutomatedValue(it.value());
429 }
430 else if (!it.key()->useControllerValue())
431 {
432 it.key()->setUseControllerValue(true);
433 }
434 }

462int Song::getExportProgress() const
463{
464 TimePos pos = getPlayPos();
465
466 if (pos >= m_exportSongEnd)
467 {
468 return 100;
469 }
470 else if (pos <= m_exportSongBegin)
471 {
472 return 0;
473 }
474 else if (pos >= m_exportLoopEnd)
475 {
476 pos = (m_exportLoopBegin-m_exportSongBegin) + (m_exportLoopEnd - m_exportLoopBegin) *
477 m_loopRenderCount + (pos - m_exportLoopEnd);
478 }
479 else if ( pos >= m_exportLoopBegin )
480 {
481 pos = (m_exportLoopBegin-m_exportSongBegin) + ((m_exportLoopEnd - m_exportLoopBegin) *
482 (m_loopRenderCount - m_loopRenderRemaining)) + (pos - m_exportLoopBegin);
483 }
484 else
485 {
486 pos = (pos - m_exportSongBegin);
487 }
488
489 return (float)pos/(float)m_exportEffectiveLength*100.0f;
490}

648void Song::stop()
649{
650 // do not stop/reset things again if we're stopped already
651 if( m_playMode == PlayMode::None )
652 {
653 return;
654 }
655
656 using gui::TimeLineWidget;
657
658 // To avoid race conditions with the processing threads
659 Engine::audioEngine()->requestChangeInModel();
660
661 auto& timeline = getTimeline();
662 m_paused = false;
663 m_recording = true;
664 m_playing = false;
665
666 switch (timeline.stopBehaviour())
667 {
668 case Timeline::StopBehaviour::BackToZero:
669 if (m_playMode == PlayMode::MidiClip)
670 {
671 getPlayPos().setTicks(std::max(0, -m_midiClipToPlay->startTimeOffset()));
672 }
673 else
674 {
675 getPlayPos().setTicks(0);
676 }
677 m_elapsedMilliSeconds[static_cast<std::size_t>(m_playMode)] = 0;
678 break;
679
680 case Timeline::StopBehaviour::BackToStart:
681 if (timeline.playStartPosition() >= 0)
682 {
683 getPlayPos().setTicks(timeline.playStartPosition().getTicks());
684 setToTime(timeline.playStartPosition());
685
686 timeline.setPlayStartPosition(-1);
687 }
688 break;
689
690 case Timeline::StopBehaviour::KeepPosition:
691 break;
692 }
693
694 m_elapsedMilliSeconds[static_cast<std::size_t>(PlayMode::None)] = m_elapsedMilliSeconds[static_cast<std::size_t>(m_playMode)];
695 getPlayPos(PlayMode::None).setTicks(getPlayPos().getTicks());
696
697 getPlayPos().setCurrentFrame( 0 );
698
699 m_vstSyncController.setPlaybackState( m_exporting );
700 m_vstSyncController.setAbsolutePosition(
701 getPlayPos().getTicks()
702 + getPlayPos().currentFrame()
703 / (double) Engine::framesPerTick() );
704
705 // remove all note-play-handles that are active
706 Engine::audioEngine()->clear();
707
708 // Moves the control of the models that were processed on the last frame
709 // back to their controllers.
710 for (auto it = m_oldAutomatedValues.begin(); it != m_oldAutomatedValues.end(); it++)
711 {
712 AutomatableModel * am = it.key();
713 am->setUseControllerValue(true);
714 }
715 m_oldAutomatedValues.clear();
716
717 m_playMode = PlayMode::None;
718
719 Engine::audioEngine()->doneChangeInModel();
720
721 emit stopped();
722 emit playbackStateChanged();
723}

728void Song::startExport()
729{
730 stop();
731
732 m_exporting = true;
733 updateLength();
734
735 const auto& timeline = getTimeline(PlayMode::Song);
736
737 if (m_renderBetweenMarkers)
738 {
739 m_exportSongBegin = m_exportLoopBegin = timeline.loopBegin();
740 m_exportSongEnd = m_exportLoopEnd = timeline.loopEnd();
741
742 getPlayPos(PlayMode::Song).setTicks(timeline.loopBegin().getTicks());
743 }
744 else
745 {
746 m_exportSongEnd = TimePos(m_length, 0);
747
748 // Handle potentially ridiculous loop points gracefully.
749 if (m_loopRenderCount > 1 && timeline.loopEnd() > m_exportSongEnd)
750 {
751 m_exportSongEnd = timeline.loopEnd();
752 }
753
754 if (!m_exportLoop)
755 m_exportSongEnd += TimePos(1,0);
756
757 m_exportSongBegin = TimePos(0,0);
758 m_exportLoopBegin = timeline.loopBegin() < m_exportSongEnd && timeline.loopEnd() <= m_exportSongEnd
759 ? timeline.loopBegin()
760 : TimePos{0};
761 m_exportLoopEnd = timeline.loopBegin() < m_exportSongEnd && timeline.loopEnd() <= m_exportSongEnd
762 ? timeline.loopEnd()
763 : TimePos{0};
764
765 getPlayPos(PlayMode::Song).setTicks( 0 );
766 }
767
768 m_exportEffectiveLength = (m_exportLoopBegin - m_exportSongBegin) + (m_exportLoopEnd - m_exportLoopBegin)
769 * m_loopRenderCount + (m_exportSongEnd - m_exportLoopEnd);
770 m_loopRenderRemaining = m_loopRenderCount;
771
772 playSong();
773
774 m_vstSyncController.setPlaybackState( true );
775}

1037 // new project

1040 cantLoadProject = true;
1041 }
1042 else

1045 // added to project bundles and loaded with "local:" paths

1054 "run malicious code."));
1055 }
1056 else
1057 {
1058 QTextStream(stderr) << tr("Can't load project: "
1059 "Project file contains local paths to plugins.")
1060#if (QT_VERSION >= QT_VERSION_CHECK(5,15,0))
1061 << Qt::endl;
1062#else
1063 << endl;
1064#endif
1065 }
1066 }
1067 }
1068
1069 if (cantLoadProject)
1070 {
1071 if( m_loadOnLaunch )
1072 {
1073 createNewProject();
1074 }
1075 setProjectFileName(m_oldFileName);

1079 m_oldFileName = m_fileName;
1080
1081 clearProject();
1082
1083 clearErrors();
1084
1085 Engine::audioEngine()->requestChangeInModel();
1086
1087 // get the header information from the DOM
1088 m_tempoModel.loadSettings( dataFile.head(), "bpm" );
1089 m_timeSigModel.loadSettings( dataFile.head(), "timesig" );
1090 m_masterVolumeModel.loadSettings( dataFile.head(), "mastervol" );
1091 m_masterPitchModel.loadSettings( dataFile.head(), "masterpitch" );
1092
1093 getTimeline(PlayMode::Song).setLoopEnabled(false);
1094
1095 //Backward compatibility for LMMS <= 0.4.15
1096 PeakController::initGetControllerBySetting();
1097
1098 // Load mixer first to be able to set the correct range for mixer channels
1099 node = dataFile.content().firstChildElement( Engine::mixer()->nodeName() );
1100 if( !node.isNull() )
1101 {
1102 Engine::mixer()->restoreState( node.toElement() );
1103 if( getGUI() != nullptr )
1104 {
1105 // refresh MixerView
1106 getGUI()->mixerView()->refreshDisplay();
1107 }
1108 }
1109
1110 node = dataFile.content().firstChild();
1111
1112 QDomNodeList tclist=dataFile.content().elementsByTagName("trackcontainer");
1113 m_nLoadingTrack=0;
1114 for( int i=0,n=tclist.count(); i<n; ++i )
1115 {
1116 QDomNode nd=tclist.at(i).firstChild();
1117 while(!nd.isNull())
1118 {
1119 if( nd.isElement() && nd.nodeName() == "track" )
1120 {
1121 ++m_nLoadingTrack;
1122 if (static_cast<Track::Type>(nd.toElement().attribute("type").toInt()) == Track::Type::Pattern)
1123 {
1124 n += nd.toElement().elementsByTagName("patterntrack").at(0)
1125 .toElement().firstChildElement().childNodes().count();
1126 }
1127 nd=nd.nextSibling();
1128 }
1129 }
1130 }
1131
1132 while( !node.isNull() && !isCancelled() )
1133 {
1134 if( node.isElement() )
1135 {
1136 if( node.nodeName() == "trackcontainer" )
1137 {
1138 ( (JournallingObject *)( this ) )->restoreState( node.toElement() );
1139 }
1140 else if( node.nodeName() == "controllers" )
1141 {
1142 restoreControllerStates( node.toElement() );
1143 }
1144 else if (node.nodeName() == "scales")
1145 {
1146 restoreScaleStates(node.toElement());
1147 }
1148 else if (node.nodeName() == "keymaps")
1149 {
1150 restoreKeymapStates(node.toElement());
1151 }
1152 else if( getGUI() != nullptr )
1153 {
1154 if( node.nodeName() == getGUI()->getControllerRackView()->nodeName() )
1155 {
1156 getGUI()->getControllerRackView()->restoreState( node.toElement() );
1157 }
1158 else if( node.nodeName() == getGUI()->pianoRoll()->nodeName() )
1159 {
1160 getGUI()->pianoRoll()->restoreState( node.toElement() );
1161 }
1162 else if( node.nodeName() == getGUI()->automationEditor()->m_editor->nodeName() )
1163 {
1164 getGUI()->automationEditor()->m_editor->restoreState( node.toElement() );
1165 }
1166 else if( node.nodeName() == getGUI()->getProjectNotes()->nodeName() )
1167 {
1168 getGUI()->getProjectNotes()->SerializingObject::restoreState( node.toElement() );
1169 }
1170 else if (node.nodeName() == getTimeline(PlayMode::Song).nodeName())
1171 {
1172 getTimeline(PlayMode::Song).restoreState(node.toElement());
1173 }
1174 }
1175 }
1176 node = node.nextSibling();
1177 }
1178
1179 // quirk for fixing projects with broken positions of Clips inside pattern tracks
1180 Engine::patternStore()->fixIncorrectPositions();
1181
1182 // Connect controller links to their controllers
1183 // now that everything is loaded
1184 ControllerConnection::finalizeConnections();
1185
1186 // Remove dummy controllers that was added for correct connections
1187 m_controllers.erase(std::remove_if(m_controllers.begin(), m_controllers.end(),
1188 [](Controller* c){return c->type() == Controller::ControllerType::Dummy;}),
1189 m_controllers.end());
1190
1191 // resolve all IDs so that autoModels are automated
1192 AutomationClip::resolveAllIDs();
1193
1194
1195 Engine::audioEngine()->doneChangeInModel();
1196
1197 ConfigManager::inst()->addRecentlyOpenedProject( fileName );
1198
1199 Engine::projectJournal()->setJournalling( true );
1200
1201 emit projectLoaded();
1202
1203 if( isCancelled() )
1204 {
1205 m_isCancelled = false;
1206 createNewProject();
1207 return;
1208 }
1209
1210 if ( hasErrors())
1211 {
1212 if ( getGUI() != nullptr )
1213 {
1214 QMessageBox::warning( nullptr, tr("LMMS Error report"), errorSummary(),
1215 QMessageBox::Ok );
1216 }
1217 else
1218 {
1219#if (QT_VERSION >= QT_VERSION_CHECK(5,15,0))
1220 QTextStream(stderr) << Engine::getSong()->errorSummary() << Qt::endl;
1221#else
1222 QTextStream(stderr) << Engine::getSong()->errorSummary() << endl;
1223#endif
1224 }
1225 }
1226
1227 m_loadingProject = false;
1228 updateLength();
1229 setModified(false);
1230 m_loadOnLaunch = false;
1231}
