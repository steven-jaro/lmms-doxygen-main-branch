35AudioFileDevice::AudioFileDevice( OutputSettings const & outputSettings,
36 const ch_cnt_t _channels,
37 const QString & _file,
38 AudioEngine* _audioEngine ) :
39 AudioDevice( _channels, _audioEngine ),
40 m_outputFile( _file ),
41 m_outputSettings(outputSettings)
42{
43 using gui::ExportProjectDialog;
44
45 setSampleRate( outputSettings.getSampleRate() );
46
47 if( m_outputFile.open( QFile::WriteOnly | QFile::Truncate ) == false )
48 {
49 QString title, message;
50 title = ExportProjectDialog::tr( "Could not open file" );
51 message = ExportProjectDialog::tr( "Could not open file %1 "
52 "for writing.\nPlease make "
53 "sure you have write "
54 "permission to the file and "
55 "the directory containing the "
56 "file and try again!"
57 ).arg( _file );
58
59 if (gui::getGUI() != nullptr)
60 {
61 QMessageBox::critical( nullptr, title, message,
62 QMessageBox::Ok,
63 QMessageBox::NoButton );
64 }
65 else
66 {
67 fprintf( stderr, "%s\n", message.toUtf8().constData() );
68 exit( EXIT_FAILURE );
69 }
70 }
71}
