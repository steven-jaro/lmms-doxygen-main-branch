49NotePlayHandle::NotePlayHandle( InstrumentTrack* instrumentTrack,
50 const f_cnt_t _offset,
51 const f_cnt_t _frames,
52 const Note& n,
53 NotePlayHandle *parent,
54 int midiEventChannel,
55 Origin origin ) :
56 PlayHandle( PlayHandle::Type::NotePlayHandle, _offset ),
57 Note(n),
58 m_pluginData( nullptr ),
59 m_instrumentTrack( instrumentTrack ),
60 m_frames( 0 ),
61 m_totalFramesPlayed( 0 ),
62 m_framesBeforeRelease( 0 ),
63 m_releaseFramesToDo( 0 ),
64 m_releaseFramesDone( 0 ),
65 m_subNotes(),
66 m_released( false ),
67 m_releaseStarted( false ),
68 m_hasMidiNote( false ),
69 m_hasParent( parent != nullptr ),
70 m_parent( parent ),
71 m_hadChildren( false ),
72 m_muted( false ),
73 m_patternTrack( nullptr ),
74 m_origTempo( Engine::getSong()->getTempo() ),
75 m_origBaseNote( instrumentTrack->baseNote() ),
76 m_frequency( 0 ),
77 m_unpitchedFrequency( 0 ),
78 m_baseDetuning( nullptr ),
79 m_songGlobalParentOffset( 0 ),
80 m_midiChannel( midiEventChannel >= 0 ? midiEventChannel : instrumentTrack->midiPort()->realOutputChannel() ),
81 m_origin( origin ),
82 m_frequencyNeedsUpdate( false )
83{
84 lock();
85 if( hasParent() == false )
86 {
87 m_baseDetuning = new BaseDetuning(detuning().get());
88 m_instrumentTrack->m_processHandles.push_back( this );
89 }
90 else
91 {
92 m_baseDetuning = parent->m_baseDetuning;
93
94 parent->m_subNotes.push_back( this );
95 parent->m_hadChildren = true;
96
97 m_patternTrack = parent->m_patternTrack;
98
99 parent->setUsesBuffer( false );
100 }
101
102 updateFrequency();
103
104 setFrames( _frames );
105
106 // inform attached components about new MIDI note (used for recording in Piano Roll)
107 if( m_origin == Origin::MidiInput )
108 {
109 m_instrumentTrack->midiNoteOn( *this );
110 }
111
112 if (m_instrumentTrack->instrument() && m_instrumentTrack->instrument()->isSingleStreamed())
113 {
114 setUsesBuffer( false );
115 }
116
117 setAudioBusHandle(instrumentTrack->audioBusHandle());
118
119 unlock();
120}

123NotePlayHandle::~NotePlayHandle()
124{
125 lock();
126 noteOff( 0 );
127
128 if( hasParent() == false )
129 {
130 delete m_baseDetuning;
131 m_instrumentTrack->m_processHandles.removeAll( this );
132 }
133 else
134 {
135 m_parent->m_subNotes.removeOne( this );
136 }
137
138 if( m_pluginData != nullptr )
139 {
140 m_instrumentTrack->deleteNotePluginData( this );
141 }
142
143 if( m_instrumentTrack->m_notes[key()] == this )
144 {
145 m_instrumentTrack->m_notes[key()] = nullptr;
146 }
147
148 m_subNotes.clear();
149
150 if( buffer() ) releaseBuffer();
151
152 unlock();
153}

186void NotePlayHandle::play( SampleFrame* _working_buffer )
187{
188 if (m_muted)
189 {
190 return;
191 }
192
193 // if the note offset falls over to next period, then don't start playback yet
194 if( offset() >= Engine::audioEngine()->framesPerPeriod() )
195 {
196 setOffset( offset() - Engine::audioEngine()->framesPerPeriod() );
197 return;
198 }
199
200 lock();
201
202 // Don't play the note if it falls outside of the user defined key range
203 // TODO: handle the range check by Microtuner, and if the key becomes "not mapped", save the current frequency
204 // so that the note release can finish playing using a valid frequency instead of a 1 Hz placeholder
205 if (key() < m_instrumentTrack->m_firstKeyModel.value() ||
206 key() > m_instrumentTrack->m_lastKeyModel.value())
207 {
208 // Release the note in case it started playing before going out of range
209 noteOff(0);
210 // Exit if the note did not start playing before going out of range (i.e. there is no need to play release)
211 if (m_totalFramesPlayed == 0)
212 {
213 unlock();
214 return;
215 }
216 }
217
218 /* It is possible for NotePlayHandle::noteOff to be called before NotePlayHandle::play,
219 * which results in a note-on message being sent without a subsequent note-off message.
220 * Therefore, we check here whether the note has already been released before sending
221 * the note-on message. */
222 if( !m_released
223 && m_totalFramesPlayed == 0 && !m_hasMidiNote
224 && ( hasParent() || ! m_instrumentTrack->isArpeggioEnabled() ) )
225 {
226 m_hasMidiNote = true;
227
228 const int baseVelocity = m_instrumentTrack->midiPort()->baseVelocity();
229
230 // send MidiNoteOn event
231 m_instrumentTrack->processOutEvent(
232 MidiEvent( MidiNoteOn, midiChannel(), midiKey(), midiVelocity( baseVelocity ) ),
233 TimePos::fromFrames( offset(), Engine::framesPerTick() ),
234 offset() );
235 }
236
237 if( m_frequencyNeedsUpdate )
238 {
239 updateFrequency();
240 }
241
242 // number of frames that can be played this period
243 f_cnt_t framesThisPeriod = m_totalFramesPlayed == 0
244 ? Engine::audioEngine()->framesPerPeriod() - offset()
245 : Engine::audioEngine()->framesPerPeriod();
246
247 // check if we start release during this period
248 if( m_released == false &&
249 instrumentTrack()->isSustainPedalPressed() == false &&
250 m_totalFramesPlayed + framesThisPeriod > m_frames )
251 {
252 noteOff( m_totalFramesPlayed == 0
253 ? ( m_frames + offset() ) // if we have noteon and noteoff during the same period, take offset in account for release frame
254 : ( m_frames - m_totalFramesPlayed ) ); // otherwise, the offset is already negated and can be ignored
255 }
256
257 // under some circumstances we're called even if there's nothing to play
258 // therefore do an additional check which fixes crash e.g. when
259 // decreasing release of an instrument-track while the note is active
260 if( framesLeft() > 0 )
261 {
262 // play note!
263 m_instrumentTrack->playNote( this, _working_buffer );
264 }
265
266 if( m_released && (!instrumentTrack()->isSustainPedalPressed() ||
267 m_releaseStarted) )
268 {
269 m_releaseStarted = true;
270
271 f_cnt_t todo = framesThisPeriod;
272
273 // if this note is base-note for arpeggio, always set
274 // m_releaseFramesToDo to bigger value than m_releaseFramesDone
275 // because we do not allow NotePlayHandle::isFinished() to be true
276 // until all sub-notes are completely played and no new ones
277 // are inserted by arpAndChordsTabWidget::processNote()
278 if( ! m_subNotes.isEmpty() )
279 {
280 m_releaseFramesToDo = m_releaseFramesDone + 2 * Engine::audioEngine()->framesPerPeriod();
281 }
282 // look whether we have frames left to be done before release
283 if( m_framesBeforeRelease )
284 {
285 // yes, then look whether these samples can be played
286 // within one audio-buffer
287 if( m_framesBeforeRelease <= framesThisPeriod )
288 {
289 // yes, then we did less releaseFramesDone
290 todo -= m_framesBeforeRelease;
291 m_framesBeforeRelease = 0;
292 }
293 else
294 {
295 // no, then just decrease framesBeforeRelease
296 // and wait for next loop... (we're not in
297 // release-phase yet)
298 todo = 0;
299 m_framesBeforeRelease -= framesThisPeriod;
300 }
301 }
302 // look whether we're in release-phase
303 if( todo && m_releaseFramesDone < m_releaseFramesToDo )
304 {
305 // check whether we have to do more frames in current
306 // loop than left in current loop
307 if( m_releaseFramesToDo - m_releaseFramesDone >= todo )
308 {
309 // yes, then increase number of release-frames
310 // done
311 m_releaseFramesDone += todo;
312 }
313 else
314 {
315 // no, we did all in this loop!
316 m_releaseFramesDone = m_releaseFramesToDo;
317 }
318 }
319 }
320
321 // update internal data
322 m_totalFramesPlayed += framesThisPeriod;
323 unlock();
324}

329f_cnt_t NotePlayHandle::framesLeft() const
330{
331 if( instrumentTrack()->isSustainPedalPressed() )
332 {
333 return 4 * Engine::audioEngine()->framesPerPeriod();
334 }
335 else if( m_released && actualReleaseFramesToDo() == 0 )
336 {
337 return m_framesBeforeRelease;
338 }
339 else if( m_released )
340 {
341 return m_framesBeforeRelease + m_releaseFramesToDo - m_releaseFramesDone;
342 }
343 return m_frames+actualReleaseFramesToDo()-m_totalFramesPlayed;
344}

369void NotePlayHandle::noteOff( const f_cnt_t _s )
370{
371 if( m_released )
372 {
373 return;
374 }
375 m_released = true;
376
377 // first note-off all sub-notes
378 for( NotePlayHandle * n : m_subNotes )
379 {
380 n->lock();
381 n->noteOff( _s );
382 n->unlock();
383 }
384
385 // then set some variables indicating release-state
386 m_framesBeforeRelease = _s;
387 m_releaseFramesToDo = std::max<f_cnt_t>(0, actualReleaseFramesToDo());
388
389 if( m_hasMidiNote )
390 {
391 m_hasMidiNote = false;
392
393 // send MidiNoteOff event
394 m_instrumentTrack->processOutEvent(
395 MidiEvent( MidiNoteOff, midiChannel(), midiKey(), 0 ),
396 TimePos::fromFrames( _s, Engine::framesPerTick() ),
397 _s );
398 }
399
400 // inform attached components about MIDI finished (used for recording in Piano Roll)
401 if (!instrumentTrack()->isSustainPedalPressed())
402 {
403 if( m_origin == Origin::MidiInput )
404 {
405 setLength( TimePos( static_cast<f_cnt_t>( totalFramesPlayed() / Engine::framesPerTick() ) ) );
406 m_instrumentTrack->midiNoteOff( *this );
407 }
408 }
409}

456int NotePlayHandle::index() const
457{
458 const PlayHandleList & playHandles = Engine::audioEngine()->playHandles();
459 int idx = 0;
460 for (const auto& playHandle : playHandles)
461 {
462 const auto nph = dynamic_cast<const NotePlayHandle*>(playHandle);
463 if( nph == nullptr || nph->m_instrumentTrack != m_instrumentTrack || nph->isReleased() || nph->hasParent() )
464 {
465 continue;
466 }
467 if( nph == this )
468 {
469 return idx;
470 }
471 ++idx;
472 }
473 return -1;
474}

520void NotePlayHandle::updateFrequency()
521{
522 int masterPitch = m_instrumentTrack->m_useMasterPitchModel.value() ? Engine::getSong()->masterPitch() : 0;
523 int baseNote = m_instrumentTrack->baseNoteModel()->value();
524 float detune = m_baseDetuning->value();
525 float instrumentPitch = m_instrumentTrack->pitchModel()->value();
526
527 if (m_instrumentTrack->m_microtuner.enabled())
528 {
529 // custom key mapping and scale: get frequency from the microtuner
530 const auto transposedKey = key() + masterPitch;
531
532 if (m_instrumentTrack->isKeyMapped(transposedKey))
533 {
534 const auto frequency = m_instrumentTrack->m_microtuner.keyToFreq(transposedKey, baseNote);
535 m_frequency = frequency * std::exp2((detune + instrumentPitch / 100) / 12.f);
536 m_unpitchedFrequency = frequency * std::exp2(detune / 12.f);
537 }
538 else
539 {
540 m_frequency = m_unpitchedFrequency = 0;
541 }
542 }
543 else
544 {
545 // default key mapping and 12-TET frequency computation with default 440 Hz base note frequency
546 const float pitch = (key() - baseNote + masterPitch + detune) / 12.0f;
547 m_frequency = DefaultBaseFreq * std::exp2(pitch + instrumentPitch / (100 * 12.0f));
548 m_unpitchedFrequency = DefaultBaseFreq * std::exp2(pitch);
549 }
550
551 for (auto it : m_subNotes)
552 {
553 it->updateFrequency();
554 }
555}

560void NotePlayHandle::processTimePos(const TimePos& time, float pitchValue, bool isRecording)
561{
562 if (!detuning() || time < songGlobalParentOffset() + pos()) { return; }
563
564 if (isRecording && m_origin == Origin::MidiInput)
565 {
566 detuning()->automationClip()->recordValue(time - songGlobalParentOffset() - pos(), pitchValue / 100);
567 }
568 else
569 {
570 const float v = detuning()->automationClip()->valueAt(time - songGlobalParentOffset() - pos());
571 if (!approximatelyEqual(v, m_baseDetuning->value()))
572 {
573 m_baseDetuning->setValue(v);
574 updateFrequency();
575 }
576 }
577}

582void NotePlayHandle::resize( const bpm_t _new_tempo )
583{
584 if (origin() == Origin::MidiInput ||
585 (origin() == Origin::NoteStacking && m_parent->origin() == Origin::MidiInput))
586 {
587 // Don't resize notes from MIDI input - they should continue to play
588 // until the key is released, and their large duration can cause
589 // overflows in this method.
590 return;
591 }
592
593 double completed = m_totalFramesPlayed / (double) m_frames;
594 double new_frames = m_origFrames * m_origTempo / (double) _new_tempo;
595 m_frames = (f_cnt_t)new_frames;
596 m_totalFramesPlayed = (f_cnt_t)( completed * new_frames );
597
598 for (const auto& subNote : m_subNotes)
599 {
600 subNote->resize(_new_tempo);
601 }
602}

627NotePlayHandle * NotePlayHandleManager::acquire( InstrumentTrack* instrumentTrack,
628 const f_cnt_t offset,
629 const f_cnt_t frames,
630 const Note& noteToPlay,
631 NotePlayHandle* parent,
632 int midiEventChannel,
633 NotePlayHandle::Origin origin )
634{
635 // TODO: use some lockless data structures
636 s_mutex.lockForWrite();
637 if (s_availableIndex < 0) { extend(NPH_CACHE_INCREMENT); }
638 NotePlayHandle * nph = s_available[s_availableIndex--];
639 s_mutex.unlock();
640
641 new( (void*)nph ) NotePlayHandle( instrumentTrack, offset, frames, noteToPlay, parent, midiEventChannel, origin );
642 return nph;
643}

655void NotePlayHandleManager::extend( int c )
656{
657 s_size += c;
658 auto tmp = new NotePlayHandle*[s_size];
659 delete[] s_available;
660 s_available = tmp;
661
662 auto n = static_cast<NotePlayHandle *>(std::malloc(sizeof(NotePlayHandle) * c));
663
664 for( int i=0; i < c; ++i )
665 {
666 s_available[++s_availableIndex] = n;
667 ++n;
