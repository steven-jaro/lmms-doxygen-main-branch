81/*
82==============================================================================
83Name: process_sample
84Description:
85 Downsamples (x2) one pair of samples, to generate one output sample.
86Input parameters:
87 - in_ptr: pointer on the two samples to decimate
88Returns: Samplerate-reduced sample.
89Throws: Nothing
90==============================================================================
91*/
92
93template <int NC, typename DT, int NCHN>
94typename Downsampler2xTpl <NC, DT, NCHN>::DataType Downsampler2xTpl <NC, DT, NCHN>::process_sample (const DataType in_ptr [2]) noexcept
95{
96 assert (in_ptr != nullptr);
97
98 DataType spl_0 (in_ptr [1]);
99 DataType spl_1 (in_ptr [0]);
100
101 StageProcTpl <NBR_COEFS, DataType>::process_sample_pos (
102 NBR_COEFS, spl_0, spl_1, _filter.data ()
103 );
104
105 return DataType (0.5f) * (spl_0 + spl_1);
106}

110/*
111==============================================================================
112Name: process_block
113Description:
114 Downsamples (x2) a block of samples.
115 Input and output blocks may overlap, see assert() for details.
116Input parameters:
117 - in_ptr: Input array, containing nbr_spl * 2 samples.
118 - nbr_spl: Number of samples to output, > 0
119Output parameters:
120 - out_ptr: Array for the output samples, capacity: nbr_spl samples.
121Throws: Nothing
122==============================================================================
123*/
124
125template <int NC, typename DT, int NCHN>
126void Downsampler2xTpl <NC, DT, NCHN>::process_block (DataType out_ptr [], const DataType in_ptr [], long nbr_spl) noexcept
127{
128 assert (in_ptr != nullptr);
129 assert (out_ptr != nullptr);
130 assert (out_ptr <= in_ptr || out_ptr >= in_ptr + nbr_spl * 2);
131 assert (nbr_spl > 0);
132
133 long pos = 0;
134 do
135 {
136 out_ptr [pos] = process_sample (&in_ptr [pos * 2]);
137 ++pos;
138 }
139 while (pos < nbr_spl);
140}

144/*
145==============================================================================
146Name: process_sample_split
147Description:
148 Split (spectrum-wise) in half a pair of samples. The lower part of the
149 spectrum is a classic downsampling, equivalent to the output of
150 process_sample().
151 The higher part is the complementary signal: original filter response
152 is flipped from left to right, becoming a high-pass filter with the same
153 cutoff frequency. This signal is then critically sampled (decimation by 2),
154 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
155Input parameters:
156 - in_ptr: pointer on the pair of input samples
157Output parameters:
158 - low: output sample, lower part of the spectrum (downsampling)
159 - high: output sample, higher part of the spectrum.
160Throws: Nothing
161==============================================================================
162*/
163
164template <int NC, typename DT, int NCHN>
165void Downsampler2xTpl <NC, DT, NCHN>::process_sample_split (DataType &low, DataType &high, const DataType in_ptr [2]) noexcept
166{
167 assert (in_ptr != nullptr);
168
169 DataType spl_0 = in_ptr [1];
170 DataType spl_1 = in_ptr [0];
171
172 StageProcTpl <NBR_COEFS, DataType>::process_sample_pos (
173 NBR_COEFS, spl_0, spl_1, _filter.data ()
174 );
175
176 low = (spl_0 + spl_1) * DataType (0.5f);
177 high = spl_0 - low; // (spl_0 - spl_1) * DataType (0.5f);
178}

182/*
183==============================================================================
184Name: process_block_split
185Description:
186 Split (spectrum-wise) in half a block of samples. The lower part of the
187 spectrum is a classic downsampling, equivalent to the output of
188 process_block().
189 The higher part is the complementary signal: original filter response
190 is flipped from left to right, becoming a high-pass filter with the same
191 cutoff frequency. This signal is then critically sampled (decimation by 2),
192 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
193 Input and output blocks may overlap, see assert() for details.
194Input parameters:
195 - in_ptr: Input array, containing nbr_spl * 2 samples.
196 - nbr_spl: Number of samples for each output, > 0
197Output parameters:
198 - out_l_ptr: Array for the output samples, lower part of the spectrum
199 (downsampling). Capacity: nbr_spl samples.
200 - out_h_ptr: Array for the output samples, higher part of the spectrum.
201 Capacity: nbr_spl samples.
202Throws: Nothing
203==============================================================================
204*/
205
206template <int NC, typename DT, int NCHN>
207void Downsampler2xTpl <NC, DT, NCHN>::process_block_split (DataType out_l_ptr [], DataType out_h_ptr [], const DataType in_ptr [], long nbr_spl) noexcept
208{
209 assert (in_ptr != nullptr);
210 assert (out_l_ptr != nullptr);
211 assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl * 2);
212 assert (out_h_ptr != nullptr);
213 assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl * 2);
214 assert (out_h_ptr != out_l_ptr);
215 assert (nbr_spl > 0);
216
217 long pos = 0;
218 do
219 {
220 process_sample_split (
221 out_l_ptr [pos],
222 out_h_ptr [pos],
223 &in_ptr [pos * 2]
224 );
225 ++pos;
226 }
227 while (pos < nbr_spl);
228}
