59auto decodeSampleSF(const QString& audioFile) -> std::optional<SampleDecoder::Result>
60{
61 SNDFILE* sndFile = nullptr;
62 auto sfInfo = SF_INFO{};
63
64 // TODO: Remove use of QFile
65 auto file = QFile{audioFile};
66 if (!file.open(QIODevice::ReadOnly)) { return std::nullopt; }
67
68 sndFile = sf_open_fd(file.handle(), SFM_READ, &sfInfo, false);
69 if (sf_error(sndFile) != 0) { return std::nullopt; }
70
71 auto buf = std::vector<sample_t>(sfInfo.channels * sfInfo.frames);
72 sf_read_float(sndFile, buf.data(), buf.size());
73
74 sf_close(sndFile);
75 file.close();
76
77 auto result = std::vector<SampleFrame>(sfInfo.frames);
78 for (int i = 0; i < static_cast<int>(result.size()); ++i)
79 {
80 if (sfInfo.channels == 1)
81 {
82 // Upmix from mono to stereo
83 result[i] = {buf[i], buf[i]};
84 }
85 else if (sfInfo.channels > 1)
86 {
87 // TODO: Add support for higher number of channels (i.e., 5.1 channel systems)
88 // The current behavior assumes stereo in all cases excluding mono.
89 // This may not be the expected behavior, given some audio files with a higher number of channels.
90 result[i] = {buf[i * sfInfo.channels], buf[i * sfInfo.channels + 1]};
91 }
92 }
93
94 return SampleDecoder::Result{std::move(result), static_cast<int>(sfInfo.samplerate)};
95}

97auto decodeSampleDS(const QString& audioFile) -> std::optional<SampleDecoder::Result>
98{
99 // Populated by DrumSynth::GetDSFileSamples
100 int_sample_t* dataPtr = nullptr;
101
102 auto ds = DrumSynth{};
103 const auto engineRate = Engine::audioEngine()->outputSampleRate();
104 const auto frames = ds.GetDSFileSamples(audioFile, dataPtr, DEFAULT_CHANNELS, engineRate);
105 const auto data = std::unique_ptr<int_sample_t[]>{dataPtr}; // NOLINT, we have to use a C-style array here
106
107 if (frames <= 0 || !data) { return std::nullopt; }
108
109 auto result = std::vector<SampleFrame>(frames);
110 src_short_to_float_array(data.get(), &result[0][0], frames * DEFAULT_CHANNELS);
111
112 return SampleDecoder::Result{std::move(result), static_cast<int>(engineRate)};
113}

115#ifdef LMMS_HAVE_OGGVORBIS
116auto decodeSampleOggVorbis(const QString& audioFile) -> std::optional<SampleDecoder::Result>
117{
118 static auto s_read = [](void* buffer, size_t size, size_t count, void* stream) -> size_t {
119 auto file = static_cast<QFile*>(stream);
120 return file->read(static_cast<char*>(buffer), size * count);
121 };
122
123 static auto s_seek = [](void* stream, ogg_int64_t offset, int whence) -> int {
124 auto file = static_cast<QFile*>(stream);
125 if (whence == SEEK_SET) { file->seek(offset); }
126 else if (whence == SEEK_CUR) { file->seek(file->pos() + offset); }
127 else if (whence == SEEK_END) { file->seek(file->size() + offset); }
128 else { return -1; }
129 return 0;
130 };
131
132 static auto s_close = [](void* stream) -> int {
133 auto file = static_cast<QFile*>(stream);
134 file->close();
135 return 0;
136 };
137
138 static auto s_tell = [](void* stream) -> long {
139 auto file = static_cast<QFile*>(stream);
140 return file->pos();
141 };
142
143 static ov_callbacks s_callbacks = {s_read, s_seek, s_close, s_tell};
144
145 // TODO: Remove use of QFile
146 auto file = QFile{audioFile};
147 if (!file.open(QIODevice::ReadOnly)) { return std::nullopt; }
148
149 auto vorbisFile = OggVorbis_File{};
150 if (ov_open_callbacks(&file, &vorbisFile, nullptr, 0, s_callbacks) < 0) { return std::nullopt; }
151
152 const auto vorbisInfo = ov_info(&vorbisFile, -1);
153 if (vorbisInfo == nullptr) { return std::nullopt; }
154
155 const auto numChannels = vorbisInfo->channels;
156 const auto sampleRate = vorbisInfo->rate;
157 const auto numSamples = ov_pcm_total(&vorbisFile, -1);
158 if (numSamples < 0) { return std::nullopt; }
159
160 auto buffer = std::vector<float>(numSamples);
161 auto output = static_cast<float**>(nullptr);
162
163 auto totalSamplesRead = 0;
164 while (true)
165 {
166 auto samplesRead = ov_read_float(&vorbisFile, &output, numSamples, 0);
167
168 if (samplesRead < 0) { return std::nullopt; }
169 else if (samplesRead == 0) { break; }
170
171 std::copy_n(*output, samplesRead, buffer.begin() + totalSamplesRead);
172 totalSamplesRead += samplesRead;
173 }
174
175 auto result = std::vector<SampleFrame>(totalSamplesRead / numChannels);
176 for (auto i = std::size_t{0}; i < result.size(); ++i)
177 {
178 if (numChannels == 1) { result[i] = {buffer[i], buffer[i]}; }
179 else if (numChannels > 1) { result[i] = {buffer[i * numChannels], buffer[i * numChannels + 1]}; }
180 }
181
182 ov_clear(&vorbisFile);
183 return SampleDecoder::Result{std::move(result), static_cast<int>(sampleRate)};
184}
185#endif // LMMS_HAVE_OGGVORBIS

188auto SampleDecoder::supportedAudioTypes() -> const std::vector<AudioType>&
189{
190 static const auto s_audioTypes = [] {
191 auto types = std::vector<AudioType>();
192
193 // Add DrumSynth by default since that support comes from us
194 types.push_back(AudioType{"DrumSynth", "ds"});
195
196 auto sfFormatInfo = SF_FORMAT_INFO{};
197 auto simpleTypeCount = 0;
198 sf_command(nullptr, SFC_GET_SIMPLE_FORMAT_COUNT, &simpleTypeCount, sizeof(int));
199
200 // TODO: Ideally, this code should be iterating over the major formats, but some important extensions such as
201 // *.ogg are not included. This is planned for future versions of sndfile.
202 for (int simple = 0; simple < simpleTypeCount; ++simple)
203 {
204 sfFormatInfo.format = simple;
205 sf_command(nullptr, SFC_GET_SIMPLE_FORMAT, &sfFormatInfo, sizeof(sfFormatInfo));
206
207 auto it = std::find_if(types.begin(), types.end(),
208 [&](const AudioType& type) { return sfFormatInfo.extension == type.extension; });
209 if (it != types.end()) { continue; }
210
211 auto name = std::string{sfFormatInfo.extension};
212 std::transform(name.begin(), name.end(), name.begin(), [](unsigned char ch) { return std::toupper(ch); });
213
214 types.push_back(AudioType{std::move(name), sfFormatInfo.extension});
215 }
216
217 std::sort(types.begin(), types.end(), [&](const AudioType& a, const AudioType& b) { return a.name < b.name; });
218 return types;
219 }();
220 return s_audioTypes;
221}

223auto SampleDecoder::decode(const QString& audioFile) -> std::optional<Result>
224{
225 auto result = std::optional<Result>{};
226 for (const auto& decoder : decoders)
227 {
228 result = decoder(audioFile);
229 if (result) { break; }
230 }
231
232 return result;
233}
