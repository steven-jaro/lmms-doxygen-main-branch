106int JackClient::Close()
107{
108 jack_log("JackClient::Close ref = %ld", GetClientControl()->fRefNum);
109 int result = 0;
110
111 Deactivate();
112
113 // Channels is stopped first to avoid receiving notifications while closing
114 fChannel->Stop();
115 // Then close client
116 fChannel->ClientClose(GetClientControl()->fRefNum, &result);
117
118 fChannel->Close();
119 assert(JackGlobals::fSynchroMutex);
120 JackGlobals::fSynchroMutex->Lock();
121 fSynchroTable[GetClientControl()->fRefNum].Disconnect();
122 JackGlobals::fSynchroMutex->Unlock();
123 JackGlobals::fClientTable[GetClientControl()->fRefNum] = NULL;
124 return result;
125}

137/*!
138 In "async" mode, the server does not synchronize itself on the output drivers, thus it would never "consume" the activations.
139 The synchronization primitives for drivers are setup in "flush" mode that to not keep unneeded activations.
140 Drivers synchro are setup in "flush" mode if server is "async" and NOT freewheel.
141*/
142void JackClient::SetupDriverSync(bool freewheel)
143{
144 if (!freewheel && !GetEngineControl()->fSyncMode) {
145 jack_log("JackClient::SetupDriverSync driver sem in flush mode");
146 for (int i = 0; i < GetEngineControl()->fDriverNum; i++) {
147 fSynchroTable[i].SetFlush(true);
148 }
149 } else {
150 jack_log("JackClient::SetupDriverSync driver sem in normal mode");
151 for (int i = 0; i < GetEngineControl()->fDriverNum; i++) {
152 fSynchroTable[i].SetFlush(false);
153 }
154 }
155}

166int JackClient::ClientNotify(int refnum, const char* name, int notify, int sync, const char* message, int value1, int value2)
167{
168 int res = 0;
169
170 jack_log("JackClient::ClientNotify ref = %ld name = %s notify = %ld", refnum, name, notify);
171
172 // Done all time: redirected on subclass implementation JackLibClient and JackInternalClient
173 switch (notify) {
174
175 case kAddClient:
176 res = ClientNotifyImp(refnum, name, notify, sync, message, value1, value2);
177 break;
178
179 case kRemoveClient:
180 res = ClientNotifyImp(refnum, name, notify, sync, message, value1, value2);
181 break;
182
183 case kActivateClient:
184 jack_log("JackClient::kActivateClient name = %s ref = %ld ", name, refnum);
185 InitAux();
186 break;
187 }
188
189 /*
190 The current semantic is that notifications can only be received when the client has been activated,
191 although is this implementation, one could imagine calling notifications as soon as the client has be opened.
192 */
193 if (IsActive()) {
194
195 switch (notify) {
196
197 case kAddClient:
198 jack_log("JackClient::kAddClient fName = %s name = %s", GetClientControl()->fName, name);
199 if (fClientRegistration && strcmp(GetClientControl()->fName, name) != 0) { // Don't call the callback for the registering client itself
200 fClientRegistration(name, 1, fClientRegistrationArg);
201 }
202 break;
203
204 case kRemoveClient:
205 jack_log("JackClient::kRemoveClient fName = %s name = %s", GetClientControl()->fName, name);
206 if (fClientRegistration && strcmp(GetClientControl()->fName, name) != 0) { // Don't call the callback for the registering client itself
207 fClientRegistration(name, 0, fClientRegistrationArg);
208 }
209 break;
210
211 case kBufferSizeCallback:
212 jack_log("JackClient::kBufferSizeCallback buffer_size = %ld", value1);
213 if (fBufferSize) {
214 res = fBufferSize(value1, fBufferSizeArg);
215 }
216 break;
217
218 case kSampleRateCallback:
219 jack_log("JackClient::kSampleRateCallback sample_rate = %ld", value1);
220 if (fSampleRate) {
221 res = fSampleRate(value1, fSampleRateArg);
222 }
223 break;
224
225 case kGraphOrderCallback:
226 jack_log("JackClient::kGraphOrderCallback");
227 if (fGraphOrder) {
228 res = fGraphOrder(fGraphOrderArg);
229 }
230 break;
231
232 case kStartFreewheelCallback:
233 jack_log("JackClient::kStartFreewheel");
234 SetupDriverSync(true);
235 // Drop RT only when the RT thread is actually running
236 if (fThread.GetStatus() == JackThread::kRunning) {
237 fThread.DropRealTime();
238 }
239 if (fFreewheel) {
240 fFreewheel(1, fFreewheelArg);
241 }
242 break;
243
244 case kStopFreewheelCallback:
245 jack_log("JackClient::kStopFreewheel");
246 SetupDriverSync(false);
247 if (fFreewheel) {
248 fFreewheel(0, fFreewheelArg);
249 }
250 // Acquire RT only when the RT thread is actually running
251 if (GetEngineControl()->fRealTime && fThread.GetStatus() == JackThread::kRunning) {
252 if (fThread.AcquireRealTime(GetEngineControl()->fClientPriority) < 0) {
253 jack_error("JackClient::AcquireRealTime error");
254 }
255 }
256 break;
257
258 case kPortRegistrationOnCallback:
259 jack_log("JackClient::kPortRegistrationOn port_index = %ld", value1);
260 if (fPortRegistration) {
261 fPortRegistration(value1, 1, fPortRegistrationArg);
262 }
263 break;
264
265 case kPortRegistrationOffCallback:
266 jack_log("JackClient::kPortRegistrationOff port_index = %ld ", value1);
267 if (fPortRegistration) {
268 fPortRegistration(value1, 0, fPortRegistrationArg);
269 }
270 break;
271
272 case kPortConnectCallback:
273 jack_log("JackClient::kPortConnectCallback src = %ld dst = %ld", value1, value2);
274 if (fPortConnect) {
275 fPortConnect(value1, value2, 1, fPortConnectArg);
276 }
277 break;
278
279 case kPortDisconnectCallback:
280 jack_log("JackClient::kPortDisconnectCallback src = %ld dst = %ld", value1, value2);
281 if (fPortConnect) {
282 fPortConnect(value1, value2, 0, fPortConnectArg);
283 }
284 break;
285
286 case kPortRenameCallback:
287 jack_log("JackClient::kPortRenameCallback port = %ld", value1);
288 if (fPortRename) {
289 fPortRename(value1, message, GetGraphManager()->GetPort(value1)->GetName(), fPortRenameArg);
290 }
291 break;
292
293 case kXRunCallback:
294 jack_log("JackClient::kXRunCallback");
295 if (fXrun) {
296 res = fXrun(fXrunArg);
297 }
298 break;
299
300 case kShutDownCallback:
301 jack_log("JackClient::kShutDownCallback");
302 ShutDown(jack_status_t(value1), message);
303 break;
304
305 case kSessionCallback:
306 jack_log("JackClient::kSessionCallback");
307 if (fSession) {
308 jack_session_event_t* event = (jack_session_event_t*)malloc( sizeof(jack_session_event_t));
309 char uuid_buf[JACK_UUID_STRING_SIZE];
310 event->type = (jack_session_event_type_t)value1;
311 event->session_dir = strdup(message);
312 event->command_line = NULL;
313 event->flags = (jack_session_flags_t)0;
314 jack_uuid_unparse(GetClientControl()->fSessionID, uuid_buf);
315 event->client_uuid = strdup(uuid_buf);
316 fSessionReply = kPendingSessionReply;
317 // Session callback may change fSessionReply by directly using jack_session_reply
318 fSession(event, fSessionArg);
319 res = fSessionReply;
320 }
321 break;
322
323 case kLatencyCallback:
324 res = HandleLatencyCallback(value1);
325 break;
326
327 case kPropertyChangeCallback: {
328 jack_uuid_t subject;
329 jack_uuid_parse(name, &subject);
330 const char* key = message;
331 jack_property_change_t change = (jack_property_change_t)value1;
332 jack_log("JackClient::kPropertyChangeCallback subject = %x key = %s change = %x", subject, key, change);
333 if (fPropertyChange)
334 fPropertyChange(subject, key, change, fPropertyChangeArg);
335 break;
336 }
337 }
338 }
339
340 return res;
341}

343int JackClient::HandleLatencyCallback(int status)
344{
345 jack_latency_callback_mode_t mode = (status == 0) ? JackCaptureLatency : JackPlaybackLatency;
346 jack_latency_range_t latency = { UINT32_MAX, 0 };
347
348 /* first setup all latency values of the ports.
349 * this is based on the connections of the ports.
350 */
351 list<jack_port_id_t>::iterator it;
352
353 for (it = fPortList.begin(); it != fPortList.end(); it++) {
354 JackPort* port = GetGraphManager()->GetPort(*it);
355 if ((port->GetFlags() & JackPortIsOutput) && (mode == JackPlaybackLatency)) {
356 GetGraphManager()->RecalculateLatency(*it, mode);
357 }
358 if ((port->GetFlags() & JackPortIsInput) && (mode == JackCaptureLatency)) {
359 GetGraphManager()->RecalculateLatency(*it, mode);
360 }
361 }
362
363 if (!fLatency) {
364 /*
365 * default action is to assume all ports depend on each other.
366 * then always take the maximum latency.
367 */
368
369 if (mode == JackPlaybackLatency) {
370 /* iterate over all OutputPorts, to find maximum playback latency
371 */
372 for (it = fPortList.begin(); it != fPortList.end(); it++) {
373 JackPort* port = GetGraphManager()->GetPort(*it);
374 if (port->GetFlags() & JackPortIsOutput) {
375 jack_latency_range_t other_latency;
376 port->GetLatencyRange(mode, &other_latency);
377 if (other_latency.max > latency.max) {
378 latency.max = other_latency.max;
379 }
380 if (other_latency.min < latency.min) {
381 latency.min = other_latency.min;
382 }
383 }
384 }
385
386 if (latency.min == UINT32_MAX) {
387 latency.min = 0;
388 }
389
390 /* now set the found latency on all input ports
391 */
392 for (it = fPortList.begin(); it != fPortList.end(); it++) {
393 JackPort* port = GetGraphManager()->GetPort(*it);
394 if (port->GetFlags() & JackPortIsInput) {
395 port->SetLatencyRange(mode, &latency);
396 }
397 }
398 }
399 if (mode == JackCaptureLatency) {
400 /* iterate over all InputPorts, to find maximum playback latency
401 */
402 for (it = fPortList.begin(); it != fPortList.end(); it++) {
403 JackPort* port = GetGraphManager()->GetPort(*it);
404 if (port->GetFlags() & JackPortIsInput) {
405 jack_latency_range_t other_latency;
406 port->GetLatencyRange(mode, &other_latency);
407 if (other_latency.max > latency.max) {
408 latency.max = other_latency.max;
409 }
410 if (other_latency.min < latency.min) {
411 latency.min = other_latency.min;
412 }
413 }
414 }
415
416 if (latency.min == UINT32_MAX) {
417 latency.min = 0;
418 }
419
420 /* now set the found latency on all output ports
421 */
422 for (it = fPortList.begin(); it != fPortList.end(); it++) {
423 JackPort* port = GetGraphManager()->GetPort(*it);
424 if (port->GetFlags() & JackPortIsOutput) {
425 port->SetLatencyRange(mode, &latency);
426 }
427 }
428 }
429 return 0;
430 }
431
432 /* we have a latency callback setup by the client,
433 * lets use it...
434 */
435 fLatency(mode, fLatencyArg);
436 return 0;
437}

439/*!
440\brief We need to start thread before activating in the server, otherwise the FW driver
441connected to the client may not be activated.
442*/
443int JackClient::Activate()
444{
445 jack_log("JackClient::Activate");
446 if (IsActive()) {
447 return 0;
448 }
449
450 // RT thread is started only when needed...
451 if (IsRealTime()) {
452 if (StartThread() < 0) {
453 return -1;
454 }
455 }
456
457 /*
458 Insertion of client in the graph will cause a kGraphOrderCallback notification
459 to be delivered by the server, the client wants to receive it.
460 */
461 GetClientControl()->fActive = true;
462
463 // Transport related callback become "active"
464 GetClientControl()->fTransportSync = true;
465 GetClientControl()->fTransportTimebase = true;
466
467 int result = -1;
468 GetClientControl()->fCallback[kRealTimeCallback] = IsRealTime();
469 fChannel->ClientActivate(GetClientControl()->fRefNum, IsRealTime(), &result);
470 return result;
471}

473/*!
474\brief Need to stop thread after deactivating in the server.
475*/
476int JackClient::Deactivate()
477{
478 jack_log("JackClient::Deactivate");
479 if (!IsActive()) {
480 return 0;
481 }
482
483 GetClientControl()->fActive = false;
484
485 // Transport related callback become "unactive"
486 GetClientControl()->fTransportSync = false;
487 GetClientControl()->fTransportTimebase = false;
488
489 // We need to wait for the new engine cycle before stopping the RT thread, but this is done by ClientDeactivate
490 int result = -1;
491 fChannel->ClientDeactivate(GetClientControl()->fRefNum, &result);
492 jack_log("JackClient::Deactivate res = %ld", result);
493
494 // RT thread is stopped only when needed...
495 if (IsRealTime()) {
496 fThread.Kill();
497 }
498 return result;
499}

513/*!
514\brief Called once when the thread starts.
515*/
516bool JackClient::Init()
517{
518 /*
519 Execute buffer_size callback.
520
521 Since StartThread uses fThread.StartSync, we are sure that buffer_size callback
522 is executed before StartThread returns (and then IsActive will be true).
523 So no RT callback can be called at the same time.
524 */
525 jack_log("JackClient::kBufferSizeCallback buffer_size = %ld", GetEngineControl()->fBufferSize);
526 if (fBufferSize) {
527 fBufferSize(GetEngineControl()->fBufferSize, fBufferSizeArg);
528 }
529
530 // Init callback
531 InitAux();
532
533 // Setup context
534 if (!jack_tls_set(JackGlobals::fRealTimeThread, this)) {
535 jack_error("Failed to set thread realtime key");
536 }
537
538 // Setup RT
539 if (GetEngineControl()->fRealTime) {
540 set_threaded_log_function();
541 SetupRealTime();
542 }
543
544 return true;
545}

639{
640 // Suspend itself: wait on the input synchro

642#ifdef __APPLE__

647 }

649 jack_error("SuspendRefNum error");

652 return true;
653 }

691int JackClient::PortRegister(const char* port_name, const char* port_type, unsigned long flags, unsigned long buffer_size)
692{
693 // Check if port name is empty
694 string port_short_name_str = string(port_name);
695 if (port_short_name_str.size() == 0) {
696 jack_error("port_name is empty");
697 return 0; // Means failure here...
698 }
699
700 // Check port name length
701 string port_full_name_str = string(GetClientControl()->fName) + string(":") + port_short_name_str;
702 if (port_full_name_str.size() >= REAL_JACK_PORT_NAME_SIZE) {
703 jack_error("\"%s:%s\" is too long to be used as a JACK port name.\n"
704 "Please use %lu characters or less",
705 GetClientControl()->fName,
706 port_name,
707 JACK_PORT_NAME_SIZE - 1);
708 return 0; // Means failure here...
709 }
710
711 int result = -1;
712 jack_port_id_t port_index = NO_PORT;
713 fChannel->PortRegister(GetClientControl()->fRefNum, port_full_name_str.c_str(), port_type, flags, buffer_size, &port_index, &result);
714
715 if (result == 0) {
716 jack_log("JackClient::PortRegister ref = %ld name = %s type = %s port_index = %ld", GetClientControl()->fRefNum, port_full_name_str.c_str(), port_type, port_index);
717 fPortList.push_back(port_index);
718 return port_index;
719 } else {
720 return 0;
721 }
722}

969 int master;
970 bool unused;

973
