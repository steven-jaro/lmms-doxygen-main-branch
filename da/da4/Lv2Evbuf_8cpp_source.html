43struct LV2_Evbuf_Impl {
44 uint32_t capacity;
45 uint32_t atom_Chunk;
46 uint32_t atom_Sequence;
47 LV2_Atom_Sequence buf;

50static inline uint32_t
51lv2_evbuf_pad_size(uint32_t size)
52{
53 return (size + 7) & (~7);
54}
55
56LV2_Evbuf*
57lv2_evbuf_new(uint32_t capacity, uint32_t atom_Chunk, uint32_t atom_Sequence)
58{
59 // FIXME: memory must be 64-bit aligned
60 auto evbuf = (LV2_Evbuf*)malloc(sizeof(LV2_Evbuf) + sizeof(LV2_Atom_Sequence) + capacity);
61 evbuf->capacity = capacity;
62 evbuf->atom_Chunk = atom_Chunk;
63 evbuf->atom_Sequence = atom_Sequence;
64 lv2_evbuf_reset(evbuf, true);
65 return evbuf;
66}

68void
69lv2_evbuf_free(LV2_Evbuf* evbuf)
70{
71 free(evbuf);
72}

74void
75lv2_evbuf_reset(LV2_Evbuf* evbuf, bool input)
76{
77 if (input) {
78 evbuf->buf.atom.size = sizeof(LV2_Atom_Sequence_Body);
79 evbuf->buf.atom.type = evbuf->atom_Sequence;
80 } else {
81 evbuf->buf.atom.size = evbuf->capacity;
82 evbuf->buf.atom.type = evbuf->atom_Chunk;
83 }
84}

86uint32_t
87lv2_evbuf_get_size(LV2_Evbuf* evbuf)
88{
89 assert(evbuf->buf.atom.type != evbuf->atom_Sequence
90 || evbuf->buf.atom.size >= sizeof(LV2_Atom_Sequence_Body));
91 return evbuf->buf.atom.type == evbuf->atom_Sequence
92 ? evbuf->buf.atom.size - sizeof(LV2_Atom_Sequence_Body)
93 : 0;
94}

109LV2_Evbuf_Iterator
110lv2_evbuf_end(LV2_Evbuf* evbuf)
111{

117bool
118lv2_evbuf_is_valid(LV2_Evbuf_Iterator iter)
119{
120 return iter.offset < lv2_evbuf_get_size(iter.evbuf);
121}
122
123LV2_Evbuf_Iterator
124lv2_evbuf_next(LV2_Evbuf_Iterator iter)
125{
126 if (!lv2_evbuf_is_valid(iter)) {
127 return iter;
128 }
129
130 LV2_Evbuf* evbuf = iter.evbuf;
131 uint32_t offset = iter.offset;
132
133 const auto contents = static_cast<char*>(LV2_ATOM_CONTENTS(LV2_Atom_Sequence, &evbuf->buf.atom)) + offset;
134 const uint32_t size = reinterpret_cast<LV2_Atom_Event*>(contents)->body.size;
135
136 offset += lv2_evbuf_pad_size(sizeof(LV2_Atom_Event) + size);
137 LV2_Evbuf_Iterator next = { evbuf, offset };
138 return next;
139}

141bool
142lv2_evbuf_get(LV2_Evbuf_Iterator iter,
143 uint32_t* frames,
144 uint32_t* type,
145 uint32_t* size,
146 uint8_t** data)
147{
148 *frames = *type = *size = 0;
149 *data = nullptr;
150
151 if (!lv2_evbuf_is_valid(iter)) {
152 return false;
153 }
154
155 LV2_Atom_Sequence* aseq = &iter.evbuf->buf;
156 auto aev = (LV2_Atom_Event*)((char*)LV2_ATOM_CONTENTS(LV2_Atom_Sequence, aseq) + iter.offset);
157
158 *frames = aev->time.frames;
159 *type = aev->body.type;
160 *size = aev->body.size;
161 *data = (uint8_t*)LV2_ATOM_BODY(&aev->body);
162
163 return true;
164}
165
166bool
167lv2_evbuf_write(LV2_Evbuf_Iterator* iter,
168 uint32_t frames,
169 uint32_t type,
170 uint32_t size,
171 const uint8_t* data)
172{
173 LV2_Atom_Sequence* aseq = &iter->evbuf->buf;
174 if (iter->evbuf->capacity - sizeof(LV2_Atom) - aseq->atom.size <
175 sizeof(LV2_Atom_Event) + size) {
176 return false;
177 }
178
179 auto aev = (LV2_Atom_Event*)((char*)LV2_ATOM_CONTENTS(LV2_Atom_Sequence, aseq) + iter->offset);
180
181 aev->time.frames = frames;
182 aev->body.type = type;
183 aev->body.size = size;
184 memcpy(LV2_ATOM_BODY(&aev->body), data, size);
185
186 size = lv2_evbuf_pad_size(sizeof(LV2_Atom_Event) + size);
187 aseq->atom.size += size;
188 iter->offset += size;
189
190 return true;
191}
