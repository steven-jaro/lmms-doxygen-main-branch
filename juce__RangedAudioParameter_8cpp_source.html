29int RangedAudioParameter::getNumSteps() const
30{
31 const auto& range = getNormalisableRange();
32
33 if (range.interval > 0)
34 return (static_cast<int> ((range.end - range.start) / range.interval) + 1);
35
36 return AudioProcessor::getDefaultNumParameterSteps();
37}
