89{

91
92 for (int i = 0; i < NBR_COEFS; ++i)

102/*
103==============================================================================
104Name: process_sample
105Description:
106 Downsamples (x2) one pair of samples, to generate one output sample.
107Input parameters:
108 - in_ptr: pointer on the two samples to decimate
109Returns: Samplerate-reduced sample.
110Throws: Nothing
111==============================================================================
112*/
113
114template <int NC>
115float Downsampler2xSse <NC>::process_sample (const float in_ptr [2]) noexcept
116{
117 assert (in_ptr != nullptr);
118
119 auto x = _mm_loadl_pi (
120 _mm_setzero_ps (), reinterpret_cast <const __m64 *> (in_ptr)
121 );
122 StageProcSseV2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
123 x = _mm_add_ss (x, _mm_shuffle_ps (x, x, 1));
124 x = _mm_mul_ss (x, _mm_set_ss (0.5f));
125
126 return _mm_cvtss_f32 (x);
127}

131/*
132==============================================================================
133Name: process_block
134Description:
135 Downsamples (x2) a block of samples.
136 Input and output blocks may overlap, see assert() for details.
137Input parameters:
138 - in_ptr: Input array, containing nbr_spl * 2 samples.
139 - nbr_spl: Number of samples to output, > 0
140Output parameters:
141 - out_ptr: Array for the output samples, capacity: nbr_spl samples.
142Throws: Nothing
143==============================================================================
144*/
145
146template <int NC>
147void Downsampler2xSse <NC>::process_block (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
148{
149 assert (in_ptr != nullptr);
150 assert (out_ptr != nullptr);
151 assert (out_ptr <= in_ptr || out_ptr >= in_ptr + nbr_spl * 2);
152 assert (nbr_spl > 0);
153
154#if ! defined (_MSC_VER)
155 // This version is slower than copy-pasta when compiled with MSVC
156 const long n4 = process_block_quad (
157 out_ptr, nullptr, in_ptr, nbr_spl, store_low, bypass
158 );
159
160 const auto half = _mm_set1_ps (0.5f);
161#else
162 const auto half = _mm_set1_ps (0.5f);
163
164 const long n4 = nbr_spl & ~(4-1);
165 for (long pos = 0; pos < n4; pos += 4)
166 {
167 auto x_0 = _mm_loadu_ps (in_ptr + pos * 2);
168 auto x_1 = _mm_movehl_ps (x_0, x_0);
169 auto x_2 = _mm_loadu_ps (in_ptr + pos * 2 + 4);
170 auto x_3 = _mm_movehl_ps (x_2, x_2);
171 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_0, _filter.data ());
172 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_1, _filter.data ());
173 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_2, _filter.data ());
174 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_3, _filter.data ());
175 const auto u_01 = _mm_unpacklo_ps (x_0, x_1); // o0, o1, e0, e1
176 const auto u_23 = _mm_unpacklo_ps (x_2, x_3); // o2, o3, e2, e3
177 const auto odd = _mm_movelh_ps (u_01, u_23);
178 const auto even = _mm_movehl_ps (u_23, u_01);
179 const auto y = _mm_mul_ps (_mm_add_ps (even, odd), half);
180 _mm_storeu_ps (out_ptr + pos, y);
181 }
182
183#endif
184 for (long pos = n4; pos < nbr_spl; ++pos)
185 {
186 auto x = _mm_loadl_pi (
187 half, reinterpret_cast <const __m64 *> (in_ptr + pos * 2)
188 );
189 StageProcSseV2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
190 x = _mm_add_ss (x, _mm_shuffle_ps (x, x, 1));
191 x = _mm_mul_ss (x, half);
192 out_ptr [pos] = _mm_cvtss_f32 (x);
193 }
194}

198/*
199==============================================================================
200Name: process_sample_split
201Description:
202 Split (spectrum-wise) in half a pair of samples. The lower part of the
203 spectrum is a classic downsampling, equivalent to the output of
204 process_sample().
205 The higher part is the complementary signal: original filter response
206 is flipped from left to right, becoming a high-pass filter with the same
207 cutoff frequency. This signal is then critically sampled (decimation by 2),
208 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
209Input parameters:
210 - in_ptr: pointer on the pair of input samples
211Output parameters:
212 - low: output sample, lower part of the spectrum (downsampling)
213 - high: output sample, higher part of the spectrum.
214Throws: Nothing
215==============================================================================
216*/
217
218template <int NC>
219void Downsampler2xSse <NC>::process_sample_split (float &low, float &high, const float in_ptr [2]) noexcept
220{
221 assert (in_ptr != nullptr);
222
223 auto x = _mm_loadl_pi (
224 _mm_setzero_ps (), reinterpret_cast <const __m64 *> (in_ptr)
225 );
226 StageProcSseV2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
227 x = _mm_mul_ps (x, _mm_set1_ps (0.5f));
228 const auto xr = _mm_shuffle_ps (x, x, 1);
229 low = _mm_cvtss_f32 (_mm_add_ss (xr, x));
230 high = _mm_cvtss_f32 (_mm_sub_ss (xr, x));
231}

236==============================================================================
237Name: process_block_split
238Description:
239 Split (spectrum-wise) in half a block of samples. The lower part of the
240 spectrum is a classic downsampling, equivalent to the output of
241 process_block().
242 The higher part is the complementary signal: original filter response
243 is flipped from left to right, becoming a high-pass filter with the same
244 cutoff frequency. This signal is then critically sampled (decimation by 2),
245 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
246 Input and output blocks may overlap, see assert() for details.
247Input parameters:
248 - in_ptr: Input array, containing nbr_spl * 2 samples.
249 Make sure that the 8 bytes following the end of the block are valid
250 readable memory locations
251 - nbr_spl: Number of samples for each output, > 0
252Output parameters:
253 - out_l_ptr: Array for the output samples, lower part of the spectrum
254 (downsampling). Capacity: nbr_spl samples.
255 - out_h_ptr: Array for the output samples, higher part of the spectrum.
256 Capacity: nbr_spl samples.
257Throws: Nothing
258==============================================================================
259*/
260
261template <int NC>
262void Downsampler2xSse <NC>::process_block_split (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl) noexcept
263{
264 assert (in_ptr != nullptr);
265 assert (out_l_ptr != nullptr);
266 assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl * 2);
267 assert (out_h_ptr != nullptr);
268 assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl * 2);
269 assert (out_h_ptr != out_l_ptr);
270 assert (nbr_spl > 0);
271
272#if ! defined (_MSC_VER)
273 // This version is slower than copy-pasta when compiled with MSVC
274 const long n4 = process_block_quad (
275 out_l_ptr, out_h_ptr, in_ptr, nbr_spl, store_low, store_high
276 );
277
278 const auto half = _mm_set1_ps (0.5f);
279#else
280 const auto half = _mm_set1_ps (0.5f);
281
282 const long n4 = nbr_spl & ~(4-1);
283 for (long pos = 0; pos < n4; pos += 4)
284 {
285 auto x_0 = _mm_loadu_ps (in_ptr + pos * 2);
286 auto x_1 = _mm_movehl_ps (x_0, x_0);
287 auto x_2 = _mm_loadu_ps (in_ptr + pos * 2 + 4);
288 auto x_3 = _mm_movehl_ps (x_2, x_2);
289 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_0, _filter.data ());
290 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_1, _filter.data ());
291 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_2, _filter.data ());
292 StageProcSseV2 <_nbr_stages>::process_sample_pos (x_3, _filter.data ());
293 const auto u_01 = _mm_unpacklo_ps (x_0, x_1); // o0, o1, e0, e1
294 const auto u_23 = _mm_unpacklo_ps (x_2, x_3); // o2, o3, e2, e3
295 const auto odd = _mm_movelh_ps (u_01, u_23);
296 const auto even = _mm_movehl_ps (u_23, u_01);
297 const auto low = _mm_mul_ps (_mm_add_ps (even, odd), half);
298 const auto high = _mm_mul_ps (_mm_sub_ps (even, odd), half);
299 _mm_storeu_ps (out_l_ptr + pos, low );
300 _mm_storeu_ps (out_h_ptr + pos, high);
301 }
302
303#endif
304 for (long pos = n4; pos < nbr_spl; ++pos)
305 {
306 auto x = _mm_loadl_pi (
307 half, reinterpret_cast <const __m64 *> (in_ptr + pos * 2)
308 );
309 StageProcSseV2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
310 x = _mm_mul_ps (x, half);
311 const auto xr = _mm_shuffle_ps (x, x, 1);
312 out_l_ptr [pos] = _mm_cvtss_f32 (_mm_add_ss (xr, x));
313 out_h_ptr [pos] = _mm_cvtss_f32 (_mm_sub_ss (xr, x));
314 }
315}
