92template <typename It>
93using HandlerMap =
94 std::unordered_map<typename PayloadEntryHeader::Key, std::function<void(It, It)>>;
95
96// Given an index of handlers and a byte range, parse the bytes as a
97// sequence of payload entries and invoke the appropriate handler for
98// each entry type. Entries that are encountered that do not have a
99// corresponding handler in the map are ignored. Throws
100// std::runtime_error if parsing fails for any entry. Note that if an
101// exception is thrown, some of the handlers may have already been called.
102template <typename It>
103void parseByteStream(HandlerMap<It>& map, It bsBegin, const It bsEnd)
104{
105 using namespace std;
106
107 while (bsBegin < bsEnd)
108 {
109 // Try to parse an entry header at this location in the byte stream
110 PayloadEntryHeader header;
111 It valueBegin;
112 tie(header, valueBegin) =
113 Deserialize<PayloadEntryHeader>::fromNetworkByteStream(bsBegin, bsEnd);
114
115 // Ensure that the reported size of the entry does not exceed the
116 // length of the byte stream
117 It valueEnd = valueBegin + header.size;
118 if (bsEnd < valueEnd)
119 {
120 throw range_error("Payload with incorrect size.");
121 }
122
123 // The next entry will start at the end of this one
124 bsBegin = valueEnd;
125
126 // Use the appropriate handler for this entry, if available
127 auto handlerIt = map.find(header.key);
128 if (handlerIt != end(map))
129 {
130 handlerIt->second(move(valueBegin), move(valueEnd));
131 }
132 }
133}

243template <typename First, typename... Rest>
244struct ParsePayload<First, Rest...>
245{
246 template <typename It, typename... Handlers>
247 static void parse(It begin, It end, Handlers... handlers)
248 {
249 detail::HandlerMap<It> map;
250 collectHandlers(map, std::move(handlers)...);
251 detail::parseByteStream(map, std::move(begin), std::move(end));
252 }
253
254 template <typename It, typename FirstHandler, typename... RestHandlers>
255 static void collectHandlers(
256 detail::HandlerMap<It>& map, FirstHandler handler, RestHandlers... rest)
257 {
258 using namespace std;
259 map[First::key] = [handler](const It begin, const It end) {
260 const auto res = First::fromNetworkByteStream(begin, end);
261 if (res.second != end)
262 {
263 std::ostringstream stringStream;
264 stringStream << "Parsing payload entry " << First::key
265 << " did not consume the expected number of bytes. "
266 << " Expected: " << distance(begin, end)
267 << ", Actual: " << distance(begin, res.second);
268 throw range_error(stringStream.str());
269 }
270 handler(res.first);
271 };
272
273 ParsePayload<Rest...>::collectHandlers(map, std::move(rest)...);
274 }
275};
