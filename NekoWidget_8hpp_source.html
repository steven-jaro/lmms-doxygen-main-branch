63 void draw(const GraphicsContext& context)
64 {
65 int x = fPos+108;
66 int y = -2;
67
68 if (fCurImage == &fImages.claw1 || fCurImage == &fImages.claw2)
69 {
70 x += 2;
71 y += 12;
72 }
73
74 fCurImage->drawAt(context, x, y);
75 }
76
77 // returns true if needs repaint
78 bool idle()
79 {
80 if (++fTimer == 10)
81 {
82 if (fCurAction == kActionNone)
83 fCurAction = static_cast<Action>(std::rand() % kActionCount);
84 else
85 fCurAction = kActionNone;
86
87 fTimer = 0;
88 }
89
90 switch (fCurAction)
91 {
92 case kActionNone:
93 if (fCurImage == &fImages.sit)
94 fCurImage = &fImages.tail;
95 else
96 fCurImage = &fImages.sit;
97 break;
98
99 case kActionClaw:
100 if (fCurImage == &fImages.claw1)
101 fCurImage = &fImages.claw2;
102 else
103 fCurImage = &fImages.claw1;
104 break;
105
106 case kActionScratch:
107 if (fCurImage == &fImages.scratch1)
108 fCurImage = &fImages.scratch2;
109 else
110 fCurImage = &fImages.scratch1;
111 break;
112
113 case kActionRunRight:
114 if (fTimer == 0 && fPos > 20*9)
115 {
116 // run the other way
117 --fTimer;
118 fCurAction = kActionRunLeft;
119 idle();
120 break;
121 }
122
123 fPos += 20;
124
125 if (fCurImage == &fImages.run1)
126 fCurImage = &fImages.run2;
127 else
128 fCurImage = &fImages.run1;
129 break;
130
131 case kActionRunLeft:
132 if (fTimer == 0 && fPos < 20*9)
133 {
134 // run the other way
135 --fTimer;
136 fCurAction = kActionRunRight;
137 idle();
138 break;
139 }
140
141 fPos -= 20;
142
143 if (fCurImage == &fImages.run3)
144 fCurImage = &fImages.run4;
145 else
146 fCurImage = &fImages.run3;
147 break;
148
149 case kActionCount:
150 break;
151 }
152
153 return true;
154 }
