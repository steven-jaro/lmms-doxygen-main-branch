189template <typename Stream>
190template <typename ConstBufferSequence>
191std::size_t buffered_write_stream<Stream>::write_some(
192 const ConstBufferSequence& buffers)
193{
194 using asio::buffer_size;
195 if (buffer_size(buffers) == 0)
196 return 0;
197
198 if (storage_.size() == storage_.capacity())
199 this->flush();
200
201 return this->copy(buffers);
202}

205template <typename ConstBufferSequence>
206std::size_t buffered_write_stream<Stream>::write_some(
207 const ConstBufferSequence& buffers, asio::error_code& ec)
208{
209 ec = asio::error_code();
210
211 using asio::buffer_size;
212 if (buffer_size(buffers) == 0)
213 return 0;
214
215 if (storage_.size() == storage_.capacity() && !flush(ec))
216 return 0;
217
218 return this->copy(buffers);

224 class buffered_write_some_handler
225 {
226 public:
227 buffered_write_some_handler(detail::buffered_stream_storage& storage,
228 const ConstBufferSequence& buffers, WriteHandler& handler)
229 : storage_(storage),
230 buffers_(buffers),
231 handler_(ASIO_MOVE_CAST(WriteHandler)(handler))
232 {
233 }
234
235#if defined(ASIO_HAS_MOVE)
236 buffered_write_some_handler(const buffered_write_some_handler& other)
237 : storage_(other.storage_),
238 buffers_(other.buffers_),
239 handler_(other.handler_)
240 {
241 }
242
243 buffered_write_some_handler(buffered_write_some_handler&& other)
244 : storage_(other.storage_),
245 buffers_(other.buffers_),
246 handler_(ASIO_MOVE_CAST(WriteHandler)(other.handler_))
247 {
248 }
249#endif // defined(ASIO_HAS_MOVE)
250
251 void operator()(const asio::error_code& ec, std::size_t)
252 {
253 if (ec)
254 {
255 const std::size_t length = 0;
256 handler_(ec, length);
257 }
258 else
259 {
260 using asio::buffer_size;
261 std::size_t orig_size = storage_.size();
262 std::size_t space_avail = storage_.capacity() - orig_size;
263 std::size_t bytes_avail = buffer_size(buffers_);
264 std::size_t length = bytes_avail < space_avail
265 ? bytes_avail : space_avail;
266 storage_.resize(orig_size + length);
267 const std::size_t bytes_copied = asio::buffer_copy(
268 storage_.data() + orig_size, buffers_, length);
269 handler_(ec, bytes_copied);
270 }
271 }
272
273 //private:
274 detail::buffered_stream_storage& storage_;
275 ConstBufferSequence buffers_;
276 WriteHandler handler_;
277 };

328 template <typename WriteHandler, typename Stream,
329 typename ConstBufferSequence>
330 void operator()(ASIO_MOVE_ARG(WriteHandler) handler,
331 buffered_stream_storage* storage, Stream* next_layer,
332 const ConstBufferSequence& buffers) const
333 {
334 // If you get an error on the following line it means that your handler
335 // does not meet the documented type requirements for a WriteHandler.
336 ASIO_WRITE_HANDLER_CHECK(WriteHandler, handler) type_check;
337
338 using asio::buffer_size;
339 non_const_lvalue<WriteHandler> handler2(handler);
340 if (buffer_size(buffers) == 0 || storage->size() < storage->capacity())
341 {
342 next_layer->async_write_some(ASIO_CONST_BUFFER(0, 0),
343 buffered_write_some_handler<ConstBufferSequence,

349 initiate_async_buffered_flush()(
