38QString SampleLoader::openAudioFile(const QString& previousFile)
39{
40 auto openFileDialog = FileDialog(nullptr, QObject::tr("Open audio file"));
41 auto dir = !previousFile.isEmpty() ? QFileInfo(PathUtil::toAbsolute(previousFile)).absolutePath() : ConfigManager::inst()->userSamplesDir();
42
43 // change dir to position of previously opened file
44 openFileDialog.setDirectory(dir);
45 openFileDialog.setFileMode(FileDialog::ExistingFiles);
46
47 // set filters
48 auto fileTypes = QStringList{};
49 auto allFileTypes = QStringList{};
50 auto nameFilters = QStringList{};
51 const auto& supportedAudioTypes = SampleDecoder::supportedAudioTypes();
52
53 for (const auto& audioType : supportedAudioTypes)
54 {
55 const auto name = QString::fromStdString(audioType.name);
56 const auto extension = QString::fromStdString(audioType.extension);
57 const auto displayExtension = QString{"*.%1"}.arg(extension);
58 fileTypes.append(QString{"%1 (%2)"}.arg(FileDialog::tr("%1 files").arg(name), displayExtension));
59 allFileTypes.append(displayExtension);
60 }
61
62 nameFilters.append(QString{"%1 (%2)"}.arg(FileDialog::tr("All audio files"), allFileTypes.join(" ")));
63 nameFilters.append(fileTypes);
64 nameFilters.append(QString("%1 (*)").arg(FileDialog::tr("Other files")));
65
66 openFileDialog.setNameFilters(nameFilters);
67
68 if (!previousFile.isEmpty())
69 {
70 // select previously opened file
71 openFileDialog.selectFile(QFileInfo{previousFile}.fileName());
72 }
73
74 if (openFileDialog.exec() == QDialog::Accepted)
75 {
76 if (openFileDialog.selectedFiles().isEmpty()) { return ""; }
77
78 return PathUtil::toShortestRelative(openFileDialog.selectedFiles()[0]);
79 }
80
81 return "";
82}

90std::shared_ptr<const SampleBuffer> SampleLoader::createBufferFromFile(const QString& filePath)
91{
92 if (filePath.isEmpty()) { return SampleBuffer::emptyBuffer(); }
93
94 try
95 {
96 return std::make_shared<SampleBuffer>(filePath);
97 }
98 catch (const std::runtime_error& error)
99 {
100 if (getGUI()) { displayError(QString::fromStdString(error.what())); }
101 return SampleBuffer::emptyBuffer();
102 }
103}
