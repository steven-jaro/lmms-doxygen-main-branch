48EffectView::EffectView( Effect * _model, QWidget * _parent ) :
49 PluginView( _model, _parent ),
50 m_bg( embed::getIconPixmap( "effect_plugin" ) ),
51 m_subWindow( nullptr ),
52 m_controlView(nullptr),
53 m_dragging(false)
54{
55 setFixedSize(EffectView::DEFAULT_WIDTH, EffectView::DEFAULT_HEIGHT);
56 setFocusPolicy(Qt::StrongFocus);
57
58 // Disable effects that are of type "DummyEffect"
59 bool isEnabled = !dynamic_cast<DummyEffect *>( effect() );
60 m_bypass = new LedCheckBox( this, "", isEnabled ? LedCheckBox::LedColor::Green : LedCheckBox::LedColor::Red );
61 m_bypass->move( 3, 3 );
62 m_bypass->setEnabled( isEnabled );
63
64 m_bypass->setToolTip(tr("On/Off"));
65
66 m_wetDry = new Knob(KnobType::Bright26, tr("W/D"), this, Knob::LabelRendering::LegacyFixedFontSize);
67 m_wetDry->move( 40 - m_wetDry->width() / 2, 5 );
68 m_wetDry->setEnabled( isEnabled );
69 m_wetDry->setHintText( tr( "Wet Level:" ), "" );
70
71
72 m_autoQuit = new TempoSyncKnob(KnobType::Bright26, tr("DECAY"), this, Knob::LabelRendering::LegacyFixedFontSize);
73 m_autoQuit->move( 78 - m_autoQuit->width() / 2, 5 );
74 m_autoQuit->setEnabled(isEnabled && effect()->autoQuitEnabled());
75 m_autoQuit->setHintText( tr( "Time:" ), "ms" );
76
77 setModel( _model );
78
79 if( effect()->controls()->controlCount() > 0 )
80 {
81 auto ctls_btn = new QPushButton(tr("Controls"), this);
82 QFont f = ctls_btn->font();
83 ctls_btn->setFont(adjustedToPixelSize(f, DEFAULT_FONT_SIZE));
84 ctls_btn->setGeometry( 150, 14, 50, 20 );
85 ctls_btn->setFocusPolicy(Qt::NoFocus);
86 connect( ctls_btn, SIGNAL(clicked()),
87 this, SLOT(editControls()));
88
89 m_controlView = effect()->controls()->createView();
90 if( m_controlView )
91 {
92 m_subWindow = getGUI()->mainWindow()->addWindowedWidget( m_controlView );
93
94 if ( !m_controlView->isResizable() )
95 {
96 m_subWindow->setSizePolicy( QSizePolicy::Fixed, QSizePolicy::Fixed );
97 if (m_subWindow->layout())
98 {
99 m_subWindow->layout()->setSizeConstraint(QLayout::SetFixedSize);
100 }
101 }
102
103 Qt::WindowFlags flags = m_subWindow->windowFlags();
104 flags &= ~Qt::WindowMaximizeButtonHint;
105 m_subWindow->setWindowFlags( flags );
106
107 connect( m_controlView, SIGNAL(closed()),
108 this, SLOT(closeEffects()));
109
110 m_subWindow->hide();
111 }
112 }
113
114 m_opacityEffect = new QGraphicsOpacityEffect(this);
115 m_opacityEffect->setOpacity(1);
116 setGraphicsEffect(m_opacityEffect);
117
118 //move above vst effect view creation
119 //setModel( _model );
120}

188void EffectView::contextMenuEvent( QContextMenuEvent * )

208void EffectView::mousePressEvent(QMouseEvent* event)
209{
210 if (event->button() == Qt::LeftButton)
211 {
212 m_dragging = true;
213 m_opacityEffect->setOpacity(0.3);
214 QCursor c(Qt::SizeVerCursor);
215 QApplication::setOverrideCursor(c);
216 update();
217 }
218}
219
220void EffectView::mouseReleaseEvent(QMouseEvent* event)
221{
222 if (event->button() == Qt::LeftButton)
223 {
224 m_dragging = false;
225 m_opacityEffect->setOpacity(1);
226 QApplication::restoreOverrideCursor();
227 update();
228 }
229}

231void EffectView::mouseMoveEvent(QMouseEvent* event)
232{
233 if (!m_dragging) { return; }
234 if (event->pos().y() < 0)
235 {
236 moveUp();
237 }
238 else if (event->pos().y() > EffectView::DEFAULT_HEIGHT)
239 {
240 moveDown();
241 }
242}

246void EffectView::paintEvent( QPaintEvent * )
247{
248 QPainter p( this );
249 p.drawPixmap( 0, 0, m_bg );
250
251 QFont f = adjustedToPixelSize(font(), DEFAULT_FONT_SIZE);
252 f.setBold( true );
253 p.setFont( f );
254
255 QString elidedText = p.fontMetrics().elidedText( model()->displayName(), Qt::ElideRight, width() - 22 );
256
257 p.setPen( palette().shadow().color() );
258 p.drawText( 6, 55, elidedText );
259 p.setPen( palette().text().color() );
260 p.drawText( 5, 54, elidedText );
261}
