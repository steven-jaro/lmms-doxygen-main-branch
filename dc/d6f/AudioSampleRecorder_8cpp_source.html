69std::shared_ptr<const SampleBuffer> AudioSampleRecorder::createSampleBuffer()
70{
71 const f_cnt_t frames = framesRecorded();
72 // create buffer to store all recorded buffers in
73 auto bigBuffer = std::vector<SampleFrame>(frames);
74
75 // now copy all buffers into big buffer
76 auto framesCopied = 0;
77 for (const auto& [buf, numFrames] : m_buffers)
78 {
79 std::copy_n(buf, numFrames, bigBuffer.begin() + framesCopied);
80 framesCopied += numFrames;
81 }
82
83 // create according sample-buffer out of big buffer
84 return std::make_shared<const SampleBuffer>(std::move(bigBuffer), sampleRate());
85}

87void AudioSampleRecorder::writeBuffer(const SampleFrame* _ab, const fpp_t _frames)
88{
89 auto buf = new SampleFrame[_frames];
90 for( fpp_t frame = 0; frame < _frames; ++frame )
91 {
92 for( ch_cnt_t chnl = 0; chnl < DEFAULT_CHANNELS; ++chnl )
93 {
94 buf[frame][chnl] = _ab[frame][chnl];
95 }
96 }
97 m_buffers.push_back( qMakePair( buf, _frames ) );
98}
