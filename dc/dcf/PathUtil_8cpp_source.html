151 {
152 //First, do no harm to absolute paths

180 QString toShortestRelative(const QString & input, bool allowLocal /* = false*/)
181 {
182 QFileInfo inputFileInfo = QFileInfo(input);
183 QString absolutePath = inputFileInfo.isAbsolute() ? input : toAbsolute(input);
184
185 Base shortestBase = Base::Absolute;
186 QString shortestPath = relativeOrAbsolute(absolutePath, shortestBase);
187 for (auto base: relativeBases)
188 {
189 // Skip local paths when searching for the shortest relative if those
190 // are not allowed for that resource
191 if (base == Base::LocalDir && !allowLocal) { continue; }
192
193 QString otherPath = relativeOrAbsolute(absolutePath, base);
194 if (otherPath.length() < shortestPath.length())
195 {
196 shortestBase = base;
197 shortestPath = otherPath;
198 }
199 }
200 return basePrefix(shortestBase) + relativeOrAbsolute(absolutePath, shortestBase);
201 }
