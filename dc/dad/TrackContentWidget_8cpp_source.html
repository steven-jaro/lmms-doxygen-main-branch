49/*! Alternate between a darker and a lighter background color every 4 bars
50 */
51const int BARS_PER_GROUP = 4;
52/* Lines between bars will disappear if zoomed too far out (i.e
53 if there are less than 4 pixels between lines)*/
54const int MIN_PIXELS_BETWEEN_LINES = 4;
55
56/*! \brief Create a new trackContentWidget
57 *
58 * Creates a new track content widget for the given track.
59 * The content widget comprises the 'grip bar' and the 'tools' button
60 * for the track's context menu.
61 *
62 * \param parent The parent track.
63 */
64TrackContentWidget::TrackContentWidget( TrackView * parent ) :
65 QWidget( parent ),
66 m_trackView( parent ),
67 m_darkerColor( Qt::SolidPattern ),
68 m_lighterColor( Qt::SolidPattern ),
69 m_coarseGridColor( Qt::SolidPattern ),
70 m_fineGridColor( Qt::SolidPattern ),
71 m_horizontalColor( Qt::SolidPattern ),
72 m_embossColor( Qt::SolidPattern ),
73 m_coarseGridWidth(2),
74 m_fineGridWidth(1),
75 m_horizontalWidth(1),
76 m_embossWidth(0),
77 m_embossOffset(0)
78{
79 setAcceptDrops( true );
80
81 connect( parent->trackContainerView(),
82 SIGNAL( positionChanged( const lmms::TimePos& ) ),
83 this, SLOT( changePosition( const lmms::TimePos& ) ) );
84
85 // Update background if snap size changes
86 connect(getGUI()->songEditor()->m_editor->snappingModel(), &Model::dataChanged,
87 this, &TrackContentWidget::updateBackground);
88
89 // Also update background if proportional snap is enabled/disabled
90 connect(getGUI()->songEditor()->m_editor, &SongEditor::proportionalSnapChanged,
91 this, &TrackContentWidget::updateBackground);
92
93 setStyle( QApplication::style() );
94
95 updateBackground();
96}

101void TrackContentWidget::updateBackground()
102{
103 // use snapSize to determine number of lines to draw
104 float snapSize = getGUI()->songEditor()->m_editor->getSnapSize();
105
106 const TrackContainerView * tcv = m_trackView->trackContainerView();
107
108 // Assume even-pixels-per-bar. Makes sense, should be like this anyways
109 int ppb = static_cast<int>( tcv->pixelsPerBar() );
110
111 // Coarse grid appears every bar (less frequently if quantization > 1 bar)
112 float coarseGridResolution = (snapSize >= 1) ? snapSize : 1;
113 // Fine grid appears within bars
114 float fineGridResolution = snapSize;
115 // Increase fine grid resolution (size between lines) if it results in less than
116 // 4 pixels between each line to avoid cluttering
117 float pixelsBetweenLines = ppb * snapSize;
118 if (pixelsBetweenLines < MIN_PIXELS_BETWEEN_LINES) {
119 // Scale fineGridResolution so that there are enough pixels between lines
120 // scaleFactor should be a power of 2
121 int scaleFactor = 1 << static_cast<int>( std::ceil( std::log2( MIN_PIXELS_BETWEEN_LINES / pixelsBetweenLines ) ) );
122 fineGridResolution *= scaleFactor;
123 }
124
125 int w = ppb * BARS_PER_GROUP;
126 int h = height();
127 m_background = QPixmap( w * 2, height() );
128 QPainter pmp( &m_background );
129
130 pmp.fillRect( 0, 0, w, h, darkerColor() );
131 pmp.fillRect( w, 0, w , h, lighterColor() );
132
133 // draw lines
134 // draw fine grid
135 pmp.setPen( QPen( fineGridColor(), fineGridWidth() ) );
136 for (float x = 0; x < w * 2; x += ppb * fineGridResolution)
137 {
138 pmp.drawLine( QLineF( x, 0.0, x, h ) );
139 }
140
141 // draw coarse grid
142 pmp.setPen( QPen( coarseGridColor(), coarseGridWidth() ) );
143 for (float x = 0; x <= w * 2; x += ppb * coarseGridResolution)
144 {
145 pmp.drawLine( QLineF( x, 0.0, x, h ) );
146 }
147
148 pmp.setPen( QPen( embossColor(), embossWidth() ) );
149 for (float x = (coarseGridWidth() + embossOffset()); x < w * 2; x += ppb * coarseGridResolution)
150 {
151 pmp.drawLine( QLineF( x, 0.0, x, h ) );
152 }
153
154 // draw horizontal line
155 pmp.setPen( QPen( horizontalColor(), horizontalWidth() ) );
156 pmp.drawLine(0, h - (horizontalWidth() + 1) / 2, w * 2, h - (horizontalWidth() + 1) / 2);
157
158 pmp.end();
159
160 // Force redraw
161 update();
162}

225// responsible for moving track-content-widgets to appropriate position after
226// change of visible viewport
227/*! \brief Move the trackContentWidget to a new place in time
228 *
229 * \param newPos The MIDI time to move to.
230 */
231void TrackContentWidget::changePosition( const TimePos & newPos )
232{
233 if (m_trackView->trackContainerView() == getGUI()->patternEditor()->m_editor)
234 {
235 const int curPattern = Engine::patternStore()->currentPattern();
236 setUpdatesEnabled( false );
237
238 // first show clip for current pattern...
239 for (const auto& clipView : m_clipViews)
240 {
241 if (clipView->getClip()->startPosition().getBar() == curPattern)
242 {
243 clipView->move(0, clipView->y());
244 clipView->raise();
245 clipView->show();
246 }
247 else { clipView->lower(); }
248 }
249 // ...then hide others to avoid flickering
250 for (const auto& clipView : m_clipViews)
251 {
252 if (clipView->getClip()->startPosition().getBar() != curPattern) { clipView->hide(); }
253 }
254 setUpdatesEnabled( true );
255 return;
256 }
257
258 TimePos pos = newPos;
259 if( pos < 0 )
260 {
261 pos = m_trackView->trackContainerView()->currentPosition();
262 }
263
264 const int begin = pos;
265 const int end = endPosition( pos );
266 const float ppb = m_trackView->trackContainerView()->pixelsPerBar();
267
268 setUpdatesEnabled( false );
269 for (const auto& clipView : m_clipViews)
270 {
271 Clip* clip = clipView->getClip();
272
273 clip->changeLength( clip->length() );
274
275 const int ts = clip->startPosition();
276 const int te = clip->endPosition()-3;
277 if( ( ts >= begin && ts <= end ) ||
278 ( te >= begin && te <= end ) ||
279 ( ts <= begin && te >= end ) )
280 {
281 clipView->move(static_cast<int>((ts - begin) * ppb / TimePos::ticksPerBar()), clipView->y());
282 if (!clipView->isVisible())
283 {
284 clipView->show();
285 }
286 }
287 else
288 {
289 clipView->move(-clipView->width() - 10, clipView->y());
290 }
291 }
292 setUpdatesEnabled( true );
293
294 // redraw background
295 updateBackground();
296// update();
297}

339/*! \brief Returns whether a selection of Clips can be pasted into this
340 *
341 * \param clipPos the position of the Clip slot being pasted on
342 * \param de the DropEvent generated
343 */
344bool TrackContentWidget::canPasteSelection( TimePos clipPos, const QDropEvent* de )
345{
346 const QMimeData * mimeData = de->mimeData();
347
348 // If the source of the DropEvent is the current instance of LMMS we don't allow pasting in the same bar
349 // if it's another instance of LMMS we allow it
350 return de->source()
351 ? canPasteSelection( clipPos, mimeData )
352 : canPasteSelection( clipPos, mimeData, true );
353}

355// Overloaded method to make it possible to call this method without a Drag&Drop event
356bool TrackContentWidget::canPasteSelection( TimePos clipPos, const QMimeData* md , bool allowSameBar )
357{
358 // For decodeKey() and decodeValue()
359 using namespace Clipboard;
360
361 Track * t = getTrack();
362 QString type = decodeKey( md );
363 QString value = decodeValue( md );
364
365 // We can only paste into tracks of the same type
366 if (type != ("clip_" + QString::number(static_cast<int>(t->type()))))
367 {
368 return false;
369 }
370
371 // value contains XML needed to reconstruct Clips and place them
372 DataFile dataFile( value.toUtf8() );
373
374 // Extract the metadata and which Clip was grabbed
375 QDomElement metadata = dataFile.content().firstChildElement( "copyMetadata" );
376 QDomAttr clipPosAttr = metadata.attributeNode( "grabbedClipPos" );
377 TimePos grabbedClipPos = clipPosAttr.value().toInt();
378 TimePos grabbedClipBar = TimePos( grabbedClipPos.getBar(), 0 );
379
380 // Extract the track index that was originally clicked
381 QDomAttr tiAttr = metadata.attributeNode( "initialTrackIndex" );
382 const int initialTrackIndex = tiAttr.value().toInt();
383
384 // Get the current track's index
385 const TrackContainer::TrackList& tracks = t->trackContainer()->tracks();
386 const auto currentTrackIt = std::find(tracks.begin(), tracks.end(), t);
387 const int currentTrackIndex = currentTrackIt != tracks.end() ? std::distance(tracks.begin(), currentTrackIt) : -1;
388
389 // Don't paste if we're on the same bar and allowSameBar is false
390 auto sourceTrackContainerId = metadata.attributeNode( "trackContainerId" ).value().toUInt();
391 if( !allowSameBar && sourceTrackContainerId == t->trackContainer()->id() &&
392 clipPos == grabbedClipBar && currentTrackIndex == initialTrackIndex )
393 {
394 return false;
395 }
396
397 // Extract the clip data
398 QDomElement clipParent = dataFile.content().firstChildElement("clips");
399 QDomNodeList clipNodes = clipParent.childNodes();
400
401 // If we are pasting into the PatternEditor, only a single Clip is allowed to be pasted
402 // so we don't have the unexpected behavior of pasting on different PatternTracks
403 if (m_trackView->trackContainerView()->fixedClips() == true &&
404 clipNodes.length() > 1)
405 {
406 return false;
407 }
408
409 // Determine if all the Clips will land on a valid track
410 for( int i = 0; i < clipNodes.length(); i++ )
411 {
412 QDomElement clipElement = clipNodes.item( i ).toElement();
413 int trackIndex = clipElement.attributeNode( "trackIndex" ).value().toInt();
414 int finalTrackIndex = trackIndex + currentTrackIndex - initialTrackIndex;
415
416 // Track must be in TrackContainer's tracks
417 if (finalTrackIndex < 0 || static_cast<std::size_t>(finalTrackIndex) >= tracks.size())
418 {
419 return false;
420 }
421
422 // Track must be of the same type
423 auto startTrackType = static_cast<Track::Type>(clipElement.attributeNode("trackType").value().toInt());
424 Track * endTrack = tracks.at( finalTrackIndex );
425 if( startTrackType != endTrack->type() )
426 {
427 return false;
428 }
429 }
430
431 return true;
432}

434/*! \brief Pastes a selection of Clips onto the track
435 *
436 * \param clipPos the position of the Clip slot being pasted on
437 * \param de the DropEvent generated
438 */
439bool TrackContentWidget::pasteSelection( TimePos clipPos, QDropEvent * de )
440{
441 const QMimeData * mimeData = de->mimeData();
442
443 if( canPasteSelection( clipPos, de ) == false )
444 {
445 return false;
446 }
447
448 // We set skipSafetyCheck to true because we already called canPasteSelection
449 return pasteSelection( clipPos, mimeData, true );
450}

452// Overloaded method so we can call it without a Drag&Drop event
453bool TrackContentWidget::pasteSelection( TimePos clipPos, const QMimeData * md, bool skipSafetyCheck )
454{
455 // For decodeKey() and decodeValue()
456 using namespace Clipboard;
457
458 // When canPasteSelection was already called before, skipSafetyCheck will skip this
459 if( !skipSafetyCheck && canPasteSelection( clipPos, md ) == false )
460 {
461 return false;
462 }
463
464 QString type = decodeKey( md );
465 QString value = decodeValue( md );
466
467 getTrack()->addJournalCheckPoint();
468
469 // value contains XML needed to reconstruct Clips and place them
470 DataFile dataFile( value.toUtf8() );
471
472 // Extract the clip data
473 QDomElement clipParent = dataFile.content().firstChildElement("clips");
474 QDomNodeList clipNodes = clipParent.childNodes();
475
476 // Extract the track index that was originally clicked
477 QDomElement metadata = dataFile.content().firstChildElement( "copyMetadata" );
478 QDomAttr tiAttr = metadata.attributeNode( "initialTrackIndex" );
479 int initialTrackIndex = tiAttr.value().toInt();
480 QDomAttr clipPosAttr = metadata.attributeNode( "grabbedClipPos" );
481 TimePos grabbedClipPos = clipPosAttr.value().toInt();
482
483 // Snap the mouse position to the beginning of the dropped bar, in ticks
484 const TrackContainer::TrackList& tracks = getTrack()->trackContainer()->tracks();
485 const auto currentTrackIt = std::find(tracks.begin(), tracks.end(), getTrack());
486 const int currentTrackIndex = currentTrackIt != tracks.end() ? std::distance(tracks.begin(), currentTrackIt) : -1;
487
488 bool wasSelection = m_trackView->trackContainerView()->rubberBand()->selectedObjects().count();
489
490 // Unselect the old group
491 const QVector<selectableObject *> so =
492 m_trackView->trackContainerView()->selectedObjects();
493 for (const auto& obj : so)
494 {
495 obj->setSelected(false);
496 }
497
498
499 // TODO -- Need to draw the hovericon either way, or ghost the Clips
500 // onto their final position.
501
502 float snapSize = getGUI()->songEditor()->m_editor->getSnapSize();
503 // All clips should be offset the same amount as the grabbed clip
504 auto offset = TimePos(clipPos - grabbedClipPos);
505 // Users expect clips to "fall" backwards, so bias the offset
506 offset -= TimePos::ticksPerBar() * snapSize / 2;
507 // The offset is quantized (rather than the positions) to preserve fine adjustments
508 offset = offset.quantize(snapSize);
509
510 // Get the leftmost Clip and fix the offset if it reaches below bar 0
511 TimePos leftmostPos = grabbedClipPos;
512 for(int i = 0; i < clipNodes.length(); ++i)
513 {
514 QDomElement outerClipElement = clipNodes.item(i).toElement();
515 QDomElement clipElement = outerClipElement.firstChildElement();
516
517 TimePos pos = clipElement.attributeNode("pos").value().toInt();
518
519 if(pos < leftmostPos) { leftmostPos = pos; }
520 }
521 // Fix offset if it sets the left most Clip to a negative position
522 offset = std::max(offset.getTicks(), -leftmostPos.getTicks());
523
524 for( int i = 0; i<clipNodes.length(); i++ )
525 {
526 QDomElement outerClipElement = clipNodes.item( i ).toElement();
527 QDomElement clipElement = outerClipElement.firstChildElement();
528
529 int trackIndex = outerClipElement.attributeNode( "trackIndex" ).value().toInt();
530 int finalTrackIndex = trackIndex + ( currentTrackIndex - initialTrackIndex );
531 Track * t = tracks.at( finalTrackIndex );
532
533 // The new position is the old position plus the offset.
534 TimePos pos = clipElement.attributeNode( "pos" ).value().toInt() + offset;
535 // If we land on ourselves, offset by one snap
536 TimePos shift = TimePos::ticksPerBar() * getGUI()->songEditor()->m_editor->getSnapSize();
537 if (offset == 0 && initialTrackIndex == currentTrackIndex) { pos += shift; }
538
539 Clip * clip = t->createClip( pos );
540 clip->restoreState( clipElement );
541 clip->movePosition(pos); // Because we restored the state, we need to move the Clip again.
542 if( wasSelection )
543 {
544 clip->selectViewOnCreate( true );
545 }
546 }
547
548 AutomationClip::resolveAllIDs();
549
550 return true;
551}

570/*! \brief Respond to a mouse press on the trackContentWidget
571 *
572 * \param me the mouse press event to respond to
573 */
574void TrackContentWidget::mousePressEvent( QMouseEvent * me )
575{
576 // Enable box select if control is held when clicking an empty space
577 // (If we had clicked a Clip it would have intercepted the mouse event)
578 if( me->modifiers() & Qt::ControlModifier ){
579 getGUI()->songEditor()->m_editor->setEditMode(SongEditor::EditMode::Select);
580 }
581 // Forward event to allow box select if the editor supports it and is in that mode
582 if( m_trackView->trackContainerView()->allowRubberband() == true )
583 {
584 QWidget::mousePressEvent( me );
585 }
586 // Forward shift clicks so tracks can be resized
587 else if( me->modifiers() & Qt::ShiftModifier )
588 {
589 QWidget::mousePressEvent( me );
590 }
591 // For an unmodified click, create a new Clip
592 else if( me->button() == Qt::LeftButton &&
593 !m_trackView->trackContainerView()->fixedClips() )
594 {
595 QVector<selectableObject*> so = m_trackView->trackContainerView()->rubberBand()->selectedObjects();
596 for( int i = 0; i < so.count(); ++i )
597 {
598 so.at( i )->setSelected( false);
599 }
600 getTrack()->addJournalCheckPoint();
601 const float snapSize = getGUI()->songEditor()->m_editor->getSnapSize();
602 const TimePos pos = TimePos(getPosition(me->x())).quantize(snapSize, true);
603 getTrack()->createClip(pos);
604 }
605}

677void TrackContentWidget::contextMenuEvent( QContextMenuEvent * cme )
678{
679 // For hasFormat(), MimeType enum class and getMimeData()
680 using namespace Clipboard;
681
682 if( cme->modifiers() )
683 {
684 return;
685 }
686
687 // If we don't have Clip data in the clipboard there's no need to create this menu
688 // since "paste" is the only action at the moment.
689 if( ! hasFormat( MimeType::StringPair ) )
690 {
691 return;
692 }
693
694 QMenu contextMenu( this );
695 QAction *pasteA = contextMenu.addAction( embed::getIconPixmap( "edit_paste" ),
696 tr( "Paste" ), [this, cme](){ contextMenuAction( cme, ContextMenuAction::Paste ); } );
697 // If we can't paste in the current TCW for some reason, disable the action so the user knows
698 pasteA->setEnabled( canPasteSelection( getPosition( cme->x() ), getMimeData() ) ? true : false );
699
700 contextMenu.exec( QCursor::pos() );
701}
702
703void TrackContentWidget::contextMenuAction( QContextMenuEvent * cme, ContextMenuAction action )
704{
705 // For getMimeData()
706 using namespace Clipboard;
707
708 switch( action )
709 {
710 case ContextMenuAction::Paste:
711 // Paste the selection on the TimePos of the context menu event
712 TimePos clipPos = getPosition( cme->x() );
713
714 pasteSelection( clipPos, getMimeData() );
715 break;
716 }
717}
