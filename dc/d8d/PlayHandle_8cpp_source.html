53void PlayHandle::doProcessing()
54{
55 if( m_usesBuffer )
56 {
57 m_bufferReleased = false;
58 zeroSampleFrames(m_playHandleBuffer, Engine::audioEngine()->framesPerPeriod());
59 play( buffer() );
60 }
61 else
62 {
63 play( nullptr );
