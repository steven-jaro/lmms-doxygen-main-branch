41{
42 static_assert(std::is_convertible<T,int32_t>::value, "int32 compatibility required.");
43 return (
44 static_cast<int32_t>(a) << 24 |
45 static_cast<int32_t>(b) << 16 |
46 static_cast<int32_t>(c) << 8 |
47 static_cast<int32_t>(d) << 0);
48}
