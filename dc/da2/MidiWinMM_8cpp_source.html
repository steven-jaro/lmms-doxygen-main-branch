55void MidiWinMM::processOutEvent( const MidiEvent& event, const TimePos& time, const MidiPort* port )
56{
57 const DWORD shortMsg = ( event.type() + event.channel() ) +
58 ( ( event.param( 0 ) & 0xff ) << 8 ) +
59 ( ( event.param( 1 ) & 0xff ) << 16 );
60
61 QStringList outDevs;
62 for( SubMap::ConstIterator it = m_outputSubs.begin(); it != m_outputSubs.end(); ++it )
63 {
64 for( MidiPortList::ConstIterator jt = it.value().begin(); jt != it.value().end(); ++jt )
65 {
66 if( *jt == port )
67 {
68 outDevs += it.key();
69 break;
70 }
71 }
72 }
73
74 for( QMap<HMIDIOUT, QString>::Iterator it = m_outputDevices.begin(); it != m_outputDevices.end(); ++it )
75 {
76 if( outDevs.contains( *it ) )
77 {
78 midiOutShortMsg( it.key(), shortMsg );
79 }
80 }
81}

86void MidiWinMM::applyPortMode( MidiPort* port )
87{
88 // make sure no subscriptions exist which are not possible with
89 // current port-mode
90 if( !port->isInputEnabled() )
91 {
92 for( SubMap::Iterator it = m_inputSubs.begin(); it != m_inputSubs.end(); ++it )
93 {
94 it.value().removeAll( port );
95 }
96 }
97
98 if( !port->isOutputEnabled() )
99 {
100 for( SubMap::Iterator it = m_outputSubs.begin(); it != m_outputSubs.end(); ++it )
101 {
102 it.value().removeAll( port );
103 }
104 }
105}

188void MidiWinMM::handleInputEvent( HMIDIIN hm, DWORD ev )
189{
190 const int cmd = ev & 0xff;
191 if( cmd == MidiActiveSensing )
192 {
193 return;
194 }
195 const int par1 = ( ev >> 8 ) & 0xff;
196 const int par2 = ev >> 16;
197 const MidiEventTypes cmdtype = static_cast<MidiEventTypes>( cmd & 0xf0 );
198 const int chan = cmd & 0x0f;
199
200 const QString d = m_inputDevices.value( hm );
201 if( d.isEmpty() || !m_inputSubs.contains( d ) )
202 {
203 return;
204 }
205
206 const MidiPortList & l = m_inputSubs[d];
207 for (MidiPortList::ConstIterator it = l.begin(); it != l.end(); ++it)
208 {
209 switch (cmdtype)
210 {
211 case MidiNoteOn:
212 case MidiNoteOff:
213 case MidiKeyPressure:
214 case MidiControlChange:
215 case MidiProgramChange:
216 case MidiChannelPressure:
217 (*it)->processInEvent(MidiEvent(cmdtype, chan, par1, par2 & 0xff, &hm));
218 break;
219
220 case MidiPitchBend:
221 (*it)->processInEvent(MidiEvent(cmdtype, chan, par1 + par2 * 128, 0, &hm));
222 break;
223
224 default:
225 qWarning("MidiWinMM: unhandled input event %d\n", cmdtype);
226 break;
227 }
228 }
229}

245void MidiWinMM::closeDevices()
246{
247 m_inputSubs.clear();
248 m_outputSubs.clear();
249
250 QMapIterator<HMIDIIN, QString> i( m_inputDevices );
251
252 HMIDIIN hInDev;
253 while( i.hasNext() )
254 {
255 hInDev = i.next().key();
256 midiInReset( hInDev );
257 midiInClose( hInDev );
258 }
259
260 QMapIterator<HMIDIOUT, QString> o( m_outputDevices );
261 while( o.hasNext() )
262 {
263 midiOutClose( o.next().key() );
264 }
265
266 m_inputDevices.clear();
267 m_outputDevices.clear();
268}

273void MidiWinMM::openDevices()
274{
275 m_inputDevices.clear();
276 for( unsigned int i = 0; i < midiInGetNumDevs(); ++i )
277 {
278 MIDIINCAPS c;
279 midiInGetDevCaps( i, &c, sizeof( c ) );
280 HMIDIIN hm = 0;
281 MMRESULT res = midiInOpen( &hm, i, (DWORD_PTR) &inputCallback,
282 (DWORD_PTR) this,
283 CALLBACK_FUNCTION );
284 if( res == MMSYSERR_NOERROR )
285 {
286 m_inputDevices[hm] = qstrdup( c.szPname );
287 midiInStart( hm );
288 }
289 }
290
291 m_outputDevices.clear();
292 for( unsigned int i = 0; i < midiOutGetNumDevs(); ++i )
293 {
294 MIDIOUTCAPS c;
295 midiOutGetDevCaps( i, &c, sizeof( c ) );
296 HMIDIOUT hm = 0;
297 MMRESULT res = midiOutOpen( &hm, i, 0, 0, CALLBACK_NULL );
298 if( res == MMSYSERR_NOERROR )
299 {
300 m_outputDevices[hm] = qstrdup( c.szPname );
301 }
302 }
303}
