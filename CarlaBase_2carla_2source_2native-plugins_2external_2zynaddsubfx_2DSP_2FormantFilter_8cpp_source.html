24FormantFilter::FormantFilter(const FilterParams *pars, Allocator *alloc, unsigned int srate, int bufsize)
25 :Filter(srate, bufsize), memory(*alloc)
26{
27 numformants = pars->Pnumformants;
28 for(int i = 0; i < numformants; ++i)
29 formant[i] = memory.alloc<AnalogFilter>(4 /*BPF*/, 1000.0f, 10.0f, pars->Pstages, srate, bufsize);
30 cleanup();
31
32 for(int j = 0; j < FF_MAX_VOWELS; ++j)
33 for(int i = 0; i < numformants; ++i) {
34 formantpar[j][i].freq = pars->getformantfreq(
35 pars->Pvowels[j].formants[i].freq);
36 formantpar[j][i].amp = pars->getformantamp(
37 pars->Pvowels[j].formants[i].amp);
38 formantpar[j][i].q = pars->getformantq(
39 pars->Pvowels[j].formants[i].q);
40 }
41
42 for(int i = 0; i < FF_MAX_FORMANTS; ++i)
43 oldformantamp[i] = 1.0f;
44 for(int i = 0; i < numformants; ++i) {
45 currentformants[i].freq = 1000.0f;
46 currentformants[i].amp = 1.0f;
47 currentformants[i].q = 2.0f;
48 }
49
50 formantslowness = powf(1.0f - (pars->Pformantslowness / 128.0f), 3.0f);
51
52 sequencesize = pars->Psequencesize;
53 if(sequencesize == 0)
54 sequencesize = 1;
55 for(int k = 0; k < sequencesize; ++k)
56 sequence[k].nvowel = pars->Psequence[k].nvowel;
57
58 vowelclearness = powf(10.0f, (pars->Pvowelclearness - 32.0f) / 48.0f);
59
60 sequencestretch = powf(0.1f, (pars->Psequencestretch - 32.0f) / 48.0f);
61 if(pars->Psequencereversed)
62 sequencestretch *= -1.0f;
63
64 outgain = dB2rap(pars->getgain());
65
66 oldinput = -1.0f;
67 Qfactor = pars->getq();
68 oldQfactor = Qfactor;
69 firsttime = 1;
70}

89void FormantFilter::setpos(float frequency)
90{
91 int p1, p2;
92
93 //Convert form real freq[Hz]
94 const float input = log_2(frequency) - 9.96578428f; //log2(1000)=9.95748f.
95
96 if(firsttime != 0)
97 slowinput = input;
98 else
99 slowinput = slowinput
100 * (1.0f - formantslowness) + input * formantslowness;
101
102 if((fabsf(oldinput - input) < 0.001f) && (fabsf(slowinput - input) < 0.001f)
103 && (fabsf(Qfactor - oldQfactor) < 0.001f)) {
104 // oldinput=input; daca setez asta, o sa faca probleme la schimbari foarte lente
105 firsttime = 0;
106 return;
107 }
108 else
109 oldinput = input;
110
111 float pos = input * sequencestretch;
112 pos -= floorf(pos);
113
114 F2I(pos * sequencesize, p2);
115 p1 = p2 - 1;
116 if(p1 < 0)
117 p1 += sequencesize;
118
119 pos = pos * sequencesize;
120 pos -= floorf(pos);
121 pos =
122 (atanf((pos * 2.0f
123 - 1.0f)
124 * vowelclearness) / atanf(vowelclearness) + 1.0f) * 0.5f;
125
126 p1 = sequence[p1].nvowel;
127 p2 = sequence[p2].nvowel;
128
129 if(firsttime != 0) {
130 for(int i = 0; i < numformants; ++i) {
131 currentformants[i].freq =
132 formantpar[p1][i].freq
133 * (1.0f - pos) + formantpar[p2][i].freq * pos;
134 currentformants[i].amp =
135 formantpar[p1][i].amp
136 * (1.0f - pos) + formantpar[p2][i].amp * pos;
137 currentformants[i].q =
138 formantpar[p1][i].q * (1.0f - pos) + formantpar[p2][i].q * pos;
139 formant[i]->setfreq_and_q(currentformants[i].freq,
140 currentformants[i].q * Qfactor);
141 oldformantamp[i] = currentformants[i].amp;
142 }
143 firsttime = 0;
144 }
145 else
146 for(int i = 0; i < numformants; ++i) {
147 currentformants[i].freq =
148 currentformants[i].freq * (1.0f - formantslowness)
149 + (formantpar[p1][i].freq
150 * (1.0f - pos) + formantpar[p2][i].freq * pos)
151 * formantslowness;
152
153 currentformants[i].amp =
154 currentformants[i].amp * (1.0f - formantslowness)
155 + (formantpar[p1][i].amp * (1.0f - pos)
156 + formantpar[p2][i].amp * pos) * formantslowness;
157
158 currentformants[i].q = currentformants[i].q
159 * (1.0f - formantslowness)
160 + (formantpar[p1][i].q * (1.0f - pos)
161 + formantpar[p2][i].q
162 * pos) * formantslowness;
163
164
165 formant[i]->setfreq_and_q(currentformants[i].freq,
166 currentformants[i].q * Qfactor);
167 }
168
169 oldQfactor = Qfactor;
170}

194void FormantFilter::filterout(float *smp)
195{
196 float inbuffer[buffersize];
197
198 memcpy(inbuffer, smp, bufferbytes);
199 memset(smp, 0, bufferbytes);
200
201 for(int j = 0; j < numformants; ++j) {
202 float tmpbuf[buffersize];
203 for(int i = 0; i < buffersize; ++i)
204 tmpbuf[i] = inbuffer[i] * outgain;
205 formant[j]->filterout(tmpbuf);
206
207 if(ABOVE_AMPLITUDE_THRESHOLD(oldformantamp[j], currentformants[j].amp))
208 for(int i = 0; i < buffersize; ++i)
209 smp[i] += tmpbuf[i]
210 * INTERPOLATE_AMPLITUDE(oldformantamp[j],
211 currentformants[j].amp,
212 i,
213 buffersize);
214 else
215 for(int i = 0; i < buffersize; ++i)
216 smp[i] += tmpbuf[i] * currentformants[j].amp;
217 oldformantamp[j] = currentformants[j].amp;
218 }
219}
