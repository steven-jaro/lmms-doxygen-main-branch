90/*! \brief Handle a note being pressed on our keyboard display
91 *
92 * \param key the key being pressed
93 */
94void Piano::handleKeyPress(int key, int midiVelocity)
95{
96 if (midiVelocity == -1)
97 {
98 midiVelocity = m_instrumentTrack->midiPort()->baseVelocity();
99 }
100 if (isValidKey(key))
101 {
102 m_midiEvProc->processInEvent(MidiEvent(MidiNoteOn, -1, key, midiVelocity));
103 m_pressedKeys[key] = true;
104 }
105}

111/*! \brief Handle a note being released on our keyboard display
112 *
113 * \param key the key being releassed
114 */
115void Piano::handleKeyRelease(int key)
116{
117 if (isValidKey(key))
118 {
119 m_midiEvProc->processInEvent(MidiEvent(MidiNoteOff, -1, key, 0));
120 m_pressedKeys[key] = false;
121 }
122}

126bool Piano::isBlackKey(int key)
127{
128 int keyCode = key % KeysPerOctave;
129
130 return KEY_ORDER[keyCode] == Piano::KeyType::Black;
131}

134bool Piano::isWhiteKey(int key)
