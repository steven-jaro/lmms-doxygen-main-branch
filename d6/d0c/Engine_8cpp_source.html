58void Engine::init( bool renderOnly )
59{
60 Engine *engine = inst();
61
62 emit engine->initProgress(tr("Generating wavetables"));
63 // generate (load from file) bandlimited wavetables
64 BandLimitedWave::generateWaves();
65 //initialize oscillators
66 Oscillator::waveTableInit();
67
68 emit engine->initProgress(tr("Initializing data structures"));
69 s_projectJournal = new ProjectJournal;
70 s_audioEngine = new AudioEngine( renderOnly );
71 s_song = new Song;
72 s_mixer = new Mixer;
73 s_patternStore = new PatternStore;
74
75#ifdef LMMS_HAVE_LV2
76 s_lv2Manager = new Lv2Manager;
77 s_lv2Manager->initPlugins();
78#endif
79 s_ladspaManager = new Ladspa2LMMS;
80
81 s_projectJournal->setJournalling( true );
82
83 emit engine->initProgress(tr("Opening audio and midi devices"));
84 s_audioEngine->initDevices();
85
86 PresetPreviewPlayHandle::init();
87
88 emit engine->initProgress(tr("Launching audio engine threads"));
89 s_audioEngine->startProcessing();
90}

95void Engine::destroy()
96{
97 s_projectJournal->stopAllJournalling();
98 s_audioEngine->stopProcessing();
99
100 PresetPreviewPlayHandle::cleanup();
101
102 s_song->clearProject();
103
104 deleteHelper( &s_patternStore );
105
106 deleteHelper( &s_mixer );
107 deleteHelper( &s_audioEngine );
108
109#ifdef LMMS_HAVE_LV2
110 deleteHelper( &s_lv2Manager );
111#endif
112 deleteHelper( &s_ladspaManager );
113
114 //delete ConfigManager::inst();
115 deleteHelper( &s_projectJournal );
116
117 deleteHelper( &s_song );
118
119 delete ConfigManager::inst();
120
121 // The oscillator FFT plans remain throughout the application lifecycle
122 // due to being expensive to create, and being used whenever a userwave form is changed
123 Oscillator::destroyFFTPlans();
124}

129float Engine::framesPerTick(sample_rate_t sampleRate)
130{
131 return sampleRate * 60.0f * 4 /
132 DefaultTicksPerBar / s_song->getTempo();
133}

138void Engine::updateFramesPerTick()
139{
140 s_framesPerTick = s_audioEngine->outputSampleRate() * 60.0f * 4 / DefaultTicksPerBar / s_song->getTempo();
