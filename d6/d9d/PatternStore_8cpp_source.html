47bool PatternStore::play(TimePos start, fpp_t frames, f_cnt_t offset, int clipNum)
48{
49 bool notePlayed = false;
50
51 if (lengthOfPattern(clipNum) <= 0)
52 {
53 return false;
54 }
55
56 start = start % (lengthOfPattern(clipNum) * TimePos::ticksPerBar());
57
58 const TrackList& tl = tracks();
59 for (Track * t : tl)
60 {
61 if (t->play(start, frames, offset, clipNum))
62 {
63 notePlayed = true;
64 }
65 }
66
67 return notePlayed;
68}

84bar_t PatternStore::lengthOfPattern(int pattern) const
85{
86 TimePos maxLength = TimePos::ticksPerBar();
87
88 const TrackList & tl = tracks();
89 for (Track * t : tl)
90 {
91 // Don't create Clips here if they don't exist
92 if (pattern < t->numOfClips())
93 {
94 maxLength = std::max(maxLength, t->getClip(pattern)->length());
95 }
96 }
97
98 return maxLength.nextFullBar();
99}

112void PatternStore::removePattern(int pattern)
113{
114 const TrackList& tl = tracks();
115 for (Track * t : tl)
116 {
117 delete t->getClip(pattern);
118 t->removeBar(pattern * DefaultTicksPerBar);
119 }
120 if (pattern <= currentPattern())
121 {
122 setCurrentPattern(std::max(currentPattern() - 1, 0));
123 }
124}

219AutomatedValueMap PatternStore::automatedValuesAt(TimePos time, int clipNum) const
220{
221 Q_ASSERT(clipNum >= 0);
222 Q_ASSERT(time.getTicks() >= 0);
223
224 auto lengthBars = lengthOfPattern(clipNum);
225 auto lengthTicks = lengthBars * TimePos::ticksPerBar();
226 if (time > lengthTicks)
227 {
228 time = lengthTicks;
229 }
230
231 return TrackContainer::automatedValuesAt(time + (TimePos::ticksPerBar() * clipNum), clipNum);
232}
