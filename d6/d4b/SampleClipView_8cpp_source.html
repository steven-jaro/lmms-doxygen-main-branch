47SampleClipView::SampleClipView( SampleClip * _clip, TrackView * _tv ) :
48 ClipView( _clip, _tv ),
49 m_clip( _clip ),
50 m_paintPixmap(),
51 m_paintPixmapXPosition(0)
52{
53 // update UI and tooltip
54 updateSample();
55
56 // track future changes of SampleClip
57 connect(m_clip, SIGNAL(sampleChanged()), this, SLOT(updateSample()));
58
59 connect(m_clip, SIGNAL(wasReversed()), this, SLOT(update()));
60
61 setStyle( QApplication::style() );
62}

64void SampleClipView::updateSample()
65{
66 update();
67
68 m_sampleThumbnail = SampleThumbnail{m_clip->m_sample};

123void SampleClipView::dropEvent( QDropEvent * _de )
124{
125 if( StringPairDrag::decodeKey( _de ) == "samplefile" )
126 {
127 m_clip->setSampleFile( StringPairDrag::decodeValue( _de ) );
128 _de->accept();
129 }
130 else if( StringPairDrag::decodeKey( _de ) == "sampledata" )
131 {
132 m_clip->setSampleBuffer(SampleLoader::createBufferFromBase64(StringPairDrag::decodeValue(_de)));
133 m_clip->updateLength();
134 update();
135 _de->accept();
136 }
137 else
138 {
139 ClipView::dropEvent( _de );
140 }
141}

146void SampleClipView::mousePressEvent( QMouseEvent * _me )
147{
148 if( _me->button() == Qt::LeftButton &&
149 _me->modifiers() & Qt::ControlModifier &&
150 _me->modifiers() & Qt::ShiftModifier )
151 {
152 m_clip->toggleRecord();
153 }
154 else
155 {
156 if( _me->button() == Qt::MiddleButton && _me->modifiers() == Qt::ControlModifier )
157 {
158 auto sClip = dynamic_cast<SampleClip*>(getClip());
159 if( sClip )
160 {
161 sClip->updateTrackClips();
162 }
163 }
164 ClipView::mousePressEvent( _me );
165 }
166}

187void SampleClipView::mouseDoubleClickEvent( QMouseEvent * )
188{
189 if (m_trackView->trackContainerView()->knifeMode()) { return; }
190
191 const QString selectedAudioFile = SampleLoader::openAudioFile();
192
193 if (selectedAudioFile.isEmpty()) { return; }
194
195 if (!m_clip->hasSampleFileLoaded(selectedAudioFile))
196 {
197 auto sampleBuffer = SampleLoader::createBufferFromFile(selectedAudioFile);
198 if (sampleBuffer != SampleBuffer::emptyBuffer())
199 {
200 m_clip->setSampleBuffer(sampleBuffer);
201 }
202 }
203 m_clip->updateLength();
204}

209void SampleClipView::paintEvent( QPaintEvent * pe )
210{
211 QPainter painter( this );
212
213 if( !needsUpdate() )
214 {
215 painter.drawPixmap(m_paintPixmapXPosition, 0, m_paintPixmap);
216 return;
217 }
218
219 setNeedsUpdate( false );
220
221 const auto trackViewWidth = getTrackView()->rect().width();
222
223 // Use the clip's height to avoid artifacts when rendering while something else is overlaying the clip.
224 const auto viewPortRect = QRect(0, 0, trackViewWidth * 2, rect().height());
225
226 m_paintPixmapXPosition = std::max(0, pe->rect().x() - trackViewWidth);
227
228 if (m_paintPixmap.isNull() || m_paintPixmap.size() != viewPortRect.size())
229 {
230 m_paintPixmap = QPixmap(viewPortRect.size());
231 }
232
233 QPainter p( &m_paintPixmap );
234
235 bool muted = m_clip->getTrack()->isMuted() || m_clip->isMuted();
236 bool selected = isSelected();
237
238 QLinearGradient lingrad(0, 0, 0, height());
239 QColor c = painter.background().color();
240 if (muted) { c = c.darker(150); }
241 if (selected) { c = c.darker(150); }
242
243 lingrad.setColorAt( 1, c.darker( 300 ) );
244 lingrad.setColorAt( 0, c );
245
246 // paint a black rectangle under the clip to prevent glitches with transparent backgrounds
247 p.fillRect( rect(), QColor( 0, 0, 0 ) );
248
249 if( gradient() )
250 {
251 p.fillRect( rect(), lingrad );
252 }
253 else
254 {
255 p.fillRect( rect(), c );
256 }
257
258 auto clipColor = m_clip->color().value_or(m_clip->getTrack()->color().value_or(painter.pen().brush().color()));
259
260 p.setPen(clipColor);
261
262 if (muted)
263 {
264 QColor penColor = p.pen().brush().color();
265 penColor.setHsv(penColor.hsvHue(), penColor.hsvSaturation() / 4, penColor.value());
266 p.setPen(penColor.darker(250));
267 }
268 if (selected)
269 {
270 p.setPen(p.pen().brush().color().darker(150));
271 }
272
273 const int spacing = BORDER_WIDTH + 1;
274 const float ppb = fixedClips() ?
275 ( parentWidget()->width() - 2 * BORDER_WIDTH )
276 / (float) m_clip->length().getBar() :
277 pixelsPerBar();
278
279 float nom = Engine::getSong()->getTimeSigModel().getNumerator();
280 float den = Engine::getSong()->getTimeSigModel().getDenominator();
281 float ticksPerBar = DefaultTicksPerBar * nom / den;
282 float offsetStart = m_clip->startTimeOffset() / ticksPerBar * pixelsPerBar();
283 float sampleLength = m_clip->sampleLength() * ppb / ticksPerBar;
284
285 const auto& sample = m_clip->m_sample;
286
287 const auto sampleRextX = static_cast<int>(offsetStart) - m_paintPixmapXPosition;
288
289 if (sample.sampleSize() > 0)
290 {
291 const auto param = SampleThumbnail::VisualizeParameters{
292 .sampleRect = QRect(sampleRextX, spacing, sampleLength, height() - spacing),
293 .viewportRect = viewPortRect,
294 .amplification = sample.amplification(),
295 .reversed = sample.reversed()
296 };
297
298 m_sampleThumbnail.visualize(param, p);
299 }
300
301 QString name = PathUtil::cleanName(m_clip->m_sample.sampleFile());
302 paintTextLabel(name, p);
303
304 // disable antialiasing for borders, since its not needed
305 p.setRenderHint( QPainter::Antialiasing, false );
306
307 // inner border
308 p.setPen( c.lighter( 135 ) );
309 p.drawRect(
310 -m_paintPixmapXPosition + 1,
311 1,
312 rect().right() - BORDER_WIDTH,
313 rect().bottom() - BORDER_WIDTH );
314
315 // outer border
316 p.setPen( c.darker( 200 ) );
317 p.drawRect(-m_paintPixmapXPosition, 0, rect().right(), rect().bottom());
318
319 // draw the 'muted' pixmap only if the clip was manually muted
320 if( m_clip->isMuted() )
321 {
322 const int spacing = BORDER_WIDTH;
323 const int size = 14;
324 p.drawPixmap( spacing, height() - ( size + spacing ),
325 embed::getIconPixmap( "muted", size, size ) );
326 }
327
328 if ( m_marker )
329 {
330 p.setPen(markerColor());
331 p.drawLine(m_markerPos, rect().bottom(), m_markerPos, rect().top());
332 }
333 // recording sample tracks is not possible at the moment
334
335 /* if( m_clip->isRecord() )
336 {
337 p.setFont( pointSize<7>( p.font() ) );
338
339 p.setPen( textShadowColor() );
340 p.drawText( 10, p.fontMetrics().height()+1, "Rec" );
341 p.setPen( textColor() );
342 p.drawText( 9, p.fontMetrics().height(), "Rec" );
343
344 p.setBrush( QBrush( textColor() ) );
345 p.drawEllipse( 4, 5, 4, 4 );
346 }*/
347
348 p.end();
349
350 painter.drawPixmap(m_paintPixmapXPosition, 0, m_paintPixmap);
351}

356void SampleClipView::reverseSample()
357{
358 m_clip->m_sample.setReversed(!m_clip->m_sample.reversed());
359 m_clip->setStartTimeOffset(m_clip->length() - m_clip->startTimeOffset() - m_clip->sampleLength());
360 Engine::getSong()->setModified();
361 update();
362}
