39LadspaControlView::LadspaControlView( QWidget * _parent,
40 LadspaControl * _ctl ) :
41 QWidget( _parent ),
42 ModelView( _ctl, this ),
43 m_ctl( _ctl )
44{
45 auto layout = new QHBoxLayout(this);
46 layout->setContentsMargins(0, 0, 0, 0);
47 layout->setSpacing( 0 );
48
49 LedCheckBox * link = nullptr;
50
51 if( m_ctl->m_link )
52 {
53 link = new LedCheckBox( "", this );
54 link->setModel( &m_ctl->m_linkEnabledModel );
55 link->setToolTip(tr("Link channels"));
56 layout->addWidget( link );
57 }
58
59 Knob * knb = nullptr;
60
61 switch( m_ctl->port()->data_type )
62 {
63 case BufferDataType::Toggled:
64 {
65 auto toggle = new LedCheckBox(m_ctl->port()->name, this, QString(), LedCheckBox::LedColor::Green);
66 toggle->setModel( m_ctl->toggledModel() );
67 layout->addWidget( toggle );
68 if( link != nullptr )
69 {
70 setFixedSize( link->width() + toggle->width(),
71 toggle->height() );
72 }
73 else
74 {
75 setFixedSize( toggle->width(),
76 toggle->height() );
77 }
78 break;
79 }
80
81 case BufferDataType::Integer:
82 case BufferDataType::Enum:
83 case BufferDataType::Floating:
84 knb = new Knob(KnobType::Bright26, m_ctl->port()->name, this, Knob::LabelRendering::WidgetFont, m_ctl->port()->name);
85 break;
86
87 case BufferDataType::Time:
88 knb = new TempoSyncKnob(KnobType::Bright26, m_ctl->port()->name, this, Knob::LabelRendering::WidgetFont, m_ctl->port()->name);
89 break;
90
91 default:
92 break;
93 }
94
95 if( knb != nullptr )
96 {
97 if( m_ctl->port()->data_type != BufferDataType::Time )
98 {
99 knb->setModel( m_ctl->knobModel() );
100 }
101 else
102 {
103 knb->setModel( m_ctl->tempoSyncKnobModel() );
104 }
105 knb->setHintText( tr( "Value:" ), "" );
106 layout->addWidget( knb );
107 if( link != nullptr )
108 {
109 setFixedSize( link->width() + knb->width(),
110 knb->height() );
111 }
112 else
113 {
114 setFixedSize( knb->width(), knb->height() );
115 }
116 }
117}
