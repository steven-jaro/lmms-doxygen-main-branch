71int JackInternalClient::Open(const char* server_name, const char* name, jack_uuid_t uuid, jack_options_t options, jack_status_t* status)
72{
73 int result;
74 jack_log("JackInternalClient::Open name = %s", name);
75
76 if (strlen(name) >= JACK_CLIENT_NAME_SIZE) {
77 jack_error("\"%s\" is too long to be used as a JACK client name.\n"
78 "Please use %lu characters or less",
79 name,
80 JACK_CLIENT_NAME_SIZE - 1);
81 return -1;
82 }
83
84 strncpy(fServerName, server_name, sizeof(fServerName));
85 fServerName[sizeof(fServerName) - 1] = 0;
86
87 // Open server/client direct channel
88 char name_res[JACK_CLIENT_NAME_SIZE + 1];
89 fChannel->ClientCheck(name, uuid, name_res, JACK_PROTOCOL_VERSION, (int)options, (int*)status, &result, false);
90 if (result < 0) {
91 int status1 = *status;
92 if (status1 & JackVersionError) {
93 jack_error("JACK protocol mismatch %d", JACK_PROTOCOL_VERSION);
94 } else {
95 jack_error("Client name = %s conflits with another running client", name);
96 }
97 goto error;
98 }
99
100 strcpy(fClientControl.fName, name_res);
101
102 // Require new client
103 fChannel->ClientOpen(name_res, &fClientControl.fRefNum, &fEngineControl, &fGraphManager, this, &result);
104 if (result < 0) {
105 jack_error("Cannot open client name = %s", name_res);
106 goto error;
107 }
108
109 SetupDriverSync(false);
110 JackGlobals::fClientTable[fClientControl.fRefNum] = this;
111 JackGlobals::fServerRunning = true;
112 jack_log("JackInternalClient::Open name = %s refnum = %ld", name_res, fClientControl.fRefNum);
113 return 0;
114
115error:
116 fChannel->Close();
117 return -1;
118}

143int JackLoadableInternalClient::Init(const char* so_name)
144{
145 char path_to_so[JACK_PATH_MAX + 1];
146 BuildClientPath(path_to_so, sizeof(path_to_so), so_name);
147
148 fHandle = LoadJackModule(path_to_so);
149 jack_log("JackLoadableInternalClient::JackLoadableInternalClient path_to_so = %s", path_to_so);
150
151 if (fHandle == NULL) {
152 PrintLoadError(so_name);
153 return -1;
154 }
155
156 fFinish = (FinishCallback)GetJackProc(fHandle, "jack_finish");
157 if (fFinish == NULL) {
158 UnloadJackModule(fHandle);
159 jack_error("symbol jack_finish cannot be found in %s", so_name);
160 return -1;
161 }
162
163 fDescriptor = (JackDriverDescFunction)GetJackProc(fHandle, "jack_get_descriptor");
164 if (fDescriptor == NULL) {
165 jack_log("No jack_get_descriptor entry-point for %s", so_name);
166 }
167 return 0;
168}
