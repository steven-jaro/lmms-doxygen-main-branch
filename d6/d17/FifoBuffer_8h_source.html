35template<typename T>
36class FifoBuffer
37{
38public:
39 FifoBuffer(int size) :
40 m_readSem(size),
41 m_writeSem(size),
42 m_readIndex(0),
43 m_writeIndex(0),
44 m_size(size)
45 {
46 m_buffer = new T[size];
47 m_readSem.acquire(size);
48 }
49
50 ~FifoBuffer()
51 {
52 delete[] m_buffer;
53 m_readSem.release(m_size);
54 }
55
56 void write(T element)
57 {
58 m_writeSem.acquire();
59 m_buffer[m_writeIndex++] = element;
60 m_writeIndex %= m_size;
61 m_readSem.release();
62 }
63
64 T read()
65 {
66 m_readSem.acquire();
67 T element = m_buffer[m_readIndex++];
68 m_readIndex %= m_size;
69 m_writeSem.release();
70 return element;
71 }
72
73 void waitUntilRead()
74 {
75 m_writeSem.acquire(m_size);
76 m_writeSem.release(m_size);
77 }
78
79 bool available()
80 {
81 return m_readSem.available();
82 }
83
84
85private:
86 QSemaphore m_readSem;
87 QSemaphore m_writeSem;
88 int m_readIndex;
89 int m_writeIndex;
90 int m_size;
91 T * m_buffer;
92} ;
