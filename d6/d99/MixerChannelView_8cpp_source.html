163void MixerChannelView::contextMenuEvent(QContextMenuEvent*)
164{
165 auto contextMenu = new CaptionMenu(mixerChannel()->m_name, this);
166
167 if (!isMasterChannel()) // no move-options in master
168 {
169 contextMenu->addAction(tr("Move &left"), this, &MixerChannelView::moveChannelLeft);
170 contextMenu->addAction(tr("Move &right"), this, &MixerChannelView::moveChannelRight);
171 }
172
173 contextMenu->addAction(tr("Rename &channel"), this, &MixerChannelView::renameChannel);
174 contextMenu->addSeparator();
175
176 if (!isMasterChannel()) // no remove-option in master
177 {
178 contextMenu->addAction(
179 embed::getIconPixmap("cancel"), tr("R&emove channel"), this, &MixerChannelView::removeChannel);
180 contextMenu->addSeparator();
181 }
182
183 contextMenu->addAction(
184 embed::getIconPixmap("cancel"), tr("Remove &unused channels"), this, &MixerChannelView::removeUnusedChannels);
185 contextMenu->addSeparator();
186
187 auto colorMenu = QMenu{tr("Color"), this};
188 colorMenu.setIcon(embed::getIconPixmap("colorize"));
189 colorMenu.addAction(tr("Change"), this, &MixerChannelView::selectColor);
190 colorMenu.addAction(tr("Reset"), this, &MixerChannelView::resetColor);
191 colorMenu.addAction(tr("Pick random"), this, &MixerChannelView::randomizeColor);
192 contextMenu->addMenu(&colorMenu);
193
194 contextMenu->exec(QCursor::pos());
195 delete contextMenu;
196}

198void MixerChannelView::paintEvent(QPaintEvent*)
199{
200 static constexpr auto innerBorderSize = 3;
201 static constexpr auto outerBorderSize = 1;
202
203 const auto channel = mixerChannel();
204 const auto isActive = m_mixerView->currentMixerChannel() == this;
205 const auto width = rect().width();
206 const auto height = rect().height();
207 auto painter = QPainter{this};
208
209 if (channel->color().has_value() && !channel->m_muteModel.value())
210 {
211 painter.fillRect(rect(), channel->color()->darker(isActive ? 120 : 150));
212 }
213 else { painter.fillRect(rect(), isActive ? backgroundActive().color() : painter.background().color()); }
214
215 // inner border
216 painter.setPen(isActive ? strokeInnerActive() : strokeInnerInactive());
217 painter.drawRect(1, 1, width - innerBorderSize, height - innerBorderSize);
218
219 // outer border
220 painter.setPen(isActive ? strokeOuterActive() : strokeOuterInactive());
221 painter.drawRect(0, 0, width - outerBorderSize, height - outerBorderSize);
222}

276void MixerChannelView::renameFinished()
277{
278 m_inRename = false;
279
280 m_renameLineEdit->deselect();
281 m_renameLineEdit->setReadOnly(true);
282 m_renameLineEdit->setFixedWidth(m_renameLineEdit->width());
283 m_channelNumberLcd->show();
284
285 auto newName = m_renameLineEdit->text();
286 setFocus();
287
288 const auto mc = mixerChannel();
289 if (!newName.isEmpty() && mc->m_name != newName)
290 {
291 mc->m_name = newName;
292 m_renameLineEdit->setText(elideName(newName));
293 Engine::getSong()->setModified();
294 }
295
296 setToolTip(mc->m_name);
297}

306void MixerChannelView::selectColor()

309

311 const auto* colorChooser = ColorChooser{this}.withPalette(ColorChooser::Palette::Mixer);

330bool MixerChannelView::confirmRemoval(int index)
331{
332 // if config variable is set to false, there is no need for user confirmation
333 bool needConfirm = ConfigManager::inst()->value("ui", "mixerchanneldeletionwarning", "1").toInt();
334 if (!needConfirm) { return true; }
335
336 // is the channel is not in use, there is no need for user confirmation
337 if (!getGUI()->mixerView()->getMixer()->isChannelInUse(index)) { return true; }
338
339 QString messageRemoveTrack = tr("This Mixer Channel is being used.\n"
340 "Are you sure you want to remove this channel?\n\n"
341 "Warning: This operation can not be undone.");
342
343 QString messageTitleRemoveTrack = tr("Confirm removal");
344 QString askAgainText = tr("Don't ask again");
345 auto askAgainCheckBox = new QCheckBox(askAgainText, nullptr);
346 connect(askAgainCheckBox, &QCheckBox::stateChanged, [](int state) {
347 // Invert button state, if it's checked we *shouldn't* ask again
348 ConfigManager::inst()->setValue("ui", "mixerchanneldeletionwarning", state ? "0" : "1");
349 });
350
351 QMessageBox mb;
352 mb.setText(messageRemoveTrack);
353 mb.setWindowTitle(messageTitleRemoveTrack);
354 mb.setIcon(QMessageBox::Warning);
355 mb.addButton(QMessageBox::Cancel);
356 mb.addButton(QMessageBox::Ok);
357 mb.setCheckBox(askAgainCheckBox);
358 mb.setDefaultButton(QMessageBox::Cancel);
359
360 int answer = mb.exec();
361
362 return answer == QMessageBox::Ok;
363}
