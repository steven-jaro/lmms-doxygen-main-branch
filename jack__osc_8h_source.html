42/*
43 * Use this as port type in jack_port_register to make it clear that
44 * this MIDI port is used to route OSC messages.
45 *
46 * jack_port_t *osc_in;
47 * osc_in = jack_port_register(client, "osc.in", JACK_DEFAULT_OSC_TYPE,
48 * JackPortIsInput, 0);
49 */
50#define JACK_DEFAULT_OSC_TYPE JACK_DEFAULT_MIDI_TYPE
51
52/*
53 * Use this as value for metadata key JACKEY_EVENT_TYPES
54 * (http://jackaudio.org/metadata/event-type) to mark/query/unmark a port
55 * as OSC carrier in jack_{set,get,remove}_property.
56 *
57 * jack_uuid_t uuid_in = jack_port_uuid(osc_in);
58 *
59 * // set port event type to OSC
60 * jack_set_property(client, uuid_in, JACKEY_EVENT_TYPES,
61 * JACK_EVENT_TYPE__OSC, NULL);
62 *
63 * // query port event type
64 * char *value = NULL;
65 * char *type = NULL;
66 * if( (jack_get_property(uuid, JACKEY_EVENT_TYPES, &value, &type) == 0)
67 * && (strstr(value, JACK_EVENT_TYPE__OSC) != NULL) )
68 * printf("This port routes OSC!\n");
69 * jack_free(value);
70 * jack_free(type);
71 *
72 * // clear port event type
73 * jack_remove_property(client, uuid_in, JACKEY_EVENT_TYPES);
74 */
75#define JACK_EVENT_TYPE__OSC "OSC"
