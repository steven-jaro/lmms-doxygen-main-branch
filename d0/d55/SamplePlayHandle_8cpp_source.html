88void SamplePlayHandle::play( SampleFrame* buffer )
89{
90 const fpp_t fpp = Engine::audioEngine()->framesPerPeriod();
91 //play( 0, _try_parallelizing );
92 if( framesDone() >= totalFrames() )
93 {
94 zeroSampleFrames(buffer, fpp);
95 return;
96 }
97
98 SampleFrame* workingBuffer = buffer;
99 f_cnt_t frames = fpp;
100
101 // apply offset for the first period
102 if( framesDone() == 0 )
103 {
104 zeroSampleFrames(buffer, offset());
105 workingBuffer += offset();
106 frames -= offset();
107 }
108
109 if( !( m_track && m_track->isMuted() )
110 && !(m_patternTrack && m_patternTrack->isMuted()))
111 {
112/* StereoVolumeVector v =
113 { { m_volumeModel->value() / DefaultVolume,
114 m_volumeModel->value() / DefaultVolume } };*/
115 // SamplePlayHandle always plays the sample at its original pitch;
116 // it is used only for previews, SampleTracks and the metronome.
117 if (!m_sample->play(workingBuffer, &m_state, frames))
118 {
119 zeroSampleFrames(workingBuffer, frames);
120 }
121 }
122
123 m_frame += frames;
124}

145f_cnt_t SamplePlayHandle::totalFrames() const
146{
147 return (m_sample->endFrame() - m_sample->startFrame()) *
148 (static_cast<float>(Engine::audioEngine()->outputSampleRate()) / m_sample->sampleRate());
149}
