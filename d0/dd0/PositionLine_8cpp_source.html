45void PositionLine::paintEvent(QPaintEvent* pe)
46{
47 QPainter p(this);
48 auto c = QColor(m_lineColor);
49
50 // If width is 1, we don't need a gradient
51 if (width() == 1)
52 {
53 c.setAlpha(153);
54 p.fillRect(rect(), c);
55 }
56 // If width > 1, we need the gradient
57 else
58 {
59 // Create the gradient trail behind the line
60 QLinearGradient gradient(rect().bottomLeft(), rect().bottomRight());
61 qreal w = (width() - 1.0) / width();
62
63 // If gradient is enabled, we're in focus and we're playing, enable gradient
64 if (m_hasTailGradient &&
65 Engine::getSong()->isPlaying() &&
66 (Engine::getSong()->playMode() == m_playMode))
67 {
68 c.setAlpha(60);
69 gradient.setColorAt(w, c);
70 }
71 else
72 {
73 c.setAlpha(0);
74 gradient.setColorAt(w, c);
75 }
76
77 // Fill in the remaining parts
78 c.setAlpha(0);
79 gradient.setColorAt(0, c);
80 c.setAlpha(153);
81 gradient.setColorAt(1, c);
82
83 // Fill line
84 p.fillRect(rect(), gradient);
85 }
86}

88// NOTE: the move() implementation fixes a bug where the position line would appear
89// in an unexpected location when positioned at the start of the track
90void PositionLine::zoomChange(float zoom)
91{
92 int playHeadPos = x() + width() - 1;
93
94 resize(std::max(8.0 * zoom, 1.0), height());
95 move(playHeadPos - width() + 1, y());
96
97 update();
98}
