112void Scale::loadSettings(const QDomElement &element)
113{
114 m_description = element.attribute("description");
115
116 QDomNode node = element.firstChild();
117 m_intervals.clear();
118
119 while (!node.isNull())
120 {
121 Interval temp;
122 temp.restoreState(node.toElement());
123 m_intervals.push_back(temp);
124 node = node.nextSibling();
125 }
126}
