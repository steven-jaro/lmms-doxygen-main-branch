42StringPairDrag::StringPairDrag( const QString & _key, const QString & _value,
43 const QPixmap & _icon, QWidget * _w ) :
44 QDrag( _w )
45{
46 // For mimeType() and MimeType enum class
47 using namespace Clipboard;
48
49 if( _icon.isNull() && _w )
50 {
51 setPixmap( _w->grab().scaled(
52 64, 64,
53 Qt::KeepAspectRatio,
54 Qt::SmoothTransformation ) );
55 }
56 else
57 {
58 setPixmap( _icon );
59 }
60 QString txt = _key + ":" + _value;
61 auto m = new QMimeData();
62 m->setData( mimeType( MimeType::StringPair ), txt.toUtf8() );
63 setMimeData( m );
64 exec( Qt::CopyAction, Qt::CopyAction );
65}

83bool StringPairDrag::processDragEnterEvent( QDragEnterEvent * _dee,
84 const QString & _allowed_keys )
85{
86 // For mimeType() and MimeType enum class
87 using namespace Clipboard;
88
89 if( !_dee->mimeData()->hasFormat( mimeType( MimeType::StringPair ) ) )
90 {
91 return( false );
92 }
93 QString txt = _dee->mimeData()->data( mimeType( MimeType::StringPair ) );
94 if( _allowed_keys.split( ',' ).contains( txt.section( ':', 0, 0 ) ) )
95 {
96 _dee->acceptProposedAction();
97 return( true );
98 }
99 _dee->ignore();
100 return( false );
101}
