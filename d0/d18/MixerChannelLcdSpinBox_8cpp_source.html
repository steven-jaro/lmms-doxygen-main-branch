44void MixerChannelLcdSpinBox::mouseDoubleClickEvent(QMouseEvent* event)
45{
46 if (!(event->modifiers() & Qt::ShiftModifier) &&
47 !(event->modifiers() & Qt::ControlModifier))
48 {
49 enterValue();
50 return;

61void MixerChannelLcdSpinBox::contextMenuEvent(QContextMenuEvent* event)
62{
63 // for the case, the user clicked right while pressing left mouse-
64 // button, the context-menu appears while mouse-cursor is still hidden
65 // and it isn't shown again until user does something which causes
66 // an QApplication::restoreOverrideCursor()-call...
67 mouseReleaseEvent(nullptr);
68
69 QPointer<CaptionMenu> contextMenu = new CaptionMenu(model()->displayName(), this);
70
71 if (QMenu *mixerMenu = m_tv->createMixerMenu(
72 tr("Assign to:"), tr("New Mixer Channel")))
73 {
74 contextMenu->addMenu(mixerMenu);
75
76 contextMenu->addSeparator();
77 }
78 addDefaultActions(contextMenu);
79 contextMenu->exec(QCursor::pos());
80}

82void MixerChannelLcdSpinBox::enterValue()
83{
84 const auto val = model()->value();
85 const auto min = model()->minValue();
86 const auto max = model()->maxValue();
87 const auto step = model()->step<int>();
88 const auto label = tr("Please enter a new value between %1 and %2:").arg(min).arg(max);
89
90 auto ok = false;
91 const auto newVal = QInputDialog::getInt(this, tr("Set value"), label, val, min, max, step, &ok);
92
93 if (ok) { model()->setValue(newVal); }
94}
