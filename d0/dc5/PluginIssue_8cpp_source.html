33const char *PluginIssue::msgFor(const PluginIssueType &it)
34{
35 switch (it)
36 {
37 case PluginIssueType::UnknownPortFlow:
38 return "unknown port flow for mandatory port";
39 case PluginIssueType::UnknownPortType:
40 return "unknown port type for mandatory port";
41 case PluginIssueType::TooManyInputChannels:
42 return "too many audio input channels";
43 case PluginIssueType::TooManyOutputChannels:
44 return "too many audio output channels";
45 case PluginIssueType::TooManyMidiInputChannels:
46 return "too many MIDI input channels";
47 case PluginIssueType::TooManyMidiOutputChannels:
48 return "too many MIDI output channels";
49 case PluginIssueType::NoOutputChannel:
50 return "no audio output channel";
51 case PluginIssueType::PortHasNoDef:
52 return "port is missing default value";
53 case PluginIssueType::PortHasNoMin:
54 return "port is missing min value";
55 case PluginIssueType::PortHasNoMax:
56 return "port is missing max value";
57 case PluginIssueType::MinGreaterMax:
58 return "port minimum is greater than maximum";
59 case PluginIssueType::DefaultValueNotInRange:
60 return "default value is not in range [min, max]";
61 case PluginIssueType::LogScaleMinMissing:
62 return "logscale requires minimum value";
63 case PluginIssueType::LogScaleMaxMissing:
64 return "logscale requires maximum value";
65 case PluginIssueType::LogScaleMinMaxDifferentSigns:
66 return "logscale with min < 0 < max";
67 case PluginIssueType::FeatureNotSupported:
68 return "required feature not supported";
69 case PluginIssueType::BadPortType:
70 return "unsupported port type";
71 case PluginIssueType::Blocked:
72 return "blocked plugin";
73 case PluginIssueType::NoIssue:
74 return nullptr;
75 }
76 return nullptr;
77}

90bool PluginIssue::operator<(const PluginIssue &other) const
91{
92 return (m_issueType != other.m_issueType)
93 ? m_issueType < other.m_issueType
94 : m_info < other.m_info;
