100 UInt32 WriteNextStateStartAux()
101 {
102 AtomicCounter old_val;
103 AtomicCounter new_val;
104 UInt32 cur_index;
105 UInt32 next_index;
106 bool need_copy;
107 do {
108 old_val = fCounter;
109 new_val = old_val;
110 cur_index = CurArrayIndex(new_val);
111 next_index = NextArrayIndex(new_val);
112 need_copy = (CurIndex(new_val) == NextIndex(new_val));
113 NextIndex(new_val) = CurIndex(new_val); // Invalidate next index
114 } while (!CAS(Counter(old_val), Counter(new_val), (UInt32*)&fCounter));
115 if (need_copy)
116 memcpy(&fState[next_index], &fState[cur_index], sizeof(T));
117 return next_index;
118 }
119
120 void WriteNextStateStopAux()
121 {
122 AtomicCounter old_val;
123 AtomicCounter new_val;
124 do {
125 old_val = fCounter;
126 new_val = old_val;
127 NextIndex(new_val)++; // Set next index
128 } while (!CAS(Counter(old_val), Counter(new_val), (UInt32*)&fCounter));
129 }

163 T* TrySwitchState()
164 {
165 AtomicCounter old_val;
166 AtomicCounter new_val;
167 do {
168 old_val = fCounter;
169 new_val = old_val;
170 CurIndex(new_val) = NextIndex(new_val); // Prepare switch
171 } while (!CAS(Counter(old_val), Counter(new_val), (UInt32*)&fCounter));
172 return &fState[CurArrayIndex(fCounter)]; // Read the counter again
173 }

178 T* TrySwitchState(bool* result)

184 new_val = old_val;
185 *result = (CurIndex(new_val) != NextIndex(new_val));
186 CurIndex(new_val) = NextIndex(new_val); // Prepare switch
187 } while (!CAS(Counter(old_val), Counter(new_val), (UInt32*)&fCounter));
188 return &fState[CurArrayIndex(fCounter)]; // Read the counter again
189 }

194 T* WriteNextStateStart()

198 : NextArrayIndex(fCounter); // We are inside a wrapping WriteNextStateStart call, NextArrayIndex can be read safely
199 return &fState[next_index];
200 }
