224 const int &gzip_compression, const int &interpolation,
225 Microtonal *microtonal_, FFTwrapper *fft_, WatchManager *wm_, const char *prefix_)

228 Plegatomode(false),
229 partoutl(new float[synth_.buffersize]),
230 partoutr(new float[synth_.buffersize]),
231 ctl(synth_, &time_),
232 microtonal(microtonal_),
233 fft(fft_),
234 wm(wm_),
235 memory(alloc),
236 synth(synth_),
237 time(time_),
238 gzip_compression(gzip_compression),
239 interpolation(interpolation)
240{
241 if(prefix_)
242 strncpy(prefix, prefix_, sizeof(prefix));
243 else
244 memset(prefix, 0, sizeof(prefix));
245
246 monomemClear();
247
248 for(int n = 0; n < NUM_KIT_ITEMS; ++n) {
249 kit[n].parent = this;
250 kit[n].Pname = new char [PART_MAX_NAME_LEN];
251 kit[n].adpars = nullptr;
252 kit[n].subpars = nullptr;
253 kit[n].padpars = nullptr;
254 }
255
256 kit[0].adpars = new ADnoteParameters(synth, fft, &time);
257
258 //Part's Insertion Effects init
259 for(int nefx = 0; nefx < NUM_PART_EFX; ++nefx) {
260 partefx[nefx] = new EffectMgr(memory, synth, 1, &time);
261 Pefxbypass[nefx] = false;
262 }
263 assert(partefx[0]);
264
265 for(int n = 0; n < NUM_PART_EFX + 1; ++n) {
266 partfxinputl[n] = new float [synth.buffersize];
267 partfxinputr[n] = new float [synth.buffersize];
268 }
269
270 killallnotes = false;
271 oldfreq = -1.0f;
272
273 cleanup();
274
275 Pname = new char[PART_MAX_NAME_LEN];
276
277 oldvolumel = oldvolumer = 0.5f;
278 lastnote = -1;
279
280 defaults();
281 assert(partefx[0]);
282}

427static int kit_usage(const Part::Kit *kits, int note, int mode)
428{
429 const bool non_kit = mode == 0;
430 const bool singl_kit = mode == 2;
431 int synth_usage = 0;
432
433 for(uint8_t i = 0; i < NUM_KIT_ITEMS; ++i) {
434 const auto &item = kits[i];
435 if(!non_kit && !item.validNote(note))
436 continue;
437
438 synth_usage += item.Padenabled;
439 synth_usage += item.Psubenabled;
440 synth_usage += item.Ppadenabled;
441
442 //Partial Kit Use
443 if(non_kit || (singl_kit && item.active()))
444 break;
445 }
446
447 return synth_usage;
448}

450/*
451 * Note On Messages
452 */
453bool Part::NoteOn(unsigned char note,
454 unsigned char velocity,
455 int masterkeyshift)
456{
457 //Verify Basic Mode and sanity
458 const bool isRunningNote = notePool.existsRunningNote();
459 const bool doingLegato = isRunningNote && isLegatoMode() &&
460 lastlegatomodevalid;
461
462 if(!Pnoteon || !inRange(note, Pminkey, Pmaxkey) || notePool.full() ||
463 notePool.synthFull(kit_usage(kit, note, Pkitmode)))
464 return false;
465
466 verifyKeyMode();
467 assert_kit_sanity(kit);
468
469 //Preserve Note Stack
470 if(isMonoMode() || isLegatoMode()) {
471 monomemPush(note);
472 monomem[note].velocity = velocity;
473 monomem[note].mkeyshift = masterkeyshift;
474
475 } else if(!monomemEmpty())
476 monomemClear();
477
478 //Mono/Legato Release old notes
479 if(isMonoMode() || (isLegatoMode() && !doingLegato))
480 notePool.releasePlayingNotes();
481
482 lastlegatomodevalid = isLegatoMode();
483
484 //Compute Note Parameters
485 const float vel = getVelocity(velocity, Pvelsns, Pveloffs);
486 const int partkeyshift = (int)Pkeyshift - 64;
487 const int keyshift = masterkeyshift + partkeyshift;
488 const float notebasefreq = getBaseFreq(note, keyshift);
489
490 if(notebasefreq < 0)
491 return false;
492
493 //Portamento
494 lastnote = note;
495 if(oldfreq < 1.0f)
496 oldfreq = notebasefreq;//this is only the first note is played
497
498 // For Mono/Legato: Force Portamento Off on first
499 // notes. That means it is required that the previous note is
500 // still held down or sustained for the Portamento to activate
501 // (that's like Legato).
502 bool portamento = false;
503 if(Ppolymode || isRunningNote)
504 portamento = ctl.initportamento(oldfreq, notebasefreq, doingLegato);
505
506 oldfreq = notebasefreq;
507
508 //Adjust Existing Notes
509 if(doingLegato) {
510 LegatoParams pars = {notebasefreq, vel, portamento, note, true};
511 notePool.applyLegato(pars);
512 return true;
513 }
514
515 if(Ppolymode)
516 notePool.makeUnsustainable(note);
517
518 //Create New Notes
519 for(uint8_t i = 0; i < NUM_KIT_ITEMS; ++i) {
520 ScratchString pre = prefix;
521 auto &item = kit[i];
522 if(Pkitmode != 0 && !item.validNote(note))
523 continue;
524
525 SynthParams pars{memory, ctl, synth, time, notebasefreq, vel,
526 portamento, note, false};
527 const int sendto = Pkitmode ? item.sendto() : 0;
528
529 try {
530 if(item.Padenabled)
531 notePool.insertNote(note, sendto,
532 {memory.alloc<ADnote>(kit[i].adpars, pars,
533 wm, (pre+"kit"+i+"/adpars/").c_str), 0, i});
534 if(item.Psubenabled)
535 notePool.insertNote(note, sendto,
536 {memory.alloc<SUBnote>(kit[i].subpars, pars), 1, i});
537 if(item.Ppadenabled)
538 notePool.insertNote(note, sendto,
539 {memory.alloc<PADnote>(kit[i].padpars, pars, interpolation, wm,
540 (pre+"kit"+i+"/padpars/").c_str), 2, i});
541 } catch (std::bad_alloc & ba) {
542 std::cerr << "dropped new note: " << ba.what() << std::endl;
543 }
544
545 //Partial Kit Use
546 if(isNonKit() || (isSingleKit() && item.active()))
547 break;
548 }
549
550 if(isLegatoMode())
551 notePool.upgradeToLegato();
552
553 //Enforce the key limit
554 setkeylimit(Pkeylimit);
555 return true;
556}

558/*
559 * Note Off Messages
560 */
561void Part::NoteOff(unsigned char note) //release the key
562{
563 // This note is released, so we remove it from the list.
564 if(!monomemEmpty())
565 monomemPop(note);
566
567 for(auto &desc:notePool.activeDesc()) {
568 if(desc.note != note || !desc.playing())
569 continue;
570 if(!ctl.sustain.sustain) { //the sustain pedal is not pushed
571 if((isMonoMode() || isLegatoMode()) && !monomemEmpty())
572 MonoMemRenote();//Play most recent still active note
573 else
574 notePool.release(desc);
575 }
576 else { //the sustain pedal is pushed
577 if(desc.canSustain())
578 desc.doSustain();
579 else
580 notePool.release(desc);
581 }
582 }
583}

585void Part::PolyphonicAftertouch(unsigned char note,
586 unsigned char velocity,
587 int masterkeyshift)
588{
589 (void) masterkeyshift;
590
591 if(!Pnoteon || !inRange(note, Pminkey, Pmaxkey) || Pdrummode)

593

595 if(!Ppolymode) // if Poly is off

602 s.note->setVelocity(vel);
603 }

609void Part::SetController(unsigned int type, int par)
610{
611 switch(type) {
612 case C_pitchwheel:
613 ctl.setpitchwheel(par);
614 break;
615 case C_expression:
616 ctl.setexpression(par);
617 setPvolume(Pvolume); //update the volume
618 break;
619 case C_portamento:
620 ctl.setportamento(par);
621 break;
622 case C_panning:
623 ctl.setpanning(par);
624 setPpanning(Ppanning); //update the panning
625 break;
626 case C_filtercutoff:
627 ctl.setfiltercutoff(par);
628 break;
629 case C_filterq:
630 ctl.setfilterq(par);
631 break;
632 case C_bandwidth:
633 ctl.setbandwidth(par);
634 break;
635 case C_modwheel:
636 ctl.setmodwheel(par);
637 break;
638 case C_fmamp:
639 ctl.setfmamp(par);
640 break;
641 case C_volume:
642 ctl.setvolume(par);
643 if(ctl.volume.receive != 0)
644 volume = ctl.volume.volume;
645 else
646 setPvolume(Pvolume);
647 break;
648 case C_sustain:
649 ctl.setsustain(par);
650 if(ctl.sustain.sustain == 0)
651 ReleaseSustainedKeys();
652 break;
653 case C_allsoundsoff:
654 AllNotesOff(); //Panic
655 break;
656 case C_resetallcontrollers:
657 ctl.resetall();
658 ReleaseSustainedKeys();
659 if(ctl.volume.receive != 0)
660 volume = ctl.volume.volume;
661 else
662 setPvolume(Pvolume);
663 setPvolume(Pvolume); //update the volume
664 setPpanning(Ppanning); //update the panning
665
666 for(int item = 0; item < NUM_KIT_ITEMS; ++item) {

789/*
790 * Compute Part samples and store them in the partoutl[] and partoutr[]
791 */
792void Part::ComputePartSmps()
793{
794 assert(partefx[0]);
795 for(unsigned nefx = 0; nefx < NUM_PART_EFX + 1; ++nefx) {
796 memset(partfxinputl[nefx], 0, synth.bufferbytes);
797 memset(partfxinputr[nefx], 0, synth.bufferbytes);
798 }
799
800 for(auto &d:notePool.activeDesc()) {
801 d.age++;
802 for(auto &s:notePool.activeNotes(d)) {
803 float tmpoutr[synth.buffersize];
804 float tmpoutl[synth.buffersize];
805 auto &note = *s.note;
806 note.noteout(&tmpoutl[0], &tmpoutr[0]);
807
808 for(int i = 0; i < synth.buffersize; ++i) { //add the note to part(mix)
809 partfxinputl[d.sendto][i] += tmpoutl[i];
810 partfxinputr[d.sendto][i] += tmpoutr[i];
811 }
812
813 if(note.finished())
814 notePool.kill(s);
815 }
816 }
817
818 //Apply part's effects and mix them
819 for(int nefx = 0; nefx < NUM_PART_EFX; ++nefx) {
820 if(!Pefxbypass[nefx]) {
821 partefx[nefx]->out(partfxinputl[nefx], partfxinputr[nefx]);
822 if(Pefxroute[nefx] == 2)
823 for(int i = 0; i < synth.buffersize; ++i) {
824 partfxinputl[nefx + 1][i] += partefx[nefx]->efxoutl[i];
825 partfxinputr[nefx + 1][i] += partefx[nefx]->efxoutr[i];
826 }
827 }
828 int routeto = ((Pefxroute[nefx] == 0) ? nefx + 1 : NUM_PART_EFX);
829 for(int i = 0; i < synth.buffersize; ++i) {
830 partfxinputl[routeto][i] += partfxinputl[nefx][i];
831 partfxinputr[routeto][i] += partfxinputr[nefx][i];
832 }
833 }
834 for(int i = 0; i < synth.buffersize; ++i) {
835 partoutl[i] = partfxinputl[NUM_PART_EFX][i];
836 partoutr[i] = partfxinputr[NUM_PART_EFX][i];
837 }
838
839 if(killallnotes) {
840 for(int i = 0; i < synth.buffersize; ++i) {
841 float tmp = (synth.buffersize_f - i) / synth.buffersize_f;
842 partoutl[i] *= tmp;
843 partoutr[i] *= tmp;
844 }
845 notePool.killAllNotes();
846 monomemClear();
847 killallnotes = false;
848 for(int nefx = 0; nefx < NUM_PART_EFX; ++nefx)
849 partefx[nefx]->cleanup();
850 }
851 ctl.updateportamento();
852}
