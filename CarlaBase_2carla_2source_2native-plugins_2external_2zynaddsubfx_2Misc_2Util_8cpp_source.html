45/*
46 * Transform the velocity according the scaling parameter (velocity sensing)
47 */
48float VelF(float velocity, unsigned char scaling)
49{
50 float x;
51 x = powf(VELOCITY_MAX_SCALE, (64.0f - scaling) / 64.0f);
52 if((scaling == 127) || (velocity > 0.99f))
53 return 1.0f;
54 else
55 return powf(velocity, x);
56}

58/*
59 * Get the detune in cents
60 */
61float getdetune(unsigned char type,
62 unsigned short int coarsedetune,
63 unsigned short int finedetune)
64{
65 float det = 0.0f, octdet = 0.0f, cdet = 0.0f, findet = 0.0f;
66 //Get Octave
67 int octave = coarsedetune / 1024;
68 if(octave >= 8)
69 octave -= 16;
70 octdet = octave * 1200.0f;
71
72 //Coarse and fine detune
73 int cdetune = coarsedetune % 1024;
74 if(cdetune > 512)
75 cdetune -= 1024;
76
77 int fdetune = finedetune - 8192;
78
79 switch(type) {
80// case 1: is used for the default (see below)
81 case 2:
82 cdet = fabs(cdetune * 10.0f);
83 findet = fabs(fdetune / 8192.0f) * 10.0f;
84 break;
85 case 3:
86 cdet = fabs(cdetune * 100.0f);
87 findet = powf(10, fabs(fdetune / 8192.0f) * 3.0f) / 10.0f - 0.1f;
88 break;
89 case 4:
90 cdet = fabs(cdetune * 701.95500087f); //perfect fifth
91 findet =
92 (powf(2, fabs(fdetune / 8192.0f) * 12.0f) - 1.0f) / 4095 * 1200;
93 break;
94 //case ...: need to update N_DETUNE_TYPES, if you'll add more
95 default:
96 cdet = fabs(cdetune * 50.0f);
97 findet = fabs(fdetune / 8192.0f) * 35.0f; //almost like "Paul's Sound Designer 2"
98 break;
99 }
100 if(finedetune < 8192)
101 findet = -findet;
102 if(cdetune < 0)
103 cdet = -cdet;
104
105 det = octdet + cdet + findet;
106 return det;
107}

160constexpr std::size_t max_pid_len = 12;
161
163std::size_t os_guess_pid_length()
164{
165 const char* pid_max_file = "/proc/sys/kernel/pid_max";
166 if(-1 == access(pid_max_file, R_OK)) {
167 return max_pid_len;
168 }
169 else {
170 std::ifstream is(pid_max_file);
171 if(!is.good())
172 return max_pid_len;
173 else {
174 std::string s;
175 is >> s;
176 for(const auto& c : s)
177 if(c < '0' || c > '9')
178 return max_pid_len;
179 return std::min(s.length(), max_pid_len);
180 }
181 }
182}

185std::string os_pid_as_padded_string()
186{
187 char result_str[max_pid_len << 1];
188 std::fill_n(result_str, max_pid_len, '0');
189 std::size_t written = snprintf(result_str + max_pid_len, max_pid_len,
190 "%d", (int)getpid());
191 // the below pointer should never cause segfaults:
192 return result_str + max_pid_len + written - os_guess_pid_length();
193}

216float interpolate(const float *data, size_t len, float pos)
217{
218 assert(len > (size_t)pos + 1);
219 const int l_pos = (int)pos,
220 r_pos = l_pos + 1;
221 const float leftness = pos - l_pos;
222 return data[l_pos] * leftness + data[r_pos] * (1.0f - leftness);
223}
224
225float cinterpolate(const float *data, size_t len, float pos)
226{
227 const unsigned int i_pos = pos,
228 l_pos = i_pos % len,
229 r_pos = l_pos + 1 < len ? l_pos + 1 : 0;
230 const float leftness = pos - i_pos;
231 return data[l_pos] * leftness + data[r_pos] * (1.0f - leftness);
232}
