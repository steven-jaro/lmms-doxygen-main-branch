57#if DISTRHO_PLUGIN_HAS_EXTERNAL_UI
58struct PluginApplication
59{
60 DGL_NAMESPACE::IdleCallback* idleCallback;
61 UI* ui;
62
63 explicit PluginApplication()
64 : idleCallback(nullptr),
65 ui(nullptr) {}
66
67 void addIdleCallback(DGL_NAMESPACE::IdleCallback* const cb)
68 {
69 DISTRHO_SAFE_ASSERT_RETURN(cb != nullptr,);
70 DISTRHO_SAFE_ASSERT_RETURN(idleCallback == nullptr,);
71
72 idleCallback = cb;
73 }
74
75 bool isQuitting() const noexcept
76 {
77 return ui->isQuitting();
78 }
79
80 bool isStandalone() const noexcept
81 {
82 return DISTRHO_UI_IS_STANDALONE;
83 }
84
85 void exec()
86 {
87 while (ui->isRunning())
88 {
89 d_msleep(30);
90 idleCallback->idleCallback();
91 }
92
93 if (! ui->isQuitting())
94 ui->close();
95 }
96
97 // these are not needed
98 void idle() {}
99 void quit() {}
100 void triggerIdleCallbacks() {}
101
102 DISTRHO_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(PluginApplication)
103};

434// UI private data fileRequestCallback, which requires PluginWindow definitions
435
436inline bool UI::PrivateData::fileRequestCallback(const char* const key)
437{
438 if (fileRequestCallbackFunc != nullptr)
439 return fileRequestCallbackFunc(callbacksPtr, key);
440
441#if DISTRHO_PLUGIN_WANT_STATE && DISTRHO_UI_FILE_BROWSER && !DISTRHO_PLUGIN_HAS_EXTERNAL_UI
442 std::free(uiStateFileKeyRequest);
443 uiStateFileKeyRequest = strdup(key);
444 DISTRHO_SAFE_ASSERT_RETURN(uiStateFileKeyRequest != nullptr, false);
445
446 char title[0xff];
447 snprintf(title, sizeof(title)-1u, DISTRHO_PLUGIN_NAME ": %s", key);
448 title[sizeof(title)-1u] = '\0';
449
450 DGL_NAMESPACE::FileBrowserOptions opts;
451 opts.title = title;
452 return window->openFileBrowser(opts);
453#endif
454
455 return false;
456}
