303PADnoteParameters::PADnoteParameters(const SYNTH_T &synth_, FFTwrapper *fft_,
304 const AbsTime *time_)
305 : Presets(), time(time_), last_update_timestamp(0), synth(synth_)
306{
307 setpresettype("Ppadsynth");
308
309 resonance = new Resonance();
310 oscilgen = new OscilGen(synth, fft_, resonance);
311 oscilgen->ADvsPAD = true;
312
313 FreqEnvelope = new EnvelopeParams(0, 0, time_);
314 FreqEnvelope->init(EnvelopeParams::ad_global_freq_env);
315 FreqLfo = new LFOParams(70, 0, 64, 0, 0, 0, 0, 0, time_);
316
317 AmpEnvelope = new EnvelopeParams(64, 1, time_);
318 AmpEnvelope->init(EnvelopeParams::ad_global_amp_env);
319 AmpLfo = new LFOParams(80, 0, 64, 0, 0, 0, 0, 1, time_);
320
321 GlobalFilter = new FilterParams(2, 94, 40, time_);
322 FilterEnvelope = new EnvelopeParams(0, 1, time_);
323 FilterEnvelope->init(EnvelopeParams::ad_global_filter_env);
324 FilterLfo = new LFOParams(80, 0, 64, 0, 0, 0, 0, 2, time_);
325
326 for(int i = 0; i < PAD_MAX_SAMPLES; ++i)
327 sample[i].smp = NULL;
328
329 defaults();
330}

430/*
431 * Get the harmonic profile (i.e. the frequency distributio of a single harmonic)
432 */
433float PADnoteParameters::getprofile(float *smp, int size)
434{
435 for(int i = 0; i < size; ++i)
436 smp[i] = 0.0f;
437 const int supersample = 16;
438 float basepar = powf(2.0f, (1.0f - Php.base.par1 / 127.0f) * 12.0f);
439 float freqmult = floor(powf(2.0f,
440 Php.freqmult / 127.0f
441 * 5.0f) + 0.000001f);
442
443 float modfreq = floor(powf(2.0f,
444 Php.modulator.freq / 127.0f
445 * 5.0f) + 0.000001f);
446 float modpar1 = powf(Php.modulator.par1 / 127.0f, 4.0f) * 5.0f / sqrt(
447 modfreq);
448 float amppar1 =
449 powf(2.0f, powf(Php.amp.par1 / 127.0f, 2.0f) * 10.0f) - 0.999f;
450 float amppar2 = (1.0f - Php.amp.par2 / 127.0f) * 0.998f + 0.001f;
451 float width = powf(150.0f / (Php.width + 22.0f), 2.0f);
452
453 for(int i = 0; i < size * supersample; ++i) {
454 bool makezero = false;
455 float x = i * 1.0f / (size * (float) supersample);
456
457 float origx = x;
458
459 //do the sizing (width)
460 x = (x - 0.5f) * width + 0.5f;
461 if(x < 0.0f) {
462 x = 0.0f;
463 makezero = true;
464 }
465 else
466 if(x > 1.0f) {
467 x = 1.0f;
468 makezero = true;
469 }
470
471 //compute the full profile or one half
472 switch(Php.onehalf) {
473 case 1:
474 x = x * 0.5f + 0.5f;
475 break;
476 case 2:
477 x = x * 0.5f;
478 break;
479 }
480
481 float x_before_freq_mult = x;
482
483 //do the frequency multiplier
484 x *= freqmult;
485
486 //do the modulation of the profile
487 x += sinf(x_before_freq_mult * 3.1415926f * modfreq) * modpar1;
488 x = fmod(x + 1000.0f, 1.0f) * 2.0f - 1.0f;
489
490
491 //this is the base function of the profile
492 float f;
493 switch(Php.base.type) {
494 case 1:
495 f = expf(-(x * x) * basepar);
496 if(f < 0.4f)
497 f = 0.0f;
498 else
499 f = 1.0f;
500 break;
501 case 2:
502 f = expf(-(fabs(x)) * sqrt(basepar));
503 break;
504 default:
505 f = expf(-(x * x) * basepar);
506 break;
507 }
508 if(makezero)
509 f = 0.0f;
510
511 float amp = 1.0f;
512 origx = origx * 2.0f - 1.0f;
513
514 //compute the amplitude multiplier
515 switch(Php.amp.type) {
516 case 1:
517 amp = expf(-(origx * origx) * 10.0f * amppar1);
518 break;
519 case 2:
520 amp = 0.5f
521 * (1.0f
522 + cosf(3.1415926f * origx * sqrt(amppar1 * 4.0f + 1.0f)));
523 break;
524 case 3:
525 amp = 1.0f
526 / (powf(origx * (amppar1 * 2.0f + 0.8f), 14.0f) + 1.0f);
527 break;
528 }
529
530 //apply the amplitude multiplier
531 float finalsmp = f;
532 if(Php.amp.type != 0)
533 switch(Php.amp.mode) {
534 case 0:
535 finalsmp = amp * (1.0f - amppar2) + finalsmp * amppar2;
536 break;
537 case 1:
538 finalsmp *= amp * (1.0f - amppar2) + amppar2;
539 break;
540 case 2:
541 finalsmp = finalsmp
542 / (amp + powf(amppar2, 4.0f) * 20.0f + 0.0001f);
543 break;
544 case 3:
545 finalsmp = amp
546 / (finalsmp
547 + powf(amppar2, 4.0f) * 20.0f + 0.0001f);
548 break;
549 }
550 ;
551
552 smp[i / supersample] += finalsmp / supersample;
553 }
554
555 //normalize the profile (make the max. to be equal to 1.0f)
556 float max = 0.0f;
557 for(int i = 0; i < size; ++i) {
558 if(smp[i] < 0.0f)
559 smp[i] = 0.0f;
560 if(smp[i] > max)
561 max = smp[i];
562 }
563 if(max < 0.00001f)
564 max = 1.0f;
565 for(int i = 0; i < size; ++i)
566 smp[i] /= max;
567
568 if(!Php.autoscale)
569 return 0.5f;
570
571 //compute the estimated perceived bandwidth
572 float sum = 0.0f;
573 int i;
574 for(i = 0; i < size / 2 - 2; ++i) {
575 sum += smp[i] * smp[i] + smp[size - i - 1] * smp[size - i - 1];
576 if(sum >= 4.0f)
577 break;
578 }
579
580 float result = 1.0f - 2.0f * i / (float) size;
581 return result;
582}

596/*
597 * Get the harmonic(overtone) position
598 */
599float PADnoteParameters::getNhr(int n) const
600{
601 float result = 1.0f;
602 const float par1 = powf(10.0f, -(1.0f - Phrpos.par1 / 255.0f) * 3.0f);
603 const float par2 = Phrpos.par2 / 255.0f;
604
605 const float n0 = n - 1.0f;
606 float tmp = 0.0f;
607 int thresh = 0;
608 switch(Phrpos.type) {
609 case 1:
610 thresh = (int)(par2 * par2 * 100.0f) + 1;
611 if(n < thresh)
612 result = n;
613 else
614 result = 1.0f + n0 + (n0 - thresh + 1.0f) * par1 * 8.0f;
615 break;
616 case 2:
617 thresh = (int)(par2 * par2 * 100.0f) + 1;
618 if(n < thresh)
619 result = n;
620 else
621 result = 1.0f + n0 - (n0 - thresh + 1.0f) * par1 * 0.90f;
622 break;
623 case 3:
624 tmp = par1 * 100.0f + 1.0f;
625 result = powf(n0 / tmp, 1.0f - par2 * 0.8f) * tmp + 1.0f;
626 break;
627 case 4:
628 result = n0
629 * (1.0f
630 - par1)
631 + powf(n0 * 0.1f, par2 * 3.0f
632 + 1.0f) * par1 * 10.0f + 1.0f;
633 break;
634 case 5:
635 result = n0
636 + sinf(n0 * par2 * par2 * PI
637 * 0.999f) * sqrt(par1) * 2.0f + 1.0f;
638 break;
639 case 6:
640 tmp = powf(par2 * 2.0f, 2.0f) + 0.1f;
641 result = n0 * powf(1.0f + par1 * powf(n0 * 0.8f, tmp), tmp) + 1.0f;
642 break;
643 case 7:
644 result = (n + Phrpos.par1 / 255.0f) / (Phrpos.par1 / 255.0f + 1);
645 break;
646 default:
647 result = n;
648 break;
649 }
650
651 const float par3 = Phrpos.par3 / 255.0f;
652
653 const float iresult = floor(result + 0.5f);
654 const float dresult = result - iresult;
655
656 return iresult + (1.0f - par3) * dresult;
657}

659//Transform non zero positive signals into ones with a max of one
660static void normalize_max(float *f, size_t len)
661{
662 float max = 0.0f;
663 for(unsigned i = 0; i < len; ++i)
664 if(f[i] > i)
665 max = f[i];
666 if(max > 0.000001f)
667 for(unsigned i = 0; i < len; ++i)
668 f[i] /= max;
669}

671//Translate Bandwidth scale integer into floating point value
672static float Pbwscale_translate(char Pbwscale)
673{
674 switch(Pbwscale) {
675 case 0: return 1.0f;
676 case 1: return 0.0f;
677 case 2: return 0.25f;
678 case 3: return 0.5f;
679 case 4: return 0.75f;
680 case 5: return 1.5f;
681 case 6: return 2.0f;
682 case 7: return -0.5f;
683 default: return 1.0;
684 }
685}

687/*
688 * Generates the long spectrum for Bandwidth mode (only amplitudes are generated; phases will be random)
689 */
690
691//Requires
692// - bandwidth scaling power
693// - bandwidth
694// - oscilator harmonics at various frequences (oodles of data)
695// - sampled resonance
696void PADnoteParameters::generatespectrum_bandwidthMode(float *spectrum,
697 int size,
698 float basefreq,
699 const float *profile,
700 int profilesize,
701 float bwadjust) const
702{
703 float harmonics[synth.oscilsize];
704 memset(spectrum, 0, sizeof(float) * size);
705 memset(harmonics, 0, sizeof(float) * synth.oscilsize);
706
707 //get the harmonic structure from the oscillator (I am using the frequency amplitudes, only)
708 oscilgen->get(harmonics, basefreq, false);
709
710 //normalize
711 normalize_max(harmonics, synth.oscilsize / 2);
712
713 //Constants across harmonics
714 const float power = Pbwscale_translate(Pbwscale);
715 const float bandwidthcents = const_cast<PADnoteParameters*>(this)->setPbandwidth(Pbandwidth);
716
717 for(int nh = 1; nh < synth.oscilsize / 2; ++nh) { //for each harmonic
718 const float realfreq = getNhr(nh) * basefreq;
719 if(realfreq > synth.samplerate_f * 0.49999f)
720 break;
721 if(realfreq < 20.0f)
722 break;
723 if(harmonics[nh - 1] < 1e-4)
724 continue;
725
726 //compute the bandwidth of each harmonic
727 const float bw =
728 ((powf(2.0f, bandwidthcents / 1200.0f) - 1.0f) * basefreq / bwadjust)
729 * powf(realfreq / basefreq, power);
730 const int ibw = (int)((bw / (synth.samplerate_f * 0.5f) * size)) + 1;
731
732 float amp = harmonics[nh - 1];
733 if(resonance->Penabled)
734 amp *= resonance->getfreqresponse(realfreq);
735
736 if(ibw > profilesize) { //if the bandwidth is larger than the profilesize
737 const float rap = sqrt((float)profilesize / (float)ibw);
738 const int cfreq =
739 (int) (realfreq
740 / (synth.samplerate_f * 0.5f) * size) - ibw / 2;
741 for(int i = 0; i < ibw; ++i) {
742 const int src = i * rap * rap;
743 const int spfreq = i + cfreq;
744 if(spfreq < 0)
745 continue;
746 if(spfreq >= size)
747 break;
748 spectrum[spfreq] += amp * profile[src] * rap;
749 }
750 }
751 else { //if the bandwidth is smaller than the profilesize
752 const float rap = sqrt((float)ibw / (float)profilesize);
753 const float ibasefreq = realfreq / (synth.samplerate_f * 0.5f) * size;
754 for(int i = 0; i < profilesize; ++i) {
755 const float idfreq = (i / (float)profilesize - 0.5f) * ibw;
756 const float freqsum = idfreq + ibasefreq;
757 const int spfreq = (int)freqsum;
758 const float fspfreq = freqsum - spfreq;
759 if(spfreq <= 0)
760 continue;
761 if(spfreq >= size - 1)
762 break;
763 spectrum[spfreq] += amp * profile[i] * rap
764 * (1.0f - fspfreq);
765 spectrum[spfreq + 1] += amp * profile[i] * rap * fspfreq;
766 }
767 }
768 }
769}

771/*
772 * Generates the long spectrum for non-Bandwidth modes (only amplitudes are generated; phases will be random)
773 */
774void PADnoteParameters::generatespectrum_otherModes(float *spectrum,
775 int size,
776 float basefreq) const
777{
778 float harmonics[synth.oscilsize];
779 memset(spectrum, 0, sizeof(float) * size);
780 memset(harmonics, 0, sizeof(float) * synth.oscilsize);
781
782 //get the harmonic structure from the oscillator (I am using the frequency amplitudes, only)
783 oscilgen->get(harmonics, basefreq, false);
784
785 //normalize
786 normalize_max(harmonics, synth.oscilsize / 2);
787
788 for(int nh = 1; nh < synth.oscilsize / 2; ++nh) { //for each harmonic
789 const float realfreq = getNhr(nh) * basefreq;
790
791 //take care of interpolation if frequency decreases
792 if(realfreq > synth.samplerate_f * 0.49999f)
793 break;
794 if(realfreq < 20.0f)
795 break;
796
797
798 float amp = harmonics[nh - 1];
799 if(resonance->Penabled)
800 amp *= resonance->getfreqresponse(realfreq);
801 const int cfreq = realfreq / (synth.samplerate_f * 0.5f) * size;
802
803 spectrum[cfreq] = amp + 1e-9;
804 }
805
806 //In continous mode the spectrum gets additional interpolation between the
807 //spectral peaks
808 if(Pmode != 1) { //continous mode
809 int old = 0;
810 for(int k = 1; k < size; ++k)
811 if((spectrum[k] > 1e-10) || (k == (size - 1))) {
812 const int delta = k - old;
813 const float val1 = spectrum[old];
814 const float val2 = spectrum[k];
815 const float idelta = 1.0f / delta;
816 for(int i = 0; i < delta; ++i) {
817 const float x = idelta * i;
818 spectrum[old + i] = val1 * (1.0f - x) + val2 * x;
819 }
820 old = k;
821 }
822 }
823}

834 unsigned max_threads)
835{

837 return;

850//Requires
851// - Pquality.samplesize
852// - Pquality.basenote
853// - Pquality.oct
854// - Pquality.smpoct
855// - spectrum at various frequencies (oodles of data)
856int PADnoteParameters::sampleGenerator(PADnoteParameters::callback cb,
857 std::function<bool()> do_abort,
858 unsigned max_threads)
859{
860 if(!max_threads)
861 max_threads = std::numeric_limits<unsigned>::max();
862
863 const int samplesize = (((int) 1) << (Pquality.samplesize + 14));
864 const int spectrumsize = samplesize / 2;
865 const int profilesize = 512;
866
867 float profile[profilesize];
868
869
870 const float bwadjust = getprofile(profile, profilesize);
871 float basefreq = 65.406f * powf(2.0f, Pquality.basenote / 2);
872 if(Pquality.basenote % 2 == 1)
873 basefreq *= 1.5f;
874
875 int samplemax = Pquality.oct + 1;
876 int smpoct = Pquality.smpoct;
877 if(Pquality.smpoct == 5)
878 smpoct = 6;
879 if(Pquality.smpoct == 6)
880 smpoct = 12;
881 if(smpoct != 0)
882 samplemax *= smpoct;
883 else
884 samplemax = samplemax / 2 + 1;
885 if(samplemax == 0)
886 samplemax = 1;
887
888 if(samplemax > PAD_MAX_SAMPLES)
889 samplemax = PAD_MAX_SAMPLES;
890
891 //this is used to compute frequency relation to the base frequency
892 float adj[samplemax];
893 for(int nsample = 0; nsample < samplemax; ++nsample)
894 adj[nsample] = (Pquality.oct + 1.0f) * (float)nsample / samplemax;
895
896 const PADnoteParameters* this_c = this;
897
898 auto thread_cb = [basefreq, bwadjust, &cb, do_abort,
899 samplesize, samplemax, spectrumsize,
900 &adj, &profile, this_c](
901 unsigned nthreads, unsigned threadno)
902 {
903 //prepare a BIG IFFT
904 FFTwrapper *fft = new FFTwrapper(samplesize);
905 fft_t *fftfreqs = new fft_t[samplesize / 2];
906 float *spectrum = new float[spectrumsize];
907
908 for(int nsample = 0; nsample < samplemax; ++nsample)
909 if(nsample % nthreads == threadno)
910 {
911 if(do_abort())
912 break;
913 const float basefreqadjust =
914 powf(2.0f, adj[nsample] - adj[samplemax - 1] * 0.5f);
915
916 if(this_c->Pmode == 0)
917 this_c->generatespectrum_bandwidthMode(spectrum,
918 spectrumsize,
919 basefreq*basefreqadjust,
920 profile,
921 profilesize,
922 bwadjust);
923 else
924 this_c->generatespectrum_otherModes(spectrum, spectrumsize,
925 basefreq * basefreqadjust);
926
927 //the last samples contains the first samples
928 //(used for linear/cubic interpolation)
929 const int extra_samples = 5;
930 PADnoteParameters::Sample newsample;
931 newsample.smp = new float[samplesize + extra_samples];
932
933 newsample.smp[0] = 0.0f;
934 for(int i = 1; i < spectrumsize; ++i) //randomize the phases
935 fftfreqs[i] = FFTpolar(spectrum[i], (float)RND * 2 * PI);
936 //that's all; here is the only ifft for the whole sample;
937 //no windows are used ;-)
938 fft->freqs2smps(fftfreqs, newsample.smp);
939
940
941 //normalize(rms)
942 float rms = 0.0f;
943 for(int i = 0; i < samplesize; ++i)
944 rms += newsample.smp[i] * newsample.smp[i];
945 rms = sqrt(rms);
946 if(rms < 0.000001f)
947 rms = 1.0f;
948 rms *= sqrt(262144.0f / samplesize);//262144=2^18
949 for(int i = 0; i < samplesize; ++i)
950 newsample.smp[i] *= 1.0f / rms * 50.0f;
951
952 //prepare extra samples used by the linear or cubic interpolation
953 for(int i = 0; i < extra_samples; ++i)
954 newsample.smp[i + samplesize] = newsample.smp[i];
955
956 //yield new sample
957 newsample.size = samplesize;
958 newsample.basefreq = basefreq * basefreqadjust;
959 cb(nsample, newsample);
960 }
961
962 //Cleanup
963 delete (fft);
964 delete[] fftfreqs;
965 delete[] spectrum;
966 };
967
968 unsigned nthreads = std::min(max_threads,
969 std::thread::hardware_concurrency());
970 std::vector<std::thread> threads(nthreads);
971 for(unsigned i = 0; i < nthreads; ++i)
972 threads[i] = std::thread(thread_cb, nthreads, i);
973 for(unsigned i = 0; i < nthreads; ++i)
974 threads[i].join();
975
976 return samplemax;
977}
