39void GlowEffect::applyEffect (Image& image, Graphics& g, float scaleFactor, float alpha)
40{
41 Image temp (image.getFormat(), image.getWidth(), image.getHeight(), true);
42
43 ImageConvolutionKernel blurKernel (roundToInt (radius * scaleFactor * 2.0f));
44
45 blurKernel.createGaussianBlur (radius);
46 blurKernel.rescaleAllValues (radius);
47
48 blurKernel.applyToImage (temp, image, image.getBounds());
49
50 g.setColour (colour.withMultipliedAlpha (alpha));
51 g.drawImageAt (temp, offset.x, offset.y, true);
52
53 g.setOpacity (alpha);
54 g.drawImageAt (image, offset.x, offset.y, false);
55}
