35// Lightweight, move-only function object wrapper.
36class executor::function
37{
38public:
39 template <typename F, typename Alloc>
40 explicit function(F f, const Alloc& a)
41 {
42 // Allocate and construct an operation to wrap the function.
43 typedef detail::executor_function<F, Alloc> func_type;
44 typename func_type::ptr p = {
45 detail::addressof(a), func_type::ptr::allocate(a), 0 };
46 func_ = new (p.v) func_type(ASIO_MOVE_CAST(F)(f), a);
47 p.v = 0;
48 }
49
50 function(function&& other) ASIO_NOEXCEPT
51 : func_(other.func_)
52 {
53 other.func_ = 0;
54 }
55
56 ~function()
57 {
58 if (func_)
59 func_->destroy();
60 }
61
62 void operator()()
63 {
64 if (func_)
65 {
66 detail::executor_function_base* func = func_;
67 func_ = 0;
68 func->complete();
69 }
70 }
71
72private:
73 detail::executor_function_base* func_;
74};

78// Not so lightweight, copyable function object wrapper.
79class executor::function
80{
81public:
82 template <typename F, typename Alloc>
83 explicit function(const F& f, const Alloc&)
84 : impl_(new impl<F>(f))
85 {
86 }
87
88 void operator()()
89 {
90 impl_->invoke_(impl_.get());
91 }
92
93private:
94 // Base class for polymorphic function implementations.
95 struct impl_base
96 {
97 void (*invoke_)(impl_base*);
98 };
99
100 // Polymorphic function implementation.
101 template <typename F>
102 struct impl : impl_base
103 {
104 impl(const F& f)
105 : function_(f)
106 {
107 invoke_ = &function::invoke<F>;
108 }
109
110 F function_;
111 };
112
113 // Helper to invoke a function.
114 template <typename F>
115 static void invoke(impl_base* i)
116 {
117 static_cast<impl<F>*>(i)->function_();
118 }
119
120 detail::shared_ptr<impl_base> impl_;
121};

128 : public executor::impl_base
129{
130public:
131 typedef ASIO_REBIND_ALLOC(Allocator, impl) allocator_type;
132
133 static impl_base* create(const Executor& e, Allocator a = Allocator())
134 {
135 raw_mem mem(a);
136 impl* p = new (mem.ptr_) impl(e, a);
137 mem.ptr_ = 0;
138 return p;
139 }
140
141 impl(const Executor& e, const Allocator& a) ASIO_NOEXCEPT
142 : impl_base(false),
143 ref_count_(1),
144 executor_(e),
145 allocator_(a)
146 {
147 }
148
149 impl_base* clone() const ASIO_NOEXCEPT
150 {
151 ++ref_count_;
152 return const_cast<impl_base*>(static_cast<const impl_base*>(this));
153 }
154

156 {
157 if (--ref_count_ == 0)
