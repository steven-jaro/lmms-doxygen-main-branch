149 template <typename Handler, typename T>
150 class coro_async_result
151 {
152 public:
153 typedef coro_handler<Handler, T> completion_handler_type;
154 typedef T return_type;
155
156 explicit coro_async_result(completion_handler_type& h)
157 : handler_(h),
158 ca_(h.ca_),
159 ready_(2)
160 {
161 h.ready_ = &ready_;
162 out_ec_ = h.ec_;
163 if (!out_ec_) h.ec_ = &ec_;
164 h.value_ = &value_;
165 }
166
167 return_type get()
168 {
169 // Must not hold shared_ptr to coro while suspended.
170 handler_.coro_.reset();
171
172 if (--ready_ != 0)
173 ca_();
174 if (!out_ec_ && ec_) throw asio::system_error(ec_);
175 return ASIO_MOVE_CAST(return_type)(value_);
176 }
177
178 private:
179 completion_handler_type& handler_;
180 typename basic_yield_context<Handler>::caller_type& ca_;
181 atomic_count ready_;
182 asio::error_code* out_ec_;
183 asio::error_code ec_;
184 return_type value_;
185 };

324 template <typename Handler, typename Function>
325 struct coro_entry_point
326 {
327 void operator()(typename basic_yield_context<Handler>::caller_type& ca)
328 {
329 shared_ptr<spawn_data<Handler, Function> > data(data_);
330#if !defined(BOOST_COROUTINES_UNIDIRECT) && !defined(BOOST_COROUTINES_V2)
331 ca(); // Yield until coroutine pointer has been initialised.
332#endif // !defined(BOOST_COROUTINES_UNIDIRECT) && !defined(BOOST_COROUTINES_V2)
333 const basic_yield_context<Handler> yield(
334 data->coro_, ca, data->handler_);
335
336 (data->function_)(yield);
337 if (data->call_handler_)
338 (data->handler_)();
339 }
340
341 shared_ptr<spawn_data<Handler, Function> > data_;
342 };

345 struct spawn_helper
346 {
347 void operator()()
348 {
349 typedef typename basic_yield_context<Handler>::callee_type callee_type;
350 coro_entry_point<Handler, Function> entry_point = { data_ };
351 shared_ptr<callee_type> coro(new callee_type(entry_point, attributes_));
352 data_->coro_ = coro;

354 }

393void spawn(ASIO_MOVE_ARG(Handler) handler,
394 ASIO_MOVE_ARG(Function) function,
395 const boost::coroutines::attributes& attributes,
396 typename enable_if<!is_executor<typename decay<Handler>::type>::value &&
397 !is_convertible<Handler&, execution_context&>::value>::type*)
398{
