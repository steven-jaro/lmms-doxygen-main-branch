29template <typename Executor>
30class executor_work_guard
31{

37
40 explicit executor_work_guard(const executor_type& e) ASIO_NOEXCEPT
41 : executor_(e),
42 owns_(true)

48 executor_work_guard(const executor_work_guard& other) ASIO_NOEXCEPT
49 : executor_(other.executor_),

67
71 ~executor_work_guard()
