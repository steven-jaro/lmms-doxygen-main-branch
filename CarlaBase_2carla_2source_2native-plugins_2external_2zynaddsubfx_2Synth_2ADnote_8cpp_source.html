32ADnote::ADnote(ADnoteParameters *pars_, SynthParams &spars,
33 WatchManager *wm, const char *prefix)
34 :SynthNote(spars), pars(*pars_)
35{
36 memory.beginTransaction();
37 tmpwavel = memory.valloc<float>(synth.buffersize);
38 tmpwaver = memory.valloc<float>(synth.buffersize);
39 bypassl = memory.valloc<float>(synth.buffersize);
40 bypassr = memory.valloc<float>(synth.buffersize);
41
42 ADnoteParameters &pars = *pars_;
43 portamento = spars.portamento;
44 midinote = spars.note;
45 NoteEnabled = ON;
46 basefreq = spars.frequency;
47 velocity = spars.velocity;
48 stereo = pars.GlobalPar.PStereo;
49
50 NoteGlobalPar.Detune = getdetune(pars.GlobalPar.PDetuneType,
51 pars.GlobalPar.PCoarseDetune,
52 pars.GlobalPar.PDetune);
53 bandwidthDetuneMultiplier = pars.getBandwidthDetuneMultiplier();
54
55 if(pars.GlobalPar.PPanning == 0)
56 NoteGlobalPar.Panning = RND;
57 else
58 NoteGlobalPar.Panning = pars.GlobalPar.PPanning / 128.0f;
59
60
61 NoteGlobalPar.Fadein_adjustment =
62 pars.GlobalPar.Fadein_adjustment / (float)FADEIN_ADJUSTMENT_SCALE;
63 NoteGlobalPar.Fadein_adjustment *= NoteGlobalPar.Fadein_adjustment;
64 if(pars.GlobalPar.PPunchStrength != 0) {
65 NoteGlobalPar.Punch.Enabled = 1;
66 NoteGlobalPar.Punch.t = 1.0f; //start from 1.0f and to 0.0f
67 NoteGlobalPar.Punch.initialvalue =
68 ((powf(10, 1.5f * pars.GlobalPar.PPunchStrength / 127.0f) - 1.0f)
69 * VelF(velocity,
70 pars.GlobalPar.PPunchVelocitySensing));
71 float time =
72 powf(10, 3.0f * pars.GlobalPar.PPunchTime / 127.0f) / 10000.0f; //0.1f .. 100 ms
73 float stretch = powf(440.0f / spars.frequency,
74 pars.GlobalPar.PPunchStretch / 64.0f);
75 NoteGlobalPar.Punch.dt = 1.0f / (time * synth.samplerate_f * stretch);
76 }
77 else
78 NoteGlobalPar.Punch.Enabled = 0;
79
80 for(int nvoice = 0; nvoice < NUM_VOICES; ++nvoice)
81 setupVoice(nvoice);
82
83 max_unison = 1;
84 for(int nvoice = 0; nvoice < NUM_VOICES; ++nvoice)
85 if(unison_size[nvoice] > max_unison)
86 max_unison = unison_size[nvoice];
87
88
89 tmpwave_unison = memory.valloc<float*>(max_unison);
90 for(int k = 0; k < max_unison; ++k) {
91 tmpwave_unison[k] = memory.valloc<float>(synth.buffersize);
92 memset(tmpwave_unison[k], 0, synth.bufferbytes);
93 }
94
95 initparameters(wm, prefix);
96 memory.endTransaction();
97}

99void ADnote::setupVoice(int nvoice)
100{
101 auto &param = pars.VoicePar[nvoice];
102 auto &voice = NoteVoicePar[nvoice];
103
104
105 for (int i = 0; i < 14; i++)
106 pinking[nvoice][i] = 0.0;
107
108 param.OscilSmp->newrandseed(prng());
109 voice.OscilSmp = NULL;
110 voice.FMSmp = NULL;
111 voice.VoiceOut = NULL;
112
113 voice.FMVoice = -1;
114 unison_size[nvoice] = 1;
115
116 if(!pars.VoicePar[nvoice].Enabled) {
117 voice.Enabled = OFF;
118 return; //the voice is disabled
119 }
120
121 const int BendAdj = pars.VoicePar[nvoice].PBendAdjust - 64;
122 if (BendAdj % 24 == 0)
123 voice.BendAdjust = BendAdj / 24;
124 else
125 voice.BendAdjust = BendAdj / 24.0f;
126
127 const float offset_val = (param.POffsetHz - 64)/64.0f;
128 voice.OffsetHz = 15.0f*(offset_val * sqrtf(fabsf(offset_val)));
129
130 unison_stereo_spread[nvoice] =
131 pars.VoicePar[nvoice].Unison_stereo_spread / 127.0f;
132
133 int unison = setupVoiceUnison(nvoice);
134
135

137 oscfreqlo[nvoice] = memory.valloc<float>(unison);
138 oscfreqhiFM[nvoice] = memory.valloc<unsigned int>(unison);
139 oscfreqloFM[nvoice] = memory.valloc<float>(unison);
140 oscposhi[nvoice] = memory.valloc<int>(unison);
141 oscposlo[nvoice] = memory.valloc<float>(unison);
142 oscposhiFM[nvoice] = memory.valloc<unsigned int>(unison);
143 oscposloFM[nvoice] = memory.valloc<float>(unison);

146 voice.fixedfreq = pars.VoicePar[nvoice].Pfixedfreq;
147 voice.fixedfreqET = pars.VoicePar[nvoice].PfixedfreqET;
148
149 setupVoiceDetune(nvoice);

152 oscposhi[nvoice][k] = 0;
153 oscposlo[nvoice][k] = 0.0f;

156 }

165 vc = pars.VoicePar[nvoice].Pextoscil;

168 int oscposhi_start =
169 pars.VoicePar[vc].OscilSmp->get(NoteVoicePar[nvoice].OscilSmp,
170 getvoicebasefreq(nvoice),
171 pars.VoicePar[nvoice].Presonance);

173 // This code was planned for biasing the carrier in MOD_RING

175 // machines run this stuff too.
176 //
177 // //Find range of generated wave
178 // float min = NoteVoicePar[nvoice].OscilSmp[0];

192

234int ADnote::setupVoiceUnison(int nvoice)
235{
236 int unison = pars.VoicePar[nvoice].Unison_size;
237 if(unison < 1)
238 unison = 1;
239
240 bool is_pwm = pars.VoicePar[nvoice].PFMEnabled == PW_MOD;
241
242 if (pars.VoicePar[nvoice].Type != 0) {
243 // Since noise unison of greater than two is touch goofy...
244 if (unison > 2)
245 unison = 2;
246 } else if (is_pwm) {
247 /* Pulse width mod uses pairs of subvoices. */
248 unison *= 2;
249 // This many is likely to sound like noise anyhow.
250 if (unison > 64)
251 unison = 64;
252 }
253
254 //compute unison
255 unison_size[nvoice] = unison;
256
257 unison_base_freq_rap[nvoice] = memory.valloc<float>(unison);
258 unison_freq_rap[nvoice] = memory.valloc<float>(unison);
259 unison_invert_phase[nvoice] = memory.valloc<bool>(unison);
260 const float unison_spread =
261 pars.getUnisonFrequencySpreadCents(nvoice);
262 const float unison_real_spread = powf(2.0f, (unison_spread * 0.5f) / 1200.0f);
263 const float unison_vibratto_a =
264 pars.VoicePar[nvoice].Unison_vibratto / 127.0f; //0.0f .. 1.0f
265
266 const int true_unison = unison / (is_pwm ? 2 : 1);
267 switch(true_unison) {
268 case 1:
269 unison_base_freq_rap[nvoice][0] = 1.0f; //if the unison is not used, always make the only subvoice to have the default note
270 break;
271 case 2: { //unison for 2 subvoices
272 unison_base_freq_rap[nvoice][0] = 1.0f / unison_real_spread;
273 unison_base_freq_rap[nvoice][1] = unison_real_spread;
274 };
275 break;
276 default: { //unison for more than 2 subvoices
277 float unison_values[true_unison];
278 float min = -1e-6, max = 1e-6;
279 for(int k = 0; k < true_unison; ++k) {
280 float step = (k / (float) (true_unison - 1)) * 2.0f - 1.0f; //this makes the unison spread more uniform
281 float val = step + (RND * 2.0f - 1.0f) / (true_unison - 1);
282 unison_values[k] = val;
283 if (min > val) {
284 min = val;
285 }
286 if (max < val) {
287 max = val;
288 }
289 }
290 const float diff = max - min;
291 for(int k = 0; k < true_unison; ++k) {
292 unison_values[k] =
293 (unison_values[k] - (max + min) * 0.5f) / diff; //the lowest value will be -1 and the highest will be 1
294 unison_base_freq_rap[nvoice][k] =
295 powf(2.0f, (unison_spread * unison_values[k]) / 1200);
296 }
297 };
298 }
299 if (is_pwm)
300 for (int i = true_unison - 1; i >= 0; i--) {
301 unison_base_freq_rap[nvoice][2*i + 1] =
302 unison_base_freq_rap[nvoice][i];
303 unison_base_freq_rap[nvoice][2*i] =
304 unison_base_freq_rap[nvoice][i];
305 }
306
307 //unison vibrattos
308 if(unison > 2 || (!is_pwm && unison > 1))
309 for(int k = 0; k < unison; ++k) //reduce the frequency difference for larger vibrattos
310 unison_base_freq_rap[nvoice][k] = 1.0f
311 + (unison_base_freq_rap[
312 nvoice][k] - 1.0f)
313 * (1.0f - unison_vibratto_a);
314 unison_vibratto[nvoice].step = memory.valloc<float>(unison);
315 unison_vibratto[nvoice].position = memory.valloc<float>(unison);
316 unison_vibratto[nvoice].amplitude =
317 (unison_real_spread - 1.0f) * unison_vibratto_a;
318
319 const float increments_per_second = synth.samplerate_f / synth.buffersize_f;
320 const float vib_speed = pars.VoicePar[nvoice].Unison_vibratto_speed / 127.0f;
321 const float vibratto_base_period = 0.25f * powf(2.0f, (1.0f - vib_speed) * 4.0f);
322 for(int k = 0; k < unison; ++k) {
323 unison_vibratto[nvoice].position[k] = RND * 1.8f - 0.9f;
324 //make period to vary randomly from 50% to 200% vibratto base period
325 const float vibratto_period = vibratto_base_period
326 * powf(2.0f, RND * 2.0f - 1.0f);
327
328 const float m = (RND < 0.5f ? -1.0f : 1.0f) *
329 4.0f / (vibratto_period * increments_per_second);
330 unison_vibratto[nvoice].step[k] = m;
331
332 // Ugly, but the alternative is likely uglier.
333 if (is_pwm)
334 for (int i = 0; i < unison; i += 2) {
335 unison_vibratto[nvoice].step[i+1] =
336 unison_vibratto[nvoice].step[i];
337 unison_vibratto[nvoice].position[i+1] =
338 unison_vibratto[nvoice].position[i];
339 }
340 }
341
342 if(unison <= 2) { //no vibratto for a single voice
343 if (is_pwm) {
344 unison_vibratto[nvoice].step[1] = 0.0f;
345 unison_vibratto[nvoice].position[1] = 0.0f;
346 }
347 if (is_pwm || unison == 1) {
348 unison_vibratto[nvoice].step[0] = 0.0f;
349 unison_vibratto[nvoice].position[0] = 0.0f;
350 unison_vibratto[nvoice].amplitude = 0.0f;
351 }
352 }
353
354 //phase invert for unison
355 unison_invert_phase[nvoice][0] = false;
356 if(unison != 1) {
357 int inv = pars.VoicePar[nvoice].Unison_invert_phase;
358 switch(inv) {
359 case 0:
360 for(int k = 0; k < unison; ++k)
361 unison_invert_phase[nvoice][k] = false;
362 break;
363 case 1:
364 for(int k = 0; k < unison; ++k)
365 unison_invert_phase[nvoice][k] = (RND > 0.5f);
366 break;
367 default:
368 for(int k = 0; k < unison; ++k)
369 unison_invert_phase[nvoice][k] =
370 (k % inv == 0) ? true : false;
371 break;
372 }
373 }
374 return unison;
375}

416void ADnote::setupVoiceMod(int nvoice, bool first_run)
417{
418 auto &param = pars.VoicePar[nvoice];
419 auto &voice = NoteVoicePar[nvoice];
420 if (param.Type != 0)
421 voice.FMEnabled = NONE;
422 else
423 switch(param.PFMEnabled) {
424 case 1:
425 voice.FMEnabled = MORPH;
426 break;
427 case 2:
428 voice.FMEnabled = RING_MOD;
429 break;
430 case 3:
431 voice.FMEnabled = PHASE_MOD;
432 break;
433 case 4:
434 voice.FMEnabled = FREQ_MOD;
435 break;
436 case 5:
437 voice.FMEnabled = PW_MOD;
438 break;
439 default:
440 voice.FMEnabled = NONE;
441 }
442
443 voice.FMFreqFixed = param.PFMFixedFreq;
444
445 //Triggers when a user enables modulation on a running voice
446 if(!first_run && voice.FMEnabled != NONE && voice.FMSmp == NULL && voice.FMVoice < 0) {
447 param.FMSmp->newrandseed(prng());
448 voice.FMSmp = memory.valloc<float>(synth.oscilsize + OSCIL_SMP_EXTRA_SAMPLES);
449 memset(voice.FMSmp, 0, sizeof(float)*(synth.oscilsize + OSCIL_SMP_EXTRA_SAMPLES));
450 int vc = nvoice;
451 if(param.PextFMoscil != -1)
452 vc = param.PextFMoscil;
453
454 float tmp = 1.0f;
455 if((pars.VoicePar[vc].FMSmp->Padaptiveharmonics != 0)
456 || (voice.FMEnabled == MORPH)
457 || (voice.FMEnabled == RING_MOD))
458 tmp = getFMvoicebasefreq(nvoice);
459
460 if(!pars.GlobalPar.Hrandgrouping)
461 pars.VoicePar[vc].FMSmp->newrandseed(prng());
462
463 for(int k = 0; k < unison_size[nvoice]; ++k)
464 oscposhiFM[nvoice][k] = (oscposhi[nvoice][k]
465 + pars.VoicePar[vc].FMSmp->get(
466 voice.FMSmp, tmp))
467 % synth.oscilsize;
468
469 for(int i = 0; i < OSCIL_SMP_EXTRA_SAMPLES; ++i)
470 voice.FMSmp[synth.oscilsize + i] = voice.FMSmp[i];
471 int oscposhiFM_add =
472 (int)((param.PFMoscilphase
473 - 64.0f) / 128.0f * synth.oscilsize
474 + synth.oscilsize * 4);
475 for(int k = 0; k < unison_size[nvoice]; ++k) {
476 oscposhiFM[nvoice][k] += oscposhiFM_add;
477 oscposhiFM[nvoice][k] %= synth.oscilsize;
478 }
479 }
480
481
482 //Compute the Voice's modulator volume (incl. damping)
483 float fmvoldamp = powf(440.0f / getvoicebasefreq(nvoice),
484 param.PFMVolumeDamp / 64.0f - 1.0f);
485 const float fmvolume_ = param.PFMVolume / 127.0f;
486 switch(voice.FMEnabled) {
487 case PHASE_MOD:
488 case PW_MOD:
489 fmvoldamp = powf(440.0f / getvoicebasefreq(nvoice),
490 param.PFMVolumeDamp / 64.0f);
491 voice.FMVolume = (expf(fmvolume_ * FM_AMP_MULTIPLIER) - 1.0f)
492 * fmvoldamp * 4.0f;
493 break;
494 case FREQ_MOD:
495 voice.FMVolume = (expf(fmvolume_ * FM_AMP_MULTIPLIER) - 1.0f)
496 * fmvoldamp * 4.0f;
497 break;
498 default:
499 if(fmvoldamp > 1.0f)
500 fmvoldamp = 1.0f;
501 voice.FMVolume = fmvolume_ * fmvoldamp;
502 }
503
504 //Voice's modulator velocity sensing
505 NoteVoicePar[nvoice].FMVolume *=
506 VelF(velocity, pars.VoicePar[nvoice].PFMVelocityScaleFunction);
507}

992/*
993 * Computes the relative frequency of each unison voice and it's vibratto
994 * This must be called before setfreq* functions
995 */
996void ADnote::compute_unison_freq_rap(int nvoice) {
997 if(unison_size[nvoice] == 1) { //no unison
998 unison_freq_rap[nvoice][0] = 1.0f;
999 return;
1000 }
1001 float relbw = ctl.bandwidth.relbw * bandwidthDetuneMultiplier;
1002 for(int k = 0; k < unison_size[nvoice]; ++k) {
1003 float pos = unison_vibratto[nvoice].position[k];
1004 float step = unison_vibratto[nvoice].step[k];
1005 pos += step;
1006 if(pos <= -1.0f) {
1007 pos = -1.0f;
1008 step = -step;
1009 }
1010 if(pos >= 1.0f) {
1011 pos = 1.0f;
1012 step = -step;
1013 }
1014 float vibratto_val = (pos - 0.333333333f * pos * pos * pos) * 1.5f; //make the vibratto lfo smoother
1015 unison_freq_rap[nvoice][k] = 1.0f
1016 + ((unison_base_freq_rap[nvoice][k]
1017 - 1.0f) + vibratto_val
1018 * unison_vibratto[nvoice].amplitude)
1019 * relbw;
1020
1021 unison_vibratto[nvoice].position[k] = pos;
1022 step = unison_vibratto[nvoice].step[k] = step;
1023 }
1024}

1060 * Get Voice base frequency
1061 */
1062float ADnote::getvoicebasefreq(int nvoice) const
1063{
1064 float detune = NoteVoicePar[nvoice].Detune / 100.0f
1065 + NoteVoicePar[nvoice].FineDetune / 100.0f
1066 * ctl.bandwidth.relbw * bandwidthDetuneMultiplier
1067 + NoteGlobalPar.Detune / 100.0f;
1068
1069 if(NoteVoicePar[nvoice].fixedfreq == 0)
1070 return this->basefreq * powf(2, detune / 12.0f);
1071 else { //the fixed freq is enabled
1072 float fixedfreq = 440.0f;
1073 int fixedfreqET = NoteVoicePar[nvoice].fixedfreqET;
1074 if(fixedfreqET != 0) { //if the frequency varies according the keyboard note

1076 (midinote
1077 - 69.0f) / 12.0f
1078 * (powf(2.0f, (fixedfreqET - 1) / 63.0f) - 1.0f);
1079 if(fixedfreqET <= 64)
1080 fixedfreq *= powf(2.0f, tmp);
1081 else
1082 fixedfreq *= powf(3.0f, tmp);
1083 }
1084 return fixedfreq * powf(2.0f, detune / 12.0f);
1085 }

1097/*
1098 * Computes all the parameters for each tick
1099 */
1100void ADnote::computecurrentparameters()
1101{
1102 int nvoice;
1103 float voicefreq, voicepitch, FMfreq,
1104 FMrelativepitch, globalpitch;
1105 globalpitch = 0.01f * (NoteGlobalPar.FreqEnvelope->envout()
1106 + NoteGlobalPar.FreqLfo->lfoout()
1107 * ctl.modwheel.relmod);
1108 globaloldamplitude = globalnewamplitude;
1109 globalnewamplitude = NoteGlobalPar.Volume
1110 * NoteGlobalPar.AmpEnvelope->envout_dB()
1111 * NoteGlobalPar.AmpLfo->amplfoout();
1112
1113 NoteGlobalPar.Filter->update(ctl.filtercutoff.relfreq,
1114 ctl.filterq.relq);
1115
1116
1117 //compute the portamento, if it is used by this note
1118 float portamentofreqrap = 1.0f;
1119 if(portamento != 0) { //this voice use portamento
1120 portamentofreqrap = ctl.portamento.freqrap;
1121 if(ctl.portamento.used == 0) //the portamento has finished
1122 portamento = 0; //this note is no longer "portamented"
1123 }
1124
1125 //compute parameters for all voices
1126 for(nvoice = 0; nvoice < NUM_VOICES; ++nvoice) {
1127 if(NoteVoicePar[nvoice].Enabled != ON)
1128 continue;
1129 NoteVoicePar[nvoice].DelayTicks -= 1;
1130 if(NoteVoicePar[nvoice].DelayTicks > 0)
1131 continue;
1132
1133 compute_unison_freq_rap(nvoice);
1134
1135 /*******************/
1136 /* Voice Amplitude */
1137 /*******************/
1138 oldamplitude[nvoice] = newamplitude[nvoice];
1139 newamplitude[nvoice] = 1.0f;
1140
1141 if(NoteVoicePar[nvoice].AmpEnvelope)
1142 newamplitude[nvoice] *= NoteVoicePar[nvoice].AmpEnvelope->envout_dB();
1143
1144 if(NoteVoicePar[nvoice].AmpLfo)
1145 newamplitude[nvoice] *= NoteVoicePar[nvoice].AmpLfo->amplfoout();
1146
1147 /****************/
1148 /* Voice Filter */
1149 /****************/
1150 auto *voiceFilter = NoteVoicePar[nvoice].Filter;
1151 if(voiceFilter) {
1152 voiceFilter->update(ctl.filtercutoff.relfreq,
1153 ctl.filterq.relq);
1154 }
1155
1156 if(NoteVoicePar[nvoice].noisetype == 0) { //compute only if the voice isn't noise
1157 /*******************/
1158 /* Voice Frequency */
1159 /*******************/
1160 voicepitch = 0.0f;
1161 if(NoteVoicePar[nvoice].FreqLfo)
1162 voicepitch += NoteVoicePar[nvoice].FreqLfo->lfoout() / 100.0f
1163 * ctl.bandwidth.relbw;
1164
1165 if(NoteVoicePar[nvoice].FreqEnvelope)
1166 voicepitch += NoteVoicePar[nvoice].FreqEnvelope->envout()
1167 / 100.0f;
1168 voicefreq = getvoicebasefreq(nvoice)
1169 * powf(2, (voicepitch + globalpitch) / 12.0f); //Hz frequency
1170 voicefreq *=
1171 powf(ctl.pitchwheel.relfreq, NoteVoicePar[nvoice].BendAdjust); //change the frequency by the controller
1172 setfreq(nvoice, voicefreq * portamentofreqrap + NoteVoicePar[nvoice].OffsetHz);
1173
1174 /***************/
1175 /* Modulator */
1176 /***************/
1177
1178
1179 if(NoteVoicePar[nvoice].FMEnabled != NONE) {
1180 FMrelativepitch = NoteVoicePar[nvoice].FMDetune / 100.0f;
1181 if(NoteVoicePar[nvoice].FMFreqEnvelope)
1182 FMrelativepitch +=
1183 NoteVoicePar[nvoice].FMFreqEnvelope->envout() / 100;
1184 if (NoteVoicePar[nvoice].FMFreqFixed)
1185 FMfreq =
1186 powf(2.0f, FMrelativepitch
1187 / 12.0f) * 440.0f;
1188 else
1189 FMfreq =
1190 powf(2.0f, FMrelativepitch
1191 / 12.0f) * voicefreq * portamentofreqrap;
1192 setfreqFM(nvoice, FMfreq);
1193
1194 FMoldamplitude[nvoice] = FMnewamplitude[nvoice];
1195 FMnewamplitude[nvoice] = NoteVoicePar[nvoice].FMVolume
1196 * ctl.fmamp.relamp;
1197 if(NoteVoicePar[nvoice].FMAmpEnvelope)
1198 FMnewamplitude[nvoice] *=
1199 NoteVoicePar[nvoice].FMAmpEnvelope->envout_dB();
1200 }
1201 }
1202 }
1203}

1208 */
1209inline void ADnote::fadein(float *smps) const
