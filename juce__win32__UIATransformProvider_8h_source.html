44 RECT rect;

46
47 rect.left = roundToInt (x);
48 rect.top = roundToInt (y);
49
50 auto bounds = Rectangle<int>::leftTopRightBottom (rect.left, rect.top, rect.right, rect.bottom);
51
52 peer->setBounds (Desktop::getInstance().getDisplays().physicalToLogical (bounds),

54 }
55
56 return S_OK;
