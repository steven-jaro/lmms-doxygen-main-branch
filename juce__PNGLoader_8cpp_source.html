367 static bool readHeader (InputStream& in, png_structp pngReadStruct, png_infop pngInfoStruct, jmp_buf& errorJumpBuf,
368 png_uint_32& width, png_uint_32& height, int& bitDepth, int& colorType, int& interlaceType) noexcept
369 {
370 if (setjmp (errorJumpBuf) == 0)
371 {
372 // read the header..
373 png_set_read_fn (pngReadStruct, &in, readCallback);
374
375 png_read_info (pngReadStruct, pngInfoStruct);
376
377 png_get_IHDR (pngReadStruct, pngInfoStruct,
378 &width, &height,
379 &bitDepth, &colorType,
380 &interlaceType, nullptr, nullptr);
381
382 if (bitDepth == 16)
383 png_set_strip_16 (pngReadStruct);
384
385 if (colorType == PNG_COLOR_TYPE_PALETTE)
386 png_set_expand (pngReadStruct);
387
388 if (bitDepth < 8)
389 png_set_expand (pngReadStruct);
390
391 if (colorType == PNG_COLOR_TYPE_GRAY || colorType == PNG_COLOR_TYPE_GRAY_ALPHA)
392 png_set_gray_to_rgb (pngReadStruct);
393
394 return true;
395 }
396
397 return false;
398 }

400 static bool readImageData (png_structp pngReadStruct, png_infop pngInfoStruct, jmp_buf& errorJumpBuf, png_bytepp rows) noexcept
401 {
402 if (setjmp (errorJumpBuf) == 0)

407 png_set_add_alpha (pngReadStruct, 0xff, PNG_FILLER_AFTER);
408
409 png_read_image (pngReadStruct, rows);
410 png_read_end (pngReadStruct, pngInfoStruct);
411 return true;
412 }
413
414 return false;

419 static Image createImageFromData (bool hasAlphaChan, int width, int height, png_bytepp rows)
420 {
421 // now convert the data to a juce image format..
422 Image image (hasAlphaChan ? Image::ARGB : Image::RGB, width, height, hasAlphaChan);
423
424 image.getProperties()->set ("originalImageHadAlpha", image.hasAlphaChannel());
425 hasAlphaChan = image.hasAlphaChannel(); // (the native image creator may not give back what we expect)
426
427 const Image::BitmapData destData (image, Image::BitmapData::writeOnly);
428
429 for (int y = 0; y < (int) height; ++y)
430 {
431 const uint8* src = rows[y];
432 uint8* dest = destData.getLinePointer (y);
433
434 if (hasAlphaChan)
435 {
436 for (int i = (int) width; --i >= 0;)
437 {
438 ((PixelARGB*) dest)->setARGB (src[3], src[0], src[1], src[2]);
439 ((PixelARGB*) dest)->premultiply();
440 dest += destData.pixelStride;
441 src += 4;
442 }
443 }
444 else
445 {
446 for (int i = (int) width; --i >= 0;)
447 {
448 ((PixelRGB*) dest)->setARGB (0, src[0], src[1], src[2]);
449 dest += destData.pixelStride;
450 src += 4;
451 }
452 }
453 }
454
455 return image;
456 }

458 static Image readImage (InputStream& in, png_structp pngReadStruct, png_infop pngInfoStruct)
459 {
460 jmp_buf errorJumpBuf;
461 png_set_error_fn (pngReadStruct, &errorJumpBuf, errorCallback, warningCallback);
462
463 png_uint_32 width = 0, height = 0;
464 int bitDepth = 0, colorType = 0, interlaceType = 0;
465
466 if (readHeader (in, pngReadStruct, pngInfoStruct, errorJumpBuf,
467 width, height, bitDepth, colorType, interlaceType))
468 {
469 // Load the image into a temp buffer..
470 const size_t lineStride = width * 4;
471 HeapBlock<uint8> tempBuffer (height * lineStride);
472 HeapBlock<png_bytep> rows (height);
473
474 for (size_t y = 0; y < height; ++y)
475 rows[y] = (png_bytep) (tempBuffer + lineStride * y);
476
477 png_bytep trans_alpha = nullptr;
478 png_color_16p trans_color = nullptr;
479 int num_trans = 0;
480 png_get_tRNS (pngReadStruct, pngInfoStruct, &trans_alpha, &num_trans, &trans_color);
481
482 if (readImageData (pngReadStruct, pngInfoStruct, errorJumpBuf, rows))
483 return createImageFromData ((colorType & PNG_COLOR_MASK_ALPHA) != 0 || num_trans != 0,
484 (int) width, (int) height, rows);
485 }
486
487 return Image();
488 }

540bool PNGImageFormat::writeImageToStream (const Image& image, OutputStream& out)
541{
542 using namespace pnglibNamespace;
543 auto width = image.getWidth();
544 auto height = image.getHeight();
545
546 auto pngWriteStruct = png_create_write_struct (PNG_LIBPNG_VER_STRING, nullptr, nullptr, nullptr);
547
548 if (pngWriteStruct == nullptr)
549 return false;
550
551 auto pngInfoStruct = png_create_info_struct (pngWriteStruct);
552
553 if (pngInfoStruct == nullptr)
554 {
555 png_destroy_write_struct (&pngWriteStruct, nullptr);
556 return false;
557 }
558
559 png_set_write_fn (pngWriteStruct, &out, PNGHelpers::writeDataCallback, nullptr);
560
561 png_set_IHDR (pngWriteStruct, pngInfoStruct, (png_uint_32) width, (png_uint_32) height, 8,
562 image.hasAlphaChannel() ? PNG_COLOR_TYPE_RGB_ALPHA
563 : PNG_COLOR_TYPE_RGB,
564 PNG_INTERLACE_NONE,
565 PNG_COMPRESSION_TYPE_BASE,
566 PNG_FILTER_TYPE_BASE);
567
568 HeapBlock<uint8> rowData (width * 4);
569
570 png_color_8 sig_bit;
571 sig_bit.red = 8;
572 sig_bit.green = 8;
573 sig_bit.blue = 8;
574 sig_bit.gray = 0;
575 sig_bit.alpha = 8;
576 png_set_sBIT (pngWriteStruct, pngInfoStruct, &sig_bit);
577
578 png_write_info (pngWriteStruct, pngInfoStruct);
579
580 png_set_shift (pngWriteStruct, &sig_bit);
581 png_set_packing (pngWriteStruct);
582
583 const Image::BitmapData srcData (image, Image::BitmapData::readOnly);
584
585 for (int y = 0; y < height; ++y)
586 {
587 uint8* dst = rowData;
588 const uint8* src = srcData.getLinePointer (y);
589
590 if (image.hasAlphaChannel())
591 {
592 for (int i = width; --i >= 0;)
593 {
594 PixelARGB p (*(const PixelARGB*) src);
595 p.unpremultiply();
596
597 *dst++ = p.getRed();
598 *dst++ = p.getGreen();
599 *dst++ = p.getBlue();
600 *dst++ = p.getAlpha();
601 src += srcData.pixelStride;
602 }
603 }
604 else
605 {
606 for (int i = width; --i >= 0;)
607 {
608 *dst++ = ((const PixelRGB*) src)->getRed();
609 *dst++ = ((const PixelRGB*) src)->getGreen();
610 *dst++ = ((const PixelRGB*) src)->getBlue();
611 src += srcData.pixelStride;
612 }
613 }
614
615 png_bytep rowPtr = rowData;
616 png_write_rows (pngWriteStruct, &rowPtr, 1);
617 }
618
619 png_write_end (pngWriteStruct, pngInfoStruct);
620 png_destroy_write_struct (&pngWriteStruct, &pngInfoStruct);
621
622 return true;
623}
