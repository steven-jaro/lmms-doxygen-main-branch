113/*
114==============================================================================
115Name: process_sample
116Description:
117 From one input sample, generates two samples with a pi/2 phase shift.
118Input parameters:
119 - input: The input sample.
120Output parameters:
121 - out_0: Output sample, ahead.
122 - out_1: Output sample, late.
123Throws: Nothing
124==============================================================================
125*/
126
127template <int NC>
128hiir_FORCEINLINE void PhaseHalfPiSse <NC>::process_sample (float &out_0, float &out_1, float input) noexcept
129{
130 StageDataSse * filter_ptr = &_filter [_phase] [0];
131
132 const auto spl_in = _mm_load_ss (&input);
133 const auto prev = _mm_load_ss (&_prev);
134 const auto comb = _mm_unpacklo_ps (prev, spl_in);
135 const auto spl_mid = _mm_load_ps (filter_ptr [_nbr_stages]._mem);
136 constexpr auto shuf = (2 << 0) | (3 << 2) | (0 << 4) | (1 << 6);
137 auto y = _mm_shuffle_ps (spl_mid, comb, shuf);
138
139 auto mem = _mm_load_ps (filter_ptr [0]._mem);
140
141 StageProcSseV4 <_nbr_stages>::process_sample_neg (&filter_ptr [0], y, mem);
142
143 _mm_store_ps (filter_ptr [_nbr_stages]._mem, y);
144
145 out_1 = _mm_cvtss_f32 (_mm_shuffle_ps (y, y, 1));
146 out_0 = _mm_cvtss_f32 (y);
147
148 _prev = input;
149 _phase = 1 - _phase;
150}

154/*
155==============================================================================
156Name: process_block
157Description:
158 From a block of samples, generates two blocks of samples, with a pi/2
159 phase shift between these signals.
160 Input and output blocks may overlap, see assert() for details.
161Input parameters:
162 - in_ptr: Input array, containing nbr_spl samples.
163 - nbr_spl: Number of input samples to process, > 0
164Output parameters:
165 - out_0_ptr: Output sample array (ahead), capacity: nbr_spl samples.
166 - out_1_ptr: Output sample array (late), capacity: nbr_spl samples.
167Throws: Nothing
168==============================================================================
169*/
170
171template <int NC>
172void PhaseHalfPiSse <NC>::process_block (float out_0_ptr [], float out_1_ptr [], const float in_ptr [], long nbr_spl) noexcept
173{
174 assert (out_0_ptr != nullptr);
175 assert (out_1_ptr != nullptr);
176 assert (in_ptr != nullptr);
177 assert (out_0_ptr <= in_ptr || out_0_ptr >= in_ptr + nbr_spl);
178 assert (out_1_ptr <= in_ptr || out_1_ptr >= in_ptr + nbr_spl);
179 assert (out_0_ptr + nbr_spl <= out_1_ptr || out_1_ptr + nbr_spl <= out_0_ptr);
180 assert (nbr_spl > 0);
181
182 if (_phase != 0)
183 {
184 process_sample (out_0_ptr [0], out_1_ptr [0], in_ptr [0]);
185 ++ out_0_ptr;
186 ++ out_1_ptr;
187 ++ in_ptr;
188 -- nbr_spl;
189 }
190
191 if (nbr_spl > 0)
192 {
193 const long n4 =
194 process_block_quad (out_0_ptr, out_1_ptr, in_ptr, nbr_spl);
195
196 for (long pos = n4; pos < nbr_spl; ++pos)
197 {
198 process_sample (out_0_ptr [pos], out_1_ptr [pos], in_ptr [pos]);
199 }
200 }
201}

248template <int NC>
249void PhaseHalfPiSse <NC>::set_single_coef (int index, double coef) noexcept
250{
251 assert (index >= 0);
252 assert (index < _nbr_stages * _stage_width);
253
254 const int stage = (index / _stage_width) + 1;
255 const int pos = (index ^ _coef_shift) & (_stage_width - 1);
256 for (int phase = 0; phase < _nbr_phases; ++phase)
257 {
258 _filter [phase] [stage]._coef [pos] = DataType (coef);
259 }
260}

264template <int NC>
265long PhaseHalfPiSse <NC>::process_block_quad (float out_0_ptr [], float out_1_ptr [], const float in_ptr [], long nbr_spl) noexcept
266{
267 assert (_phase == 0);
268
269 constexpr auto shuf_y = (2 << 0) | (3 << 2);
270 constexpr auto shuf_1 = shuf_y | (0 << 4) | (1 << 6);
271 constexpr auto shuf_2 = shuf_y | (1 << 4) | (2 << 6);
272 constexpr auto shuf_3 = shuf_y | (2 << 4) | (3 << 6);
273
274 const long n4 = nbr_spl & ~(4-1);
275 auto prev = _mm_load_ss (&_prev);
276 auto y_2 = _mm_load_ps (_filter [0] [_nbr_stages]._mem);
277 auto y_3 = _mm_load_ps (_filter [1] [_nbr_stages]._mem);
278 for (long pos = 0; pos < n4; pos += 4)
279 {
280 const auto x = _mm_loadu_ps (in_ptr + pos);
281
282 const auto comb_0 = _mm_unpacklo_ps (prev, x);
283 auto y_0 = _mm_shuffle_ps (y_2, comb_0, shuf_1);
284 auto mem_0 = _mm_load_ps (_filter [0] [0]._mem);
285 StageProcSseV4 <_nbr_stages>::process_sample_neg (_filter [0].data (), y_0, mem_0);
286 _mm_store_ps (_filter [0] [_nbr_stages]._mem, y_0);
287
288 auto y_1 = _mm_shuffle_ps (y_3, x, shuf_1);
289 auto mem_1 = _mm_load_ps (_filter [1] [0]._mem);
290 StageProcSseV4 <_nbr_stages>::process_sample_neg (_filter [1].data (), y_1, mem_1);
291 _mm_store_ps (_filter [1] [_nbr_stages]._mem, y_1);
292
293 y_2 = _mm_shuffle_ps (y_0, x, shuf_2);
294 auto mem_2 = _mm_load_ps (_filter [0] [0]._mem);
295 StageProcSseV4 <_nbr_stages>::process_sample_neg (_filter [0].data (), y_2, mem_2);
296 _mm_store_ps (_filter [0] [_nbr_stages]._mem, y_2);
297
298 y_3 = _mm_shuffle_ps (y_1, x, shuf_3);
299 auto mem_3 = _mm_load_ps (_filter [1] [0]._mem);
300 StageProcSseV4 <_nbr_stages>::process_sample_neg (_filter [1].data (), y_3, mem_3);
301 _mm_store_ps (_filter [1] [_nbr_stages]._mem, y_3);
302
303 prev = _mm_shuffle_ps (x, x, 3);
304
305 const auto u_01 = _mm_unpacklo_ps (y_0, y_1); // o0, o1, e0, e1
306 const auto u_23 = _mm_unpacklo_ps (y_2, y_3); // o2, o3, e2, e3
307 const auto odd = _mm_movelh_ps (u_01, u_23);
308 const auto even = _mm_movehl_ps (u_23, u_01);
309 _mm_storeu_ps (out_0_ptr + pos, even);
310 _mm_storeu_ps (out_1_ptr + pos, odd);
311 }
312 _prev = _mm_cvtss_f32 (prev);
313
314 return n4;
315}
