24#ifndef rEffPar
25#define rEffPar(name, idx, ...) \
26 {STRINGIFY(name) "::i", rProp(parameter) rDefaultDepends(preset) \
27 DOC(__VA_ARGS__), NULL, rEffParCb(idx)}
28#define rEffParTF(name, idx, ...) \
29 {STRINGIFY(name) "::T:F", rProp(parameter) rDefaultDepends(preset) \
30 DOC(__VA_ARGS__), NULL, rEffParTFCb(idx)}
31#define rEffParCb(idx) \
32 [](const char *msg, rtosc::RtData &d) {\
33 rObject &obj = *(rObject*)d.obj; \
34 if(rtosc_narguments(msg)) \
35 obj.changepar(idx, rtosc_argument(msg, 0).i); \
36 else \
37 d.reply(d.loc, "i", obj.getpar(idx));}
38#define rEffParTFCb(idx) \
39 [](const char *msg, rtosc::RtData &d) {\
40 rObject &obj = *(rObject*)d.obj; \
41 if(rtosc_narguments(msg)) \
42 obj.changepar(idx, rtosc_argument(msg, 0).T*127); \
43 else \
44 d.reply(d.loc, obj.getpar(idx)?"T":"F");}
45#endif
46
47#define rEffParCommon(pname, rshort, rdoc, idx, ...) \
48{STRINGIFY(pname) "::i", rProp(parameter) rLinear(0,127) \
49 rShort(rshort) rDoc(rdoc), \
50 0, \
51 [](const char *msg, rtosc::RtData &d) \
52 { \
53 rObject& eff = *(rObject*)d.obj; \
54 if(!rtosc_narguments(msg)) \
55 d.reply(d.loc, "i", eff.getpar(idx)); \
56 else { \
57 eff.changepar(0, rtosc_argument(msg, 0).i); \
58 d.broadcast(d.loc, "i", eff.getpar(idx)); \
59 } \
60 }}
