29// A utility handler for passing to async functions that may call the
30// handler past the lifetime of the wrapped delegate object.
31// The need for this is particularly driven by boost::asio timer
32// objects, which explicitly document that they may be called without
33// an error code after they have been cancelled. This has led to
34// several crashes. This handler wrapper implements a useful idiom for
35// avoiding this problem.
36
37template <typename Delegate>
38struct SafeAsyncHandler
39{
40 SafeAsyncHandler(const std::shared_ptr<Delegate>& pDelegate)
41 : mpDelegate(pDelegate)
42 {
43 }
44
45 template <typename... T>
46 void operator()(T&&... t) const
47 {
48 std::shared_ptr<Delegate> pDelegate = mpDelegate.lock();
49 if (pDelegate)
50 {
51 (*pDelegate)(std::forward<T>(t)...);
52 }
53 }
54
55 std::weak_ptr<Delegate> mpDelegate;
56};
57
58// Factory function for easily wrapping a shared_ptr to a handler
59template <typename Delegate>
60SafeAsyncHandler<Delegate> makeAsyncSafe(const std::shared_ptr<Delegate>& pDelegate)
61{
62 return {pDelegate};
63}
