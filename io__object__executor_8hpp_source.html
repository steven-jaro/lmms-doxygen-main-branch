28// Wrap the (potentially polymorphic) executor so that we can bypass it when
29// dispatching on a target executor that has a native I/O implementation.
30template <typename Executor>
31class io_object_executor
32{
33public:
34 io_object_executor(const Executor& ex,
35 bool native_implementation) ASIO_NOEXCEPT
36 : executor_(ex),
37 has_native_impl_(native_implementation)
38 {
39 }

44 {
45 }

78 void on_work_started() const ASIO_NOEXCEPT
79 {
80 if (is_same<Executor, io_context::executor_type>::value
81 || has_native_impl_)
82 {
83 // When using a native implementation, work is already counted by the

85 }

92 void on_work_finished() const ASIO_NOEXCEPT

94 if (is_same<Executor, io_context::executor_type>::value
95 || has_native_impl_)

97 // When using a native implementation, work is already counted by the

99 }

103 }
104 }

106 template <typename F, typename A>
107 void dispatch(ASIO_MOVE_ARG(F) f, const A& a) const
108 {
109 if (is_same<Executor, io_context::executor_type>::value
110 || has_native_impl_)
111 {
112 // When using a native implementation, I/O completion handlers are
113 // already dispatched according to the execution context's executor's
114 // rules. We can call the function directly.
115 typename decay<F>::type function(ASIO_MOVE_CAST(F)(f));
116 asio_handler_invoke_helpers::invoke(function, function);
117 }
118 else
119 {
120 executor_.dispatch(ASIO_MOVE_CAST(F)(f), a);
121 }
122 }
123
124 template <typename F, typename A>
125 void post(ASIO_MOVE_ARG(F) f, const A& a) const
126 {
127 executor_.post(ASIO_MOVE_CAST(F)(f), a);
128 }
129
130 template <typename F, typename A>
131 void defer(ASIO_MOVE_ARG(F) f, const A& a) const
132 {
133 executor_.defer(ASIO_MOVE_CAST(F)(f), a);
134 }
135
136 friend bool operator==(const io_object_executor& a,
137 const io_object_executor& b) ASIO_NOEXCEPT
138 {
139 return a.executor_ == b.executor_
140 && a.has_native_impl_ == b.has_native_impl_;
141 }
142
143 friend bool operator!=(const io_object_executor& a,
144 const io_object_executor& b) ASIO_NOEXCEPT
145 {
146 return a.executor_ != b.executor_
147 || a.has_native_impl_ != b.has_native_impl_;
148 }
149
150private:
151 Executor executor_;
152 const bool has_native_impl_;
153};
