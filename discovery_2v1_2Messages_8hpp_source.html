70 template <typename It>
71 static std::pair<MessageHeader, It> fromNetworkByteStream(It begin, const It end)
72 {
73 using namespace std;
74
75 MessageHeader header;
76 tie(header.messageType, begin) =
77 Deserialize<decltype(header.messageType)>::fromNetworkByteStream(begin, end);
78 tie(header.ttl, begin) =
79 Deserialize<decltype(header.ttl)>::fromNetworkByteStream(begin, end);
80 tie(header.groupId, begin) =
81 Deserialize<decltype(header.groupId)>::fromNetworkByteStream(begin, end);
82 tie(header.ident, begin) =
83 Deserialize<decltype(header.ident)>::fromNetworkByteStream(begin, end);
84
85 return make_pair(move(header), move(begin));
86 }

92// Types that are only used in the sending/parsing of messages, not
93// publicly exposed.
94using ProtocolHeader = std::array<char, 8>;
95const ProtocolHeader kProtocolHeader = {{'_', 'a', 's', 'd', 'p', '_', 'v', 1}};
96
97// Must have at least kMaxMessageSize bytes available in the output stream
98template <typename NodeId, typename Payload, typename It>
99It encodeMessage(NodeId from,
100 const uint8_t ttl,
101 const MessageType messageType,
102 const Payload& payload,
103 It out)
104{
105 using namespace std;
106 const MessageHeader<NodeId> header = {messageType, ttl, 0, std::move(from)};
107 const auto messageSize =
108 kProtocolHeader.size() + sizeInByteStream(header) + sizeInByteStream(payload);
109
110 if (messageSize < kMaxMessageSize)
111 {
112 return toNetworkByteStream(
113 payload, toNetworkByteStream(
114 header, copy(begin(kProtocolHeader), end(kProtocolHeader), move(out))));
115 }
116 else
117 {
118 throw range_error("Exceeded maximum message size");
119 }
120}
121
122} // namespace detail

143template <typename NodeId, typename It>
144std::pair<MessageHeader<NodeId>, It> parseMessageHeader(It bytesBegin, const It bytesEnd)
145{
146 using namespace std;
147 using ItDiff = typename iterator_traits<It>::difference_type;
148
149 MessageHeader<NodeId> header = {};
150 const auto protocolHeaderSize = discovery::sizeInByteStream(detail::kProtocolHeader);
151 const auto minMessageSize =
152 static_cast<ItDiff>(protocolHeaderSize + sizeInByteStream(header));
153
154 // If there are enough bytes in the stream to make a header and if
155 // the first bytes in the stream are the protocol header, then
156 // proceed to parse the stream.
157 if (distance(bytesBegin, bytesEnd) >= minMessageSize
158 && equal(begin(detail::kProtocolHeader), end(detail::kProtocolHeader), bytesBegin))
159 {
160 tie(header, bytesBegin) = MessageHeader<NodeId>::fromNetworkByteStream(
161 bytesBegin + protocolHeaderSize, bytesEnd);
162 }
163 return make_pair(move(header), move(bytesBegin));
164}
