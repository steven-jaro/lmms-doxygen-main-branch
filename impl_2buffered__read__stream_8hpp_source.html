30template <typename Stream>
31std::size_t buffered_read_stream<Stream>::fill()
32{
33 detail::buffer_resize_guard<detail::buffered_stream_storage>
34 resize_guard(storage_);
35 std::size_t previous_size = storage_.size();
36 storage_.resize(storage_.capacity());
37 storage_.resize(previous_size + next_layer_.read_some(buffer(
38 storage_.data() + previous_size,
39 storage_.size() - previous_size)));
40 resize_guard.commit();
41 return storage_.size() - previous_size;
42}

207template <typename Stream>
208template <typename MutableBufferSequence>
209std::size_t buffered_read_stream<Stream>::read_some(
210 const MutableBufferSequence& buffers)
211{
212 using asio::buffer_size;
213 if (buffer_size(buffers) == 0)
214 return 0;
215
216 if (storage_.empty())
217 this->fill();
218
219 return this->copy(buffers);
220}

222template <typename Stream>
223template <typename MutableBufferSequence>
224std::size_t buffered_read_stream<Stream>::read_some(
225 const MutableBufferSequence& buffers, asio::error_code& ec)
226{
227 ec = asio::error_code();
228
229 using asio::buffer_size;
230 if (buffer_size(buffers) == 0)
231 return 0;
232
233 if (storage_.empty() && !this->fill(ec))
234 return 0;
235
236 return this->copy(buffers);
237}

260
261 buffered_read_some_handler(buffered_read_some_handler&& other)

265 {
266 }
267#endif // defined(ASIO_HAS_MOVE)
268
269 void operator()(const asio::error_code& ec, std::size_t)
270 {
271 if (ec || storage_.empty())
272 {
273 const std::size_t length = 0;
274 handler_(ec, length);
275 }
276 else
277 {
278 const std::size_t bytes_copied = asio::buffer_copy(
279 buffers_, storage_.data(), storage_.size());
280 storage_.consume(bytes_copied);
281 handler_(ec, bytes_copied);
282 }
283 }

287 MutableBufferSequence buffers_;
288 ReadHandler handler_;

338 struct initiate_async_buffered_read_some
339 {
340 template <typename ReadHandler, typename Stream,
341 typename MutableBufferSequence>
342 void operator()(ASIO_MOVE_ARG(ReadHandler) handler,
343 buffered_stream_storage* storage, Stream* next_layer,
344 const MutableBufferSequence& buffers) const
345 {
346 // If you get an error on the following line it means that your handler
347 // does not meet the documented type requirements for a ReadHandler.
348 ASIO_READ_HANDLER_CHECK(ReadHandler, handler) type_check;
349
350 using asio::buffer_size;
351 non_const_lvalue<ReadHandler> handler2(handler);
352 if (buffer_size(buffers) == 0 || !storage->empty())
353 {
354 next_layer->async_read_some(ASIO_MUTABLE_BUFFER(0, 0),
355 buffered_read_some_handler<MutableBufferSequence,
356 typename decay<ReadHandler>::type>(
357 *storage, buffers, handler2.value));
358 }
359 else
360 {
361 initiate_async_buffered_fill()(
362 buffered_read_some_handler<MutableBufferSequence,
363 typename decay<ReadHandler>::type>(
364 *storage, buffers, handler2.value),
365 storage, next_layer);
366 }
367 }
