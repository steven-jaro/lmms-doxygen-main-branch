55Timer* MultiTimer::getCallback (int timerID) const noexcept
56{
57 for (int i = timers.size(); --i >= 0;)
58 {
59 MultiTimerCallback* const t = static_cast<MultiTimerCallback*> (timers.getUnchecked(i));
