317void FilterParams::defaults()
318{
319 Ptype = Dtype;
320 Pfreq = Dfreq;
321 Pq = Dq;
322
323 Pstages = 0;
324 basefreq = (Pfreq / 64.0f - 1.0f) * 5.0f;
325 basefreq = powf(2.0f, basefreq + 9.96578428f);
326 baseq = expf(powf((float) Pq / 127.0f, 2) * logf(1000.0f)) - 0.9f;
327
328 gain = 0.0f;
329 freqtracking = 0.0f;
330
331 Pcategory = 0;
332
333 Pnumformants = 3;
334 Pformantslowness = 64;
335 for(int j = 0; j < FF_MAX_VOWELS; ++j)
336 defaults(j);
337
338
339 Psequencesize = 3;
340 for(int i = 0; i < FF_MAX_SEQUENCE; ++i)
341 Psequence[i].nvowel = i % FF_MAX_VOWELS;
342
343 Psequencestretch = 40;
344 Psequencereversed = 0;
345 Pcenterfreq = 64; //1 kHz
346 Poctavesfreq = 64;
347 Pvowelclearness = 64;
348}

403/*
404 * Parameter control
405 */
406float FilterParams::getfreq() const
407{
408 return log2(basefreq) - log2f(1000.0f);
409}
410
411float FilterParams::getq() const
412{
413 return baseq;
414}
415float FilterParams::getfreqtracking(float notefreq) const
416{
417 return log2f(notefreq / 440.0f) * (freqtracking / 100.0);
418}
419
420float FilterParams::getgain() const
421{
422 return gain;
423}
424
425/*
426 * Get the center frequency of the formant's graph
427 */
428float FilterParams::getcenterfreq() const
429{
430 return 10000.0f * powf(10, -(1.0f - Pcenterfreq / 127.0f) * 2.0f);
431}
432
433/*
434 * Get the number of octave that the formant functions applies to
435 */
436float FilterParams::getoctavesfreq() const
437{
438 return 0.25f + 10.0f * Poctavesfreq / 127.0f;
439}
440
441/*
442 * Get the frequency from x, where x is [0..1]
443 */
444float FilterParams::getfreqx(float x) const
445{
446 if(x > 1.0f)
447 x = 1.0f;
448 float octf = powf(2.0f, getoctavesfreq());
449 return getcenterfreq() / sqrt(octf) * powf(octf, x);
450}
451
452/*
453 * Get the x coordinate from frequency (used by the UI)
454 */
455float FilterParams::getfreqpos(float freq) const
456{
457 return (logf(freq) - logf(getfreqx(0.0f))) / logf(2.0f) / getoctavesfreq();
458}
459
460/*
461 * Transforms a parameter to the real value
462 */
463float FilterParams::getformantfreq(unsigned char freq) const
464{
465 return getfreqx(freq / 127.0f);
466}
467
468float FilterParams::getformantamp(unsigned char amp) const
469{
470 return powf(0.1f, (1.0f - amp / 127.0f) * 4.0f);
471}
472
473float FilterParams::getformantq(unsigned char q) const
474{
475 //temp
476 return powf(25.0f, (q - 32.0f) / 64.0f);
477}

493void FilterParams::add2XML(XMLwrapper& xml)
494{
495 //filter parameters
496 xml.addpar("category", Pcategory);
497 xml.addpar("type", Ptype);
498 xml.addparreal("basefreq", basefreq);
499 xml.addparreal("baseq", baseq);
500 xml.addpar("stages", Pstages);
501 xml.addparreal("freq_tracking", freqtracking);
502 xml.addparreal("gain", gain);
503
504 //formant filter parameters
505 if((Pcategory == 1) || (!xml.minimal)) {
506 xml.beginbranch("FORMANT_FILTER");
507 xml.addpar("num_formants", Pnumformants);
508 xml.addpar("formant_slowness", Pformantslowness);
509 xml.addpar("vowel_clearness", Pvowelclearness);
510 xml.addpar("center_freq", Pcenterfreq);
511 xml.addpar("octaves_freq", Poctavesfreq);
512 for(int nvowel = 0; nvowel < FF_MAX_VOWELS; ++nvowel) {
513 xml.beginbranch("VOWEL", nvowel);
514 add2XMLsection(xml, nvowel);
515 xml.endbranch();
516 }
517 xml.addpar("sequence_size", Psequencesize);
518 xml.addpar("sequence_stretch", Psequencestretch);
519 xml.addparbool("sequence_reversed", Psequencereversed);
520 for(int nseq = 0; nseq < FF_MAX_SEQUENCE; ++nseq) {
521 xml.beginbranch("SEQUENCE_POS", nseq);
522 xml.addpar("vowel_id", Psequence[nseq].nvowel);
523 xml.endbranch();
524 }
525 xml.endbranch();
526 }
527}

550void FilterParams::getfromXML(XMLwrapper& xml)
551{
552 const bool upgrade_3_0_2 = (xml.fileversion() < version_type(3,0,2)) && (xml.getparreal("basefreq", -1) < 0);
553
554 //filter parameters
555 Pcategory = xml.getpar127("category", Pcategory);
556 Ptype = xml.getpar127("type", Ptype);
557 Pstages = xml.getpar127("stages", Pstages);
558 if(upgrade_3_0_2) {
559 int Pfreq = xml.getpar127("freq", 0);
560 basefreq = (Pfreq / 64.0f - 1.0f) * 5.0f;
561 basefreq = powf(2.0f, basefreq + 9.96578428f);
562 int Pq = xml.getpar127("q", 0);
563 baseq = expf(powf((float) Pq / 127.0f, 2) * logf(1000.0f)) - 0.9f;
564 int Pgain = xml.getpar127("gain", 0);
565 gain = (Pgain / 64.0f - 1.0f) * 30.0f; //-30..30dB
566 int Pfreqtracking = xml.getpar127("freq_track", 0);
567 freqtracking = 100 * (Pfreqtracking - 64.0f) / (64.0f);
568 } else {
569 basefreq = xml.getparreal("basefreq", 1000);
570 baseq = xml.getparreal("baseq", 10);
571 gain = xml.getparreal("gain", 0);
572 freqtracking = xml.getparreal("freq_tracking", 0);
573 }
574
575 //formant filter parameters
576 if(xml.enterbranch("FORMANT_FILTER")) {
577 Pnumformants = xml.getpar127("num_formants", Pnumformants);
578 Pformantslowness = xml.getpar127("formant_slowness", Pformantslowness);
579 Pvowelclearness = xml.getpar127("vowel_clearness", Pvowelclearness);
580 Pcenterfreq = xml.getpar127("center_freq", Pcenterfreq);
581 Poctavesfreq = xml.getpar127("octaves_freq", Poctavesfreq);
582
583 for(int nvowel = 0; nvowel < FF_MAX_VOWELS; ++nvowel) {
584 if(xml.enterbranch("VOWEL", nvowel) == 0)
585 continue;
586 getfromXMLsection(xml, nvowel);
587 xml.exitbranch();
588 }
589 Psequencesize = xml.getpar127("sequence_size", Psequencesize);
590 Psequencestretch = xml.getpar127("sequence_stretch", Psequencestretch);
591 Psequencereversed = xml.getparbool("sequence_reversed",
592 Psequencereversed);
593 for(int nseq = 0; nseq < FF_MAX_SEQUENCE; ++nseq) {
594 if(xml.enterbranch("SEQUENCE_POS", nseq) == 0)
595 continue;
596 Psequence[nseq].nvowel = xml.getpar("vowel_id",
597 Psequence[nseq].nvowel,
598 0,
599 FF_MAX_VOWELS - 1);
600 xml.exitbranch();
601 }
602 xml.exitbranch();
603 }
