297static inline
298uint32_t d_nextPowerOf2(uint32_t size) noexcept
299{
300 DISTRHO_SAFE_ASSERT_RETURN(size > 0, 0);
301
302 // http://graphics.stanford.edu/~seander/bithacks.html#RoundUpPowerOf2
303 --size;
304 size |= size >> 1;
305 size |= size >> 2;
306 size |= size >> 4;
307 size |= size >> 8;
308 size |= size >> 16;
309 return ++size;
310}
