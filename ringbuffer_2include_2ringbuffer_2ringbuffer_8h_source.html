162 std::size_t write(const T *src, size_t cnt) {

167 template<class Func>
168 std::size_t write_func(Func& f, size_t cnt)
169 {
170 std::size_t w, to_write; // w: write ptr, to_write: actually writable
171 std::size_t n1, n2; // n1 + n2 = to_write (1st and 2nd halve)
172 init_variables_for_write(cnt, w, to_write, n1, n2);
173
174 //std::copy_n(src, n1, &(buf[w]));
175 f(0, n1, buf + w);
176 w = (w + n1) & size_mask;
177 // update so readers are already informed:
178 w_ptr.store(w);
179
180 if (n2) {
181 //std::copy_n(src + n1, n2, &(buf[w]));
182 f(n1, n2, buf + w);
183 w = (w + n2) & size_mask;
184 w_ptr.store(w);
185 }
186
187 return to_write;
188 }
