43TrackGrip::TrackGrip(Track* track, QWidget* parent) :
44 QWidget(parent),
45 m_track(track)
46{
47 if (!s_grabbedPixmap)
48 {
49 s_grabbedPixmap = new QPixmap(embed::getIconPixmap("track_op_grip_c"));
50 }
51
52 if (!s_releasedPixmap)
53 {
54 s_releasedPixmap = new QPixmap(embed::getIconPixmap("track_op_grip"));
55 }
56
57 setSizePolicy(QSizePolicy::Minimum, QSizePolicy::Expanding);
58
59 setCursor(Qt::OpenHandCursor);
60
61 setFixedWidth(std::max(s_grabbedPixmap->width(), s_releasedPixmap->width()) + 2 * c_margin);
62}

88void TrackGrip::paintEvent(QPaintEvent*)
89{
90 QPainter p(this);
91
92 // Check if the color of the track should be used for the background
93 const auto color = m_track->color();
94 const auto muted = m_track->getMutedModel()->value();
95
96 if (color.has_value() && !muted)
97 {
98 p.fillRect(rect(), color.value());
99 }
100
101 // Paint the pixmap
102 auto r = rect().marginsRemoved(QMargins(c_margin, c_margin, c_margin, c_margin));
103 p.drawTiledPixmap(r, m_isGrabbed ? *s_grabbedPixmap : *s_releasedPixmap);
104}
