104/*
105==============================================================================
106Name: process_sample
107Description:
108 Downsamples (x2) one pair of samples, to generate one output sample.
109Input parameters:
110 - in_ptr: pointer on the two samples to decimate
111Returns: Samplerate-reduced sample.
112Throws: Nothing
113==============================================================================
114*/
115
116template <int NC>
117float Downsampler2xNeonOld <NC>::process_sample (const float in_ptr [2]) noexcept
118{
119 assert (in_ptr != nullptr);
120
121 // Combines two input samples and two mid-processing data
122 const auto spl_in = vreinterpret_f32_u8 (
123 vld1_u8 (reinterpret_cast <const uint8_t *> (in_ptr))
124 );
125 const auto spl_m = vget_high_f32 (load4a (_filter [_nbr_stages]._mem));
126 auto y = vcombine_f32 (spl_m, spl_in);
127 auto mem = load4a (_filter [0]._mem);
128
129 // Processes each stage
130 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter.data (), y, mem);
131
132 // Averages both paths and outputs the result
133 const float even = vgetq_lane_f32 (y, 1);
134 const float odd = vgetq_lane_f32 (y, 0);
135 const float out = (even + odd) * 0.5f;
136
137 return out;
138}

142/*
143==============================================================================
144Name: process_block
145Description:
146 Downsamples (x2) a block of samples.
147 Input and output blocks may overlap, see assert() for details.
148Input parameters:
149 - in_ptr: Input array, containing nbr_spl * 2 samples.
150 - nbr_spl: Number of samples to output, > 0
151Output parameters:
152 - out_ptr: Array for the output samples, capacity: nbr_spl samples.
153Throws: Nothing
154==============================================================================
155*/
156
157template <int NC>
158void Downsampler2xNeonOld <NC>::process_block (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
159{
160 assert (in_ptr != nullptr);
161 assert (out_ptr != nullptr);
162 assert (out_ptr <= in_ptr || out_ptr >= in_ptr + nbr_spl * 2);
163 assert (nbr_spl > 0);
164
165 const long n4 = process_block_quad (
166 out_ptr, nullptr, in_ptr, nbr_spl, store_low, bypass
167 );
168
169 for (long pos = n4; pos < nbr_spl; ++pos)
170 {
171 out_ptr [pos] = process_sample (in_ptr + pos * 2);
172 }
173}

177/*
178==============================================================================
179Name: process_sample_split
180Description:
181 Split (spectrum-wise) in half a pair of samples. The lower part of the
182 spectrum is a classic downsampling, equivalent to the output of
183 process_sample().
184 The higher part is the complementary signal: original filter response
185 is flipped from left to right, becoming a high-pass filter with the same
186 cutoff frequency. This signal is then critically sampled (decimation by 2),
187 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
188Input parameters:
189 - in_ptr: pointer on the pair of input samples
190Output parameters:
191 - low: output sample, lower part of the spectrum (downsampling)
192 - high: output sample, higher part of the spectrum.
193Throws: Nothing
194==============================================================================
195*/
196
197template <int NC>
198void Downsampler2xNeonOld <NC>::process_sample_split (float &low, float &high, const float in_ptr [2]) noexcept
199{
200 assert (in_ptr != nullptr);
201
202 // Combines two input samples and two mid-processing data
203 const auto spl_in = vreinterpret_f32_u8 (
204 vld1_u8 (reinterpret_cast <const uint8_t *> (in_ptr))
205 );
206 const auto spl_m = vget_high_f32 (load4a (_filter [_nbr_stages]._mem));
207 auto y = vcombine_f32 (spl_m, spl_in);
208 auto mem = load4a (_filter [0]._mem);
209
210 // Processes each stage
211 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter.data (), y, mem);
212
213 // Outputs the result
214 const auto both = vget_low_f32 (y) * vdup_n_f32 (0.5f);
215 const float even = vget_lane_f32 (both, 1);
216 const float odd = vget_lane_f32 (both, 0);
217 low = even + odd;
218 high = even - odd;
219}

223/*
224==============================================================================
225Name: process_block_split
226Description:
227 Split (spectrum-wise) in half a block of samples. The lower part of the
228 spectrum is a classic downsampling, equivalent to the output of
229 process_block().
230 The higher part is the complementary signal: original filter response
231 is flipped from left to right, becoming a high-pass filter with the same
232 cutoff frequency. This signal is then critically sampled (decimation by 2),
233 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
234 Input and output blocks may overlap, see assert() for details.
235Input parameters:
236 - in_ptr: Input array, containing nbr_spl * 2 samples.
237 - nbr_spl: Number of samples for each output, > 0
238Output parameters:
239 - out_l_ptr: Array for the output samples, lower part of the spectrum
240 (downsampling). Capacity: nbr_spl samples.
241 - out_h_ptr: Array for the output samples, higher part of the spectrum.
242 Capacity: nbr_spl samples.
243Throws: Nothing
244==============================================================================
245*/
246
247template <int NC>
248void Downsampler2xNeonOld <NC>::process_block_split (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl) noexcept
249{
250 assert (in_ptr != nullptr);
251 assert (out_l_ptr != nullptr);
252 assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl * 2);
253 assert (out_h_ptr != nullptr);
254 assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl * 2);
255 assert (out_h_ptr != out_l_ptr);
256 assert (nbr_spl > 0);
257
258 const long n4 = process_block_quad (
259 out_l_ptr, out_h_ptr, in_ptr, nbr_spl, store_low, store_high
260 );
261
262 for (long pos = n4; pos < nbr_spl; ++pos)
263 {
264 process_sample_split (out_l_ptr [pos], out_h_ptr [pos], in_ptr + pos * 2);
265 }
266}

308template <int NC>
309void Downsampler2xNeonOld <NC>::set_single_coef (int index, double coef) noexcept
310{
311 assert (index >= 0);
312 assert (index < _nbr_stages * _stage_width);
313
314 const int stage = (index / _stage_width) + 1;
315 const int pos = (index ^ _coef_shift) & (_stage_width - 1);
316 _filter [stage]._coef [pos] = DataType (coef);
317}

321template <int NC>
322template <typename FL, typename FH>
323long Downsampler2xNeonOld <NC>::process_block_quad (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl, FL fnc_l, FH fnc_h) noexcept
324{
325 assert (_phase == 0);
326
327 const auto half = vdupq_n_f32 (0.5f);
328
329 const long n4 = nbr_spl & ~(4-1);
330 auto y_3 = load4a (_filter [_nbr_stages]._mem);
331 for (long pos = 0; pos < n4; pos += 4)
332 {
333 const auto x_0 = load4u (in_ptr + pos * 2 );
334 const auto x_2 = load4u (in_ptr + pos * 2 + 4);
335
336 auto y_0 = vcombine_f32 (vget_high_f32 (y_3), vget_low_f32 (x_0));
337 auto mem_0 = load4a (_filter [0]._mem);
338 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_0, mem_0);
339 storea (_filter [_nbr_stages]._mem, y_0);
340
341 auto y_1 = vcombine_f32 (vget_high_f32 (y_0), vget_high_f32 (x_0));
342 auto mem_1 = load4a (_filter [0]._mem);
343 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_1, mem_1);
344 storea (_filter [_nbr_stages]._mem, y_1);
345
346 auto y_2 = vcombine_f32 (vget_high_f32 (y_1), vget_low_f32 (x_2));
347 auto mem_2 = load4a (_filter [0]._mem);
348 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_2, mem_2);
349 storea (_filter [_nbr_stages]._mem, y_2);
350
351 y_3 = vcombine_f32 (vget_high_f32 (y_2), vget_high_f32 (x_2));
352 auto mem_3 = load4a (_filter [0]._mem);
353 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_3, mem_3);
354 storea (_filter [_nbr_stages]._mem, y_3);
355
356 const auto u_02 = vcombine_f32 (vget_low_f32 (y_0), vget_low_f32 (y_2));
357 const auto u_13 = vcombine_f32 (vget_low_f32 (y_1), vget_low_f32 (y_3));
358 const auto both = vtrnq_f32 (u_02, u_13);
359 const auto odd = both.val [0];
360 const auto even = both.val [1];
361 fnc_l (out_l_ptr + pos, even, odd, half);
362 fnc_h (out_h_ptr + pos, even, odd, half);
363 }
364
365 return n4;
366}
