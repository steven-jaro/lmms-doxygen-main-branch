106void StepRecorderWidget::paintEvent(QPaintEvent * pe)
107{
108 QPainter painter(this);
109
110 updateBoundaries();
111
112 move(0, 0);
113
114 //draw steps ruler
115 painter.setPen(m_colorLineEnd);
116
117 TimePos curPos = m_curStepEndPos;
118 int x = xCoordOfTick(curPos);
119 while(x <= m_right)
120 {
121 const int w = 2;
122 const int h = 4;
123 painter.drawRect(x - 1, m_top, w, h);
124 curPos += m_stepsLength;
125 x = xCoordOfTick(curPos);
126 }
127
128 //draw current step start/end position lines
129 if(m_curStepStartPos != m_curStepEndPos)
130 {
131 drawVerLine(&painter, m_curStepStartPos, m_colorLineStart, m_top, m_bottom);
132 }
133
134 drawVerLine(&painter, m_curStepEndPos, m_colorLineEnd, m_top, m_bottom);
135
136 //if the line is adjacent to the keyboard at the left - it cannot be seen.
137 //add another line to make it clearer
138 if(m_curStepEndPos == 0)
139 {
140 drawVerLine(&painter, xCoordOfTick(m_curStepEndPos) + 1, m_colorLineEnd, m_top, m_bottom);
141 }
142}

144int StepRecorderWidget::xCoordOfTick(int tick)
145{
146 return m_marginLeft + ((tick - m_currentPosition) * m_ppb / TimePos::ticksPerBar());
147}
