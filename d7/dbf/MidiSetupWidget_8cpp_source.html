37MidiSetupWidget::MidiSetupWidget(const QString & caption, const QString & configSection,
38 const QString & devName, QWidget * parent) :
39 QGroupBox(QGroupBox::tr("Settings for %1").arg(tr(caption.toUtf8())), parent),
40 m_configSection(configSection),
41 m_device(nullptr)
42{
43 // supply devName=QString() (distinct from QString(""))
44 // to indicate that there is no editable device field
45 if (!devName.isNull())
46 {
47 QFormLayout * form = new QFormLayout(this);
48
49 m_device = new QLineEdit(devName, this);
50
51 form->addRow(tr("Device"), m_device);
52 }
53}

55void MidiSetupWidget::saveSettings()
56{
57 if (!m_configSection.isEmpty() && m_device)
58 {
59 ConfigManager::inst()->setValue(m_configSection, "device",
60 m_device->text());
61 }
62}

64void MidiSetupWidget::show()
65{
66 // the setup widget should only be visible if the device has some configurable attributes
67 bool visible = !m_configSection.isEmpty();
68 parentWidget()->setVisible(visible);
69 QWidget::setVisible(visible);
