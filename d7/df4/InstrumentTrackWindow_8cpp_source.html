351void InstrumentTrackWindow::modelChanged()
352{
353 m_track = castModel<InstrumentTrack>();
354
355 m_nameLineEdit->setText( m_track->name() );
356
357 m_track->disconnect( SIGNAL(nameChanged()), this );
358 m_track->disconnect( SIGNAL(instrumentChanged()), this );
359
360 connect( m_track, SIGNAL(nameChanged()),
361 this, SLOT(updateName()));
362 connect( m_track, SIGNAL(instrumentChanged()),
363 this, SLOT(updateInstrumentView()));
364
365 m_volumeKnob->setModel( &m_track->m_volumeModel );
366 m_panningKnob->setModel( &m_track->m_panningModel );
367 m_mixerChannelNumber->setModel( &m_track->m_mixerChannelModel );
368 m_pianoView->setModel( &m_track->m_piano );
369
370 if (m_track->instrument() && m_track->instrument()->isBendable())
371 {
372 m_pitchKnob->setModel( &m_track->m_pitchModel );
373 m_pitchRangeSpinBox->setModel( &m_track->m_pitchRangeModel );
374 m_pitchKnob->show();
375 m_pitchLabel->show();
376 m_pitchRangeSpinBox->show();
377 m_pitchRangeLabel->show();
378 }
379 else
380 {
381 m_pitchKnob->hide();
382 m_pitchLabel->hide();
383 m_pitchKnob->setModel( nullptr );
384 m_pitchRangeSpinBox->hide();
385 m_pitchRangeLabel->hide();
386 }
387
388 if (m_track->instrument() && m_track->instrument()->isMidiBased())
389 {
390 m_tuningView->microtunerNotSupportedLabel()->show();
391 m_tuningView->microtunerGroupBox()->hide();
392 m_track->m_microtuner.enabledModel()->setValue(false);
393 }
394 else
395 {
396 m_tuningView->microtunerNotSupportedLabel()->hide();
397 m_tuningView->microtunerGroupBox()->show();
398 }
399
400 m_ssView->setModel(&m_track->m_soundShaping);
401 m_noteStackingView->setModel(&m_track->m_noteStacking);
402 m_arpeggioView->setModel(&m_track->m_arpeggio);
403 m_midiView->setModel(&m_track->m_midiPort);
404 m_effectView->setModel(m_track->m_audioBusHandle.effects());
405 m_tuningView->pitchGroupBox()->setModel(&m_track->m_useMasterPitchModel);
406 m_tuningView->microtunerGroupBox()->setModel(m_track->m_microtuner.enabledModel());
407 m_tuningView->scaleCombo()->setModel(m_track->m_microtuner.scaleModel());

409 m_tuningView->rangeImportCheckbox()->setModel(m_track->m_microtuner.keyRangeImportModel());
410 updateName();
411
412 updateSubWindow();
413}

418void InstrumentTrackWindow::saveSettingsBtnClicked()
419{
420 FileDialog sfd(this, tr("Save preset"), "", tr("XML preset file (*.xpf)"));
421
422 QString presetRoot = ConfigManager::inst()->userPresetsDir();
423 if(!QDir(presetRoot).exists())
424 {
425 QDir().mkdir(presetRoot);
426 }
427 if(!QDir(presetRoot + m_track->instrumentName()).exists())
428 {
429 QDir(presetRoot).mkdir(m_track->instrumentName());
430 }
431
432 sfd.setAcceptMode(FileDialog::AcceptSave);
433 sfd.setDirectory(presetRoot + m_track->instrumentName());
434 sfd.setFileMode( FileDialog::AnyFile );
435 QString fname = m_track->name();
436 sfd.selectFile(fname.remove(QRegularExpression(FILENAME_FILTER)));
437 sfd.setDefaultSuffix( "xpf");
438
439 if( sfd.exec() == QDialog::Accepted &&
440 !sfd.selectedFiles().isEmpty() &&
441 !sfd.selectedFiles().first().isEmpty() )
442 {
443 DataFile dataFile(DataFile::Type::InstrumentTrackSettings);
444 QDomElement& content(dataFile.content());
445
446 m_track->savePreset(dataFile, content);
447 //We don't want to save muted & solo settings when we're saving a preset
448 content.setAttribute("muted", 0);
449 content.setAttribute("solo", 0);
450 content.setAttribute("mutedBeforeSolo", 0);
451 QString f = sfd.selectedFiles()[0];
452 dataFile.writeFile(f);
453 }
454}

576
577void InstrumentTrackWindow::dropEvent( QDropEvent* event )
578{
579 QString type = StringPairDrag::decodeKey( event );
580 QString value = StringPairDrag::decodeValue( event );
581
582 if( type == "instrument" )
583 {
584 m_track->loadInstrument( value, nullptr, true /* DnD */ );
585
586 Engine::getSong()->setModified();
587
588 event->accept();
589 setFocus();
590 }
591 else if( type == "presetfile" )
592 {
593 DataFile dataFile(value);
594 m_track->replaceInstrument(dataFile);
595 event->accept();
596 setFocus();
597 }
598 else if( type == "pluginpresetfile" )
599 {
600 const QString ext = FileItem::extension( value );
601 Instrument * i = m_track->instrument();
602
603 if( !i->descriptor()->supportsFileType( ext ) )
604 {
605 PluginFactory::PluginInfoAndKey piakn =
606 getPluginFactory()->pluginSupportingExtension(ext);
607 i = m_track->loadInstrument(piakn.info.name(), &piakn.key);
608 }
609
610 i->loadFile( value );
611
612 event->accept();
613 setFocus();
614 }
615}

639void InstrumentTrackWindow::viewInstrumentInDirection(int d)
640{
641 // helper routine for viewNextInstrument, viewPrevInstrument
642 // d=-1 to view the previous instrument,
643 // d=+1 to view the next instrument
644
645 const QList<TrackView *> &trackViews = m_itv->trackContainerView()->trackViews();
646 int idxOfMe = trackViews.indexOf(m_itv);
647
648 // search for the next InstrumentTrackView (i.e. skip AutomationViews, etc)
649 // sometimes, the next InstrumentTrackView may already be open, in which case
650 // replace our window contents with the *next* closed Instrument Track and
651 // give focus to the InstrumentTrackView we skipped.
652 int idxOfNext = idxOfMe;
653 InstrumentTrackView *newView = nullptr;
654 InstrumentTrackView *bringToFront = nullptr;
655 do
656 {
657 idxOfNext = (idxOfNext + d + trackViews.size()) % trackViews.size();
658 newView = dynamic_cast<InstrumentTrackView*>(trackViews[idxOfNext]);
659 // the window that should be brought to focus is the FIRST InstrumentTrackView that comes after us
660 if (bringToFront == nullptr && newView != nullptr)
661 {
662 bringToFront = newView;
663 }
664 // if the next instrument doesn't have an active window, then exit loop & load that one into our window.
665 if (newView != nullptr && !newView->m_tlb->isChecked())
666 {
667 break;
668 }
669 } while (idxOfNext != idxOfMe);
670
671 // avoid reloading the window if there is only one instrument, as that will just change the active tab
672 if (idxOfNext != idxOfMe)
673 {
674 // save current window pos and then hide the window by unchecking its button in the track list
675 QPoint curPos = parentWidget()->pos();
676 m_itv->m_tlb->setChecked(false);
677
678 // enable the new window by checking its track list button & moving it to where our window just was
679 newView->m_tlb->setChecked(true);
680 newView->getInstrumentTrackWindow()->parentWidget()->move(curPos);
681
682 // scroll the SongEditor/PatternEditor to make sure the new trackview label is visible
683 bringToFront->trackContainerView()->scrollToTrackView(bringToFront);
684
685 // get the instrument window to refresh
686 modelChanged();
687 }
688 Q_ASSERT(bringToFront);
689 bringToFront->getInstrumentTrackWindow()->setFocus();
690 Qt::WindowFlags flags = windowFlags();
691 if (!m_instrumentView->isResizable()) {
692 flags |= Qt::MSWindowsFixedSizeDialogHint;
693 } else {
694 flags &= ~Qt::MSWindowsFixedSizeDialogHint;
695 }
696 setWindowFlags( flags );
697}

738void InstrumentTrackWindow::updateSubWindow()
739{
740 auto subWindow = findSubWindowInParents();
741 if (subWindow && m_instrumentView)
742 {
743 Qt::WindowFlags flags = subWindow->windowFlags();
744
745 const auto instrumentViewResizable = m_instrumentView->isResizable();
746
747 if (instrumentViewResizable)
748 {
749 // TODO As of writing SlicerT is the only resizable instrument. Is this code specific to SlicerT?
750 const auto extraSpace = QSize(12, 208);
751 subWindow->setMaximumSize(m_instrumentView->maximumSize() + extraSpace);
752 subWindow->setMinimumSize(m_instrumentView->minimumSize() + extraSpace);
753
754 flags &= ~Qt::MSWindowsFixedSizeDialogHint;
755 flags |= Qt::WindowMaximizeButtonHint;
756 }
757 else
758 {
759 flags |= Qt::MSWindowsFixedSizeDialogHint;
760 flags &= ~Qt::WindowMaximizeButtonHint;
761
762 // The sub window might be reused from an instrument that was maximized. Show the sub window
763 // as normal, i.e. not maximized, if the instrument view is not resizable.
764 if (subWindow->isMaximized())
765 {
766 subWindow->showNormal();
767 }
768 }
769
770 subWindow->setWindowFlags(flags);
771
772 // Show or hide the Size and Maximize options from the system menu depending on whether the view is resizable or not
773 QMenu * systemMenu = subWindow->systemMenu();
774 systemMenu->actions().at(2)->setVisible(instrumentViewResizable); // Size
775 systemMenu->actions().at(4)->setVisible(instrumentViewResizable); // Maximize
776
777 // TODO This is only needed if the sub window is implemented with LMMS' own SubWindow class.
778 // If an QMdiSubWindow is used everything works automatically. It seems that SubWindow is
779 // missing some implementation details that QMdiSubWindow has.
780 auto subWin = dynamic_cast<SubWindow*>(subWindow);
781 if (subWin)
782 {
783 subWin->updateTitleBar();
784 }
785 }
786}
