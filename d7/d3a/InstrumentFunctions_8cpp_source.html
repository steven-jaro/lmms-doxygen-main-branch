152{
153 for( m_size = 0; m_size < MAX_CHORD_POLYPHONY; m_size++ )
154 {

158 }
159

161 }

230void InstrumentFunctionNoteStacking::processNote( NotePlayHandle * _n )
231{
232 const int base_note_key = _n->key();
233 const ChordTable & chord_table = ChordTable::getInstance();
234 // we add chord-subnotes to note if either note is a base-note and
235 // arpeggio is not used or note is part of an arpeggio
236 // at the same time we only add sub-notes if nothing of the note was
237 // played yet, because otherwise we would add chord-subnotes every
238 // time an audio-buffer is rendered...
239 if( ( _n->origin() == NotePlayHandle::Origin::Arpeggio || ( _n->hasParent() == false && _n->instrumentTrack()->isArpeggioEnabled() == false ) ) &&
240 _n->totalFramesPlayed() == 0 &&
241 m_chordsEnabledModel.value() == true && ! _n->isReleased() )
242 {
243 // then insert sub-notes for chord
244 const int selected_chord = m_chordsModel.value();
245
246 for( int octave_cnt = 0; octave_cnt < m_chordRangeModel.value(); ++octave_cnt )
247 {
248 const int sub_note_key_base = base_note_key + octave_cnt * KeysPerOctave;
249
250 // process all notes in the chord
251 for( int i = 0; i < chord_table.chords()[selected_chord].size(); ++i )
252 {
253 // add interval to sub-note-key
254 const int sub_note_key = sub_note_key_base + (int) chord_table.chords()[selected_chord][i];
255 // maybe we're out of range -> let's get outta
256 // here!
257 if( sub_note_key > NumKeys )
258 {
259 break;
260 }
261 // create copy of base-note
262 Note note_copy( _n->length(), 0, sub_note_key, _n->getVolume(), _n->getPanning(), _n->detuning() );
263
264 // create sub-note-play-handle, only note is
265 // different
266 Engine::audioEngine()->addPlayHandle(
267 NotePlayHandleManager::acquire( _n->instrumentTrack(), _n->offset(), _n->frames(), note_copy,
268 _n, -1, NotePlayHandle::Origin::NoteStacking )
269 );
270 }
271 }
272 }
273}

338void InstrumentFunctionArpeggio::processNote( NotePlayHandle * _n )
339{
340 const int base_note_key = _n->key();
341 if( _n->origin() == NotePlayHandle::Origin::Arpeggio ||
342 _n->origin() == NotePlayHandle::Origin::NoteStacking ||
343 !m_arpEnabledModel.value() ||
344 _n->isReleased() )
345 {
346 return;
347 }
348
349 // Set master note if not playing arp note or it will play as an ordinary note
350 _n->setMasterNote();
351
352 const int selected_arp = m_arpModel.value();
353 const auto arpMode = static_cast<ArpMode>(m_arpModeModel.value());
354
355 ConstNotePlayHandleList cnphv = NotePlayHandle::nphsOfInstrumentTrack(_n->instrumentTrack());
356
357 if(arpMode != ArpMode::Free && cnphv.size() == 0 )
358 {
359 // maybe we're playing only a preset-preview-note?
360 cnphv = PresetPreviewPlayHandle::nphsOfInstrumentTrack( _n->instrumentTrack() );
361 if( cnphv.size() == 0 )
362 {
363 // still nothing found here, so lets return
364 //return;
365 cnphv.push_back( _n );
366 }
367 }
368
369 // avoid playing same key for all
370 // currently playing notes if sort mode is enabled
371 if (arpMode == ArpMode::Sort && _n != cnphv.first()) { return; }
372
373 const InstrumentFunctionNoteStacking::ChordTable & chord_table = InstrumentFunctionNoteStacking::ChordTable::getInstance();
374 const int cur_chord_size = chord_table.chords()[selected_arp].size();
375 const int total_chord_size = cur_chord_size * cnphv.size();
376 // how many notes are in a single chord (multiplied by range)
377 const int singleNoteRange = static_cast<int>(cur_chord_size * m_arpRangeModel.value() * m_arpRepeatsModel.value());
378 // how many notes are in the final chord
379 const int range = arpMode == ArpMode::Sort ? singleNoteRange * cnphv.size() : singleNoteRange;
380
381 if (arpMode == ArpMode::Sort)
382 {
383 std::sort(cnphv.begin(), cnphv.end(), [](const NotePlayHandle* a, const NotePlayHandle* b)
384 {
385 return a->key() < b->key();
386 });
387 }
388
389 // number of frames that every note should be played
390 const auto arp_frames = (f_cnt_t)(m_arpTimeModel.value() / 1000.0f * Engine::audioEngine()->outputSampleRate());
391 const auto gated_frames = (f_cnt_t)(m_arpGateModel.value() * arp_frames / 100.0f);
392
393 // used for calculating remaining frames for arp-note, we have to add
394 // arp_frames-1, otherwise the first arp-note will not be setup
395 // correctly... -> arp_frames frames silence at the start of every note!
396 int cur_frame = (arpMode != ArpMode::Free ?
397 cnphv.first()->totalFramesPlayed() :
398 _n->totalFramesPlayed()) + arp_frames - 1;
399 // used for loop
400 f_cnt_t frames_processed = arpMode != ArpMode::Free ? cnphv.first()->noteOffset() : _n->noteOffset();
401
402 while( frames_processed < Engine::audioEngine()->framesPerPeriod() )
403 {
404 const f_cnt_t remaining_frames_for_cur_arp = arp_frames - ( cur_frame % arp_frames );
405 // does current arp-note fill whole audio-buffer or is the remaining time just
406 // a short bit that we can discard?
407 if( remaining_frames_for_cur_arp > Engine::audioEngine()->framesPerPeriod() ||
408 _n->frames() - _n->totalFramesPlayed() < arp_frames / 5 )
409 {
410 // then we don't have to do something!
411 break;
412 }
413
414 frames_processed += remaining_frames_for_cur_arp;
415
416 // Skip notes randomly
417 if( m_arpSkipModel.value() )
418 {
419 if (100 * static_cast<float>(rand()) / (static_cast<float>(RAND_MAX) + 1.0f) < m_arpSkipModel.value())
420 {
421 // update counters
422 frames_processed += arp_frames;
423 cur_frame += arp_frames;
424 continue;
425 }
426 }
427
428 auto dir = static_cast<ArpDirection>(m_arpDirectionModel.value());
429
430 // Miss notes randomly. We intercept int dir and abuse it
431 // after need. :)
432
433 if( m_arpMissModel.value() )
434 {
435 if (100 * static_cast<float>(rand()) / (static_cast<float>(RAND_MAX) + 1.0f) < m_arpMissModel.value())
436 {
437 dir = ArpDirection::Random;
438 }
439 }
440
441 int cur_arp_idx = 0;
442 // process according to arpeggio-direction...
443 if (dir == ArpDirection::Up || dir == ArpDirection::Down)
444 {
445 cur_arp_idx = (cur_frame / arp_frames) % range;
446 }
447 else if ((dir == ArpDirection::UpAndDown || dir == ArpDirection::DownAndUp) && range > 1)
448 {
449 // imagine, we had to play the arp once up and then
450 // once down -> makes 2 * range possible notes...
451 // because we don't play the lower and upper notes
452 // twice, we have to subtract 2
453 cur_arp_idx = (cur_frame / arp_frames) % (range * 2 - (2 * static_cast<int>(m_arpRepeatsModel.value())));
454 // if greater than range, we have to play down...
455 // looks like the code for arp_dir==DOWN... :)
456 if (cur_arp_idx >= range)
457 {
458 cur_arp_idx = range - cur_arp_idx % (range - 1) - static_cast<int>(m_arpRepeatsModel.value());
459 }
460 }
461 else if( dir == ArpDirection::Random )
462 {
463 // just pick a random chord-index
464 cur_arp_idx = static_cast<int>(range * static_cast<float>(rand()) / static_cast<float>(RAND_MAX));
465 }
466
467 // Divide cur_arp_idx with wanted repeats. The repeat feature will not affect random notes.
468 cur_arp_idx = static_cast<int>(cur_arp_idx / m_arpRepeatsModel.value());
469
470 // Cycle notes
471 if( m_arpCycleModel.value() && dir != ArpDirection::Random )
472 {
473 cur_arp_idx *= m_arpCycleModel.value() + 1;
474 cur_arp_idx %= static_cast<int>(range / m_arpRepeatsModel.value());
475 }
476
477 // If ArpDirection::Down or ArpDirection::DownAndUp, invert the final range.
478 if (dir == ArpDirection::Down || dir == ArpDirection::DownAndUp)
479 {
480 cur_arp_idx = static_cast<int>(range / m_arpRepeatsModel.value()) - cur_arp_idx - 1;
481 }
482
483 // now calculate final key for our arp-note
484 int sub_note_key = 0;
485 if (arpMode != ArpMode::Sort)
486 {
487 sub_note_key = base_note_key + (cur_arp_idx / cur_chord_size) *
488 KeysPerOctave + chord_table.chords()[selected_arp][cur_arp_idx % cur_chord_size];
489 }
490 else
491 {
492 const auto octaveDiv = std::div(cur_arp_idx, total_chord_size);
493 const int octave = octaveDiv.quot;
494 const auto arpDiv = std::div(octaveDiv.rem, cnphv.size());
495 const int arpIndex = arpDiv.rem;
496 const int chordIndex = arpDiv.quot;
497 sub_note_key = cnphv[arpIndex]->key()
498 + chord_table.chords()[selected_arp][chordIndex]
499 + octave * KeysPerOctave;
500 }
501
502 // range-checking
503 if( sub_note_key >= NumKeys ||
504 sub_note_key < 0 ||
505 Engine::audioEngine()->criticalXRuns() )
506 {
507 continue;
508 }
509
510 // create new arp-note
511
512 // create sub-note-play-handle, only ptr to note is different
513 // and is_arp_note=true
514 Engine::audioEngine()->addPlayHandle(
515 NotePlayHandleManager::acquire( _n->instrumentTrack(),
516 frames_processed,
517 gated_frames,
518 Note( TimePos( 0 ), TimePos( 0 ), sub_note_key, _n->getVolume(),
519 _n->getPanning(), _n->detuning() ),
520 _n, -1, NotePlayHandle::Origin::Arpeggio )
521 );
522
523 // update counters
524 frames_processed += arp_frames;
525 cur_frame += arp_frames;
526 }
527}
