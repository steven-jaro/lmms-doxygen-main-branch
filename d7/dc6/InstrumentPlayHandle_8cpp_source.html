43void InstrumentPlayHandle::play(SampleFrame* working_buffer)
44{
45 InstrumentTrack * instrumentTrack = m_instrument->instrumentTrack();
46
47 // ensure that all our nph's have been processed first
48 auto nphv = NotePlayHandle::nphsOfInstrumentTrack(instrumentTrack, true);
49
50 bool nphsLeft;
51 do
52 {
53 nphsLeft = false;
54 for (const auto& handle : nphv)
55 {
56 if (handle->state() != ThreadableJob::ProcessingState::Done && !handle->isFinished())
57 {
58 nphsLeft = true;
59 const_cast<NotePlayHandle*>(handle)->process();
60 }
61 }
62 }
63 while (nphsLeft);
64
65 m_instrument->play(working_buffer);
66
67 // Process the audio buffer that the instrument has just worked on...
68 const fpp_t frames = Engine::audioEngine()->framesPerPeriod();
69 instrumentTrack->processAudioBuffer(working_buffer, frames, nullptr);
70}
