297 QDir dir(ConfigManager::inst()->localeDir());
298 QStringList fileNames = dir.entryList(QStringList("*.qm"));
299 for(int i = 0; i < fileNames.size(); ++i)
300 {
301 // Extract ISO-639 language code from filename
302 fileNames[i].truncate(fileNames[i].lastIndexOf('.'));
303 // Skip invalid language codes
304 QLocale locale(fileNames[i]);
305 if (locale.language() == QLocale::C)
306 {
307 continue;
308 }
309 // Display the native language name or fallback to the English name
310 QString langName = locale.nativeLanguageName();
311 if (langName.isEmpty())
312 {
313 langName = QLocale::languageToString(locale.language());
314 }
315 m_languages.append(fileNames[i]);
316 changeLang->addItem(langName);
317 }
318
319 // If language unset, fallback to system language when available.
320 if(m_lang == "")
321 {
322 QString tmp = QLocale::system().name().left(2);
323 if(m_languages.contains(tmp))
324 {
325 m_lang = tmp;
326 }
327 else
328 {
329 m_lang = "en";
330 }
331 }
332
333 for(int i = 0; i < changeLang->count(); ++i)
334 {
335 if(m_lang == m_languages.at(i))
336 {
337 changeLang->setCurrentIndex(i);
338 break;
339 }
340 }

343 this, SLOT(setLanguage(int)));

1220void SetupDialog::updateBufferSizeWarning(int value)
1221{
1222 QString text = "<ul>";
1223 // 'value' is not a power of 2 (for value > 0) and under 256. On buffer sizes larger than 256
1224 // lmms works with chunks of size 256 and only the final mix will use the actual buffer size.
1225 // Plugins don't see a larger buffer size than 256 so anything larger than this is functionally
1226 // a 'power of 2' value.
1227 if(((value & (value - 1)) != 0) && value < 256)
1228 {
1229 text += "<li>" + tr("The currently selected value is not a power of 2 "
1230 "(32, 64, 128, 256). Some plugins may not be available.") + "</li>";
1231 }
1232 if(value <= 32)
1233 {
1234 text += "<li>" + tr("The currently selected value is less than or equal to 32. "
1235 "Some plugins may not be available.") + "</li>";
1236 }
1237 text += "</ul>";
1238 m_bufferSizeWarnLbl->setText(text);
1239}

1242void SetupDialog::setBufferSize(int value)
1243{
1244 const int step = DEFAULT_BUFFER_SIZE / BUFFERSIZE_RESOLUTION;
1245 if(value > step && value % step)
1246 {
1247 int mod_value = value % step;
1248 if(mod_value < step / 2)
1249 {
1250 m_bufferSizeSlider->setValue(value - mod_value);
1251 }
1252 else
1253 {
1254 m_bufferSizeSlider->setValue(value + step - mod_value);
1255 }
1256 return;
1257 }
1258
1259 if(m_bufferSizeSlider->value() != value)
1260 {
1261 m_bufferSizeSlider->setValue(value);
1262 }
1263
1264 m_bufferSize = value * BUFFERSIZE_RESOLUTION;
1265 m_bufferSizeLbl->setText(tr("Frames: %1\nLatency: %2 ms").arg(m_bufferSize).arg(
1266 1000.0f * m_bufferSize / Engine::audioEngine()->outputSampleRate(), 0, 'f', 1));
1267 updateBufferSizeWarning(m_bufferSize);
1268}
