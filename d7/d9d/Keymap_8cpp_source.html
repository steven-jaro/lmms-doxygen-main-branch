67/**
68 * \brief Return scale degree for a given key, based on current map and first/middle/last notes
69 * \param MIDI key to be mapped
70 * \return Scale degree defined by the mapping on success, -1 if key isn't mapped
71 */
72int Keymap::getDegree(int key) const
73{
74 if (key < m_firstKey || key > m_lastKey) {return -1;}
75 if (m_map.empty()) {return key;} // exception: empty mapping table means linear (1:1) mapping
76
77 const int keyOffset = key - m_middleKey; // -127..127
78 const int key_rem = keyOffset % static_cast<int>(m_map.size()); // remainder
79 const int key_mod = key_rem >= 0 ? key_rem : key_rem + m_map.size(); // true modulo
80 return m_map[key_mod];
81}

84/**
85 * \brief Return octave offset for a given key, based on current map and the middle note
86 * \param MIDI key to be mapped
87 * \return Octave offset defined by the mapping on success, 0 if key isn't mapped
88 */
89int Keymap::getOctave(int key) const
90{
91 // The keymap wraparound cannot cause an octave transition if a key isn't mapped or the map is empty â†’ return 0
92 if (m_map.empty() || getDegree(key) == -1) {return 0;}
93
94 const int keyOffset = key - m_middleKey;
95 if (keyOffset >= 0)
96 {
97 return keyOffset / static_cast<int>(m_map.size());
98 }
99 else
100 {
101 return (keyOffset + 1) / static_cast<int>(m_map.size()) - 1;
102 }
103}

118void Keymap::saveSettings(QDomDocument &document, QDomElement &element)
119{
120 element.setAttribute("description", m_description);
121
122 element.setAttribute("first_key", m_firstKey);
123 element.setAttribute("last_key", m_lastKey);
124 element.setAttribute("middle_key", m_middleKey);
125 element.setAttribute("base_key", m_baseKey);
126 element.setAttribute("base_freq", m_baseFreq);
127
128 for (int value : m_map)
129 {
130 QDomElement degree = document.createElement("degree");
131 element.appendChild(degree);
132 degree.setAttribute("value", value);
133 }
134}

137void Keymap::loadSettings(const QDomElement &element)
138{
139 m_description = element.attribute("description");
140
141 m_firstKey = element.attribute("first_key").toInt();
142 m_lastKey = element.attribute("last_key").toInt();
143 m_middleKey = element.attribute("middle_key").toInt();
144 m_baseKey = element.attribute("base_key").toInt();
145 m_baseFreq = element.attribute("base_freq").toDouble();
146
147 QDomNode node = element.firstChild();
148 m_map.clear();
149
150 while (!node.isNull())
151 {
152 m_map.push_back(node.toElement().attribute("value").toInt());
153 node = node.nextSibling();
154 }
155}
