67void AutomatableSlider::mousePressEvent( QMouseEvent * _me )
68{
69 if( _me->button() == Qt::LeftButton &&
70 ! ( _me->modifiers() & Qt::ControlModifier ) )
71 {
72 m_showStatus = true;
73 QSlider::mousePressEvent( _me );
