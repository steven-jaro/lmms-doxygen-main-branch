31template <typename MutableBufferSequence>
32class reactive_socket_recv_op_base : public reactor_op
33{
34public:
35 reactive_socket_recv_op_base(socket_type socket,
36 socket_ops::state_type state, const MutableBufferSequence& buffers,
37 socket_base::message_flags flags, func_type complete_func)
38 : reactor_op(&reactive_socket_recv_op_base::do_perform, complete_func),
39 socket_(socket),
40 state_(state),
41 buffers_(buffers),
42 flags_(flags)
43 {
44 }
45
46 static status do_perform(reactor_op* base)
47 {
48 reactive_socket_recv_op_base* o(
49 static_cast<reactive_socket_recv_op_base*>(base));
50
51 buffer_sequence_adapter<asio::mutable_buffer,
52 MutableBufferSequence> bufs(o->buffers_);
53
54 status result = socket_ops::non_blocking_recv(o->socket_,
55 bufs.buffers(), bufs.count(), o->flags_,
56 (o->state_ & socket_ops::stream_oriented) != 0,
57 o->ec_, o->bytes_transferred_) ? done : not_done;
58
59 if (result == done)
60 if ((o->state_ & socket_ops::stream_oriented) != 0)
61 if (o->bytes_transferred_ == 0)
62 result = done_and_exhausted;
63
64 ASIO_HANDLER_REACTOR_OPERATION((*o, "non_blocking_recv",
65 o->ec_, o->bytes_transferred_));
66
67 return result;
68 }
69
70private:
71 socket_type socket_;
72 socket_ops::state_type state_;
73 MutableBufferSequence buffers_;
74 socket_base::message_flags flags_;
75};

77template <typename MutableBufferSequence, typename Handler, typename IoExecutor>
78class reactive_socket_recv_op :
79 public reactive_socket_recv_op_base<MutableBufferSequence>
80{
81public:
82 ASIO_DEFINE_HANDLER_PTR(reactive_socket_recv_op);
83
84 reactive_socket_recv_op(socket_type socket, socket_ops::state_type state,
85 const MutableBufferSequence& buffers, socket_base::message_flags flags,
86 Handler& handler, const IoExecutor& io_ex)
87 : reactive_socket_recv_op_base<MutableBufferSequence>(socket, state,
88 buffers, flags, &reactive_socket_recv_op::do_complete),
89 handler_(ASIO_MOVE_CAST(Handler)(handler)),
90 io_executor_(io_ex)
91 {
92 handler_work<Handler, IoExecutor>::start(handler_, io_executor_);
93 }
94
95 static void do_complete(void* owner, operation* base,
96 const asio::error_code& /*ec*/,
97 std::size_t /*bytes_transferred*/)
98 {
99 // Take ownership of the handler object.
100 reactive_socket_recv_op* o(static_cast<reactive_socket_recv_op*>(base));
101 ptr p = { asio::detail::addressof(o->handler_), o, o };
102 handler_work<Handler, IoExecutor> w(o->handler_, o->io_executor_);
103
104 ASIO_HANDLER_COMPLETION((*o));
105
106 // Make a copy of the handler so that the memory can be deallocated before
107 // the upcall is made. Even if we're not about to make an upcall, a
108 // sub-object of the handler may be the true owner of the memory associated
109 // with the handler. Consequently, a local copy of the handler is required
110 // to ensure that any owning sub-object remains valid until after we have
111 // deallocated the memory here.
112 detail::binder2<Handler, asio::error_code, std::size_t>
113 handler(o->handler_, o->ec_, o->bytes_transferred_);
114 p.h = asio::detail::addressof(handler.handler_);
115 p.reset();
116
117 // Make the upcall if required.
118 if (owner)
119 {
120 fenced_block b(fenced_block::half);
121 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
122 w.complete(handler, handler.handler_);
123 ASIO_HANDLER_INVOCATION_END;
124 }
125 }
126
127private:
128 Handler handler_;
129 IoExecutor io_executor_;
130};
