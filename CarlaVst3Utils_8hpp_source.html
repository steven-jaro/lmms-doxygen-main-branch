144static inline constexpr
145uint32_t v3_cconst(const uint8_t a, const uint8_t b, const uint8_t c, const uint8_t d) noexcept
146{
147 return static_cast<uint32_t>((a << 24) | (b << 16) | (c << 8) | (d << 0));
148}

150static inline
151const char* tuid2str(const v3_tuid iid) noexcept
152{
153 static char buf[44];
154 std::snprintf(buf, sizeof(buf), "0x%08X,0x%08X,0x%08X,0x%08X",
155 v3_cconst(iid[ 0], iid[ 1], iid[ 2], iid[ 3]),
156 v3_cconst(iid[ 4], iid[ 5], iid[ 6], iid[ 7]),
157 v3_cconst(iid[ 8], iid[ 9], iid[10], iid[11]),
158 v3_cconst(iid[12], iid[13], iid[14], iid[15]));
159 return buf;
160}

182template<class v3_class, const v3_tuid cid>
183static inline
184v3_result V3_API v3_query_interface(void* const self, const v3_tuid iid, void** const iface)
185{
186 v3_class* const cls = *static_cast<v3_class**>(self);
187
188 if (v3_tuid_match(iid, v3_funknown_iid) || v3_tuid_match(iid, cid))
189 {
190 ++cls->refcounter;
191 *iface = self;
192 return V3_OK;
193 }
194
195 *iface = nullptr;
196 return V3_NO_INTERFACE;
197}
198
199template<class v3_class>
200static inline
201uint32_t V3_API v3_ref(void* const self)
202{
203 v3_class* const cls = *static_cast<v3_class**>(self);
204 return ++cls->refcounter;
205}
206
207template<class v3_class>
208static inline
209uint32_t V3_API v3_unref(void* const self)
210{
211 v3_class** const clsptr = static_cast<v3_class**>(self);
212 v3_class* const cls = *clsptr;
213
214 if (const int refcount = --cls->refcounter)
215 return refcount;
216
217 delete cls;
218 delete clsptr;
219 return 0;
220}
