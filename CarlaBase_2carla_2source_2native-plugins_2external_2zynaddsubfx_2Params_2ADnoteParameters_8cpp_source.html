577/*
578 * Get the Multiplier of the fine detunes of the voices
579 */
580float ADnoteParameters::getBandwidthDetuneMultiplier() const
581{
582 float bw = (GlobalPar.PBandwidth - 64.0f) / 64.0f;
583 bw = powf(2.0f, bw * powf(fabs(bw), 0.2f) * 5.0f);
584
585 return bw;
586}
