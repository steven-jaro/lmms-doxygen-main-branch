74 :Effect(pars),
75 lfo(pars.srate, pars.bufsize),
76 maxdelay((int)(MAX_CHORUS_DELAY / 1000.0f * samplerate_f)),

95//get the delay value in samples; xlfo is the current lfo value
96float Chorus::getdelay(float xlfo)
97{
98 float result =
99 (Pflangemode) ? 0 : (delay + xlfo * depth) * samplerate_f;
100
101 //check if delay is too big (caused by bad setdelay() and setdepth()
102 if((result + 0.5f) >= maxdelay) {
103 cerr
104 <<
105 "WARNING: Chorus.cpp::getdelay(..) too big delay (see setdelay and setdepth funcs.)"
106 << endl;
107 result = maxdelay - 1.0f;
108 }
109 return result;
110}

112//Apply the effect
113void Chorus::out(const Stereo<float *> &input)
114{
115 dl1 = dl2;
116 dr1 = dr2;
117 lfo.effectlfoout(&lfol, &lfor);
118
119 dl2 = getdelay(lfol);
120 dr2 = getdelay(lfor);
121
122 for(int i = 0; i < buffersize; ++i) {
123 float inL = input.l[i];
124 float inR = input.r[i];
125 //LRcross
126 Stereo<float> tmpc(inL, inR);
127 inL = tmpc.l * (1.0f - lrcross) + tmpc.r * lrcross;
128 inR = tmpc.r * (1.0f - lrcross) + tmpc.l * lrcross;
129
130 //Left channel
131
132 //compute the delay in samples using linear interpolation between the lfo delays
133 float mdel =
134 (dl1 * (buffersize - i) + dl2 * i) / buffersize_f;
135 if(++dlk >= maxdelay)
136 dlk = 0;
137 float tmp = dlk - mdel + maxdelay * 2.0f; //where should I get the sample from
138
139 dlhi = (int) tmp;
140 dlhi %= maxdelay;
141
142 float dlhi2 = (dlhi - 1 + maxdelay) % maxdelay;
143 float dllo = 1.0f + floorf(tmp) - tmp;
144 efxoutl[i] = cinterpolate(delaySample.l, maxdelay, dlhi2) * dllo
145 + cinterpolate(delaySample.l, maxdelay,
146 dlhi) * (1.0f - dllo);
147 delaySample.l[dlk] = inL + efxoutl[i] * fb;
148
149 //Right channel
150
151 //compute the delay in samples using linear interpolation between the lfo delays
152 mdel = (dr1 * (buffersize - i) + dr2 * i) / buffersize_f;
153 if(++drk >= maxdelay)
154 drk = 0;
155 tmp = drk * 1.0f - mdel + maxdelay * 2.0f; //where should I get the sample from
156
157 dlhi = (int) tmp;
158 dlhi %= maxdelay;
159
160 dlhi2 = (dlhi - 1 + maxdelay) % maxdelay;
161 dllo = 1.0f + floorf(tmp) - tmp;
162 efxoutr[i] = cinterpolate(delaySample.r, maxdelay, dlhi2) * dllo
163 + cinterpolate(delaySample.r, maxdelay,
164 dlhi) * (1.0f - dllo);
165 delaySample.r[dlk] = inR + efxoutr[i] * fb;
166 }
167
168 if(Poutsub)
169 for(int i = 0; i < buffersize; ++i) {
170 efxoutl[i] *= -1.0f;
171 efxoutr[i] *= -1.0f;
172 }
173
174 for(int i = 0; i < buffersize; ++i) {
175 efxoutl[i] *= pangainL;
176 efxoutr[i] *= pangainR;
177 }
178}

231 //Flange2
232 {64, 64, 33, 34, 1, 40, 35, 3, 109, 0, 0, 0},

234 {64, 64, 53, 34, 1, 94, 35, 3, 54, 0, 0, 1},
235 //Flange4
236 {64, 64, 40, 0, 1, 62, 12, 19, 97, 0, 0, 0},

240
241 if(npreset >= NUM_PRESETS)

249void Chorus::changepar(int npar, unsigned char value)
250{
251 switch(npar) {
252 case 0:
253 setvolume(value);
254 break;
255 case 1:
256 setpanning(value);
257 break;
258 case 2:
259 lfo.Pfreq = value;
260 lfo.updateparams();
261 break;
262 case 3:
263 lfo.Prandomness = value;
264 lfo.updateparams();
265 break;
266 case 4:
267 lfo.PLFOtype = value;
268 lfo.updateparams();
269 break;
270 case 5:
271 lfo.Pstereo = value;
272 lfo.updateparams();
273 break;
274 case 6:
275 setdepth(value);
276 break;
277 case 7:
278 setdelay(value);
279 break;
280 case 8:
281 setfb(value);
282 break;
283 case 9:
284 setlrcross(value);
285 break;
286 case 10:
287 Pflangemode = (value > 1) ? 1 : value;
288 break;
289 case 11:
290 Poutsub = (value > 1) ? 1 : value;
291 break;
292 }
293}
