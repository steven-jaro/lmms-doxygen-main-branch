27static inline int CAS(register UInt32 value, register UInt32 newvalue, register volatile void* addr)
28{
29 register int result;
30 register UInt32 tmp;
31 asm volatile (
32 "# CAS \n"
33 " lwarx %4, 0, %1 \n" // creates a reservation on addr
34 " cmpw %4, %2 \n" // test value at addr
35 " bne- 1f \n"
36 " sync \n" // synchronize instructions
37 " stwcx. %3, 0, %1 \n" // if the reservation is not altered
38 // stores the new value at addr
39 " bne- 1f \n"
40 " li %0, 1 \n"
41 " b 2f \n"
42 "1: \n"
43 " li %0, 0 \n"
44 "2: \n"
45 : "=r" (result)
46 : "r" (addr), "r" (value), "r" (newvalue), "r" (tmp)
47 );
48 return result;
49}
