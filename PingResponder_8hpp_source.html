114 // Operator to handle incoming messages on the interface
115 template <typename It>
116 void operator()(const asio::ip::udp::endpoint& from, const It begin, const It end)
117 {
118 using namespace discovery;
119
120 // Decode Ping Message
121 const auto result = link::v1::parseMessageHeader(begin, end);
122 const auto& header = result.first;
123 const auto payloadBegin = result.second;
124
125 // Check Payload size
126 const auto payloadSize = static_cast<std::size_t>(std::distance(payloadBegin, end));
127 const auto maxPayloadSize =
128 sizeInByteStream(makePayload(HostTime{}, PrevGHostTime{}));
129 if (header.messageType == v1::kPing && payloadSize <= maxPayloadSize)
130 {
131 debug(mLog) << " Received ping message from " << from;
132
133 try
134 {
135 reply(std::move(payloadBegin), std::move(end), from);
136 }
137 catch (const std::runtime_error& err)
138 {
139 info(mLog) << " Failed to send pong to " << from << ". Reason: " << err.what();
140 }
141 }
142 else
143 {
144 info(mLog) << " Received invalid Message from " << from << ".";
145 }
146 listen();
147 }
148
149 template <typename It>
150 void reply(It begin, It end, const asio::ip::udp::endpoint& to)
151 {
152 using namespace discovery;
153
154 // Encode Pong Message
155 const auto id = SessionMembership{mSessionId};
156 const auto currentGt = GHostTime{mGhostXForm.hostToGhost(mClock.micros())};
157 const auto pongPayload = makePayload(id, currentGt);
158
159 v1::MessageBuffer pongBuffer;
160 const auto pongMsgBegin = std::begin(pongBuffer);
161 auto pongMsgEnd = v1::pongMessage(pongPayload, pongMsgBegin);
162 // Append ping payload to pong message.
163 pongMsgEnd = std::copy(begin, end, pongMsgEnd);
164
165 const auto numBytes =
166 static_cast<std::size_t>(std::distance(pongMsgBegin, pongMsgEnd));
167 mSocket.send(pongBuffer.data(), numBytes, to);
168 }
169
170 SessionId mSessionId;
171 GhostXForm mGhostXForm;
172 Clock mClock;
173 typename IoType::type::Log mLog;
174 Socket mSocket;
175 };
176
177 IoType mIo;
178 std::shared_ptr<Impl> mpImpl;
179};
