88 void operator()(const ::asio::error_code& error, const std::size_t numBytes)
89 {
90 if (!error && numBytes > 0 && numBytes <= MaxPacketSize)
91 {
92 const auto bufBegin = begin(mReceiveBuffer);
93 mHandler(mSenderEndpoint, bufBegin, bufBegin + static_cast<ptrdiff_t>(numBytes));
94 }
95 }
