116static inline bool

159static inline bool
160lv2_event_write(LV2_Event_Iterator* iter,
161 uint32_t frames,
162 uint32_t subframes,
163 uint16_t type,
164 uint16_t size,
165 const uint8_t* data)
166{
167 if (!iter->buf) {
168 return false;
169 }
170
171 if (iter->buf->capacity - iter->buf->size < sizeof(LV2_Event) + size) {
172 return false;
173 }
174
175 LV2_Event* const ev = (LV2_Event*)(iter->buf->data + iter->offset);
176
177 ev->frames = frames;
178 ev->subframes = subframes;
179 ev->type = type;
180 ev->size = size;
181 memcpy((uint8_t*)ev + sizeof(LV2_Event), data, size);
182 ++iter->buf->event_count;
183
184 size = lv2_event_pad_size((uint16_t)(sizeof(LV2_Event) + size));
185 iter->buf->size += size;
186 iter->offset += size;
187
188 return true;
189}

207
208 ev->frames = frames;
209 ev->subframes = subframes;
210 ev->type = type;
211 ev->size = size;
212 ++iter->buf->event_count;
213
214 const uint16_t padded_size = lv2_event_pad_size(total_size);
215 iter->buf->size += padded_size;
216 iter->offset += padded_size;

243 iter->buf->size += padded_size;
244 iter->offset += padded_size;
245
246 return true;
