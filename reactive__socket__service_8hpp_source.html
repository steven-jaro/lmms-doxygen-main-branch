245 const ConstBufferSequence& buffers,

256 op::ptr::allocate(handler), 0 };

264 p.v = p.p = 0;

289 // Receive a datagram with the endpoint of the sender. Returns the number of
290 // bytes received.
291 template <typename MutableBufferSequence>
292 size_t receive_from(implementation_type& impl,
293 const MutableBufferSequence& buffers,
294 endpoint_type& sender_endpoint, socket_base::message_flags flags,
295 asio::error_code& ec)
296 {
297 buffer_sequence_adapter<asio::mutable_buffer,
298 MutableBufferSequence> bufs(buffers);
299
300 std::size_t addr_len = sender_endpoint.capacity();
301 std::size_t bytes_recvd = socket_ops::sync_recvfrom(
302 impl.socket_, impl.state_, bufs.buffers(), bufs.count(),
303 flags, sender_endpoint.data(), &addr_len, ec);
304
305 if (!ec)
306 sender_endpoint.resize(addr_len);
307
308 return bytes_recvd;
309 }

311 // Wait until data can be received without blocking.
312 size_t receive_from(implementation_type& impl, const null_buffers&,
313 endpoint_type& sender_endpoint, socket_base::message_flags,
314 asio::error_code& ec)
315 {
316 // Wait for socket to become ready.
317 socket_ops::poll_read(impl.socket_, impl.state_, -1, ec);
318
319 // Reset endpoint since it can be given no sensible value at this time.
320 sender_endpoint = endpoint_type();
321
322 return 0;
323 }

385 // Accept a new connection.
386 template <typename Socket>
387 asio::error_code accept(implementation_type& impl,
388 Socket& peer, endpoint_type* peer_endpoint, asio::error_code& ec)
389 {
390 // We cannot accept a socket that is already open.
391 if (peer.is_open())
392 {
393 ec = asio::error::already_open;
394 return ec;
395 }
396
397 std::size_t addr_len = peer_endpoint ? peer_endpoint->capacity() : 0;
398 socket_holder new_socket(socket_ops::sync_accept(impl.socket_,
399 impl.state_, peer_endpoint ? peer_endpoint->data() : 0,
400 peer_endpoint ? &addr_len : 0, ec));
401
402 // On success, assign new connection to peer socket object.
403 if (new_socket.get() != invalid_socket)
404 {
405 if (peer_endpoint)
406 peer_endpoint->resize(addr_len);
407 peer.assign(impl.protocol_, new_socket.get(), ec);
408 if (!ec)
409 new_socket.release();
410 }
411
412 return ec;
413 }
