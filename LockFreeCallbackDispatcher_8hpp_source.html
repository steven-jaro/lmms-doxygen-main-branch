31// Utility to signal invocation of a callback on another thread in a lock free manner.
32// The callback is evoked on a thread owned by the instance of this class.
33//
34// A condition variable is used to notify a waiting thread, but only if the required
35// lock can be acquired immediately. If that fails, we fall back on signaling
36// after a timeout. This gives us a guaranteed minimum signalling rate which is defined
37// by the fallbackPeriod parameter.
38
39template <typename Callback, typename Duration>
40class LockFreeCallbackDispatcher
41{
42public:
43 LockFreeCallbackDispatcher(Callback callback, Duration fallbackPeriod)
44 : mCallback(std::move(callback))
45 , mFallbackPeriod(std::move(fallbackPeriod))
46 , mRunning(true)
47 , mThread([this] { run(); })
48 {
49 }
50
51 ~LockFreeCallbackDispatcher()
52 {
53 mRunning = false;
54 mCondition.notify_one();
55 mThread.join();
56 }
57
58 void invoke()
59 {
60 if (mMutex.try_lock())
61 {
62 mCondition.notify_one();
63 mMutex.unlock();
64 }
65 }
66
67private:
68 void run()
69 {
70 while (mRunning.load())
71 {
72 {
73 std::unique_lock<std::mutex> lock(mMutex);
74 mCondition.wait_for(lock, mFallbackPeriod);
75 }
76 mCallback();
77 }
78 }
79
80 Callback mCallback;
81 Duration mFallbackPeriod;
82 std::atomic<bool> mRunning;
83 std::mutex mMutex;
84 std::condition_variable mCondition;
85 std::thread mThread;
86};
