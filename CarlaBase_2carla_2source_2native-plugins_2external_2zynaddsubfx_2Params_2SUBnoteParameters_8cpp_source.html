227float SUBnoteParameters::convertBandwidth(int bw_, int stages, float freq,
228 int scale, int relbw)
229{
230 //the bandwidth is not absolute(Hz); it is relative to frequency
231 float bw = powf(10, (bw_ - 127.0f) / 127.0f * 4) * stages;
232
233 //Bandwidth Scale
234 bw *= powf(1000 / freq, (scale - 64.0f) / 64.0f * 3.0f);
235
236 //Relative BandWidth
237 bw *= powf(100, (relbw - 64.0f) / 64.0f);
238
239 if(bw > 25.0f)
240 bw = 25.0f;
241
242 return bw;
243}

245float SUBnoteParameters::convertHarmonicMag(int mag, int type)
246{
247 const float hmagnew = 1.0f - mag / 127.0f;
248
249 switch(type) {
250 case 1:
251 return expf(hmagnew * logf(0.01f));
252 break;
253 case 2:
254 return expf(hmagnew * logf(0.001f));
255 break;
256 case 3:
257 return expf(hmagnew * logf(0.0001f));
258 break;
259 case 4:
260 return expf(hmagnew * logf(0.00001f));
261 break;
262 default:
263 return 1.0f - hmagnew;
264 }
265}

408void SUBnoteParameters::updateFrequencyMultipliers(void) {
409 float par1 = POvertoneSpread.par1 / 255.0f;
410 float par1pow = powf(10.0f,
411 -(1.0f - POvertoneSpread.par1 / 255.0f) * 3.0f);
412 float par2 = POvertoneSpread.par2 / 255.0f;
413 float par3 = 1.0f - POvertoneSpread.par3 / 255.0f;
414 float result;
415 float tmp = 0.0f;
416 int thresh = 0;
417
418 for(int n = 0; n < MAX_SUB_HARMONICS; ++n) {
419 float n1 = n + 1.0f;
420 switch(POvertoneSpread.type) {
421 case 1:
422 thresh = (int)(100.0f * par2 * par2) + 1;
423 if (n1 < thresh)
424 result = n1;
425 else
426 result = n1 + 8.0f * (n1 - thresh) * par1pow;
427 break;
428 case 2:
429 thresh = (int)(100.0f * par2 * par2) + 1;
430 if (n1 < thresh)
431 result = n1;
432 else
433 result = n1 + 0.9f * (thresh - n1) * par1pow;
434 break;
435 case 3:
436 tmp = par1pow * 100.0f + 1.0f;
437 result = powf(n / tmp, 1.0f - 0.8f * par2) * tmp + 1.0f;
438 break;
439 case 4:
440 result = n * (1.0f - par1pow) +
441 powf(0.1f * n, 3.0f * par2 + 1.0f) *
442 10.0f * par1pow + 1.0f;
443 break;
444
445 case 5:
446 result = n1 + 2.0f * sinf(n * par2 * par2 * PI * 0.999f) *
447 sqrt(par1pow);
448 break;
449 case 6:
450 tmp = powf(2.0f * par2, 2.0f) + 0.1f;
451 result = n * powf(par1 * powf(0.8f * n, tmp) + 1.0f, tmp) +
452 1.0f;
453 break;
454
455 case 7:
456 result = (n1 + par1) / (par1 + 1);
457 break;
458 default:
459 result = n1;
460 }
461 float iresult = floor(result + 0.5f);
462 POvertoneFreqMult[n] = iresult + par3 * (result - iresult);
463 }
464}
