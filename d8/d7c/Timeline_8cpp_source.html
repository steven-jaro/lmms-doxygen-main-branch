64void Timeline::saveSettings(QDomDocument& doc, QDomElement& element)
65{
66 element.setAttribute("lp0pos", static_cast<int>(loopBegin()));
67 element.setAttribute("lp1pos", static_cast<int>(loopEnd()));
68 element.setAttribute("lpstate", static_cast<int>(loopEnabled()));
69 element.setAttribute("stopbehaviour", static_cast<int>(stopBehaviour()));
70}
