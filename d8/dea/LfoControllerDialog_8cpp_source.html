209void LfoControllerDialog::askUserDefWave()
210{
211 const auto fileName = SampleLoader::openWaveformFile();
212 if (fileName.isEmpty()) { return; }
213
214 auto lfoModel = dynamic_cast<LfoController*>(model());
215 auto& buffer = lfoModel->m_userDefSampleBuffer;
216 buffer = SampleLoader::createBufferFromFile(fileName);
217
218 m_userWaveBtn->setToolTip(buffer->audioFile());
219}
