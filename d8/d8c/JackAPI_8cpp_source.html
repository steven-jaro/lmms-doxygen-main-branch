305static inline void WaitGraphChange()
306{
307 /*
308 TLS key that is set only in RT thread, so never waits for pending
309 graph change in RT context (just read the current graph state).
310 */
311
312 if (jack_tls_get(JackGlobals::fRealTimeThread) == NULL) {
313 JackGraphManager* manager = GetGraphManager();
314 JackEngineControl* control = GetEngineControl();
315 assert(manager);
316 assert(control);
317 if (manager->IsPendingChange()) {
318 jack_log("WaitGraphChange...");
319 JackSleep(int(control->fPeriodUsecs * 1.1f));
320 }
321 }
322}

334LIB_EXPORT jack_client_t* jack_client_new(const char* client_name)
335{
336 JackGlobals::CheckContext("jack_client_new");
337
338 try {
339 assert(JackGlobals::fOpenMutex);
340 JackGlobals::fOpenMutex->Lock();
341 jack_error("jack_client_new: deprecated");
342 int options = JackUseExactName;
343 if (getenv("JACK_START_SERVER") == NULL) {
344 options |= JackNoStartServer;
345 }
346 jack_client_t* res = jack_client_new_aux(client_name, (jack_options_t)options, NULL);
347 JackGlobals::fOpenMutex->Unlock();
348 return res;
349 } catch (std::bad_alloc& e) {
350 jack_error("Memory allocation error...");
351 return NULL;
352 } catch (...) {
353 jack_error("Unknown error...");
354 return NULL;
355 }
356}

496 return 0;
497 } else {
498 return manager->IsConnected(src, dst);

503LIB_EXPORT int jack_port_tie(jack_port_t* src, jack_port_t* dst)
504{
505 JackGlobals::CheckContext("jack_port_tie");
506
507 uintptr_t src_aux = (uintptr_t)src;
508 jack_port_id_t mysrc = (jack_port_id_t)src_aux;
509 if (!CheckPort(mysrc)) {
510 jack_error("jack_port_tie called with a NULL src port");
511 return -1;
512 }
513 uintptr_t dst_aux = (uintptr_t)dst;
514 jack_port_id_t mydst = (jack_port_id_t)dst_aux;
515 if (!CheckPort(mydst)) {
516 jack_error("jack_port_tie called with a NULL dst port");
517 return -1;
518 }
519 JackGraphManager* manager = GetGraphManager();
520 if (manager && manager->GetPort(mysrc)->GetRefNum() != manager->GetPort(mydst)->GetRefNum()) {
521 jack_error("jack_port_tie called with ports not belonging to the same client");
522 return -1;
523 } else {
524 return (manager ? manager->GetPort(mydst)->Tie(mysrc) : -1);
525 }
526}
