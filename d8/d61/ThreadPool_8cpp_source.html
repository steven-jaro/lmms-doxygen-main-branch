29ThreadPool::ThreadPool(size_t numWorkers)
30{
31 assert(numWorkers > 0);
32
33 m_workers.reserve(numWorkers);
34 for (size_t i = 0; i < numWorkers; ++i)
35 {
36 m_workers.emplace_back([this] { run(); });
37 }
38}

40ThreadPool::~ThreadPool()
41{
42 {
43 const auto lock = std::unique_lock{m_runMutex};
44 m_done = true;
45 }
46
47 m_runCond.notify_all();
48
49 for (auto& worker : m_workers)
50 {
51 if (worker.joinable()) { worker.join(); }
52 }
53}

60void ThreadPool::run()
61{
62 while (!m_done)
63 {
64 std::function<void()> task;
65 {
66 auto lock = std::unique_lock{m_runMutex};
67 m_runCond.wait(lock, [this] { return !m_queue.empty() || m_done; });
68
69 if (m_done) { break; }
70 task = m_queue.front();
71 m_queue.pop();
72 }
73 task();
74 }
75}
