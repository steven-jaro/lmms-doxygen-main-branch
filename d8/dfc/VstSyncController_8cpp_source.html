40VstSyncController::VstSyncController()
41{
42 try
43 {
44 m_syncData.create();
45 }
46 catch (const std::runtime_error& error)
47 {
48 qCritical() << "Failed to allocate shared memory for VST sync:" << error.what();
49 return;
50 }
51
52 m_syncData->isPlaying = false;
53 m_syncData->bufferSize = Engine::audioEngine()->framesPerPeriod();
54 m_syncData->timeSigNumer = 4;
55 m_syncData->timeSigDenom = 4;
56
57 connect(Engine::audioEngine(), &AudioEngine::sampleRateChanged, this, &VstSyncController::updateSampleRate);
58 updateSampleRate();
59}

63void VstSyncController::setAbsolutePosition(double ticks)
64{
65 if (!m_syncData) { return; }
66
67#ifdef VST_SNC_LATENCY
68 m_syncData->ppqPos = ((ticks + 0) / 48.0) - m_syncData->latency;
69#else
70 m_syncData->ppqPos = ((ticks + 0) / 48.0);
71#endif
72}

138void VstSyncController::update()
139{
140 if (!m_syncData) { return; }
141
142 m_syncData->bufferSize = Engine::audioEngine()->framesPerPeriod();
143
144#ifdef VST_SNC_LATENCY
145 m_syncData->latency = m_syncData->bufferSize * m_syncData->bpm / (static_cast<float>(m_syncData->sampleRate) * 60);
146#endif
147}
