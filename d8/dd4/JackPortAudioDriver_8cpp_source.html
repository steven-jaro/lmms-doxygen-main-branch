54static bool name_to_num(const char* paDeviceName, char* entry)
55{
56 if (const char* sep = strstr(paDeviceName, " (hw:"))
57 {
58 sep += 5;
59 while (*sep != '\0' && *sep != ',' && *sep != ')')
60 *entry++ = *sep++;

78int JackPortAudioDriver::Render(const void* inputBuffer, void* outputBuffer, PaStreamCallbackFlags statusFlags)
79{
80 fInputBuffer = (jack_default_audio_sample_t**)inputBuffer;
81 fOutputBuffer = (jack_default_audio_sample_t**)outputBuffer;
82
83 if (statusFlags) {
84 if (statusFlags & paOutputUnderflow)
85 jack_error("JackPortAudioDriver::Render paOutputUnderflow");
86 if (statusFlags & paInputUnderflow)
87 jack_error("JackPortAudioDriver::Render paInputUnderflow");
88 if (statusFlags & paOutputOverflow)
89 jack_error("JackPortAudioDriver::Render paOutputOverflow");
90 if (statusFlags & paInputOverflow)
91 jack_error("JackPortAudioDriver::Render paInputOverflow");
92 if (statusFlags & paPrimingOutput)
93 jack_error("JackPortAudioDriver::Render paOutputUnderflow");
94
95 if (statusFlags != paPrimingOutput) {
96 jack_time_t cur_time = GetMicroSeconds();
97 NotifyXRun(cur_time, float(cur_time - fBeginDateUst)); // Better this value than nothing...
98 }
99 }
100
101 // Setup threaded based log function
102 set_threaded_log_function();
103 CycleTakeBeginTime();
104 return (Process() == 0) ? paContinue : paAbort;
105}

123PaError JackPortAudioDriver::OpenStream(jack_nframes_t buffer_size)
124{
125 PaStreamParameters inputParameters;
126 PaStreamParameters outputParameters;
127
128 jack_log("JackPortAudioDriver::OpenStream buffer_size = %d", buffer_size);
129
130 // Update parameters
131 inputParameters.device = fInputDevice;
132 inputParameters.channelCount = fCaptureChannels;
133 inputParameters.sampleFormat = paFloat32 | paNonInterleaved; // 32 bit floating point input
134 inputParameters.suggestedLatency = (fInputDevice != paNoDevice) // TODO: check how to setup this on ASIO
135 ? ((fPaDevices->GetHostFromDevice(fInputDevice) == "ASIO") ? 0 : Pa_GetDeviceInfo(inputParameters.device)->defaultLowInputLatency)
136 : 0;
137 inputParameters.hostApiSpecificStreamInfo = NULL;
138
139 outputParameters.device = fOutputDevice;
140 outputParameters.channelCount = fPlaybackChannels;
141 outputParameters.sampleFormat = paFloat32 | paNonInterleaved; // 32 bit floating point output
142 outputParameters.suggestedLatency = (fOutputDevice != paNoDevice) // TODO: check how to setup this on ASIO
143 ? ((fPaDevices->GetHostFromDevice(fOutputDevice) == "ASIO") ? 0 : Pa_GetDeviceInfo(outputParameters.device)->defaultLowOutputLatency)
144 : 0;
145 outputParameters.hostApiSpecificStreamInfo = NULL;
146
147 return Pa_OpenStream(&fStream,
148 (fInputDevice == paNoDevice) ? 0 : &inputParameters,
149 (fOutputDevice == paNoDevice) ? 0 : &outputParameters,
150 fEngineControl->fSampleRate,
151 buffer_size,
152 paNoFlag, // Clipping is on...
153 Render,
154 this);
155}

157void JackPortAudioDriver::UpdateLatencies()
158{
159 jack_latency_range_t input_range;
160 jack_latency_range_t output_range;
161 jack_latency_range_t monitor_range;
162
163 const PaStreamInfo* info = Pa_GetStreamInfo(fStream);
164 assert(info);
165
166 for (int i = 0; i < fCaptureChannels; i++) {
167 input_range.max = input_range.min = fEngineControl->fBufferSize + (info->inputLatency * fEngineControl->fSampleRate) + fCaptureLatency;
168 fGraphManager->GetPort(fCapturePortList[i])->SetLatencyRange(JackCaptureLatency, &input_range);
169 }
170
171 for (int i = 0; i < fPlaybackChannels; i++) {
172 output_range.max = output_range.min = (info->outputLatency * fEngineControl->fSampleRate) + fPlaybackLatency;
173 if (fEngineControl->fSyncMode) {
174 output_range.max = output_range.min += fEngineControl->fBufferSize;
175 } else {
176 output_range.max = output_range.min += fEngineControl->fBufferSize * 2;
177 }
178 fGraphManager->GetPort(fPlaybackPortList[i])->SetLatencyRange(JackPlaybackLatency, &output_range);
179 if (fWithMonitorPorts) {
180 monitor_range.min = monitor_range.max = fEngineControl->fBufferSize;
181 fGraphManager->GetPort(fMonitorPortList[i])->SetLatencyRange(JackCaptureLatency, &monitor_range);
182 }
183 }
184}

186int JackPortAudioDriver::Open(jack_nframes_t buffer_size,
187 jack_nframes_t samplerate,
188 bool capturing,
189 bool playing,
190 int inchannels,
191 int outchannels,
192 bool monitor,
193 const char* capture_driver_uid,
194 const char* playback_driver_uid,
195 jack_nframes_t capture_latency,
196 jack_nframes_t playback_latency)
197{
198 int in_max = 0;
199 int out_max = 0;
200 PaError err = paNoError;
201
202 if (!fPaDevices) {
203 fPaDevices = new PortAudioDevices();
204 }
205
206 fCaptureLatency = capture_latency;
207 fPlaybackLatency = playback_latency;
208
209 jack_log("JackPortAudioDriver::Open nframes = %ld in = %ld out = %ld capture name = %s playback name = %s samplerate = %ld",
210 buffer_size, inchannels, outchannels, capture_driver_uid, playback_driver_uid, samplerate);
211
212 // Get devices
213 if (capturing) {
214 if (fPaDevices->GetInputDeviceFromName(capture_driver_uid, fInputDevice, in_max) < 0) {
215 goto error;
216 }
217 }
218 if (playing) {
219 if (fPaDevices->GetOutputDeviceFromName(playback_driver_uid, fOutputDevice, out_max) < 0) {
220 goto error;
221 }
222 }
223
224 // If ASIO, request for preferred size (assuming fInputDevice and fOutputDevice are the same)
225 if (buffer_size == 0) {
226 buffer_size = fPaDevices->GetPreferredBufferSize(fInputDevice);
227 jack_log("JackPortAudioDriver::Open preferred buffer_size = %d", buffer_size);
228 }
229
230 // Generic JackAudioDriver Open
231 char capture_driver_name[JACK_CLIENT_NAME_SIZE];
232 char playback_driver_name[JACK_CLIENT_NAME_SIZE];
233 snprintf(capture_driver_name, sizeof(capture_driver_name), "%s", capture_driver_uid);
234 snprintf(playback_driver_name, sizeof(playback_driver_name), "%s", playback_driver_uid);
235 if (JackAudioDriver::Open(buffer_size, samplerate, capturing, playing, inchannels, outchannels, monitor,
236 capture_driver_name, playback_driver_name, capture_latency, playback_latency) != 0) {
237 return -1;
238 }
239
240 jack_log("JackPortAudioDriver::Open fInputDevice = %d, fOutputDevice %d", fInputDevice, fOutputDevice);
241
242 // Default channels number required
243 if (inchannels == 0) {
244 jack_log("JackPortAudioDriver::Open setup max in channels = %ld", in_max);
245 inchannels = in_max;
246 }
247 if (outchannels == 0) {
248 jack_log("JackPortAudioDriver::Open setup max out channels = %ld", out_max);
249 outchannels = out_max;
250 }
251
252 // Too many channels required, take max available
253 if (inchannels > in_max) {
254 jack_error("This device has only %d available input channels.", in_max);
255 inchannels = in_max;
256 }
257 if (outchannels > out_max) {
258 jack_error("This device has only %d available output channels.", out_max);
259 outchannels = out_max;
260 }
261
262 // Core driver may have changed the in/out values
263 fCaptureChannels = inchannels;
264 fPlaybackChannels = outchannels;
265
266#ifdef __linux__
267 if (JackServerGlobals::on_device_acquire != NULL) {
268 char audio_name[32];
269 snprintf(audio_name, sizeof(audio_name), "Audio");
270
271 if (name_to_num(capture_driver_name, audio_name+5)) {
272 if (!JackServerGlobals::on_device_acquire(audio_name)) {
273 jack_error("Audio device %s cannot be acquired...", capture_driver_name);
274 return -1;
275 }
276 }
277
278 if (strcmp(capture_driver_name, playback_driver_name) && name_to_num(playback_driver_name, audio_name+5)) {
279 if (!JackServerGlobals::on_device_acquire(audio_name)) {
280 jack_error("Audio device %s cannot be acquired...", playback_driver_name);
281 return -1;
282 }
283 }
284 }
285#endif
286
287 err = OpenStream(buffer_size);
288 if (err != paNoError) {
289 jack_error("Pa_OpenStream error = %s", Pa_GetErrorText(err));
290 goto error;
291 }
292
293#ifdef __APPLE__
294 fEngineControl->fPeriod = fEngineControl->fPeriodUsecs * 1000;
295 fEngineControl->fComputation = JackTools::ComputationMicroSec(fEngineControl->fBufferSize) * 1000;
296 fEngineControl->fConstraint = fEngineControl->fPeriodUsecs * 1000;
297#endif
298
299#ifdef __linux__
300 if (JackServerGlobals::on_device_reservation_loop != NULL) {
301 device_reservation_loop_running = true;
302 if (JackPosixThread::StartImp(&fReservationLoopThread, 0, 0, on_device_reservation_loop, NULL) != 0) {
303 device_reservation_loop_running = false;
304 }
305 }
306#endif
307
308 return 0;
309
310error:
311 JackAudioDriver::Close();
312 jack_error("Can't open default PortAudio device");
313 return -1;
314}

316int JackPortAudioDriver::Close()
317{
318 // Generic audio driver close
319 jack_log("JackPortAudioDriver::Close");
320 JackAudioDriver::Close();
321 PaError err = Pa_CloseStream(fStream);
322 if (err != paNoError) {
323 jack_error("Pa_CloseStream error = %s", Pa_GetErrorText(err));
324 }
325
326#ifdef __linux__
327 if (device_reservation_loop_running) {
328 device_reservation_loop_running = false;
329 JackPosixThread::StopImp(fReservationLoopThread);
330 }
331
332 if (JackServerGlobals::on_device_release != NULL)
333 {
334 char audio_name[32];
335 snprintf(audio_name, sizeof(audio_name), "Audio");
336
337 if (name_to_num(fCaptureDriverName, audio_name+5)) {
338 JackServerGlobals::on_device_release(audio_name);
339 }
340
341 if (strcmp(fCaptureDriverName, fPlaybackDriverName) && name_to_num(fPlaybackDriverName, audio_name+5)) {
342 JackServerGlobals::on_device_release(audio_name);
343 }
344 }
345#endif
346
347 delete fPaDevices;
348 fPaDevices = NULL;
349 return (err != paNoError) ? -1 : 0;
350}

485 SERVER_EXPORT Jack::JackDriverClientInterface* driver_initialize(Jack::JackLockedEngine* engine, Jack::JackSynchro* table, const JSList* params)
486 {
487 jack_nframes_t srate = 44100;
488 jack_nframes_t frames_per_interrupt = 512;
489 const char* capture_pcm_name = "";
490 const char* playback_pcm_name = "";
491 bool capture = false;
492 bool playback = false;
493 int chan_in = 0;
494 int chan_out = 0;
495 bool monitor = false;
496 const JSList *node;
497 const jack_driver_param_t *param;
498 jack_nframes_t systemic_input_latency = 0;
499 jack_nframes_t systemic_output_latency = 0;
500 PortAudioDevices* pa_devices = new PortAudioDevices();
501
502 for (node = params; node; node = jack_slist_next(node))
503 {
504 param = (const jack_driver_param_t *) node->data;
505
506 switch (param->character) {
507
508 case 'd':
509 capture_pcm_name = param->value.str;
510 playback_pcm_name = param->value.str;
511 break;
512
513 case 'D':
514 capture = true;
515 playback = true;
516 break;
517
518 case 'c':
519 chan_in = chan_out = (int)param->value.ui;
520 break;
521
522 case 'i':
523 chan_in = (int)param->value.ui;
524 break;
525
526 case 'o':
527 chan_out = (int)param->value.ui;
528 break;
529
530 case 'C':
531 capture = true;
532 if (strcmp(param->value.str, "none") != 0) {
533 capture_pcm_name = param->value.str;
534 }
535 break;
536
537 case 'P':
538 playback = true;
539 if (strcmp(param->value.str, "none") != 0) {
540 playback_pcm_name = param->value.str;
541 }
542 break;
543
544 case 'm':
545 monitor = param->value.i;
546 break;
547
548 case 'r':
549 srate = param->value.ui;
550 break;
551
552 case 'p':
553 frames_per_interrupt = (unsigned int)param->value.ui;
554 break;
555
556 case 'I':
557 systemic_input_latency = param->value.ui;
558 break;
559
560 case 'O':
561 systemic_output_latency = param->value.ui;
562 break;
563
564 case 'l':
565 pa_devices->DisplayDevicesNames();
566 // Stops the server in this case
567 return NULL;
568 }
569 }
570
571 // duplex is the default
572 if (!capture && !playback) {
573 capture = true;
574 playback = true;
575 }
576
577 Jack::JackDriverClientInterface* driver = new Jack::JackPortAudioDriver("system", "portaudio", engine, table, pa_devices);
578 if (driver->Open(frames_per_interrupt, srate, capture, playback,
579 chan_in, chan_out, monitor, capture_pcm_name,
580 playback_pcm_name, systemic_input_latency,
581 systemic_output_latency) == 0) {
582 return driver;
583 } else {
584 delete driver;
585 return NULL;
586 }
587 }
