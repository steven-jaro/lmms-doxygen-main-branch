67void PeakIndicator::updatePeakDisplay()
68{
69 // Treat everything below -144 dbFS as -inf. Otherwise some residual signals show up
70 // in the form of very small dbFS values, e.g. -857.1 dbFS.
71 // TODO Make the threshold configurable in the settings?
72 if (m_peak <= dbfsToAmp(-144.))
73 {
74 setText(tr("-inf"));
75 }
76 else
77 {
78 auto dbfs = ampToDbfs(m_peak);
79 setText(QString::number(dbfs, 'f', 1));
80 }
81}
