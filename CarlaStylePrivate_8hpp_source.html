114 QColor outline(const QPalette& pal) const
115 {
116 if (! pal.window().texture().isNull())
117 return QColor(0, 0, 0, 160);
118 const QColor& col = qt_palette_bg_color(pal);
119 return col.blackF() > 0.4 ? col.lighter(160) : col.darker(140);
120 }
121
122 QColor highlightedOutline(const QPalette &pal) const
123 {
124 QColor highlightedOutline = highlight(pal).darker(125);
125 if (highlightedOutline.value() > 160)
126 highlightedOutline.setHsl(highlightedOutline.hue(), highlightedOutline.saturation(), 160);
127 return highlightedOutline;
128 }
129
130 QColor tabFrameColor(const QPalette& pal) const
131 {
132 if (! pal.button().texture().isNull())
133 return QColor(255, 255, 255, 8);
134 return buttonColor(pal).lighter(104);
135 }
136
137 QColor buttonColor(const QPalette& pal) const
138 {
139 QColor buttonColor = pal.button().color();
140 const int val = qGray(buttonColor.rgb());
141 buttonColor = buttonColor.lighter(100 + qMax(1, (180 - val)/6));
142 buttonColor.setHsv(buttonColor.hue(), buttonColor.saturation() * 0.75, buttonColor.value());
143 return buttonColor;
144 }

158 void startAnimation(CarlaStyleAnimation* animation) const
159 {
160 stopAnimation(animation->target());
161 fStyle->connect(animation, SIGNAL(destroyed()), SLOT(_removeAnimation()), Qt::UniqueConnection);
162 fAnimations.insert(animation->target(), animation);
163 animation->start();
164 }
165
166 void stopAnimation(const QObject* target) const
167 {
168 CarlaStyleAnimation* const animation = fAnimations.take(target);
169 if (animation != nullptr && animation->state() != QAbstractAnimation::Stopped)
170 animation->stop();
171 }
