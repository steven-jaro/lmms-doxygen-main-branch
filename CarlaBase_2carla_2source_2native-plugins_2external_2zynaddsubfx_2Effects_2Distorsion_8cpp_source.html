138//Effect output
139void Distorsion::out(const Stereo<float *> &smp)
140{
141 float inputvol = powf(5.0f, (Pdrive - 32.0f) / 127.0f);
142 if(Pnegate)
143 inputvol *= -1.0f;
144
145 if(Pstereo) //Stereo
146 for(int i = 0; i < buffersize; ++i) {
147 efxoutl[i] = smp.l[i] * inputvol * pangainL;
148 efxoutr[i] = smp.r[i] * inputvol * pangainR;
149 }
150 else //Mono
151 for(int i = 0; i < buffersize; ++i)
152 efxoutl[i] = (smp.l[i] * pangainL + smp.r[i] * pangainR) * inputvol;
153
154 if(Pprefiltering)
155 applyfilters(efxoutl, efxoutr);
156
157 waveShapeSmps(buffersize, efxoutl, Ptype + 1, Pdrive);
158 if(Pstereo)
159 waveShapeSmps(buffersize, efxoutr, Ptype + 1, Pdrive);
160
161 if(!Pprefiltering)
162 applyfilters(efxoutl, efxoutr);
163
164 if(!Pstereo)
165 memcpy(efxoutr, efxoutl, bufferbytes);
166
167 float level = dB2rap(60.0f * Plevel / 127.0f - 40.0f);
168 for(int i = 0; i < buffersize; ++i) {
169 float lout = efxoutl[i];
170 float rout = efxoutr[i];
171 float l = lout * (1.0f - lrcross) + rout * lrcross;
172 float r = rout * (1.0f - lrcross) + lout * lrcross;
173 lout = l;
174 rout = r;
175
176 efxoutl[i] = lout * 2.0f * level;
177 efxoutr[i] = rout * 2.0f * level;
178 }
179}

182//Parameter control
183void Distorsion::setvolume(unsigned char _Pvolume)
184{
185 Pvolume = _Pvolume;
186
187 if(insertion == 0) {
188 outvolume = powf(0.01f, (1.0f - Pvolume / 127.0f)) * 4.0f;
189 volume = 1.0f;
190 }
191 else
192 volume = outvolume = Pvolume / 127.0f;
193 if(Pvolume == 0)
194 cleanup();
195}

214void Distorsion::setpreset(unsigned char npreset)
215{
216 const int PRESET_SIZE = 11;
217 const int NUM_PRESETS = 6;
218 unsigned char presets[NUM_PRESETS][PRESET_SIZE] = {
219 //Overdrive 1
220 {127, 64, 35, 56, 70, 0, 0, 96, 0, 0, 0},
221 //Overdrive 2
222 {127, 64, 35, 29, 75, 1, 0, 127, 0, 0, 0},
223 //A. Exciter 1
224 {64, 64, 35, 75, 80, 5, 0, 127, 105, 1, 0},
225 //A. Exciter 2
226 {64, 64, 35, 85, 62, 1, 0, 127, 118, 1, 0},
227 //Guitar Amp
228 {127, 64, 35, 63, 75, 2, 0, 55, 0, 0, 0},
229 //Quantisize
230 {127, 64, 35, 88, 75, 4, 0, 127, 0, 1, 0}
231 };
232
233 if(npreset >= NUM_PRESETS)
234 npreset = NUM_PRESETS - 1;
235 for(int n = 0; n < PRESET_SIZE; ++n)
236 changepar(n, presets[npreset][n]);
237 if(!insertion) //lower the volume if this is system effect
238 changepar(0, (int) (presets[npreset][0] / 1.5f));
239 Ppreset = npreset;
240 cleanup();
241}

244void Distorsion::changepar(int npar, unsigned char value)
245{
246 switch(npar) {
247 case 0:
248 setvolume(value);
249 break;
250 case 1:
251 setpanning(value);
252 break;
253 case 2:
254 setlrcross(value);
255 break;
256 case 3:
257 Pdrive = value;
258 break;
259 case 4:
260 Plevel = value;
261 break;
262 case 5:
263 if(value > 13)
264 Ptype = 13; //this must be increased if more distorsion types are added
265 else
266 Ptype = value;
267 break;
268 case 6:
269 if(value > 1)
270 Pnegate = 1;
271 else
272 Pnegate = value;
273 break;
274 case 7:
275 setlpf(value);
276 break;
277 case 8:
278 sethpf(value);
279 break;
280 case 9:
281 Pstereo = (value > 1) ? 1 : value;
282 break;
283 case 10:
284 Pprefiltering = value;
285 break;
286 }
287}
