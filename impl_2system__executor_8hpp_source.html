41template <typename Function, typename Allocator>
42void system_executor::post(
43 ASIO_MOVE_ARG(Function) f, const Allocator& a) const
44{
45 typedef typename decay<Function>::type function_type;
46
47 system_context& ctx = detail::global<system_context>();
48
49 // Allocate and construct an operation to wrap the function.
50 typedef detail::executor_op<function_type, Allocator> op;
51 typename op::ptr p = { detail::addressof(a), op::ptr::allocate(a), 0 };
52 p.p = new (p.v) op(ASIO_MOVE_CAST(Function)(f), a);
53
54 ASIO_HANDLER_CREATION((ctx, *p.p,
55 "system_executor", &this->context(), 0, "post"));
56
57 ctx.scheduler_.post_immediate_completion(p.p, false);
58 p.v = p.p = 0;
59}
