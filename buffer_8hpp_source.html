544 || is_same<BufferSequence, const_buffers_1>::value

556template <typename Iterator>
557inline std::size_t buffer_size(multiple_buffers,
558 Iterator begin, Iterator end) ASIO_NOEXCEPT
559{
560 std::size_t total_buffer_size = 0;
561
562 Iterator iter = begin;
563 for (; iter != end; ++iter)
564 {
565 const_buffer b(*iter);
566 total_buffer_size += b.size();
567 }
