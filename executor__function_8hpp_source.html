56template <typename Function, typename Alloc>
57class executor_function : public executor_function_base

59public:
60 ASIO_DEFINE_TAGGED_HANDLER_ALLOCATOR_PTR(

62
63 template <typename F>
64 executor_function(ASIO_MOVE_ARG(F) f, const Alloc& allocator)
65 : executor_function_base(&executor_function::do_complete),
66 function_(ASIO_MOVE_CAST(F)(f)),
67 allocator_(allocator)
68 {
69 }
70
71 static void do_complete(executor_function_base* base, bool call)
72 {
73 // Take ownership of the function object.
74 executor_function* o(static_cast<executor_function*>(base));
75 Alloc allocator(o->allocator_);
76 ptr p = { detail::addressof(allocator), o, o };
77
78 // Make a copy of the function so that the memory can be deallocated before
79 // the upcall is made. Even if we're not about to make an upcall, a
80 // sub-object of the function may be the true owner of the memory
81 // associated with the function. Consequently, a local copy of the function
82 // is required to ensure that any owning sub-object remains valid until
83 // after we have deallocated the memory here.
84 Function function(ASIO_MOVE_CAST(Function)(o->function_));
85 p.reset();
86
87 // Make the upcall if required.
88 if (call)
89 {
90 function();
91 }
92 }
93
94private:
95 Function function_;
96 Alloc allocator_;
97};
