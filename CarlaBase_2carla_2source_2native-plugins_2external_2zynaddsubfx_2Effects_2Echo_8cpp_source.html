104//Initialize the delays
105void Echo::initdelays(void)
106{
107 cleanup();
108 //number of seconds to delay left chan
109 float dl = avgDelay - lrdelay;
110
111 //number of seconds to delay right chan
112 float dr = avgDelay + lrdelay;
113
114 ndelta.l = max(1, (int) (dl * samplerate));
115 ndelta.r = max(1, (int) (dr * samplerate));
116 delta = ndelta;
117}

119//Effect output
120void Echo::out(const Stereo<float *> &input)
121{
122 for(int i = 0; i < buffersize; ++i) {
123 float ldl = delay.l[pos.l];
124 float rdl = delay.r[pos.r];
125 ldl = ldl * (1.0f - lrcross) + rdl * lrcross;
126 rdl = rdl * (1.0f - lrcross) + ldl * lrcross;
127
128 efxoutl[i] = ldl * 2.0f;
129 efxoutr[i] = rdl * 2.0f;
130
131 ldl = input.l[i] * pangainL - ldl * fb;
132 rdl = input.r[i] * pangainR - rdl * fb;
133
134 //LowPass Filter
135 old.l = delay.l[(pos.l + delta.l) % (MAX_DELAY * samplerate)] =
136 ldl * hidamp + old.l * (1.0f - hidamp);
137 old.r = delay.r[(pos.r + delta.r) % (MAX_DELAY * samplerate)] =
138 rdl * hidamp + old.r * (1.0f - hidamp);
139
140 //increment
141 ++pos.l; // += delta.l;
142 ++pos.r; // += delta.r;
143
144 //ensure that pos is still in bounds
145 pos.l %= MAX_DELAY * samplerate;
146 pos.r %= MAX_DELAY * samplerate;
147
148 //adjust delay if needed
149 delta.l = (15 * delta.l + ndelta.l) / 16;
150 delta.r = (15 * delta.r + ndelta.r) / 16;
151 }
152}

155//Parameter control
156void Echo::setvolume(unsigned char _Pvolume)
157{
158 Pvolume = _Pvolume;
159
160 if(insertion == 0) {
161 if (Pvolume == 0) {
162 outvolume = 0.0f;
163 } else {
164 outvolume = powf(0.01f, (1.0f - Pvolume / 127.0f)) * 4.0f;
165 }
166 volume = 1.0f;
167 }
168 else
169 volume = outvolume = Pvolume / 127.0f;
170 if(Pvolume == 0)
171 cleanup();
172}

181void Echo::setlrdelay(unsigned char _Plrdelay)
182{
183 float tmp;
184 Plrdelay = _Plrdelay;
185 tmp =
186 (powf(2.0f, fabsf(Plrdelay - 64.0f) / 64.0f * 9.0f) - 1.0f) / 1000.0f;
187 if(Plrdelay < 64.0f)
188 tmp = -tmp;
189 lrdelay = tmp;
190 initdelays();
191}

205void Echo::setpreset(unsigned char npreset)
206{
207 const int PRESET_SIZE = 7;
208 const int NUM_PRESETS = 9;
209 unsigned char presets[NUM_PRESETS][PRESET_SIZE] = {
210 {67, 64, 35, 64, 30, 59, 0 }, //Echo 1
211 {67, 64, 21, 64, 30, 59, 0 }, //Echo 2
212 {67, 75, 60, 64, 30, 59, 10}, //Echo 3
213 {67, 60, 44, 64, 30, 0, 0 }, //Simple Echo
214 {67, 60, 102, 50, 30, 82, 48}, //Canyon
215 {67, 64, 44, 17, 0, 82, 24}, //Panning Echo 1
216 {81, 60, 46, 118, 100, 68, 18}, //Panning Echo 2
217 {81, 60, 26, 100, 127, 67, 36}, //Panning Echo 3
218 {62, 64, 28, 64, 100, 90, 55} //Feedback Echo
219 };
220
221 if(npreset >= NUM_PRESETS)
222 npreset = NUM_PRESETS - 1;
223 for(int n = 0; n < PRESET_SIZE; ++n)
224 changepar(n, presets[npreset][n]);
225 if(insertion)
226 setvolume(presets[npreset][0] / 2); //lower the volume if this is insertion effect
227 Ppreset = npreset;
228}
