36/* base class, prepares common parameters */
37class RBJ
38{
39 public:
40 double Q, alpha, sin, cos;
41 double a[3], b[3];
42
43 public:
44 RBJ (double f, double _Q)
45 {
46 Q = _Q;
47
48 double w = 2 * M_PI * f;
49
50 sin = ::sin (w);
51 cos = ::cos (w);
52
53 alpha = sin / (2 * Q);
54 }
55
56 /* templated so we can set double and float coefficients from the same
57 * piece of code */
58 template <class T>
59 void make_direct_I (T * ca, T * cb)
60 {
61 double a0i = 1 / a[0];
62
63 ca[0] = b[0] * a0i;
64 ca[1] = b[1] * a0i;
65 ca[2] = b[2] * a0i;
66
67 /* our bi-quad implementation /adds/ b[i] * y[i] so we need to
68 * toggle the sign for the b[] coefficients.
69 */
70 cb[0] = 0;
71 cb[1] = -a[1] * a0i;
72 cb[2] = -a[2] * a0i;
73 }
74};

79class LP
80: public RBJ

83 LP (double f, double Q, BiQuad & bq) : RBJ (f, Q)
84 { ab (bq.a, bq.b); }

88 { ab (ca, cb); }
89
90 template <class T>
91 void ab (T * ca, T * cb)
92 {
93 b[0] = (1 - cos) * .5;
94 b[1] = (1 - cos);
95 b[2] = (1 - cos) * .5;
96
97 a[0] = 1 + alpha;
98 a[1] = -2 * cos;
99 a[2] = 1 - alpha;
100
101 make_direct_I (ca, cb);
102 }
103};

185class PeakShelve
186: public RBJ
187{
188 public:
189 double A, beta;
190
191 public:
192 PeakShelve (double f, double Q, double dB)
193 : RBJ (f, Q)
194 {
195 A = pow (10, dB * .025);
196 double S = Q; /* slope */
197 beta = sqrt ((A * A + 1) / S - (A - 1) * (A - 1));
198 }
199};
200
201class LoShelve
202: public PeakShelve
203{
204 public:
205 LoShelve (double f, double Q, double dB, BiQuad & bq)
206 : PeakShelve (f, Q, dB)
207 { ab (bq.a, bq.b); }
208
209 template <class T>
210 LoShelve (double f, double Q, double dB, T * ca, T * cb)
211 : PeakShelve (f, Q, dB)
212 { ab (ca, cb); }
213
214 template <class T>
215 void ab (T * ca, T * cb)
216 {
217 double Ap1 = A + 1, Am1 = A - 1;
218 double beta_sin = beta * sin;
219
220 b[0] = A * (Ap1 - Am1 * cos + beta_sin);
221 b[1] = 2 * A * (Am1 - Ap1 * cos);
222 b[2] = A * (Ap1 - Am1 * cos - beta_sin);
223
224 a[0] = Ap1 + Am1 * cos + beta_sin;
225 a[1] = -2 * (Am1 + Ap1 * cos);
226 a[2] = Ap1 + Am1 * cos - beta_sin;
227
228 make_direct_I (ca, cb);
229 }
230};
