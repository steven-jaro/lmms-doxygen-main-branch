63void TempoSyncBarModelEditor::modelChanged()
64{
65 TempoSyncKnobModel * tempoSyncModel = model();
66
67 if(tempoSyncModel == nullptr)
68 {
69 qWarning("no TempoSyncKnobModel has been set!");
70 }
71
72 if(m_custom != nullptr)
73 {
74 m_custom->setModel(&tempoSyncModel->getCustomMeterModel());
75 }
76
77 connect(tempoSyncModel, &TempoSyncKnobModel::syncModeChanged, this, &TempoSyncBarModelEditor::updateDescAndIcon);
78 connect(this, SIGNAL(sliderMoved(float)), tempoSyncModel, SLOT(disableSync()));
79
80 updateDescAndIcon();
81}

84void TempoSyncBarModelEditor::contextMenuEvent(QContextMenuEvent *)
85{
86 mouseReleaseEvent(nullptr);
87
88 TempoSyncKnobModel * tempoSyncModel = model();
89
90 CaptionMenu contextMenu(tempoSyncModel->displayName(), this);
91 addDefaultActions(&contextMenu);
92 contextMenu.addAction(QPixmap(),
93 model()->isScaleLogarithmic() ? tr("Set linear") : tr("Set logarithmic"),
94 this, SLOT(toggleScale()));
95
96 contextMenu.addSeparator();
97
98 float limit = 60000.0f / (Engine::getSong()->getTempo() * tempoSyncModel->scale());
99
100 QMenu * syncMenu = contextMenu.addMenu(m_tempoSyncIcon, m_tempoSyncDescription);
101
102 float const maxValue = tempoSyncModel->maxValue();
103
104 if(limit / 8.0f <= maxValue)
105 {
106 connect(syncMenu, SIGNAL(triggered(QAction*)), tempoSyncModel, SLOT(setTempoSync(QAction*)));
107
108 syncMenu->addAction(embed::getIconPixmap("note_none"),
109 tr("No Sync"))->setData((int) TempoSyncKnobModel::SyncMode::None);
110
111 if(limit / 0.125f <= maxValue)
112 {
113 syncMenu->addAction(embed::getIconPixmap("note_double_whole"),
114 tr("Eight beats"))->setData((int) TempoSyncKnobModel::SyncMode::DoubleWholeNote);
115 }
116
117 if(limit / 0.25f <= maxValue)
118 {
119 syncMenu->addAction(embed::getIconPixmap("note_whole"),
120 tr("Whole note"))->setData((int) TempoSyncKnobModel::SyncMode::WholeNote);
121 }
122
123 if(limit / 0.5f <= maxValue)
124 {
125 syncMenu->addAction(embed::getIconPixmap("note_half"),
126 tr("Half note"))->setData((int) TempoSyncKnobModel::SyncMode::HalfNote);
127 }
128
129 if(limit <= maxValue)
130 {
131 syncMenu->addAction(embed::getIconPixmap("note_quarter"),
132 tr("Quarter note"))->setData((int) TempoSyncKnobModel::SyncMode::QuarterNote);
133 }
134
135 if(limit / 2.0f <= maxValue)
136 {
137 syncMenu->addAction(embed::getIconPixmap("note_eighth"),
138 tr("8th note"))->setData((int) TempoSyncKnobModel::SyncMode::EighthNote);
139 }
140
141 if(limit / 4.0f <= maxValue)
142 {
143 syncMenu->addAction(embed::getIconPixmap("note_sixteenth"),
144 tr("16th note"))->setData((int) TempoSyncKnobModel::SyncMode::SixteenthNote);
145 }
146
147 syncMenu->addAction(embed::getIconPixmap("note_thirtysecond"),
148 tr("32nd note"))->setData((int) TempoSyncKnobModel::SyncMode::ThirtysecondNote);
149
150 syncMenu->addAction(embed::getIconPixmap("dont_know"),
151 tr("Custom..."), this, SLOT(showCustom()))->setData((int) TempoSyncKnobModel::SyncMode::Custom);
152
153 contextMenu.addSeparator();
154 }
155
156 contextMenu.exec(QCursor::pos());
157
158 delete syncMenu;
159}

203void TempoSyncBarModelEditor::showCustom()
204{
205 if(m_custom == nullptr)
206 {
207 m_custom = new MeterDialog(getGUI()->mainWindow()->workspace());
208 QMdiSubWindow * subWindow = getGUI()->mainWindow()->addWindowedWidget(m_custom);
209 Qt::WindowFlags flags = subWindow->windowFlags();
210 flags &= ~Qt::WindowMaximizeButtonHint;
211 subWindow->setWindowFlags(flags);
212 subWindow->setFixedSize(subWindow->size());
213 m_custom->setWindowTitle("Meter");
214 m_custom->setModel(&model()->getCustomMeterModel());
215 }
216
217 m_custom->parentWidget()->show();
218 model()->setTempoSync(TempoSyncKnobModel::SyncMode::Custom);
219}

222void TempoSyncBarModelEditor::updateTextDescription()
223{
224 TempoSyncKnobModel * tempoSyncModel = model();
225
226 auto const syncMode = tempoSyncModel->syncMode();
227
228 switch(syncMode)
229 {
230 case TempoSyncKnobModel::SyncMode::None:
231 m_tempoSyncDescription = tr("Tempo Sync");
232 break;
233 case TempoSyncKnobModel::SyncMode::Custom:
234 m_tempoSyncDescription = tr("Custom ") +
235 "(" +
236 QString::number(tempoSyncModel->getCustomMeterModel().numeratorModel().value()) +
237 "/" +
238 QString::number(tempoSyncModel->getCustomMeterModel().denominatorModel().value()) +
239 ")";
240 break;
241 case TempoSyncKnobModel::SyncMode::DoubleWholeNote:
242 m_tempoSyncDescription = tr("Synced to Eight Beats");
243 break;
244 case TempoSyncKnobModel::SyncMode::WholeNote:
245 m_tempoSyncDescription = tr("Synced to Whole Note");
246 break;
247 case TempoSyncKnobModel::SyncMode::HalfNote:
248 m_tempoSyncDescription = tr("Synced to Half Note");
249 break;
250 case TempoSyncKnobModel::SyncMode::QuarterNote:
251 m_tempoSyncDescription = tr("Synced to Quarter Note");
252 break;
253 case TempoSyncKnobModel::SyncMode::EighthNote:
254 m_tempoSyncDescription = tr("Synced to 8th Note");
255 break;
256 case TempoSyncKnobModel::SyncMode::SixteenthNote:
257 m_tempoSyncDescription = tr("Synced to 16th Note");
258 break;
259 case TempoSyncKnobModel::SyncMode::ThirtysecondNote:
260 m_tempoSyncDescription = tr("Synced to 32nd Note");
261 break;
262 default: ;
263 }
264}
