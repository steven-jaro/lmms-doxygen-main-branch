54/**
55 * Register (or unregister) as a slow-sync client, one that cannot
56 * respond immediately to transport position changes.
57 *
58 * The @a sync_callback will be invoked at the first available
59 * opportunity after its registration is complete. If the client is
60 * currently active this will be the following process cycle,
61 * otherwise it will be the first cycle after calling jack_activate().
62 * After that, it runs according to the ::JackSyncCallback rules.
63 * Clients that don't set a @a sync_callback are assumed to be ready
64 * immediately any time the transport wants to start.
65 *
66 * @param client the JACK client structure.
67 * @param sync_callback is a realtime function that returns TRUE when
68 * the client is ready. Setting @a sync_callback to NULL declares that
69 * this client no longer requires slow-sync processing.
70 * @param arg an argument for the @a sync_callback function.
71 *
72 * @return 0 on success, otherwise a non-zero error code.
73 */
74int jack_set_sync_callback (jack_client_t *client,
75 JackSyncCallback sync_callback,
76 void *arg) JACK_OPTIONAL_WEAK_EXPORT;

78/**
79 * Set the timeout value for slow-sync clients.
80 *
81 * This timeout prevents unresponsive slow-sync clients from
82 * completely halting the transport mechanism. The default is two
83 * seconds. When the timeout expires, the transport starts rolling,
84 * even if some slow-sync clients are still unready. The @a
85 * sync_callbacks of these clients continue being invoked, giving them
86 * a chance to catch up.
87 *
88 * @see jack_set_sync_callback
89 *
90 * @param client the JACK client structure.
91 * @param timeout is delay (in microseconds) before the timeout expires.
92 *
93 * @return 0 on success, otherwise a non-zero error code.
94 */
95int jack_set_sync_timeout (jack_client_t *client,
96 jack_time_t timeout) JACK_OPTIONAL_WEAK_EXPORT;

98/**
99 * Register as timebase master for the JACK subsystem.
100 *
101 * The timebase master registers a callback that updates extended
102 * position information such as beats or timecode whenever necessary.
103 * Without this extended information, there is no need for this
104 * function.
105 *
106 * There is never more than one master at a time. When a new client
107 * takes over, the former @a timebase_callback is no longer called.
108 * Taking over the timebase may be done conditionally, so it fails if
109 * there was a master already.
110 *
111 * @param client the JACK client structure.
112 * @param conditional non-zero for a conditional request.
113 * @param timebase_callback is a realtime function that returns
114 * position information.
115 * @param arg an argument for the @a timebase_callback function.
116 *
117 * @return
118 * - 0 on success;
119 * - EBUSY if a conditional request fails because there was already a
120 * timebase master;
121 * - other non-zero error code.
122 */
123int jack_set_timebase_callback (jack_client_t *client,
124 int conditional,
125 JackTimebaseCallback timebase_callback,
126 void *arg) JACK_OPTIONAL_WEAK_EXPORT;

128/**
129 * Reposition the transport to a new frame number.
130 *
131 * May be called at any time by any client. The new position takes
132 * effect in two process cycles. If there are slow-sync clients and
133 * the transport is already rolling, it will enter the
134 * ::JackTransportStarting state and begin invoking their @a
135 * sync_callbacks until ready. This function is realtime-safe.
136 *
137 * @see jack_transport_reposition, jack_set_sync_callback
138 *
139 * @param client the JACK client structure.
140 * @param frame frame number of new transport position.
141 *
142 * @return 0 if valid request, non-zero otherwise.
143 */
144int jack_transport_locate (jack_client_t *client,
