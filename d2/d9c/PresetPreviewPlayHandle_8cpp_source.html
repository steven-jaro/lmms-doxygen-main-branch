118PresetPreviewPlayHandle::PresetPreviewPlayHandle( const QString & _preset_file, bool _load_by_plugin, DataFile *dataFile ) :
119 PlayHandle( Type::PresetPreviewHandle ),
120 m_previewNote(nullptr)
121{
122 setUsesBuffer( false );
123
124 s_previewTC->lockData();
125
126 Engine::audioEngine()->requestChangeInModel();
127 s_previewTC->setPreviewNote( nullptr );
128 s_previewTC->previewInstrumentTrack()->silenceAllNotes();
129 Engine::audioEngine()->doneChangeInModel();
130
131 const bool j = Engine::projectJournal()->isJournalling();
132 Engine::projectJournal()->setJournalling( false );
133
134 if( _load_by_plugin )
135 {
136 Instrument * i = s_previewTC->previewInstrumentTrack()->instrument();
137 const QString ext = QFileInfo( _preset_file ).
138 suffix().toLower();
139 if( i == nullptr || !i->descriptor()->supportsFileType( ext ) )
140 {
141 const PluginFactory::PluginInfoAndKey& infoAndKey =
142 getPluginFactory()->pluginSupportingExtension(ext);
143 i = s_previewTC->previewInstrumentTrack()->
144 loadInstrument(infoAndKey.info.name(), &infoAndKey.key);
145 }
146 if( i != nullptr )
147 {
148 i->loadFile( _preset_file );
149 }
150 }
151 else
152 {
153 bool dataFileCreated = false;
154 if( dataFile == 0 )
155 {
156 dataFile = new DataFile( _preset_file );
157 dataFileCreated = true;
158 }
159
160 s_previewTC->previewInstrumentTrack()->loadTrackSpecificSettings(
161 dataFile->content().firstChild().toElement());
162
163 if( dataFileCreated )
164 {
165 delete dataFile;
166 }
167 }
168 dataFile = 0;
169 // make sure, our preset-preview-track does not appear in any MIDI-
170 // devices list, so just disable receiving/sending MIDI-events at all
171 s_previewTC->previewInstrumentTrack()->
172 midiPort()->setMode( MidiPort::Mode::Disabled );
173
174 Engine::audioEngine()->requestChangeInModel();
175 // create note-play-handle for it
176 m_previewNote = NotePlayHandleManager::acquire(
177 s_previewTC->previewInstrumentTrack(), 0,
178 std::numeric_limits<f_cnt_t>::max() / 2,
179 Note( 0, 0, DefaultKey, 100 ) );
180
181 setAudioBusHandle(s_previewTC->previewInstrumentTrack()->audioBusHandle());
182
183 s_previewTC->setPreviewNote( m_previewNote );
184
185 Engine::audioEngine()->addPlayHandle( m_previewNote );
186
187 Engine::audioEngine()->doneChangeInModel();
188 s_previewTC->unlockData();
189 Engine::projectJournal()->setJournalling( j );
190}

195PresetPreviewPlayHandle::~PresetPreviewPlayHandle()
196{
197 Engine::audioEngine()->requestChangeInModel();
198 // not muted by other preset-preview-handle?
199 if (s_previewTC->testAndSetPreviewNote(m_previewNote, nullptr))
200 {
201 m_previewNote->noteOff();
202 }
203 Engine::audioEngine()->doneChangeInModel();
204}
