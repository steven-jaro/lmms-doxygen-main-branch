83void PatternClipView::paintEvent(QPaintEvent*)
84{
85 QPainter painter( this );
86
87 if( !needsUpdate() )
88 {
89 painter.drawPixmap( 0, 0, m_paintPixmap );
90 return;
91 }
92
93 setNeedsUpdate( false );
94
95 if (m_paintPixmap.isNull() || m_paintPixmap.size() != size())
96 {
97 m_paintPixmap = QPixmap(size());
98 }
99
100 QPainter p( &m_paintPixmap );
101
102 QLinearGradient lingrad( 0, 0, 0, height() );
103 QColor c = getColorForDisplay( painter.background().color() );
104
105 lingrad.setColorAt( 0, c.lighter( 130 ) );
106 lingrad.setColorAt( 1, c.lighter( 70 ) );
107
108 // paint a black rectangle under the clip to prevent glitches with transparent backgrounds
109 p.fillRect( rect(), QColor( 0, 0, 0 ) );
110
111 if( gradient() )
112 {
113 p.fillRect( rect(), lingrad );
114 }
115 else
116 {
117 p.fillRect( rect(), c );
118 }
119
120 // bar lines
121 const int lineSize = 3;
122 int pixelsPerPattern = Engine::patternStore()->lengthOfPattern(m_patternClip->patternIndex()) * pixelsPerBar();
123 int offset = static_cast<int>(m_patternClip->startTimeOffset() * (pixelsPerBar() / TimePos::ticksPerBar()))
124 % pixelsPerPattern;
125 if (offset < 2) {
126 offset += pixelsPerPattern;
127 }
128
129 p.setPen( c.darker( 200 ) );
130
131 if (pixelsPerPattern > 0)
132 {
133 for (int x = offset; x < width() - 2; x += pixelsPerPattern)
134 {
135 p.drawLine( x, BORDER_WIDTH, x, BORDER_WIDTH + lineSize );
136 p.drawLine( x, rect().bottom() - ( BORDER_WIDTH + lineSize ),
137 x, rect().bottom() - BORDER_WIDTH );
138 }
139 }
140
141 // clip name
142 paintTextLabel(m_patternClip->name(), p);
143
144 // inner border
145 p.setPen( c.lighter( 130 ) );
146 p.drawRect( 1, 1, rect().right() - BORDER_WIDTH,
147 rect().bottom() - BORDER_WIDTH );
148
149 // outer border
150 p.setPen( c.darker( 300 ) );
151 p.drawRect( 0, 0, rect().right(), rect().bottom() );
152
153 // draw the 'muted' pixmap only if the clip was manually muted
154 if (m_patternClip->isMuted())
155 {
156 const int spacing = BORDER_WIDTH;
157 const int size = 14;
158 p.drawPixmap( spacing, height() - ( size + spacing ),
159 embed::getIconPixmap( "muted", size, size ) );
160 }
161
162 if (m_marker)
163 {
164 p.setPen(markerColor());
165 p.drawLine(m_markerPos, rect().bottom(), m_markerPos, rect().top());
166 }
167
168 p.end();
169
170 painter.drawPixmap( 0, 0, m_paintPixmap );
171}
