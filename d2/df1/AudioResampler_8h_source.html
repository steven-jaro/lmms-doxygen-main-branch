34/**
35 * @class AudioResampler
36 * @brief A utility class for resampling interleaved audio buffers using various resampling algorithms.
37 *
38 * This class provides support for zero-order hold, linear, and several levels of sinc-based resampling.
39 */
40class LMMS_EXPORT AudioResampler
41{
42public:
43 /**
44 * @enum Mode
45 * @brief Defines the resampling method to use.
46 */
47 enum class Mode
48 {
49 ZOH, //!< Zero Order Hold (nearest-neighbor) interpolation.
50 Linear, //!< Linear interpolation.
51 SincFastest, //!< Fastest sinc-based resampling.
52 SincMedium, //!< Medium quality sinc-based resampling.
53 SincBest //!< Highest quality sinc-based resampling.
54 };
55
56 /**
57 * @struct Result
58 * @brief Result of a resampling operation.
59 */
60 struct Result
61 {
62 f_cnt_t inputFramesUsed; //!< The number of input frames used during processing.
63 f_cnt_t outputFramesGenerated; //!< The number of output frames generated during processing.
64 };
65
66 /**
67 * @brief Constructs an `AudioResampler` instance.
68 * @param mode The resampling mode to use.
69 * @param channels Number of audio channels. Defaults to `2` (stereo).
70 */
71 AudioResampler(Mode mode, ch_cnt_t channels = 2);
72
73 /**
74 * @brief Process a block of interleaved audio input from `input` and resample it into `output`.
75 *
76 * @param input The interleaved audio input.
77 * @param output The interleaved audio output.
78 *
79 * @throws `std::invalid_argument` if a channel mismatch has been detected.
80 * @throws `std::runtime_error` if the resampling process has failed.
81 *
82 * @remark This utility class does not cache the input and output buffers, making it stateless. In other words,
83 * `input` is directly resampled into the `output`.
84 *
85 * @returns the result of the resampling process. See @ref Result for more details.
86 */
87 [[nodiscard]] auto process(InterleavedBufferView<const float> input, InterleavedBufferView<float> output) -> Result;
88
89 /**
90 * @brief Resets the internal resampler state.
91 * Useful when working with unreleated pieces of audio.
92 */
93 void reset();
94
95 /**
96 * @brief Sets the resampling ratio to `ratio`.
97 * @param ratio Output sample rate divided by input sample rate.
98 */
99 void setRatio(double ratio) { m_ratio = ratio; }
100
101 /**
102 * @brief Sets the resampling ratio to `output / input`.
103 * @param input Input sample rate.
104 * @param output Output sample rate.
105 */
106 void setRatio(sample_rate_t input, sample_rate_t output) { m_ratio = static_cast<double>(output) / input; }
107
108 //! @returns the resampling ratio.
109 auto ratio() const -> double { return m_ratio; }
110
111 //! @returns the number of channels expected by the resampler.
112 auto channels() const -> ch_cnt_t { return m_channels; }
113
114 //! @returns the interpolation mode used by this resampler.
115 auto mode() const -> Mode { return m_mode; }
116
117private:
118 struct LMMS_EXPORT StateDeleter { void operator()(void* state); };
119 std::unique_ptr<void, StateDeleter> m_state;
120 Mode m_mode;
121 ch_cnt_t m_channels = 0;
122 double m_ratio = 1.0;
123 int m_error = 0;
124};
125
126} // namespace lmms
