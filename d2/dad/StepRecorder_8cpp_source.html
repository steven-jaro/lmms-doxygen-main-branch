52void StepRecorder::start(const TimePos& currentPosition, const TimePos& stepLength)
53{
54 m_isRecording = true;
55
56 setStepsLength(stepLength);
57
58 // quantize current position to get start recording position
59 const int q = m_pianoRoll.quantization();
60 const int curPosTicks = currentPosition.getTicks();
61 const int QuantizedPosTicks = (curPosTicks / q) * q;
62 const TimePos& QuantizedPos = TimePos(QuantizedPosTicks);
63
64 m_curStepStartPos = QuantizedPos;
65 m_curStepLength = 0;
66
67 m_stepRecorderWidget.show();
68
69 m_stepRecorderWidget.showHint();
70
71 prepareNewStep();
72}

80void StepRecorder::notePressed(const Note & n)
81{
82 //if this is the first pressed note in step, advance position
83 if(!m_isStepInProgress)
84 {
85 m_isStepInProgress = true;
86
87 //move cursor one step forwards
88 stepForwards();
89 }
90
91 StepNote* stepNote = findCurStepNote(n.key());
92 if(stepNote == nullptr)
93 {
94 m_curStepNotes.push_back(new StepNote(Note(m_curStepLength, m_curStepStartPos, n.key(), n.getVolume(), n.getPanning())));
95 m_pianoRoll.update();
96 }
97 else if (stepNote->isReleased())
98 {
99 stepNote->setPressed();
100 }
101}

103void StepRecorder::noteReleased(const Note & n)
104{
105 StepNote* stepNote = findCurStepNote(n.key());
106
107 if(stepNote != nullptr && stepNote->isPressed())
108 {
109 stepNote->setReleased();
110
111 //if m_updateReleasedTimer is not already active, activate it
112 //(when activated, the timer will re-set itself as long as there are released notes)
113 if(!m_updateReleasedTimer.isActive())
114 {
115 m_updateReleasedTimer.start(REMOVE_RELEASED_NOTE_TIME_THRESHOLD_MS);
116 }
117
118 //check if all note are released, apply notes to clips (or dismiss if length is zero) and prepare to record next step
119 if(allCurStepNotesReleased())
120 {
121 if(m_curStepLength > 0)
122 {
123 applyStep();
124 }
125 else
126 {
127 dismissStep();
128 }
129 }
130 }
131}

133bool StepRecorder::keyPressEvent(QKeyEvent* ke)
134{
135 bool event_handled = false;
136
137 switch(ke->key())
138 {
139 case Qt::Key_Right:
140 {
141 if(!ke->isAutoRepeat())
142 {
143 stepForwards();
144 }
145 event_handled = true;
146 break;
147 }
148
149 case Qt::Key_Left:
150 {
151 if(!ke->isAutoRepeat())
152 {
153 stepBackwards();
154 }
155 event_handled = true;
156 break;
157 }
158 }
159
160 return event_handled;
161}

163void StepRecorder::setStepsLength(const TimePos& newLength)
164{
165 if(m_isStepInProgress)
166 {
167 //update current step length by the new amount : (number_of_steps * newLength)
168 m_curStepLength = (m_curStepLength / m_stepsLength) * newLength;
169
170 updateCurStepNotes();
171 }
172
173 m_stepsLength = newLength;
174
175 updateWidget();

193void StepRecorder::stepForwards()
194{
195 if(m_isStepInProgress)
196 {
197 m_curStepLength += m_stepsLength;
198
199 updateCurStepNotes();
200 }
201 else
202 {
203 m_curStepStartPos += m_stepsLength;
204 }
205
206 updateWidget();

209void StepRecorder::stepBackwards()
210{
211 if(m_isStepInProgress)
212 {
213 if(m_curStepLength > 0)
214 {
215 m_curStepLength = max(m_curStepLength - m_stepsLength, 0);
216 }
217 else
218 {
219 //if length is already zero - move starting position backwards
220 m_curStepStartPos = max(m_curStepStartPos - m_stepsLength, 0);
221 }
222
223 updateCurStepNotes();
224 }
225 else
226 {
227 m_curStepStartPos = max(m_curStepStartPos - m_stepsLength, 0);
228 }
229
230 updateWidget();
231}

233void StepRecorder::applyStep()
234{
235 m_midiClip->addJournalCheckPoint();
236
237 for (const StepNote* stepNote : m_curStepNotes)
238 {
239 m_midiClip->addNote(stepNote->m_note, false);
240 }
241
242 m_midiClip->rearrangeAllNotes();
243 m_midiClip->updateLength();
244 m_midiClip->dataChanged();
245 Engine::getSong()->setModified();
246
247 prepareNewStep();
248}

260void StepRecorder::prepareNewStep()
261{
262 for(StepNote* stepNote : m_curStepNotes)

267
268 m_isStepInProgress = false;

286void StepRecorder::removeNotesReleasedForTooLong()
287{
288 int nextTimout = std::numeric_limits<int>::max();
289 bool notesRemoved = false;
290
291 for (const auto& stepNote : m_curStepNotes)
292 {
293 if (stepNote->isReleased())
294 {
295 const int timeSinceReleased = stepNote->timeSinceReleased(); // capture value to avoid wraparound when calculating nextTimout
296 if (timeSinceReleased >= REMOVE_RELEASED_NOTE_TIME_THRESHOLD_MS)
297 {
298 notesRemoved = true;
299 }
300 else
301 {
302 nextTimout = min(nextTimout, REMOVE_RELEASED_NOTE_TIME_THRESHOLD_MS - timeSinceReleased);
303 }
304 }
305 }
306
307 m_curStepNotes.erase(std::remove_if(m_curStepNotes.begin(), m_curStepNotes.end(), [](auto stepNote)
308 {
309 bool shouldRemove = stepNote->isReleased() && stepNote->timeSinceReleased() >= REMOVE_RELEASED_NOTE_TIME_THRESHOLD_MS;
310 if (shouldRemove)
311 {
312 delete stepNote;
313 }
314
315 return shouldRemove;
316 }), m_curStepNotes.end());
317
318 if(notesRemoved)
319 {
320 m_pianoRoll.update();
321 }
322
323 if(nextTimout != std::numeric_limits<int>::max())
324 {
325 m_updateReleasedTimer.start(nextTimout);
326 }
327 else
328 {
329 // no released note found for next timeout, stop timer
330 m_updateReleasedTimer.stop();
331 }
332}
