28bool JackRestartThreadedDriver::Execute()
29{
30 try {
31 // Keep running even in case of error
32 while (fThread.GetStatus() == JackThread::kRunning) {
33 Process();
34 }
35 return false;
36 } catch (JackNetException& e) {
37 e.PrintMessage();
38 jack_info("Driver is restarted");
39 fThread.DropSelfRealTime();
40 // Thread in kIniting status again...
41 fThread.SetStatus(JackThread::kIniting);
42 if (Init()) {
43 // Thread in kRunning status again...
44 fThread.SetStatus(JackThread::kRunning);
45 return true;
46 } else {
47 return false;
48 }
49 }
50}
