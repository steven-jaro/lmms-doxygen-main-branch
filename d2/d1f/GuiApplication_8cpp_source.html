89GuiApplication::GuiApplication()
90{
91 // Immediately register our SIGINT handler
92 createSocketNotifier();
93
94 // prompt the user to create the LMMS working directory (e.g. ~/Documents/lmms) if it doesn't exist
95 if ( !ConfigManager::inst()->hasWorkingDir() &&
96 QMessageBox::question( nullptr,
97 tr( "Working directory" ),
98 tr( "The LMMS working directory %1 does not "
99 "exist. Create it now? You can change the directory "
100 "later via Edit -> Settings." ).arg( ConfigManager::inst()->workingDir() ),
101 QMessageBox::Yes | QMessageBox::No, QMessageBox::Yes ) == QMessageBox::Yes)
102 {
103 ConfigManager::inst()->createWorkingDir();
104 }
105 // Init style and palette
106 QDir::addSearchPath("artwork", ConfigManager::inst()->themeDir());
107 QDir::addSearchPath("artwork", ConfigManager::inst()->defaultThemeDir());
108 QDir::addSearchPath("artwork", ":/artwork");
109
110 auto lmmsstyle = new LmmsStyle();
111 QApplication::setStyle(lmmsstyle);
112
113 auto lmmspal = new LmmsPalette(nullptr, lmmsstyle);
114 auto lpal = new QPalette(lmmspal->palette());
115
116 QApplication::setPalette( *lpal );
117 LmmsStyle::s_palette = lpal;
118
119#ifdef LMMS_BUILD_APPLE
120 QApplication::setAttribute(Qt::AA_DontShowIconsInMenus, true);
121#endif
122
123 // Show splash screen
124 QSplashScreen splashScreen( embed::getIconPixmap( "splash" ) );
125 splashScreen.setFixedSize(splashScreen.pixmap().size());
126 splashScreen.show();
127
128 QHBoxLayout layout;
129 layout.setAlignment(Qt::AlignBottom);
130 splashScreen.setLayout(&layout);
131
132 // Create a left-aligned label for loading progress
133 // & a right-aligned label for version info
134 QLabel loadingProgressLabel;
135 m_loadingProgressLabel = &loadingProgressLabel;
136 QLabel versionLabel(MainWindow::tr( "Version %1" ).arg( LMMS_VERSION ));
137
138 loadingProgressLabel.setAlignment(Qt::AlignLeft);
139 versionLabel.setAlignment(Qt::AlignRight);
140
141 layout.addWidget(&loadingProgressLabel);
142 layout.addWidget(&versionLabel);
143
144 // may have long gaps between future frames, so force update now
145 splashScreen.update();
146 qApp->processEvents();
147
148 connect(Engine::inst(), SIGNAL(initProgress(const QString&)),
149 this, SLOT(displayInitProgress(const QString&)));
150
151 // Init central engine which handles all components of LMMS
152 Engine::init(false);
153
154 s_instance = this;
155
156 displayInitProgress(tr("Preparing UI"));
157
158 m_mainWindow = new MainWindow;
159 connect(m_mainWindow, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
160 connect(m_mainWindow, SIGNAL(initProgress(const QString&)),
161 this, SLOT(displayInitProgress(const QString&)));
162
163 displayInitProgress(tr("Preparing song editor"));
164 m_songEditor = new SongEditorWindow(Engine::getSong());
165 connect(m_songEditor, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
166
167 displayInitProgress(tr("Preparing mixer"));
168 m_mixerView = new MixerView(Engine::mixer());
169 connect(m_mixerView, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
170
171 displayInitProgress(tr("Preparing controller rack"));
172 m_controllerRackView = new ControllerRackView;
173 connect(m_controllerRackView, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
174
175 displayInitProgress(tr("Preparing project notes"));
176 m_projectNotes = new ProjectNotes;
177 connect(m_projectNotes, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
178
179 displayInitProgress(tr("Preparing microtuner"));
180 m_microtunerConfig = new MicrotunerConfig;
181 connect(m_microtunerConfig, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
182
183 displayInitProgress(tr("Preparing pattern editor"));
184 m_patternEditor = new PatternEditorWindow(Engine::patternStore());
185 connect(m_patternEditor, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
186
187 displayInitProgress(tr("Preparing piano roll"));
188 m_pianoRoll = new PianoRollWindow();
189 connect(m_pianoRoll, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
190
191 displayInitProgress(tr("Preparing automation editor"));
192 m_automationEditor = new AutomationEditorWindow;
193 connect(m_automationEditor, SIGNAL(destroyed(QObject*)), this, SLOT(childDestroyed(QObject*)));
194
195 splashScreen.finish(m_mainWindow);
196 m_mainWindow->finalize();
197
198 m_loadingProgressLabel = nullptr;
199}

259/** \brief Called from main when SIGINT is fired
260 *
261 * Unix signal handlers can only call async-signal-safe functions:
262 * write(fd) --> QSocketNotifier --> SLOT sigintOccurred()
263 *
264 * See https://doc.qt.io/qt-6/unix-signals.html
265 */
266void GuiApplication::sigintHandler(int)
267{
268#ifdef LMMS_BUILD_WIN32
269 char message[] = "Sorry, SIGINT is unhandled on this platform\n";
270 std::ignore = _write(_fileno(stderr), message, sizeof(message));
271#else
272 char a = 1;
273 std::ignore = ::write(s_sigintFd[0], &a, sizeof(a));
274#endif
275}

277// Create our unix signal notifiers
278void GuiApplication::createSocketNotifier()
279{
280#ifdef LMMS_BUILD_WIN32
281 // no-op
282#else
283 if (::socketpair(AF_UNIX, SOCK_STREAM, 0, s_sigintFd))
284 {
285 qFatal("Couldn't create SIGINT socketpair");
286 return;
287 }
288
289 // Listen on the file descriptor for SIGINT
290 m_sigintNotifier = new QSocketNotifier(s_sigintFd[1], QSocketNotifier::Read, this);
291 connect(m_sigintNotifier, SIGNAL(activated(QSocketDescriptor)), this, SLOT(sigintOccurred()), Qt::QueuedConnection);
292#endif
293}

295// Handle the shutdown event
296void GuiApplication::sigintOccurred()
297{
298 m_sigintNotifier->setEnabled(false);
299 qDebug() << "Shutting down...";
300 // cleanup, etc
301 qApp->exit(3);
302 m_sigintNotifier->setEnabled(true);
303}

306/*!
307 * @brief Returns the Windows System font.
308 */
309QFont GuiApplication::getWin32SystemFont()
310{
311 NONCLIENTMETRICS metrics = { sizeof( NONCLIENTMETRICS ) };
312 SystemParametersInfo( SPI_GETNONCLIENTMETRICS, sizeof( NONCLIENTMETRICS ), &metrics, 0 );
313 int pointSize = metrics.lfMessageFont.lfHeight;
314 if ( pointSize < 0 )
315 {
316 // height is in pixels, convert to points
317 HDC hDC = GetDC( nullptr );
318 pointSize = MulDiv(std::abs(pointSize), 72, GetDeviceCaps(hDC, LOGPIXELSY));
319 ReleaseDC( nullptr, hDC );
320 }
321
322 return QFont( QString::fromUtf8( metrics.lfMessageFont.lfFaceName ), pointSize );
323}
324#endif
