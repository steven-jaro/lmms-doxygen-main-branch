103void ConfigManager::upgrade_1_1_90()
104{
105 // Remove trailing " (bad latency!)" string which was once saved with PulseAudio
106 if(value("mixer", "audiodev").startsWith("PulseAudio ("))
107 {
108 setValue("mixer", "audiodev", "PulseAudio");
109 }
110
111 // MidiAlsaRaw used to store the device info as "Device" instead of "device"
112 if (value("MidiAlsaRaw", "device").isNull())
113 {
114 // copy "device" = "Device" and then delete the old "Device" (further down)
115 QString oldDevice = value("MidiAlsaRaw", "Device");
116 setValue("MidiAlsaRaw", "device", oldDevice);
117 }
118 if (!value("MidiAlsaRaw", "device").isNull())
119 {
120 // delete the old "Device" in the case that we just copied it to "device"
121 // or if the user somehow set both the "Device" and "device" fields
122 deleteValue("MidiAlsaRaw", "Device");
123 }
124}

136void ConfigManager::upgrade_1_2_2()
137{
138 // Since mixer has been renamed to audioengine, we need to transfer the
139 // attributes from the old element to the new one
140 std::vector<QString> attrs = {
141 "audiodev", "mididev", "framesperaudiobuffer", "hqaudio", "samplerate"
142 };
143
144 for (auto attr : attrs)
145 {
146 if (!value("mixer", attr).isNull())
147 {
148 setValue("audioengine", attr, value("mixer", attr));
149 deleteValue("mixer", attr);
150 }
151 }
152
153 m_settings.remove("mixer");

156void ConfigManager::upgrade()
157{
158 // Skip the upgrade if versions match
159 if (m_version == LMMS_VERSION)
160 {
161 return;
162 }
163
164 // Runs all necessary upgrade methods
165 std::size_t max = std::min(static_cast<std::size_t>(m_configVersion), UPGRADE_METHODS.size());
166 std::for_each( UPGRADE_METHODS.begin() + max, UPGRADE_METHODS.end(),
167 [this](UpgradeMethod um)
168 {
169 (this->*um)();
170 }
171 );
172
173 ProjectVersion createdWith = m_version;
174
175 // Don't use old themes as they break the UI (i.e. 0.4 != 1.0, etc)
176 if (createdWith.setCompareType(ProjectVersion::CompareType::Minor) != LMMS_VERSION)
177 {
178 m_themeDir = defaultThemeDir();
179 }
180
181 // Bump the version, now that we are upgraded
182 m_version = LMMS_VERSION;
183 m_configVersion = UPGRADE_METHODS.size();
184}

357QString ConfigManager::value(const QString& cls, const QString& attribute, const QString& defaultVal) const
358{
359 if (m_settings.find(cls) != m_settings.end())
360 {
361 for (const auto& setting : m_settings[cls])
362 {
363 if (setting.first == attribute)
364 {
365 return setting.second;
366 }
367 }
368 }
369 return defaultVal;

375void ConfigManager::setValue(const QString & cls,
376 const QString & attribute,
377 const QString & value)
378{
379 if(m_settings.contains(cls))
380 {
381 for(QPair<QString, QString>& pair : m_settings[cls])
382 {
383 if(pair.first == attribute)
384 {
385 if (pair.second != value)
386 {
387 pair.second = value;
388 emit valueChanged(cls, attribute, value);
389 }
390 return;
391 }
392 }
393 }
394 // not in map yet, so we have to add it...
395 m_settings[cls].push_back(qMakePair(attribute, value));
396}

400{
401 if(m_settings.contains(cls))
402 {
403 for(stringPairVector::iterator it = m_settings[cls].begin();
404 it != m_settings[cls].end(); ++it)

410 }

416void ConfigManager::loadConfigFile(const QString & configFile)
417{
418 // read the XML file and create DOM tree
419 // Allow configuration file override through --config commandline option
420 if (!configFile.isEmpty())
421 {
422 m_lmmsRcFile = configFile;
423 }
424
425 QFile cfg_file(m_lmmsRcFile);
426 QDomDocument dom_tree;
427
428 if(cfg_file.open(QIODevice::ReadOnly))
429 {
430 QString errorString;
431 int errorLine, errorCol;
432 if(dom_tree.setContent(&cfg_file, false, &errorString, &errorLine, &errorCol))
433 {
434 // get the head information from the DOM
435 QDomElement root = dom_tree.documentElement();
436
437 QDomNode node = root.firstChild();
438
439 // Cache LMMS version
440 if (!root.attribute("version").isNull()) {
441 m_version = root.attribute("version");
442 }
443
444 // Get the version of the configuration file (for upgrade purposes)
445 if( root.attribute("configversion").isNull() )
446 {
447 m_configVersion = legacyConfigVersion(); // No configversion attribute found
448 }
449 else
450 {
451 bool success;
452 m_configVersion = root.attribute("configversion").toUInt(&success);
453 if( !success ) qWarning("Config Version conversion failure.");
454 }
455
456 // create the settings-map out of the DOM
457 while(!node.isNull())
458 {
459 if(node.isElement() &&
460 node.toElement().hasAttributes ())
461 {
462 stringPairVector attr;
463 QDomNamedNodeMap node_attr =
464 node.toElement().attributes();
465 for(int i = 0; i < node_attr.count();
466 ++i)
467 {
468 QDomNode n = node_attr.item(i);
469 if(n.isAttr())
470 {
471 attr.push_back(qMakePair(n.toAttr().name(),
472 n.toAttr().value()));
473 }
474 }
475 m_settings[node.nodeName()] = attr;
476 }
477 else if(node.nodeName() == "recentfiles")
478 {
479 m_recentlyOpenedProjects.clear();
480 QDomNode n = node.firstChild();
481 while(!n.isNull())
482 {
483 if(n.isElement() && n.toElement().hasAttributes())
484 {
485 m_recentlyOpenedProjects << n.toElement().attribute("path");
486 }
487 n = n.nextSibling();
488 }
489 }
490 else if (node.nodeName() == "favoriteitems")
491 {
492 m_favoriteItems.clear();
493 QDomNode n = node.firstChild();
494 while (!n.isNull())
495 {
496 if (n.isElement() && n.toElement().hasAttributes())
497 {
498 m_favoriteItems << n.toElement().attribute("path");
499 }
500 n = n.nextSibling();
501 }
502 }
503 node = node.nextSibling();
504 }
505
506 if(value("paths", "theme") != "")
507 {
508 m_themeDir = value("paths", "theme");
509#ifdef LMMS_BUILD_WIN32
510 // Detect a QDir/QFile hang on Windows
511 // see issue #3417 on github
512 bool badPath = (m_themeDir == "/" || m_themeDir == "\\");
513#else
514 bool badPath = false;
515#endif
516
517 if(badPath || !QDir(m_themeDir).exists() ||
518 !QFile(m_themeDir + "/style.css").exists())
519 {
520 m_themeDir = defaultThemeDir();
521 }
522 m_themeDir = ensureTrailingSlash(m_themeDir);
523 }
524 setWorkingDir(value("paths", "workingdir"));
525
526 setGIGDir(value("paths", "gigdir") == "" ? gigDir() : value("paths", "gigdir"));
527 setSF2Dir(value("paths", "sf2dir") == "" ? sf2Dir() : value("paths", "sf2dir"));
528 setVSTDir(value("paths", "vstdir"));
529 setLADSPADir(value("paths", "ladspadir"));
530 #ifdef LMMS_HAVE_STK
531 setSTKDir(value("paths", "stkdir"));
532 #endif
533 #ifdef LMMS_HAVE_FLUIDSYNTH
534 setSF2File(value("paths", "defaultsf2"));
535 #endif
536 setBackgroundPicFile(value("paths", "backgroundtheme"));
537 }
538 else if (gui::getGUI() != nullptr)
539 {
540 QMessageBox::warning(nullptr, gui::MainWindow::tr("Configuration file"),
541 gui::MainWindow::tr("Error while parsing configuration file at line %1:%2: %3").
542 arg(errorLine).
543 arg(errorCol).
544 arg(errorString));
545 }
546 cfg_file.close();
547 }
548
549 // Plugins are searched recursively, block problematic locations
550 if( m_vstDir.isEmpty() || m_vstDir == QDir::separator() || m_vstDir == "/" ||
551 m_vstDir == ensureTrailingSlash( QDir::homePath() ) ||
552 !QDir( m_vstDir ).exists() )
553 {
554#ifdef LMMS_BUILD_WIN32
555 QString programFiles = QString::fromLocal8Bit(getenv("ProgramFiles"));
556 m_vstDir = programFiles + "/VstPlugins/";
557#else
558 m_vstDir = m_workingDir + "plugins/vst/";
559#endif
560 }
561
562 if(m_ladspaDir.isEmpty() )
563 {
564 m_ladspaDir = userLadspaDir();
565 }
566
567#ifdef LMMS_HAVE_STK
568 if(m_stkDir.isEmpty() || m_stkDir == QDir::separator() || m_stkDir == "/" ||
569 !QDir(m_stkDir).exists())
570 {
571#if defined(LMMS_BUILD_WIN32)
572 m_stkDir = m_dataDir + "stk/rawwaves/";
573#else
574 // Look for bundled raw waves first
575 m_stkDir = qApp->applicationDirPath() + "/../share/stk/rawwaves/";
576 // Try system installations if not exists
577 if (!QDir(m_stkDir).exists())
578 {
579 m_stkDir = "/usr/local/share/stk/rawwaves/";
580 }
581 if (!QDir(m_stkDir).exists())
582 {
583 m_stkDir = "/usr/share/stk/rawwaves/";
584 }
585#endif
586 }
587#endif // LMMS_HAVE_STK
588
589 upgrade();
590
591 QStringList searchPaths;
592 if (std::getenv("LMMS_THEME_PATH"))
593 searchPaths << std::getenv("LMMS_THEME_PATH");
594 searchPaths << themeDir() << defaultThemeDir();
595 QDir::setSearchPaths("resources", searchPaths);
596
597 // Create any missing subdirectories in the working dir, but only if the working dir exists
598 if(hasWorkingDir())
599 {
600 createWorkingDir();
601 }
602
603 for (auto& file : m_recentlyOpenedProjects)
604 {
605 file = PathUtil::toAbsolute(file);
606 }
607
608 for (auto& file : m_favoriteItems)
609 {
610 file = PathUtil::toAbsolute(file);
611 }
612}

617void ConfigManager::saveConfigFile()
618{
619 setValue("paths", "theme", m_themeDir);
620 setValue("paths", "workingdir", m_workingDir);
621 setValue("paths", "vstdir", m_vstDir);
622 setValue("paths", "gigdir", m_gigDir);
623 setValue("paths", "sf2dir", m_sf2Dir);
624 setValue("paths", "ladspadir", m_ladspaDir);
625#ifdef LMMS_HAVE_STK
626 setValue("paths", "stkdir", m_stkDir);
627#endif
628#ifdef LMMS_HAVE_FLUIDSYNTH
629 setValue("paths", "defaultsf2", m_sf2File);
630#endif
631 setValue("paths", "backgroundtheme", m_backgroundPicFile);
632
633 QDomDocument doc("lmms-config-file");
634
635 QDomElement lmms_config = doc.createElement("lmms");
636 lmms_config.setAttribute("version", m_version);
637 lmms_config.setAttribute("configversion", m_configVersion);
638 doc.appendChild(lmms_config);
639
640 for (auto it = m_settings.begin(); it != m_settings.end(); ++it)
641 {
642 QDomElement n = doc.createElement(it.key());
643 for (const auto& [first, second] : *it)
644 {
645 n.setAttribute(first, second);
646 }
647 lmms_config.appendChild(n);
648 }
649
650 QDomElement recent_files = doc.createElement("recentfiles");
651
652 for (const auto& recentlyOpenedProject : m_recentlyOpenedProjects)
653 {
654 QDomElement n = doc.createElement("file");
655 n.setAttribute("path", PathUtil::toShortestRelative(recentlyOpenedProject));
656 recent_files.appendChild(n);
657 }
658 lmms_config.appendChild(recent_files);
659
660 QDomElement favorite_items = doc.createElement("favoriteitems");
661
662 for (const auto& favoriteItem : m_favoriteItems)
663 {
664 QDomElement n = doc.createElement("item");
665 n.setAttribute("path", PathUtil::toShortestRelative(favoriteItem));
666 favorite_items.appendChild(n);
667 }
668
669 lmms_config.appendChild(favorite_items);
670
671 QString xml = "<?xml version=\"1.0\"?>\n" + doc.toString(2);
672
673 QFile outfile(m_lmmsRcFile);
674 if(!outfile.open(QIODevice::WriteOnly | QIODevice::Truncate))
675 {
676 using gui::MainWindow;
677
678 QString title, message;
679 title = MainWindow::tr("Could not open file");
680 message = MainWindow::tr("Could not open file %1 "
681 "for writing.\nPlease make "
682 "sure you have write "
683 "permission to the file and "
684 "the directory containing the "
685 "file and try again!"
686 ).arg(m_lmmsRcFile);
687 if (gui::getGUI() != nullptr)
688 {
689 QMessageBox::critical(nullptr, title, message,
690 QMessageBox::Ok,
691 QMessageBox::NoButton);
692 }
693 return;
694 }
695
696 outfile.write(xml.toUtf8());
697 outfile.close();
698}

716void ConfigManager::initDevelopmentWorkingDir()
717{
718 // If we're in development (lmms is not installed) let's get the source and
719 // binary directories by reading the CMake Cache
720 QDir appPath = qApp->applicationDirPath();
721 // If in tests, get parent directory
722 if (appPath.dirName() == "tests") {
723 appPath.cdUp();
724 }
725 QFile cmakeCache(appPath.absoluteFilePath("CMakeCache.txt"));
726 if (cmakeCache.exists()) {
727 cmakeCache.open(QFile::ReadOnly);
728 QTextStream stream(&cmakeCache);
729
730 // Find the lines containing something like lmms_SOURCE_DIR:static=<dir>
731 // and lmms_BINARY_DIR:static=<dir>
732 int done = 0;
733 while(! stream.atEnd())
734 {
735 QString line = stream.readLine();
736
737 if (line.startsWith("lmms_SOURCE_DIR:")) {
738 QString srcDir = line.section('=', -1).trimmed();
739 QDir::addSearchPath("data", srcDir + "/data/");
740 done++;
741 }
742 if (line.startsWith("lmms_BINARY_DIR:")) {
743 m_lmmsRcFile = line.section('=', -1).trimmed() + QDir::separator() +
744 ".lmmsrc.xml";
745 done++;
746 }
747 if (done == 2)
748 {
749 break;
750 }
751 }
752
753 cmakeCache.close();
754 }
755}

757// If configversion is not present, we will convert the LMMS version to the appropriate
758// configuration file version for backwards compatibility.
759unsigned int ConfigManager::legacyConfigVersion()
760{
761 ProjectVersion createdWith = m_version;
762
763 createdWith.setCompareType(ProjectVersion::CompareType::Build);
764
765 if( createdWith < "1.1.90" )
766 {
767 return 0;
768 }
769 else if( createdWith < "1.1.91" )
770 {
771 return 1;
772 }
773 else
774 {
775 return 2;
776 }
