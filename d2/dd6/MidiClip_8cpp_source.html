88void MidiClip::resizeToFirstTrack()
89{
90 // Resize this track to be the same as existing tracks in the pattern
91 const TrackContainer::TrackList & tracks =
92 m_instrumentTrack->trackContainer()->tracks();
93 for (const auto& track : tracks)
94 {
95 if (track->type() == Track::Type::Instrument)
96 {
97 if (track != m_instrumentTrack)
98 {
99 const auto& instrumentTrackClips = m_instrumentTrack->getClips();
100 const auto currentClipIt = std::find(instrumentTrackClips.begin(), instrumentTrackClips.end(), this);
101 unsigned int currentClip = currentClipIt != instrumentTrackClips.end() ?
102 std::distance(instrumentTrackClips.begin(), currentClipIt) : -1;
103 m_steps = static_cast<MidiClip*>(track->getClip(currentClip))->m_steps;
104 }
105 break;
106 }
107 }
108}

126void MidiClip::updateLength()
127{
128 if( m_clipType == Type::BeatClip )
129 {
130 changeLength( beatClipLength() );
131 updatePatternTrack();
132 return;
133 }
134
135 // If the clip hasn't already been manually resized, automatically resize it.
136 if (getAutoResize())
137 {
138 tick_t max_length = TimePos::ticksPerBar();
139
140 for (const auto& note : m_notes)
141 {
142 if (note->length() > 0)
143 {
144 max_length = std::max<tick_t>(max_length, note->endPos());
145 }
146 }
147 changeLength( TimePos( max_length ).nextFullBar() *
148 TimePos::ticksPerBar() );
149 setStartTimeOffset(TimePos(0));
150 updatePatternTrack();
151 }
152}

157TimePos MidiClip::beatClipLength() const
158{
159 tick_t max_length = TimePos::ticksPerBar();
160
161 for (const auto& note : m_notes)
162 {
163 if (note->type() == Note::Type::Step)
164 {
165 max_length = std::max<tick_t>(max_length, note->pos() + 1);
166 }
167 }
168
169 if (m_steps != TimePos::stepsPerBar())
170 {
171 max_length = m_steps * TimePos::ticksPerBar() / TimePos::stepsPerBar();
172 }
173
174 return TimePos{max_length}.nextFullBar() * TimePos::ticksPerBar();
175}

180Note * MidiClip::addNote( const Note & _new_note, const bool _quant_pos )
181{
182 auto new_note = _new_note.clone();
183 if (_quant_pos && gui::getGUI()->pianoRoll())
184 {
185 new_note->quantizePos(gui::getGUI()->pianoRoll()->quantization());
186 }
187
188 instrumentTrack()->lock();
189 m_notes.insert(std::upper_bound(m_notes.begin(), m_notes.end(), new_note, Note::lessThan), new_note);
190 instrumentTrack()->unlock();
191
192 checkType();
193 updateLength();
194
195 emit dataChanged();
196
197 return new_note;

310void MidiClip::reverseNotes(const NoteVector& notes)
311{
312 if (notes.empty()) { return; }
313
314 addJournalCheckPoint();
315
316 // Find the very first start position and the very last end position of all the notes.
317 TimePos firstPos = (*std::min_element(notes.begin(), notes.end(), [](const Note* n1, const Note* n2){ return Note::lessThan(n1, n2); }))->pos();
318 TimePos lastPos = (*std::max_element(notes.begin(), notes.end(), [](const Note* n1, const Note* n2){ return n1->endPos() < n2->endPos(); }))->endPos();
319
320 for (auto note : notes)
321 {
322 TimePos newStart = lastPos - (note->pos() - firstPos) - note->length();
323 note->setPos(newStart);
324 }
325
326 rearrangeAllNotes();
327 emit dataChanged();
328}

332void MidiClip::splitNotes(const NoteVector& notes, TimePos pos)
333{
334 if (notes.empty()) { return; }
335
336 addJournalCheckPoint();
337
338 for (const auto& note : notes)
339 {
340 int leftLength = pos.getTicks() - note->pos();
341 int rightLength = note->length() - leftLength;
342
343 // Split out of bounds
344 if (leftLength <= 0 || rightLength <= 0)
345 {
346 continue;
347 }
348
349 // Reduce note length
350 note->setLength(leftLength);
351
352 // Add new note with the remaining length
353 Note newNote = Note(*note);
354 newNote.setLength(rightLength);
355 newNote.setPos(note->pos() + leftLength);
356
357 addNote(newNote, false);
358 }

361void MidiClip::splitNotesAlongLine(const NoteVector notes, TimePos pos1, int key1, TimePos pos2, int key2, bool deleteShortEnds)
362{
363 if (notes.empty()) { return; }
364
365 // Don't split if the line is horitzontal
366 if (key1 == key2) { return; }
367
368 addJournalCheckPoint();
369
370 const auto slope = 1.f * (pos2 - pos1) / (key2 - key1);
371 const auto& [minKey, maxKey] = std::minmax(key1, key2);
372
373 for (const auto& note : notes)
374 {
375 // Skip if the key is <= to minKey, since the line is drawn from the top of minKey to the top of maxKey, but only passes through maxKey - minKey - 1 total keys.
376 if (note->key() <= minKey || note->key() > maxKey) { continue; }
377
378 // Subtracting 0.5 to get the line's intercept at the "center" of the key, not the top.
379 const TimePos keyIntercept = slope * (note->key() - 0.5 - key1) + pos1;
380 if (note->pos() < keyIntercept && note->endPos() > keyIntercept)
381 {
382 auto newNote1 = Note{*note};
383 newNote1.setLength(keyIntercept - note->pos());
384
385 auto newNote2 = Note{*note};
386 newNote2.setPos(keyIntercept);
387 newNote2.setLength(note->endPos() - keyIntercept);
388
389 if (deleteShortEnds)
390 {
391 addNote(newNote1.length() >= newNote2.length() ? newNote1 : newNote2, false);
392 }
393 else
394 {
395 addNote(newNote1, false);
396 addNote(newNote2, false);
397 }
398
399 removeNote(note);
400 }
401 }

427void MidiClip::exportToXML(QDomDocument& doc, QDomElement& midiClipElement, bool onlySelectedNotes)
428{
429 midiClipElement.setAttribute("type", static_cast<int>(m_clipType));
430 midiClipElement.setAttribute("name", name());
431 midiClipElement.setAttribute("autoresize", QString::number(getAutoResize()));
432 midiClipElement.setAttribute("off", startTimeOffset());
433
434 if (const auto& c = color())
435 {
436 midiClipElement.setAttribute("color", c->name());
437 }
438 // as the target of copied/dragged MIDI clip is always an existing
439 // MIDI clip, we must not store actual position, instead we store -1
440 // which tells loadSettings() not to mess around with position
441 if (midiClipElement.parentNode().nodeName() == "clipboard" ||
442 midiClipElement.parentNode().nodeName() == "dnddata")
443 {
444 midiClipElement.setAttribute("pos", -1);
445 }
446 else
447 {
448 midiClipElement.setAttribute("pos", startPosition());
449 }
450 midiClipElement.setAttribute("muted", isMuted());
451 midiClipElement.setAttribute("steps", m_steps);
452 midiClipElement.setAttribute("len", length());
453
454 // now save settings of all notes
455 for (auto& note : m_notes)
456 {
457 if (!onlySelectedNotes || note->selected())
458 {
459 note->saveState(doc, midiClipElement);
460 }
461 }

473void MidiClip::loadSettings( const QDomElement & _this )
474{
475 m_clipType = static_cast<Type>( _this.attribute( "type"
476 ).toInt() );
477 setName( _this.attribute( "name" ) );
478
479 if (_this.hasAttribute("color"))
480 {
481 setColor(QColor{_this.attribute("color")});
482 }
483
484 if( _this.attribute( "pos" ).toInt() >= 0 )
485 {
486 movePosition( _this.attribute( "pos" ).toInt() );
487 }
488 if (static_cast<bool>(_this.attribute("muted").toInt()) != isMuted())
489 {
490 toggleMute();
491 }
492
493 clearNotes();
494
495 QDomNode node = _this.firstChild();
496 while( !node.isNull() )
497 {
498 if( node.isElement() &&
499 !node.toElement().attribute( "metadata" ).toInt() )
500 {
501 auto n = new Note;
502 n->restoreState( node.toElement() );
503 m_notes.push_back( n );
504 }
505 node = node.nextSibling();
506 }
507
508 m_steps = _this.attribute( "steps" ).toInt();
509 if( m_steps == 0 )
510 {
511 m_steps = TimePos::stepsPerBar();
512 }
513
514 checkType();
515
516 int len = _this.attribute("len").toInt();
517 if (len <= 0)
518 {
519 // TODO: Handle with an upgrade method
520 updateLength();
521 }
522 else
523 {
524 changeLength(len);
525 }
526
527 setAutoResize(_this.attribute("autoresize", "1").toInt());
528 setStartTimeOffset(_this.attribute("off").toInt());
529
530 emit dataChanged();
531}

579void MidiClip::cloneSteps()
580{
581 int oldLength = m_steps;
582 m_steps *= 2; // cloning doubles the track
583 for(int i = 0; i < oldLength; ++i )
584 {
585 Note *toCopy = noteAtStep( i );
586 if( toCopy )
587 {
588 setStep( oldLength + i, true );
589 Note *newNote = noteAtStep( oldLength + i );
590 newNote->setKey( toCopy->key() );
591 newNote->setLength( toCopy->length() );
592 newNote->setPanning( toCopy->getPanning() );
593 newNote->setVolume( toCopy->getVolume() );
594 }
595 }
596 updateLength();
597 emit dataChanged();
598}

661
662void MidiClip::changeTimeSignature()
663{
664 TimePos last_pos = TimePos::ticksPerBar() - 1;
665 for (const auto& note : m_notes)
666 {
667 if (note->length() < 0 && note->pos() > last_pos)
668 {
669 last_pos = note->pos() + TimePos::ticksPerBar() / TimePos::stepsPerBar();
670 }
671 }
672 last_pos = last_pos.nextFullBar() * TimePos::ticksPerBar();
673 m_steps = std::max<tick_t>(TimePos::stepsPerBar(),
674 last_pos.getBar() * TimePos::stepsPerBar());
675 updateLength();
676}
