156TimePos Note::quantized( const TimePos & m, const int qGrid )
157{
158 float p = ( (float) m / qGrid );
159 if( p - floorf( p ) < 0.5f )
160 {
161 return static_cast<int>( p ) * qGrid;
162 }
163 return static_cast<int>( p + 1 ) * qGrid;
164}

207void Note::loadSettings( const QDomElement & _this )
208{
209 const int oldKey = _this.attribute( "tone" ).toInt() + _this.attribute( "oct" ).toInt() * KeysPerOctave;
210 m_key = std::max(oldKey, _this.attribute("key").toInt());
211 m_volume = _this.attribute( "vol" ).toInt();
212 m_panning = _this.attribute( "pan" ).toInt();
213 m_length = _this.attribute( "len" ).toInt();
214 m_pos = _this.attribute( "pos" ).toInt();
215 // Default m_type value is 0, which corresponds to RegularNote
216 static_assert(0 == static_cast<int>(Type::Regular));
217 m_type = static_cast<Type>(_this.attribute("type", "0").toInt());
218
219 if( _this.hasChildNodes() )
220 {
221 createDetuning();
222 m_detuning->loadSettings( _this );
223 }
224}

260/*! \brief Get the start/end/bottom/top positions of notes in a vector
261 *
262 * Returns no value if there are no notes
263 */
264std::optional<NoteBounds> boundsForNotes(const NoteVector& notes)
265{
266 if (notes.empty()) { return std::nullopt; }
267
268 TimePos start = notes.front()->pos();
269 TimePos end = start;
270 int lower = notes.front()->key();
271 int upper = lower;
272
273 for (const Note* note: notes)
274 {
275 // TODO should we assume that NoteVector is always sorted correctly,
276 // so first() always has the lowest time position?
277 start = std::min(start, note->pos());
278 end = std::max(end, note->endPos());
279 lower = std::min(lower, note->key());
280 upper = std::max(upper, note->key());
281 }
282
283 return NoteBounds{start, end, lower, upper};
284}
