43void LinkedModelGroup::linkControls(LinkedModelGroup *other)
44{
45 foreach_model([&other](const std::string& id, ModelInfo& inf)

47 auto itr2 = other->m_models.find(id);

79void LinkedModelGroup::addModel(AutomatableModel *model, const QString &name)
80{
81 model->setObjectName(name);
82 m_models.emplace(std::string(name.toUtf8().data()), ModelInfo(name, model));
83 connect(model, &AutomatableModel::destroyed,
84 this, [this, model](jo_id_t){
85 if(containsModel(model->objectName()))
86 {
87 emit modelRemoved(model);
88 eraseModel(model->objectName());
89 }
90 },
91 Qt::DirectConnection);
92
93 // View needs to create another child view, e.g. a new knob:
94 emit modelAdded(model);
95 emit dataChanged();
96}

155void LinkedModelGroups::saveSettings(QDomDocument& doc, QDomElement& that)
156{
157 LinkedModelGroup* grp0 = getGroup(0);
158 if (grp0)
159 {
160 QDomElement models = doc.createElement("models");

171{
172 QDomElement models = that.firstChildElement("models");
173 if (auto grp0 = getGroup(0); !models.isNull() && grp0)
174 {
175 // only load the first group, the others are linked to the first
176 grp0->loadValues(models);
177 }
