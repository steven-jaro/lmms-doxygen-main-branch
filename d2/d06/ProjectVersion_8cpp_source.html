35ProjectVersion::ProjectVersion(QString version, CompareType c) :
36 m_version(version),
37 m_compareType(c)
38{
39 // Version numbers may have build data, prefixed with a '+',
40 // but this mustn't affect version precedence in comparisons
41 QString metadataStripped = version.split("+").first();
42 // They must have an obligatory initial segment, and may have
43 // optional identifiers prefaced by a '-'. Both parts affect precedence
44 QString obligatorySegment = metadataStripped.section('-', 0, 0);
45 QString prereleaseSegment = metadataStripped.section('-', 1);
46
47 // The obligatory segment consists of three identifiers: MAJOR.MINOR.PATCH
48 QStringList mainVersion = obligatorySegment.split(".");
49 // HACK: Pad invalid versions in order to prevent crashes
50 while (mainVersion.size() < 3){ mainVersion.append("0"); }
51 m_major = mainVersion.at(0).toInt();
52 m_minor = mainVersion.at(1).toInt();
53 m_patch = mainVersion.at(2).toInt();
54
55 // Any # of optional pre-release identifiers may follow, separated by '.'s
56 if (!prereleaseSegment.isEmpty()){ m_labels = prereleaseSegment.split("."); }
57
58 // HACK: Handle old (1.2.2 and earlier), non-standard versions of the form
59 // MAJOR.MINOR.PATCH.COMMITS, used for non-release builds from source.
60 if (mainVersion.size() >= 4 && m_major <= 1 && m_minor <= 2 && m_patch <= 2){
61 // Drop the standard version identifiers. erase(a, b) removes [a,b)
62 mainVersion.erase(mainVersion.begin(), mainVersion.begin() + 3);
63 // Prepend the remaining identifiers as prerelease versions
64 m_labels = mainVersion + m_labels;
65 // Bump the patch version. x.y.z-a < x.y.z, but we want x.y.z.a > x.y.z
66 m_patch += 1;
67 }
68}

80//! @param c Determines the number of identifiers to check when comparing
81int ProjectVersion::compare(const ProjectVersion & a, const ProjectVersion & b, CompareType c)
82{
83 // How many identifiers to compare before we consider the versions equal
84 const int limit = static_cast<int>(c);
85
86 // Use the value of limit to zero out identifiers we don't care about
87 int aMaj = 0, bMaj = 0, aMin = 0, bMin = 0, aPat = 0, bPat = 0;
88 if (limit >= 1){ aMaj = a.getMajor(); bMaj = b.getMajor(); }
89 if (limit >= 2){ aMin = a.getMinor(); bMin = b.getMinor(); }
90 if (limit >= 3){ aPat = a.getPatch(); bPat = b.getPatch(); }
91
92 // Then we can compare as if we care about every identifier
93 if(aMaj != bMaj){ return aMaj - bMaj; }
94 if(aMin != bMin){ return aMin - bMin; }
95 if(aPat != bPat){ return aPat - bPat; }
96
97 // Decide how many optional identifiers we care about
98 const int maxLabels = std::max(0, limit - 3);
99 const auto aLabels = a.getLabels().mid(0, maxLabels);
100 const auto bLabels = b.getLabels().mid(0, maxLabels);
101
102 // We can only compare identifiers if both versions have them
103 const int commonLabels = std::min(aLabels.size(), bLabels.size());
104 // If one version has optional labels and the other doesn't,
105 // the one without them is bigger
106 if (commonLabels == 0){ return bLabels.size() - aLabels.size(); }
107
108 // Otherwise, compare as many labels as we can
109 for (int i = 0; i < commonLabels; i++){
110 const QString& labelA = aLabels.at(i);
111 const QString& labelB = bLabels.at(i);
112 // If both labels are the same, skip
113 if (labelA == labelB){ continue; }
114 // Numeric and non-numeric identifiers compare differently
115 bool aIsNumeric = false, bIsNumeric = false;
116 const int numA = labelA.toInt(&aIsNumeric);
117 const int numB = labelB.toInt(&bIsNumeric);
118 // toInt reads '-x' as a negative number, semver says it's non-numeric
119 aIsNumeric &= !labelA.startsWith("-");
120 bIsNumeric &= !labelB.startsWith("-");
121 // If only one identifier is numeric, that one is smaller
122 if (aIsNumeric != bIsNumeric){ return aIsNumeric ? -1 : 1; }
123 // If both are numeric, compare as numbers
124 if (aIsNumeric && bIsNumeric){ return numA - numB; }
125 // Otherwise, compare lexically
126 return labelA.compare(labelB);
127 }
128
129 // If everything else matches, the version with more labels is bigger
130 return aLabels.size() - bLabels.size();
131}
