102/*
103==============================================================================
104Name: process_sample
105Description:
106 From one 4-channel input vector, generates two vectors with a pi/2 phase
107 shift between these signals.
108Input parameters:
109 - input: The input sample.
110Output parameters:
111 - out_0: Output sample, ahead.
112 - out_1: Output sample, late.
113Throws: Nothing
114==============================================================================
115*/
116
117template <int NC>
118void PhaseHalfPi4Neon <NC>::process_sample (float32x4_t &out_0, float32x4_t &out_1, float32x4_t input) noexcept
119{
120 out_0 = input; // Even coefs
121 out_1 = load4a (_prev); // Odd coefs
122
123 StageProc4Neon <NBR_COEFS>::process_sample_neg (
124 NBR_COEFS, out_0, out_1, &_bifilter [_phase] [0]
125 );
126
127 storea (_prev, input);
128 _phase = 1 - _phase;
129}

133/*
134==============================================================================
135Name: process_block
136Description:
137 From a block of 4-channel input vectors, generates two blocks of vectors
138 with a pi/2 phase shift between these signals.
139 Input and output blocks may overlap, see assert() for details.
140Input parameters:
141 - in_ptr: Input array, containing nbr_spl samples.
142 - nbr_spl: Number of input samples to process, > 0
143Output parameters:
144 - out_0_ptr: Output sample array (ahead), capacity: nbr_spl samples.
145 - out_1_ptr: Output sample array (late), capacity: nbr_spl samples.
146Throws: Nothing
147==============================================================================
148*/
149
150template <int NC>
151void PhaseHalfPi4Neon <NC>::process_block (float out_0_ptr [], float out_1_ptr [], const float in_ptr [], long nbr_spl) noexcept
152{
153 assert (out_0_ptr != nullptr);
154 assert (out_1_ptr != nullptr);
155 assert (in_ptr != nullptr);
156 assert (out_0_ptr <= in_ptr || out_0_ptr >= in_ptr + nbr_spl * _nbr_chn);
157 assert (out_1_ptr <= in_ptr || out_1_ptr >= in_ptr + nbr_spl * _nbr_chn);
158 assert ( out_0_ptr + nbr_spl * _nbr_chn <= out_1_ptr
159 || out_1_ptr + nbr_spl * _nbr_chn <= out_0_ptr);
160 assert (nbr_spl > 0);
161
162 long pos = 0;
163 if (_phase == 1)
164 {
165 float32x4_t out_0;
166 float32x4_t out_1;
167 process_sample (out_0, out_1, load4u (in_ptr));
168 storeu (out_0_ptr, out_0);
169 storeu (out_1_ptr, out_1);
170 ++ pos;
171 }
172
173 const long end = ((nbr_spl - pos) & -_nbr_phases) + pos;
174 float32x4_t prev = load4a (_prev);
175 while (pos < end)
176 {
177 const auto ofs_0 = pos * _nbr_chn;
178 const auto input_0 = load4u (in_ptr + ofs_0);
179 auto out_0 = input_0;
180 auto out_1 = prev;
181 StageProc4Neon <NBR_COEFS>::process_sample_neg (
182 NBR_COEFS, out_0, out_1, &_bifilter [0] [0]
183 );
184 storeu (out_0_ptr + ofs_0, out_0);
185 storeu (out_1_ptr + ofs_0, out_1);
186
187 const auto ofs_1 = ofs_0 + _nbr_chn;
188 const auto input_1 = load4u (in_ptr + ofs_1);
189 out_0 = input_1;
190 out_1 = input_0; // prev
191 StageProc4Neon <NBR_COEFS>::process_sample_neg (
192 NBR_COEFS, out_0, out_1, &_bifilter [1] [0]
193 );
194 storeu (out_0_ptr + ofs_1, out_0);
195 storeu (out_1_ptr + ofs_1, out_1);
196
197 prev = input_1;
198 pos += 2;
199 }
200 storea (_prev, prev);
201
202 if (pos < nbr_spl)
203 {
204 assert (pos + 1 == nbr_spl);
205 float32x4_t out_0;
206 float32x4_t out_1;
207 const auto ofs = pos * _nbr_chn;
208 process_sample (out_0, out_1, load4u (in_ptr + ofs));
209 storeu (out_0_ptr + ofs, out_0);
210 storeu (out_1_ptr + ofs, out_1);
211 }
212}

216/*
217==============================================================================
218Name: clear_buffers
219Description:
220 Clears filter memory, as if it processed silence since an infinite amount
221 of time.
222Throws: Nothing
223==============================================================================
224*/
225
226template <int NC>
227void PhaseHalfPi4Neon <NC>::clear_buffers () noexcept
228{
229 _phase = 0;
230 for (int i = 0; i < NBR_COEFS + 2; ++i)
231 {
232 storea (_bifilter [0] [i]._mem, vdupq_n_f32 (0));
233 storea (_bifilter [1] [i]._mem, vdupq_n_f32 (0));
234 }
235 storea (_prev, vdupq_n_f32 (0));
236}
