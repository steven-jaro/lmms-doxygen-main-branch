43/*
44 * The "std::polar" template has no clear definition for the range of
45 * the input parameters, and some C++ standard library implementations
46 * don't accept negative amplitude among others. Define our own
47 * FFTpolar template, which works like we expect it to.
48 */
49template<class _Tp>
50std::complex<_Tp>
51FFTpolar(const _Tp& __rho, const _Tp& __theta = _Tp(0))
52{
53 _Tp __x = __rho * cos(__theta);
54 if (std::isnan(__x))
55 __x = 0;
56 _Tp __y = __rho * sin(__theta);
57 if (std::isnan(__y))
58 __y = 0;
59 return std::complex<_Tp>(__x, __y);
60}
