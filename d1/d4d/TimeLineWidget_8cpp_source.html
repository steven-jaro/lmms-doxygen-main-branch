142void TimeLineWidget::paintEvent( QPaintEvent * )
143{
144 QPainter p( this );
145
146 // Draw background
147 p.fillRect( 0, 0, width(), height(), p.background() );
148
149 // Clip so that we only draw everything starting from the offset
150 p.setClipRect(m_xOffset, 0, width() - m_xOffset, height());
151
152 // Variables for the loop rectangle
153 int const loopRectMargin = getLoopRectangleVerticalPadding();
154 int const loopRectHeight = this->height() - 2 * loopRectMargin;
155 int const loopStart = markerX(m_timeline->loopBegin());
156 int const loopEndR = markerX(m_timeline->loopEnd());
157 int const loopRectWidth = loopEndR - loopStart;
158
159 bool const loopPointsActive = m_timeline->loopEnabled();
160
161 // Draw the main loop rectangle (inner fill only)
162 QRect outerRectangle( loopStart, loopRectMargin, loopRectWidth - 1, loopRectHeight - 1 );
163 p.fillRect( outerRectangle, loopPointsActive ? getActiveLoopBrush() : getInactiveLoopBrush());
164
165 // Draw the bar lines and numbers
166 // Activate hinting on the font
167 QFont font = p.font();
168 font.setHintingPreference( QFont::PreferFullHinting );
169 p.setFont(font);
170 int const fontAscent = p.fontMetrics().ascent();
171 int const fontHeight = p.fontMetrics().height();
172
173 QColor const & barLineColor = getBarLineColor();
174 QColor const & barNumberColor = getBarNumberColor();
175
176 bar_t barNumber = m_begin.getBar();
177 int const x = m_xOffset - ((static_cast<int>(m_begin * m_ppb) / TimePos::ticksPerBar()) % static_cast<int>(m_ppb));
178
179 // Double the interval between bar numbers until they are far enough apart
180 int barLabelInterval = 1;
181 while (barLabelInterval * m_ppb < MIN_BAR_LABEL_DISTANCE) { barLabelInterval *= 2; }
182
183 for( int i = 0; x + i * m_ppb < width(); ++i )
184 {
185 ++barNumber;
186 if ((barNumber - 1) % barLabelInterval == 0)
187 {
188 const int cx = x + qRound( i * m_ppb );
189 p.setPen( barLineColor );
190 p.drawLine( cx, 5, cx, height() - 6 );
191
192 const QString s = QString::number( barNumber );
193 p.setPen( barNumberColor );
194 p.drawText( cx + 5, ((height() - fontHeight) / 2) + fontAscent, s );
195 }
196 }
197
198 // Draw the loop rectangle's outer outline
199 p.setPen( loopPointsActive ? getActiveLoopColor() : getInactiveLoopColor() );
200 p.setBrush( Qt::NoBrush );
201 p.drawRect( outerRectangle );
202
203 // Draw the loop rectangle's inner outline
204 QRect innerRectangle = outerRectangle.adjusted( 1, 1, -1, -1 );
205 p.setPen( loopPointsActive ? getActiveLoopInnerColor() : getInactiveLoopInnerColor() );
206 p.setBrush( Qt::NoBrush );
207 p.drawRect( innerRectangle );
208
209 // Draw loop handles if necessary
210 const auto handleMode = ConfigManager::inst()->value("app", "loopmarkermode") == "handles";
211 if (handleMode && underMouse() && QGuiApplication::keyboardModifiers().testFlag(Qt::ShiftModifier))
212 {
213 const auto handleWidth = std::min(m_loopHandleWidth, loopRectWidth / 2 - 1);
214 const auto leftHandle = QRectF(loopStart - .5, loopRectMargin - .5, handleWidth, loopRectHeight);
215 const auto rightHandle = QRectF(loopEndR - handleWidth - .5, loopRectMargin - .5, handleWidth, loopRectHeight);
216 const auto color = loopPointsActive ? m_activeLoopHandleColor : m_inactiveLoopHandleColor;
217
218 p.fillRect(leftHandle, color);
219 p.fillRect(rightHandle, color);
220 }
221
222 // Only draw the position marker if the position line is in view
223 if (markerX(m_pos) >= m_xOffset && markerX(m_pos) < width() - m_posMarkerPixmap.width() / 2)
224 {
225 // Let the position marker extrude to the left
226 p.setClipping(false);
227 p.setOpacity(0.6);
228 p.drawPixmap(markerX(m_pos) - (m_posMarkerPixmap.width() / 2),
229 height() - m_posMarkerPixmap.height(), m_posMarkerPixmap);
230 }
231}

233auto TimeLineWidget::getClickedTime(const int xPosition) const -> TimePos
234{
235 // How far into the timeline we clicked, measuring pixels from the leftmost part of the editor
236 const auto pixelDelta = std::max(xPosition - m_xOffset, 0);
237 return m_begin + static_cast<int>(pixelDelta * TimePos::ticksPerBar() / m_ppb);
238}

240auto TimeLineWidget::getLoopAction(QMouseEvent* event) const -> TimeLineWidget::Action
241{
242 const auto mode = ConfigManager::inst()->value("app", "loopmarkermode");
243 const auto xPos = event->x();
244 const auto button = event->button();
245
246 if (mode == "handles")
247 {
248 // Loop start and end pos, or closest edge of screen if loop extends off it
249 const auto leftMost = std::max(markerX(m_timeline->loopBegin()), m_xOffset);
250 const auto rightMost = std::min(markerX(m_timeline->loopEnd()), width());
251 // Distance from click to handle, positive aimed towards center of loop
252 const auto deltaLeft = xPos - leftMost;
253 const auto deltaRight = rightMost - xPos;
254
255 if (deltaLeft < 0 || deltaRight < 0) { return Action::NoAction; } // Clicked outside loop
256 else if (deltaLeft <= m_loopHandleWidth && deltaLeft < deltaRight) { return Action::MoveLoopBegin; }
257 else if (deltaRight <= m_loopHandleWidth) { return Action::MoveLoopEnd; }
258 else { return Action::MoveLoop; }
259 }
260 else if (mode == "closest")
261 {
262 const TimePos loopMid = (m_timeline->loopBegin() + m_timeline->loopEnd()) / 2;
263 return getClickedTime(xPos) < loopMid ? Action::MoveLoopBegin : Action::MoveLoopEnd;
264 }
265 else // Default to dual-button mode
266 {
267 if (button == Qt::LeftButton) { return Action::MoveLoopBegin; }
268 else if (button == Qt::RightButton) { return Action::MoveLoopEnd; }
269 return Action::NoAction;
270 }
271}

273auto TimeLineWidget::actionCursor(Action action) const -> QCursor
274{
275 switch (action) {
276 case Action::MoveLoop: return Qt::SizeHorCursor;
277 case Action::MoveLoopBegin: return m_cursorSelectLeft;
278 case Action::MoveLoopEnd: return m_cursorSelectRight;
279 // Fall back to normal cursor if no action or action cursor not specified
280 default: return Qt::ArrowCursor;
281 }

284void TimeLineWidget::mousePressEvent(QMouseEvent* event)
285{
286 if (event->x() < m_xOffset) { return; }
287
288 const auto shift = event->modifiers() & Qt::ShiftModifier;
289 const auto ctrl = event->modifiers() & Qt::ControlModifier;
290
291 if (shift) // loop marker manipulation
292 {
293 m_action = getLoopAction(event);
294 setCursor(actionCursor(m_action));
295
296 if (m_action == Action::MoveLoop)
297 {
298 m_dragStartPos = getClickedTime(event->x());
299 m_oldLoopPos = {m_timeline->loopBegin(), m_timeline->loopEnd()};
300 }
301 }
302 else if (event->button() == Qt::LeftButton && ctrl) // selection
303 {
304 m_action = Action::SelectSongClip;
305 m_initalXSelect = event->x();
306 }
307 else if (event->button() == Qt::LeftButton && !ctrl) // move playhead
308 {
309 m_action = Action::MovePositionMarker;
310 }
311
312 if (m_action == Action::MoveLoopBegin || m_action == Action::MoveLoopEnd)
313 {
314 delete m_hint;
315 m_hint = TextFloat::displayMessage(tr("Hint"),
316 tr("Press <%1> to disable magnetic loop points.").arg(UI_CTRL_KEY),
317 embed::getIconPixmap("hint"), 0);
318 }
319
320 setContextMenuPolicy(m_action == Action::NoAction ? Qt::DefaultContextMenu : Qt::PreventContextMenu);
321
322 mouseMoveEvent(event);
323}

325void TimeLineWidget::mouseMoveEvent( QMouseEvent* event )
326{
327 parentWidget()->update(); // essential for widgets that this timeline had taken their mouse move event from.
328
329 auto timeAtCursor = getClickedTime(event->x());
330 const auto control = event->modifiers() & Qt::ControlModifier;
331
332 switch( m_action )
333 {
334 case Action::MovePositionMarker:
335 m_pos.setTicks(timeAtCursor.getTicks());
336 Engine::getSong()->setToTime(timeAtCursor, m_mode);
337 if (!( Engine::getSong()->isPlaying()))
338 {
339 //Song::PlayMode::None is used when nothing is being played.
340 Engine::getSong()->setToTime(timeAtCursor, Song::PlayMode::None);
341 }
342 m_pos.setCurrentFrame( 0 );
343 m_pos.setJumped( true );
344 updatePosition();
345 break;
346
347 case Action::MoveLoopBegin:
348 case Action::MoveLoopEnd:
349 {
350 const auto otherPoint = m_action == Action::MoveLoopBegin
351 ? m_timeline->loopEnd()
352 : m_timeline->loopBegin();
353 if (control)
354 {
355 // no ctrl-press-hint when having ctrl pressed
356 delete m_hint;
357 m_hint = nullptr;
358 }
359 else
360 {
361 timeAtCursor = timeAtCursor.quantize(m_snapSize);
362 }
363 // Catch begin == end
364 if (timeAtCursor == otherPoint)
365 {
366 const int offset = control ? 1 : m_snapSize * TimePos::ticksPerBar();
367 if (m_action == Action::MoveLoopBegin) { timeAtCursor -= offset; }
368 else { timeAtCursor += offset; }
369 }
370 // Update m_action so we still move the correct point even if it is
371 // dragged past the other.
372 m_action = timeAtCursor < otherPoint ? Action::MoveLoopBegin : Action::MoveLoopEnd;
373 m_timeline->setLoopPoints(timeAtCursor, otherPoint);
374 update();
375 break;
376 }
377 case Action::MoveLoop:
378 {
379 const TimePos dragDelta = timeAtCursor - m_dragStartPos;
380 auto loopPos = m_oldLoopPos;
381 for (auto& point : loopPos)
382 {
383 point += dragDelta;
384 if (!control) { point = point.quantize(m_snapSize); }
385 }
386 m_timeline->setLoopPoints(loopPos[0], loopPos[1]);
387 break;
388 }
389 case Action::SelectSongClip:
390 emit regionSelectedFromPixels( m_initalXSelect , event->x() );
391 break;
392
393 default:
394 break;
395 }
396
397 if (event->buttons() == Qt::NoButton)
398 {
399 setCursor(QGuiApplication::keyboardModifiers().testFlag(Qt::ShiftModifier)
400 ? actionCursor(getLoopAction(event))
401 : Qt::ArrowCursor
402 );
403 }
404}

414void TimeLineWidget::contextMenuEvent(QContextMenuEvent* event)
415{
416 if (event->x() < m_xOffset) { return; }
417
418 auto menu = QMenu{};
419
420 menu.addAction(tr("Set loop begin here"), [this, event] {
421 auto begin = getClickedTime(event->x());
422 const auto end = m_timeline->loopEnd();
423 if (!QGuiApplication::keyboardModifiers().testFlag(Qt::ControlModifier)) { begin = begin.quantize(m_snapSize); }
424 if (begin == end) { m_timeline->setLoopEnd(end + m_snapSize * TimePos::ticksPerBar()); }
425 m_timeline->setLoopBegin(begin);
426 update();
427 });
428 menu.addAction(tr("Set loop end here"), [this, event] {
429 const auto begin = m_timeline->loopBegin();
430 auto end = getClickedTime(event->x());
431 if (!QGuiApplication::keyboardModifiers().testFlag(Qt::ControlModifier)) { end = end.quantize(m_snapSize); }
432 if (begin == end) { m_timeline->setLoopBegin(begin - m_snapSize * TimePos::ticksPerBar()); }
433 m_timeline->setLoopEnd(end);
434 update();
435 });
436
437 menu.addSeparator();
438
439 const auto loopMenu = menu.addMenu(tr("Loop edit mode (hold shift)"));
440 const auto loopMode = ConfigManager::inst()->value("app", "loopmarkermode", "dual");
441 const auto addLoopModeAction = [loopMenu, &loopMode](QString text, QString mode) {
442 const auto action = loopMenu->addAction(text, [mode] {
443 ConfigManager::inst()->setValue("app", "loopmarkermode", mode);
444 });
445 action->setCheckable(true);
446 if (loopMode == mode) { action->setChecked(true); }
447 };
448 addLoopModeAction(tr("Dual-button"), "dual");
449 addLoopModeAction(tr("Grab closest"), "closest");
450 addLoopModeAction(tr("Handles"), "handles");
451
452 menu.exec(event->globalPos());
453}
