67/**
68 * @brief Sets the inValue of an automation node
69 * @param Float value to be assigned
70*/
71void AutomationNode::setInValue(float value)
72{
73 m_inValue = value;
74
75 // Recalculate the tangents from neighbor nodes
76 AutomationClip::timeMap & tm = m_clip->getTimeMap();
77
78 // Get an iterator pointing to this node
79 AutomationClip::timeMap::iterator it = tm.lowerBound(m_pos);
80 // If it's not the first node, get the one immediately behind it
81 if (it != tm.begin()) { --it; }
82
83 // Generate tangents from the previously, current and next nodes
84 m_clip->generateTangents(it, 3);
85}
