47ControllerView::ControllerView( Controller * _model, QWidget * _parent ) :
48 QFrame( _parent ),
49 ModelView( _model, this ),
50 m_subWindow( nullptr ),
51 m_controllerDlg( nullptr ),
52 m_show( true )
53{
54 this->setFrameStyle( QFrame::StyledPanel );
55 this->setFrameShadow( QFrame::Raised );
56 setFocusPolicy(Qt::StrongFocus);
57
58 auto vBoxLayout = new QVBoxLayout(this);
59
60 auto hBox = new QHBoxLayout();
61 vBoxLayout->addLayout(hBox);
62
63 auto label = new QLabel("<b>" + _model->displayName() + "</b>", this);
64 QSizePolicy sizePolicy = label->sizePolicy();
65 sizePolicy.setHorizontalStretch(1);
66 label->setSizePolicy(sizePolicy);
67
68 hBox->addWidget(label);
69
70 auto controlsButton = new QPushButton(tr("Controls"), this);
71 connect( controlsButton, SIGNAL(clicked()), SLOT(editControls()));
72
73 hBox->addWidget(controlsButton);
74
75 m_nameLabel = new QLabel(_model->name(), this);
76 vBoxLayout->addWidget(m_nameLabel);
77
78
79 m_controllerDlg = getController()->createDialog( getGUI()->mainWindow()->workspace() );
80
81 m_subWindow = getGUI()->mainWindow()->addWindowedWidget( m_controllerDlg );
82
83 Qt::WindowFlags flags = m_subWindow->windowFlags();
84 flags &= ~Qt::WindowMaximizeButtonHint;
85 m_subWindow->setWindowFlags( flags );
86 m_subWindow->setFixedSize( m_subWindow->size() );
87
88 m_subWindow->setWindowIcon( m_controllerDlg->windowIcon() );
89
90 connect( m_controllerDlg, SIGNAL(closed()),
91 this, SLOT(closeControls()));
92
93 m_subWindow->hide();
94
95 setModel( _model );
96}

142void ControllerView::renameController()
143{
144 bool ok;
145 auto c = castModel<Controller>();
146 QString new_name = QInputDialog::getText( this,
147 tr( "Rename controller" ),
148 tr( "Enter the new name for this controller" ),
149 QLineEdit::Normal, c->name() , &ok );
150 if( ok && !new_name.isEmpty() )
151 {
152 c->setName( new_name );
153 if( getController()->type() == Controller::ControllerType::Lfo )
154 {
155 m_controllerDlg->setWindowTitle( tr( "LFO" ) + " (" + new_name + ")" );
156 }
157 m_nameLabel->setText( new_name );
158 }
159}

175void ControllerView::contextMenuEvent( QContextMenuEvent * )
176{
177 Controller* c = castModel<Controller>();
178 QPointer<CaptionMenu> contextMenu = new CaptionMenu(c->name(), this);
179 contextMenu->addAction(embed::getIconPixmap("arp_up"), tr("Move &up"), this, &ControllerView::moveUp);
180 contextMenu->addAction(embed::getIconPixmap("arp_down"), tr("Move &down"), this, &ControllerView::moveDown);
181 contextMenu->addSeparator();
182 contextMenu->addAction(
183 embed::getIconPixmap("cancel"), tr("&Remove this controller"), this, &ControllerView::removeController);
184 contextMenu->addAction( tr("Re&name this controller"), this, SLOT(renameController()));
185 contextMenu->addSeparator();
186 contextMenu->exec( QCursor::pos() );
187 delete contextMenu;
188}
