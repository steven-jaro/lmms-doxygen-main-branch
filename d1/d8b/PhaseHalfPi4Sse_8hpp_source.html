101/*
102==============================================================================
103Name: process_sample
104Description:
105 From one 4-channel input vector, generates two vectors with a pi/2 phase
106 shift between these signals.
107Input parameters:
108 - input: The input sample.
109Output parameters:
110 - out_0: Output sample, ahead.
111 - out_1: Output sample, late.
112Throws: Nothing
113==============================================================================
114*/
115
116template <int NC>
117void PhaseHalfPi4Sse <NC>::process_sample (__m128 &out_0, __m128 &out_1, __m128 input) noexcept
118{
119 out_0 = input; // Even coefs
120 out_1 = _mm_load_ps (_prev); // Odd coefs
121
122 StageProc4Sse <NBR_COEFS>::process_sample_neg (
123 NBR_COEFS, out_0, out_1, &_bifilter [_phase] [0]
124 );
125
126 _mm_store_ps (_prev, input);
127 _phase = 1 - _phase;
128}

132/*
133==============================================================================
134Name: process_block
135Description:
136 From a block of 4-channel input vectors, generates two blocks of vectors
137 with a pi/2 phase shift between these signals.
138 Input and output blocks may overlap, see assert() for details.
139Input parameters:
140 - in_ptr: Input array, containing nbr_spl samples.
141 - nbr_spl: Number of input samples to process, > 0
142Output parameters:
143 - out_0_ptr: Output sample array (ahead), capacity: nbr_spl samples.
144 - out_1_ptr: Output sample array (late), capacity: nbr_spl samples.
145Throws: Nothing
146==============================================================================
147*/
148
149template <int NC>
150void PhaseHalfPi4Sse <NC>::process_block (float out_0_ptr [], float out_1_ptr [], const float in_ptr [], long nbr_spl) noexcept
151{
152 assert (out_0_ptr != nullptr);
153 assert (out_1_ptr != nullptr);
154 assert (in_ptr != nullptr);
155 assert (out_0_ptr <= in_ptr || out_0_ptr >= in_ptr + nbr_spl * _nbr_chn);
156 assert (out_1_ptr <= in_ptr || out_1_ptr >= in_ptr + nbr_spl * _nbr_chn);
157 assert ( out_0_ptr + nbr_spl * _nbr_chn <= out_1_ptr
158 || out_1_ptr + nbr_spl * _nbr_chn <= out_0_ptr);
159 assert (nbr_spl > 0);
160
161 long pos = 0;
162 if (_phase == 1)
163 {
164 __m128 out_0;
165 __m128 out_1;
166 process_sample (out_0, out_1, _mm_loadu_ps (in_ptr));
167 _mm_storeu_ps (out_0_ptr, out_0);
168 _mm_storeu_ps (out_1_ptr, out_1);
169 ++ pos;
170 }
171
172 const long end = ((nbr_spl - pos) & -_nbr_phases) + pos;
173 __m128 prev = _mm_load_ps (_prev);
174 while (pos < end)
175 {
176 const auto ofs_0 = pos * _nbr_chn;
177 const __m128 input_0 = _mm_loadu_ps (in_ptr + ofs_0);
178 __m128 out_0 = input_0;
179 __m128 out_1 = prev;
180 StageProc4Sse <NBR_COEFS>::process_sample_neg (
181 NBR_COEFS, out_0, out_1, &_bifilter [0] [0]
182 );
183 _mm_storeu_ps (out_0_ptr + ofs_0, out_0);
184 _mm_storeu_ps (out_1_ptr + ofs_0, out_1);
185
186 const auto ofs_1 = ofs_0 + _nbr_chn;
187 const __m128 input_1 = _mm_loadu_ps (in_ptr + ofs_1);
188 out_0 = input_1;
189 out_1 = input_0; // prev
190 StageProc4Sse <NBR_COEFS>::process_sample_neg (
191 NBR_COEFS, out_0, out_1, &_bifilter [1] [0]
192 );
193 _mm_storeu_ps (out_0_ptr + ofs_1, out_0);
194 _mm_storeu_ps (out_1_ptr + ofs_1, out_1);
195
196 prev = input_1;
197 pos += 2;
198 }
199 _mm_store_ps (_prev, prev);
200
201 if (pos < nbr_spl)
202 {
203 assert (pos + 1 == nbr_spl);
204 __m128 out_0;
205 __m128 out_1;
206 const auto ofs = pos * _nbr_chn;
207 process_sample (out_0, out_1, _mm_loadu_ps (in_ptr + ofs));
208 _mm_storeu_ps (out_0_ptr + ofs, out_0);
209 _mm_storeu_ps (out_1_ptr + ofs, out_1);
210 }
211}

215/*
216==============================================================================
217Name: clear_buffers
218Description:
219 Clears filter memory, as if it processed silence since an infinite amount
220 of time.
221Throws: Nothing
222==============================================================================
223*/
224
225template <int NC>
226void PhaseHalfPi4Sse <NC>::clear_buffers () noexcept
227{
228 _phase = 0;
229 for (int i = 0; i < NBR_COEFS + 2; ++i)
230 {
231 _mm_store_ps (_bifilter [0] [i]._mem, _mm_setzero_ps ());
232 _mm_store_ps (_bifilter [1] [i]._mem, _mm_setzero_ps ());
233 }
234 _mm_store_ps (_prev, _mm_setzero_ps ());
235}
