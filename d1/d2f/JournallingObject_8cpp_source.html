90void JournallingObject::restoreState( const QDomElement & _this )
91{
92 SerializingObject::restoreState( _this );
93
94 saveJournallingState( false );
95
96 // search for journal-node
97 QDomNode node = _this.firstChild();
98 while( !node.isNull() )
99 {
100 if( node.isElement() && node.nodeName() == "journal" )
101 {
102 const jo_id_t new_id = node.toElement().attribute( "id" ).toInt();
103 if( new_id )
104 {
105 changeID( new_id );
106 }
107 }
108 node = node.nextSibling();
109 }
110
111 restoreJournallingState();
112}

117void JournallingObject::changeID( jo_id_t _id )
118{
119 if( id() != _id )
120 {
121 JournallingObject * jo = Engine::projectJournal()->
122 journallingObject( _id );
123 if( jo != nullptr )
124 {
125 QString used_by = jo->nodeName();
126 if( used_by == "automatablemodel" &&
127 dynamic_cast<AutomatableModel *>( jo ) )
128 {
129 used_by += ":" +
130 dynamic_cast<AutomatableModel *>( jo )->
131 displayName();
132 }
133 fprintf( stderr, "JO-ID %d already in use by %s!\n",
134 (int) _id, used_by.toUtf8().constData() );
135 return;
136 }
137
138 Engine::projectJournal()->freeID( m_id );
139 Engine::projectJournal()->reallocID( _id, this );
140 m_id = _id;
141 }
142}
