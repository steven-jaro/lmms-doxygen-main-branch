41class SharedMemoryData
42{
43public:
44 SharedMemoryData() noexcept;
45 SharedMemoryData(std::string&& key, bool readOnly);
46 SharedMemoryData(std::string&& key, std::size_t size, bool readOnly);
47 SharedMemoryData(std::size_t size, bool readOnly);
48 ~SharedMemoryData();
49
50 SharedMemoryData(SharedMemoryData&& other) noexcept;
51 SharedMemoryData& operator=(SharedMemoryData&& other) noexcept
52 {
53 auto temp = std::move(other);
54 swap(*this, temp);
55 return *this;
56 }
57
58 friend void swap(SharedMemoryData& a, SharedMemoryData& b) noexcept
59 {
60 using std::swap;
61 swap(a.m_key, b.m_key);
62 swap(a.m_impl, b.m_impl);
63 swap(a.m_ptr, b.m_ptr);
64 }
65
66 const std::string& key() const noexcept { return m_key; }
67 void* get() const noexcept { return m_ptr; }
68 std::size_t size_bytes() const noexcept;
69
70private:
71 std::string m_key;
72 std::unique_ptr<SharedMemoryImpl> m_impl;
73 void* m_ptr = nullptr;
74};

79template<typename T>
80class SharedMemory
81{
82 // This is stricter than necessary, but keeps things easy for now
83 static_assert(std::is_trivial_v<T>, "objects held in shared memory must be trivial");
84
85public:
86 SharedMemory() = default;
87 SharedMemory(SharedMemory&&) = default;
88 SharedMemory& operator=(SharedMemory&&) = default;
89
90 void attach(std::string key)
91 {
92 m_data = detail::SharedMemoryData{std::move(key), std::is_const_v<T>};
93 }
94
95 void create(std::string key)
96 {
97 m_data = detail::SharedMemoryData{std::move(key), sizeof(T), std::is_const_v<T>};
98 }
99
100 void create()
101 {
102 m_data = detail::SharedMemoryData{sizeof(T), std::is_const_v<T>};
103 }
104
105 void detach() noexcept
106 {
107 m_data = detail::SharedMemoryData{};
108 }
109
110 const std::string& key() const noexcept { return m_data.key(); }
111 T* get() const noexcept { return static_cast<T*>(m_data.get()); }
112
113 std::size_t size() const noexcept { return get() ? 1 : 0; }
114 std::size_t size_bytes() const noexcept { return get() ? sizeof(T) : 0; }
115
116 T* operator->() const noexcept { return get(); }
117 T& operator*() const noexcept { return *get(); }
118 explicit operator bool() const noexcept { return get() != nullptr; }
119
120private:
121 detail::SharedMemoryData m_data;
122};
