95 JackEngineControl(bool sync, bool temporary, long timeout, bool rt, long priority, bool verbose, jack_timer_type_t clock, const char* server_name)
96 {
97 static_assert(offsetof(JackEngineControl, fTransport) % sizeof(UInt32) == 0,
98 "fTransport must be aligned within JackEngineControl");
99 static_assert(offsetof(JackEngineControl, fFrameTimer) % sizeof(UInt32) == 0,
100 "fFrameTimer must be aligned within JackEngineControl");
101 fBufferSize = 512;
102 fSampleRate = 48000;
103 fPeriodUsecs = jack_time_t(1000000.f / fSampleRate * fBufferSize);
104 fSyncMode = sync;
105 fTemporary = temporary;
106 fTimeOut = (timeout > 0);
107 fTimeOutUsecs = timeout * 1000;
108 fRealTime = rt;
109 fSavedRealTime = false;
110 fServerPriority = priority;
111
112 #ifdef WIN32
113 fClientPriority = (rt) ? priority - 3 : 0;
114 #else
115 fClientPriority = (rt) ? priority - 5 : 0;
116 #endif
117 fMaxClientPriority = (rt) ? priority - 1 : 0;
118 fVerbose = verbose;
119 fPrevCycleTime = 0;
120 fCurCycleTime = 0;
121 fSpareUsecs = 0;
122 fMaxUsecs = 0;
123 ResetRollingUsecs();
124 strncpy(fServerName, server_name, sizeof(fServerName));
125 fServerName[sizeof(fServerName) - 1] = 0;
126 fCPULoad = 0.f;
127 fPeriod = 0;
128 fComputation = 0;
129 fConstraint = 0;
130 fMaxDelayedUsecs = 0.f;
