210inline sample_t EnvelopeAndLfoParameters::lfoShapeSample( fpp_t _frame_offset )
211{
212 f_cnt_t frame = ( m_lfoFrame + _frame_offset ) % m_lfoOscillationFrames;
213 const float phase = frame / static_cast<float>(
214 m_lfoOscillationFrames );
215 sample_t shape_sample;
216 switch( static_cast<LfoShape>(m_lfoWaveModel.value()) )
217 {
218 case LfoShape::TriangleWave:
219 shape_sample = Oscillator::triangleSample( phase );
220 break;
221 case LfoShape::SquareWave:
222 shape_sample = Oscillator::squareSample( phase );
223 break;
224 case LfoShape::SawWave:
225 shape_sample = Oscillator::sawSample( phase );
226 break;
227 case LfoShape::UserDefinedWave:
228 shape_sample = Oscillator::userWaveSample(m_userWave.get(), phase);
229 break;
230 case LfoShape::RandomWave:
231 if( frame == 0 )
232 {
233 m_random = Oscillator::noiseSample( 0.0f );
234 }
235 shape_sample = m_random;
236 break;
237 case LfoShape::SineWave:
238 default:
239 shape_sample = Oscillator::sinSample( phase );
240 break;
241 }
242 return shape_sample * m_lfoAmount;
243}

261inline void EnvelopeAndLfoParameters::fillLfoLevel( float * _buf,
262 f_cnt_t _frame,
263 const fpp_t _frames )
264{
265 if( m_lfoAmountIsZero || _frame <= m_lfoPredelayFrames )
266 {
267 for( fpp_t offset = 0; offset < _frames; ++offset )
268 {
269 *_buf++ = 0.0f;
270 }
271 return;
272 }
273 _frame -= m_lfoPredelayFrames;
274
275 if( m_bad_lfoShapeData )
276 {
277 updateLfoShapeData();
278 }
279
280 fpp_t offset = 0;
281 const float lafI = 1.0f / std::max(minimumFrames, m_lfoAttackFrames);
282 for( ; offset < _frames && _frame < m_lfoAttackFrames; ++offset,
283 ++_frame )
284 {
285 *_buf++ = m_lfoShapeData[offset] * _frame * lafI;
286 }
287 for( ; offset < _frames; ++offset )
288 {
289 *_buf++ = m_lfoShapeData[offset];
290 }
291}

296void EnvelopeAndLfoParameters::fillLevel( float * _buf, f_cnt_t _frame,
297 const f_cnt_t _release_begin,
298 const fpp_t _frames )
299{
300 QMutexLocker m(&m_paramMutex);
301
302 fillLfoLevel( _buf, _frame, _frames );
303
304 for( fpp_t offset = 0; offset < _frames; ++offset, ++_buf, ++_frame )
305 {
306 float env_level;
307 if( _frame < _release_begin )
308 {
309 if( _frame < m_pahdFrames )
310 {
311 env_level = m_pahdEnv[_frame];
312 }
313 else
314 {
315 env_level = m_sustainLevel;
316 }
317 }
318 else if( ( _frame - _release_begin ) < m_rFrames )
319 {
320 env_level = m_rEnv[_frame - _release_begin] *
321 ( ( _release_begin < m_pahdFrames ) ?
322 m_pahdEnv[_release_begin] : m_sustainLevel );
323 }
324 else
325 {
326 env_level = 0.0f;
327 }
328
329 // at this point, *_buf is LFO level
330 *_buf = m_controlEnvAmountModel.value() ?
331 env_level * ( 0.5f + *_buf ) :
332 env_level + *_buf;
333 }
334}

402
403void EnvelopeAndLfoParameters::updateSampleVars()
404{
405 QMutexLocker m(&m_paramMutex);
406
407 const float frames_per_env_seg = SECS_PER_ENV_SEGMENT *
408 Engine::audioEngine()->outputSampleRate();
409
410 // TODO: Remove the expKnobVals, time should be linear
411 const auto predelay_frames = static_cast<f_cnt_t>(frames_per_env_seg * expKnobVal(m_predelayModel.value()));
412
413 const f_cnt_t attack_frames = std::max(minimumFrames,
414 static_cast<f_cnt_t>(frames_per_env_seg *
415 expKnobVal(m_attackModel.value())));
416
417 const auto hold_frames = static_cast<f_cnt_t>(frames_per_env_seg * expKnobVal(m_holdModel.value()));
418
419 const f_cnt_t decay_frames = std::max(minimumFrames,
420 static_cast<f_cnt_t>(frames_per_env_seg *
421 expKnobVal(m_decayModel.value() *
422 (1 - m_sustainModel.value()))));
423
424 m_sustainLevel = m_sustainModel.value();
425 m_amount = m_amountModel.value();
426 if( m_amount >= 0 )
427 {
428 m_amountAdd = ( 1.0f - m_amount ) * m_valueForZeroAmount;
429 }
430 else
431 {
432 m_amountAdd = m_valueForZeroAmount;
433 }
434
435 m_pahdFrames = predelay_frames + attack_frames + hold_frames +
436 decay_frames;
437 m_rFrames = static_cast<f_cnt_t>( frames_per_env_seg *
438 expKnobVal( m_releaseModel.value() ) );
439 m_rFrames = std::max(minimumFrames, m_rFrames);
440
441 if( static_cast<int>( floorf( m_amount * 1000.0f ) ) == 0 )
442 {
443 m_rFrames = minimumFrames;
444 }
445
446 // if the buffers are too small, make bigger ones - so we only alloc new memory when necessary
447 if( m_pahdBufSize < m_pahdFrames )
448 {
449 sample_t * tmp = m_pahdEnv;
450 m_pahdEnv = new sample_t[m_pahdFrames];
451 delete[] tmp;
452 m_pahdBufSize = m_pahdFrames;
453 }
454 if( m_rBufSize < m_rFrames )
455 {
456 sample_t * tmp = m_rEnv;
457 m_rEnv = new sample_t[m_rFrames];
458 delete[] tmp;
459 m_rBufSize = m_rFrames;
460 }
461
462 const float aa = m_amountAdd;
463 for( f_cnt_t i = 0; i < predelay_frames; ++i )
464 {
465 m_pahdEnv[i] = aa;
466 }
467
468 f_cnt_t add = predelay_frames;
469
470 const float afI = ( 1.0f / attack_frames ) * m_amount;
471 for( f_cnt_t i = 0; i < attack_frames; ++i )
472 {
473 m_pahdEnv[add+i] = i * afI + aa;
474 }
475
476 add += attack_frames;
477 const float amsum = m_amount + m_amountAdd;
478 for( f_cnt_t i = 0; i < hold_frames; ++i )
479 {
480 m_pahdEnv[add + i] = amsum;
481 }
482
483 add += hold_frames;
484 const float dfI = ( 1.0 / decay_frames ) * ( m_sustainLevel -1 ) * m_amount;
485 for( f_cnt_t i = 0; i < decay_frames; ++i )
486 {
487/*
488 m_pahdEnv[add + i] = ( m_sustainLevel + ( 1.0f -
489 (float)i / decay_frames ) *
490 ( 1.0f - m_sustainLevel ) ) *
491 m_amount + m_amountAdd;
492*/
493 m_pahdEnv[add + i] = amsum + i*dfI;
494 }
495
496 const float rfI = ( 1.0f / m_rFrames ) * m_amount;
497 for( f_cnt_t i = 0; i < m_rFrames; ++i )
498 {
499 m_rEnv[i] = (float)( m_rFrames - i ) * rfI;
500 }
501
502 // save this calculation in real-time-part
503 m_sustainLevel = m_sustainLevel * m_amount + m_amountAdd;
504
505
506 const float frames_per_lfo_oscillation = SECS_PER_LFO_OSCILLATION *
507 Engine::audioEngine()->outputSampleRate();
508 m_lfoPredelayFrames = static_cast<f_cnt_t>( frames_per_lfo_oscillation *
509 expKnobVal( m_lfoPredelayModel.value() ) );
510 m_lfoAttackFrames = static_cast<f_cnt_t>( frames_per_lfo_oscillation *
511 expKnobVal( m_lfoAttackModel.value() ) );
512 m_lfoOscillationFrames = static_cast<f_cnt_t>(
513 frames_per_lfo_oscillation *
514 m_lfoSpeedModel.value() );
515 if( m_x100Model.value() )
516 {
517 m_lfoOscillationFrames /= 100;
518 }
519 m_lfoAmount = m_lfoAmountModel.value() * 0.5f;
520
521 m_used = true;
522 if( static_cast<int>( floorf( m_lfoAmount * 1000.0f ) ) == 0 )
523 {
524 m_lfoAmountIsZero = true;
525 if( static_cast<int>( floorf( m_amount * 1000.0f ) ) == 0 )
526 {
527 m_used = false;
528 }
529 }
530 else
531 {
532 m_lfoAmountIsZero = false;
533 }
534
535 m_bad_lfoShapeData = true;
536
537 emit dataChanged();
538
539}
