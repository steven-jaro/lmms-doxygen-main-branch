30int JackWinNamedPipeNotifyChannel::Open(const char* name)
31{
32 jack_log("JackWinNamedPipeNotifyChannel::Open name = %s", name);
33
34 // Connect to client listen pipe
35 if (fNotifyPipe.Connect(jack_client_dir, name, 0) < 0) {
36 jack_error("Cannot connect client pipe");
37 return -1;
38 }
39 // TODO : use a time out for notifications
40 fNotifyPipe.SetReadTimeOut(SOCKET_TIME_OUT);
41 return 0;
42}

50void JackWinNamedPipeNotifyChannel::ClientNotify(int refnum, const char* name, int notify, int sync, const char* message, int value1, int value2, int* result)
51{
52 JackClientNotification event(name, refnum, notify, sync, message, value1, value2);
53 JackResult res;
54
55 // Send notification
56 if (event.Write(&fNotifyPipe) < 0) {
57 jack_error("Could not write notification");
58 *result = -1;
59 return;
60 }
61
62 // Read the result in "synchronous" mode only
63 if (sync) {
64 // Get result : use a time out
65 if (res.Read(&fNotifyPipe) < 0) {
66 jack_error("Could not read result");
67 *result = -1;
68 } else {
69 *result = res.fResult;
70 }
71 } else {
72 *result = 0;
73 }
74}
