38PatternClip::PatternClip(Track* track) :
39 Clip(track)
40{
41 bar_t t = Engine::patternStore()->lengthOfPattern(patternIndex());
42 if( t > 0 )
43 {
44 saveJournallingState( false );
45 changeLength( TimePos( t, 0 ) );
46 restoreJournallingState();
47 }

50void PatternClip::saveSettings(QDomDocument& doc, QDomElement& element)
51{
52 element.setAttribute( "name", name() );
53 if( element.parentNode().nodeName() == "clipboard" )
54 {
55 element.setAttribute( "pos", -1 );
56 }
57 else
58 {
59 element.setAttribute( "pos", startPosition() );
60 }
61 element.setAttribute( "len", length() );
62 element.setAttribute("off", startTimeOffset());
63 element.setAttribute( "muted", isMuted() );
64 element.setAttribute("autoresize", QString::number(getAutoResize()));
65 if (const auto& c = color())
66 {
67 element.setAttribute("color", c->name());
68 }
69}

74void PatternClip::loadSettings(const QDomElement& element)
75{
76 setName( element.attribute( "name" ) );
77 if( element.attribute( "pos" ).toInt() >= 0 )
78 {
79 movePosition( element.attribute( "pos" ).toInt() );
80 }
81 changeLength( element.attribute( "len" ).toInt() );
82 setAutoResize(element.attribute("autoresize", "1").toInt());
83 setStartTimeOffset(element.attribute("off").toInt());
84 if (static_cast<bool>(element.attribute("muted").toInt()) != isMuted())
85 {
86 toggleMute();
87 }
88
89 if (element.hasAttribute("color"))
90 {
91 if (!element.hasAttribute("usestyle"))
92 {
93 // for colors saved in 1.3-onwards
94 setColor(QColor{element.attribute("color")});
95 }
96 else if (element.attribute("usestyle").toUInt() == 0)
97 {
98 // for colors saved before 1.3
99 setColor(QColor{element.attribute("color").toUInt()});
100 }
101 }
102}
