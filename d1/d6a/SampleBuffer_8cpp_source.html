39SampleBuffer::SampleBuffer(const QString& audioFile)
40{
41 if (audioFile.isEmpty()) { throw std::runtime_error{"Failure loading audio file: Audio file path is empty."}; }
42 const auto absolutePath = PathUtil::toAbsolute(audioFile);
43
44 if (auto decodedResult = SampleDecoder::decode(absolutePath))
45 {
46 auto& [data, sampleRate] = *decodedResult;
47 m_data = std::move(data);
48 m_sampleRate = sampleRate;
49 m_audioFile = PathUtil::toShortestRelative(audioFile);
50 return;
51 }
52
53 throw std::runtime_error{
54 "Failed to decode audio file: Either the audio codec is unsupported, or the file is corrupted."};
55}
