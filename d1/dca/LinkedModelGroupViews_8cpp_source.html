58void LinkedModelGroupView::modelChanged(LinkedModelGroup *group)
59{
60 // reconnect models
61 group->foreach_model([this](const std::string& str,
62 const LinkedModelGroup::ModelInfo& minf)
63 {
64 auto itr = m_widgets.find(str);
65 // in case there are new or deleted widgets, the subclass has already
66 // modified m_widgets, so this will go into the else case
67 if (itr == m_widgets.end())
68 {
69 // no widget? this can happen when the whole view is being destroyed
70 // (for some strange reasons)
71 }
72 else
73 {
74 itr->second->setModel(minf.m_model);
75 }
76 });
77
78 m_model = group;
79}

84void LinkedModelGroupView::addControl(Control* ctrl, const std::string& id,
85 const std::string &display, bool removable)
86{
87 if (ctrl)
88 {
89 auto box = new QWidget(this);
90 auto boxLayout = new QHBoxLayout(box);
91 boxLayout->addWidget(ctrl->topWidget());
92
93 if (removable)
94 {
95 auto removeBtn = new QPushButton;
96 removeBtn->setIcon( embed::getIconPixmap( "discard" ) );
97 QObject::connect(removeBtn, &QPushButton::clicked,
98 this, [this,ctrl](bool){
99 AutomatableModel* controlModel = ctrl->model();
100 // remove control out of model group
101 // (will also remove it from the UI)
102 m_model->removeControl(controlModel);
103 // delete model (includes disconnecting all connections)
104 delete controlModel;
105 },
106 Qt::DirectConnection);
107 boxLayout->addWidget(removeBtn);
108 }
109
110 // required, so the Layout knows how to sort/filter widgets by string
111 box->setObjectName(QString::fromStdString(display));
112 m_layout->addWidget(box);
113
114 // take ownership of control and add it
115 m_widgets.emplace(id, std::unique_ptr<Control>(ctrl));
116 }
117
118 if (isHidden()) { setHidden(false); }
119}

124void LinkedModelGroupView::removeControl(const QString& key)
125{
126 auto itr = m_widgets.find(key.toStdString());
127 if (itr != m_widgets.end())
128 {
129 QLayoutItem* item = m_layout->itemByString(key);
130 Q_ASSERT(!!item);
131 QWidget* wdg = item->widget();
132 Q_ASSERT(!!wdg);
133
134 // remove item from layout
135 m_layout->removeItem(item);
136 // the widget still exists and is visible - remove it now
137 delete wdg;
138 // erase widget pointer from dictionary
139 m_widgets.erase(itr);
140 // repaint immediately, so we don't have dangling model views
141 m_layout->update();
142 }
143}
