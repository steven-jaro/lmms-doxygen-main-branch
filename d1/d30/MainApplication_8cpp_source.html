60bool MainApplication::event(QEvent* event)
61{
62 switch(event->type())
63 {
64 case QEvent::FileOpen:
65 {
66 auto fileEvent = static_cast<QFileOpenEvent*>(event);
67 // Handle the project file
68 m_queuedFile = fileEvent->file();
69 if(Engine::getSong())
70 {
71 if(getGUI()->mainWindow()->mayChangeProject(true))
72 {
73 qDebug() << "Loading file " << m_queuedFile;
74 Engine::getSong()->loadProject(m_queuedFile);
75 }
76 }
77 else
78 {
79 qDebug() << "Queuing file " << m_queuedFile;
80 }
81 return true;
82 }
83 default:
84 return QApplication::event(event);
85 }
86}

89// This can be moved into nativeEventFilter once Qt4 support has been dropped
90bool MainApplication::winEventFilter(MSG* msg, long* result)
91{
92 switch(msg->message)
93 {
94 case WM_STYLECHANGING:
95 if (msg->wParam == static_cast<WPARAM>(GWL_EXSTYLE))
96 {
97 // Prevent plugins making the main window transparent
98 STYLESTRUCT * style = reinterpret_cast<STYLESTRUCT *>(msg->lParam);
99 if(!(style->styleOld & WS_EX_LAYERED))
100 {
101 style->styleNew &= ~WS_EX_LAYERED;
102 }
103 *result = 0;
104 return true;
105 }
106 return false;
107 default:
108 return false;
109 }
110}

112bool MainApplication::nativeEventFilter(const QByteArray& eventType,
113 void* message, long* result)
114{
