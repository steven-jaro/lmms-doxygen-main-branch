94bool Effect::processAudioBuffer(SampleFrame* buf, const fpp_t frames)
95{
96 if (!isOkay() || dontRun() || !isEnabled() || !isRunning())
97 {
98 processBypassedImpl();
99 return false;
100 }
101
102 const auto status = processImpl(buf, frames);
103 switch (status)
104 {
105 case ProcessStatus::Continue:
106 break;
107 case ProcessStatus::ContinueIfNotQuiet:
108 handleAutoQuit({buf, frames});
109 break;
110 case ProcessStatus::Sleep:
111 return false;
112 default:
113 break;
114 }
115
116 return isRunning();
117}

145void Effect::handleAutoQuit(std::span<const SampleFrame> output)
146{
147 if (!m_autoQuitEnabled)
148 {
149 return;
150 }
151
152 /*
153 * In the past, the RMS was calculated then compared with a threshold of 10^(-10).
154 * Now we use a different algorithm to determine whether a buffer is non-quiet, so
155 * a new threshold is needed for the best compatibility. The following is how it's derived.
156 *
157 * Old method:
158 * RMS = average (L^2 + R^2) across stereo buffer.
159 * RMS threshold = 10^(-10)
160 *
161 * So for a single channel, it would be:
162 * RMS/2 = average M^2 across single channel buffer.
163 * RMS/2 threshold = 5^(-11)
164 *
165 * The new algorithm for determining whether a buffer is non-silent compares M with the threshold,
166 * not M^2, so the square root of M^2's threshold should give us the most compatible threshold for
167 * the new algorithm:
168 *
169 * (RMS/2)^0.5 = (5^(-11))^0.5 = 0.0001431 (approx.)
170 *
171 * In practice though, the exact value shouldn't really matter so long as it's sufficiently small.
172 */
173 static constexpr auto threshold = 0.0001431f;
174
175 // Check whether we need to continue processing input. Restart the
176 // counter if the threshold has been exceeded.
177
178 for (const SampleFrame& frame : output)
179 {
180 const auto abs = frame.abs();
181 if (abs.left() >= threshold || abs.right() >= threshold)
182 {
183 // The output buffer is not quiet
184 m_quietBufferCount = 0;
185 return;
186 }
187 }
188
189 // The output buffer is quiet, so check if auto-quit should be activated yet
190 if (++m_quietBufferCount > timeout())
191 {
192 // Activate auto-quit
193 stopRunning();
194 }
195}
