74==============================================================================
75Name: set_coefs
76Description:
77 Sets filter coefficients. Generate them with the PolyphaseIir2Designer
78 class.
79 Call this function before doing any processing.
80Input parameters:
81 - coef_arr: Array of coefficients. There should be as many coefficients as

83Throws: Nothing
84==============================================================================

86
87template <int NC>
88void Downsampler2xF64Sse2 <NC>::set_coefs (const double coef_arr []) noexcept
89{
90 assert (coef_arr != nullptr);
91
92 for (int i = 0; i < NBR_COEFS; ++i)
93 {
94 const int stage = (i / _stage_width) + 1;

96 _filter [stage]._coef [pos] = DataType (coef_arr [i]);

102/*
103==============================================================================
104Name: process_sample
105Description:
106 Downsamples (x2) one pair of samples, to generate one output sample.
107Input parameters:
108 - in_ptr: pointer on the two samples to decimate
109Returns: Samplerate-reduced sample.
110Throws: Nothing
111==============================================================================
112*/
113
114template <int NC>
115double Downsampler2xF64Sse2 <NC>::process_sample (const double in_ptr [2]) noexcept
116{
117 assert (in_ptr != nullptr);
118
119 auto x = _mm_loadu_pd (in_ptr);
120 StageProcF64Sse2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
121 x = _mm_add_sd (x, _mm_shuffle_pd (x, x, 1));
122 x = _mm_mul_sd (x, _mm_set_sd (0.5f));
123
124 return _mm_cvtsd_f64 (x);
125}

146{

163/*
164==============================================================================
165Name: process_sample_split
166Description:
167 Split (spectrum-wise) in half a pair of samples. The lower part of the
168 spectrum is a classic downsampling, equivalent to the output of
169 process_sample().
170 The higher part is the complementary signal: original filter response
171 is flipped from left to right, becoming a high-pass filter with the same
172 cutoff frequency. This signal is then critically sampled (decimation by 2),
173 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
174Input parameters:
175 - in_ptr: pointer on the pair of input samples
176Output parameters:
177 - low: output sample, lower part of the spectrum (downsampling)
178 - high: output sample, higher part of the spectrum.
179Throws: Nothing
180==============================================================================
181*/
182
183template <int NC>
184void Downsampler2xF64Sse2 <NC>::process_sample_split (double &low, double &high, const double in_ptr [2]) noexcept
185{
186 assert (in_ptr != nullptr);
187
188 auto x = _mm_loadu_pd (in_ptr);
189 StageProcF64Sse2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
190 x = _mm_mul_pd (x, _mm_set1_pd (0.5f));
191 const auto xr = _mm_shuffle_pd (x, x, 1);
192 low = _mm_cvtsd_f64 (_mm_add_sd (xr, x));
193 high = _mm_cvtsd_f64 (_mm_sub_sd (xr, x));
194}

198/*
199==============================================================================
200Name: process_block_split
201Description:
202 Split (spectrum-wise) in half a block of samples. The lower part of the
203 spectrum is a classic downsampling, equivalent to the output of
204 process_block().
205 The higher part is the complementary signal: original filter response
206 is flipped from left to right, becoming a high-pass filter with the same
207 cutoff frequency. This signal is then critically sampled (decimation by 2),
208 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
209 Input and output blocks may overlap, see assert() for details.
210Input parameters:
211 - in_ptr: Input array, containing nbr_spl * 2 samples.
212 - nbr_spl: Number of samples for each output, > 0
213Output parameters:
214 - out_l_ptr: Array for the output samples, lower part of the spectrum
215 (downsampling). Capacity: nbr_spl samples.
216 - out_h_ptr: Array for the output samples, higher part of the spectrum.
217 Capacity: nbr_spl samples.
218Throws: Nothing
219==============================================================================
220*/
221
222template <int NC>
223void Downsampler2xF64Sse2 <NC>::process_block_split (double out_l_ptr [], double out_h_ptr [], const double in_ptr [], long nbr_spl) noexcept
224{
225 assert (in_ptr != nullptr);
226 assert (out_l_ptr != nullptr);
227 assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl * 2);
228 assert (out_h_ptr != nullptr);
229 assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl * 2);
230 assert (out_h_ptr != out_l_ptr);
231 assert (nbr_spl > 0);
232
233 const long n2 = process_block_double (
234 out_l_ptr, out_h_ptr, in_ptr, nbr_spl, store_low, store_high
235 );
236 if (n2 < nbr_spl)
237 {
238 process_sample_split (out_l_ptr [n2], out_h_ptr [n2], in_ptr + n2 * 2);
239 }
240}

280template <int NC>
281template <typename FL, typename FH>
282long Downsampler2xF64Sse2 <NC>::process_block_double (double out_l_ptr [], double out_h_ptr [], const double in_ptr [], long nbr_spl, FL fnc_l, FH fnc_h) noexcept
283{
284 const auto half = _mm_set1_pd (0.5f);
285
286 const long n2 = nbr_spl & ~(2-1);
287 for (long pos = 0; pos < n2; pos += 2)
288 {
289 auto x_0 = _mm_loadu_pd (in_ptr + pos * 2);
290 auto x_1 = _mm_loadu_pd (in_ptr + pos * 2 + 2);
291 StageProcF64Sse2 <_nbr_stages>::process_sample_pos (x_0, _filter.data ());
292 StageProcF64Sse2 <_nbr_stages>::process_sample_pos (x_1, _filter.data ());
293 const auto odd = _mm_unpacklo_pd (x_0, x_1);
294 const auto even = _mm_unpackhi_pd (x_0, x_1);
295 fnc_l (out_l_ptr + pos, even, odd, half);
296 fnc_h (out_h_ptr + pos, even, odd, half);
297 }
298
299 return n2;
300}
