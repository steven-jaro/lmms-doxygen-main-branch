144QMap<QString, QLayoutItem*>::const_iterator
145ControlLayout::pairAt(int index) const
146{
147 if (index < 0) { return m_itemMap.cend(); }
148
149 auto skip = [&](QLayoutItem* item) -> bool
150 {
151 return item->widget()->objectName() == s_searchBarName;
152 };
153
154 QMap<QString, QLayoutItem*>::const_iterator itr = m_itemMap.cbegin();
155 for (; itr != m_itemMap.cend() && (index > 0 || skip(itr.value())); ++itr)
156 {
157 if(!skip(itr.value())) { index--; }
158 }
159 return itr;
160}

162// linear time :-(
163QLayoutItem *ControlLayout::itemAt(int index) const
164{
165 auto itr = pairAt(index);
166 return (itr == m_itemMap.end()) ? nullptr : itr.value();
167}

214QSize ControlLayout::minimumSize() const
215{
216 // original formula from Qt's FlowLayout example:
217 // get maximum height and width for all children.
218 // as Qt will later call heightForWidth, only the width here really matters
219 QSize size;
220 for (const QLayoutItem *item : std::as_const(m_itemMap))
221 {
222 size = size.expandedTo(item->minimumSize());
223 }
224 const QMargins margins = contentsMargins();
225 size += QSize(margins.left() + margins.right(), margins.top() + margins.bottom());
226
227 // the original formula would leed to ~1 widget per row
228 // bash it at least to 400 so we have ~4 knobs per row
229 size.setWidth(qMax(size.width(), m_minWidth));
230 return size;
231}

233int ControlLayout::doLayout(const QRect &rect, bool testOnly) const
234{
235 int left, top, right, bottom;
236 getContentsMargins(&left, &top, &right, &bottom);
237 QRect effectiveRect = rect.adjusted(+left, +top, -right, -bottom);
238 int x = effectiveRect.x();
239 int y = effectiveRect.y();
240 int lineHeight = 0;
241
242 const QString filterText = m_searchBar->text();
243 bool first = true;
244
245 QMapIterator<QString, QLayoutItem*> itr(m_itemMap);
246 while (itr.hasNext())
247 {
248 itr.next();
249 QLayoutItem* item = itr.value();
250 QWidget *wid = item->widget();
251 if (wid)
252 {
253 if ( first || // do not filter search bar
254 filterText.isEmpty() || // no filter - pass all
255 itr.key().contains(filterText, Qt::CaseInsensitive))
256 {
257 if (first)
258 {
259 // for the search bar, only show it if there are at least
260 // five control widgets (i.e. at least 6 widgets)
261 if (m_itemMap.size() > 5) { wid->show(); }
262 else { wid->hide(); }
263 }
264 else { wid->show(); }
265
266 int spaceX = horizontalSpacing();
267 if (spaceX == -1)
268 {
269 spaceX = wid->style()->layoutSpacing(
270 QSizePolicy::PushButton, QSizePolicy::PushButton, Qt::Horizontal);
271 }
272 int spaceY = verticalSpacing();
273 if (spaceY == -1)
274 {
275 spaceY = wid->style()->layoutSpacing(
276 QSizePolicy::PushButton, QSizePolicy::PushButton, Qt::Vertical);
277 }
278 int nextX = x + item->sizeHint().width() + spaceX;
279 if (nextX - spaceX > effectiveRect.right() && lineHeight > 0)
280 {
281 x = effectiveRect.x();
282 y = y + lineHeight + spaceY;
283 nextX = x + item->sizeHint().width() + spaceX;
284 lineHeight = 0;
285 }
286
287 if (!testOnly)
288 {
289 item->setGeometry(QRect(QPoint(x, y), item->sizeHint()));
290 }
291
292 x = nextX;
293 lineHeight = qMax(lineHeight, item->sizeHint().height());
294 first = false;
295 }
296 else
297 {
298 wid->hide();
299 }
300 }
301 }
302 return y + lineHeight - rect.y() + bottom;
303}

305int ControlLayout::smartSpacing(QStyle::PixelMetric pm) const
306{
307 QObject *parent = this->parent();
308 if (!parent) { return -1; }
309 else if (parent->isWidgetType())
310 {
311 auto pw = static_cast<QWidget*>(parent);
312 return pw->style()->pixelMetric(pm, nullptr, pw);
313 }
314 else { return static_cast<QLayout *>(parent)->spacing(); }
315}
