68{
69 _this.setTagName( classNodeName() );
70 _this.setAttribute( "type", nodeName() );

73 m_tracksMutex.lockForRead();

84void TrackContainer::loadSettings( const QDomElement & _this )
85{
86 bool journalRestore = _this.parentNode().nodeName() == "journaldata";
87 if( journalRestore )
88 {
89 clearAllTracks();
90 }
91
92 static QProgressDialog * pd = nullptr;
93 bool was_null = ( pd == nullptr );
94 if (!journalRestore && gui::getGUI() != nullptr)
95 {
96 if( pd == nullptr )
97 {
98 pd = new QProgressDialog( tr( "Loading project..." ),
99 tr( "Cancel" ), 0,
100 Engine::getSong()->getLoadingTrackCount(),
101 gui::getGUI()->mainWindow());
102 pd->setWindowModality( Qt::ApplicationModal );
103 pd->setWindowTitle( tr( "Please wait..." ) );
104 pd->show();
105 }
106 }
107
108 QDomNode node = _this.firstChild();
109 while( !node.isNull() )
110 {
111 if( pd != nullptr )
112 {
113 pd->setValue( pd->value() + 1 );
114 QCoreApplication::instance()->processEvents(
115 QEventLoop::AllEvents, 100 );
116 if( pd->wasCanceled() )
117 {
118 if (gui::getGUI() != nullptr)
119 {
120 gui::TextFloat::displayMessage( tr( "Loading cancelled" ),
121 tr( "Project loading was cancelled." ),
122 embed::getIconPixmap( "project_file", 24, 24 ),
123 2000 );
124 }
125 Engine::getSong()->loadingCancelled();
126 break;
127 }
128 }
129
130 if( node.isElement() &&
131 !node.toElement().attribute( "metadata" ).toInt() )
132 {
133 QString trackName = node.toElement().hasAttribute( "name" ) ?
134 node.toElement().attribute( "name" ) :
135 node.firstChild().toElement().attribute( "name" );
136 if( pd != nullptr )
137 {
138 pd->setLabelText( tr("Loading Track %1 (%2/Total %3)").arg( trackName ).
139 arg( pd->value() + 1 ).arg( Engine::getSong()->getLoadingTrackCount() ) );
140 }
141 Track::create( node.toElement(), this );
142 }
143 node = node.nextSibling();
144 }
145
146 if( pd != nullptr )
147 {
148 if( was_null )
149 {
150 delete pd;
151 pd = nullptr;
152 }
153 }
154}

159int TrackContainer::countTracks( Track::Type _tt ) const
160{
161 int cnt = 0;
162 m_tracksMutex.lockForRead();
163 for (const auto& track : m_tracks)
164 {
165 if (track->type() == _tt || _tt == Track::Type::Count)
166 {
167 ++cnt;
168 }
169 }
170 m_tracksMutex.unlock();
171 return( cnt );
172}

193void TrackContainer::removeTrack( Track * _track )
194{
195 // need a read locker to ensure that m_tracks doesn't change after reading index.
196 // After checking that index != -1, we need to upgrade the lock to a write locker before changing m_tracks.
197 // But since Qt offers no function to promote a read lock to a write lock, we must start with the write locker.
198 QWriteLocker lockTracksAccess(&m_tracksMutex);
199 auto it = std::find(m_tracks.begin(), m_tracks.end(), _track);
200 if (it != m_tracks.end())
201 {
202 // If the track is solo, all other tracks are muted. Change this before removing the solo track:
203 if (_track->isSolo()) {
204 _track->setSolo(false);
205 }
206 m_tracks.erase(it);
207 lockTracksAccess.unlock();
208
209 if( Engine::getSong() )
210 {
211 Engine::getSong()->setModified();
212 }
213 }
214}

259AutomatedValueMap TrackContainer::automatedValuesFromTracks(const TrackList &tracks, TimePos time, int clipNum)
260{
261 Track::clipVector clips;
262
263 for (Track* track: tracks)
264 {
265 if (track->isMuted()) {
266 continue;
267 }
268
269 switch(track->type())
270 {
271 case Track::Type::Automation:
272 case Track::Type::HiddenAutomation:
273 case Track::Type::Pattern:
274 if (clipNum < 0) {
275 track->getClipsInRange(clips, 0, time);
276 } else {
277 Q_ASSERT(track->numOfClips() > clipNum);
278 clips.push_back(track->getClip(clipNum));
279 }
280 default:
281 break;
282 }
283 }
284
285 AutomatedValueMap valueMap;
286
287 Q_ASSERT(std::is_sorted(clips.begin(), clips.end(), Clip::comparePosition));
288
289 for(Clip* clip : clips)
290 {
291 if (clip->isMuted() || clip->startPosition() > time) {
292 continue;
293 }
294
295 if (auto* p = dynamic_cast<AutomationClip *>(clip))
296 {
297 if (! p->hasAutomation()) {
298 continue;
299 }
300 TimePos relTime = time - p->startPosition() - p->startTimeOffset();
301 if (!p->isInPattern()) {
302 relTime = std::min(static_cast<int>(relTime), p->length() - p->startTimeOffset());
303 }
304 float value = p->valueAt(relTime);
305
306 for (AutomatableModel* model : p->objects())
307 {
308 valueMap[model] = value;
309 }
310 }
311 else if (auto* pattern = dynamic_cast<PatternClip*>(clip))
312 {
313 auto patIndex = dynamic_cast<class PatternTrack*>(pattern->getTrack())->patternIndex();
314 auto patStore = Engine::patternStore();
315
316 TimePos patTime = time - clip->startPosition();
317 patTime = std::min(patTime, clip->length());
318 patTime = patTime % (patStore->lengthOfPattern(patIndex) * TimePos::ticksPerBar());
319
320 auto patValues = patStore->automatedValuesAt(patTime, patIndex);
321 for (auto it=patValues.begin(); it != patValues.end(); it++)
322 {
323 // override old values, pattern track with the highest index takes precedence
324 valueMap[it.key()] = it.value();
325 }
326 }
327 else
328 {
329 continue;
330 }
331 }
332
333 return valueMap;
334};
