127 UInt32 WriteNextStateStartAux(int state, bool* result)
128 {
129 AtomicArrayCounter old_val;
130 AtomicArrayCounter new_val;
131 UInt32 cur_index;
132 UInt32 next_index;
133 bool need_copy;
134 do {
135 old_val = fCounter;
136 new_val = old_val;
137 *result = GetIndex1(new_val, state);
138 cur_index = GetIndex1(new_val, 0);
139 next_index = SwapIndex1(fCounter, state);
140 need_copy = (GetIndex1(new_val, state) == 0); // Written = false, switch just occurred
141 SetIndex1(new_val, state, 0); // Written = false, invalidate state
142 } while (!CAS(Counter1(old_val), Counter1(new_val), (UInt32*)&fCounter));
143 if (need_copy)
144 memcpy(&fState[next_index], &fState[cur_index], sizeof(T));
145 return next_index;
146 }
147
148 void WriteNextStateStopAux(int state)
149 {
150 AtomicArrayCounter old_val;
151 AtomicArrayCounter new_val;
152 do {
153 old_val = fCounter;
154 new_val = old_val;
155 SetIndex1(new_val, state, 1); // Written = true, state becomes "switchable"
156 } while (!CAS(Counter1(old_val), Counter1(new_val), (UInt32*)&fCounter));
157 }

159 public:

193 T* TrySwitchState(int state)
194 {
195 AtomicArrayCounter old_val;
196 AtomicArrayCounter new_val;
197 do {
198 old_val = fCounter;
199 new_val = old_val;
200 if (GetIndex1(new_val, state)) { // If state has been written
201 SetIndex1(new_val, 0, SwapIndex1(new_val, state)); // Prepare switch
202 SetIndex1(new_val, state, 0); // Invalidate the state "state"
203 IncIndex1(new_val, 3); // Inc switch
204 }
205 } while (!CAS(Counter1(old_val), Counter1(new_val), (UInt32*)&fCounter));
206 return &fState[GetIndex1(fCounter, 0)]; // Read the counter again
207 }

225 } while (!CAS(Counter1(old_val), Counter1(new_val), (UInt32*)&fCounter));
226 return &fState[GetIndex1(fCounter, 0)]; // Read the counter again
227 }
