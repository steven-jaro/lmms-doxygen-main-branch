197bool DataFile::validate( QString extension )
198{
199 switch( m_type )
200 {
201 case Type::SongProject:
202 if( extension == "mmp" || extension == "mmpz" )
203 {
204 return true;
205 }
206 break;
207 case Type::SongProjectTemplate:
208 if( extension == "mpt" )
209 {
210 return true;
211 }
212 break;
213 case Type::InstrumentTrackSettings:
214 if ( extension == "xpf" || extension == "xml" )
215 {
216 return true;
217 }
218 break;
219 case Type::MidiClip:
220 if (extension == "xpt" || extension == "xptz")
221 {
222 return true;
223 }
224 break;
225 case Type::Unknown:
226 if (! ( extension == "mmp" || extension == "mpt" || extension == "mmpz" ||
227 extension == "xpf" || extension == "xml" ||
228 ( extension == "xiz" && ! getPluginFactory()->pluginSupportingExtension(extension).isNull()) ||
229 extension == "sf2" || extension == "sf3" || extension == "pat" || extension == "mid" ||
230 extension == "dll"
231#ifdef LMMS_BUILD_LINUX
232 || extension == "so"
233#endif
234#ifdef LMMS_HAVE_LV2
235 || extension == "lv2"
236#endif
237 ) )
238 {
239 return true;
240 }
241 if( extension == "wav" || extension == "ogg" || extension == "ds"
242#ifdef LMMS_HAVE_SNDFILE_MP3
243 || extension == "mp3"
244#endif
245 )
246 {
247 return true;
248 }
249 break;
250 default:
251 return false;
252 }
253 return false;
254}

259QString DataFile::nameWithExtension( const QString & _fn ) const
260{
261 const QString extension = _fn.section( '.', -1 );
262
263 switch( type() )
264 {
265 case Type::SongProject:
266 if( extension != "mmp" &&
267 extension != "mpt" &&
268 extension != "mmpz" )
269 {
270 if( ConfigManager::inst()->value( "app",
271 "nommpz" ).toInt() == 0 )
272 {
273 return _fn + ".mmpz";
274 }
275 return _fn + ".mmp";
276 }
277 break;
278 case Type::SongProjectTemplate:
279 if( extension != "mpt" )
280 {
281 return _fn + ".mpt";
282 }
283 break;
284 case Type::InstrumentTrackSettings:
285 if( extension != "xpf" )
286 {
287 return _fn + ".xpf";
288 }
289 break;
290 default: ;
291 }
292 return _fn;
293}

312bool DataFile::writeFile(const QString& filename, bool withResources)
313{
314 // Small lambda function for displaying errors
315 auto showError = [](QString title, QString body){
316 if (gui::getGUI() != nullptr)
317 {
318 QMessageBox mb;
319 mb.setWindowTitle(title);
320 mb.setText(body);
321 mb.setIcon(QMessageBox::Warning);
322 mb.setStandardButtons(QMessageBox::Ok);
323 mb.exec();
324 }
325 else
326 {
327 qWarning() << body;
328 }
329 };
330
331 // If we are saving without resources, filename is just the file we are
332 // saving to. If we are saving with resources (project bundle), filename
333 // will be used (discarding extensions) to create a folder where the
334 // bundle will be saved in
335
336 QFileInfo fInfo(filename);
337
338 const QString bundleDir = fInfo.path() + "/" + fInfo.fileName().section('.', 0, 0);
339 const QString resourcesDir = bundleDir + "/resources";
340 const QString fullName = withResources
341 ? nameWithExtension(bundleDir + "/" + fInfo.fileName())
342 : nameWithExtension(filename);
343 const QString fullNameTemp = fullName + ".new";
344 const QString fullNameBak = fullName + ".bak";
345
346 using gui::SongEditor;
347
348 // If we are saving with resources, setup the bundle folder first
349 if (withResources)
350 {
351 // First check if there's a bundle folder with the same name in
352 // the path already. If so, warns user that we can't overwrite a
353 // project bundle.
354 if (QDir(bundleDir).exists())
355 {
356 showError(SongEditor::tr("Operation denied"),
357 SongEditor::tr("A bundle folder with that name already exists on the "
358 "selected path. Can't overwrite a project bundle. Please select a different "
359 "name."));
360
361 return false;
362 }
363
364 // Create bundle folder
365 if (!QDir().mkdir(bundleDir))
366 {
367 showError(SongEditor::tr("Error"),
368 SongEditor::tr("Couldn't create bundle folder."));
369 return false;
370 }
371
372 // Create resources folder
373 if (!QDir().mkdir(resourcesDir))
374 {
375 showError(SongEditor::tr("Error"),
376 SongEditor::tr("Couldn't create resources folder."));
377 return false;
378 }
379
380 // Copy resources to folder and update paths
381 if (!copyResources(resourcesDir))
382 {
383 showError(SongEditor::tr("Error"),
384 SongEditor::tr("Failed to copy resources."));
385 return false;
386 }
387 }
388
389 QSaveFile outfile(fullNameTemp);
390
391 if (!outfile.open(QIODevice::WriteOnly | QIODevice::Truncate))
392 {
393 showError(SongEditor::tr("Could not write file"),
394 SongEditor::tr("Could not open %1 for writing. You probably are not permitted to "
395 "write to this file. Please make sure you have write-access to "
396 "the file and try again.").arg(fullName));
397
398 return false;
399 }
400
401 const QString extension = fullName.section('.', -1);
402 if (extension == "mmpz" || extension == "xptz")
403 {
404 QString xml;
405 QTextStream ts( &xml );
406 write( ts );
407 outfile.write( qCompress( xml.toUtf8() ) );
408 }
409 else
410 {
411 QTextStream ts( &outfile );
412 write( ts );
413 }
414
415 if (!outfile.commit())
416 {
417 showError(SongEditor::tr("Could not write file"),
418 SongEditor::tr("An unknown error has occurred and the file could not be saved."));
419 return false;
420 }
421
422 if (ConfigManager::inst()->value("app", "disablebackup").toInt())
423 {
424 // remove current file
425 QFile::remove(fullName);
426 }
427 else
428 {
429 // remove old backup file
430 QFile::remove(fullNameBak);
431 // move current file to backup file
432 QFile::rename(fullName, fullNameBak);
433 }
434 // move temporary file to current file
435 QFile::rename(fullNameTemp, fullName);
436
437 return true;
438}

443bool DataFile::copyResources(const QString& resourcesDir)
444{
445 // List of filenames used so we can append a counter to any
446 // repeating filenames
447 std::list<QString> namesList;
448
449 auto it = ELEMENTS_WITH_RESOURCES.begin();
450
451 // Copy resources and manipulate the DataFile to have local paths to them
452 while (it != ELEMENTS_WITH_RESOURCES.end())
453 {
454 QDomNodeList list = elementsByTagName(it->first);
455
456 // Go through all elements with the tagname from our map
457 for (int i = 0; !list.item(i).isNull(); ++i)
458 {
459 QDomElement el = list.item(i).toElement();
460
461 auto res = it->second.begin();
462
463 // Search for attributes that point to resources
464 while (res != it->second.end())
465 {
466 // If the element has that attribute
467 if (el.hasAttribute(*res))
468 {
469 // Get absolute path to resource
470 bool error;
471 QString resPath = PathUtil::toAbsolute(el.attribute(*res), &error);
472 // If we are running without the project loaded (from CLI), "local:" base
473 // prefixes aren't converted, so we need to convert it ourselves
474 if (error)
475 {
476 resPath = QFileInfo(m_fileName).path() + "/" + resPath.remove(0,
477 PathUtil::basePrefix(PathUtil::Base::LocalDir).length());
478 }
479
480 // Check if we need to add a counter to the filename
481 QString finalFileName = QFileInfo(resPath).fileName();
482 QString extension = resPath.section('.', -1);
483 int repeatedNames = 0;
484 for (QString name : namesList)
485 {
486 if (finalFileName == name)
487 {
488 ++repeatedNames;
489 }
490 }
491 // Add the name to the list before modifying it
492 namesList.push_back(finalFileName);
493 if (repeatedNames)
494 {
495 // Remove the extension, add the counter and add the
496 // extension again to get the final file name
497 finalFileName.truncate(finalFileName.lastIndexOf('.'));
498 finalFileName = finalFileName + "-" + QString::number(repeatedNames) + "." + extension;
499 }
500
501 // Final path is our resources dir + the new file name
502 QString finalPath = resourcesDir + "/" + finalFileName;
503
504 // Copy resource file to the resources folder
505 if(!QFile::copy(resPath, finalPath))
506 {
507 qWarning("ERROR: Failed to copy resource");
508 return false;
509 }
510
511 // Update attribute path to point to the bundle file
512 QString newAtt = PathUtil::basePrefix(PathUtil::Base::LocalDir) + "resources/" + finalFileName;
513 el.setAttribute(*res, newAtt);
514 }
515 ++res;
516 }
517 }
518 ++it;
519 }
520
521 return true;
522}

527/**
528 * @brief This recursive method will go through all XML nodes of the DataFile
529 * and check whether any of them have local paths. If they are not on
530 * our list of elements that can have local paths we return true,
531 * indicating that we potentially have plugins with local paths that
532 * would be a security issue. The Song class can then abort loading
533 * this project.
534 * @param parent The parent node being iterated. When called
535 * without arguments, this will be an empty element that will be
536 * ignored (since the second parameter will be true).
537 * @param firstCall Defaults to true, and indicates to this recursive
538 * method whether this is the first call. If it is it will use the
539 * root element as the parent.
540 */
541bool DataFile::hasLocalPlugins(QDomElement parent /* = QDomElement()*/, bool firstCall /* = true*/) const
542{
543 // If this is the first iteration of the recursion we use the root element
544 if (firstCall) { parent = documentElement(); }
545
546 auto children = parent.childNodes();
547 for (int i = 0; i < children.size(); ++i)
548 {
549 QDomNode child = children.at(i);
550 QDomElement childElement = child.toElement();
551
552 bool skipNode = false;
553 // Skip the nodes allowed to have "local:" attributes, but
554 // still check its children
555 for (const auto& element : ELEMENTS_WITH_RESOURCES)
556 {
557 if (childElement.tagName() == element.first)
558 {
559 skipNode = true;
560 break;
561 }
562 }
563
564 // Check if they have "local:" attribute (unless they are allowed to
565 // and skipNode is true)
566 if (!skipNode)
567 {
568 auto attributes = childElement.attributes();
569 for (int i = 0; i < attributes.size(); ++i)
570 {
571 QDomNode attribute = attributes.item(i);
572 QDomAttr attr = attribute.toAttr();
573 if (attr.value().startsWith(PathUtil::basePrefix(PathUtil::Base::LocalDir),
574 Qt::CaseInsensitive))
575 {
576 return true;
577 }
578 }
579 }
580
581 // Now we check the children of this node (recursively)
582 // and if any return true we return true.
583 if (hasLocalPlugins(childElement, false))
584 {
585 return true;
586 }
587 }
588
589 // If we got here none of the nodes had the "local:" path.
590 return false;
591}

650void DataFile::mapSrcAttributeInElementsWithResources(const QMap<QString, QString>& map)
651{
652 for (const auto& [elem, srcAttrs] : ELEMENTS_WITH_RESOURCES)
653 {
654 auto elements = elementsByTagName(elem);
655
656 for (const auto& srcAttr : srcAttrs)
657 {
658 for (int i = 0; i < elements.length(); ++i)
659 {
660 auto item = elements.item(i).toElement();
661
662 if (item.isNull() || !item.hasAttribute(srcAttr)) { continue; }
663
664 const QString srcVal = item.attribute(srcAttr);
665
666 const auto it = map.constFind(srcVal);
667 if (it != map.constEnd())
668 {
669 item.setAttribute(srcAttr, *it);
670 }
671 }
672 }

1250/**
1251 * Helper function to call a functor for all effect ports' DomElements,
1252 * providing the functor with lists to add and remove DomElements. Helpful for
1253 * patching port values from savefiles.
1254 */
1255template<class Ftor>
1256void iterate_ladspa_ports(QDomElement& effect, Ftor& ftor)
1257{
1258 // Head back up the DOM to upgrade ports
1259 QDomNodeList ladspacontrols = effect.elementsByTagName( "ladspacontrols" );
1260 for( int m = 0; !ladspacontrols.item( m ).isNull(); ++m )
1261 {
1262 QList<QDomElement> addList, removeList;
1263 QDomElement ladspacontrol = ladspacontrols.item( m ).toElement();
1264 for( QDomElement port = ladspacontrol.firstChild().toElement();
1265 !port.isNull(); port = port.nextSibling().toElement() )
1266 {
1267 QStringList parts = port.tagName().split("port");
1268 // Not a "port"
1269 if ( parts.size() < 2 )
1270 {
1271 continue;
1272 }
1273 int num = parts[1].toInt();
1274
1275 // From Qt's docs of QDomNode:
1276 // * copying a QDomNode is OK, they still have the same
1277 // pointer to the "internal" QDomNodePrivate.
1278 // * Also, they are using linked lists, which means
1279 // deleting or appending QDomNode does not invalidate
1280 // any other pointers.
1281 // => Inside ftor, you can (and should) push back the
1282 // QDomElements by value, not references
1283 // => The loops below for adding and removing don't
1284 // invalidate any other QDomElements
1285 ftor(port, num, addList, removeList);
1286 }
1287
1288 // Add ports marked for adding
1289 for ( QDomElement e : addList )
1290 {
1291 ladspacontrol.appendChild( e );
1292 }
1293 // Remove ports marked for removal
1294 for ( QDomElement e : removeList )
1295 {
1296 ladspacontrol.removeChild( e );
1297 }
1298 }
1299}
