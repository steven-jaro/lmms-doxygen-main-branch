66void SampleRecordHandle::play( SampleFrame* /*_working_buffer*/ )
67{
68 const SampleFrame* recbuf = Engine::audioEngine()->inputBuffer();
69 const f_cnt_t frames = Engine::audioEngine()->inputBufferFrames();
70 writeBuffer( recbuf, frames );
71 m_framesRecorded += frames;
72
73 TimePos len = (tick_t)( m_framesRecorded / Engine::framesPerTick() );
74 if( len > m_minLength )

76// m_clip->changeLength( len );

108std::shared_ptr<const SampleBuffer> SampleRecordHandle::createSampleBuffer()
109{
110 const f_cnt_t frames = framesRecorded();
111 // create buffer to store all recorded buffers in
112 auto bigBuffer = std::vector<SampleFrame>(frames);
113
114 // now copy all buffers into big buffer
115 auto framesCopied = 0;
116 for (const auto& [buf, numFrames] : m_buffers)
117 {
118 std::copy_n(buf, numFrames, bigBuffer.begin() + framesCopied);
119 framesCopied += numFrames;
120 }
121
122 // create according sample-buffer out of big buffer
123 return std::make_shared<const SampleBuffer>(std::move(bigBuffer), Engine::audioEngine()->inputSampleRate());
124}

129void SampleRecordHandle::writeBuffer( const SampleFrame* _ab, const f_cnt_t _frames )
130{
131 auto buf = new SampleFrame[_frames];
132 for( f_cnt_t frame = 0; frame < _frames; ++frame )
133 {
134 for( ch_cnt_t chnl = 0; chnl < DEFAULT_CHANNELS; ++chnl )
135 {
136 buf[frame][chnl] = _ab[frame][chnl];
137 }
138 }
139 m_buffers.push_back( qMakePair( buf, _frames ) );
140}
