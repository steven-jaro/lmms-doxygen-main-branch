34constexpr auto converterType(AudioResampler::Mode mode) -> int
35{
36 switch (mode)
37 {
38 case AudioResampler::Mode::ZOH:
39 return SRC_ZERO_ORDER_HOLD;
40 case AudioResampler::Mode::Linear:
41 return SRC_LINEAR;
42 case AudioResampler::Mode::SincFastest:
43 return SRC_SINC_FASTEST;
44 case AudioResampler::Mode::SincMedium:
45 return SRC_SINC_MEDIUM_QUALITY;
46 case AudioResampler::Mode::SincBest:
47 return SRC_SINC_BEST_QUALITY;
48 default:
49 throw std::invalid_argument{"Invalid interpolation mode"};
50 }
51}

54AudioResampler::AudioResampler(Mode mode, ch_cnt_t channels)

63auto AudioResampler::process(InterleavedBufferView<const float> input, InterleavedBufferView<float> output) -> Result
64{
65 if (input.channels() != m_channels || output.channels() != m_channels)
66 {
67 throw std::invalid_argument{"Invalid channel count"};
68 }
69
70 auto data = SRC_DATA{};
71
72 data.data_in = input.data();
73 data.input_frames = input.frames();
74
75 data.data_out = output.data();
76 data.output_frames = output.frames();
77
78 data.src_ratio = m_ratio;
79 data.end_of_input = 0;
80
81 if ((m_error = src_process(static_cast<SRC_STATE*>(m_state.get()), &data)))
82 {
83 throw std::runtime_error{src_strerror(m_error)};
84 }
85
86 return {static_cast<f_cnt_t>(data.input_frames_used), static_cast<f_cnt_t>(data.output_frames_gen)};
87}
