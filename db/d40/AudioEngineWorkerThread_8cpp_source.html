57void AudioEngineWorkerThread::JobQueue::addJob( ThreadableJob * _job )
58{
59 if( _job->requiresProcessing() )
60 {
61 // update job state
62 _job->queue();
63 // actually queue the job via atomic operations
64 auto index = m_writeIndex++;
65 if (index < JOB_QUEUE_SIZE) {
66 m_items[index] = _job;
67 } else {
68 qWarning() << "Job queue is full!";
69 ++m_itemsDone;
70 }
71 }
72}

76void AudioEngineWorkerThread::JobQueue::run()
77{
78 bool processedJob = true;
79 while (processedJob && m_itemsDone < m_writeIndex)
80 {
81 processedJob = false;
82 for (auto i = std::size_t{0}; i < m_writeIndex && i < JOB_QUEUE_SIZE; ++i)
83 {
84 ThreadableJob * job = m_items[i].exchange(nullptr);
85 if( job )
86 {
87 job->process();
88 processedJob = true;
89 ++m_itemsDone;
90 }
91 }
92 // always exit loop if we're not in dynamic mode
93 processedJob = processedJob && ( m_opMode == OperationMode::Dynamic );
94 }
95}

116AudioEngineWorkerThread::AudioEngineWorkerThread( AudioEngine* audioEngine ) :
117 QThread( audioEngine ),
118 m_quit( false )
119{
120 // initialize global static data
121 if( queueReadyWaitCond == nullptr )

126 // keep track of all instantiated worker threads - this is used for
127 // processing the last worker thread "inline", see comments in
128 // AudioEngineWorkerThread::startAndWaitForJobs() for details
129 workerThreads << this;

154void AudioEngineWorkerThread::startAndWaitForJobs()
155{
156 queueReadyWaitCond->wakeAll();
157 // The last worker-thread is never started. Instead it's processed "inline"
158 // i.e. within the global AudioEngine thread. This way we can reduce latencies
159 // that otherwise would be caused by synchronizing with another thread.
160 globalJobQueue.run();
161 globalJobQueue.wait();
162}

167void AudioEngineWorkerThread::run()
168{
169 disable_denormals();
170
171 QMutex m;
172 while( m_quit == false )
173 {
174 m.lock();
175 queueReadyWaitCond->wait( &m );
176 globalJobQueue.run();
177 m.unlock();
178 }
179}
