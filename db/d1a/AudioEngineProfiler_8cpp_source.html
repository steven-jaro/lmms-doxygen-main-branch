41void AudioEngineProfiler::finishPeriod( sample_rate_t sampleRate, fpp_t framesPerPeriod )
42{
43 // Time taken to process all data and fill the audio buffer.
44 const unsigned int periodElapsed = m_periodTimer.elapsed();
45 // Maximum time the processing can take before causing buffer underflow. Convert to us.
46 const uint64_t timeLimit = static_cast<uint64_t>(1000000) * framesPerPeriod / sampleRate;
47
48 // Compute new overall CPU load and apply exponential averaging.
49 // The result is used for overload detection in AudioEngine::criticalXRuns()
50 // → the weight of a new sample must be high enough to allow relatively fast changes!
51 const auto newCpuLoad = 100.f * periodElapsed / timeLimit;
52 m_cpuLoad = newCpuLoad * 0.1f + m_cpuLoad * 0.9f;
53
54 // Compute detailed load analysis. Can use stronger averaging to get more stable readout.
55 for (std::size_t i = 0; i < DetailCount; i++)
56 {
57 const auto newLoad = 100.f * m_detailTime[i] / timeLimit;
58 const auto oldLoad = m_detailLoad[i].load(std::memory_order_relaxed);
59 m_detailLoad[i].store(newLoad * 0.05f + oldLoad * 0.95f, std::memory_order_relaxed);
60 }
61
62 if( m_outputFile.isOpen() )
63 {
64 m_outputFile.write( QString( "%1\n" ).arg( periodElapsed ).toLatin1() );
65 }
66}
