148void PatternEditor::dropEvent(QDropEvent* de)
149{
150 QString type = StringPairDrag::decodeKey( de );
151 QString value = StringPairDrag::decodeValue( de );
152
153 if( type.left( 6 ) == "track_" )
154 {
155 DataFile dataFile( value.toUtf8() );
156 Track * t = Track::create( dataFile.content().firstChild().toElement(), model() );
157
158 // Ensure pattern clips exist
159 bool hasValidPatternClips = false;
160 if (t->getClips().size() == static_cast<std::size_t>(m_ps->numOfPatterns()))
161 {
162 hasValidPatternClips = true;
163 for (auto i = std::size_t{0}; i < t->getClips().size(); ++i)
164 {
165 if (t->getClips()[i]->startPosition() != TimePos(i, 0))
166 {
167 hasValidPatternClips = false;
168 break;
169 }
170 }
171 }
172 if (!hasValidPatternClips)
173 {
174 t->deleteClips();
175 t->createClipsForPattern(m_ps->numOfPatterns() - 1);
176 }
177 m_ps->updateAfterTrackAdd();
178
179 de->accept();
180 }
181 else
182 {
183 TrackContainerView::dropEvent( de );
184 }
185 updateMaxSteps();
186}

205void PatternEditor::updatePixelsPerBar()
206{
207 setPixelsPerBar(m_maxClipLength != 0
208 ? (width() - m_trackHeadWidth) * TimePos::ticksPerBar() / m_maxClipLength
209 : (width() - m_trackHeadWidth));

213void PatternEditor::updateMaxSteps()
214{
215 const TrackContainer::TrackList& tl = model()->tracks();
216
217 m_maxClipLength = 0;
218 for (const auto& track : tl)
219 {
220 if (track->type() == Track::Type::Instrument)
221 {
222 auto mClip = static_cast<MidiClip*>(track->getClip(m_ps->currentPattern()));
223 m_maxClipLength = std::max(m_maxClipLength, static_cast<tick_t>(mClip->length()));
224 }
225 }
226 updatePixelsPerBar();
227}

230void PatternEditor::makeSteps( bool clone )
231{
232 const TrackContainer::TrackList& tl = model()->tracks();
233
234 for (const auto& track : tl)
235 {
236 if (track->type() == Track::Type::Instrument)
237 {
238 auto p = static_cast<MidiClip*>(track->getClip(m_ps->currentPattern()));
239 if( clone )
240 {
241 p->cloneSteps();
242 } else
243 {
244 p->addSteps();
245 }
246 }
247 }
248 updateMaxSteps();
249}

251// Creates a clone of the current pattern track with the same content, but no clips in the song editor
252// TODO: Avoid repeated code from cloneTrack and clearTrack in TrackOperationsWidget somehow
253void PatternEditor::cloneClip()
254{
255 // Get the current PatternTrack id
256 auto ps = static_cast<PatternStore*>(model());
257 const int currentPattern = ps->currentPattern();
258
259 PatternTrack* pt = PatternTrack::findPatternTrack(currentPattern);
260
261 if (pt)
262 {
263 // Clone the track
264 Track* newTrack = pt->clone();
265 ps->setCurrentPattern(static_cast<PatternTrack*>(newTrack)->patternIndex());
266
267 // Track still have the clips which is undesirable in this case, clear the track
268 newTrack->lock();
269 newTrack->deleteClips();
270 newTrack->unlock();
271 }
272}
