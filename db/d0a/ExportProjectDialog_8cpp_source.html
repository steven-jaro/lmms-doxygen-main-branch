37ExportProjectDialog::ExportProjectDialog( const QString & _file_name,
38 QWidget * _parent, bool multi_export=false ) :
39 QDialog( _parent ),
40 Ui::ExportProjectDialog(),
41 m_fileName( _file_name ),
42 m_fileExtension(),
43 m_multiExport( multi_export ),
44 m_renderManager( nullptr )
45{
46 setupUi( this );
47 setWindowTitle( tr( "Export project to %1" ).arg(
48 QFileInfo( _file_name ).fileName() ) );
49
50 // Get the extension of the chosen file.
51 QStringList parts = _file_name.split( '.' );
52 QString fileExt;
53 if( parts.size() > 0 )
54 {
55 fileExt = "." + parts[parts.size()-1];
56 }
57
58 int cbIndex = 0;
59 for (auto i = std::size_t{0}; i < ProjectRenderer::NumFileFormats; ++i)
60 {
61 if( ProjectRenderer::fileEncodeDevices[i].isAvailable() )
62 {
63 // Get the extension of this format.
64 QString renderExt = ProjectRenderer::fileEncodeDevices[i].m_extension;
65
66 // Add to combo box.
67 fileFormatCB->addItem( ProjectRenderer::tr(
68 ProjectRenderer::fileEncodeDevices[i].m_description ),
69 QVariant( static_cast<int>(ProjectRenderer::fileEncodeDevices[i].m_fileFormat) ) // Format tag; later used for identification.
70 );
71
72 // If this is our extension, select it.
73 if( QString::compare( renderExt, fileExt,
74 Qt::CaseInsensitive ) == 0 )
75 {
76 fileFormatCB->setCurrentIndex( cbIndex );
77 }
78
79 cbIndex++;
80 }
81 }
82
83 int const MAX_LEVEL=8;
84 for(int i=0; i<=MAX_LEVEL; ++i)
85 {
86 QString info="";
87 if ( i==0 ){ info = tr( "( Fastest - biggest )" ); }
88 else if ( i==MAX_LEVEL ){ info = tr( "( Slowest - smallest )" ); }
89
90 compLevelCB->addItem(
91 QString::number(i)+" "+info,
92 QVariant(i/static_cast<double>(MAX_LEVEL))
93 );
94 }
95 compLevelCB->setCurrentIndex(5);
96#ifndef LMMS_HAVE_SF_COMPLEVEL
97 // Disable this widget; the setting would be ignored by the renderer.
98 compressionWidget->setVisible(false);
99#endif
100
101 for (const auto sampleRate : SUPPORTED_SAMPLERATES)
102 {
103 samplerateCB->addItem(tr("%1 Hz").arg(sampleRate), sampleRate);
104 }
105
106 const auto currentIndex = std::max(0, samplerateCB->findData(Engine::audioEngine()->outputSampleRate()));
107 samplerateCB->setCurrentIndex(currentIndex);
108
109 connect( startButton, SIGNAL(clicked()),
110 this, SLOT(startBtnClicked()));
111}

163void ExportProjectDialog::startExport()
164{
165 const auto bitrates = std::array{64, 128, 160, 192, 256, 320};
166
167 OutputSettings os = OutputSettings(samplerateCB->currentData().toInt(), bitrates[bitrateCB->currentIndex()],
168 static_cast<OutputSettings::BitDepth>(depthCB->currentIndex()),
169 mapToStereoMode(stereoModeComboBox->currentIndex()));
170
171 if (compressionWidget->isVisible())
172 {
173 double level = compLevelCB->itemData(compLevelCB->currentIndex()).toDouble();
174 os.setCompressionLevel(level);
175 }
176
177 // Make sure we have the the correct file extension
178 // so there's no confusion about the codec in use.
179 auto output_name = m_fileName;
180 if (!(m_multiExport || output_name.endsWith(m_fileExtension,Qt::CaseInsensitive)))
181 {
182 output_name+=m_fileExtension;
183 }
184
185 m_renderManager.reset(new RenderManager(os, m_ft, output_name));
186
187 Engine::getSong()->setExportLoop( exportLoopCB->isChecked() );
188 Engine::getSong()->setRenderBetweenMarkers( renderMarkersCB->isChecked() );
189 Engine::getSong()->setLoopRenderCount(loopCountSB->value());
190
191 connect( m_renderManager.get(), SIGNAL(progressChanged(int)),
192 progressBar, SLOT(setValue(int)));
193 connect( m_renderManager.get(), SIGNAL(progressChanged(int)),
194 this, SLOT(updateTitleBar(int)));
195 connect( m_renderManager.get(), SIGNAL(finished()),
196 this, SLOT(accept())) ;
197 connect( m_renderManager.get(), SIGNAL(finished()),
198 getGUI()->mainWindow(), SLOT(resetWindowTitle()));
199
200 if ( m_multiExport )
201 {
202 m_renderManager->renderTracks();
203 }
204 else
205 {
206 m_renderManager->renderProject();
207 }
208}

211void ExportProjectDialog::onFileFormatChanged(int index)
212{
213 // Extract the format tag from the currently selected item,
214 // and adjust the UI properly.
215 QVariant format_tag = fileFormatCB->itemData(index);
216 bool successful_conversion = false;
217 auto exportFormat = static_cast<ProjectRenderer::ExportFileFormat>(
218 format_tag.toInt(&successful_conversion)
219 );
220 Q_ASSERT(successful_conversion);
221
222 bool stereoModeVisible = (exportFormat == ProjectRenderer::ExportFileFormat::MP3);
223
224 bool sampleRateControlsVisible = (exportFormat != ProjectRenderer::ExportFileFormat::MP3);
225
226 bool bitRateControlsEnabled =
227 (exportFormat == ProjectRenderer::ExportFileFormat::Ogg ||
228 exportFormat == ProjectRenderer::ExportFileFormat::MP3);
229
230 bool bitDepthControlEnabled =
231 (exportFormat == ProjectRenderer::ExportFileFormat::Wave ||
232 exportFormat == ProjectRenderer::ExportFileFormat::Flac);
233
234#ifdef LMMS_HAVE_SF_COMPLEVEL
235 bool compressionLevelVisible = (exportFormat == ProjectRenderer::ExportFileFormat::Flac);
236 compressionWidget->setVisible(compressionLevelVisible);
237#endif
238
239 stereoModeWidget->setVisible(stereoModeVisible);
240 sampleRateWidget->setVisible(sampleRateControlsVisible);
241
242 bitrateWidget->setVisible(bitRateControlsEnabled);
243
244 depthWidget->setVisible(bitDepthControlEnabled);
245}

247void ExportProjectDialog::startBtnClicked()
248{
249 m_ft = ProjectRenderer::ExportFileFormat::Count;
250
251 // Get file format from current menu selection.
252 bool successful_conversion = false;
253 QVariant tag = fileFormatCB->itemData(fileFormatCB->currentIndex());
254 m_ft = static_cast<ProjectRenderer::ExportFileFormat>(
255 tag.toInt(&successful_conversion)
256 );
257
258 if( !successful_conversion )
259 {
260 QMessageBox::information( this, tr( "Error" ),
261 tr( "Error while determining file-encoder device. "
262 "Please try to choose a different output "
263 "format." ) );
264 reject();
265 return;
266 }
267
268 // Find proper file extension.
269 for (auto i = std::size_t{0}; i < ProjectRenderer::NumFileFormats; ++i)
270 {
271 if (m_ft == ProjectRenderer::fileEncodeDevices[i].m_fileFormat)
272 {
273 m_fileExtension = QString( QLatin1String( ProjectRenderer::fileEncodeDevices[i].m_extension ) );
274 break;
275 }
276 }
277
278 startButton->setEnabled( false );
279 progressBar->setEnabled( true );
280
281 updateTitleBar( 0 );
282
283 startExport();
284}
