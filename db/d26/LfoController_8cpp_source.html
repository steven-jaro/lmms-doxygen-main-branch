41LfoController::LfoController( Model * _parent ) :
42 Controller( ControllerType::Lfo, _parent, tr( "LFO Controller" ) ),
43 m_baseModel(0.5f, 0.f, 1.f, 0.001f, this, tr("Base value")),
44 m_speedModel(2.f, 0.01f, 20.f, 0.0001f, 20000.f, this, tr("Oscillator speed")),
45 m_amountModel(1.f, -1.f, 1.f, 0.005f, this, tr("Oscillator amount")),
46 m_phaseModel( 0.0, 0.0, 360.0, 4.0, this, tr( "Oscillator phase" ) ),
47 m_waveModel( static_cast<int>(Oscillator::WaveShape::Sine), 0, Oscillator::NumWaveShapes,
48 this, tr( "Oscillator waveform" ) ),
49 m_multiplierModel( 0, 0, 2, this, tr( "Frequency Multiplier" ) ),
50 m_duration( 1000 ),
51 m_phaseOffset( 0 ),
52 m_currentPhase( 0 ),
53 m_sampleFunction( &Oscillator::sinSample ),
54 m_userDefSampleBuffer(std::make_shared<SampleBuffer>())
55{
56 setSampleExact( true );
57 connect( &m_waveModel, SIGNAL(dataChanged()),
58 this, SLOT(updateSampleFunction()), Qt::DirectConnection );
59
60 connect( &m_speedModel, SIGNAL(dataChanged()),
61 this, SLOT(updateDuration()), Qt::DirectConnection );
62 connect( &m_multiplierModel, SIGNAL(dataChanged()),
63 this, SLOT(updateDuration()), Qt::DirectConnection );
64 connect( Engine::audioEngine(), SIGNAL(sampleRateChanged()),
65 this, SLOT(updateDuration()));
66
67 connect( Engine::getSong(), SIGNAL(playbackStateChanged()),
68 this, SLOT(updatePhase()));
69 connect( Engine::getSong(), SIGNAL(playbackPositionChanged()),
70 this, SLOT(updatePhase()));
71
72 updateDuration();
73}

89void LfoController::updateValueBuffer()
90{
91 m_phaseOffset = m_phaseModel.value() / 360.0;
92 float phase = m_currentPhase + m_phaseOffset;
93 float phasePrev = 0.0f;
94
95 // roll phase up until we're in sync with period counter
96 m_bufferLastUpdated++;
97 if( m_bufferLastUpdated < s_periods )
98 {
99 int diff = s_periods - m_bufferLastUpdated;
100 phase += static_cast<float>( Engine::audioEngine()->framesPerPeriod() * diff ) / m_duration;
101 m_bufferLastUpdated += diff;
102 }
103
104 float amount = m_amountModel.value();
105 ValueBuffer *amountBuffer = m_amountModel.valueBuffer();
106 int amountInc = amountBuffer ? 1 : 0;
107 float *amountPtr = amountBuffer ? &(amountBuffer->values()[ 0 ] ) : &amount;
108 Oscillator::WaveShape waveshape = static_cast<Oscillator::WaveShape>(m_waveModel.value());
109
110 for( float& f : m_valueBuffer )
111 {
112 float currentSample = 0;
113 switch (waveshape)
114 {
115 case Oscillator::WaveShape::WhiteNoise:
116 {
117 if (absFraction(phase) < absFraction(phasePrev))
118 {
119 // Resample when phase period has completed
120 m_heldSample = m_sampleFunction(phase);
121 }
122 currentSample = m_heldSample;
123 break;
124 }
125 case Oscillator::WaveShape::UserDefined:
126 {
127 currentSample = Oscillator::userWaveSample(m_userDefSampleBuffer.get(), phase);
128 break;
129 }
130 default:
131 {
132 if (m_sampleFunction != nullptr)
133 {
134 currentSample = m_sampleFunction(phase);
135 }
136 }
137 }
138
139 f = std::clamp(m_baseModel.value() + (*amountPtr * currentSample / 2.0f), 0.0f, 1.0f);
140
141 phasePrev = phase;
142 phase += 1.0 / m_duration;
143 amountPtr += amountInc;
144 }
145
146 m_currentPhase = absFraction(phase - m_phaseOffset);
147 m_bufferLastUpdated = s_periods;
148}

150void LfoController::updatePhase()
151{
152 m_currentPhase = ( Engine::getSong()->getFrames() ) / m_duration;
153 m_bufferLastUpdated = s_periods - 1;

157void LfoController::updateDuration()
158{
159 float newDurationF = Engine::audioEngine()->outputSampleRate() * m_speedModel.value();
160
161 switch(m_multiplierModel.value() )
162 {
163 case 1:
164 newDurationF /= 100.0;
165 break;
166
167 case 2:
168 newDurationF *= 100.0;
169 break;
170
171 default:
172 break;
173 }
174
175 m_duration = newDurationF;
176}

178void LfoController::updateSampleFunction()
179{
180 switch( static_cast<Oscillator::WaveShape>(m_waveModel.value()) )
181 {
182 case Oscillator::WaveShape::Sine:
183 default:
184 m_sampleFunction = &Oscillator::sinSample;
185 break;
186 case Oscillator::WaveShape::Triangle:
187 m_sampleFunction = &Oscillator::triangleSample;
188 break;
189 case Oscillator::WaveShape::Saw:
190 m_sampleFunction = &Oscillator::sawSample;
191 break;
192 case Oscillator::WaveShape::Square:
193 m_sampleFunction = &Oscillator::squareSample;
194 break;
195 case Oscillator::WaveShape::MoogSaw:
196 m_sampleFunction = &Oscillator::moogSawSample;
197 break;
198 case Oscillator::WaveShape::Exponential:
199 m_sampleFunction = &Oscillator::expSample;
200 break;
201 case Oscillator::WaveShape::WhiteNoise:
202 m_sampleFunction = &Oscillator::noiseSample;
203 break;
204 case Oscillator::WaveShape::UserDefined:
205 m_sampleFunction = nullptr;
206 /*TODO: If C++11 is allowed, should change the type of
207 m_sampleFunction be std::function<sample_t(const float)>
208 and use the line below:
209 */
210 //m_sampleFunction = &(m_userDefSampleBuffer->userWaveSample)
211 break;
212 }
213}

232void LfoController::loadSettings( const QDomElement & _this )
233{
234 Controller::loadSettings( _this );
235
236 m_baseModel.loadSettings( _this, "base" );
237 m_speedModel.loadSettings( _this, "speed" );
238 m_amountModel.loadSettings( _this, "amount" );
239 m_phaseModel.loadSettings( _this, "phase" );
240 m_waveModel.loadSettings( _this, "wave" );
241 m_multiplierModel.loadSettings( _this, "multiplier" );
242
243 if (const auto userWaveFile = _this.attribute("userwavefile"); !userWaveFile.isEmpty())
244 {
245 if (QFileInfo(PathUtil::toAbsolute(userWaveFile)).exists())
246 {
247 m_userDefSampleBuffer = gui::SampleLoader::createBufferFromFile(_this.attribute("userwavefile"));
248 }
249 else { Engine::getSong()->collectError(QString("%1: %2").arg(tr("Sample not found"), userWaveFile)); }
250 }
251
252 updateSampleFunction();
253}
