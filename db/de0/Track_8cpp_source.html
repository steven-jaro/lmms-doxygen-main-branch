49/*! \brief Create a new (empty) track object
50 *
51 * The track object is the whole track, linking its contents, its
52 * automation, name, type, and so forth.
53 *
54 * \param type The type of track (Song Editor or Pattern Editor)
55 * \param tc The track Container object to encapsulate in this track.
56 *
57 * \todo check the definitions of all the properties - are they OK?
58 */
59Track::Track( Type type, TrackContainer * tc ) :
60 Model( tc ), /*!< The track Model */
61 m_trackContainer( tc ), /*!< The track container object */
62 m_type( type ), /*!< The track type */
63 m_name(), /*!< The track's name */
64 m_mutedModel( false, this, tr( "Mute" ) ), /*!< For controlling track muting */
65 m_soloModel( false, this, tr( "Solo" ) ), /*!< For controlling track soloing */
66 m_clips() /*!< The clips (segments) */
67{
68 m_trackContainer->addTrack( this );
69 m_height = -1;
70}

96/*! \brief Create a track based on the given track type and container.
97 *
98 * \param tt The type of track to create
99 * \param tc The track container to attach to
100 */
101Track * Track::create( Type tt, TrackContainer * tc )
102{
103 Engine::audioEngine()->requestChangeInModel();
104
105 Track * t = nullptr;
106
107 switch( tt )
108 {
109 case Type::Instrument: t = new class InstrumentTrack( tc ); break;
110 case Type::Pattern: t = new class PatternTrack( tc ); break;
111 case Type::Sample: t = new class SampleTrack( tc ); break;
112// case Type::Event:
113// case Type::Video:
114 case Type::Automation: t = new class AutomationTrack( tc ); break;
115 case Type::HiddenAutomation:
116 t = new class AutomationTrack( tc, true ); break;
117 default: break;
118 }
119
120 if (tc == Engine::patternStore() && t)
121 {
122 t->createClipsForPattern(Engine::patternStore()->numOfPatterns() - 1);
123 }
124
125 tc->updateAfterTrackAdd();
126
127 Engine::audioEngine()->doneChangeInModel();
128
129 return t;
130}

176/*! \brief Save this track's settings to file
177 *
178 * We save the track type and its muted state and solo state, then append the track-
179 * specific settings. Then we iterate through the clips
180 * and save all their states in turn.
181 *
182 * \param doc The QDomDocument to use to save
183 * \param element The The QDomElement to save into
184 * \param presetMode Describes whether to save the track as a preset or not.
185 * \todo Does this accurately describe the parameters? I think not!?
186 * \todo Save the track height
187 */
188void Track::saveTrack(QDomDocument& doc, QDomElement& element, bool presetMode)
189{
190 if (!presetMode)
191 {
192 element.setTagName( "track" );
193 }
194 element.setAttribute( "type", static_cast<int>(type()) );
195 element.setAttribute( "name", name() );
196 m_mutedModel.saveSettings( doc, element, "muted" );
197 m_soloModel.saveSettings( doc, element, "solo" );
198 // Save the mutedBeforeSolo value so we can recover the muted state if any solo was active (issue 5562)
199 element.setAttribute( "mutedBeforeSolo", int(m_mutedBeforeSolo) );
200
201 if( m_height >= MINIMAL_TRACK_HEIGHT )
202 {
203 element.setAttribute( "trackheight", m_height );
204 }
205
206 if (m_color.has_value())
207 {
208 element.setAttribute("color", m_color->name());
209 }
210
211 QDomElement tsDe = doc.createElement( nodeName() );
212 // let actual track (InstrumentTrack, PatternTrack, SampleTrack etc.) save its settings
213 element.appendChild( tsDe );
214 saveTrackSpecificSettings(doc, tsDe, presetMode);
215
216 if (presetMode)
217 {
218 return;
219 }
220
221 // now save settings of all Clip's
222 for (const auto& clip : m_clips)
223 {
224 clip->saveState(doc, element);
225 }
226}

228/*! \brief Load the settings from a file
229 *
230 * We load the track's type and muted state and solo state, then clear out our
231 * current Clip.
232 *
233 * Then we step through the QDomElement's children and load the
234 * track-specific settings and clip states from it
235 * one at a time.
236 *
237 * \param element the QDomElement to load track settings from
238 * \param presetMode Indicates if a preset or a full track is loaded
239 * \todo Load the track height.
240 */
241void Track::loadTrack(const QDomElement& element, bool presetMode)
242{
243 if( static_cast<Type>(element.attribute( "type" ).toInt()) != type() )
244 {
245 qWarning( "Current track-type does not match track-type of "
246 "settings-node!\n" );
247 }
248
249 setName( element.hasAttribute( "name" ) ? element.attribute( "name" ) :
250 element.firstChild().toElement().attribute( "name" ) );
251
252 m_mutedModel.loadSettings( element, "muted" );
253 m_soloModel.loadSettings( element, "solo" );
254 // Get the mutedBeforeSolo value so we can recover the muted state if any solo was active.
255 // Older project files that didn't have this attribute will set the value to false (issue 5562)
256 m_mutedBeforeSolo = QVariant( element.attribute( "mutedBeforeSolo", "0" ) ).toBool();
257
258 if (element.hasAttribute("color"))
259 {
260 setColor(QColor{element.attribute("color")});
261 }
262
263 if (presetMode)
264 {
265 QDomNode node = element.firstChild();
266 while( !node.isNull() )
267 {
268 if( node.isElement() && node.nodeName() == nodeName() )
269 {
270 loadTrackSpecificSettings( node.toElement() );
271 break;
272 }
273 node = node.nextSibling();
274 }
275
276 return;
277 }
278
279 {
280 auto guard = Engine::audioEngine()->requestChangesGuard();
281 deleteClips();
282 }
283
284 QDomNode node = element.firstChild();
285 while( !node.isNull() )
286 {
287 if( node.isElement() )
288 {
289 if( node.nodeName() == nodeName() )
290 {
291 loadTrackSpecificSettings( node.toElement() );
292 }
293 else if( node.nodeName() != "muted"
294 && node.nodeName() != "solo"
295 && !node.toElement().attribute( "metadata" ).toInt() )
296 {
297 Clip * clip = createClip(
298 TimePos( 0 ) );
299 clip->restoreState( node.toElement() );
300 }
301 }
302 node = node.nextSibling();
303 }
304
305 int storedHeight = element.attribute( "trackheight" ).toInt();
306 if( storedHeight >= MINIMAL_TRACK_HEIGHT )
307 {
308 m_height = storedHeight;
309 }
310}

445/*! \brief Retrieve a list of clips that fall within a period.
446 *
447 * Here we're interested in a range of clips that intersect
448 * the given time period.
449 *
450 * We return the Clips we find in order by time, earliest Clips first.
451 *
452 * \param clipV The list to contain the found clips.
453 * \param start The MIDI start time of the range.
454 * \param end The MIDI endi time of the range.
455 */
456void Track::getClipsInRange( clipVector & clipV, const TimePos & start,
457 const TimePos & end )
458{
459 for( Clip* clip : m_clips )
460 {
461 int s = clip->startPosition();
462 int e = clip->endPosition();
463 if( ( s <= end ) && ( e >= start ) )
464 {
465 // Clip is within given range
466 // Insert sorted by Clip's position
467 clipV.insert(std::upper_bound(clipV.begin(), clipV.end(), clip, Clip::comparePosition),
468 clip);
469 }
470 }
471}

510/*! \brief Move all the clips after a certain time later by one bar.
511 *
512 * \param pos The time at which we want to insert the bar.
513 * \todo if we stepped through this list last to first, and the list was
514 * in ascending order by Clip time, once we hit a Clip that was earlier
515 * than the insert time, we could fall out of the loop early.
516 */
517void Track::insertBar( const TimePos & pos )
518{
519 // we'll increase the position of every Clip, positioned behind pos, by
520 // one bar
521 for (const auto& clip : m_clips)
522 {
523 if (clip->startPosition() >= pos)

553/*! \brief Return the length of the entire track in bars
554 *
555 * We step through our list of Clips and determine their end position,
556 * keeping track of the latest time found in ticks. Then we return
557 * that in bars by dividing by the number of ticks per bar.
558 */
559bar_t Track::length() const
560{
561 // find last end-position
562 tick_t last = 0;
563 for (const auto& clip : m_clips)
564 {
565 if (Engine::getSong()->isExporting() && clip->isMuted())
566 {
567 continue;
568 }
569
570 const tick_t cur = clip->endPosition();
571 if( cur > last )
572 {
573 last = cur;
574 }
575 }
576
577 return last / TimePos::ticksPerBar();
578}

582/*! \brief Invert the track's solo state.
583 *
584 * We have to go through all the tracks determining if any other track
585 * is already soloed. Then we have to save the mute state of all tracks,
586 * and set our mute state to on and all the others to off.
587 */
588void Track::toggleSolo()
589{
590 const TrackContainer::TrackList & tl = m_trackContainer->tracks();
591
592 bool soloBefore = false;
593 for (const auto& track : tl)
594 {
595 if (track != this)
596 {
597 if (track->m_soloModel.value())
598 {
599 soloBefore = true;
600 break;
601 }
602 }
603 }
604
605 const bool solo = m_soloModel.value();
606 // Should we use the new behavior of solo or the older/legacy one?
607 const bool soloLegacyBehavior = ConfigManager::inst()->value("app", "sololegacybehavior", "0").toInt();
608
609 for (const auto& track : tl)
610 {
611 if (solo)
612 {
613 // save mute-state in case no track was solo before
614 if (!soloBefore)
615 {
616 track->m_mutedBeforeSolo = track->isMuted();
617 }
618 // Don't mute AutomationTracks (keep their original state) unless we are on the sololegacybehavior mode
619 if (track == this)
620 {
621 track->setMuted(false);
622 }
623 else if (soloLegacyBehavior || track->type() != Type::Automation)
624 {
625 track->setMuted(true);
626 }
627 if (track != this)
628 {
629 track->m_soloModel.setValue(false);
630 }
631 }
632 else if (!soloBefore)
633 {
634 // Unless we are on the sololegacybehavior mode, only restores the
635 // mute state if the track isn't an Automation Track
636 if (soloLegacyBehavior || track->type() != Type::Automation)
637 {
638 track->setMuted(track->m_mutedBeforeSolo);
639 }
640 }
641 }
642}
