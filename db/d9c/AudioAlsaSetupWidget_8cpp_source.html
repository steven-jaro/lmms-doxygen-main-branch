39AudioAlsaSetupWidget::AudioAlsaSetupWidget( QWidget * _parent ) :
40 AudioDeviceSetupWidget( AudioAlsa::name(), _parent ),
41 m_selectedDevice(-1)
42{
43 QFormLayout * form = new QFormLayout(this);
44
45 m_deviceInfos = AudioAlsa::getAvailableDevices();
46
47 QString deviceText = ConfigManager::inst()->value( "audioalsa", "device" );
48
49 m_deviceComboBox = new QComboBox(this);
50 for (size_t i = 0; i < m_deviceInfos.size(); ++i)
51 {
52 AudioAlsa::DeviceInfo const & currentDeviceInfo = m_deviceInfos[i];
53 QString comboBoxText = currentDeviceInfo.getDeviceName();
54 m_deviceComboBox->addItem(comboBoxText, QVariant(static_cast<uint>(i)));
55
56 QString toolTipText = currentDeviceInfo.getDeviceDescription();
57 m_deviceComboBox->setItemData(i, toolTipText, Qt::ToolTipRole);
58
59 if (currentDeviceInfo.getDeviceName() == deviceText)
60 {
61 m_deviceComboBox->setCurrentIndex(static_cast<int>(i));
62 }
63 }
64
65 m_selectedDevice = m_deviceComboBox->currentIndex();
66
67 connect(m_deviceComboBox,
68 SIGNAL(currentIndexChanged(int)),
69 SLOT(onCurrentIndexChanged(int)));
70
71 form->addRow(tr("Device"), m_deviceComboBox);
72
73 auto m = new LcdSpinBoxModel(/* this */);
74 m->setRange(DEFAULT_CHANNELS, DEFAULT_CHANNELS);
75 m->setStep( 2 );
76 m->setValue( ConfigManager::inst()->value( "audioalsa",
77 "channels" ).toInt() );
78
79 m_channels = new LcdSpinBox( 1, this );
80 m_channels->setModel( m );
81
82 form->addRow(tr("Channels"), m_channels);
83}

96void AudioAlsaSetupWidget::saveSettings()
97{
98 QString deviceText;
99
100 if (m_selectedDevice != -1)
101 {
102 AudioAlsa::DeviceInfo const & selectedDevice = m_deviceInfos[m_selectedDevice];
103 deviceText = selectedDevice.getDeviceName();
104 }
105
106 ConfigManager::inst()->setValue( "audioalsa", "device", deviceText );
107 ConfigManager::inst()->setValue( "audioalsa", "channels",
108 QString::number( m_channels->value<int>() ) );
109}
