40CPULoadWidget::CPULoadWidget( QWidget * _parent ) :
41 QWidget( _parent ),
42 m_currentLoad( 0 ),
43 m_temp(),
44 m_background( embed::getIconPixmap( "cpuload_bg" ) ),
45 m_leds( embed::getIconPixmap( "cpuload_leds" ) ),
46 m_changed( true ),
47 m_updateTimer()
48{
49 setFixedSize( m_background.width(), m_background.height() );
50
51 m_temp = QPixmap( width(), height() );
52
53
54 connect( &m_updateTimer, SIGNAL(timeout()),
55 this, SLOT(updateCpuLoad()));
56 m_updateTimer.start( 100 ); // update cpu-load at 10 fps
57}

65void CPULoadWidget::paintEvent( QPaintEvent * )
66{
67 if( m_changed == true )
68 {
69 m_changed = false;
70
71 m_temp.fill( QColor(0,0,0,0) );
72 QPainter p( &m_temp );
73 p.drawPixmap( 0, 0, m_background );
74
75 // Normally the CPU load indicator moves smoothly, with 1 pixel resolution. However, some themes may want to
76 // draw discrete elements (like LEDs), so the stepSize property can be used to specify a larger step size.
77 int w = (m_leds.width() * std::min(m_currentLoad, 100) / (stepSize() * 100)) * stepSize();
78 if( w > 0 )
79 {
80 p.drawPixmap( 23, 3, m_leds, 0, 0, w,
81 m_leds.height() );
82 }
83 }
84 QPainter p( this );
85 p.drawPixmap( 0, 0, m_temp );
86}

91void CPULoadWidget::updateCpuLoad()
92{
93 // Additional display smoothing for the main load-value. Stronger averaging
94 // cannot be used directly in the profiler: cpuLoad() must react fast enough
95 // to be useful as overload indicator in AudioEngine::criticalXRuns().
96 const int new_load = (m_currentLoad + Engine::audioEngine()->cpuLoad()) / 2;
97
98 if (new_load != m_currentLoad)
99 {
100 auto engine = Engine::audioEngine();
101 setToolTip(
102 tr("DSP total: %1%").arg(new_load) + "\n"
103 + tr(" - Notes and setup: %1%").arg(engine->detailLoad(AudioEngineProfiler::DetailType::NoteSetup)) + "\n"
104 + tr(" - Instruments: %1%").arg(engine->detailLoad(AudioEngineProfiler::DetailType::Instruments)) + "\n"
105 + tr(" - Effects: %1%").arg(engine->detailLoad(AudioEngineProfiler::DetailType::Effects)) + "\n"
106 + tr(" - Mixing: %1%").arg(engine->detailLoad(AudioEngineProfiler::DetailType::Mixing))
107 );
108 m_currentLoad = new_load;
109 m_changed = true;
110 update();
111 }
112}
