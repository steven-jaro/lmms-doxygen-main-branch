55TimePos TimePos::quantize(float bars, bool forceRoundDown) const
56{
57 //The intervals we should snap to, our new position should be a factor of this
58 int interval = s_ticksPerBar * bars;
59 //The lower position we could snap to
60 int lowPos = m_ticks / interval;
61 //Offset from the lower position
62 int offset = m_ticks % interval;
63 //1 if we should snap up, 0 if we shouldn't
64 // Ternary expression is making sure that the snap happens in the direction to
65 // the right even if m_ticks is negative and the offset is exactly half-way
66 // More details on issue #5840 and PR #5847
67 int snapUp = forceRoundDown || ((2 * offset) == -interval)
68 ? 0
69 : (2 * offset) / interval;
70
71 return (lowPos + snapUp) * interval;
72}
