75inline JUCE_COMRESULT addHandlersToArray (const std::vector<const AccessibilityHandler*>& handlers, SAFEARRAY** pRetVal)
76{
77 auto numHandlers = handlers.size();
78
79 *pRetVal = SafeArrayCreateVector (VT_UNKNOWN, 0, (ULONG) numHandlers);
80
81 if (pRetVal != nullptr)
82 {
83 for (LONG i = 0; i < (LONG) numHandlers; ++i)
84 {
85 auto* handler = handlers[(size_t) i];
86
87 if (handler == nullptr)
88 continue;
89
90 ComSmartPtr<IRawElementProviderSimple> provider;
91 JUCE_BEGIN_IGNORE_WARNINGS_GCC_LIKE ("-Wlanguage-extension-token")
92 handler->getNativeImplementation()->QueryInterface (IID_PPV_ARGS (provider.resetAndGetPointerAddress()));
93 JUCE_END_IGNORE_WARNINGS_GCC_LIKE
94
95 auto hr = SafeArrayPutElement (*pRetVal, &i, provider);
96
97 if (FAILED (hr))
98 return E_FAIL;
99 }
100 }
101
102 return S_OK;
103}
