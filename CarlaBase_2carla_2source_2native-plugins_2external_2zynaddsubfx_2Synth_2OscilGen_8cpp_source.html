263inline void clearAll(fft_t *freqs, int oscilsize)
264{
265 memset(freqs, 0, oscilsize / 2 * sizeof(fft_t));
266}

268inline void clearDC(fft_t *freqs)
269{
270 freqs[0] = fft_t(0.0f, 0.0f);
271}

273//return magnitude squared
274inline float normal(const fft_t *freqs, off_t x)
275{
276 return norm(freqs[x]);
277}
278
279//return magnitude
280inline float abs(const fft_t *freqs, off_t x)
281{
282 return abs(freqs[x]);
283}
284
285//return angle aka phase from a sine (not cosine wave)
286inline float arg(const fft_t *freqs, off_t x)
287{
288 const fft_t tmp(freqs[x].imag(), freqs[x].real());
289 return arg(tmp);
290}

296void normalize(fft_t *freqs, int oscilsize)
297{
298 float normMax = 0.0f;
299 for(int i = 0; i < oscilsize / 2; ++i) {
300 //magnitude squared
301 const float norm = normal(freqs, i);
302 if(normMax < norm)
303 normMax = norm;
304 }
305
306 const float max = sqrt(normMax);
307 if(max < 1e-8) //data is all ~zero, do not amplify noise
308 return;
309
310 for(int i = 0; i < oscilsize / 2; ++i)
311 freqs[i] /= max;
312}
313
314//Full RMS normalize
315void rmsNormalize(fft_t *freqs, int oscilsize)
316{
317 float sum = 0.0f;
318 for(int i = 1; i < oscilsize / 2; ++i)
319 sum += normal(freqs, i);
320
321 if(sum < 0.000001f)
322 return; //data is all ~zero, do not amplify noise
323
324 const float gain = 1.0f / sqrt(sum);
325
326 for(int i = 1; i < oscilsize / 2; ++i)
327 freqs[i] *= gain;
328}

437void OscilGen::convert2sine()
438{
439 float mag[MAX_AD_HARMONICS], phase[MAX_AD_HARMONICS];
440 float oscil[synth.oscilsize];
441 fft_t *freqs = new fft_t[synth.oscilsize / 2];
442 get(oscil, -1.0f);
443 FFTwrapper *fft = new FFTwrapper(synth.oscilsize);
444 fft->smps2freqs(oscil, freqs);
445 delete (fft);
446
447 normalize(freqs, synth.oscilsize);
448
449 mag[0] = 0;
450 phase[0] = 0;
451 for(int i = 0; i < MAX_AD_HARMONICS; ++i) {
452 mag[i] = abs(freqs, i + 1);
453 phase[i] = arg(freqs, i + 1);
454 }
455
456 defaults();
457
458 for(int i = 0; i < MAX_AD_HARMONICS - 1; ++i) {
459 float newmag = mag[i];
460 float newphase = phase[i];
461
462 Phmag[i] = (int) ((newmag) * 63.0f) + 64;
463
464 Phphase[i] = 64 - (int) (64.0f * newphase / PI);
465 if(Phphase[i] > 127)
466 Phphase[i] = 127;
467
468 if(Phmag[i] == 64)
469 Phphase[i] = 64;
470 }
471 delete[] freqs;
472 prepare();
473}

488/*
489 * Get the base function
490 */
491void OscilGen::getbasefunction(float *smps)
492{
493 float par = (Pbasefuncpar + 0.5f) / 128.0f;
494 if(Pbasefuncpar == 64)
495 par = 0.5f;
496
497 float p1 = Pbasefuncmodulationpar1 / 127.0f,
498 p2 = Pbasefuncmodulationpar2 / 127.0f,
499 p3 = Pbasefuncmodulationpar3 / 127.0f;
500
501 switch(Pbasefuncmodulation) {
502 case 1:
503 p1 = (powf(2, p1 * 5.0f) - 1.0f) / 10.0f;
504 p3 = floor(powf(2, p3 * 5.0f) - 1.0f);
505 if(p3 < 0.9999f)
506 p3 = -1.0f;
507 break;
508 case 2:
509 p1 = (powf(2, p1 * 5.0f) - 1.0f) / 10.0f;
510 p3 = 1.0f + floor(powf(2, p3 * 5.0f) - 1.0f);
511 break;
512 case 3:
513 p1 = (powf(2, p1 * 7.0f) - 1.0f) / 10.0f;
514 p3 = 0.01f + (powf(2, p3 * 16.0f) - 1.0f) / 10.0f;
515 break;
516 }
517
518 base_func func = getBaseFunction(Pcurrentbasefunc);
519
520 for(int i = 0; i < synth.oscilsize; ++i) {
521 float t = i * 1.0f / synth.oscilsize;
522
523 switch(Pbasefuncmodulation) {
524 case 1: //rev
525 t = t * p3 + sinf((t + p2) * 2.0f * PI) * p1;
526 break;
527 case 2: //sine
528 t += sinf( (t * p3 + p2) * 2.0f * PI) * p1;
529 break;
530 case 3: //power
531 t += powf((1.0f - cosf((t + p2) * 2.0f * PI)) * 0.5f, p3) * p1;
532 break;
533 case 4: //chop
534 t = t * (powf(2.0, Pbasefuncmodulationpar1/32.0 +
535 Pbasefuncmodulationpar2/2048.0)) + p3;
536 }
537
538 t = t - floor(t);
539
540 if(func)
541 smps[i] = func(t, par);
542 else if (Pcurrentbasefunc == 0)
543 smps[i] = -sinf(2.0f * PI * i / synth.oscilsize);
544 else
545 smps[i] = userfunc(t);
546 }
547}

550/*
551 * Filter the oscillator
552 */
553void OscilGen::oscilfilter(fft_t *freqs)
554{
555 if(Pfiltertype == 0)
556 return;
557
558 const float par = 1.0f - Pfilterpar1 / 128.0f;
559 const float par2 = Pfilterpar2 / 127.0f;
560 filter_func filter = getFilter(Pfiltertype);
561
562 for(int i = 1; i < synth.oscilsize / 2; ++i)
563 freqs[i] *= filter(i, par, par2);
564
565 normalize(freqs, synth.oscilsize);
566}

591inline void normalize(float *smps, size_t N)
592{
593 //Find max
594 float max = 0.0f;
595 for(size_t i = 0; i < N; ++i)
596 if(max < fabs(smps[i]))
597 max = fabs(smps[i]);
598 if(max < 0.00001f)
599 max = 1.0f;
600
601 //Normalize to +-1
602 for(size_t i = 0; i < N; ++i)
603 smps[i] /= max;
604}

606/*
607 * Waveshape
608 */
609void OscilGen::waveshape(fft_t *freqs)
610{
611 oldwaveshapingfunction = Pwaveshapingfunction;
612 oldwaveshaping = Pwaveshaping;
613 if(Pwaveshapingfunction == 0)
614 return;
615
616 clearDC(freqs);
617 //reduce the amplitude of the freqs near the nyquist
618 for(int i = 1; i < synth.oscilsize / 8; ++i) {
619 float gain = i / (synth.oscilsize / 8.0f);
620 freqs[synth.oscilsize / 2 - i] *= gain;
621 }
622 fft->freqs2smps(freqs, tmpsmps);
623
624 //Normalize
625 normalize(tmpsmps, synth.oscilsize);
626
627 //Do the waveshaping
628 waveShapeSmps(synth.oscilsize, tmpsmps, Pwaveshapingfunction, Pwaveshaping);
629
630 fft->smps2freqs(tmpsmps, freqs); //perform FFT
631}

634/*
635 * Do the Frequency Modulation of the Oscil
636 */
637void OscilGen::modulation(fft_t *freqs)
638{
639 oldmodulation = Pmodulation;
640 oldmodulationpar1 = Pmodulationpar1;
641 oldmodulationpar2 = Pmodulationpar2;
642 oldmodulationpar3 = Pmodulationpar3;
643 if(Pmodulation == 0)
644 return;
645
646
647 float modulationpar1 = Pmodulationpar1 / 127.0f,
648 modulationpar2 = 0.5f - Pmodulationpar2 / 127.0f,
649 modulationpar3 = Pmodulationpar3 / 127.0f;
650
651 switch(Pmodulation) {
652 case 1:
653 modulationpar1 = (powf(2, modulationpar1 * 7.0f) - 1.0f) / 100.0f;
654 modulationpar3 = floor((powf(2, modulationpar3 * 5.0f) - 1.0f));
655 if(modulationpar3 < 0.9999f)
656 modulationpar3 = -1.0f;
657 break;
658 case 2:
659 modulationpar1 = (powf(2, modulationpar1 * 7.0f) - 1.0f) / 100.0f;
660 modulationpar3 = 1.0f
661 + floor((powf(2, modulationpar3 * 5.0f) - 1.0f));
662 break;
663 case 3:
664 modulationpar1 = (powf(2, modulationpar1 * 9.0f) - 1.0f) / 100.0f;
665 modulationpar3 = 0.01f
666 + (powf(2, modulationpar3 * 16.0f) - 1.0f) / 10.0f;
667 break;
668 }
669
670 clearDC(freqs); //remove the DC
671 //reduce the amplitude of the freqs near the nyquist
672 for(int i = 1; i < synth.oscilsize / 8; ++i) {
673 const float tmp = i / (synth.oscilsize / 8.0f);
674 freqs[synth.oscilsize / 2 - i] *= tmp;
675 }
676 fft->freqs2smps(freqs, tmpsmps);
677 const int extra_points = 2;
678 float *in = new float[synth.oscilsize + extra_points];
679
680 //Normalize
681 normalize(tmpsmps, synth.oscilsize);
682
683 for(int i = 0; i < synth.oscilsize; ++i)
684 in[i] = tmpsmps[i];
685 for(int i = 0; i < extra_points; ++i)
686 in[i + synth.oscilsize] = tmpsmps[i];
687
688 //Do the modulation
689 for(int i = 0; i < synth.oscilsize; ++i) {
690 float t = i * 1.0f / synth.oscilsize;
691
692 switch(Pmodulation) {
693 case 1:
694 t = t * modulationpar3
695 + sinf((t + modulationpar2) * 2.0f * PI) * modulationpar1; //rev
696 break;
697 case 2:
698 t = t
699 + sinf((t * modulationpar3
700 + modulationpar2) * 2.0f * PI) * modulationpar1; //sine
701 break;
702 case 3:
703 t = t + powf((1.0f - cosf(
704 (t + modulationpar2) * 2.0f * PI)) * 0.5f,
705 modulationpar3) * modulationpar1; //power
706 break;
707 }
708
709 t = (t - floor(t)) * synth.oscilsize;
710
711 const int poshi = (int) t;
712 const float poslo = t - floor(t);
713
714 tmpsmps[i] = in[poshi] * (1.0f - poslo) + in[poshi + 1] * poslo;
715 }
716
717 delete [] in;
718 fft->smps2freqs(tmpsmps, freqs); //perform FFT
719}

722/*
723 * Adjust the spectrum
724 */
725void OscilGen::spectrumadjust(fft_t *freqs)
726{
727 if(Psatype == 0)
728 return;
729 float par = Psapar / 127.0f;
730 switch(Psatype) {
731 case 1:
732 par = 1.0f - par * 2.0f;
733 if(par >= 0.0f)
734 par = powf(5.0f, par);
735 else
736 par = powf(8.0f, par);
737 break;
738 case 2:
739 par = powf(10.0f, (1.0f - par) * 3.0f) * 0.001f;
740 break;
741 case 3:
742 par = powf(10.0f, (1.0f - par) * 3.0f) * 0.001f;
743 break;
744 }
745
746 normalize(freqs, synth.oscilsize);
747
748 for(int i = 0; i < synth.oscilsize / 2; ++i) {
749 float mag = abs(freqs, i);
750 float phase = M_PI_2 - arg(freqs, i);
751
752 switch(Psatype) {
753 case 1:
754 mag = powf(mag, par);
755 break;
756 case 2:
757 if(mag < par)
758 mag = 0.0f;
759 break;
760 case 3:
761 mag /= par;
762 if(mag > 1.0f)
763 mag = 1.0f;
764 break;
765 }
766 freqs[i] = FFTpolar<fftw_real>(mag, phase);
767 }
768}

770void OscilGen::shiftharmonics(fft_t *freqs)
771{
772 if(Pharmonicshift == 0)
773 return;
774
775 int harmonicshift = -Pharmonicshift;
776 fft_t h;
777
778 if(harmonicshift > 0)
779 for(int i = synth.oscilsize / 2 - 2; i >= 0; i--) {
780 int oldh = i - harmonicshift;
781 if(oldh < 0)
782 h = 0.0f;
783 else
784 h = freqs[oldh + 1];
785 freqs[i + 1] = h;
786 }
787 else
788 for(int i = 0; i < synth.oscilsize / 2 - 1; ++i) {
789 int oldh = i + ::abs(harmonicshift);
790 if(oldh >= (synth.oscilsize / 2 - 1))
791 h = 0.0f;
792 else {
793 h = freqs[oldh + 1];
794 if(abs(h) < 0.000001f)
795 h = 0.0f;
796 }
797
798 freqs[i + 1] = h;
799 }
800
801 clearDC(freqs);
802}

812void OscilGen::prepare(fft_t *freqs)
813{
814 if((oldbasepar != Pbasefuncpar) || (oldbasefunc != Pcurrentbasefunc)
815 || DIFF(basefuncmodulation) || DIFF(basefuncmodulationpar1)
816 || DIFF(basefuncmodulationpar2) || DIFF(basefuncmodulationpar3))
817 changebasefunction();
818
819 for(int i = 0; i < MAX_AD_HARMONICS; ++i)
820 hphase[i] = (Phphase[i] - 64.0f) / 64.0f * PI / (i + 1);
821
822 for(int i = 0; i < MAX_AD_HARMONICS; ++i) {
823 const float hmagnew = 1.0f - fabs(Phmag[i] / 64.0f - 1.0f);
824 switch(Phmagtype) {
825 case 1:
826 hmag[i] = expf(hmagnew * logf(0.01f));
827 break;
828 case 2:
829 hmag[i] = expf(hmagnew * logf(0.001f));
830 break;
831 case 3:
832 hmag[i] = expf(hmagnew * logf(0.0001f));
833 break;
834 case 4:
835 hmag[i] = expf(hmagnew * logf(0.00001f));
836 break;
837 default:
838 hmag[i] = 1.0f - hmagnew;
839 break;
840 }
841
842 if(Phmag[i] < 64)
843 hmag[i] = -hmag[i];
844 }
845
846 //remove the harmonics where Phmag[i]==64
847 for(int i = 0; i < MAX_AD_HARMONICS; ++i)
848 if(Phmag[i] == 64)
849 hmag[i] = 0.0f;
850
851
852 clearAll(freqs, synth.oscilsize);
853 if(Pcurrentbasefunc == 0) //the sine case
854 for(int i = 0; i < MAX_AD_HARMONICS - 1; ++i) {
855 freqs[i + 1] =
856 std::complex<float>(-hmag[i] * sinf(hphase[i] * (i + 1)) / 2.0f,
857 hmag[i] * cosf(hphase[i] * (i + 1)) / 2.0f);
858 }
859 else
860 for(int j = 0; j < MAX_AD_HARMONICS; ++j) {
861 if(Phmag[j] == 64)
862 continue;
863 for(int i = 1; i < synth.oscilsize / 2; ++i) {
864 int k = i * (j + 1);
865 if(k >= synth.oscilsize / 2)
866 break;
867 freqs[k] += basefuncFFTfreqs[i] * FFTpolar<fftw_real>(
868 hmag[j],
869 hphase[j] * k);
870 }
871 }
872
873 if(Pharmonicshiftfirst != 0)
874 shiftharmonics(freqs);
875
876 if(Pfilterbeforews) {
877 oscilfilter(freqs);
878 waveshape(freqs);
879 } else {
880 waveshape(freqs);
881 oscilfilter(freqs);
882 }
883
884 modulation(freqs);
885 spectrumadjust(freqs);
886 if(Pharmonicshiftfirst == 0)
887 shiftharmonics(freqs);
888
889 clearDC(freqs);
890
891 oldhmagtype = Phmagtype;
892 oldharmonicshift = Pharmonicshift + Pharmonicshiftfirst * 256;
893
894 oscilprepared = 1;
895}

902void OscilGen::adaptiveharmonic(fft_t *f, float freq)
903{
904 if(Padaptiveharmonics == 0 /*||(freq<1.0f)*/)
905 return;
906 if(freq < 1.0f)
907 freq = 440.0f;
908
909 fft_t *inf = new fft_t[synth.oscilsize / 2];
910 for(int i = 0; i < synth.oscilsize / 2; ++i)
911 inf[i] = f[i];
912 clearAll(f, synth.oscilsize);
913 clearDC(inf);
914
915 float basefreq = 30.0f * powf(10.0f, Padaptiveharmonicsbasefreq / 128.0f);
916 float power = (Padaptiveharmonicspower + 1.0f) / 101.0f;
917
918 float rap = freq / basefreq;
919
920 rap = powf(rap, power);
921
922 bool down = false;
923 if(rap > 1.0f) {
924 rap = 1.0f / rap;
925 down = true;
926 }
927
928 for(int i = 0; i < synth.oscilsize / 2 - 2; ++i) {
929 const int high = (int)(i * rap);
930 const float low = fmod(i * rap, 1.0f);
931
932 if(high >= (synth.oscilsize / 2 - 2))
933 break;
934
935 if(down) {
936 f[high] += (1.0f - low) * inf[i];
937 f[high + 1] += low * inf[i];
938 }
939 else {
940 f[i] = (1.0f - low) * inf[high] + low * inf[high + 1];
941 }
942 }
943 if(!down)//corect the aplitude of the first harmonic
944 f[0] *= rap;
945
946 f[1] += f[0];
947 clearDC(f);
948 delete[] inf;
949}

951void OscilGen::adaptiveharmonicpostprocess(fft_t *f, int size)
952{
953 if(Padaptiveharmonics <= 1)
954 return;
955 fft_t *inf = new fft_t[size];
956 float par = Padaptiveharmonicspar * 0.01f;
957 par = 1.0f - powf((1.0f - par), 1.5f);
958
959 for(int i = 0; i < size; ++i) {
960 inf[i] = f[i] * double(par);
961 f[i] *= (1.0f - par);
962 }
963
964
965 if(Padaptiveharmonics == 2) { //2n+1
966 for(int i = 0; i < size; ++i)
967 if((i % 2) == 0)
968 f[i] += inf[i]; //i=0 first harmonic,etc.
969 }
970 else { //other ways
971 int nh = (Padaptiveharmonics - 3) / 2 + 2;
972 int sub_vs_add = (Padaptiveharmonics - 3) % 2;
973 if(sub_vs_add == 0) {
974 for(int i = 0; i < size; ++i)
975 if(((i + 1) % nh) == 0)
976 f[i] += inf[i];
977 }
978 else
979 for(int i = 0; i < size / nh - 1; ++i)
980 f[(i + 1) * nh - 1] += inf[i];
981 }
982
983 delete [] inf;
984}

1031/*
1032 * Get the oscillator function
1033 */
1034short int OscilGen::get(float *smps, float freqHz, int resonance)
1035{
1036 if(needPrepare())
1037 prepare();
1038
1039 fft_t *input = freqHz > 0.0f ? oscilFFTfreqs : pendingfreqs;
1040
1041 int outpos =
1042 (int)((RND * 2.0f
1043 - 1.0f) * synth.oscilsize_f * (Prand - 64.0f) / 64.0f);
1044 outpos = (outpos + 2 * synth.oscilsize) % synth.oscilsize;
1045
1046
1047 clearAll(outoscilFFTfreqs, synth.oscilsize);
1048
1049 int nyquist = (int)(0.5f * synth.samplerate_f / fabs(freqHz)) + 2;
1050 if(ADvsPAD)
1051 nyquist = (int)(synth.oscilsize / 2);
1052 if(nyquist > synth.oscilsize / 2)
1053 nyquist = synth.oscilsize / 2;
1054
1055 //Process harmonics
1056 {
1057 int realnyquist = nyquist;
1058
1059 if(Padaptiveharmonics != 0)
1060 nyquist = synth.oscilsize / 2;
1061 for(int i = 1; i < nyquist - 1; ++i)
1062 outoscilFFTfreqs[i] = input[i];
1063
1064 adaptiveharmonic(outoscilFFTfreqs, freqHz);
1065 adaptiveharmonicpostprocess(&outoscilFFTfreqs[1],
1066 synth.oscilsize / 2 - 1);
1067
1068 nyquist = realnyquist;
1069 }
1070
1071 if(Padaptiveharmonics) //do the antialiasing in the case of adaptive harmonics
1072 for(int i = nyquist; i < synth.oscilsize / 2; ++i)
1073 outoscilFFTfreqs[i] = fft_t(0.0f, 0.0f);
1074
1075 // Randomness (each harmonic), the block type is computed
1076 // in ADnote by setting start position according to this setting
1077 if((Prand > 64) && (freqHz >= 0.0f) && (!ADvsPAD)) {
1078 const float rnd = PI * powf((Prand - 64.0f) / 64.0f, 2.0f);
1079 for(int i = 1; i < nyquist - 1; ++i) //to Nyquist only for AntiAliasing
1080 outoscilFFTfreqs[i] *=
1081 FFTpolar<fftw_real>(1.0f, (float)(rnd * i * RND));
1082 }
1083
1084 //Harmonic Amplitude Randomness
1085 if((freqHz > 0.1f) && (!ADvsPAD)) {
1086 unsigned int realrnd = prng();
1087 sprng(randseed);
1088 float power = Pamprandpower / 127.0f;
1089 float normalize = 1.0f / (1.2f - power);
1090 switch(Pamprandtype) {
1091 case 1:
1092 power = power * 2.0f - 0.5f;
1093 power = powf(15.0f, power);
1094 for(int i = 1; i < nyquist - 1; ++i)
1095 outoscilFFTfreqs[i] *= powf(RND, power) * normalize;
1096 break;
1097 case 2:
1098 power = power * 2.0f - 0.5f;
1099 power = powf(15.0f, power) * 2.0f;
1100 float rndfreq = 2 * PI * RND;
1101 for(int i = 1; i < nyquist - 1; ++i)
1102 outoscilFFTfreqs[i] *= powf(fabs(sinf(i * rndfreq)), power)
1103 * normalize;
1104 break;
1105 }
1106 sprng(realrnd + 1);
1107 }
1108
1109 if((freqHz > 0.1f) && (resonance != 0))
1110 res->applyres(nyquist - 1, outoscilFFTfreqs, freqHz);
1111
1112 rmsNormalize(outoscilFFTfreqs, synth.oscilsize);
1113
1114 if((ADvsPAD) && (freqHz > 0.1f)) //in this case the smps will contain the freqs
1115 for(int i = 1; i < synth.oscilsize / 2; ++i)
1116 smps[i - 1] = abs(outoscilFFTfreqs, i);
1117 else {
1118 fft->freqs2smps(outoscilFFTfreqs, smps);
1119 for(int i = 0; i < synth.oscilsize; ++i)
1120 smps[i] *= 0.25f; //correct the amplitude
1121 }
1122
1123 if(Prand < 64)
1124 return outpos;
1125 else
1126 return 0;
1127}
