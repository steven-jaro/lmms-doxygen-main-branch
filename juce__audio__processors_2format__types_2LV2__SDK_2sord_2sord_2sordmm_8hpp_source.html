101 inline void add(const std::string& name, const std::string& uri)
102 {
103 const SerdNode name_node = string_to_node(SERD_LITERAL, name);
104 const SerdNode uri_node = string_to_node(SERD_URI, uri);
105 serd_env_set_prefix(_c_obj, &name_node, &uri_node);

108 inline std::string qualify(std::string uri) const
109 {
110 const SerdNode uri_node = string_to_node(SERD_URI, uri);
111 SerdNode prefix;
112 SerdChunk suffix;
113 if (serd_env_qualify(_c_obj, &uri_node, &prefix, &suffix)) {
114 std::string ret(reinterpret_cast<const char*>(prefix.buf),
115 prefix.n_bytes);
116 ret.append(":").append(reinterpret_cast<const char*>(suffix.buf),
117 suffix.len);
118 return ret;
119 }
120 return uri;
121 }
122
123 inline std::string expand(const std::string& curie) const
124 {
125 assert(curie.find(':') != std::string::npos);
126 SerdNode curie_node = string_to_node(SERD_CURIE, curie);
127 SerdChunk uri_prefix;
128 SerdChunk uri_suffix;
129 if (!serd_env_expand(_c_obj, &curie_node, &uri_prefix, &uri_suffix)) {
130 std::string ret(reinterpret_cast<const char*>(uri_prefix.buf),
131 uri_prefix.len);
132 ret.append(reinterpret_cast<const char*>(uri_suffix.buf), uri_suffix.len);
133 return ret;
134 }
135 std::cerr << "CURIE `" << curie << "' has unknown prefix." << std::endl;
136 return curie;
137 }

331inline Node::Node(World& world, Type type, const std::string& s)
332 : _world(&world)
333{
334 switch (type) {

347 reinterpret_cast<const unsigned char*>(s.c_str()));

606inline SerdStatus
607Model::write_to_file(const std::string& uri, SerdSyntax syntax, SerdStyle style)
608{
609 uint8_t* path =
610 serd_file_uri_parse(reinterpret_cast<const uint8_t*>(uri.c_str()), nullptr);
611 if (!path) {
612 fprintf(stderr, "Failed to parse file URI <%s>\n", uri.c_str());
613 return SERD_ERR_BAD_ARG;
614 }
615
616 FILE* const fd = fopen(reinterpret_cast<const char*>(path), "w");
617 if (!fd) {
618 fprintf(stderr, "Failed to open file %s\n", path);
619 serd_free(path);
620 return SERD_ERR_UNKNOWN;
621 }
622 serd_free(path);
623
624 SerdURI base_uri = SERD_URI_NULL;
625 if (serd_uri_parse(reinterpret_cast<const uint8_t*>(uri.c_str()),
626 &base_uri)) {
627 fprintf(stderr, "Invalid base URI <%s>\n", uri.c_str());
628 fclose(fd);
629 return SERD_ERR_BAD_ARG;
630 }
631
632 SerdWriter* writer = serd_writer_new(
633 syntax, style, _world.prefixes().c_obj(), &base_uri, serd_file_sink, fd);
634
635 serd_env_foreach(_world.prefixes().c_obj(),
636 reinterpret_cast<SerdPrefixSink>(serd_writer_set_prefix),
637 writer);
638
639 sord_write(_c_obj, writer, nullptr);
640 serd_writer_free(writer);
641 fclose(fd);
642
643 return SERD_SUCCESS;
644}
