78 inline sample_t process (sample_t s)
79 {
80 int z = h;
81
82 sample_t r = s * a[0];
83
84 r += a[1] * x[z];
85 r += b[1] * y[z];
86
87 z ^= 1;
88 r += a[2] * x[z];
89 r += b[2] * y[z];
90
91 y[z] = r;
92 x[z] = s;
93
94 h = z;
95
96 return r;
97 }
