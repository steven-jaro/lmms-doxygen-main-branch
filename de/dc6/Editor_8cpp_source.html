94Editor::Editor(bool record, bool stepRecord) :
95 m_toolBar(new DropToolBar(this)),
96 m_playAction(nullptr),
97 m_recordAction(nullptr),
98 m_recordAccompanyAction(nullptr),
99 m_toggleStepRecordingAction(nullptr),
100 m_stopAction(nullptr)
101{
102 m_toolBar = addDropToolBarToTop(tr("Transport controls"));
103
104 auto addButton = [this](QAction* action, QString objectName) {
105 m_toolBar->addAction(action);
106 m_toolBar->widgetForAction(action)->setObjectName(objectName);
107 };
108
109 // Set up play and record actions
110 m_playAction = new QAction(embed::getIconPixmap("play"), tr("Play (Space)"), this);
111 m_stopAction = new QAction(embed::getIconPixmap("stop"), tr("Stop (Space)"), this);
112
113 m_recordAction = new QAction(embed::getIconPixmap("record"), tr("Record"), this);
114 m_recordAccompanyAction = new QAction(embed::getIconPixmap("record_accompany"), tr("Record while playing"), this);
115 m_toggleStepRecordingAction = new QAction(embed::getIconPixmap("record_step_off"), tr("Toggle Step Recording"), this);
116
117 // Set up connections
118 connect(m_playAction, SIGNAL(triggered()), this, SLOT(play()));
119 connect(m_recordAction, SIGNAL(triggered()), this, SLOT(record()));
120 connect(m_recordAccompanyAction, SIGNAL(triggered()), this, SLOT(recordAccompany()));
121 connect(m_toggleStepRecordingAction, SIGNAL(triggered()), this, SLOT(toggleStepRecording()));
122 connect(m_stopAction, SIGNAL(triggered()), this, SLOT(stop()));
123 new QShortcut(QKeySequence(combine(Qt::SHIFT, Qt::Key_F11)), this, SLOT(toggleMaximize()));
124
125 // Add actions to toolbar
126 addButton(m_playAction, "playButton");
127 if (record)
128 {
129 addButton(m_recordAction, "recordButton");
130 addButton(m_recordAccompanyAction, "recordAccompanyButton");
131 }
132 if(stepRecord)
133 {
134 addButton(m_toggleStepRecordingAction, "stepRecordButton");
135 }
136 addButton(m_stopAction, "stopButton");
137}

158 void Editor::keyPressEvent(QKeyEvent *ke)
159 {
160 if (ke->key() == Qt::Key_Space)
161 {
162 if (ke->modifiers() & Qt::ShiftModifier)
163 {
164 togglePause();
165 }
166 else
167 {
168 togglePlayStop();
169 }
170 return;
171 }
172 ke->ignore();
173 }
