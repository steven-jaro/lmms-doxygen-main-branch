36/* Returns biggest value from abs_spectrum[spec_size] array.
37 *
38 * return -1 on error, otherwise the maximum value
39 */
40float maximum(const float *abs_spectrum, unsigned int spec_size)
41{
42 if (abs_spectrum == nullptr) {return -1;}
43 if (spec_size == 0) {return -1;}
44
45 float maxi = 0;
46
47 for (unsigned int i = 0; i < spec_size; i++)
48 {
49 if (abs_spectrum[i] > maxi) {maxi = abs_spectrum[i];}
50 }
51 return maxi;
52}

60/* Normalize the array of absolute magnitudes to a 0..1 range.
61 * Block size refers to FFT block size before any zero padding.
62 *
63 * return -1 on error, 0 on success
64 */
65int normalize(const float *abs_spectrum, float *norm_spectrum, unsigned int bin_count, unsigned int block_size)
66{
67 if (abs_spectrum == nullptr || norm_spectrum == nullptr) {return -1;}
68 if (bin_count == 0 || block_size == 0) {return -1;}
69
70 block_size /= 2;
71 for (unsigned int i = 0; i < bin_count; i++)
72 {
73 norm_spectrum[i] = abs_spectrum[i] / block_size;
74 }
75 return 0;
76}

79{
80 if (abs_spectrum.size() != norm_spectrum.size()) {return -1;}

101/* Precompute an FFT window function for later real-time use.
102 *
103 * return -1 on error
104 */
105int precomputeWindow(float *window, unsigned int length, FFTWindow type, bool normalized)
106{
107 using namespace std::numbers;
108 if (window == nullptr) {return -1;}
109
110 float gain = 0;
111 float a0;
112 float a1;
113 float a2;
114 float a3;
115
116 // constants taken from
117 // https://en.wikipedia.org/wiki/Window_function#AList_of_window_functions
118 switch (type)
119 {
120 default:
121 case FFTWindow::Rectangular:
122 for (unsigned int i = 0; i < length; i++) {window[i] = 1.0;}
123 gain = 1;
124 return 0;
125 case FFTWindow::BlackmanHarris:
126 a0 = 0.35875f;
127 a1 = 0.48829f;
128 a2 = 0.14128f;
129 a3 = 0.01168f;
130 break;
131 case FFTWindow::Hamming:
132 a0 = 0.54f;
133 a1 = 1.0 - a0;
134 a2 = 0;
135 a3 = 0;
136 break;
137 case FFTWindow::Hanning:
138 a0 = 0.5;
139 a1 = 1.0 - a0;
140 a2 = 0;
141 a3 = 0;
142 break;
143 }
144
145 // common computation for cosine-sum based windows
146 for (unsigned int i = 0; i < length; i++)
147 {
148 window[i] = (a0 - a1 * std::cos(2 * pi_v<float> * i / (static_cast<float>(length) - 1.0))
149 + a2 * std::cos(4 * pi_v<float> * i / (static_cast<float>(length) - 1.0))
150 - a3 * std::cos(6 * pi_v<float> * i / (static_cast<float>(length) - 1.0)));
151 gain += window[i];
152 }
153
154 // apply amplitude correction
155 gain /= (float) length;
156 for (unsigned int i = 0; i < length; i++) {window[i] /= gain;}
157
158 return 0;
159}

162/* Compute absolute values of complex_buffer, save to absspec_buffer.
163 * Take care that - compl_len is not bigger than complex_buffer!
164 * - absspec buffer is big enough!
165 *
166 * return 0 on success, else -1
167 */
168int absspec(const fftwf_complex *complex_buffer, float *absspec_buffer, unsigned int compl_length)
169{
170 if (complex_buffer == nullptr || absspec_buffer == nullptr) {return -1;}
171 if (compl_length == 0) {return -1;}
172
173 for (unsigned int i = 0; i < compl_length; i++)
174 {
175 absspec_buffer[i] = (float)sqrt(complex_buffer[i][0] * complex_buffer[i][0]
176 + complex_buffer[i][1] * complex_buffer[i][1]);
177 }
178
179 return 0;
180}

183/* Build fewer subbands from many absolute spectrum values.
184 * Take care that - compressedbands[] array num_new elements long
185 * - num_old > num_new
186 *
187 * return 0 on success, else -1
188 */
189int compressbands(const float *absspec_buffer, float *compressedband, int num_old, int num_new, int bottom, int top)
190{
191 if (absspec_buffer == nullptr || compressedband == nullptr) {return -1;}
192 if (num_old < num_new) {return -1;}
193 if (num_old <= 0 || num_new <= 0) {return -1;}
194 if (bottom < 0) {bottom = 0;}
195 if (top >= num_old) {top = num_old - 1;}
196
197 int usefromold = num_old - (num_old - top) - bottom;
198
199 float ratio = (float)usefromold / (float)num_new;
200
201 // for each new subband
202 for (int i = 0; i < num_new; i++)
203 {
204 compressedband[i] = 0;
205
206 float j_min = (i * ratio) + bottom;
207
208 if (j_min < 0) { j_min = static_cast<float>(bottom); }
209
210 float j_max = j_min + ratio;
211
212 for (float j = std::floor(j_min); j <= j_max; j++)
213 {
214 compressedband[i] += absspec_buffer[(int)j];
215 }
216 }
217
218 return 0;
219}
