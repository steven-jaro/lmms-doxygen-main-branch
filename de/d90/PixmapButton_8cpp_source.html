53void PixmapButton::paintEvent(QPaintEvent*)
54{
55 QPainter p(this);
56
57 QPixmap* pixmapToDraw = isActive() ? &m_activePixmap : &m_inactivePixmap;
58
59 if (!pixmapToDraw->isNull())
60 {
61 p.drawPixmap(0, 0, *pixmapToDraw);
62 }
63}
