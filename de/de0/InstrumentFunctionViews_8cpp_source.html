41InstrumentFunctionNoteStackingView::InstrumentFunctionNoteStackingView( InstrumentFunctionNoteStacking* cc, QWidget* parent ) :
42 QWidget( parent ),
43 ModelView( nullptr, this ),
44 m_cc( cc ),
45 m_chordsGroupBox( new GroupBox( tr( "STACKING" ) ) ),
46 m_chordsComboBox( new ComboBox() ),

55 mainLayout->setColumnStretch( 0, 1 );
56 mainLayout->setHorizontalSpacing( 20 );
57 mainLayout->setVerticalSpacing( 1 );
58

63
64 mainLayout->addWidget( chordLabel, 0, 0 );
65 mainLayout->addWidget( m_chordsComboBox, 1, 0 );
66 mainLayout->addWidget( m_chordRangeKnob, 0, 1, 2, 1, Qt::AlignHCenter );
