80void LcdSpinBox::mousePressEvent( QMouseEvent* event )
81{
82 if( event->button() == Qt::LeftButton &&
83 ! (event->modifiers() & KBD_COPY_MODIFIER) &&
84 event->y() < cellHeight() + 2 )
85 {
86 m_mouseMoving = true;
87 m_lastMousePos = event->globalPos();
88
89 AutomatableModel *thisModel = model();
90 if( thisModel )
91 {
92 thisModel->addJournalCheckPoint();
93 thisModel->saveJournallingState( false );
94 }
95 }
96 else
97 {
98 IntModelView::mousePressEvent( event );
99 }
100}

105void LcdSpinBox::mouseMoveEvent( QMouseEvent* event )
106{
107 if( m_mouseMoving )
108 {
109 int dy = event->globalY() - m_lastMousePos.y();
110 if( dy )
111 {
112 auto fdy = static_cast<float>(dy);
113 if( event->modifiers() & Qt::ShiftModifier ) {
114 fdy = qBound( -4.f, fdy/4.f, 4.f );
115 }
116 float floatValNotRounded =
117 model()->value() + m_remainder - fdy / 2.f * model()->step<int>();
118 float floatValRounded = roundf( floatValNotRounded );
119 m_remainder = floatValNotRounded - floatValRounded;
120 model()->setValue( floatValRounded );
121 emit manualChange();
122 m_lastMousePos = event->globalPos();
123 }
124 }
125}

142void LcdSpinBox::wheelEvent(QWheelEvent * we)
143{
144 we->accept();
145 const int direction = (we->angleDelta().y() > 0 ? 1 : -1) * (we->inverted() ? -1 : 1);
146
147 model()->setValue(model()->value() + direction * model()->step<int>());
148 emit manualChange();
149}

156void LcdSpinBox::enterValue()
157{
158 bool ok;
159 int new_val;
160
161 new_val = QInputDialog::getInt(
162 this, tr( "Set value" ),
163 tr( "Please enter a new value between %1 and %2:" ).
164 arg( model()->minValue() ).
165 arg( model()->maxValue() ),
166 model()->value(),
167 model()->minValue(),
168 model()->maxValue(),
169 model()->step<int>(), &ok );
170
171 if( ok )
172 {
173 model()->setValue( new_val );
174 }
175}
