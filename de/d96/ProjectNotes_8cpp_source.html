52ProjectNotes::ProjectNotes() :
53 QMainWindow( getGUI()->mainWindow()->workspace() )
54{
55 m_edit = new QTextEdit( this );
56 m_edit->setAutoFillBackground( true );
57 QPalette pal;
58 pal.setColor( m_edit->backgroundRole(), QColor( 64, 64, 64 ) );
59 m_edit->setPalette( pal );
60 m_edit->show();
61
62 clear();
63
64 connect( m_edit,
65 SIGNAL( currentCharFormatChanged( const QTextCharFormat& ) ),
66 this, SLOT( formatChanged( const QTextCharFormat& ) ) );
67// connect( m_edit, SIGNAL(currentAlignmentChanged(int)),
68// this, SLOT(alignmentChanged(int)));
69 connect( m_edit, SIGNAL(textChanged()),
70 Engine::getSong(), SLOT(setModified()));
71
72 setupActions();
73
74 setCentralWidget( m_edit );
75 setWindowTitle( tr( "Project Notes" ) );
76 setWindowIcon( embed::getIconPixmap( "project_notes" ) );
77
78 getGUI()->mainWindow()->addWindowedWidget( this );
79 parentWidget()->setAttribute( Qt::WA_DeleteOnClose, false );
80 parentWidget()->move( 700, 10 );
81 parentWidget()->resize( 400, 300 );
82 parentWidget()->hide();
83}

331void ProjectNotes::formatChanged( const QTextCharFormat & _f )
332{
333 QFont font = _f.font();
334 m_comboFont->lineEdit()->setText( font.family() );
335 m_comboSize->lineEdit()->setText( QString::number( font.pointSize() ) );
336 m_actionTextBold->setChecked( font.bold() );
337 m_actionTextItalic->setChecked( font.italic() );
338 m_actionTextUnderline->setChecked( font.underline() );
339
340 QPixmap pix( 16, 16 );
341 pix.fill( _f.foreground().color() );
342 m_actionTextColor->setIcon( pix );
343
344 Engine::getSong()->setModified();
345}
