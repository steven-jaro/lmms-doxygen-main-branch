234static __inline__
235JSList*
236jack_slist_sort_merge (JSList *l1,
237 JSList *l2,
238 JCompareFunc compare_func)
239{
240 JSList list, *l;
241
242 l = &list;
243
244 while (l1 && l2) {
245 if (compare_func(l1->data, l2->data) < 0) {
246 l = l->next = l1;
247 l1 = l1->next;
248 } else {
249 l = l->next = l2;
250 l2 = l2->next;
251 }
252 }
253 l->next = l1 ? l1 : l2;
254
255 return list.next;
256}

258static __inline__
259JSList*
260jack_slist_sort (JSList *list,
261 JCompareFunc compare_func)
262{
263 JSList *l1, *l2;
264
265 if (!list)
266 return NULL;
267 if (!list->next)
268 return list;
269
270 l1 = list;
271 l2 = list->next;
272
273 while ((l2 = l2->next) != NULL) {
274 if ((l2 = l2->next) == NULL)
275 break;
276 l1 = l1->next;
277 }
278 l2 = l1->next;
279 l1->next = NULL;
280
281 return jack_slist_sort_merge (jack_slist_sort (list, compare_func),
282 jack_slist_sort (l2, compare_func),
283 compare_func);
284}
