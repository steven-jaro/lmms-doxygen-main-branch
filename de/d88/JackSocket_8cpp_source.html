167int JackClientSocket::Read(void* data, int len)
168{
169 int res;
170
171#if defined(__sun__) || defined(sun)
172 if (fTimeOut > 0) {
173
174 struct timeval tv;
175 fd_set fdset;
176 ssize_t res;
177
178 tv.tv_sec = fTimeOut;
179 tv.tv_usec = 0;
180
181 FD_ZERO(&fdset);
182 FD_SET(fSocket, &fdset);
183
184 do {
185 res = select(fSocket + 1, &fdset, NULL, NULL, &tv);
186 } while (res < 0 && errno == EINTR);
187
188 if (res < 0) {
189 return res;
190 } else if (res == 0) {
191 return -1;
192 }
193 }
194#endif
195
196 if ((res = read(fSocket, data, len)) != len) {
197 if (errno == EWOULDBLOCK || errno == EAGAIN) {
198 jack_log("JackClientSocket::Read time out");
199 // For a non blocking socket, a read failure is not considered as an error
200 return 0;
201 } else if (res != 0) {
202 jack_error("Cannot read socket fd = %d res = %d err = %s", fSocket, res, strerror(errno));
203 //return 0;
204 return -1;
205 } else if (errno == 0) {
206 // aborted reading due to shutdown
207 return JACK_REQUEST_ERR_ABORTED;
208 } else {
209 jack_error("Cannot read socket fd = %d err = %s", fSocket, strerror(errno));
210 return -1;
211 }
212 } else {
213 return 0;
214 }

270int JackServerSocket::Bind(const char* dir, const char* name, int which) // A revoir : utilisation de "which"
271{
272 struct sockaddr_un addr;
273
274 if ((fSocket = socket(AF_UNIX, SOCK_STREAM, 0)) < 0) {
275 jack_error("Cannot create server socket err = %s", strerror(errno));
276 return -1;
277 }
278
279 addr.sun_family = AF_UNIX;
280 // Socket name has to be kept in fName to be "unlinked".
281 BuildName(name, fName, dir, which, sizeof(addr.sun_path), fPromiscuous);
282 strncpy(addr.sun_path, fName, sizeof(addr.sun_path) - 1);
283
284 jack_log("JackServerSocket::Bind : addr.sun_path %s", addr.sun_path);
285 unlink(fName); // Security...
286
287 if (bind(fSocket, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
288 jack_error("Cannot bind server to socket err = %s", strerror(errno));
289 goto error;
290 }
291
292 if (listen(fSocket, 100) < 0) {
293 jack_error("Cannot enable listen on server socket err = %s", strerror(errno));
294 goto error;
295 }
296
297 if (fPromiscuous && (jack_promiscuous_perms(-1, fName, fPromiscuousGid) < 0))
298 goto error;
299
300 return 0;
301
302error:
303 unlink(fName);
304 close(fSocket);
305 fSocket = -1;
306 return -1;
307}
