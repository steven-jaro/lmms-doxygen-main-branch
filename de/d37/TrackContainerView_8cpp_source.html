190void TrackContainerView::moveTrackView( TrackView * trackView, int indexTo )
191{
192 // Can't move out of bounds
193 if ( indexTo >= m_trackViews.size() || indexTo < 0 ) { return; }
194
195 // Does not need to move to itself
196 int indexFrom = m_trackViews.indexOf( trackView );
197 if ( indexFrom == indexTo ) { return; }
198
199 PatternTrack::swapPatternTracks( trackView->getTrack(),
200 m_trackViews[indexTo]->getTrack() );
201
202 m_scrollLayout->removeWidget( trackView );
203 m_scrollLayout->insertWidget( indexTo, trackView );
204
205 Track * track = m_tc->m_tracks[indexFrom];
206
207 m_tc->m_tracks.erase(m_tc->m_tracks.begin() + indexFrom);
208 m_tc->m_tracks.insert(m_tc->m_tracks.begin() + indexTo, track);
209 m_trackViews.move( indexFrom, indexTo );
210
211 realignTracks();
212}

234void TrackContainerView::scrollToTrackView( TrackView * _tv )
235{
236 if (!m_trackViews.contains(_tv))
237 {
238 qWarning("TrackContainerView::scrollToTrackView: TrackView is not owned by this");
239 }
240 else
241 {
242 int currentScrollTop = m_scrollArea->verticalScrollBar()->value();
243 int scrollAreaHeight = m_scrollArea->size().height();
244 int trackViewTop = _tv->pos().y();
245 int trackViewBottom = trackViewTop + _tv->size().height();
246
247 // displayed_location = widget_location - currentScrollTop
248 // want to make sure that the widget top has displayed location > 0,
249 // and widget bottom < scrollAreaHeight
250 // trackViewTop - scrollY > 0 && trackViewBottom - scrollY < scrollAreaHeight
251 // therefore scrollY < trackViewTop && scrollY > trackViewBottom - scrollAreaHeight
252 int newScroll = std::max( trackViewBottom-scrollAreaHeight, std::min(currentScrollTop, trackViewTop) );
253 m_scrollArea->verticalScrollBar()->setValue(newScroll);
254 }
255}

306const TrackView * TrackContainerView::trackViewAt( const int _y ) const
307{
308 const int abs_y = _y + m_scrollArea->verticalScrollBar()->value();
309 int y_cnt = 0;
310
311// debug code
312// qDebug( "abs_y %d", abs_y );
313
314 for (const auto& trackView : m_trackViews)
315 {
316 const int y_cnt1 = y_cnt;
317 y_cnt += trackView->height();
318 if (abs_y >= y_cnt1 && abs_y < y_cnt) { return trackView; }
319 }
320 return( nullptr );
321}

392void TrackContainerView::dropEvent( QDropEvent * _de )
393{
394 QString type = StringPairDrag::decodeKey( _de );
395 QString value = StringPairDrag::decodeValue( _de );
396 if( type == "instrument" )
397 {
398 auto it = dynamic_cast<InstrumentTrack*>(Track::create(Track::Type::Instrument, m_tc));
399 auto ilt = new InstrumentLoaderThread(this, it, value);
400 ilt->start();
401 //it->toggledInstrumentTrackButton( true );
402 _de->accept();
403 }
404 else if( type == "samplefile" || type == "pluginpresetfile"
405 || type == "soundfontfile" || type == "vstpluginfile"
406 || type == "patchfile" )
407 {
408 auto it = dynamic_cast<InstrumentTrack*>(Track::create(Track::Type::Instrument, m_tc));
409 PluginFactory::PluginInfoAndKey piakn =
410 getPluginFactory()->pluginSupportingExtension(FileItem::extension(value));
411 Instrument * i = it->loadInstrument(piakn.info.name(), &piakn.key);
412 i->loadFile( value );
413 //it->toggledInstrumentTrackButton( true );
414 _de->accept();
415 }
416 else if( type == "presetfile" )
417 {
418 DataFile dataFile( value );
419 auto it = dynamic_cast<InstrumentTrack*>(Track::create(Track::Type::Instrument, m_tc));
420 it->loadPreset(dataFile.content().toElement());
421
422 //it->toggledInstrumentTrackButton( true );
423 _de->accept();
424 }
425 else if( type == "importedproject" )
426 {
427 ImportFilter::import( value, m_tc );
428 _de->accept();
429 }
430
431 else if( type == "projectfile")
432 {
433 if( getGUI()->mainWindow()->mayChangeProject(true) )
434 {
435 Engine::getSong()->loadProject( value );
436 }
437 _de->accept();
438 }
439
440 else if( type.left( 6 ) == "track_" )
441 {
442 DataFile dataFile( value.toUtf8() );
443 Track::create( dataFile.content().firstChild().toElement(), m_tc );
444 _de->accept();
445 }
446}
