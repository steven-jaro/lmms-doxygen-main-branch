112/* This routine is called whenever the stream state changes */
113static void stream_state_callback( pa_stream *s, void * userdata )
114{
115 switch( pa_stream_get_state( s ) )
116 {
117 case PA_STREAM_CREATING:
118 case PA_STREAM_TERMINATED:
119 break;
120
121 case PA_STREAM_READY:
122 qDebug( "Stream successfully created\n" );
123 break;
124
125 case PA_STREAM_FAILED:
126 default:
127 qCritical( "Stream error: %s\n",
128 pa_strerror(pa_context_errno(
129 pa_stream_get_context( s ) ) ) );
130 }
131}

135/* This is called whenever the context status changes */
136static void context_state_callback(pa_context *c, void *userdata)
137{
138 auto _this = static_cast<AudioPulseAudio*>(userdata);
139 switch( pa_context_get_state( c ) )
140 {
141 case PA_CONTEXT_CONNECTING:
142 case PA_CONTEXT_AUTHORIZING:
143 case PA_CONTEXT_SETTING_NAME:
144 break;
145
146 case PA_CONTEXT_READY:
147 {
148 qDebug( "Connection established.\n" );
149 _this->m_s = pa_stream_new( c, "lmms", &_this->m_sampleSpec, nullptr);
150 pa_stream_set_state_callback( _this->m_s, stream_state_callback, _this );
151 pa_stream_set_write_callback( _this->m_s, stream_write_callback, _this );
152
153 pa_buffer_attr buffer_attr;
154
155 buffer_attr.maxlength = (uint32_t)(-1);
156
157 // play silence in case of buffer underrun instead of using default rewind
158 buffer_attr.prebuf = 0;
159
160 buffer_attr.minreq = (uint32_t)(-1);
161 buffer_attr.fragsize = (uint32_t)(-1);
162
163 double latency = (double)( Engine::audioEngine()->framesPerPeriod() ) / (double)_this->sampleRate();
164
165 // ask PulseAudio for the desired latency (which might not be approved)
166 buffer_attr.tlength = pa_usec_to_bytes( latency * PA_USEC_PER_MSEC,
167 &_this->m_sampleSpec );
168
169 pa_stream_connect_playback( _this->m_s, nullptr, &buffer_attr,
170 PA_STREAM_ADJUST_LATENCY,
171 nullptr, // volume
172 nullptr );
173 _this->signalConnected( true );
174 break;
175 }
176
177 case PA_CONTEXT_TERMINATED:
178 break;
179
180 case PA_CONTEXT_FAILED:
181 default:
182 qCritical( "Connection failure: %s\n", pa_strerror( pa_context_errno( c ) ) );
183 _this->signalConnected( false );
184 }
185}

190void AudioPulseAudio::run()
191{
192 pa_mainloop * mainLoop = pa_mainloop_new();
193 if( !mainLoop )
194 {
195 qCritical( "pa_mainloop_new() failed.\n" );
196 return;
197 }
198 pa_mainloop_api * mainloop_api = pa_mainloop_get_api( mainLoop );
199
200 pa_context *context = pa_context_new( mainloop_api, "lmms" );
201 if ( context == nullptr )
202 {
203 qCritical( "pa_context_new() failed." );
204 return;
205 }
206
207 m_connected = false;
208
209 pa_context_set_state_callback( context, context_state_callback, this );
210 // connect the context
211 pa_context_connect( context, nullptr, (pa_context_flags) 0, nullptr );
212
213 while (!m_connectedSemaphore.tryAcquire()) {
214 pa_mainloop_iterate(mainLoop, 1, nullptr);
215 }
216
217 // run the main loop
218 if( m_connected )
219 {
220 int ret = 0;
221 m_quit = false;
222 while( m_quit == false
223 && pa_mainloop_iterate( mainLoop, 1, &ret ) >= 0 )
224 {
225 }
226
227 pa_stream_disconnect( m_s );
228 pa_stream_unref( m_s );
229 }
230 else
231 {
232 const fpp_t fpp = audioEngine()->framesPerPeriod();
233 auto temp = new SampleFrame[fpp];
234 while( getNextBuffer( temp ) )
235 {
236 }
237 delete[] temp;
238 }
239
240 pa_context_disconnect( context );
241 pa_context_unref( context );
242
243 pa_mainloop_free( mainLoop );
244}

249void AudioPulseAudio::streamWriteCallback( pa_stream *s, size_t length )
250{
251 const fpp_t fpp = audioEngine()->framesPerPeriod();
252 auto temp = new SampleFrame[fpp];
253 auto pcmbuf = (int_sample_t*)pa_xmalloc(fpp * channels() * sizeof(int_sample_t));
254
255 size_t fd = 0;
256 while( fd < length/4 && m_quit == false )
257 {
258 const fpp_t frames = getNextBuffer( temp );
259 if( !frames )
260 {
261 m_quit = true;
262 break;
263 }
264 int bytes = convertToS16(temp, frames, pcmbuf, m_convertEndian);
265 if( bytes > 0 )
266 {
267 pa_stream_write( m_s, pcmbuf, bytes, nullptr, 0,
268 PA_SEEK_RELATIVE );
269 }
270 fd += frames;
271 }
272
273 pa_xfree( pcmbuf );
274 delete[] temp;
275}
