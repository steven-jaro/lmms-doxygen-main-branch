46SampleThumbnail::Thumbnail::Thumbnail(const float* buffer, size_t size, size_t width)
47 : m_peaks(width)
48 , m_samplesPerPeak(std::max(static_cast<double>(size) / width, 1.0))
49{
50 for (auto peakIndex = std::size_t{0}; peakIndex < width; ++peakIndex)
51 {
52 const auto beginSample = buffer + static_cast<size_t>(std::floor(peakIndex * m_samplesPerPeak));
53 const auto endSample = buffer + static_cast<size_t>(std::ceil((peakIndex + 1) * m_samplesPerPeak));
54 const auto [min, max] = std::minmax_element(beginSample, endSample);
55 m_peaks[peakIndex] = Peak{*min, *max};
56 }
57}

59SampleThumbnail::Thumbnail SampleThumbnail::Thumbnail::zoomOut(float factor) const
60{
61 assert(factor >= 1 && "Invalid zoom out factor");
62
63 auto peaks = std::vector<Peak>(m_peaks.size() / factor);
64 for (auto peakIndex = std::size_t{0}; peakIndex < peaks.size(); ++peakIndex)
65 {
66 const auto beginAggregationAt = m_peaks.begin() + static_cast<size_t>(std::floor(peakIndex * factor));
67 const auto endAggregationAt = m_peaks.begin() + static_cast<size_t>(std::ceil((peakIndex + 1) * factor));
68 peaks[peakIndex] = std::accumulate(beginAggregationAt, endAggregationAt, Peak{});
69 }
70
71 return Thumbnail{std::move(peaks), m_samplesPerPeak * factor};
72}

74SampleThumbnail::SampleThumbnail(const Sample& sample)
75 : m_buffer(sample.buffer())
76{
77 auto entry = SampleThumbnailEntry{sample.sampleFile(), QFileInfo{sample.sampleFile()}.lastModified()};
78 if (!entry.filePath.isEmpty())
79 {
80 const auto it = s_sampleThumbnailCacheMap.find(entry);
81 if (it != s_sampleThumbnailCacheMap.end())
82 {
83 m_thumbnailCache = it->second;
84 return;
85 }
86
87 if (s_sampleThumbnailCacheMap.size() == MaxSampleThumbnailCacheSize)
88 {
89 const auto leastUsed = std::min_element(s_sampleThumbnailCacheMap.begin(), s_sampleThumbnailCacheMap.end(),
90 [](const auto& a, const auto& b) { return a.second.use_count() < b.second.use_count(); });
91 s_sampleThumbnailCacheMap.erase(leastUsed->first);
92 }
93
94 s_sampleThumbnailCacheMap[std::move(entry)] = m_thumbnailCache;
95 }
96
97 const auto flatBuffer = m_buffer->data()->data();
98 const auto flatBufferSize = m_buffer->size() * DEFAULT_CHANNELS;
99 m_thumbnailCache->emplace_back(flatBuffer, flatBufferSize, flatBufferSize / AggregationPerZoomStep);
100
101 while (m_thumbnailCache->back().width() >= AggregationPerZoomStep)
102 {
103 auto zoomedOutThumbnail = m_thumbnailCache->back().zoomOut(AggregationPerZoomStep);
104 m_thumbnailCache->emplace_back(std::move(zoomedOutThumbnail));
105 }
106}

108void SampleThumbnail::visualize(VisualizeParameters parameters, QPainter& painter) const
109{
110 const auto& sampleRect = parameters.sampleRect;
111 const auto& viewportRect = parameters.viewportRect.isNull() ? sampleRect : parameters.viewportRect;
112
113 const auto renderRect = sampleRect.intersected(viewportRect);
114 if (renderRect.isNull()) { return; }
115
116 const auto sampleRange = parameters.sampleEnd - parameters.sampleStart;
117 if (sampleRange <= 0.0f || sampleRange > 1.0f) { return; }
118
119 const auto targetThumbnailWidth = static_cast<int>(sampleRect.width() / sampleRange);
120 const auto finerThumbnail = std::find_if(m_thumbnailCache->rbegin(), m_thumbnailCache->rend(),
121 [&](const auto& thumbnail) { return thumbnail.width() >= targetThumbnailWidth; });
122
123 const auto useOriginalBuffer = finerThumbnail == m_thumbnailCache->rend();
124 const auto drawOriginalBuffer = static_cast<size_t>(targetThumbnailWidth) == m_buffer->size();
125
126 painter.save();
127 painter.setRenderHint(QPainter::Antialiasing, true);
128
129 const auto thumbnailBeginForward = std::max<int>(renderRect.x() - sampleRect.x(), parameters.sampleStart * targetThumbnailWidth);
130 const auto thumbnailEndForward = std::max<int>(renderRect.x() + renderRect.width() - sampleRect.x(), parameters.sampleEnd * targetThumbnailWidth);
131 const auto thumbnailBegin = parameters.reversed ? targetThumbnailWidth - thumbnailBeginForward - 1 : thumbnailBeginForward;
132 const auto thumbnailEnd = parameters.reversed ? targetThumbnailWidth - thumbnailEndForward : thumbnailEndForward;
133 const auto advanceThumbnailBy = parameters.reversed ? -1 : 1;
134
135 const auto finerThumbnailWidth = useOriginalBuffer ? m_buffer->size() : finerThumbnail->width();
136 const auto finerThumbnailScaleFactor = static_cast<double>(finerThumbnailWidth) / targetThumbnailWidth;
137 const auto yScale = renderRect.height() / 2 * parameters.amplification;
138
139 for (auto x = renderRect.x(), i = thumbnailBegin; x < renderRect.x() + renderRect.width() && i != thumbnailEnd;
140 ++x, i += advanceThumbnailBy)
141 {
142 if (useOriginalBuffer && drawOriginalBuffer)
143 {
144 const auto value = m_buffer->data()->data()[i];
145 painter.drawPoint(x, renderRect.center().y() - value * yScale);
146 continue;
147 }
148 else
149 {
150 const auto beginIndex = std::clamp<size_t>(std::floor(i * finerThumbnailScaleFactor), 0, finerThumbnail->width() - 1);
151 const auto endIndex = std::clamp<size_t>(std::ceil((i + 1) * finerThumbnailScaleFactor), 0, finerThumbnail->width() - 1);
152
153 auto minPeak = 0.f;
154 auto maxPeak = 0.f;
155
156 if (useOriginalBuffer)
157 {
158 const auto flatBuffer = m_buffer->data()->data();
159 const auto [min, max] = std::minmax_element(flatBuffer + beginIndex, flatBuffer + endIndex);
160 minPeak = *min;
161 maxPeak = *max;
162 }
163 else
164 {
165 const auto beginAggregationAt = finerThumbnail->data() + beginIndex;
166 const auto endAggregationAt = finerThumbnail->data() + endIndex;
167 const auto peak = std::accumulate(beginAggregationAt, endAggregationAt, Thumbnail::Peak{});
168 minPeak = peak.min;
169 maxPeak = peak.max;
170 }
171
172 const auto yMin = renderRect.center().y() - minPeak * yScale;
173 const auto yMax = renderRect.center().y() - maxPeak * yScale;
174 painter.drawLine(x, yMin, x, yMax);
175 }
176 }
177
178 painter.restore();
179}
