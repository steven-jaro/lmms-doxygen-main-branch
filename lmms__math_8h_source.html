152inline float logToLinearScale(float min, float max, float value)
153{
154 using namespace std::numbers;
155 if (min < 0)
156 {
157 const float mmax = std::max(std::abs(min), std::abs(max));
158 const float val = value * (max - min) + min;
159 float result = signedPowf(val / mmax, e_v<float>) * mmax;
160 return std::isnan(result) ? 0 : result;
161 }
162 float result = std::pow(value, e_v<float>) * (max - min) + min;
163 return std::isnan(result) ? 0 : result;
164}

168inline float linearToLogScale(float min, float max, float value)
169{
170 constexpr auto inv_e = static_cast<float>(1.0 / std::numbers::e);
171 const float valueLimited = std::clamp(value, min, max);
172 const float val = (valueLimited - min) / (max - min);
173 if (min < 0)
174 {
175 const float mmax = std::max(std::abs(min), std::abs(max));
176 float result = signedPowf(valueLimited / mmax, inv_e) * mmax;
177 return std::isnan(result) ? 0 : result;
178 }
179 float result = std::pow(val, inv_e) * (max - min) + min;
180 return std::isnan(result) ? 0 : result;
181}

240// @brief Calculate number of digits which LcdSpinBox would show for a given number
241inline int numDigitsAsInt(float f)
242{
243 // use rounding:
244 // LcdSpinBox sometimes uses std::round(), sometimes cast rounding
245 // we use rounding to be on the "safe side"
246 int asInt = static_cast<int>(std::round(f));
247 int digits = 1; // always at least 1
248 if(asInt < 0)
249 {
250 ++digits;
251 asInt = -asInt;
252 }
253 // "asInt" is positive from now
254 int power = 1;
255 for (int i = 1; i < 10; ++i)
256 {
257 power *= 10;
258 if (asInt >= power) { ++digits; } // 2 digits for >=10, 3 for >=100
259 else { break; }
260 }
261 return digits;
262}

267public:
268 LinearMap(T x1, T y1, T x2, T y2)
269 {
270 T const dx = x2 - x1;
271 assert (dx != T(0));
272
273 m_a = (y2 - y1) / dx;
274 m_b = y1 - m_a * x1;
275 }
276
277 T map(T x) const
278 {
279 return m_a * x + m_b;
280 }

282private:

288// exp approximation for SSE2: https://stackoverflow.com/a/47025627/5759631
289// Maximum relative error of 1.72863156e-3 on [-87.33654, 88.72283]
290inline __m128 fastExp(__m128 x)
291{
292 __m128 f, p, r;
293 __m128i t, j;
294 const __m128 a = _mm_set1_ps (12102203.0f); /* (1 << 23) / log(2) */
295 const __m128i m = _mm_set1_epi32 (0xff800000); /* mask for integer bits */
296 const __m128 ttm23 = _mm_set1_ps (1.1920929e-7f); /* exp2(-23) */
297 const __m128 c0 = _mm_set1_ps (0.3371894346f);
298 const __m128 c1 = _mm_set1_ps (0.657636276f);
299 const __m128 c2 = _mm_set1_ps (1.00172476f);
300
301 t = _mm_cvtps_epi32 (_mm_mul_ps (a, x));
302 j = _mm_and_si128 (t, m); /* j = (int)(floor (x/log(2))) << 23 */
303 t = _mm_sub_epi32 (t, j);
304 f = _mm_mul_ps (ttm23, _mm_cvtepi32_ps (t)); /* f = (x/log(2)) - floor (x/log(2)) */
305 p = c0; /* c0 */
306 p = _mm_mul_ps (p, f); /* c0 * f */
307 p = _mm_add_ps (p, c1); /* c0 * f + c1 */
308 p = _mm_mul_ps (p, f); /* (c0 * f + c1) * f */
309 p = _mm_add_ps (p, c2); /* p = (c0 * f + c1) * f + c2 ~= 2^f */
310 r = _mm_castsi128_ps (_mm_add_epi32 (j, _mm_castps_si128 (p))); /* r = p * 2^i*/
311 return r;
312}

314// Lost Robot's SSE2 adaptation of Kari's vectorized log approximation: https://stackoverflow.com/a/65537754/5759631
315// Maximum relative error of 7.922410e-4 on [1.0279774e-38f, 3.4028235e+38f]
316inline __m128 fastLog(__m128 a)
317{
318 __m128i aInt = _mm_castps_si128(a);
319 __m128i e = _mm_sub_epi32(aInt, _mm_set1_epi32(0x3f2aaaab));
320 e = _mm_and_si128(e, _mm_set1_epi32(0xff800000));
321 __m128i subtr = _mm_sub_epi32(aInt, e);
322 __m128 m = _mm_castsi128_ps(subtr);
323 __m128 i = _mm_mul_ps(_mm_cvtepi32_ps(e), _mm_set1_ps(1.19209290e-7f));
324 __m128 f = _mm_sub_ps(m, _mm_set1_ps(1.0f));
325 __m128 s = _mm_mul_ps(f, f);
326 __m128 r = _mm_add_ps(_mm_mul_ps(_mm_set1_ps(0.230836749f), f), _mm_set1_ps(-0.279208571f));
327 __m128 t = _mm_add_ps(_mm_mul_ps(_mm_set1_ps(0.331826031f), f), _mm_set1_ps(-0.498910338f));
328 r = _mm_add_ps(_mm_mul_ps(r, s), t);
329 r = _mm_add_ps(_mm_mul_ps(r, s), f);
330 r = _mm_add_ps(_mm_mul_ps(i, _mm_set1_ps(0.693147182f)), r);
331 return r;
332}

339inline __m128 sse2Floor(__m128 x)
340{
341 __m128 t = _mm_cvtepi32_ps(_mm_cvttps_epi32(x)); // trunc toward 0
342 __m128 needs_correction = _mm_cmplt_ps(x, t); // checks if x < trunc
343 return _mm_sub_ps(t, _mm_and_ps(needs_correction, _mm_set1_ps(1.0f)));
344}

346inline __m128 sse2Round(__m128 x)
347{
348 __m128 sign_mask = _mm_cmplt_ps(x, _mm_setzero_ps());// checks if x < 0
349 __m128 bias_pos = _mm_set1_ps(0.5f);
350 __m128 bias_neg = _mm_set1_ps(-0.5f);
351 __m128 bias = _mm_or_ps(_mm_and_ps(sign_mask, bias_neg), _mm_andnot_ps(sign_mask, bias_pos));
352 __m128 y = _mm_add_ps(x, bias);
353 return _mm_cvtepi32_ps(_mm_cvttps_epi32(y));
354}
