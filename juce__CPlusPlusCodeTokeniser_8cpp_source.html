39{
40 struct Type
41 {

43 uint32 colour;
44 };
45
46 const Type types[] =
47 {
48 { "Error", 0xffcc0000 },
49 { "Comment", 0xff00aa00 },

51 { "Operator", 0xff225500 },
52 { "Identifier", 0xff000000 },

57 { "Punctuation", 0xff004400 },
58 { "Preprocessor Text", 0xff660000 }
59 };
60
61 CodeEditorComponent::ColourScheme cs;
