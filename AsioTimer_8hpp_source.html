33// This implementation is based on the boost::asio::system_timer concept.
34// Since boost::system_timer doesn't support move semantics, we create a wrapper
35// with a unique_ptr to get a movable type. It also handles an inconvenient
36// aspect of asio timers, which is that you must explicitly guard against the
37// handler firing after cancellation. We handle this by use of the SafeAsyncHandler
38// utility. AsioTimer therefore guarantees that a handler will not be called after
39// the destruction of the timer, or after the timer has been canceled.
40
41class AsioTimer
42{
43public:
44 using ErrorCode = ::asio::error_code;
45 using TimePoint = std::chrono::system_clock::time_point;
46
47 AsioTimer(::asio::io_service& io)
48 : mpTimer(new ::asio::system_timer(io))
49 , mpAsyncHandler(std::make_shared<AsyncHandler>())
50 {
51 }
52
53 ~AsioTimer()
54 {
55 // The timer may not be valid anymore if this instance was moved from
56 if (mpTimer != nullptr)
57 {
58 // Ignore errors during cancellation
59 cancel();
60 }
61 }
62
63 AsioTimer(const AsioTimer&) = delete;
64 AsioTimer& operator=(const AsioTimer&) = delete;
65
66 // Enable move construction but not move assignment. Move assignment
67 // would get weird - would have to handle outstanding handlers
68 AsioTimer(AsioTimer&& rhs)
69 : mpTimer(std::move(rhs.mpTimer))
70 , mpAsyncHandler(std::move(rhs.mpAsyncHandler))
71 {
72 }
73
74 void expires_at(std::chrono::system_clock::time_point tp)
75 {
76 mpTimer->expires_at(std::move(tp));
77 }
78
79 template <typename T>
80 void expires_from_now(T duration)
81 {
82 mpTimer->expires_from_now(std::move(duration));
83 }
84
85 ErrorCode cancel()
86 {
87 ErrorCode ec;
88 mpTimer->cancel(ec);
89 mpAsyncHandler->mpHandler = nullptr;
90 return ec;
91 }
92
93 template <typename Handler>
94 void async_wait(Handler handler)
95 {
96 *mpAsyncHandler = std::move(handler);
97 mpTimer->async_wait(util::makeAsyncSafe(mpAsyncHandler));
98 }
99
100 TimePoint now() const
101 {
102 return std::chrono::system_clock::now();
103 }
104
105private:
106 struct AsyncHandler
107 {
108 template <typename Handler>
109 AsyncHandler& operator=(Handler handler)
110 {
111 mpHandler = [handler](ErrorCode ec) { handler(std::move(ec)); };
112 return *this;
113 }
114
115 void operator()(ErrorCode ec)
116 {
117 if (mpHandler)
118 {
119 mpHandler(std::move(ec));
120 }
121 }
122
123 std::function<void(const ErrorCode)> mpHandler;
124 };
125
126 std::unique_ptr<::asio::system_timer> mpTimer;
127 std::shared_ptr<AsyncHandler> mpAsyncHandler;
128};
