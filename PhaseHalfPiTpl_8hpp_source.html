84Description:

95template <int NC, typename DT, int NCHN>
96void PhaseHalfPiTpl <NC, DT, NCHN>::process_sample (DataType &out_0, DataType &out_1, DataType input) noexcept
97{
98 out_0 = input; // Even coefs
99 out_1 = _prev; // Odd coefs

101 StageProcTpl <NBR_COEFS, DataType>::process_sample_neg (

106 _phase = 1 - _phase;
107}

111/*
112==============================================================================
113Name: process_block
114Description:
115 From a block of samples, generates two blocks of samples, with a pi/2
116 phase shift between these signals.
117 Input and output blocks may overlap, see assert() for details.
118Input parameters:
119 - in_ptr: Input array, containing nbr_spl samples.
120 - nbr_spl: Number of input samples to process, > 0
121Output parameters:
122 - out_0_ptr: Output sample array (ahead), capacity: nbr_spl samples.
123 - out_1_ptr: Output sample array (late), capacity: nbr_spl samples.
124Throws: Nothing
125==============================================================================
126*/
127
128template <int NC, typename DT, int NCHN>
129void PhaseHalfPiTpl <NC, DT, NCHN>::process_block (DataType out_0_ptr [], DataType out_1_ptr [], const DataType in_ptr [], long nbr_spl) noexcept
130{
131 assert (out_0_ptr != nullptr);
132 assert (out_1_ptr != nullptr);
133 assert (in_ptr != nullptr);
134 assert (out_0_ptr <= in_ptr || out_0_ptr >= in_ptr + nbr_spl);
135 assert (out_1_ptr <= in_ptr || out_1_ptr >= in_ptr + nbr_spl);
136 assert (out_0_ptr + nbr_spl <= out_1_ptr || out_1_ptr + nbr_spl <= out_0_ptr);
137 assert (nbr_spl > 0);
138
139 long pos = 0;
140 if (_phase == 1)
141 {
142 process_sample (out_0_ptr [0], out_1_ptr [0], in_ptr [0]);
143 ++ pos;
144 }
145
146 const long end = ((nbr_spl - pos) & -_nbr_phases) + pos;
147 while (pos < end)
148 {
149 const DataType input_0 = in_ptr [pos];
150 out_0_ptr [pos] = input_0;
151 out_1_ptr [pos] = _prev;
152 StageProcTpl <NBR_COEFS, DataType>::process_sample_neg (
153 NBR_COEFS,
154 out_0_ptr [pos],
155 out_1_ptr [pos],
156 _bifilter [0].data ()
157 );
158
159 const DataType input_1 = in_ptr [pos + 1];
160 out_0_ptr [pos + 1] = input_1;
161 out_1_ptr [pos + 1] = input_0; // _prev
162 StageProcTpl <NBR_COEFS, DataType>::process_sample_neg (
163 NBR_COEFS,
164 out_0_ptr [pos + 1],
165 out_1_ptr [pos + 1],
166 _bifilter [1].data ()
167 );
168 _prev = input_1;
169
170 pos += 2;
171 }
172
173 if (pos < nbr_spl)
174 {
175 assert (pos + 1 == nbr_spl);
176 process_sample (out_0_ptr [pos], out_1_ptr [pos], in_ptr [pos]);
177 }
178}

200

202 _phase = 0;
203}
