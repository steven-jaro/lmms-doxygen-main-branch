42 float getSinePhase(float x) {
43 return ((-std::cos(x)+1)/2);
44 }
45 float getSawPhase(float x) {
46 return (-((2/M_PI * std::atan(1/std::tan(x/2)))-1)/2);
47 }
48 float getRevSawPhase(float x) {
49 return (((2/M_PI * std::atan(1/std::tan(x/2)))+1)/2);
50 }
51 float getSquarePhase(float x) {
52 return (std::round((std::sin(x)+1)/2));
53 }
54
55 //saw, sqr, sin, revSaw
56 float getBlendedPhase(float x, float wave)
57 {
58 //wave = 2;
59 if (wave>=1 && wave<2) {
60 /* saw vs sqr */
61 waveBlend = wave-1;
62 return (getSawPhase(x)*(1-waveBlend) + getSquarePhase(x)*waveBlend);
63 } else if (wave>=2 && wave<3) {
64 /* sqr vs sin */
65 waveBlend = wave-2;
66 return (getSquarePhase(x)*(1-waveBlend) + getSinePhase(x)*waveBlend);
67 } else if (wave>=3 && wave<=4) {
68 /* sin vs revSaw */
69 waveBlend = wave-3;
70 return (getSinePhase(x)*(1-waveBlend) + getRevSawPhase(x)*waveBlend);
71 } else {
72 return 0.0f;
73 }
74 }
