104void AutomationClipView::disconnectObject( QAction * _a )
105{
106 JournallingObject * j = Engine::projectJournal()->
107 journallingObject( _a->data().toInt() );
108 if( j && dynamic_cast<AutomatableModel *>( j ) )
109 {
110 float oldMin = m_clip->getMin();
111 float oldMax = m_clip->getMax();
112
113 m_clip->m_objects.erase( std::find( m_clip->m_objects.begin(),
114 m_clip->m_objects.end(),
115 dynamic_cast<AutomatableModel *>( j ) ) );
116 update();
117
118 //If automation editor is opened, update its display after disconnection
119 if( getGUI()->automationEditor() )
120 {
121 getGUI()->automationEditor()->m_editor->updateAfterClipChange();
122 }
123
124 //if there is no more connection connected to the AutomationClip
125 if( m_clip->m_objects.size() == 0 )
126 {
127 //scale the points to fit the new min. and max. value
128 this->scaleTimemapToFit( oldMin, oldMax );
129 }
130 }
131}

176 _cm->addAction( embed::getIconPixmap( "edit_rename" ),

178 this, SLOT(changeName()));
179 _cm->addAction( embed::getIconPixmap( "record" ),
180 tr( "Set/clear record" ),
181 this, SLOT(toggleRecording()));
182 _cm->addAction( embed::getIconPixmap( "flip_y" ),
183 tr( "Flip Vertically (Visible)" ),
184 this, SLOT(flipY()));
185 _cm->addAction( embed::getIconPixmap( "flip_x" ),
186 tr( "Flip Horizontally (Visible)" ),
187 this, SLOT(flipX()));
188
189 if (!m_clip->m_objects.empty())
190 {
191 _cm->addSeparator();
192 auto m = new QMenu(tr("%1 Connections").arg(m_clip->m_objects.size()), _cm);
193 for (const auto& object : m_clip->m_objects)
194 {
195 if (object)
196 {
197 a = new QAction(tr("Disconnect \"%1\"").arg(object->fullDisplayName()), m);
198 a->setData(object->id());
199 m->addAction( a );
200 }
201 }
202 connect( m, SIGNAL(triggered(QAction*)),
203 this, SLOT(disconnectObject(QAction*)));
204 _cm->addMenu( m );
205 }

226void AutomationClipView::paintEvent( QPaintEvent * )
227{
228 QPainter painter( this );
229
230 if( !needsUpdate() )
231 {
232 painter.drawPixmap( 0, 0, m_paintPixmap );
233 return;
234 }
235
236 setNeedsUpdate( false );
237
238 if (m_paintPixmap.isNull() || m_paintPixmap.size() != size())
239 {
240 m_paintPixmap = QPixmap(size());
241 }
242
243 QPainter p( &m_paintPixmap );
244
245 QLinearGradient lingrad( 0, 0, 0, height() );
246 QColor c = getColorForDisplay( painter.background().color() );
247 bool muted = m_clip->getTrack()->isMuted() || m_clip->isMuted();
248 bool current = getGUI()->automationEditor()->currentClip() == m_clip;
249
250 lingrad.setColorAt( 1, c.darker( 300 ) );
251 lingrad.setColorAt( 0, c );
252
253 // paint a black rectangle under the clip to prevent glitches with transparent backgrounds
254 p.fillRect( rect(), QColor( 0, 0, 0 ) );
255
256 if( gradient() )
257 {
258 p.fillRect( rect(), lingrad );
259 }
260 else
261 {
262 p.fillRect( rect(), c );
263 }
264
265 // pixels per bar
266 const float ppb = fixedClips() ?
267 ( parentWidget()->width() - 2 * BORDER_WIDTH )
268 / (float) m_clip->timeMapLength().getBar() :
269 pixelsPerBar();
270
271 const auto min = m_clip->firstObject()->minValue<float>();
272 const auto max = m_clip->firstObject()->maxValue<float>();
273
274 const float y_scale = max - min;
275 const float h = ( height() - 2 * BORDER_WIDTH ) / y_scale;
276 const float ppTick = ppb / TimePos::ticksPerBar();
277 const int offset = m_clip->startTimeOffset() * ppTick;
278
279 p.translate( 0.0f, max * height() / y_scale - BORDER_WIDTH );
280 p.scale( 1.0f, -h );
281
282 QLinearGradient lin2grad( 0, min, 0, max );
283 QColor col;
284
285 col = !muted ? painter.pen().brush().color() : mutedColor();
286
287 lin2grad.setColorAt( 1, col.lighter( 150 ) );
288 lin2grad.setColorAt( 0.5, col );
289 lin2grad.setColorAt( 0, col.darker( 150 ) );
290
291 p.setRenderHints( QPainter::Antialiasing, true );
292 for( AutomationClip::timeMap::const_iterator it =
293 m_clip->getTimeMap().begin();
294 it != m_clip->getTimeMap().end(); ++it )
295 {
296 if( it+1 == m_clip->getTimeMap().end() )
297 {
298 const float x1 = POS(it) * ppTick + offset;
299 const auto x2 = (float)(width() - BORDER_WIDTH);
300 if( x1 > ( width() - BORDER_WIDTH ) ) break;
301 // We are drawing the space after the last node, so we use the outValue
302 if( gradient() )
303 {
304 p.fillRect(QRectF(x1, 0.0f, x2 - x1, OUTVAL(it)), lin2grad);
305 }
306 else
307 {
308 p.fillRect(QRectF(x1, 0.0f, x2 - x1, OUTVAL(it)), col);
309 }
310 break;
311 }
312
313 float *values = m_clip->valuesAfter(POS(it));
314
315 // We are creating a path to draw a polygon representing the values between two
316 // nodes. When we have two nodes with discrete progression, we will basically have
317 // a rectangle with the outValue of the first node (that's why nextValue will match
318 // the outValue of the current node). When we have nodes with linear or cubic progression
319 // the value of the end of the shape between the two nodes will be the inValue of
320 // the next node.
321 float nextValue = m_clip->progressionType() == AutomationClip::ProgressionType::Discrete
322 ? OUTVAL(it)
323 : INVAL(it + 1);
324
325 QPainterPath path;
326 QPointF origin = QPointF(POS(it) * ppTick + offset, 0.0f);
327 path.moveTo(origin);
328 path.moveTo(QPointF(POS(it) * ppTick + offset, values[0]));
329 for (int i = POS(it) + 1; i < POS(it + 1); i++)
330 {
331 float x = i * ppTick + offset;
332 if(x > (width() - BORDER_WIDTH)) break;
333 float value = values[i - POS(it)];
334 path.lineTo(QPointF(x, value));
335 }
336 path.lineTo((POS(it + 1)) * ppTick + offset, nextValue);
337 path.lineTo((POS(it + 1)) * ppTick + offset, 0.0f);
338 path.lineTo(origin);
339
340 if( gradient() )
341 {
342 p.fillPath( path, lin2grad );
343 }
344 else
345 {
346 p.fillPath( path, col );
347 }
348 delete [] values;
349 }
350
351 p.setRenderHints( QPainter::Antialiasing, false );
352 p.resetTransform();
353
354 // bar lines
355 const int lineSize = 3;
356 p.setPen( c.darker( 300 ) );
357
358 for (bar_t b = 1; b < width() - BORDER_WIDTH; ++b)
359 {
360 const int bx = BORDER_WIDTH + static_cast<int>(ppb * b) - 2;
361
362 //top line
363 p.drawLine(bx + offset, BORDER_WIDTH, bx + offset, BORDER_WIDTH + lineSize);
364
365 //bottom line
366 p.drawLine(bx + offset, rect().bottom() - (lineSize + BORDER_WIDTH), bx + offset, rect().bottom() - BORDER_WIDTH);
367 }
368
369 // recording icon for when recording automation
370 if( m_clip->isRecording() )
371 {
372 static auto s_clipRec = embed::getIconPixmap("clip_rec");
373 p.drawPixmap(1, rect().bottom() - s_clipRec.height(), s_clipRec);
374 }
375
376 // clip name
377 paintTextLabel(m_clip->name(), p);
378
379 // inner border
380 p.setPen( c.lighter( current ? 160 : 130 ) );
381 p.drawRect( 1, 1, rect().right() - BORDER_WIDTH,
382 rect().bottom() - BORDER_WIDTH );
383
384 // outer border
385 p.setPen( current? c.lighter( 130 ) : c.darker( 300 ) );
386 p.drawRect( 0, 0, rect().right(), rect().bottom() );
387
388 // draw the 'muted' pixmap only if the clip was manually muted
389 if( m_clip->isMuted() )
390 {
391 const int spacing = BORDER_WIDTH;
392 const int size = 14;
393 p.drawPixmap( spacing, height() - ( size + spacing ),
394 embed::getIconPixmap( "muted", size, size ) );
395 }
396
397 if (m_marker)
398 {
399 p.setPen(markerColor());
400 p.drawLine(m_markerPos, rect().bottom(), m_markerPos, rect().top());
401 }
402
403 p.end();
404
405 painter.drawPixmap( 0, 0, m_paintPixmap );
406}

423void AutomationClipView::dropEvent( QDropEvent * _de )
424{
425 QString type = StringPairDrag::decodeKey( _de );
426 QString val = StringPairDrag::decodeValue( _de );
427 if( type == "automatable_model" )
428 {
429 auto mod = dynamic_cast<AutomatableModel*>(Engine::projectJournal()->journallingObject(val.toInt()));
430 if( mod != nullptr )
431 {
432 bool added = m_clip->addObject( mod );
433 if ( !added )
434 {
435 TextFloat::displayMessage( mod->displayName(),
436 tr( "Model is already connected "
437 "to this clip." ),
438 embed::getIconPixmap( "automation" ),
439 2000 );
440 }
441 }
442 update();
443
444 if( getGUI()->automationEditor() &&
445 getGUI()->automationEditor()->currentClip() == m_clip )
446 {
447 getGUI()->automationEditor()->setCurrentClip( m_clip );
448 }
449 }
450 else
451 {
452 ClipView::dropEvent( _de );
453 }
454}

459/**
460 * @brief Preserves the auto points over different scale
461 */
462void AutomationClipView::scaleTimemapToFit( float oldMin, float oldMax )
463{
464 float newMin = m_clip->getMin();
465 float newMax = m_clip->getMax();
466
467 if( oldMin == newMin && oldMax == newMax )
468 {
469 return;
470 }
471
472 // TODO: Currently when rescaling the timeMap values to fit the new range of values (newMin and newMax)
473 // only the inValue is being considered and the outValue is being reset to the inValue (so discrete jumps
474 // are discarded). Possibly later we will want discrete jumps to be maintained so we will need to upgrade
475 // the logic to account for them.
476 for( AutomationClip::timeMap::iterator it = m_clip->m_timeMap.begin();
477 it != m_clip->m_timeMap.end(); ++it )
478 {
479 // If the values are out of the previous range, fix them so they are
480 // between oldMin and oldMax.
481 if (INVAL(it) < oldMin)
482 {
483 it.value().setInValue(oldMin);
484 }
485 else if (INVAL(it) > oldMax)
486 {
487 it.value().setInValue(oldMax);
488 }
489 // Calculate what the value would be proportionally in the new range
490 it.value().setInValue((INVAL(it) - oldMin) * (newMax - newMin) / (oldMax - oldMin) + newMin);
491 // Read earlier TODO comment: For now I'm discarding the discrete jumps during the rescaling
492 it.value().setOutValue(INVAL(it));
493 }
494
495 m_clip->generateTangents();
496}
