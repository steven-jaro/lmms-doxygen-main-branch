135QMenu * SampleTrackView::createMixerMenu(QString title, QString newMixerLabel)
136{
137 int channelIndex = model()->mixerChannelModel()->value();
138
139 MixerChannel *mixerChannel = Engine::mixer()->mixerChannel(channelIndex);
140
141 // If title allows interpolation, pass channel index and name
142 if (title.contains("%2"))
143 {
144 title = title.arg(channelIndex).arg(mixerChannel->m_name);
145 }
146
147 auto mixerMenu = new QMenu(title);
148
149 mixerMenu->addAction(newMixerLabel, this, SLOT(createMixerLine()));
150 mixerMenu->addSeparator();
151
152 for (int i = 0; i < Engine::mixer()->numChannels(); ++i)
153 {
154 MixerChannel * currentChannel = Engine::mixer()->mixerChannel(i);
155
156 if (currentChannel != mixerChannel)
157 {
158 const auto index = currentChannel->index();
159 QString label = tr("%1: %2").arg(index).arg(currentChannel->m_name);
160 mixerMenu->addAction(label, [this, index](){
161 assignMixerLine(index);
162 });
163 }
164 }
165
166 return mixerMenu;
167}

199void SampleTrackView::dropEvent(QDropEvent *de)
200{
201 QString type = StringPairDrag::decodeKey(de);
202 QString value = StringPairDrag::decodeValue(de);
203
204 if (type == "samplefile")
205 {
206 int trackHeadWidth = ConfigManager::inst()->value("ui", "compacttrackbuttons").toInt()==1
207 ? DEFAULT_SETTINGS_WIDGET_WIDTH_COMPACT + TRACK_OP_WIDTH_COMPACT
208 : DEFAULT_SETTINGS_WIDGET_WIDTH + TRACK_OP_WIDTH;
209
210 int xPos = de->pos().x() < trackHeadWidth
211 ? trackHeadWidth
212 : de->pos().x();
213
214 const float snapSize = getGUI()->songEditor()->m_editor->getSnapSize();
215 TimePos clipPos = trackContainerView()->fixedClips()
216 ? TimePos(0)
217 : TimePos(((xPos - trackHeadWidth) / trackContainerView()->pixelsPerBar()
218 * TimePos::ticksPerBar()) + trackContainerView()->currentPosition()
219 ).quantize(snapSize, true);
220
221 auto sClip = static_cast<SampleClip*>(getTrack()->createClip(clipPos));
222 if (sClip) { sClip->setSampleFile(value); }
223 }
224}

229/*! \brief Create and assign a new mixer Channel for this track */
230void SampleTrackView::createMixerLine()
231{
232 int channelIndex = getGUI()->mixerView()->addNewChannel();
233 auto channel = Engine::mixer()->mixerChannel(channelIndex);
234
235 channel->m_name = getTrack()->name();
236 channel->setColor(getTrack()->color());
237
238 assignMixerLine(channelIndex);
239}
