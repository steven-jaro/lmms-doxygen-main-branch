62void LfoGraph::paintEvent(QPaintEvent*)
63{
64 QPainter p{this};
65 p.setRenderHint(QPainter::Antialiasing);
66
67 // Draw the graph background
68 p.drawPixmap(rect(), m_lfoGraph);
69
70 const auto* params = castModel<EnvelopeAndLfoParameters>();
71 if (!params) { return; }
72
73 const float amount = params->getLfoAmountModel().value();
74 const float lfoSpeed = params->getLfoSpeedModel().value();
75 const f_cnt_t predelayFrames = params->getLfoPredelayFrames();
76 const f_cnt_t attackFrames = params->getLfoAttackFrames();
77 const f_cnt_t oscillationFrames = params->getLfoOscillationFrames();
78 const bool x100 = params->getX100Model().value();
79 const int lfoWaveModel = params->getLfoWaveModel().value();
80 const auto * userWave = params->getLfoUserWave().get();
81
82 const int margin = 3;
83 const int lfoGraphWidth = width() - margin; // subtract margin
84 const int lfoGraphHeight = height() - 2 * margin; // subtract margin
85 int graphBaseX = 2;
86 int graphBaseY = margin + lfoGraphHeight / 2;
87
88 const float framesForGraph =
89 SECS_PER_LFO_OSCILLATION * Engine::audioEngine()->baseSampleRate() / 10;
90
91 float oscFrames = oscillationFrames * (x100 ? 100. : 1.);
92
93 QPolygonF polyLine;
94 polyLine << QPointF(graphBaseX - 1, graphBaseY);
95
96 // Collect the points for the poly line by sampling the LFO according to its shape
97 for (int x = 0; x <= lfoGraphWidth; ++x)
98 {
99 float value = 0.0;
100 float currentSample = x * framesForGraph / lfoGraphWidth;
101 const auto sampleAsFrameCount = static_cast<f_cnt_t>(currentSample);
102 if (sampleAsFrameCount > predelayFrames)
103 {
104 currentSample -= predelayFrames;
105 const float phase = currentSample / oscFrames;
106
107 const auto lfoShape = static_cast<EnvelopeAndLfoParameters::LfoShape>(lfoWaveModel);
108 switch (lfoShape)
109 {
110 case EnvelopeAndLfoParameters::LfoShape::SineWave:
111 default:
112 value = Oscillator::sinSample(phase);
113 break;
114 case EnvelopeAndLfoParameters::LfoShape::TriangleWave:
115 value = Oscillator::triangleSample(phase);
116 break;
117 case EnvelopeAndLfoParameters::LfoShape::SawWave:
118 value = Oscillator::sawSample(phase);
119 break;
120 case EnvelopeAndLfoParameters::LfoShape::SquareWave:
121 value = Oscillator::squareSample(phase);
122 break;
123 case EnvelopeAndLfoParameters::LfoShape::RandomWave:
124 if (x % (int)(900 * lfoSpeed + 1) == 0)
125 {
126 m_randomGraph = Oscillator::noiseSample(0.0);
127 }
128 value = m_randomGraph;
129 break;
130 case EnvelopeAndLfoParameters::LfoShape::UserDefinedWave:
131 value = Oscillator::userWaveSample(userWave, phase);
132 break;
133 }
134
135 if (sampleAsFrameCount <= attackFrames)
136 {
137 value *= currentSample / attackFrames;
138 }
139 }
140
141 const float currentY = -lfoGraphHeight / 2.0f * value;
142
143 polyLine << QPointF(graphBaseX + x, graphBaseY + currentY);
144 }
145
146 // Compute the color of the lines based on the amount of the LFO
147 const float absAmount = std::abs(amount);
148 const QColor lineColor{ColorHelper::interpolateInRgb(m_noAmountColor, m_fullAmountColor, absAmount)};
149
150 p.setPen(QPen(lineColor, 1.5));
151
152 p.drawPolyline(polyLine);
153
154 drawInfoText(*params);
155}

157void LfoGraph::drawInfoText(const EnvelopeAndLfoParameters& params)
158{
159 QPainter p(this);
160
161 const float lfoSpeed = params.getLfoSpeedModel().value();
162 const bool x100 = params.getX100Model().value();
163
164 const float hertz = 1. / (SECS_PER_LFO_OSCILLATION * lfoSpeed) * (x100 ? 100. : 1.);
165 const auto infoText = tr("%1 Hz").arg(hertz, 0, 'f', 3);
166
167 // First configure the font so that we get correct results for the font metrics used below
168 QFont f = p.font();
169 p.setFont(adjustedToPixelSize(f, height() * 0.2));
170
171 // This is the position where the text and its rectangle will be rendered
172 const QPoint textPosition(4, height() - 6);
173
174 // Draw a slightly transparent black rectangle underneath the text to keep it legible
175 const QFontMetrics fontMetrics(f);
176 // This gives the bounding rectangle if the text was rendered at the origin ...
177 const auto boundingRect = fontMetrics.boundingRect(infoText);
178 // ... so we translate it to the actual position where the text will be rendered.
179 p.fillRect(boundingRect.translated(textPosition), QColor{0, 0, 0, 192});
180
181 // Now draw the actual info text
182 p.setPen(QColor(201, 201, 225));
183 p.drawText(textPosition, infoText);
184}
