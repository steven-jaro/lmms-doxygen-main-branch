85void TempoSyncKnob::modelChanged()
86{
87 if( model() == nullptr )
88 {
89 qWarning( "no TempoSyncKnobModel has been set!" );
90 }
91 if( m_custom != nullptr )
92 {
93 m_custom->setModel( &model()->m_custom );
94 }
95 connect(model(), &TempoSyncKnobModel::syncModeChanged, this, &TempoSyncKnob::updateDescAndIcon);
96 connect( this, SIGNAL(sliderMoved(float)),
97 model(), SLOT(disableSync()));
98 updateDescAndIcon();
99}

104void TempoSyncKnob::contextMenuEvent( QContextMenuEvent * )
105{
106 mouseReleaseEvent( nullptr );
107
108 CaptionMenu contextMenu( model()->displayName(), this );
109 addDefaultActions( &contextMenu );
110 contextMenu.addAction(QPixmap(),
111 model()->isScaleLogarithmic() ? tr("Set linear") : tr("Set logarithmic"),
112 this, SLOT(toggleScale()));
113 contextMenu.addSeparator();
114
115 float limit = 60000.0f / ( Engine::getSong()->getTempo() *
116 model()->m_scale );
117
118 QMenu * syncMenu = contextMenu.addMenu( m_tempoSyncIcon,
119 m_tempoSyncDescription );
120 if( limit / 8.0f <= model()->maxValue() )
121 {
122
123 connect( syncMenu, SIGNAL(triggered(QAction*)),
124 model(), SLOT(setTempoSync(QAction*)));
125 syncMenu->addAction( embed::getIconPixmap( "note_none" ),
126 tr( "No Sync" ) )->setData( (int) TempoSyncKnobModel::SyncMode::None );
127 if( limit / 0.125f <= model()->maxValue() )
128 {
129 syncMenu->addAction( embed::getIconPixmap( "note_double_whole" ),
130 tr( "Eight beats" ) )->setData(
131 (int) TempoSyncKnobModel::SyncMode::DoubleWholeNote );
132 }
133 if( limit / 0.25f <= model()->maxValue() )
134 {
135 syncMenu->addAction( embed::getIconPixmap( "note_whole" ),
136 tr( "Whole note" ) )->setData(
137 (int) TempoSyncKnobModel::SyncMode::WholeNote );
138 }
139 if( limit / 0.5f <= model()->maxValue() )
140 {
141 syncMenu->addAction( embed::getIconPixmap( "note_half" ),
142 tr( "Half note" ) )->setData(
143 (int) TempoSyncKnobModel::SyncMode::HalfNote );
144 }
145 if( limit <= model()->maxValue() )
146 {
147 syncMenu->addAction( embed::getIconPixmap( "note_quarter" ),
148 tr( "Quarter note" ) )->setData(
149 (int) TempoSyncKnobModel::SyncMode::QuarterNote );
150 }
151 if( limit / 2.0f <= model()->maxValue() )
152 {
153 syncMenu->addAction( embed::getIconPixmap( "note_eighth" ),
154 tr( "8th note" ) )->setData(
155 (int) TempoSyncKnobModel::SyncMode::EighthNote );
156 }
157 if( limit / 4.0f <= model()->maxValue() )
158 {
159 syncMenu->addAction( embed::getIconPixmap( "note_sixteenth" ),
160 tr( "16th note" ) )->setData(
161 (int) TempoSyncKnobModel::SyncMode::SixteenthNote );
162 }
163 syncMenu->addAction( embed::getIconPixmap( "note_thirtysecond" ),
164 tr( "32nd note" ) )->setData(
165 (int) TempoSyncKnobModel::SyncMode::ThirtysecondNote );
166 syncMenu->addAction( embed::getIconPixmap( "dont_know" ),
167 tr( "Custom..." ),
168 this, SLOT(showCustom())
169 )->setData(
170 (int) TempoSyncKnobModel::SyncMode::Custom );
171 contextMenu.addSeparator();
172
173 }
174 contextMenu.exec( QCursor::pos() );
175
176 delete syncMenu;
177}

182void TempoSyncKnob::updateDescAndIcon()
183{
184 if( model()->m_tempoSyncMode != TempoSyncKnobModel::SyncMode::None )
185 {
186 switch( model()->m_tempoSyncMode )
187 {
188 case TempoSyncKnobModel::SyncMode::Custom:
189 m_tempoSyncDescription = tr( "Custom " ) +
190 "(" +
191 QString::number( model()->m_custom.numeratorModel().value() ) +
192 "/" +
193 QString::number( model()->m_custom.denominatorModel().value() ) +
194 ")";
195 break;
196 case TempoSyncKnobModel::SyncMode::DoubleWholeNote:
197 m_tempoSyncDescription = tr(
198 "Synced to Eight Beats" );
199 break;
200 case TempoSyncKnobModel::SyncMode::WholeNote:
201 m_tempoSyncDescription = tr(
202 "Synced to Whole Note" );
203 break;
204 case TempoSyncKnobModel::SyncMode::HalfNote:
205 m_tempoSyncDescription = tr(
206 "Synced to Half Note" );
207 break;
208 case TempoSyncKnobModel::SyncMode::QuarterNote:
209 m_tempoSyncDescription = tr(
210 "Synced to Quarter Note" );
211 break;
212 case TempoSyncKnobModel::SyncMode::EighthNote:
213 m_tempoSyncDescription = tr(
214 "Synced to 8th Note" );
215 break;
216 case TempoSyncKnobModel::SyncMode::SixteenthNote:
217 m_tempoSyncDescription = tr(
218 "Synced to 16th Note" );
219 break;
220 case TempoSyncKnobModel::SyncMode::ThirtysecondNote:
221 m_tempoSyncDescription = tr(
222 "Synced to 32nd Note" );
223 break;
224 default: ;
225 }
226 }
227 else
228 {
229 m_tempoSyncDescription = tr( "Tempo Sync" );
230 }
231 if( m_custom != nullptr &&
232 model()->m_tempoSyncMode != TempoSyncKnobModel::SyncMode::Custom )
233 {
234 m_custom->parentWidget()->hide();
235 }
236
237 switch( model()->m_tempoSyncMode )
238 {
239 case TempoSyncKnobModel::SyncMode::None:
240 m_tempoSyncIcon = embed::getIconPixmap( "tempo_sync" );
241 break;
242 case TempoSyncKnobModel::SyncMode::Custom:
243 m_tempoSyncIcon = embed::getIconPixmap( "dont_know" );
244 break;
245 case TempoSyncKnobModel::SyncMode::DoubleWholeNote:
246 m_tempoSyncIcon = embed::getIconPixmap( "note_double_whole" );
247 break;
248 case TempoSyncKnobModel::SyncMode::WholeNote:
249 m_tempoSyncIcon = embed::getIconPixmap( "note_whole" );
250 break;
251 case TempoSyncKnobModel::SyncMode::HalfNote:
252 m_tempoSyncIcon = embed::getIconPixmap( "note_half" );
253 break;
254 case TempoSyncKnobModel::SyncMode::QuarterNote:
255 m_tempoSyncIcon = embed::getIconPixmap( "note_quarter" );
256 break;
257 case TempoSyncKnobModel::SyncMode::EighthNote:
258 m_tempoSyncIcon = embed::getIconPixmap( "note_eighth" );
259 break;
260 case TempoSyncKnobModel::SyncMode::SixteenthNote:
261 m_tempoSyncIcon = embed::getIconPixmap( "note_sixteenth" );
262 break;
263 case TempoSyncKnobModel::SyncMode::ThirtysecondNote:
264 m_tempoSyncIcon = embed::getIconPixmap( "note_thirtysecond" );
265 break;
266 default:
267 qWarning( "TempoSyncKnob::calculateTempoSyncTime:"
268 "invalid TempoSyncMode" );
269 break;
270 }
271
272 emit syncDescriptionChanged( m_tempoSyncDescription );
273 emit syncIconChanged();
274}

313void TempoSyncKnob::showCustom()
314{
315 if( m_custom == nullptr )
316 {
317 m_custom = new MeterDialog( getGUI()->mainWindow()->workspace() );
318 QMdiSubWindow * subWindow = getGUI()->mainWindow()->addWindowedWidget( m_custom );
319 Qt::WindowFlags flags = subWindow->windowFlags();
320 flags &= ~Qt::WindowMaximizeButtonHint;
321 subWindow->setWindowFlags( flags );
322 subWindow->setFixedSize( subWindow->size() );
323 m_custom->setWindowTitle( "Meter" );
324 m_custom->setModel( &model()->m_custom );
325 }
326 m_custom->parentWidget()->show();
327 model()->setTempoSync( TempoSyncKnobModel::SyncMode::Custom );
328}
