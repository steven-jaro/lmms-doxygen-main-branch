99void ControllerConnection::setController( Controller * _controller )
100{
101 if( m_ownsController && m_controller )
102 {
103 delete m_controller;
104 m_controller = nullptr;
105 }
106
107 if( m_controller && m_controller->type() != Controller::ControllerType::Dummy )
108 {
109 m_controller->removeConnection( this );
110 }
111
112 if( !_controller )
113 {
114 m_controller = Controller::create( Controller::ControllerType::Dummy, nullptr );
115 }
116 else
117 {
118 m_controller = _controller;
119 }
120 m_controllerId = -1;
121
122 if( _controller->type() != Controller::ControllerType::Dummy )
123 {
124 _controller->addConnection( this );
125 QObject::connect( _controller, SIGNAL(valueChanged()),
126 this, SIGNAL(valueChanged()), Qt::DirectConnection );
127 }
128
129 m_ownsController =
130 (_controller->type() == Controller::ControllerType::Midi);
131
132 // If we don't own the controller, allow deletion of controller
133 // to delete the connection
134 if( !m_ownsController ) {
135 QObject::connect( _controller, SIGNAL(destroyed()),
136 this, SLOT(deleteConnection()));
137 }
138}

153/*
154 * A connection may not be finalized. This means, the connection should exist,
155 * but the controller does not yet exist. This happens when loading. Even
156 * loading connections last won't help, since there can be connections BETWEEN
157 * controllers. So, we remember the controller-ID and use a dummyController
158 * instead. Once the song is loaded, finalizeConnections() connects to the proper controllers
159 */
160void ControllerConnection::finalizeConnections()
161{
162 for (auto i = std::size_t{0}; i < s_connections.size(); ++i)
163 {
164 ControllerConnection * c = s_connections[i];
165 if (!c->isFinalized() && static_cast<std::size_t>(c->m_controllerId) < Engine::getSong()->controllers().size())
166 {
167 c->setController( Engine::getSong()->
168 controllers().at( c->m_controllerId ) );
169 }
170 else if (c->getController()->type() == Controller::ControllerType::Dummy)
171 {
172 delete c;
173 --i;
174 }
175 }
176}

181void ControllerConnection::saveSettings( QDomDocument & _doc, QDomElement & _this )
182{
183 if( Engine::getSong() )
184 {
185 if( m_ownsController )
186 {
187 m_controller->saveState( _doc, _this );
188 }
189 else
190 {
191 const auto& controllers = Engine::getSong()->controllers();
192 const auto it = std::find(controllers.begin(), controllers.end(), m_controller);
193 if (it != controllers.end())
194 {
195 const int id = std::distance(controllers.begin(), it);
196 _this.setAttribute("id", id);
197 }
198 }
199 }

205void ControllerConnection::loadSettings( const QDomElement & _this )
206{
207 QDomNode node = _this.firstChild();
208 if( !node.isNull() )
209 {
210 setController( Controller::create( node.toElement(), Engine::getSong() ) );
211 }
212 else
213 {
214 m_controllerId = _this.attribute( "id", "-1" ).toInt();
215 if( m_controllerId < 0 )
216 {
217 qWarning( "controller index invalid\n" );
218 m_controllerId = -1;
219 }
220
221 if (!Engine::getSong()->isLoadingProject()
222 && m_controllerId != -1
223 && static_cast<std::size_t>(m_controllerId) < Engine::getSong()->controllers().size())
224 {
225 setController( Engine::getSong()->
226 controllers().at( m_controllerId ) );
227 }
228 else
229 {
230 m_controller = Controller::create( Controller::ControllerType::Dummy, nullptr );
231 }
232 }
233}
