54void ProjectJournal::undo()
55{
56 while( !m_undoCheckPoints.isEmpty() )
57 {
58 CheckPoint c = m_undoCheckPoints.pop();
59 JournallingObject *jo = m_joIDs[c.joID];
60
61 if( jo )
62 {
63 DataFile curState( DataFile::Type::JournalData );
64 jo->saveState( curState, curState.content() );
65 m_redoCheckPoints.push( CheckPoint( c.joID, curState ) );
66
67 bool prev = isJournalling();
68 setJournalling( false );
69 jo->restoreState( c.data.content().firstChildElement() );
70 setJournalling( prev );
71 Engine::getSong()->setModified();
72
73 // loading AutomationClip connections correctly
74 if (!c.data.content().elementsByTagName("automationclip").isEmpty())
75 {
76 AutomationClip::resolveAllIDs();
77 }
78 break;
79 }
80 }
81}

120void ProjectJournal::addJournalCheckPoint( JournallingObject *jo )
121{
122 if( isJournalling() )
123 {
124 m_redoCheckPoints.clear();
125
126 DataFile dataFile( DataFile::Type::JournalData );
127 jo->saveState( dataFile, dataFile.content() );
128
129 m_undoCheckPoints.push( CheckPoint( jo->id(), dataFile ) );
130 if( m_undoCheckPoints.size() > MAX_UNDO_STATES )
131 {
132 m_undoCheckPoints.remove( 0, m_undoCheckPoints.size() - MAX_UNDO_STATES );
133 }
134 }
135}

181void ProjectJournal::clearJournal()
182{
183 m_undoCheckPoints.clear();
184 m_redoCheckPoints.clear();
185
186 for( JoIdMap::Iterator it = m_joIDs.begin(); it != m_joIDs.end(); )
187 {
188 if( it.value() == nullptr )
189 {
190 it = m_joIDs.erase( it );
191 }
192 else
193 {
194 ++it;
195 }
196 }
197}

199void ProjectJournal::stopAllJournalling()
200{
201 for( JoIdMap::Iterator it = m_joIDs.begin(); it != m_joIDs.end(); ++it)
202 {
203 if( it.value() != nullptr )
204 {
205 it.value()->setJournalling(false);
206 }
207 }
208 setJournalling(false);
209}
