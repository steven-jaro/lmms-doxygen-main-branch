46PeakController::PeakController( Model * _parent,
47 PeakControllerEffect * _peak_effect ) :
48 Controller( ControllerType::Peak, _parent, tr( "Peak Controller" ) ),
49 m_peakEffect( _peak_effect ),
50 m_currentSample( 0.0f )
51{
52 setSampleExact( true );
53 if( m_peakEffect )
54 {
55 connect( m_peakEffect, SIGNAL(destroyed()),
56 this, SLOT(handleDestroyedEffect()));
57 }
58 connect( Engine::audioEngine(), SIGNAL(sampleRateChanged()), this, SLOT(updateCoeffs()));
59 connect( m_peakEffect->attackModel(), SIGNAL(dataChanged()),
60 this, SLOT(updateCoeffs()), Qt::DirectConnection );
61 connect( m_peakEffect->decayModel(), SIGNAL(dataChanged()),
62 this, SLOT(updateCoeffs()), Qt::DirectConnection );
63 m_coeffNeedsUpdate = true;
64}

78void PeakController::updateValueBuffer()
79{
80 if( m_coeffNeedsUpdate )
81 {
82 const float ratio = 44100.0f / Engine::audioEngine()->outputSampleRate();
83 m_attackCoeff = 1.0f - powf( 2.0f, -0.3f * ( 1.0f - m_peakEffect->attackModel()->value() ) * ratio );
84 m_decayCoeff = 1.0f - powf( 2.0f, -0.3f * ( 1.0f - m_peakEffect->decayModel()->value() ) * ratio );
85 m_coeffNeedsUpdate = false;
86 }
87
88 if( m_peakEffect )
89 {
90 float targetSample = m_peakEffect->lastSample();
91 if( m_currentSample != targetSample )
92 {
93 const f_cnt_t frames = Engine::audioEngine()->framesPerPeriod();
94 float * values = m_valueBuffer.values();
95
96 for( f_cnt_t f = 0; f < frames; ++f )
97 {
98 const float diff = ( targetSample - m_currentSample );
99 if( m_currentSample < targetSample ) // going up...
100 {
101 m_currentSample += diff * m_attackCoeff;
102 }
103 else if( m_currentSample > targetSample ) // going down
104 {
105 m_currentSample += diff * m_decayCoeff;
106 }
107 values[f] = m_currentSample;
108 }
109 }
110 else
111 {
112 m_valueBuffer.fill( m_currentSample );
113 }
114 }
115 else
116 {
117 m_valueBuffer.fill( 0 );
118 }
119 m_bufferLastUpdated = s_periods;
120}

153void PeakController::loadSettings( const QDomElement & _this )
154{
155 Controller::loadSettings( _this );
156
157 int effectId = _this.attribute( "effectId" ).toInt();
158 if( m_buggedFile == true )
159 {
160 effectId = m_loadCount++;
161 }
162
163 for (const auto& effect : s_effects)
164 {
165 if (effect->m_effectId == effectId)
166 {
167 m_peakEffect = effect;
168 return;
169 }
170 }

187PeakController * PeakController::getControllerBySetting(const QDomElement & _this )
188{
189 int effectId = _this.attribute( "effectId" ).toInt();
190
191 //Backward compatibility for bug in <= 0.4.15 . For >= 1.0.0 ,
192 //foundCount should always be 1 because m_effectId is initialized with rand()
193 int foundCount = 0;
194 if( m_buggedFile == false )
195 {
196 for (const auto& effect : s_effects)
197 {
198 if (effect->m_effectId == effectId)
199 {
200 foundCount++;
201 }
202 }
203 if( foundCount >= 2 )
204 {
205 m_buggedFile = true;
206 int newEffectId = 0;
207 for (const auto& effect : s_effects)
208 {
209 effect->m_effectId = newEffectId++;
210 }
211 QMessageBox msgBox;
212 msgBox.setIcon( QMessageBox::Information );
213 msgBox.setWindowTitle( tr("Peak Controller Bug") );
214 msgBox.setText( tr("Due to a bug in older version of LMMS, the peak "
215 "controllers may not be connect properly. "
216 "Please ensure that peak controllers are connected "
217 "properly and re-save this file. "
218 "Sorry for any inconvenience caused.") );
219 msgBox.setStandardButtons(QMessageBox::Ok);
220 msgBox.exec();
221 }
222 }
223
224 if( m_buggedFile == true )
225 {
226 effectId = m_getCount;
227 }
228 m_getCount++; //NB: m_getCount should be increased even m_buggedFile is false
229
230 for (const auto& effect : s_effects)
231 {
232 if (effect->m_effectId == effectId)
233 {
234 return effect->controller();
235 }
236 }
237
238 return nullptr;
239}
