50 static void do_complete(void* owner, operation* base,
51 const asio::error_code& /*ec*/,
52 std::size_t /*bytes_transferred*/)
53 {
54 // Take ownership of the handler object.
55 reactive_null_buffers_op* o(static_cast<reactive_null_buffers_op*>(base));
56 ptr p = { asio::detail::addressof(o->handler_), o, o };
57 handler_work<Handler, IoExecutor> w(o->handler_, o->io_executor_);
58
59 ASIO_HANDLER_COMPLETION((*o));
60
61 // Make a copy of the handler so that the memory can be deallocated before
62 // the upcall is made. Even if we're not about to make an upcall, a
63 // sub-object of the handler may be the true owner of the memory associated
64 // with the handler. Consequently, a local copy of the handler is required
65 // to ensure that any owning sub-object remains valid until after we have
66 // deallocated the memory here.
67 detail::binder2<Handler, asio::error_code, std::size_t>
68 handler(o->handler_, o->ec_, o->bytes_transferred_);
69 p.h = asio::detail::addressof(handler.handler_);
70 p.reset();
71
72 // Make the upcall if required.
73 if (owner)
74 {
75 fenced_block b(fenced_block::half);
76 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
77 w.complete(handler, handler.handler_);
78 ASIO_HANDLER_INVOCATION_END;
79 }
80 }
81
82private:
83 Handler handler_;
84 IoExecutor io_executor_;
85};
