44int Fl_Osc_TSlider::handle(int event)
45{
46 int super = 1;
47
48 super = Fl_Osc_Slider::handle(event);
49
50 switch(event) {
51 case FL_PUSH:
52 case FL_MOUSEWHEEL:
53 if (!Fl::event_inside(x(),y(),w(),h()))
54 return(1);
55 tipwin->position(Fl::event_x_root()-Fl::event_x()+x(),
56 Fl::event_y_root()-Fl::event_y()+h()+y()+5);
57 case FL_DRAG:
58 tipwin->showValue(transform(value()));
59 break;
60 case FL_RELEASE:
61 case FL_HIDE:
62 case FL_LEAVE:
63 if (tipwin)
64 tipwin->hide();
65 return 1;
66 }
67
68 return super;
69}
