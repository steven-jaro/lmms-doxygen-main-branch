168// A function object adapter to invoke a nullary function object and capture
169// any exception thrown into a promise.
170template <typename T, typename F>
171class promise_invoker
172{
173public:
174 promise_invoker(const shared_ptr<std::promise<T> >& p,
175 ASIO_MOVE_ARG(F) f)
176 : p_(p), f_(ASIO_MOVE_CAST(F)(f))
177 {
178 }
179
180 void operator()()
181 {
182#if !defined(ASIO_NO_EXCEPTIONS)
183 try
184#endif // !defined(ASIO_NO_EXCEPTIONS)
185 {
186 f_();
187 }
188#if !defined(ASIO_NO_EXCEPTIONS)
189 catch (...)
190 {
191 p_->set_exception(std::current_exception());
192 }
193#endif // !defined(ASIO_NO_EXCEPTIONS)
194 }
195
196private:
197 shared_ptr<std::promise<T> > p_;
198 typename decay<F>::type f_;
199};

201// An executor that adapts the system_executor to capture any exeption thrown
202// by a submitted function object and save it into a promise.
203template <typename T>
204class promise_executor
205{
206public:
207 explicit promise_executor(const shared_ptr<std::promise<T> >& p)
208 : p_(p)
209 {
210 }
211
212 execution_context& context() const ASIO_NOEXCEPT
213 {
214 return system_executor().context();
215 }
216
217 void on_work_started() const ASIO_NOEXCEPT {}
218 void on_work_finished() const ASIO_NOEXCEPT {}
219
220 template <typename F, typename A>
221 void dispatch(ASIO_MOVE_ARG(F) f, const A&) const
222 {
223 promise_invoker<T, F>(p_, ASIO_MOVE_CAST(F)(f))();
224 }
225
226 template <typename F, typename A>
227 void post(ASIO_MOVE_ARG(F) f, const A& a) const
228 {
229 system_executor().post(
230 promise_invoker<T, F>(p_, ASIO_MOVE_CAST(F)(f)), a);
231 }
232
233 template <typename F, typename A>
234 void defer(ASIO_MOVE_ARG(F) f, const A& a) const
235 {
236 system_executor().defer(
237 promise_invoker<T, F>(p_, ASIO_MOVE_CAST(F)(f)), a);
238 }
239
240 friend bool operator==(const promise_executor& a,
241 const promise_executor& b) ASIO_NOEXCEPT
242 {
243 return a.p_ == b.p_;
244 }
245
246 friend bool operator!=(const promise_executor& a,
247 const promise_executor& b) ASIO_NOEXCEPT
248 {
249 return a.p_ != b.p_;
250 }
251
252private:
253 shared_ptr<std::promise<T> > p_;
254};

256// The base class for all completion handlers that create promises.
257template <typename T>
258class promise_creator
259{
260public:
261 typedef promise_executor<T> executor_type;
262
263 executor_type get_executor() const ASIO_NOEXCEPT
264 {
265 return executor_type(p_);
266 }
267
268 typedef std::future<T> future_type;
269
270 future_type get_future()
271 {
272 return p_->get_future();
273 }
274
275protected:
276 template <typename Allocator>
277 void create_promise(const Allocator& a)
278 {
279 ASIO_REBIND_ALLOC(Allocator, char) b(a);
280 p_ = std::allocate_shared<std::promise<T>>(b, std::allocator_arg, b);
281 }
282
283 shared_ptr<std::promise<T> > p_;
284};

585// Completion handlers produced from the use_future completion token, when not
586// using use_future::operator().
587template <typename Signature, typename Allocator>
588class promise_handler
589 : public promise_handler_selector<Signature>
590{
591public:
592 typedef Allocator allocator_type;
593 typedef void result_type;
594
595 promise_handler(use_future_t<Allocator> u)
596 : allocator_(u.get_allocator())
597 {
598 this->create_promise(allocator_);
599 }
600
601 allocator_type get_allocator() const ASIO_NOEXCEPT
602 {
603 return allocator_;
604 }
605
606private:
607 Allocator allocator_;
608};

610template <typename Function, typename Signature, typename Allocator>
611inline void asio_handler_invoke(Function& f,
612 promise_handler<Signature, Allocator>* h)
613{
614 typename promise_handler<Signature, Allocator>::executor_type
615 ex(h->get_executor());
616 ex.dispatch(ASIO_MOVE_CAST(Function)(f), std::allocator<void>());
617}
