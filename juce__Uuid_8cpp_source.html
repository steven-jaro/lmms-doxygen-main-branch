26Uuid::Uuid()
27{
28 Random r;
29
30 for (size_t i = 0; i < sizeof (uuid); ++i)
31 uuid[i] = (uint8) (r.nextInt (256));
32
33 // To make it RFC 4122 compliant, need to force a few bits...
34 uuid[6] = (uuid[6] & 0x0f) | 0x40;
35 uuid[8] = (uuid[8] & 0x3f) | 0x80;
36}
