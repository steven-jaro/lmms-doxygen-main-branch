42 template <typename SyncReadStream, typename MutableBufferSequence,
43 typename MutableBufferIterator, typename CompletionCondition>
44 std::size_t read_buffer_sequence(SyncReadStream& s,
45 const MutableBufferSequence& buffers, const MutableBufferIterator&,
46 CompletionCondition completion_condition, asio::error_code& ec)
47 {
48 ec = asio::error_code();
49 asio::detail::consuming_buffers<mutable_buffer,
50 MutableBufferSequence, MutableBufferIterator> tmp(buffers);
51 while (!tmp.empty())
52 {
53 if (std::size_t max_size = detail::adapt_completion_condition_result(
54 completion_condition(ec, tmp.total_consumed())))
55 tmp.consume(s.read_some(tmp.prepare(max_size), ec));
56 else
57 break;
58 }
59 return tmp.total_consumed();;
60 }

237template <typename SyncReadStream, typename DynamicBuffer_v2,
238 typename CompletionCondition>
239std::size_t read(SyncReadStream& s, DynamicBuffer_v2 buffers,
240 CompletionCondition completion_condition, asio::error_code& ec,
241 typename enable_if<
242 is_dynamic_buffer_v2<DynamicBuffer_v2>::value
243 >::type*)
244{
245 DynamicBuffer_v2& b = buffers;
246
247 ec = asio::error_code();
248 std::size_t total_transferred = 0;
249 std::size_t max_size = detail::adapt_completion_condition_result(
250 completion_condition(ec, total_transferred));
251 std::size_t bytes_available = std::min<std::size_t>(
252 std::max<std::size_t>(512, b.capacity() - b.size()),
253 std::min<std::size_t>(max_size, b.max_size() - b.size()));
254 while (bytes_available > 0)
255 {
256 std::size_t pos = b.size();
257 b.grow(bytes_available);
258 std::size_t bytes_transferred = s.read_some(
259 b.data(pos, bytes_available), ec);
260 b.shrink(bytes_available - bytes_transferred);
261 total_transferred += bytes_transferred;
262 max_size = detail::adapt_completion_condition_result(
263 completion_condition(ec, total_transferred));
264 bytes_available = std::min<std::size_t>(
265 std::max<std::size_t>(512, b.capacity() - b.size()),
266 std::min<std::size_t>(max_size, b.max_size() - b.size()));
267 }
268 return total_transferred;
269}

316 class read_op
317 : detail::base_from_completion_cond<CompletionCondition>
318 {
319 public:
320 read_op(AsyncReadStream& stream, const MutableBufferSequence& buffers,
321 CompletionCondition& completion_condition, ReadHandler& handler)
322 : detail::base_from_completion_cond<
323 CompletionCondition>(completion_condition),
324 stream_(stream),
325 buffers_(buffers),
326 start_(0),
327 handler_(ASIO_MOVE_CAST(ReadHandler)(handler))

331#if defined(ASIO_HAS_MOVE)
332 read_op(const read_op& other)
333 : detail::base_from_completion_cond<CompletionCondition>(other),
334 stream_(other.stream_),
335 buffers_(other.buffers_),
336 start_(other.start_),
337 handler_(other.handler_)
338 {
339 }
340
341 read_op(read_op&& other)
342 : detail::base_from_completion_cond<CompletionCondition>(
343 ASIO_MOVE_CAST(detail::base_from_completion_cond<
344 CompletionCondition>)(other)),
345 stream_(other.stream_),
346 buffers_(ASIO_MOVE_CAST(buffers_type)(other.buffers_)),
347 start_(other.start_),
348 handler_(ASIO_MOVE_CAST(ReadHandler)(other.handler_))
349 {
350 }
351#endif // defined(ASIO_HAS_MOVE)
352
353 void operator()(const asio::error_code& ec,
354 std::size_t bytes_transferred, int start = 0)
355 {
356 std::size_t max_size;
357 switch (start_ = start)
358 {
359 case 1:
360 max_size = this->check_for_completion(ec, buffers_.total_consumed());
361 do
362 {
363 stream_.async_read_some(buffers_.prepare(max_size),
364 ASIO_MOVE_CAST(read_op)(*this));
365 return; default:
366 buffers_.consume(bytes_transferred);
367 if ((!ec && bytes_transferred == 0) || buffers_.empty())
368 break;
369 max_size = this->check_for_completion(ec, buffers_.total_consumed());
370 } while (max_size > 0);
371
372 handler_(ec, buffers_.total_consumed());
373 }
374 }

376 //private:
377 typedef asio::detail::consuming_buffers<mutable_buffer,
378 MutableBufferSequence, MutableBufferIterator> buffers_type;
379
380 AsyncReadStream& stream_;
381 buffers_type buffers_;
382 int start_;
383 ReadHandler handler_;
384 };

863 read_dynbuf_v2_op(read_dynbuf_v2_op&& other)
864 : detail::base_from_completion_cond<CompletionCondition>(
865 ASIO_MOVE_CAST(detail::base_from_completion_cond<
866 CompletionCondition>)(other)),
867 stream_(other.stream_),
868 buffers_(ASIO_MOVE_CAST(DynamicBuffer_v2)(other.buffers_)),
869 start_(other.start_),
870 total_transferred_(other.total_transferred_),

876
877 void operator()(const asio::error_code& ec,
878 std::size_t bytes_transferred, int start = 0)
879 {
880 std::size_t max_size, pos;
881 switch (start_ = start)
882 {
883 case 1:
884 max_size = this->check_for_completion(ec, total_transferred_);
885 bytes_available_ = std::min<std::size_t>(
886 std::max<std::size_t>(512,
887 buffers_.capacity() - buffers_.size()),
888 std::min<std::size_t>(max_size,
889 buffers_.max_size() - buffers_.size()));
890 for (;;)
891 {
892 pos = buffers_.size();
893 buffers_.grow(bytes_available_);
894 stream_.async_read_some(buffers_.data(pos, bytes_available_),
895 ASIO_MOVE_CAST(read_dynbuf_v2_op)(*this));
896 return; default:
897 total_transferred_ += bytes_transferred;
898 buffers_.shrink(bytes_available_ - bytes_transferred);
899 max_size = this->check_for_completion(ec, total_transferred_);
900 bytes_available_ = std::min<std::size_t>(
901 std::max<std::size_t>(512,
902 buffers_.capacity() - buffers_.size()),
903 std::min<std::size_t>(max_size,
904 buffers_.max_size() - buffers_.size()));
905 if ((!ec && bytes_transferred == 0) || bytes_available_ == 0)
906 break;
907 }
908
909 handler_(ec, static_cast<const std::size_t&>(total_transferred_));
910 }
911 }
