49InstrumentTrack::InstrumentTrack(TrackContainer* tc) :
50 Track(Track::Type::Instrument, tc),
51 MidiEventProcessor(),
52 m_midiPort(tr("unnamed_track"), Engine::audioEngine()->midiClient(), this, this),
53 m_notes(),
54 m_sustainPedalPressed(false),
55 m_silentBuffersProcessed(false),
56 m_previewMode(false),
57 m_baseNoteModel(0, 0, NumKeys - 1, this, tr("Base note")),
58 m_firstKeyModel(0, 0, NumKeys - 1, this, tr("First note")),
59 m_lastKeyModel(0, 0, NumKeys - 1, this, tr("Last note")),
60 m_hasAutoMidiDev(false),
61 m_volumeModel(DefaultVolume, MinVolume, MaxVolume, 0.1f, this, tr("Volume")),
62 m_panningModel(DefaultPanning, PanningLeft, PanningRight, 0.1f, this, tr("Panning")),
63 m_audioBusHandle(tr("unnamed_track"), true, &m_volumeModel, &m_panningModel, &m_mutedModel),
64 m_pitchModel(0, MinPitchDefault, MaxPitchDefault, 1, this, tr("Pitch")),
65 m_pitchRangeModel(1, 1, 60, this, tr("Pitch range")),
66 m_mixerChannelModel(0, 0, 0, this, tr("Mixer channel")),
67 m_useMasterPitchModel(true, this, tr("Master pitch")),
68 m_instrument(nullptr),
69 m_soundShaping(this),
70 m_arpeggio(this),
71 m_noteStacking(this),
72 m_piano(this),
73 m_microtuner()
74{
75 m_pitchModel.setCenterValue( 0 );
76 m_pitchModel.setStrictStepSize(true);
77 m_panningModel.setCenterValue( DefaultPanning );
78 m_baseNoteModel.setInitValue( DefaultKey );
79 m_firstKeyModel.setInitValue(0);
80 m_lastKeyModel.setInitValue(NumKeys - 1);
81
82 m_mixerChannelModel.setRange( 0, Engine::mixer()->numChannels()-1, 1);
83
84 for( int i = 0; i < NumKeys; ++i )
85 {
86 m_notes[i] = nullptr;
87 m_runningMidiNotes[i] = 0;
88 }
89
90
91 // Initialize the m_midiCCEnabled variable, but it's actually going to be connected
92 // to a LedButton
93 m_midiCCEnable = std::make_unique<BoolModel>(false, nullptr, tr("Enable/Disable MIDI CC"));
94
95 // Initialize the MIDI CC controller models and connect them to the method that processes
96 // the midi cc events
97 for (int i = 0; i < MidiControllerCount; ++i)
98 {
99 m_midiCCModel[i] = std::make_unique<FloatModel>(0.0f, 0.0f, 127.0f, 1.0f,
100 nullptr, tr("CC Controller %1").arg(i));
101
102 connect(m_midiCCModel[i].get(), &FloatModel::dataChanged,
103 this, [this, i]{ processCCEvent(i); }, Qt::DirectConnection);
104 }
105
106 setName( tr( "Default preset" ) );
107
108 connect(&m_baseNoteModel, SIGNAL(dataChanged()), this, SLOT(updateBaseNote()), Qt::DirectConnection);
109 connect(&m_pitchModel, SIGNAL(dataChanged()), this, SLOT(updatePitch()), Qt::DirectConnection);
110 connect(&m_pitchRangeModel, SIGNAL(dataChanged()), this, SLOT(updatePitchRange()), Qt::DirectConnection);
111 connect(&m_mixerChannelModel, SIGNAL(dataChanged()), this, SLOT(updateMixerChannel()), Qt::DirectConnection);
112
113 autoAssignMidiDevice(true);
114}

124/** \brief Check if there is a valid mapping for the given key and it is within defined of range.
125 */
126bool InstrumentTrack::isKeyMapped(int key) const
127{
128 if (key < firstKey() || key > lastKey()) {return false;}
129 if (!m_microtuner.enabled()) {return true;}
130
131 Song *song = Engine::getSong();
132 if (!song) {return false;}
133
134 return song->getKeymap(m_microtuner.currentKeymap())->getDegree(key) != -1;
135}

224void InstrumentTrack::processAudioBuffer( SampleFrame* buf, const fpp_t frames, NotePlayHandle* n )
225{
226 // we must not play the sound if this InstrumentTrack is muted...
227 if( isMuted() || ( Engine::getSong()->playMode() != Song::PlayMode::MidiClip &&
228 n && n->isPatternTrackMuted() ) || ! m_instrument )
229 {
230 return;
231 }
232
233 // Test for silent input data if instrument provides a single stream only (i.e. driven by InstrumentPlayHandle)
234 // We could do that in all other cases as well but the overhead for silence test is bigger than
235 // what we potentially save. While playing a note, a NotePlayHandle-driven instrument will produce sound in
236 // 99 of 100 cases so that test would be a waste of time.
237 if (m_instrument->isSingleStreamed() && MixHelpers::isSilent(buf, frames))
238 {
239 // at least pass one silent buffer to allow
240 if( m_silentBuffersProcessed )
241 {
242 // skip further processing
243 return;
244 }
245 m_silentBuffersProcessed = true;
246 }
247 else
248 {
249 m_silentBuffersProcessed = false;
250 }
251
252 // if effects "went to sleep" because there was no input, wake them up
253 // now
254 m_audioBusHandle.effects()->startRunning();
255
256 // get volume knob data
257 static const float DefaultVolumeRatio = 1.0f / DefaultVolume;
258 /*ValueBuffer * volBuf = m_volumeModel.valueBuffer();
259 float v_scale = volBuf
260 ? 1.0f
261 : getVolume() * DefaultVolumeRatio;*/
262
263 // instruments using instrument-play-handles will call this method
264 // without any knowledge about notes, so they pass NULL for n, which
265 // is no problem for us since we just bypass the envelopes+LFOs
266 if (!m_instrument->isSingleStreamed() && n != nullptr)
267 {
268 const f_cnt_t offset = n->noteOffset();
269 m_soundShaping.processAudioBuffer( buf + offset, frames - offset, n );
270 const float vol = ( (float) n->getVolume() * DefaultVolumeRatio );
271 const panning_t pan = std::clamp(n->getPanning(), PanningLeft, PanningRight);
272 StereoVolumeVector vv = panningToVolumeVector( pan, vol );
273 for( f_cnt_t f = offset; f < frames; ++f )
274 {
275 for( int c = 0; c < 2; ++c )
276 {
277 buf[f][c] *= vv.vol[c];
278 }
279 }
280 }
281}

286MidiEvent InstrumentTrack::applyMasterKey( const MidiEvent& event )
287{
288 MidiEvent copy( event );
289 switch( event.type() )
290 {
291 case MidiNoteOn:
292 case MidiNoteOff:
293 case MidiKeyPressure:
294 copy.setKey( masterKey( event.key() ) );
295 break;
296 default:
297 break;
298 }
299 return copy;

305void InstrumentTrack::processCCEvent(int controller)
306{
307 // Does nothing if the LED is disabled
308 if (!m_midiCCEnable->value()) { return; }
309
310 auto channel = static_cast<uint8_t>(midiPort()->realOutputChannel());
311 auto cc = static_cast<uint16_t>(controller);
312 auto value = static_cast<uint16_t>(m_midiCCModel[controller]->value());
313
314 // Process the MIDI CC event as an input event but with source set to Internal
315 // so we can know LMMS generated the event, not a controller, and can process it during
316 // the project export
317 processInEvent(MidiEvent(MidiControlChange, channel, cc, value, nullptr, MidiEvent::Source::Internal));
318}

323void InstrumentTrack::processInEvent( const MidiEvent& event, const TimePos& time, f_cnt_t offset )
324{
325 if (Engine::getSong()->isExporting() && event.source() == MidiEvent::Source::External)
326 {
327 return;
328 }
329
330 bool eventHandled = false;
331
332 switch( event.type() )
333 {
334 // we don't send MidiNoteOn, MidiNoteOff and MidiKeyPressure
335 // events to instrument as NotePlayHandle will send them on its
336 // own
337 case MidiNoteOn:
338 if( event.velocity() > 0 )
339 {
340 // play a note only if it is not already playing and if it is within configured bounds
341 if (m_notes[event.key()] == nullptr && event.key() >= firstKey() && event.key() <= lastKey())
342 {
343 NotePlayHandle* nph =
344 NotePlayHandleManager::acquire(
345 this, offset,
346 std::numeric_limits<f_cnt_t>::max() / 2,
347 Note(TimePos(), Engine::getSong()->getPlayPos(Engine::getSong()->playMode()),
348 event.key(), event.volume(midiPort()->baseVelocity())),
349 nullptr, event.channel(),
350 NotePlayHandle::Origin::MidiInput);
351 m_notes[event.key()] = nph;
352 if( ! Engine::audioEngine()->addPlayHandle( nph ) )
353 {
354 m_notes[event.key()] = nullptr;
355 }
356 }
357 eventHandled = true;
358 break;
359 }
360
361 case MidiNoteOff:
362 if( m_notes[event.key()] != nullptr )
363 {
364 // do actual note off and remove internal reference to NotePlayHandle (which itself will
365 // be deleted later automatically)
366 Engine::audioEngine()->requestChangeInModel();
367 m_notes[event.key()]->noteOff( offset );
368 if (isSustainPedalPressed() &&
369 m_notes[event.key()]->origin() ==
370 NotePlayHandle::Origin::MidiInput)
371 {
372 m_sustainedNotes << m_notes[event.key()];
373 }
374 m_notes[event.key()] = nullptr;
375 Engine::audioEngine()->doneChangeInModel();
376 }
377 eventHandled = true;
378 break;
379
380 case MidiKeyPressure:
381 if( m_notes[event.key()] != nullptr )
382 {
383 // setVolume() calls processOutEvent() with MidiKeyPressure so the
384 // attached instrument will receive the event as well
385 m_notes[event.key()]->setVolume( event.volume( midiPort()->baseVelocity() ) );
386 }
387 eventHandled = true;
388 break;
389
390 case MidiPitchBend:
391 // updatePitch() is connected to m_pitchModel::dataChanged() which will send out
392 // MidiPitchBend events
393 m_pitchModel.setValue( m_pitchModel.minValue() + event.pitchBend() * m_pitchModel.range() / MidiMaxPitchBend );
394 break;
395
396 case MidiControlChange:
397 if( event.controllerNumber() == MidiControllerSustain )
398 {
399 if( event.controllerValue() > MidiMaxControllerValue/2 )
400 {
401 m_sustainPedalPressed = true;
402 }
403 else if (isSustainPedalPressed())
404 {
405 for (NotePlayHandle* nph : m_sustainedNotes)
406 {
407 if (nph && nph->isReleased())
408 {
409 if( nph->origin() ==
410 NotePlayHandle::Origin::MidiInput)
411 {
412 nph->setLength(
413 TimePos( static_cast<f_cnt_t>(
414 nph->totalFramesPlayed() /
415 Engine::framesPerTick() ) ) );
416 midiNoteOff( *nph );
417 }
418 }
419 }
420 m_sustainedNotes.clear();
421 m_sustainPedalPressed = false;
422 }
423 }
424 if( event.controllerNumber() == MidiControllerAllSoundOff ||
425 event.controllerNumber() == MidiControllerAllNotesOff ||
426 event.controllerNumber() == MidiControllerOmniOn ||
427 event.controllerNumber() == MidiControllerOmniOff ||
428 event.controllerNumber() == MidiControllerMonoOn ||
429 event.controllerNumber() == MidiControllerPolyOn )
430 {
431 silenceAllNotes();
432 }
433 break;
434
435 case MidiMetaEvent:
436 // handle special cases such as note panning
437 switch( event.metaEvent() )
438 {
439 case MidiNotePanning:
440 if( m_notes[event.key()] != nullptr )
441 {
442 eventHandled = true;
443 m_notes[event.key()]->setPanning( event.panning() );
444 }
445 break;
446 default:
447 qWarning( "InstrumentTrack: unhandled MIDI meta event: %i", event.metaEvent() );
448 break;
449 }
450 break;
451
452 default:
453 break;
454 }
455
456 // If the event wasn't handled, check if there's a loaded instrument and if so send the
457 // event to it. If it returns false means the instrument didn't handle the event, so we trigger a warning.
458 if (eventHandled == false && !(instrument() && instrument()->handleMidiEvent(event, time, offset)))
459 {
460 qWarning("InstrumentTrack: unhandled MIDI event %d", event.type());
461 }
462
463}

468void InstrumentTrack::processOutEvent( const MidiEvent& event, const TimePos& time, f_cnt_t offset )
469{
470 // do nothing if we do not have an instrument instance (e.g. when loading settings)
471 if( m_instrument == nullptr )
472 {
473 return;
474 }
475
476 const MidiEvent transposedEvent = applyMasterKey( event );
477 const int key = transposedEvent.key();
478
479 // If we have a selected output midi channel between 1-16, we will use that channel to handle the midi event.
480 // But if our selected midi output channel is 0 ("--"), we will use the event channel instead.
481 const auto handleEventOutputChannel = midiPort()->outputChannel() == 0
482 ? event.channel()
483 : midiPort()->realOutputChannel();
484
485 switch( event.type() )
486 {
487 case MidiNoteOn:
488 m_midiNotesMutex.lock();
489 m_piano.setKeyState( event.key(), true ); // event.key() = original key
490
491 if( key >= 0 && key < NumKeys )
492 {
493 if( m_runningMidiNotes[key] > 0 )
494 {
495 m_instrument->handleMidiEvent( MidiEvent( MidiNoteOff, handleEventOutputChannel, key, 0 ), time, offset );
496 }
497 ++m_runningMidiNotes[key];
498 m_instrument->handleMidiEvent( MidiEvent( MidiNoteOn, handleEventOutputChannel, key, event.velocity() ), time, offset );
499
500 }
501 m_midiNotesMutex.unlock();
502 emit newNote();
503 break;
504
505 case MidiNoteOff:
506 m_midiNotesMutex.lock();
507 m_piano.setKeyState( event.key(), false ); // event.key() = original key
508
509 if( key >= 0 && key < NumKeys && --m_runningMidiNotes[key] <= 0 )
510 {
511 m_instrument->handleMidiEvent( MidiEvent( MidiNoteOff, handleEventOutputChannel, key, 0 ), time, offset );
512 }
513 m_midiNotesMutex.unlock();
514 emit endNote();
515 break;
516
517 default:
518 m_instrument->handleMidiEvent( transposedEvent, time, offset );
519 break;
520 }
521
522 // if appropriate, midi-port does further routing
523 m_midiPort.processOutEvent( event, time );
524}

529void InstrumentTrack::silenceAllNotes( bool removeIPH )
530{
531 m_midiNotesMutex.lock();
532 for( int i = 0; i < NumKeys; ++i )
533 {
534 m_notes[i] = nullptr;
535 m_runningMidiNotes[i] = 0;
536 }
537 m_midiNotesMutex.unlock();
538
539 Engine::audioEngine()->requestChangeInModel();
540 // invalidate all NotePlayHandles and PresetPreviewHandles linked to this track
541 m_processHandles.clear();
542
543 auto flags = PlayHandle::Type::NotePlayHandle | PlayHandle::Type::PresetPreviewHandle;
544 if( removeIPH )
545 {
546 flags |= PlayHandle::Type::InstrumentPlayHandle;
547 }
548 Engine::audioEngine()->removePlayHandlesOfTypes( this, flags );
549 Engine::audioEngine()->doneChangeInModel();
550}

571void InstrumentTrack::playNote( NotePlayHandle* n, SampleFrame* workingBuffer )
572{
573 // Note: under certain circumstances the working buffer is a nullptr.
574 // These cases are triggered in PlayHandle::doProcessing when the play method is called with a nullptr.
575 // TODO: Find out if we can skip processing at a higher level if the buffer is nullptr.
576
577 // arpeggio- and chord-widget has to do its work -> adding sub-notes
578 // for chords/arpeggios
579 m_noteStacking.processNote( n );
580 m_arpeggio.processNote( n );
581
582 if( n->isMasterNote() == false && m_instrument != nullptr )
583 {
584 // all is done, so now lets play the note!
585 m_instrument->playNote( n, workingBuffer );
586
587 // This is effectively the same as checking if workingBuffer is not a nullptr.
588 // Calling processAudioBuffer with a nullptr leads to crashes. Hence the check.
589 if (n->usesBuffer())
590 {
591 const fpp_t frames = n->framesLeftForCurrentPeriod();
592 const f_cnt_t offset = n->noteOffset();
593 processAudioBuffer(workingBuffer, frames + offset, n);
594 }
595 }
596}

701bool InstrumentTrack::play( const TimePos & _start, const fpp_t _frames,
702 const f_cnt_t _offset, int _clip_num )
703{
704 if( ! m_instrument || ! tryLock() )
705 {
706 return false;
707 }
708 const float frames_per_tick = Engine::framesPerTick();
709
710 clipVector clips;
711 class PatternTrack * pattern_track = nullptr;
712 if( _clip_num >= 0 )
713 {
714 Clip * clip = getClip( _clip_num );
715 clips.push_back( clip );
716 if (trackContainer() == Engine::patternStore())
717 {
718 pattern_track = PatternTrack::findPatternTrack(_clip_num);
719 }
720 }
721 else
722 {
723 getClipsInRange( clips, _start, _start + static_cast<int>(
724 _frames / frames_per_tick ) );
725 }
726
727 // Handle automation: detuning
728 for (const auto& processHandle : m_processHandles)
729 {
730 processHandle->processTimePos(
731 _start, m_pitchModel.value(), gui::getGUI() && gui::getGUI()->pianoRoll()->isRecording());
732 }
733
734 if ( clips.size() == 0 )
735 {
736 unlock();
737 return false;
738 }
739
740 bool played_a_note = false; // will be return variable
741
742 for (const auto& clip : clips)
743 {
744 auto c = dynamic_cast<MidiClip*>(clip);
745 // everything which is not a MIDI clip won't be played
746 // A MIDI clip playing in the Piano Roll window will always play
747 if (c == nullptr || (Engine::getSong()->playMode() != Song::PlayMode::MidiClip && clip->isMuted()))
748 {
749 continue;
750 }
751 TimePos cur_start = _start;
752 if( _clip_num < 0 )
753 {
754 cur_start -= c->startPosition() + c->startTimeOffset();
755 }
756
757 // get all notes from the given clip...
758 const NoteVector & notes = c->notes();
759 // ...and set our index to zero
760 auto nit = notes.begin();
761
762 // very effective algorithm for playing notes that are
763 // posated within the current sample-frame
764
765 if( cur_start > 0 )
766 {
767 // skip notes which end before start-bar
768 while( nit != notes.end() && ( *nit )->endPos() < cur_start )
769 {
770 ++nit;
771 }
772 }
773
774 while (nit != notes.end() && (*nit)->pos() < c->length() - c->startTimeOffset())
775 {
776 const auto currentNote = *nit;
777 // Skip any notes note at the current time pos or not overlapping with the start.
778 if (!(currentNote->pos() == cur_start
779 || (cur_start == -c->startTimeOffset() && (*nit)->pos() < cur_start && (*nit)->endPos() > cur_start)))
780 {
781 ++nit;
782 continue;
783 }
784
785 // Calculate the overlap of the note over the clip end.
786 const auto noteOverlap = std::max(0, currentNote->endPos() - (c->length() - c->startTimeOffset()));
787 // If the note is a Step Note, frames will be 0 so the NotePlayHandle
788 // plays for the whole length of the sample
789 const auto noteFrames = currentNote->type() == Note::Type::Step
790 ? 0
791 : (currentNote->endPos() - cur_start - noteOverlap) * frames_per_tick;
792
793 NotePlayHandle* notePlayHandle = NotePlayHandleManager::acquire(this, _offset, noteFrames, *currentNote);
794 notePlayHandle->setPatternTrack(pattern_track);
795 // are we playing global song?
796 if( _clip_num < 0 )
797 {
798 // then set song-global offset of clip in order to
799 // properly perform the note detuning
800 notePlayHandle->setSongGlobalParentOffset( c->startPosition() + c->startTimeOffset());
801 }
802
803 Engine::audioEngine()->addPlayHandle( notePlayHandle );
804 played_a_note = true;
805 ++nit;
806 }
807 }
808 unlock();
809 return played_a_note;
810}

833void InstrumentTrack::saveTrackSpecificSettings(QDomDocument& doc, QDomElement& thisElement, bool presetMode)
834{
835 m_volumeModel.saveSettings( doc, thisElement, "vol" );
836 m_panningModel.saveSettings( doc, thisElement, "pan" );
837 m_pitchModel.saveSettings( doc, thisElement, "pitch" );
838 m_pitchRangeModel.saveSettings( doc, thisElement, "pitchrange" );
839
840 m_mixerChannelModel.saveSettings( doc, thisElement, "mixch" );
841 m_baseNoteModel.saveSettings( doc, thisElement, "basenote" );
842 m_firstKeyModel.saveSettings(doc, thisElement, "firstkey");
843 m_lastKeyModel.saveSettings(doc, thisElement, "lastkey");
844 m_useMasterPitchModel.saveSettings( doc, thisElement, "usemasterpitch");
845 m_microtuner.saveSettings(doc, thisElement);
846
847 // Save MIDI CC stuff
848 m_midiCCEnable->saveSettings(doc, thisElement, "enablecc");
849 QDomElement midiCC = doc.createElement("midicontrollers");
850 thisElement.appendChild(midiCC);
851 for (int i = 0; i < MidiControllerCount; ++i)
852 {
853 m_midiCCModel[i]->saveSettings(doc, midiCC, "cc" + QString::number(i));
854 }
855
856 if( m_instrument != nullptr )
857 {
858 QDomElement i = doc.createElement( "instrument" );
859 i.setAttribute( "name", m_instrument->descriptor()->name );
860 QDomElement ins = m_instrument->saveState( doc, i );
861 if(m_instrument->key().isValid()) {
862 ins.appendChild( m_instrument->key().saveXML( doc ) );
863 }
864 thisElement.appendChild( i );
865 }
866 m_soundShaping.saveState( doc, thisElement );
867 m_noteStacking.saveState( doc, thisElement );
868 m_arpeggio.saveState( doc, thisElement );
869
870 // Save the midi port info if we are not in song saving mode, e.g. in
871 // track cloning mode or if we are in song saving mode and the user
872 // has chosen not to discard the MIDI connections.
873 if (!Engine::getSong()->isSavingProject() ||
874 !Engine::getSong()->getSaveOptions().discardMIDIConnections.value())
875 {
876 // Don't save auto assigned midi device connection
877 bool hasAuto = m_hasAutoMidiDev;
878 autoAssignMidiDevice(false);
879
880 // Only save the MIDI port information if we are not saving a preset.
881 if (!presetMode)
882 {
883 m_midiPort.saveState(doc, thisElement);
884 }
885
886 autoAssignMidiDevice(hasAuto);
887 }
888
889 m_audioBusHandle.effects()->saveState(doc, thisElement);
890}

895void InstrumentTrack::loadTrackSpecificSettings( const QDomElement & thisElement )
896{
897 // don't delete instrument in preview mode if it's the same
898 // we can't do this for other situations due to some issues with linked models
899 bool reuseInstrument = m_previewMode && m_instrument && m_instrument->nodeName() == getSavedInstrumentName(thisElement);
900 // remove the InstrumentPlayHandle if and only if we need to delete the instrument
901 silenceAllNotes(!reuseInstrument);
902
903 lock();
904
905 m_volumeModel.loadSettings( thisElement, "vol" );
906 m_panningModel.loadSettings( thisElement, "pan" );
907 m_pitchRangeModel.loadSettings( thisElement, "pitchrange" );
908 m_pitchModel.loadSettings( thisElement, "pitch" );
909 m_mixerChannelModel.setRange( 0, Engine::mixer()->numChannels()-1 );
910 if ( !m_previewMode )
911 {
912 m_mixerChannelModel.loadSettings( thisElement, "mixch" );
913 }
914 m_baseNoteModel.loadSettings( thisElement, "basenote" );
915 m_firstKeyModel.loadSettings(thisElement, "firstkey");
916 m_lastKeyModel.loadSettings(thisElement, "lastkey");
917 m_useMasterPitchModel.loadSettings( thisElement, "usemasterpitch");
918 m_microtuner.loadSettings(thisElement);
919
920 // clear effect-chain just in case we load an old preset without FX-data
921 m_audioBusHandle.effects()->clear();
922
923 // We set MIDI CC enable to false so the knobs don't trigger MIDI CC events while
924 // they are being loaded. After all knobs are loaded we load the right value of m_midiCCEnable.
925 m_midiCCEnable->setValue(false);
926
927 QDomNode node = thisElement.firstChild();
928 while( !node.isNull() )
929 {
930 if( node.isElement() )
931 {
932 if( m_soundShaping.nodeName() == node.nodeName() )
933 {
934 m_soundShaping.restoreState( node.toElement() );
935 }
936 else if( m_noteStacking.nodeName() == node.nodeName() )
937 {
938 m_noteStacking.restoreState( node.toElement() );
939 }
940 else if( m_arpeggio.nodeName() == node.nodeName() )
941 {
942 m_arpeggio.restoreState( node.toElement() );
943 }
944 else if( m_midiPort.nodeName() == node.nodeName() )
945 {
946 m_midiPort.restoreState( node.toElement() );
947 }
948 else if (m_audioBusHandle.effects()->nodeName() == node.nodeName())
949 {
950 m_audioBusHandle.effects()->restoreState(node.toElement());
951 }
952 else if(node.nodeName() == "instrument")
953 {
954 using PluginKey = Plugin::Descriptor::SubPluginFeatures::Key;
955 PluginKey key(node.toElement().elementsByTagName("key").item(0).toElement());
956
957 if (reuseInstrument)
958 {
959 m_instrument->restoreState(node.firstChildElement());
960 }
961 else
962 {
963 delete m_instrument;
964 m_instrument = nullptr;
965 m_instrument = Instrument::instantiate(
966 node.toElement().attribute("name"), this, &key);
967 m_instrument->restoreState(node.firstChildElement());
968 emit instrumentChanged();
969 }
970 }
971 else if (node.nodeName() == "midicontrollers")
972 {
973 for (int i = 0; i < MidiControllerCount; ++i)
974 {
975 m_midiCCModel[i]->loadSettings(node.toElement(), "cc" + QString::number(i));
976 }
977 }
978 // compat code - if node-name doesn't match any known
979 // one, we assume that it is an instrument-plugin
980 // which we'll try to load
981 else if(AutomationClip::classNodeName() != node.nodeName() &&
982 ControllerConnection::classNodeName() != node.nodeName() &&
983 !node.toElement().hasAttribute( "id" ))
984 {
985 delete m_instrument;
986 m_instrument = nullptr;
987 m_instrument = Instrument::instantiate(
988 node.nodeName(), this, nullptr, true);
989 if (m_instrument->nodeName() == node.nodeName())
990 {
991 m_instrument->restoreState(node.toElement());
992 }
993 emit instrumentChanged();
994 }
995 }
996 node = node.nextSibling();
997 }
998
999 // Load the right value of m_midiCCEnable
1000 m_midiCCEnable->loadSettings(thisElement, "enablecc");
1001
1002 updatePitchRange();
1003 unlock();
1004}
