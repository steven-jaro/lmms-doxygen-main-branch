38SideBarWidget::SideBarWidget( const QString & _title, const QPixmap & _icon,
39 QWidget * _parent ) :
40 QWidget( _parent ),
41 m_title( _title ),
42 m_icon(_icon),
43 m_buttonSize(17, 17)
44{
45 m_contents = new QWidget( this );
46 m_layout = new QVBoxLayout( m_contents );
47 m_layout->setSpacing( 5 );
48 m_layout->setContentsMargins(0, 0, 0, 0);
49 m_closeBtn = new QPushButton(embed::getIconPixmap("close"), QString(), this);
50 m_closeBtn->resize(m_buttonSize);
51 m_closeBtn->setToolTip(tr("Close"));
52 connect(m_closeBtn, &QPushButton::clicked,
53 [this]() { this->closeButtonClicked(); });
54}

59void SideBarWidget::paintEvent( QPaintEvent * )
60{
61 QPainter p( this );
62 p.fillRect( 0, 0, width(), 27, palette().highlight().color() );
63
64 QFont f = p.font();
65 f.setBold( true );
66 f.setUnderline(false);
67 p.setFont(adjustedToPixelSize(f, LARGE_FONT_SIZE));
68
69 p.setPen( palette().highlightedText().color() );
70
71 const int tx = m_icon.width() + 8;
72
73 QFontMetrics metrics( f );
74 const int ty = (metrics.ascent() + m_icon.height()) / 2;
75 p.drawText( tx, ty, m_title );
76
77 p.drawPixmap( 2, 2, m_icon.transformed( QTransform().rotate( -90 ) ) );
78}
