42#ifdef LMMS_BUILD_APPLE
43Semaphore::Semaphore(unsigned val)
44{
45 kern_return_t rval = semaphore_create(mach_task_self(), &m_sem, SYNC_POLICY_FIFO, val);
46 if(rval != 0) {
47 throw std::system_error(rval, std::system_category(), "Could not create semaphore");
48 }
49}

61void Semaphore::wait()
62{
63 kern_return_t rval = semaphore_wait(m_sem);
64 if (rval != KERN_SUCCESS) {
65 throw std::system_error(rval, std::system_category(), "Waiting for semaphore failed");
66 }
67}

69bool Semaphore::tryWait()

94void Semaphore::wait()
95{
96 if (WaitForSingleObject(m_sem, INFINITE) != WAIT_OBJECT_0) {
97 throw std::system_error(GetLastError(), std::system_category(), "Waiting for semaphore failed");
98 }
99}

101bool Semaphore::tryWait()
102{

125void Semaphore::wait()
126{
127 while (sem_wait(&m_sem) != 0) {
128 if (errno != EINTR) {
129 throw std::system_error(errno, std::generic_category(), "Waiting for semaphore failed");
130 }
131 /* Otherwise, interrupted, so try again. */
132 }
133}
134
135bool Semaphore::tryWait()
136{
137 return (sem_trywait(&m_sem) == 0);
138}
