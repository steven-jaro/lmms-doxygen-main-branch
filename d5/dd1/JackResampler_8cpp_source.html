58unsigned int JackRingBuffer::Read(jack_default_audio_sample_t* buffer, unsigned int frames)
59{
60 size_t len = jack_ringbuffer_read_space(fRingBuffer);
61 jack_log("JackRingBuffer::Read input available = %ld", len / sizeof(jack_default_audio_sample_t));
62
63 if (len < frames * sizeof(jack_default_audio_sample_t)) {
64 jack_error("JackRingBuffer::Read : producer too slow, missing frames = %d", frames);
65 return 0;
66 } else {
67 jack_ringbuffer_read(fRingBuffer, (char*)buffer, frames * sizeof(jack_default_audio_sample_t));
68 return frames;
69 }
70}
