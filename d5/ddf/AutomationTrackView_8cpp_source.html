58void AutomationTrackView::dropEvent( QDropEvent * _de )
59{
60 QString type = StringPairDrag::decodeKey( _de );
61 QString val = StringPairDrag::decodeValue( _de );
62 if( type == "automatable_model" )
63 {
64 auto mod = dynamic_cast<AutomatableModel*>(Engine::projectJournal()->journallingObject(val.toInt()));
65 if( mod != nullptr )
66 {
67 TimePos pos = TimePos( trackContainerView()->
68 currentPosition() +
69 ( _de->pos().x() -
70 getTrackContentWidget()->x() ) *
71 TimePos::ticksPerBar() /
72 static_cast<int>( trackContainerView()->pixelsPerBar() ) )
73 .toAbsoluteBar();
74
75 if( pos.getTicks() < 0 )
76 {
77 pos.setTicks( 0 );
78 }
79
80 Clip * clip = getTrack()->createClip( pos );
81 auto autoClip = dynamic_cast<AutomationClip*>(clip);
82 autoClip->addObject( mod );
83 }
84 }
85
86 update();
87}
