78ProjectRenderer::ProjectRenderer(
79 const OutputSettings& outputSettings, ExportFileFormat exportFileFormat, const QString& outputFilename)
80 : QThread(Engine::audioEngine())
81 , m_fileDev(nullptr)
82 , m_progress(0)
83 , m_abort(false)
84{
85 AudioFileDeviceInstantiaton audioEncoderFactory = fileEncodeDevices[static_cast<std::size_t>(exportFileFormat)].m_getDevInst;
86
87 if (audioEncoderFactory)
88 {
89 bool successful = false;
90
91 m_fileDev = audioEncoderFactory(
92 outputFilename, outputSettings, DEFAULT_CHANNELS,
93 Engine::audioEngine(), successful );
94 if( !successful )
95 {
96 delete m_fileDev;
97 m_fileDev = nullptr;
98 }
99 }
100}

105// Little help function for getting file format from a file extension
106// (only for registered file-encoders).
107ProjectRenderer::ExportFileFormat ProjectRenderer::getFileFormatFromExtension(
108 const QString & _ext )
109{
110 int idx = 0;
111 while( fileEncodeDevices[idx].m_fileFormat != ExportFileFormat::Count )
112 {
113 if( QString( fileEncodeDevices[idx].m_extension ) == _ext )
114 {
115 return( fileEncodeDevices[idx].m_fileFormat );
116 }
117 ++idx;
118 }
119
120 return( ExportFileFormat::Wave ); // Default.
121}

154void ProjectRenderer::run()
155{
156 PerfLogTimer perfLog("Project Render");
157
158 Engine::getSong()->startExport();
159 // Skip first empty buffer.
160 Engine::audioEngine()->nextBuffer();
161
162 m_progress = 0;
163
164 // Now start processing
165 Engine::audioEngine()->startProcessing(false);
166
167 // Continually track and emit progress percentage to listeners.
168 while (!Engine::getSong()->isExportDone() && !m_abort)
169 {
170 m_fileDev->processNextBuffer();
171 const int nprog = Engine::getSong()->getExportProgress();
172 if (m_progress != nprog)
173 {
174 m_progress = nprog;
175 emit progressChanged( m_progress );
176 }
177 }
178
179 // Notify the audio engine of the end of processing.
180 Engine::audioEngine()->stopProcessing();
181
182 Engine::getSong()->stopExport();
183
184 perfLog.end();
185
186 // If the user aborted export-process, the file has to be deleted.
187 const QString f = m_fileDev->outputFile();
188 if( m_abort )
189 {
190 QFile( f ).remove();
191 }
192}

205void ProjectRenderer::updateConsoleProgress()
206{
207 constexpr int cols = 50;
208 static int rot = 0;
209 auto buf = std::array<char, 80>{};
210 auto prog = std::array<char, cols + 1>{};
211
212 for( int i = 0; i < cols; ++i )
213 {
214 prog[i] = ( i*100/cols <= m_progress ? '-' : ' ' );
215 }
216 prog[cols] = 0;
217
218 const auto activity = "|/-\\";
219 std::fill(buf.begin(), buf.end(), 0);
220 std::snprintf(buf.data(), buf.size(), "\r|%s| %3d%% %c ", prog.data(), m_progress,
221 activity[rot] );
222 rot = ( rot+1 ) % 4;
223
224 fprintf( stderr, "%s", buf.data() );
225 fflush( stderr );
226}
