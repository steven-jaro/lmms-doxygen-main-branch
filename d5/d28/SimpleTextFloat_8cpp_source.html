38SimpleTextFloat::SimpleTextFloat() :
39 QWidget(getGUI()->mainWindow(), Qt::ToolTip)
40{
41 QHBoxLayout * layout = new QHBoxLayout(this);
42 layout->setMargin(3);
43 setLayout(layout);
44
45 m_textLabel = new QLabel(this);
46 layout->addWidget(m_textLabel);
47
48 m_showTimer = new QTimer(this);
49 m_showTimer->setSingleShot(true);
50 QObject::connect(m_showTimer, &QTimer::timeout, this, &SimpleTextFloat::show);
51
52 m_hideTimer = new QTimer(this);

55}

62void SimpleTextFloat::showWithDelay(int msecBeforeDisplay, int msecDisplayTime)
63{
64 if (msecBeforeDisplay > 0)
65 {
66 m_showTimer->start(msecBeforeDisplay);
67 }
68 else
69 {
70 show();
71 }
72
73 if (msecDisplayTime > 0)
74 {
75 m_hideTimer->start(msecBeforeDisplay + msecDisplayTime);

77}
