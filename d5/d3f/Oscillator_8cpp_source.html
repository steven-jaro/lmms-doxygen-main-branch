80void Oscillator::update(SampleFrame* ab, const fpp_t frames, const ch_cnt_t chnl, bool modulator)
81{
82 if (m_freq >= Engine::audioEngine()->outputSampleRate() / 2)
83 {
84 zeroSampleFrames(ab, frames);
85 return;
86 }
87 // If this oscillator is used to PM or PF modulate another oscillator, take a note.
88 // The sampling functions will check this variable and avoid using band-limited
89 // wavetables, since they contain ringing that would lead to unexpected results.
90 m_isModulator = modulator;
91 if (m_subOsc != nullptr)
92 {
93 switch (static_cast<ModulationAlgo>(m_modulationAlgoModel->value()))
94 {
95 case ModulationAlgo::PhaseModulation:
96 updatePM(ab, frames, chnl);
97 break;
98 case ModulationAlgo::AmplitudeModulation:
99 updateAM(ab, frames, chnl);
100 break;
101 case ModulationAlgo::SignalMix:
102 default:
103 updateMix(ab, frames, chnl);
104 break;
105 case ModulationAlgo::SynchronizedBySubOsc:
106 updateSync(ab, frames, chnl);
107 break;
108 case ModulationAlgo::FrequencyModulation:
109 updateFM(ab, frames, chnl);
110 }
111 }
112 else
113 {
114 updateNoSub(ab, frames, chnl);
115 }
116}

119void Oscillator::generateSawWaveTable(int bands, sample_t* table, int firstBand)
120{
121 using namespace std::numbers;
122 // sawtooth wave contain both even and odd harmonics
123 // hence sinewaves are added for all bands
124 // https://en.wikipedia.org/wiki/Sawtooth_wave
125 for (int i = 0; i < OscillatorConstants::WAVETABLE_LENGTH; i++)
126 {
127 // add offset to the position index to match phase of the non-wavetable saw wave; precompute "/ period"
128 const float imod = (i - OscillatorConstants::WAVETABLE_LENGTH / 2.f) / OscillatorConstants::WAVETABLE_LENGTH;
129 for (int n = firstBand; n <= bands; n++)
130 {
131 table[i] += (n % 2 ? 1.0f : -1.0f) / n * std::sin(2 * pi_v<float> * n * imod) / (pi_v<float> * 0.5f);
132 }
133 }
134}

137void Oscillator::generateTriangleWaveTable(int bands, sample_t* table, int firstBand)
138{
139 using namespace std::numbers;
140 constexpr float pi_sqr = pi_v<float> * pi_v<float>;
141 // triangle waves contain only odd harmonics
142 // hence sinewaves are added for alternate bands
143 // https://en.wikipedia.org/wiki/Triangle_wave
144 for (int i = 0; i < OscillatorConstants::WAVETABLE_LENGTH; i++)
145 {
146 for (int n = firstBand | 1; n <= bands; n += 2)
147 {
148 table[i] += (n & 2 ? -1.0f : 1.0f) / (n * n)
149 * std::sin(2 * pi_v<float> * n * i / (float)OscillatorConstants::WAVETABLE_LENGTH) / (pi_sqr / 8.f);
150 }
151 }
152}

154
155void Oscillator::generateSquareWaveTable(int bands, sample_t* table, int firstBand)
156{
157 using namespace std::numbers;
158 // square waves only contain odd harmonics,
159 // at different levels when compared to triangle waves
160 // https://en.wikipedia.org/wiki/Square_wave
161 for (int i = 0; i < OscillatorConstants::WAVETABLE_LENGTH; i++)
162 {
163 for (int n = firstBand | 1; n <= bands; n += 2)
164 {
165 table[i] += (1.0f / n)
166 * std::sin(2 * pi_v<float> * i * n / OscillatorConstants::WAVETABLE_LENGTH)
167 / (pi_v<float> / 4.f);
168 }
169 }

174// Expects waveform converted to frequency domain to be present in the spectrum buffer
175void Oscillator::generateFromFFT(int bands, sample_t* table)
176{
177 // Keep only specified number of bands, set the rest to zero.
178 // Add a +1 offset to the requested number of bands, since the first "useful" frequency falls into bin 1.
179 // I.e., for bands = 1, keeping just bin 0 (center 0 Hz, +- 4 Hz) makes no sense, it would not produce any tone.
180 for (int i = bands + 1; i < OscillatorConstants::WAVETABLE_LENGTH * 2 - bands; i++)
181 {
182 s_specBuf[i][0] = 0.0f;
183 s_specBuf[i][1] = 0.0f;
184 }
185 //ifft
186 fftwf_execute(s_ifftPlan);
187 //normalize and copy to result buffer
188 normalize(s_sampleBuffer.data(), table, OscillatorConstants::WAVETABLE_LENGTH, 2*OscillatorConstants::WAVETABLE_LENGTH + 1);
189}

191std::unique_ptr<OscillatorConstants::waveform_t> Oscillator::generateAntiAliasUserWaveTable(const SampleBuffer* sampleBuffer)
192{
193 auto userAntiAliasWaveTable = std::make_unique<OscillatorConstants::waveform_t>();
194 for (int i = 0; i < OscillatorConstants::WAVE_TABLES_PER_WAVEFORM_COUNT; ++i)
195 {
196 // TODO: This loop seems to be doing the same thing for each iteration of the outer loop,
197 // and could probably be moved out of it
198 for (int j = 0; j < OscillatorConstants::WAVETABLE_LENGTH; ++j)
199 {
200 s_sampleBuffer[j] = Oscillator::userWaveSample(
201 sampleBuffer, static_cast<float>(j) / OscillatorConstants::WAVETABLE_LENGTH);
202 }
203 fftwf_execute(s_fftPlan);
204 Oscillator::generateFromFFT(OscillatorConstants::MAX_FREQ / freqFromWaveTableBand(i), (*userAntiAliasWaveTable)[i].data());
205 }
206
207 return userAntiAliasWaveTable;
208}

223void Oscillator::createFFTPlans()
224{
225 Oscillator::s_specBuf = ( fftwf_complex * ) fftwf_malloc( ( OscillatorConstants::WAVETABLE_LENGTH * 2 + 1 ) * sizeof( fftwf_complex ) );
226 Oscillator::s_fftPlan = fftwf_plan_dft_r2c_1d(OscillatorConstants::WAVETABLE_LENGTH, s_sampleBuffer.data(), s_specBuf, FFTW_MEASURE );
227 Oscillator::s_ifftPlan = fftwf_plan_dft_c2r_1d(OscillatorConstants::WAVETABLE_LENGTH, s_specBuf, s_sampleBuffer.data(), FFTW_MEASURE);
228 // initialize s_specBuf content to zero, since the values are used in a condition inside generateFromFFT()
229 for (int i = 0; i < OscillatorConstants::WAVETABLE_LENGTH * 2 + 1; i++)
230 {
231 s_specBuf[i][0] = 0.0f;
232 s_specBuf[i][1] = 0.0f;
233 }
234}

243void Oscillator::generateWaveTables()
244{
245 // Generate tables for simple shaped (constructed by summing sine waves).
246 // Start from the table that contains the least number of bands, and re-use each table in the following
247 // iteration, adding more bands in each step and avoiding repeated computation of earlier bands.
248 using generator_t = void (*)(int, sample_t*, int);
249 auto simpleGen = [](WaveShape shape, generator_t generator)
250 {
251 const int shapeID = static_cast<std::size_t>(shape) - FirstWaveShapeTable;
252 int lastBands = 0;
253
254 // Clear the first wave table
255 std::fill(
256 std::begin(s_waveTables[shapeID][OscillatorConstants::WAVE_TABLES_PER_WAVEFORM_COUNT - 1]),
257 std::end(s_waveTables[shapeID][OscillatorConstants::WAVE_TABLES_PER_WAVEFORM_COUNT - 1]),
258 0.f);
259
260 for (int i = OscillatorConstants::WAVE_TABLES_PER_WAVEFORM_COUNT - 1; i >= 0; i--)
261 {
262 const int bands = OscillatorConstants::MAX_FREQ / freqFromWaveTableBand(i);
263 generator(bands, s_waveTables[shapeID][i], lastBands + 1);
264 lastBands = bands;
265 if (i)
266 {
267 std::copy(
268 s_waveTables[shapeID][i],
269 s_waveTables[shapeID][i] + OscillatorConstants::WAVETABLE_LENGTH,
270 s_waveTables[shapeID][i - 1]);
271 }
272 }
273 };
274
275 // FFT-based wave shapes: make standard wave table without band limit, convert to frequency domain, remove bands
276 // above maximum frequency and convert back to time domain.
277 auto fftGen = []()
278 {
279 // Generate moogSaw tables
280 for (int i = 0; i < OscillatorConstants::WAVE_TABLES_PER_WAVEFORM_COUNT; ++i)
281 {
282 for (int i = 0; i < OscillatorConstants::WAVETABLE_LENGTH; ++i)
283 {
284 Oscillator::s_sampleBuffer[i] = moogSawSample((float)i / (float)OscillatorConstants::WAVETABLE_LENGTH);
285 }
286 fftwf_execute(s_fftPlan);
287 generateFromFFT(OscillatorConstants::MAX_FREQ / freqFromWaveTableBand(i), s_waveTables[static_cast<std::size_t>(WaveShape::MoogSaw) - FirstWaveShapeTable][i]);
288 }
289
290 // Generate exponential tables
291 for (int i = 0; i < OscillatorConstants::WAVE_TABLES_PER_WAVEFORM_COUNT; ++i)
292 {
293 for (int i = 0; i < OscillatorConstants::WAVETABLE_LENGTH; ++i)
294 {
295 s_sampleBuffer[i] = expSample((float)i / (float)OscillatorConstants::WAVETABLE_LENGTH);
296 }
297 fftwf_execute(s_fftPlan);
298 generateFromFFT(OscillatorConstants::MAX_FREQ / freqFromWaveTableBand(i), s_waveTables[static_cast<std::size_t>(WaveShape::Exponential) - FirstWaveShapeTable][i]);
299 }
300 };
301
302// TODO: Mingw compilers currently do not support std::thread. There are some 3rd-party workarounds available,
303// but since threading is not essential in this case, it is easier and more reliable to simply generate
304// the wavetables serially. Remove the the check and #else branch once std::thread is well supported.
305#if !defined(__MINGW32__) && !defined(__MINGW64__)
306 std::thread sawThread(simpleGen, WaveShape::Saw, generateSawWaveTable);
307 std::thread squareThread(simpleGen, WaveShape::Square, generateSquareWaveTable);
308 std::thread triangleThread(simpleGen, WaveShape::Triangle, generateTriangleWaveTable);
309 std::thread fftThread(fftGen);
310 sawThread.join();
311 squareThread.join();
312 triangleThread.join();
313 fftThread.join();
314#else
315 simpleGen(WaveShape::Saw, generateSawWaveTable);
316 simpleGen(WaveShape::Square, generateSquareWaveTable);
317 simpleGen(WaveShape::Triangle, generateTriangleWaveTable);
318 fftGen();
319#endif
320}

541// should be called every time phase-offset is changed...
542inline void Oscillator::recalcPhase()
543{
544 if (!approximatelyEqual(m_phaseOffset, m_ext_phaseOffset))
545 {
546 m_phase -= m_phaseOffset;
547 m_phaseOffset = m_ext_phaseOffset;
548 m_phase += m_phaseOffset;
549 }
550 m_phase = absFraction( m_phase );
551}

556inline bool Oscillator::syncOk( float _osc_coeff )
557{
558 const float v1 = m_phase;
559 m_phase += _osc_coeff;
560 // check whether m_phase is in next period
561 return( floorf( m_phase ) > floorf( v1 ) );
562}

658// sync with sub-osc (every time sub-osc starts new period, we also start new
659// period)
660template<Oscillator::WaveShape W>
661void Oscillator::updateSync( SampleFrame* _ab, const fpp_t _frames,
662 const ch_cnt_t _chnl )
663{
664 const float sub_osc_coeff = m_subOsc->syncInit( _ab, _frames, _chnl );
665 recalcPhase();
666 const float osc_coeff = m_freq * m_detuning_div_samplerate;
667
668 for( fpp_t frame = 0; frame < _frames ; ++frame )
669 {
670 if( m_subOsc->syncOk( sub_osc_coeff ) )
671 {
672 m_phase = m_phaseOffset;
673 }

683template<Oscillator::WaveShape W>
684void Oscillator::updateFM( SampleFrame* _ab, const fpp_t _frames,
685 const ch_cnt_t _chnl )
686{
687 m_subOsc->update( _ab, _frames, _chnl, true );
688 recalcPhase();
689 const float osc_coeff = m_freq * m_detuning_div_samplerate;
690 const float sampleRateCorrection = 44100.0f / Engine::audioEngine()->outputSampleRate();
691
692 for( fpp_t frame = 0; frame < _frames; ++frame )
693 {
694 m_phase += _ab[frame][_chnl] * sampleRateCorrection;
695 _ab[frame][_chnl] = getSample<W>( m_phase ) * m_volume;
696 m_phase += osc_coeff;
697 }
698}
