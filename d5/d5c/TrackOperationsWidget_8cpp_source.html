55/*! \brief Create a new trackOperationsWidget
56 *
57 * The trackOperationsWidget is the grip and the mute button of a track.
58 *
59 * \param parent the trackView to contain this widget
60 */
61TrackOperationsWidget::TrackOperationsWidget( TrackView * parent ) :
62 QWidget( parent ), /*!< The parent widget */
63 m_trackView( parent ) /*!< The parent track view */
64{
65 setToolTip(tr("Press <%1> while clicking on move-grip "
66 "to begin a new drag'n'drop action." ).arg(UI_CTRL_KEY) );
67
68 auto toMenu = new QMenu(this);
69 connect( toMenu, SIGNAL(aboutToShow()), this, SLOT(updateMenu()));
70
71
72 setObjectName( "automationEnabled" );
73
74 auto layout = new QHBoxLayout(this);
75 layout->setContentsMargins(0, 0, 0, 0);
76 layout->setSpacing(0);
77 layout->setAlignment(Qt::AlignTop);
78
79 m_trackGrip = new TrackGrip(m_trackView->getTrack(), this);
80 layout->addWidget(m_trackGrip);
81
82 // This widget holds the gear icon and the mute and solo
83 // buttons in a layout.
84 auto operationsWidget = new QWidget(this);
85 auto operationsLayout = new QHBoxLayout(operationsWidget);
86 operationsLayout->setContentsMargins(2, 6, 0, 6);
87 operationsLayout->setSpacing(2);
88
89 m_trackOps = new QPushButton(operationsWidget);
90 m_trackOps->setFocusPolicy( Qt::NoFocus );
91 m_trackOps->setMenu( toMenu );
92 m_trackOps->setToolTip(tr("Actions"));
93 m_trackOps->setCursor(Qt::PointingHandCursor);

95 m_muteBtn = new AutomatableButton(operationsWidget, tr("Mute"));
96 m_muteBtn->setCheckable(true);
97 m_muteBtn->setToolTip(tr("Mute"));

100 m_soloBtn->setCheckable(true);
101 m_soloBtn->setToolTip(tr("Solo"));
102 m_soloBtn->setObjectName("btn-solo");
103
104 operationsLayout->addWidget(m_trackOps);
105 operationsLayout->addWidget(m_muteBtn);
106 operationsLayout->addWidget(m_soloBtn);
107
108 layout->addWidget(operationsWidget, 0, Qt::AlignTop | Qt::AlignLeading);
109
110 connect( this, SIGNAL(trackRemovalScheduled(lmms::gui::TrackView*)),
111 m_trackView->trackContainerView(),
112 SLOT(deleteTrackView(lmms::gui::TrackView*)),
113 Qt::QueuedConnection );
114
115 connect( m_trackView->getTrack()->getMutedModel(), SIGNAL(dataChanged()),
116 this, SLOT(update()));
117
118 connect(m_trackView->getTrack(), SIGNAL(colorChanged()), this, SLOT(update()));
119}

122/*! \brief Respond to trackOperationsWidget mouse events
123 *
124 * If it's the left mouse button, and Ctrl is held down, and we're
125 * not a Pattern Editor track, then start a new drag event to
126 * copy this track.
127 *
128 * Otherwise, ignore all other events.
129 *
130 * \param me The mouse event to respond to.
131 */
132void TrackOperationsWidget::mousePressEvent( QMouseEvent * me )
133{
134 if (me->button() == Qt::LeftButton && me->modifiers() & KBD_COPY_MODIFIER &&
135 m_trackView->getTrack()->type() != Track::Type::Pattern)
136 {
137 DataFile dataFile( DataFile::Type::DragNDropData );
138 m_trackView->getTrack()->saveState( dataFile, dataFile.content() );
139 new StringPairDrag( QString( "track_%1" ).arg(
140 static_cast<int>(m_trackView->getTrack()->type()) ),
141 dataFile.toString(), m_trackView->getTrackSettingsWidget()->grab(),
142 this );
143 }
144 else if( me->button() == Qt::LeftButton )
145 {
146 // track-widget (parent-widget) initiates track-move
147 me->ignore();
148 }
149}

166/*! \brief Show a message box warning the user that this track is about to be closed */
167bool TrackOperationsWidget::confirmRemoval()
168{
169 bool needConfirm = ConfigManager::inst()->value("ui", "trackdeletionwarning", "1").toInt();
170 if (!needConfirm){ return true; }
171
172 QString messageRemoveTrack = tr("After removing a track, it can not "
173 "be recovered. Are you sure you want to remove track \"%1\"?")
174 .arg(m_trackView->getTrack()->name());
175 QString messageTitleRemoveTrack = tr("Confirm removal");
176 QString askAgainText = tr("Don't ask again");
177 auto askAgainCheckBox = new QCheckBox(askAgainText, nullptr);
178 connect(askAgainCheckBox, &QCheckBox::stateChanged, [](int state){
179 // Invert button state, if it's checked we *shouldn't* ask again
180 ConfigManager::inst()->setValue("ui", "trackdeletionwarning", state ? "0" : "1");
181 });
182
183 QMessageBox mb;
184 mb.setText(messageRemoveTrack);
185 mb.setWindowTitle(messageTitleRemoveTrack);
186 mb.setIcon(QMessageBox::Warning);
187 mb.addButton(QMessageBox::Cancel);
188 mb.addButton(QMessageBox::Ok);
189 mb.setCheckBox(askAgainCheckBox);
190 mb.setDefaultButton(QMessageBox::Cancel);
191
192 return mb.exec() == QMessageBox::Ok;
193}

195/*! \brief Clone this track
196 *
197 */
198void TrackOperationsWidget::cloneTrack()
199{
200 TrackContainerView *tcView = m_trackView->trackContainerView();
201
202 Track *newTrack = m_trackView->getTrack()->clone();
203 TrackView *newTrackView = tcView->createTrackView( newTrack );
204
205 int index = tcView->trackViews().indexOf( m_trackView );
206 int i = tcView->trackViews().size();
207 while ( i != index + 1 )
208 {
209 tcView->moveTrackView( newTrackView, i - 1 );
210 i--;
211 }
212
213 if (m_soloBtn->model()->value())
214 {
215 // if this track was solo, make the new track the new solo
216 newTrack->toggleSolo();
217 }
218}

243void TrackOperationsWidget::selectTrackColor()
244{
245 const auto newColor = ColorChooser{this}
246 .withPalette(ColorChooser::Palette::Track)
247 ->getColor(m_trackView->getTrack()->color().value_or(Qt::white));
248
249 if (!newColor.isValid()) { return; }
250
251 const auto track = m_trackView->getTrack();
252 track->addJournalCheckPoint();
253 track->setColor(newColor);
254 Engine::getSong()->setModified();
255}

286/*! \brief Update the trackOperationsWidget context menu
287 *
288 * For all track types, we have the Clone and Remove options.
289 * For instrument-tracks we also offer the MIDI-control-menu
290 * For automation tracks, extra options: turn on/off recording
291 * on all Clips (same should be added for sample tracks when
292 * sampletrack recording is implemented)
293 */
294void TrackOperationsWidget::updateMenu()
295{
296 QMenu * toMenu = m_trackOps->menu();
297 toMenu->clear();
298 toMenu->addAction( embed::getIconPixmap( "edit_copy", 16, 16 ),
299 tr( "Clone this track" ),
300 this, SLOT(cloneTrack()));
301 toMenu->addAction( embed::getIconPixmap( "cancel", 16, 16 ),
302 tr( "Remove this track" ),
303 this, SLOT(removeTrack()));
304
305 if( ! m_trackView->trackContainerView()->fixedClips() )
306 {
307 toMenu->addAction( tr( "Clear this track" ), this, SLOT(clearTrack()));
308 }
309 if (QMenu *mixerMenu = m_trackView->createMixerMenu(tr("Channel %1: %2"), tr("Assign to new Mixer Channel")))
310 {
311 toMenu->addMenu(mixerMenu);
312 }
313
314 if (auto trackView = dynamic_cast<InstrumentTrackView*>(m_trackView))
315 {
316 toMenu->addSeparator();
317 toMenu->addMenu(trackView->midiMenu());
318 }
319 if( dynamic_cast<AutomationTrackView *>( m_trackView ) )
320 {
321 toMenu->addAction( tr( "Turn all recording on" ), this, SLOT(recordingOn()));
322 toMenu->addAction( tr( "Turn all recording off" ), this, SLOT(recordingOff()));
323 }
324
325 toMenu->addSeparator();
326
327 QMenu* colorMenu = toMenu->addMenu(tr("Track color"));
328 colorMenu->setIcon(embed::getIconPixmap("colorize"));
329 colorMenu->addAction(tr("Change"), this, SLOT(selectTrackColor()));
330 colorMenu->addAction(tr("Reset"), this, SLOT(resetTrackColor()));
331 colorMenu->addAction(tr("Pick random"), this, SLOT(randomizeTrackColor()));
332 colorMenu->addSeparator();
333 colorMenu->addAction(tr("Reset clip colors"), this, SLOT(resetClipColors()));
334}

337void TrackOperationsWidget::toggleRecording( bool on )
338{
339 auto atv = dynamic_cast<AutomationTrackView*>(m_trackView);
340 if( atv )
341 {
342 for( Clip * clip : atv->getTrack()->getClips() )
343 {
344 auto ap = dynamic_cast<AutomationClip*>(clip);
345 if( ap ) { ap->setRecording( on ); }
346 }
347 atv->update();
348 }
349}
