104void EffectRackView::moveUp( EffectView* view )
105{
106 fxChain()->moveUp( view->effect() );
107 if( view != m_effectViews.first() )
108 {
109 int i = 0;
110 for( QVector<EffectView *>::Iterator it = m_effectViews.begin();
111 it != m_effectViews.end(); it++, i++ )
112 {
113 if( *it == view )
114 {
115 break;
116 }
117 }
118
119 EffectView * temp = m_effectViews[ i - 1 ];
120
121 m_effectViews[i - 1] = view;
122 m_effectViews[i] = temp;
123
124 update();
125 }
126}

156void EffectRackView::update()
157{
158 QWidget * w = m_scrollArea->widget();
159 QVector<bool> view_map( qMax<int>( fxChain()->m_effects.size(),
160 m_effectViews.size() ), false );
161
162 for (const auto& effect : fxChain()->m_effects)
163 {
164 int i = 0;
165 for (const auto& effectView : m_effectViews)
166 {
167 if (effectView->model() == effect)
168 {
169 view_map[i] = true;
170 break;
171 }
172 ++i;
173 }
174 if( i >= m_effectViews.size() )
175 {
176 auto view = new EffectView(effect, w);
177 connect(view, &EffectView::movedUp, this, &EffectRackView::moveUp);
178 connect(view, &EffectView::movedDown, this, &EffectRackView::moveDown);
179 connect(view, &EffectView::deletedPlugin, this, &EffectRackView::deletePlugin, Qt::QueuedConnection);
180
181 QAction* moveUpAction = new QAction(view);
182 moveUpAction->setShortcut(combine(Qt::Key_Up, Qt::AltModifier));
183 moveUpAction->setShortcutContext(Qt::WidgetShortcut);
184 connect(moveUpAction, &QAction::triggered, view, &EffectView::moveUp);
185 view->addAction(moveUpAction);
186
187 QAction* moveDownAction = new QAction(view);
188 moveDownAction->setShortcut(combine(Qt::Key_Down, Qt::AltModifier));
189 moveDownAction->setShortcutContext(Qt::WidgetShortcut);
190 connect(moveDownAction, &QAction::triggered, view, &EffectView::moveDown);
191 view->addAction(moveDownAction);
192
193 view->show();
194 m_effectViews.append( view );
195 if( i < view_map.size() )
196 {
197 view_map[i] = true;
198 }
199 else
200 {
201 view_map.append( true );
202 }
203
204 }
205 }
206
207 int i = 0, nView = 0;
208
209 const int EffectViewMargin = 3;
210 m_lastY = EffectViewMargin;
211
212 for( QVector<EffectView *>::Iterator it = m_effectViews.begin();
213 it != m_effectViews.end(); i++ )
214 {
215 if( i < view_map.size() && view_map[i] == false )
216 {
217 delete m_effectViews[nView];
218 it = m_effectViews.erase( it );
219 }
220 else
221 {
222 ( *it )->move( EffectViewMargin, m_lastY );
223 m_lastY += ( *it )->height();
224 ++nView;
225 ++it;
226 }
227 }
228
229 w->setFixedSize(EffectView::DEFAULT_WIDTH + 2 * EffectViewMargin, m_lastY);
230
231 QWidget::update();
232}

237void EffectRackView::addEffect()
238{
239 EffectSelectDialog esd( this );
240 esd.exec();
241
242 if( esd.result() == QDialog::Rejected )
243 {
244 return;
245 }
246
247 Effect * fx = esd.instantiateSelectedPlugin( fxChain() );
248
249 fxChain()->appendEffect( fx );
250 update();
251
252 // Find the effectView, and show the controls
253 for (const auto& effectView : m_effectViews)
254 {
255 if (effectView->effect() == fx)
256 {
257 effectView->editControls();
258 break;
259 }
260 }
261
262
263}
