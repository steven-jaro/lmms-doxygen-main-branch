81void LcdFloatSpinBox::layoutSetup(const QString &style)
82{
83 // Assemble the LCD parts
84 auto lcdLayout = new QHBoxLayout();
85
86 m_wholeDisplay.setSeamless(false, true);
87 m_fractionDisplay.setSeamless(true, false);
88
89 lcdLayout->addWidget(&m_wholeDisplay);
90
91 auto dotLabel = new QLabel("", this);
92 QPixmap dotPixmap(embed::getIconPixmap(QString("lcd_" + style + "_dot").toUtf8().constData()));
93 dotLabel->setPixmap(dotPixmap.copy(0, 0, dotPixmap.size().width(), dotPixmap.size().height() / 2));
94 lcdLayout->addWidget(dotLabel);
95
96 lcdLayout->addWidget(&m_fractionDisplay);
97
98 lcdLayout->setContentsMargins(0, 0, 0, 0);
99 lcdLayout->setSpacing(0);
100
101 // Add space for label
102 auto outerLayout = new QVBoxLayout();
103 outerLayout->addLayout(lcdLayout);
104 outerLayout->addSpacing(9);
105 outerLayout->setContentsMargins(0, 0, 0, 0);
106 outerLayout->setSizeConstraint(QLayout::SetFixedSize);
107 this->setLayout(outerLayout);
108 this->setFixedHeight(32);
109}

112void LcdFloatSpinBox::update()
113{
114 const int digitValue = fastPow10f(m_fractionDisplay.numDigits());
115 float value = model()->value();
116 int fraction = std::abs(std::round((value - static_cast<int>(value)) * digitValue));
117 if (fraction == digitValue)
118 {
119 value += std::copysign(1, value);
120 fraction = 0;
121 }
122 m_wholeDisplay.setValue(value);
123 m_fractionDisplay.setValue(fraction);
124
125 QWidget::update();
126}

137void LcdFloatSpinBox::mousePressEvent(QMouseEvent* event)
138{
139 // switch between integer and fractional step based on cursor position
140 m_intStep = event->x() < m_wholeDisplay.width();
141
142 if (event->button() == Qt::LeftButton &&
143 !(event->modifiers() & KBD_COPY_MODIFIER) &&
144 event->y() < m_wholeDisplay.cellHeight() + 2)
145 {
146 m_mouseMoving = true;
147 m_origMousePos = event->globalPos();
148
149 AutomatableModel *thisModel = model();
150 if (thisModel)
151 {
152 thisModel->addJournalCheckPoint();
153 thisModel->saveJournallingState(false);
154 }
155 }
156 else
157 {
158 FloatModelView::mousePressEvent(event);
159 }
160}

163void LcdFloatSpinBox::mouseMoveEvent(QMouseEvent* event)
164{
165 if (m_mouseMoving)
166 {
167 int dy = event->globalY() - m_origMousePos.y();
168 if (getGUI()->mainWindow()->isShiftPressed()) { dy = qBound(-4, dy/4, 4); }
169 if (dy > 1 || dy < -1)
170 {
171 model()->setValue(model()->value() - dy / 2 * getStep());
172 emit manualChange();
173 m_origMousePos = event->globalPos();
174 }
175 }
176}

189void LcdFloatSpinBox::wheelEvent(QWheelEvent *event)
190{
191 // switch between integer and fractional step based on cursor position
192 if (position(event).x() < m_wholeDisplay.width()) { m_intStep = true; }
193 else { m_intStep = false; }
194
195 event->accept();
196 model()->setValue(model()->value() + ((event->angleDelta().y() > 0) ? 1 : -1) * getStep());
197 emit manualChange();
198}

207void LcdFloatSpinBox::enterValue()
208{
209 bool ok;
210 float newVal;
211
212 newVal = QInputDialog::getDouble(
213 this, tr("Set value"),
214 tr("Please enter a new value between %1 and %2:").
215 arg(model()->minValue()).
216 arg(model()->maxValue()),
217 model()->value(),
218 model()->minValue(),
219 model()->maxValue(),
220 m_fractionDisplay.numDigits(), &ok);
221
222 if (ok)
223 {
224 model()->setValue(newVal);
225 }
226}

236void LcdFloatSpinBox::paintEvent(QPaintEvent*)
237{
238 QPainter p(this);
239
240 // Border
241 QStyleOptionFrame opt;
242 opt.initFrom(this);
243 opt.state = QStyle::State_Sunken;
244 opt.rect = QRect(0, 0, width() - 1, m_wholeDisplay.height());
245 style()->drawPrimitive(QStyle::PE_Frame, &opt, &p, this);
246
247 // Label
248 if (!m_label.isEmpty())
249 {
250 p.setFont(adjustedToPixelSize(p.font(), DEFAULT_FONT_SIZE));
251 p.setPen(m_wholeDisplay.textShadowColor());
252 p.drawText(width() / 2 - p.fontMetrics().boundingRect(m_label).width() / 2 + 1, height(), m_label);
253 p.setPen(m_wholeDisplay.textColor());
254 p.drawText(width() / 2 - p.fontMetrics().boundingRect(m_label).width() / 2, height() - 1, m_label);
255 }
256}
