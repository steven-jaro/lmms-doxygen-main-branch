47InstrumentTuningView::InstrumentTuningView(InstrumentTrack *it, QWidget *parent) :
48 QWidget(parent)
49{
50 auto layout = new QVBoxLayout(this);
51 layout->setContentsMargins(5, 5, 5, 5);
52
53 // Master pitch toggle
54 m_pitchGroupBox = new GroupBox(tr("GLOBAL TRANSPOSITION"));
55 m_pitchGroupBox->setModel(&it->m_useMasterPitchModel);
56 layout->addWidget(m_pitchGroupBox);
57
58 auto masterPitchLayout = new QHBoxLayout(m_pitchGroupBox);
59 masterPitchLayout->setContentsMargins(8, 18, 8, 8);
60
61 auto tlabel = new QLabel(tr("Enables the use of global transposition"));
62 tlabel->setWordWrap(true);
63 tlabel->setFont(adjustedToPixelSize(tlabel->font(), DEFAULT_FONT_SIZE));
64 masterPitchLayout->addWidget(tlabel);
65
66 // Microtuner settings
67 m_microtunerNotSupportedLabel = new QLabel(tr("Microtuner is not available for MIDI-based instruments."));
68 m_microtunerNotSupportedLabel->setWordWrap(true);
69 m_microtunerNotSupportedLabel->hide();
70 layout->addWidget(m_microtunerNotSupportedLabel);
71
72 m_microtunerGroupBox = new GroupBox(tr("MICROTUNER"));
73 m_microtunerGroupBox->setModel(it->m_microtuner.enabledModel());
74 layout->addWidget(m_microtunerGroupBox);
75
76 auto microtunerLayout = new QVBoxLayout(m_microtunerGroupBox);
77 microtunerLayout->setContentsMargins(8, 18, 8, 8);
78
79 auto scaleEditLayout = new QHBoxLayout();
80 scaleEditLayout->setContentsMargins(0, 0, 4, 0);
81 microtunerLayout->addLayout(scaleEditLayout);
82
83 auto scaleLabel = new QLabel(tr("Active scale:"));
84 scaleEditLayout->addWidget(scaleLabel);
85
86 QPixmap editPixmap(embed::getIconPixmap("edit_draw_small"));
87 auto editPixButton = new PixmapButton(this, tr("Edit scales and keymaps"));
88 editPixButton->setToolTip(tr("Edit scales and keymaps"));
89 editPixButton->setInactiveGraphic(editPixmap);
90 editPixButton->setActiveGraphic(editPixmap);
91 editPixButton->setFixedSize(16, 16);
92 connect(editPixButton, SIGNAL(clicked()), getGUI()->mainWindow(), SLOT(toggleMicrotunerWin()));
93
94 scaleEditLayout->addWidget(editPixButton);
95
96 m_scaleCombo = new ComboBox();
97 m_scaleCombo->setModel(it->m_microtuner.scaleModel());
98 microtunerLayout->addWidget(m_scaleCombo);
99
100 auto keymapLabel = new QLabel(tr("Active keymap:"));
101 microtunerLayout->addWidget(keymapLabel);
102
103 m_keymapCombo = new ComboBox();
104 m_keymapCombo->setModel(it->m_microtuner.keymapModel());
105 microtunerLayout->addWidget(m_keymapCombo);
106
107 m_rangeImportCheckbox = new LedCheckBox(tr("Import note ranges from keymap"), this);
108 m_rangeImportCheckbox->setModel(it->m_microtuner.keyRangeImportModel());
109 m_rangeImportCheckbox->setToolTip(tr("When enabled, the first, last and base notes of this instrument will be overwritten with values specified by the selected keymap."));
110 m_rangeImportCheckbox->setCheckable(true);
111 microtunerLayout->addWidget(m_rangeImportCheckbox);
112
113 // Fill remaining space
114 layout->addStretch();
115}
