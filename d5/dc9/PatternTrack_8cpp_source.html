61PatternTrack::~PatternTrack()
62{
63 Engine::audioEngine()->removePlayHandlesOfTypes( this,
64 PlayHandle::Type::NotePlayHandle
65 | PlayHandle::Type::InstrumentPlayHandle
66 | PlayHandle::Type::SamplePlayHandle );
67
68 const int pattern = s_infoMap[this];
69 Engine::patternStore()->removePattern(pattern);
70 for( infoMap::iterator it = s_infoMap.begin(); it != s_infoMap.end();
71 ++it )
72 {
73 if (it.value() > pattern)
74 {
75 --it.value();
76 }
77 }
78 s_infoMap.remove( this );
79
80 // remove us from the Song and update the pattern selection combobox to reflect the change
81 trackContainer()->removeTrack( this );
82 Engine::patternStore()->updateComboBox();
83}

88// play _frames frames of given Clip within starting with _start
89bool PatternTrack::play( const TimePos & _start, const fpp_t _frames,
90 const f_cnt_t _offset, int _clip_num )
91{
92 if( isMuted() )
93 {
94 return false;
95 }
96
97 if( _clip_num >= 0 )
98 {
99 return Engine::patternStore()->play(_start, _frames, _offset, s_infoMap[this]);
100 }
101
102 clipVector clips;
103 getClipsInRange( clips, _start, _start + static_cast<int>( _frames / Engine::framesPerTick() ) );
104
105 if( clips.size() == 0 )
106 {
107 return false;
108 }
109
110 TimePos lastPosition;
111 TimePos lastLength;
112 tick_t lastOffset = 0;
113 for (const auto& clip : clips)
114 {
115 if (!clip->isMuted() && clip->startPosition() >= lastPosition)
116 {
117 lastPosition = clip->startPosition();
118 lastLength = clip->length();
119 tick_t patternLength = Engine::patternStore()->lengthOfPattern(static_cast<PatternClip*>(clip)->patternIndex())
120 * TimePos::ticksPerBar();
121 lastOffset = patternLength - (clip->startTimeOffset() % patternLength);
122 if (lastOffset == patternLength)
123 {
124 lastOffset = 0;
125 }
126 }
127 }
128
129 if( _start - lastPosition < lastLength )
130 {
131 return Engine::patternStore()->play(_start - lastPosition + lastOffset, _frames, _offset, s_infoMap[this]);
132 }
133 return false;
134}

178void PatternTrack::loadTrackSpecificSettings(const QDomElement& _this)
179{
180/* if( _this.attribute( "icon" ) != "" )
181 {
182 m_trackLabel->setPixmapFile( _this.attribute( "icon" ) );
183 }*/
184
185 // If data was created by Track::clone(), do not add any tracks to the pattern(-editor)
186 // instead create a new copy of the clip on each track
187 if (_this.hasAttribute("sourcepattern"))
188 {
189 const int src = _this.attribute("sourcepattern").toInt();
190 const int dst = s_infoMap[this];
191 // copy clips of all tracks from source pattern (at bar "src") to destination
192 // clips (which are created if they do not exist yet)
193 for (const auto& track : Engine::patternStore()->tracks())
194 {
195 Clip::copyStateTo(track->getClip(src), track->getClip(dst));
196 }
197 setName( tr( "Clone of %1" ).arg(
198 _this.parentNode().toElement().attribute( "name" ) ) );
199 }
200 else
201 {
202 QDomNode node = _this.namedItem(
203 TrackContainer::classNodeName() );
204 if( node.isElement() )
205 {
206 Engine::patternStore()->restoreState(node.toElement());
207 }
208 }
209/* doesn't work yet because PatternTrack-ctor also sets current pattern so if
210 pattern tracks are created after this function is called, this doesn't
211 help at all....
212 if( _this.attribute( "current" ).toInt() )
213 {
214 engine::getPatternEditor()->setCurrentPattern( s_infoMap[this] );
215 }*/
216}

222PatternTrack* PatternTrack::findPatternTrack(int pattern_num)
223{
224 for( infoMap::iterator it = s_infoMap.begin(); it != s_infoMap.end();
225 ++it )
