51void jack_log_function(int level, const char *message)
52{
53 void (* log_callback)(const char *);
54
55 switch (level)
56 {
57 case LOG_LEVEL_INFO:
58 log_callback = jack_info_callback;
59 break;
60 case LOG_LEVEL_ERROR:
61 log_callback = jack_error_callback;
62 break;
63 default:
64 return;
65 }
66
67 log_callback(message);
68}

70static void jack_format_and_log(int level, const char *prefix, const char *fmt, va_list ap)
71{
72 char buffer[256];
73 size_t len;
74 jack_log_function_t log_function;
75
76 if (prefix != NULL) {
77 len = strlen(prefix);
78 assert(len < 256);
79 memcpy(buffer, prefix, len);
80 } else {
81 len = 0;
82 }
83
84 vsnprintf(buffer + len, sizeof(buffer) - len, fmt, ap);
85
86 log_function = (jack_log_function_t)jack_tls_get(JackGlobals::fKeyLogFunction);
87
88 /* if log function is not overridden for thread, use default one */
89 if (log_function == NULL)
90 {
91 log_function = jack_log_function;
92 //log_function(LOG_LEVEL_INFO, "------ Using default log function");
93 }
94 else
95 {
96 //log_function(LOG_LEVEL_INFO, "++++++ Using thread-specific log function");
97 }
98
99 log_function(level, buffer);
100}
