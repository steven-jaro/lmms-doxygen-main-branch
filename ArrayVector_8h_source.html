351 friend void swap(ArrayVector& a, ArrayVector& b)
352 noexcept(std::is_nothrow_swappable_v<T> && std::is_nothrow_move_constructible_v<T>)
353 {
354 const auto toSwap = std::min(a.size(), b.size());
355 const auto aSwapEnd = a.begin() + toSwap;
356 const auto bSwapEnd = b.begin() + toSwap;
357 std::swap_ranges(a.begin(), aSwapEnd, b.begin());
358 std::uninitialized_move(aSwapEnd, a.end(), bSwapEnd);
359 std::uninitialized_move(bSwapEnd, b.end(), aSwapEnd);
360 std::destroy(aSwapEnd, a.end());
361 std::destroy(bSwapEnd, b.end());
362 std::swap(a.m_size, b.m_size);
363 }
