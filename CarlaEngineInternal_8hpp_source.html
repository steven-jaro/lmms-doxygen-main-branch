208static inline
209const char* EnginePostAction2Str(const EnginePostAction action)
210{
211 switch (action)
212 {
213 case kEnginePostActionNull:
214 return "kEnginePostActionNull";
215 case kEnginePostActionZeroCount:
216 return "kEnginePostActionZeroCount";
217#ifndef BUILD_BRIDGE_ALTERNATIVE_ARCH
218 case kEnginePostActionRemovePlugin:
219 return "kEnginePostActionRemovePlugin";
220 case kEnginePostActionSwitchPlugins:
221 return "kEnginePostActionSwitchPlugins";
222#endif
223 }
224
225 carla_stderr("CarlaBackend::EnginePostAction2Str(%i) - invalid action", action);
226 return nullptr;
227}
