68 static void do_complete(void* owner, operation* base,
69 const asio::error_code& /*ec*/,
70 std::size_t /*bytes_transferred*/)
71 {
72 // Take ownership of the operation object.
73 resolve_endpoint_op* o(static_cast<resolve_endpoint_op*>(base));
74 ptr p = { asio::detail::addressof(o->handler_), o, o };
75 handler_work<Handler, IoExecutor> w(o->handler_, o->io_executor_);
76
77 if (owner && owner != &o->scheduler_)
78 {
79 // The operation is being run on the worker io_context. Time to perform
80 // the resolver operation.
81
82 // Perform the blocking endpoint resolution operation.
83 char host_name[NI_MAXHOST];
84 char service_name[NI_MAXSERV];
85 socket_ops::background_getnameinfo(o->cancel_token_, o->endpoint_.data(),
86 o->endpoint_.size(), host_name, NI_MAXHOST, service_name, NI_MAXSERV,
87 o->endpoint_.protocol().type(), o->ec_);
88 o->results_ = results_type::create(o->endpoint_, host_name, service_name);
89
90 // Pass operation back to main io_context for completion.
91 o->scheduler_.post_deferred_completion(o);
92 p.v = p.p = 0;
93 }
94 else
95 {
96 // The operation has been returned to the main io_context. The completion
97 // handler is ready to be delivered.
98
99 ASIO_HANDLER_COMPLETION((*o));
100
101 // Make a copy of the handler so that the memory can be deallocated
102 // before the upcall is made. Even if we're not about to make an upcall,
103 // a sub-object of the handler may be the true owner of the memory
104 // associated with the handler. Consequently, a local copy of the handler
105 // is required to ensure that any owning sub-object remains valid until
106 // after we have deallocated the memory here.
107 detail::binder2<Handler, asio::error_code, results_type>
108 handler(o->handler_, o->ec_, o->results_);
109 p.h = asio::detail::addressof(handler.handler_);
110 p.reset();
111
112 if (owner)
113 {
114 fenced_block b(fenced_block::half);
115 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, "..."));
116 w.complete(handler, handler.handler_);
117 ASIO_HANDLER_INVOCATION_END;
118 }
119 }
120 }

127 IoExecutor io_executor_;
128 results_type results_;
129};
