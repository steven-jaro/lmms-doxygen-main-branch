99EQ::EQ(EffectParams pars)
100 :Effect(pars)
101{
102 for(int i = 0; i < MAX_EQ_BANDS; ++i) {
103 filter[i].Ptype = 0;
104 filter[i].Pfreq = 64;
105 filter[i].Pgain = 64;
106 filter[i].Pq = 64;
107 filter[i].Pstages = 0;
108 filter[i].l = memory.alloc<AnalogFilter>(6, 1000.0f, 1.0f, 0, pars.srate, pars.bufsize);
109 filter[i].r = memory.alloc<AnalogFilter>(6, 1000.0f, 1.0f, 0, pars.srate, pars.bufsize);
110 }
111 //default values
112 Pvolume = 50;
113
114 setpreset(Ppreset);
115 cleanup();
116}

161void EQ::setpreset(unsigned char npreset)
162{
163 const int PRESET_SIZE = 1;
164 const int NUM_PRESETS = 2;
165 unsigned char presets[NUM_PRESETS][PRESET_SIZE] = {
166 {67}, //EQ 1
167 {67} //EQ 2
168 };
169
170 if(npreset >= NUM_PRESETS)
171 npreset = NUM_PRESETS - 1;
172 for(int n = 0; n < PRESET_SIZE; ++n)
173 changepar(n, presets[npreset][n]);
174 Ppreset = npreset;

178void EQ::changepar(int npar, unsigned char value)
179{
180 switch(npar) {
181 case 0:
182 setvolume(value);
183 break;
184 }
185 if(npar < 10)
186 return;
187
188 int nb = (npar - 10) / 5; //number of the band (filter)
189 if(nb >= MAX_EQ_BANDS)
190 return;
191 int bp = npar % 5; //band paramenter
192
193 float tmp;
194 switch(bp) {
195 case 0:
196 filter[nb].Ptype = value;
197 if(value > 9)
198 filter[nb].Ptype = 0; //has to be changed if more filters will be added
199 if(filter[nb].Ptype != 0) {
200 filter[nb].l->settype(value - 1);
201 filter[nb].r->settype(value - 1);
202 }
203 break;
204 case 1:
205 filter[nb].Pfreq = value;
206 tmp = 600.0f * powf(30.0f, (value - 64.0f) / 64.0f);
207 filter[nb].l->setfreq(tmp);
208 filter[nb].r->setfreq(tmp);
209 break;
210 case 2:
211 filter[nb].Pgain = value;
212 tmp = 30.0f * (value - 64.0f) / 64.0f;
213 filter[nb].l->setgain(tmp);
214 filter[nb].r->setgain(tmp);
215 break;
216 case 3:
217 filter[nb].Pq = value;
218 tmp = powf(30.0f, (value - 64.0f) / 64.0f);
219 filter[nb].l->setq(tmp);
220 filter[nb].r->setq(tmp);
221 break;
222 case 4:
223 filter[nb].Pstages = value;
224 if(value >= MAX_FILTER_STAGES)
225 filter[nb].Pstages = MAX_FILTER_STAGES - 1;
226 filter[nb].l->setstages(value);
227 filter[nb].r->setstages(value);
228 break;
229 }
230}

232unsigned char EQ::getpar(int npar) const
233{
234 switch(npar) {
235 case 0:
236 return Pvolume;
237 break;
238 }
239
240 if(npar < 10)
241 return 0;
242
243 int nb = (npar - 10) / 5; //number of the band (filter)
244 if(nb >= MAX_EQ_BANDS)
245 return 0;
246 int bp = npar % 5; //band paramenter
247 switch(bp) {
248 case 0:
249 return filter[nb].Ptype;
250 break;
251 case 1:
252 return filter[nb].Pfreq;
253 break;
254 case 2:
255 return filter[nb].Pgain;
256 break;
257 case 3:
258 return filter[nb].Pq;
259 break;
260 case 4:
261 return filter[nb].Pstages;
262 break;
263 default: return 0; //in case of bogus parameter number
264 }

268float EQ::getfreqresponse(float freq)
269{
270 float resp = 1.0f;
271 for(int i = 0; i < MAX_EQ_BANDS; ++i) {
272 if(filter[i].Ptype == 0)
273 continue;
274 resp *= filter[i].l->H(freq);
275 }
276 return rap2dB(resp * outvolume);
277}

279//Not exactly the most efficient manner to derive the total taps, but it should
280//be fast enough in practice
281void EQ::getFilter(float *a, float *b) const
282{
283 a[0] = 1;
284 b[0] = 1;
285 off_t off=0;
286 for(int i = 0; i < MAX_EQ_BANDS; ++i) {
287 auto &F = filter[i];
288 if(F.Ptype == 0)
289 continue;
290 const double Fb[3] = {F.l->coeff.c[0], F.l->coeff.c[1], F.l->coeff.c[2]};
291 const double Fa[3] = {1.0f, -F.l->coeff.d[1], -F.l->coeff.d[2]};
292
293 for(int j=0; j<F.Pstages+1; ++j) {
294 for(int k=0; k<3; ++k) {
295 a[off] = Fa[k];
296 b[off] = Fb[k];
297 off++;
298 }
299 }
300 }
301}
