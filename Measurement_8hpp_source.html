93 void resetTimer()
94 {
95 mTimer.cancel();
96 mTimer.expires_from_now(std::chrono::milliseconds(50));
97 mTimer.async_wait([this](const typename Timer::ErrorCode e) {
98 if (!e)
99 {
100 if (mMeasurementsStarted < kNumberMeasurements)
101 {
102 const auto ht = HostTime{mClock.micros()};
103 sendPing(mEndpoint, discovery::makePayload(ht));
104 ++mMeasurementsStarted;
105 resetTimer();
106 }
107 else
108 {
109 fail();
110 }
111 }
112 });
113 }

120 // Operator to handle incoming messages on the interface
121 template <typename It>
122 void operator()(
123 const asio::ip::udp::endpoint& from, const It messageBegin, const It messageEnd)
124 {
125 using namespace std;
126 const auto result = v1::parseMessageHeader(messageBegin, messageEnd);
127 const auto& header = result.first;
128 const auto payloadBegin = result.second;
129
130 if (header.messageType == v1::kPong)
131 {
132 debug(mLog) << "Received Pong message from " << from;
133
134 // parse for all entries
135 SessionId sessionId{};
136 std::chrono::microseconds ghostTime{0};
137 std::chrono::microseconds prevGHostTime{0};
138 std::chrono::microseconds prevHostTime{0};
139
140 try
141 {
142 discovery::parsePayload<SessionMembership, GHostTime, PrevGHostTime, HostTime>(
143 payloadBegin, messageEnd,
144 [&sessionId](const SessionMembership& sms) { sessionId = sms.sessionId; },
145 [&ghostTime](GHostTime gt) { ghostTime = std::move(gt.time); },
146 [&prevGHostTime](PrevGHostTime gt) { prevGHostTime = std::move(gt.time); },
147 [&prevHostTime](HostTime ht) { prevHostTime = std::move(ht.time); });
148 }
149 catch (const std::runtime_error& err)
150 {
151 warning(mLog) << "Failed parsing payload, caught exception: " << err.what();
152 listen();
153 return;
154 }
155
156 if (mSessionId == sessionId)
157 {
158 const auto hostTime = mClock.micros();
159
160 const auto payload =
161 discovery::makePayload(HostTime{hostTime}, PrevGHostTime{ghostTime});
162
163 sendPing(from, payload);
164 listen();
165
166 if (prevGHostTime != Micros{0})
167 {
168 mData.push_back(
169 std::make_pair(static_cast<double>((hostTime + prevHostTime).count()) * 0.5,
170 static_cast<double>(ghostTime.count())));
171 mData.push_back(std::make_pair(static_cast<double>(prevHostTime.count()),
172 static_cast<double>((ghostTime + prevGHostTime).count()) * 0.5));
173 }
174
175 if (mData.size() > kNumberDataPoints)
176 {
177 finish();
178 }
179 else
180 {
181 resetTimer();
182 }
183 }
184 else
185 {
186 fail();
187 }
188 }
189 else
190 {
191 debug(mLog) << "Received invalid message from " << from;
192 listen();
193 }
194 }

196 template <typename Payload>
197 void sendPing(asio::ip::udp::endpoint to, const Payload& payload)
198 {
199 v1::MessageBuffer buffer;
200 const auto msgBegin = std::begin(buffer);
201 const auto msgEnd = v1::pingMessage(payload, msgBegin);
202 const auto numBytes = static_cast<size_t>(std::distance(msgBegin, msgEnd));
203
204 try
205 {
206 mSocket.send(buffer.data(), numBytes, to);
207 }
208 catch (const std::runtime_error& err)
209 {
210 info(mLog) << "Failed to send Ping to " << to.address().to_string() << ": "
211 << err.what();
212 }
213 }
214
215 void finish()
216 {
217 mTimer.cancel();
218 mCallback(std::move(mData));
219 mData = {};
220 mSuccess = true;
221 debug(mLog) << "Measuring " << mEndpoint << " done.";
222 }
223
224 void fail()
225 {
226 mCallback(std::vector<Point>{});
227 mData = {};
228 debug(mLog) << "Measuring " << mEndpoint << " failed.";
229 }
230
231 Socket mSocket;
232 SessionId mSessionId;
233 asio::ip::udp::endpoint mEndpoint;
234 std::vector<std::pair<double, double>> mData;
235 Callback mCallback;
236 Clock mClock;
237 Timer mTimer;
238 std::size_t mMeasurementsStarted;
239 Log mLog;
240 bool mSuccess;
241 };
