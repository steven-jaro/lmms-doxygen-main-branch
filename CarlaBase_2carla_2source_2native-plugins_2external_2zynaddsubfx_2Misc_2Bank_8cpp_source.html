211 */
212int Bank::loadbank(string bankdirname)
213{
214 normalizedirsuffix(bankdirname);
215 DIR *dir = opendir(bankdirname.c_str());
216 clearbank();
217
218 if(dir == NULL)
219 return -1;
220
221 //set msb when possible
222 bank_msb = 0;
223 for(unsigned i=0; i<banks.size(); i++)
224 if(banks[i].dir == bankdirname)
225 bank_msb = i;
226
227 dirname = bankdirname;
228
229 bankfiletitle = dirname;
230
231 struct dirent *fn;
232
233 while((fn = readdir(dir))) {
234 const char *filename = fn->d_name;
235
236 //check for extension
237 if(strstr(filename, INSTRUMENT_EXTENSION) == NULL)
238 continue;
239
240 //verify if the name is like this NNNN-name (where N is a digit)
241 int no = 0;
242 unsigned int startname = 0;
243
244 for(unsigned int i = 0; i < 4; ++i) {
245 if(strlen(filename) <= i)
246 break;
247
248 if((filename[i] >= '0') && (filename[i] <= '9')) {
249 no = no * 10 + (filename[i] - '0');
250 startname++;
251 }
252 }
253
254 if((startname + 1) < strlen(filename))
255 startname++; //to take out the "-"
256
257 string name = filename;
258
259 //remove the file extension
260 for(int i = name.size() - 1; i >= 2; i--)
261 if(name[i] == '.') {
262 name = name.substr(0, i);
263 break;
264 }
265
266 if(no != 0) //the instrument position in the bank is found
267 addtobank(no - 1, filename, name.substr(startname));
268 else
269 addtobank(-1, filename, name);
270 }
271
272 closedir(dir);
273
274 if(!dirname.empty())
275 config->cfg.currentBankDir = dirname;
276
277 return 0;
278}

316/*
317 * Swaps a slot with another
318 */
319int Bank::swapslot(unsigned int n1, unsigned int n2)
320{
321 int err = 0;
322 if((n1 == n2) || (locked()))
323 return 0;
324 if(emptyslot(n1) && (emptyslot(n2)))
325 return 0;
326 if(emptyslot(n1)) //change n1 to n2 in order to make
327 swap(n1, n2);
328
329 if(emptyslot(n2)) { //this is just a movement from slot1 to slot2
330 err |= setname(n1, getname(n1), n2);
331 if(err)
332 return err;
333 ins[n2] = ins[n1];
334 ins[n1] = ins_t();
335 }
336 else { //if both slots are used
337 if(ins[n1].name == ins[n2].name) //change the name of the second instrument if the name are equal
338 ins[n2].name += "2";
339
340 err |= setname(n1, getname(n1), n2);
341 err |= setname(n2, getname(n2), n1);
342 if(err)
343 return err;
344 swap(ins[n2], ins[n1]);
345 }
346 return err;
347}

359void Bank::rescanforbanks()
360{
361 db->clear();
362 //remove old banks
363 banks.clear();
364
365 for(int i = 0; i < MAX_BANK_ROOT_DIRS; ++i)
366 if(!config->cfg.bankRootDirList[i].empty())
367 scanrootdir(config->cfg.bankRootDirList[i]);
368#ifdef WIN32
369 {
370 //Search the VST Directory for banks/preset/etc
371 char path[1024];
372 GetModuleFileName(GetModuleHandle("ZynAddSubFX.dll"), path, sizeof(path));
373 if(strstr(path, "ZynAddSubFX.dll")) {
374 strstr(path, "ZynAddSubFX.dll")[0] = 0;
375 strcat(path, "banks");
376 scanrootdir(path);
377 }
378 }
379#endif
380
381 //sort the banks
382 sort(banks.begin(), banks.end());
383
384 for(int i = 0; i < (int) banks.size(); ++i)
385 db->addBankDir(banks[i].dir);
386
387 //remove duplicate bank names
388 for(int j = 0; j < (int) banks.size() - 1; ++j) {
389 int dupl = 0;
390 for(int i = j + 1; i < (int) banks.size(); ++i) {
391 if(banks[i].name == banks[j].name) {
392 //add a [1] to the first bankname and [n] to others
393 banks[i].name = banks[i].name + '['
394 + stringFrom(dupl + 2) + ']';
395 dupl++;
396 }
397 }
398 if(dupl != 0)
399 banks[j].name += "[1]";
400 if(dupl)
401 j += dupl;
402 }
403 db->scanBanks();
404}

421void Bank::scanrootdir(string rootdir)
422{
423 expanddirname(rootdir);
424
425 DIR *dir = opendir(rootdir.c_str());
426 if(dir == NULL)
427 return;
428
429 bankstruct bank;
430
431 const char *separator = "/";
432 if(rootdir.size()) {
433 char tmp = rootdir[rootdir.size() - 1];
434 if((tmp == '/') || (tmp == '\\'))
435 separator = "";
436 }
437
438 struct dirent *fn;
439 while((fn = readdir(dir))) {
440 const char *dirname = fn->d_name;
441 if(dirname[0] == '.')
442 continue;
443
444 bank.dir = rootdir + separator + dirname + '/';
445 bank.name = dirname;
446 //find out if the directory contains at least 1 instrument
447 bool isbank = false;
448
449 DIR *d = opendir(bank.dir.c_str());
450 if(d == NULL)
451 continue;
452
453 struct dirent *fname;
454
455 while((fname = readdir(d))) {
456 if((strstr(fname->d_name, INSTRUMENT_EXTENSION) != NULL)
457 || (strstr(fname->d_name, FORCE_BANK_DIR_FILE) != NULL)) {
458 isbank = true;
459 break; //could put a #instrument counter here instead
460 }
461 }
462
463 if(isbank)
464 banks.push_back(bank);
465
466 closedir(d);
467 }
468
469 closedir(dir);
470}

506int Bank::addtobank(int pos, string filename, string name)
507{
508 if((pos >= 0) && (pos < BANK_SIZE)) {
509 if(!ins[pos].filename.empty())
510 pos = -1; //force it to find a new free position
511 }
512 else
513 if(pos >= BANK_SIZE)
514 pos = -1;
515
516
517 if(pos < 0) //find a free position
518 for(int i = BANK_SIZE - 1; i >= 0; i--)
519 if(ins[i].filename.empty()) {
520 pos = i;
521 break;
522 }
523
524 if(pos < 0)
525 return -1; //the bank is full
526
527 deletefrombank(pos);
528
529 ins[pos].name = name;
530 ins[pos].filename = dirname + filename;
531 return 0;
532}
