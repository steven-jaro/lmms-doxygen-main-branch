202//Change the effect
203void EffectMgr::changeeffectrt(int _nefx, bool avoidSmash)
204{
205 cleanup();
206 if(nefx == _nefx && efx != NULL)
207 return;
208 nefx = _nefx;
209 memset(efxoutl, 0, synth.bufferbytes);
210 memset(efxoutr, 0, synth.bufferbytes);
211 memory.dealloc(efx);
212 EffectParams pars(memory, insertion, efxoutl, efxoutr, 0,
213 synth.samplerate, synth.buffersize, filterpars, avoidSmash);
214 try {
215 switch (nefx) {
216 case 1:
217 efx = memory.alloc<Reverb>(pars);
218 break;
219 case 2:
220 efx = memory.alloc<Echo>(pars);
221 break;
222 case 3:
223 efx = memory.alloc<Chorus>(pars);
224 break;
225 case 4:
226 efx = memory.alloc<Phaser>(pars);
227 break;
228 case 5:
229 efx = memory.alloc<Alienwah>(pars);
230 break;
231 case 6:
232 efx = memory.alloc<Distorsion>(pars);
233 break;
234 case 7:
235 efx = memory.alloc<EQ>(pars);
236 break;
237 case 8:
238 efx = memory.alloc<DynamicFilter>(pars, time);
239 break;
240 //put more effect here
241 default:
242 efx = NULL;
243 break; //no effect (thru)
244 }
245 } catch (std::bad_alloc &ba) {
246 std::cerr << "failed to change effect " << _nefx << ": " << ba.what() << std::endl;
247 return;
248 }
249
250 if(!avoidSmash)
251 for(int i=0; i<128; ++i)
252 settings[i] = geteffectparrt(i);
253}

361// Apply the effect
362void EffectMgr::out(float *smpsl, float *smpsr)
363{
364 if(!efx) {
365 if(!insertion)
366 for(int i = 0; i < synth.buffersize; ++i) {
367 smpsl[i] = 0.0f;
368 smpsr[i] = 0.0f;
369 efxoutl[i] = 0.0f;
370 efxoutr[i] = 0.0f;
371 }
372 return;
373 }
374 for(int i = 0; i < synth.buffersize; ++i) {
375 smpsl[i] += synth.denormalkillbuf[i];
376 smpsr[i] += synth.denormalkillbuf[i];
377 efxoutl[i] = 0.0f;
378 efxoutr[i] = 0.0f;
379 }
380 efx->out(smpsl, smpsr);
381
382 float volume = efx->volume;
383
384 if(nefx == 7) { //this is need only for the EQ effect
385 memcpy(smpsl, efxoutl, synth.bufferbytes);
386 memcpy(smpsr, efxoutr, synth.bufferbytes);
387 return;
388 }
389
390 //Insertion effect
391 if(insertion != 0) {
392 float v1, v2;
393 if(volume < 0.5f) {
394 v1 = 1.0f;
395 v2 = volume * 2.0f;
396 }
397 else {
398 v1 = (1.0f - volume) * 2.0f;
399 v2 = 1.0f;
400 }
401 if((nefx == 1) || (nefx == 2))
402 v2 *= v2; //for Reverb and Echo, the wet function is not liniar
403
404 if(dryonly) //this is used for instrument effect only
405 for(int i = 0; i < synth.buffersize; ++i) {
406 smpsl[i] *= v1;
407 smpsr[i] *= v1;
408 efxoutl[i] *= v2;
409 efxoutr[i] *= v2;
410 }
411 else // normal instrument/insertion effect
412 for(int i = 0; i < synth.buffersize; ++i) {
413 smpsl[i] = smpsl[i] * v1 + efxoutl[i] * v2;
414 smpsr[i] = smpsr[i] * v1 + efxoutr[i] * v2;
415 }
416 }
417 else // System effect
418 for(int i = 0; i < synth.buffersize; ++i) {
419 efxoutl[i] *= 2.0f * volume;
420 efxoutr[i] *= 2.0f * volume;
421 smpsl[i] = efxoutl[i];
422 smpsr[i] = efxoutr[i];
423 }
424}

459void EffectMgr::add2XML(XMLwrapper& xml)
460{
461 xml.addpar("type", geteffect());
462
463 if(!geteffect())
464 return;
465 xml.addpar("preset", preset);
466
467 xml.beginbranch("EFFECT_PARAMETERS");
468 for(int n = 0; n < 128; ++n) {
469 int par = 0;
470 if(efx)
471 par = efx->getpar(n);
472 else if(n<128)
473 par = settings[n];
474
475 if(par == 0)
476 continue;
477 xml.beginbranch("par_no", n);
478 xml.addpar("par", par);
479 xml.endbranch();
480 }
481 assert(filterpars);
482 if(nefx == 8) {
483 xml.beginbranch("FILTER");
484 filterpars->add2XML(xml);
485 xml.endbranch();
486 }
487 xml.endbranch();
488}

490void EffectMgr::getfromXML(XMLwrapper& xml)
491{
492 changeeffect(xml.getpar127("type", geteffect()));
493
494 if(!geteffect())
495 return;
496
497 preset = xml.getpar127("preset", preset);
498
499 if(xml.enterbranch("EFFECT_PARAMETERS")) {
500 for(int n = 0; n < 128; ++n) {
501 seteffectpar(n, 0); //erase effect parameter
502 if(xml.enterbranch("par_no", n) == 0)
503 continue;
504 int par = geteffectpar(n);
505 seteffectpar(n, xml.getpar127("par", par));
506 xml.exitbranch();
507 }
508 assert(filterpars);
509 if(xml.enterbranch("FILTER")) {
510 filterpars->getfromXML(xml);
511 xml.exitbranch();
512 }
513 xml.exitbranch();
514 }
515 cleanup();
516}
