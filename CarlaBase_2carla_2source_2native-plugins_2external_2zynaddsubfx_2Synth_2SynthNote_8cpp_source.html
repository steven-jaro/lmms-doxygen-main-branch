26SynthNote::Legato::Legato(const SYNTH_T &synth_, float freq, float vel, int port,
27 int note, bool quiet)
28 :synth(synth_)
29{
30 // Initialise some legato-specific vars
31 msg = LM_Norm;
32 fade.length = (int)(synth.samplerate_f * 0.005f); // 0.005f seems ok.
33 if(fade.length < 1)
34 fade.length = 1; // (if something's fishy)
35 fade.step = (1.0f / fade.length);
36 decounter = -10;
37 param.freq = freq;
38 param.vel = vel;
39 param.portamento = port;
40 param.midinote = note;
41 lastfreq = 0.0f;
42 silent = quiet;

45int SynthNote::Legato::update(LegatoParams pars)
46{
47 if(pars.externcall)
48 msg = LM_Norm;
49 if(msg != LM_CatchUp) {
50 lastfreq = param.freq;
51 param.freq = pars.frequency;
52 param.vel = pars.velocity;
53 param.portamento = pars.portamento;
54 param.midinote = pars.midinote;
55 if(msg == LM_Norm) {
56 if(silent) {
57 fade.m = 0.0f;
58 msg = LM_FadeIn;
59 }
60 else {
61 fade.m = 1.0f;
62 msg = LM_FadeOut;
63 return 1;
64 }
65 }
66 if(msg == LM_ToNorm)
67 msg = LM_Norm;
68 }
69 return 0;
70}

72void SynthNote::Legato::apply(SynthNote &note, float *outl, float *outr)
73{
74 if(silent) // Silencer
75 if(msg != LM_FadeIn) {
76 memset(outl, 0, synth.bufferbytes);
77 memset(outr, 0, synth.bufferbytes);
78 }
79 try {
80 switch (msg) {
81 case LM_CatchUp: // Continue the catch-up...
82 if (decounter == -10)
83 decounter = fade.length;
84 //Yea, could be done without the loop...
85 for (int i = 0; i < synth.buffersize; ++i) {
86 decounter--;
87 if (decounter < 1) {
88 // Catching-up done, we can finally set
89 // the note to the actual parameters.
90 decounter = -10;
91 msg = LM_ToNorm;
92 LegatoParams pars{param.freq, param.vel, param.portamento,
93 param.midinote, false};
94 note.legatonote(pars);
95 break;
96 }
97 }
98 break;
99 case LM_FadeIn: // Fade-in
100 if (decounter == -10)
101 decounter = fade.length;
102 silent = false;
103 for (int i = 0; i < synth.buffersize; ++i) {
104 decounter--;
105 if (decounter < 1) {
106 decounter = -10;
107 msg = LM_Norm;
108 break;
109 }
110 fade.m += fade.step;
111 outl[i] *= fade.m;
112 outr[i] *= fade.m;
113 }
114 break;
115 case LM_FadeOut: // Fade-out, then set the catch-up
116 if (decounter == -10)
117 decounter = fade.length;
118 for (int i = 0; i < synth.buffersize; ++i) {
119 decounter--;
120 if (decounter < 1) {
121 for (int j = i; j < synth.buffersize; ++j) {
122 outl[j] = 0.0f;
123 outr[j] = 0.0f;
124 }
125 decounter = -10;
126 silent = true;
127 // Fading-out done, now set the catch-up :
128 decounter = fade.length;
129 msg = LM_CatchUp;
130 //This freq should make this now silent note to catch-up/resync
131 //with the heard note for the same length it stayed at the
132 //previous freq during the fadeout.
133 float catchupfreq = param.freq * (param.freq / lastfreq);
134 LegatoParams pars{catchupfreq, param.vel, param.portamento,
135 param.midinote, false};
136 note.legatonote(pars);
137 break;
138 }
139 fade.m -= fade.step;
140 outl[i] *= fade.m;
141 outr[i] *= fade.m;
142 }
143 break;
144 default:
145 break;
146 }
147 } catch (std::bad_alloc &ba) {
148 std::cerr << "failed to apply legato: " << ba.what() << std::endl;
149 }
150}

152void SynthNote::setVelocity(float velocity_) {
153 legato.setSilent(true); //Let legato.update(...) returns 0.
154 LegatoParams pars{legato.getFreq(), velocity_,
155 legato.getPortamento(), legato.getMidinote(), true};
156 try {
157 legatonote(pars);
158 } catch (std::bad_alloc &ba) {
159 std::cerr << "failed to set velocity to legato note: " << ba.what() << std::endl;
160 }
161 legato.setDecounter(0); //avoid chopping sound due fade-in
162}
