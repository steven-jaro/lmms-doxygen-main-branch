256/*
257 * Get the frequency according the note number
258 */
259float Microtonal::getnotefreq(int note, int keyshift) const
260{
261 // in this function will appears many times things like this:
262 // var=(a+b*100)%b
263 // I had written this way because if I use var=a%b gives unwanted results when a<0
264 // This is the same with divisions.
265
266 if((Pinvertupdown != 0) && ((Pmappingenabled == 0) || (Penabled == 0)))
267 note = (int) Pinvertupdowncenter * 2 - note;
268
269 //compute global fine detune
270 float globalfinedetunerap =
271 powf(2.0f, (Pglobalfinedetune - 64.0f) / 1200.0f); //-64.0f .. 63.0f cents
272
273 if(Penabled == 0) //12tET
274 return powf(2.0f,
275 (note - PAnote
276 + keyshift) / 12.0f) * PAfreq * globalfinedetunerap;
277
278 int scaleshift =
279 ((int)Pscaleshift - 64 + (int) octavesize * 100) % octavesize;
280
281 //compute the keyshift
282 float rap_keyshift = 1.0f;
283 if(keyshift != 0) {
284 int kskey = (keyshift + (int)octavesize * 100) % octavesize;
285 int ksoct = (keyshift + (int)octavesize * 100) / octavesize - 100;
286 rap_keyshift = (kskey == 0) ? (1.0f) : (octave[kskey - 1].tuning);
287 rap_keyshift *= powf(octave[octavesize - 1].tuning, ksoct);
288 }
289
290 //if the mapping is enabled
291 if(Pmappingenabled) {
292 if((note < Pfirstkey) || (note > Plastkey))
293 return -1.0f;
294 //Compute how many mapped keys are from middle note to reference note
295 //and find out the proportion between the freq. of middle note and "A" note
296 int tmp = PAnote - Pmiddlenote, minus = 0;
297 if(tmp < 0) {
298 tmp = -tmp;
299 minus = 1;
300 }
301 int deltanote = 0;
302 for(int i = 0; i < tmp; ++i)
303 if(Pmapping[i % Pmapsize] >= 0)
304 deltanote++;
305 float rap_anote_middlenote =
306 (deltanote ==
307 0) ? (1.0f) : (octave[(deltanote - 1) % octavesize].tuning);
308 if(deltanote)
309 rap_anote_middlenote *=
310 powf(octave[octavesize - 1].tuning,
311 (deltanote - 1) / octavesize);
312 if(minus)
313 rap_anote_middlenote = 1.0f / rap_anote_middlenote;
314
315 //Convert from note (midi) to degree (note from the tunning)
316 int degoct =
317 (note - (int)Pmiddlenote + (int) Pmapsize
318 * 200) / (int)Pmapsize - 200;
319 int degkey = (note - Pmiddlenote + (int)Pmapsize * 100) % Pmapsize;
320 degkey = Pmapping[degkey];
321 if(degkey < 0)
322 return -1.0f; //this key is not mapped
323
324 //invert the keyboard upside-down if it is asked for
325 //TODO: do the right way by using Pinvertupdowncenter
326 if(Pinvertupdown != 0) {
327 degkey = octavesize - degkey - 1;
328 degoct = -degoct;
329 }
330 //compute the frequency of the note
331 degkey = degkey + scaleshift;
332 degoct += degkey / octavesize;
333 degkey %= octavesize;
334
335 float freq = (degkey == 0) ? (1.0f) : octave[degkey - 1].tuning;
336 freq *= powf(octave[octavesize - 1].tuning, degoct);
337 freq *= PAfreq / rap_anote_middlenote;
338 freq *= globalfinedetunerap;
339 if(scaleshift)
340 freq /= octave[scaleshift - 1].tuning;
341 return freq * rap_keyshift;
342 }
343 else { //if the mapping is disabled
344 int nt = note - PAnote + scaleshift;
345 int ntkey = (nt + (int)octavesize * 100) % octavesize;
346 int ntoct = (nt - ntkey) / octavesize;
347
348 float oct = octave[octavesize - 1].tuning;
349 float freq =
350 octave[(ntkey + octavesize - 1) % octavesize].tuning * powf(oct,
351 ntoct)
352 * PAfreq;
353 if(!ntkey)
354 freq /= oct;
355 if(scaleshift)
356 freq /= octave[scaleshift - 1].tuning;
357 freq *= globalfinedetunerap;
358 return freq * rap_keyshift;
359 }
360}

367bool Microtonal::operator!=(const Microtonal &micro) const
368{
369 //A simple macro to test equality MiCRotonal EQuals (not the perfect
370 //approach, but good enough)
371#define MCREQ(x) if(x != micro.x) \
372 return true
373
374 //for floats
375#define FMCREQ(x) if(!((x < micro.x + 0.0001f) && (x > micro.x - 0.0001f))) \
376 return true
377
378 MCREQ(Pinvertupdown);
379 MCREQ(Pinvertupdowncenter);
380 MCREQ(octavesize);
381 MCREQ(Penabled);
382 MCREQ(PAnote);
383 FMCREQ(PAfreq);
384 MCREQ(Pscaleshift);
385
386 MCREQ(Pfirstkey);
387 MCREQ(Plastkey);
388 MCREQ(Pmiddlenote);
389 MCREQ(Pmapsize);
390 MCREQ(Pmappingenabled);
391
392 for(int i = 0; i < 128; ++i)
393 MCREQ(Pmapping[i]);
394
395 for(int i = 0; i < octavesize; ++i) {
396 FMCREQ(octave[i].tuning);
397 MCREQ(octave[i].type);
398 MCREQ(octave[i].x1);
399 MCREQ(octave[i].x2);
400 }
401 if(strcmp((const char *)this->Pname, (const char *)micro.Pname))
402 return true;
403 if(strcmp((const char *)this->Pcomment, (const char *)micro.Pcomment))
404 return true;
405 MCREQ(Pglobalfinedetune);
406 return false;
407
408 //undefine macros, as they are no longer needed
409#undef MCREQ
410#undef FMCREQ
411}

414/*
415 * Convert a line to tunings; returns -1 if it ok
416 */
417int Microtonal::linetotunings(OctaveTuning &octave, const char *line)
418{
419 int x1 = -1, x2 = -1, type = -1;
420 float x = -1.0f, tmp, tuning = 1.0f;
421 if(strstr(line, "/") == NULL) {
422 if(strstr(line, ".") == NULL) { // M case (M=M/1)
423 sscanf(line, "%d", &x1);
424 x2 = 1;
425 type = 2; //division
426 }
427 else { // float number case
428 sscanf(line, "%f", &x);
429 if(x < 0.000001f)
430 return 1;
431 type = 1; //float type(cents)
432 }
433 }
434 else { // M/N case
435 sscanf(line, "%d/%d", &x1, &x2);
436 if((x1 < 0) || (x2 < 0))
437 return 1;
438 if(x2 == 0)
439 x2 = 1;
440 type = 2; //division
441 }
442
443 if(x1 <= 0)
444 x1 = 1; //not allow zero frequency sounds (consider 0 as 1)
445
446 //convert to float if the number are too big
447 if((type == 2)
448 && ((x1 > (128 * 128 * 128 - 1)) || (x2 > (128 * 128 * 128 - 1)))) {
449 type = 1;
450 x = ((float) x1) / x2;
451 }
452 switch(type) {
453 case 1:
454 x1 = (int) floor(x);
455 tmp = fmod(x, 1.0f);
456 x2 = (int) (floor(tmp * 1e6));
457 tuning = powf(2.0f, x / 1200.0f);
458 break;
459 case 2:
460 x = ((float)x1) / x2;
461 tuning = x;
462 break;
463 }
464
465 octave.tuning = tuning;
466 octave.type = type;
467 octave.x1 = x1;
468 octave.x2 = x2;
469
470 return -1; //ok
471}

473/*
474 * Convert the text to tunnings
475 */
476int Microtonal::texttotunings(const char *text)
477{
478 unsigned int k = 0, nl = 0;
479 char *lin = new char[MAX_LINE_SIZE + 1];
480 OctaveTuning tmpoctave[MAX_OCTAVE_SIZE];
481 while(k < strlen(text)) {
482 int i;
483 for(i = 0; i < MAX_LINE_SIZE; ++i) {
484 lin[i] = text[k++];
485 if(lin[i] < 0x20)
486 break;
487 }
488 lin[i] = '\0';
489 if(strlen(lin) == 0)
490 continue;
491 int err = linetotunings(tmpoctave[nl], lin);
492 if(err != -1) {
493 delete [] lin;
494 return nl; //Parse error
495 }
496 nl++;
497 }
498 delete [] lin;
499 if(nl > MAX_OCTAVE_SIZE)
500 nl = MAX_OCTAVE_SIZE;
501 if(nl == 0)
502 return -2; //the input is empty
503 octavesize = nl;
504 for(int i = 0; i < octavesize; ++i) {
505 octave[i].tuning = tmpoctave[i].tuning;
506 octave[i].type = tmpoctave[i].type;
507 octave[i].x1 = tmpoctave[i].x1;
508 octave[i].x2 = tmpoctave[i].x2;
509 }
510 return -1; //ok
511}

513/*
514 * Convert the text to mapping
515 */
516void Microtonal::texttomapping(const char *text)
517{
518 unsigned int i, k = 0;
519 char *lin;
520 lin = new char[MAX_LINE_SIZE + 1];
521 for(i = 0; i < 128; ++i)
522 Pmapping[i] = -1;
523 int tx = 0;
524 while(k < strlen(text)) {
525 for(i = 0; i < MAX_LINE_SIZE; ++i) {
526 lin[i] = text[k++];
527 if(lin[i] < 0x20)
528 break;
529 }
530 lin[i] = '\0';
531 if(strlen(lin) == 0)
532 continue;
533
534 int tmp = 0;
535 if(sscanf(lin, "%d", &tmp) == 0)
536 tmp = -1;
537 if(tmp < -1)
538 tmp = -1;
539 Pmapping[tx] = tmp;
540
541 if((tx++) > 127)
542 break;
543 }
544 delete [] lin;
545
546 if(tx == 0)
547 tx = 1;
548 Pmapsize = tx;
549}

578/*
579 * Loads the tunnings from a scl file
580 */
581int Microtonal::loadscl(SclInfo &scl, const char *filename)
582{
583 FILE *file = fopen(filename, "r");
584 char tmp[500];
585 OctaveTuning tmpoctave[MAX_OCTAVE_SIZE];
586
587 if(!file)
588 return 2;
589
590 fseek(file, 0, SEEK_SET);
591
592 //loads the short description
593 if(loadline(file, &tmp[0]) != 0)
594 return 2;
595
596 for(int i = 0; i < 500; ++i)
597 if(tmp[i] < 32)
598 tmp[i] = 0;
599
600 snprintf(scl.Pname, MICROTONAL_MAX_NAME_LEN, "%s", tmp);
601 snprintf(scl.Pcomment, MICROTONAL_MAX_NAME_LEN, "%s", tmp);
602
603 //loads the number of the notes
604 if(loadline(file, &tmp[0]) != 0)
605 return 2;
606 int nnotes = MAX_OCTAVE_SIZE;
607 sscanf(&tmp[0], "%d", &nnotes);
608 if(nnotes > MAX_OCTAVE_SIZE)
609 return 2;
610
611 //load the tunnings
612 for(int nline = 0; nline < nnotes; ++nline) {
613 if(loadline(file, &tmp[0]) != 0)
614 return 2;
615 linetotunings(tmpoctave[nline], tmp);
616 }
617 fclose(file);
618
619 scl.octavesize = nnotes;
620 for(int i = 0; i < scl.octavesize; ++i) {
621 scl.octave[i].tuning = tmpoctave[i].tuning;
622 scl.octave[i].type = tmpoctave[i].type;
623 scl.octave[i].x1 = tmpoctave[i].x1;
624 scl.octave[i].x2 = tmpoctave[i].x2;
625 }
626
627 return 0;
628}

631/*
632 * Loads the mapping from a kbm file
633 */
634int Microtonal::loadkbm(KbmInfo &kbm, const char *filename)
635{
636 FILE *file = fopen(filename, "r");
637 int x;
638 float tmpPAfreq = 440.0f;
639 char tmp[500];
640
641 if(!file)
642 return 2;
643
644 fseek(file, 0, SEEK_SET);
645 //loads the mapsize
646 if(loadline(file, tmp) != 0 || sscanf(tmp, "%d", &x) == 0)
647 return 2;
648 kbm.Pmapsize = limit(x, 0, 127);
649
650 //loads first MIDI note to retune
651 if(loadline(file, tmp) != 0 || sscanf(tmp, "%d", &x) == 0)
652 return 2;
653 kbm.Pfirstkey = limit(x, 0, 127);
654
655 //loads last MIDI note to retune
656 if(loadline(file, tmp) != 0 || sscanf(tmp, "%d", &x) == 0)
657 return 2;
658 kbm.Plastkey = limit(x, 0, 127);
659
660 //loads last the middle note where scale fro scale degree=0
661 if(loadline(file, tmp) != 0 || sscanf(tmp, "%d", &x) == 0)
662 return 2;
663 kbm.Pmiddlenote = limit(x, 0, 127);
664
665 //loads the reference note
666 if(loadline(file, tmp) != 0 || sscanf(tmp, "%d", &x) == 0)
667 return 2;
668 kbm.PAnote = limit(x,0,127);
669
670 //loads the reference freq.
671 if(loadline(file, tmp) != 0 || sscanf(tmp, "%f", &tmpPAfreq) == 0)
672 return 2;
673 kbm.PAfreq = tmpPAfreq;
674
675 //the scale degree(which is the octave) is not loaded,
676 //it is obtained by the tunnings with getoctavesize() method
677 if(loadline(file, &tmp[0]) != 0)
678 return 2;
679
680 //load the mappings
681 if(kbm.Pmapsize != 0) {
682 for(int nline = 0; nline < kbm.Pmapsize; ++nline) {
683 if(loadline(file, tmp) != 0)
684 return 2;
685 if(sscanf(tmp, "%d", &x) == 0)
686 x = -1;
687 kbm.Pmapping[nline] = x;
688 }
689 kbm.Pmappingenabled = 1;
690 }
691 else {
692 kbm.Pmappingenabled = 0;

754void Microtonal::getfromXML(XMLwrapper& xml)
755{
756 xml.getparstr("name", (char *) Pname, MICROTONAL_MAX_NAME_LEN);
757 xml.getparstr("comment", (char *) Pcomment, MICROTONAL_MAX_NAME_LEN);
758
759 Pinvertupdown = xml.getparbool("invert_up_down", Pinvertupdown);
760 Pinvertupdowncenter = xml.getpar127("invert_up_down_center",
761 Pinvertupdowncenter);
762
763 Penabled = xml.getparbool("enabled", Penabled);
764 Pglobalfinedetune = xml.getpar127("global_fine_detune", Pglobalfinedetune);
765
766 PAnote = xml.getpar127("a_note", PAnote);
767 PAfreq = xml.getparreal("a_freq", PAfreq, 1.0f, 10000.0f);
768
769 if(xml.enterbranch("SCALE")) {
770 Pscaleshift = xml.getpar127("scale_shift", Pscaleshift);
771 Pfirstkey = xml.getpar127("first_key", Pfirstkey);
772 Plastkey = xml.getpar127("last_key", Plastkey);
773 Pmiddlenote = xml.getpar127("middle_note", Pmiddlenote);
774
775 if(xml.enterbranch("OCTAVE")) {
776 octavesize = xml.getpar127("octave_size", octavesize);
777 for(int i = 0; i < octavesize; ++i) {
778 if(xml.enterbranch("DEGREE", i) == 0)
779 continue;
780 octave[i].x2 = 0;
781 octave[i].tuning = xml.getparreal("cents", octave[i].tuning);
782 octave[i].x1 = xml.getpar127("numerator", octave[i].x1);
783 octave[i].x2 = xml.getpar127("denominator", octave[i].x2);
784
785 if(octave[i].x2 != 0)
786 octave[i].type = 2;
787 else {
788 octave[i].type = 1;
789 //populate fields for display
790 float x = logf(octave[i].tuning) / LOG_2 * 1200.0f;
791 octave[i].x1 = (int) floor(x);
792 octave[i].x2 = (int) (floor((x-octave[i].x1) * 1.0e6));
793 }
794
795
796 xml.exitbranch();
797 }
798 xml.exitbranch();
799 }
800
801 if(xml.enterbranch("KEYBOARD_MAPPING")) {
802 Pmapsize = xml.getpar127("map_size", Pmapsize);
803 Pmappingenabled = xml.getpar127("mapping_enabled", Pmappingenabled);
804 for(int i = 0; i < Pmapsize; ++i) {
805 if(xml.enterbranch("KEYMAP", i) == 0)
806 continue;
807 Pmapping[i] = xml.getpar127("degree", Pmapping[i]);
808 xml.exitbranch();
809 }
810 xml.exitbranch();
811 }
812 xml.exitbranch();
813 }
814 apply();
815}
