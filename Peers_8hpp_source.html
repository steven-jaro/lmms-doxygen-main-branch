31// SessionMembershipCallback is invoked when any change to session
32// membership occurs (when any peer joins or leaves a session)
33//
34// SessionTimelineCallback is invoked with a session id and a timeline
35// whenever a new combination of these values is seen
36//
37// SessionStartStopStateCallback is invoked with a session id and a startStopState
38// whenever a new combination of these values is seen

61 // The set of peers for a given session, ordered by (peerId, addr).
62 // The result will possibly contain multiple entries for the same
63 // peer if it is visible through multiple gateways.
64 std::vector<Peer> sessionPeers(const SessionId& sid) const
65 {
66 using namespace std;
67 vector<Peer> result;
68 auto& peerVec = mpImpl->mPeers;
69 copy_if(begin(peerVec), end(peerVec), back_inserter(result), SessionMemberPred{sid});
70 return result;
71 }
72
73 // Number of individual for a given session.
74 std::size_t uniqueSessionPeerCount(const SessionId& sid) const
75 {
76 using namespace std;
77 auto peerVec = sessionPeers(sid);
78 auto last = unique(begin(peerVec), end(peerVec),
79 [](const Peer& a, const Peer& b) { return a.first.ident() == b.first.ident(); });
80 return static_cast<size_t>(distance(begin(peerVec), last));
81 }
82
83 void setSessionTimeline(const SessionId& sid, const Timeline& tl)
84 {
85 // Set the cached timeline for all peers to a new client-specified
86 // timeline. When we make a timeline change, we do so
87 // optimistically and clients assume that all peers in a session
88 // have adopted the newly specified timeline. We must represent
89 // this in our cache or else we risk failing to notify about a
90 // higher-priority peer timeline that was already seen.
91 for (auto& peer : mpImpl->mPeers)
92 {
93 if (peer.first.sessionId() == sid)
94 {
95 peer.first.nodeState.timeline = tl;
96 }
97 }
98 }
99
100 // Purge all cached peers that are members of the given session
101 void forgetSession(const SessionId& sid)
102 {
103 using namespace std;
104 auto& peerVec = mpImpl->mPeers;
105 peerVec.erase(
106 remove_if(begin(peerVec), end(peerVec), SessionMemberPred{sid}), end(peerVec));
107 }
108
109 void resetPeers()
110 {
111 mpImpl->mPeers.clear();
112 }

213 void sawPeerOnGateway(PeerState peerState, asio::ip::address gatewayAddr)
214 {
215 using namespace std;
216
217 const auto peerSession = peerState.sessionId();
218 const auto peerTimeline = peerState.timeline();
219 const auto peerStartStopState = peerState.startStopState();
220 bool isNewSessionTimeline = false;
221 bool isNewSessionStartStopState = false;
222 bool didSessionMembershipChange = false;
223 {
224 isNewSessionTimeline = !sessionTimelineExists(peerSession, peerTimeline);
225 isNewSessionStartStopState =
226 !sessionStartStopStateExists(peerSession, peerStartStopState);
227
228 auto peer = make_pair(move(peerState), move(gatewayAddr));
229 const auto idRange = equal_range(begin(mPeers), end(mPeers), peer, PeerIdComp{});
230
231 if (idRange.first == idRange.second)
232 {
233 // This peer is not currently known on any gateway
234 didSessionMembershipChange = true;
235 mPeers.insert(move(idRange.first), move(peer));
236 }
237 else
238 {
239 // We've seen this peer before... does it have a new session?
240 didSessionMembershipChange =
241 all_of(idRange.first, idRange.second, [&peerSession](const Peer& test) {
242 return test.first.sessionId() != peerSession;
243 });
244
245 // was it on this gateway?
246 const auto addrRange =
247 equal_range(idRange.first, idRange.second, peer, AddrComp{});
248
249 if (addrRange.first == addrRange.second)
250 {
251 // First time on this gateway, add it
252 mPeers.insert(move(addrRange.first), move(peer));
253 }
254 else
255 {
256 // We have an entry for this peer on this gateway, update it
257 *addrRange.first = move(peer);
258 }
259 }
260 } // end lock
261
262 // Invoke callbacks outside the critical section
263 if (isNewSessionTimeline)
264 {
265 mSessionTimelineCallback(peerSession, peerTimeline);
266 }
267
268 // Pass the start stop state to the Controller after it processed the timeline.
269 // A new timeline can cause a session Id change which will prevent processing the
270 // new start stop state. By handling the start stop state after the timeline we
271 // assure that the start stop state is processed with the correct session Id.
272 if (isNewSessionStartStopState)
273 {
274 mSessionStartStopStateCallback(peerSession, peerStartStopState);
275 }
276
277 if (didSessionMembershipChange)
278 {
279 mSessionMembershipCallback();
280 }
281 }

283 void peerLeftGateway(const NodeId& nodeId, const asio::ip::address& gatewayAddr)
284 {
285 using namespace std;
286
287 bool didSessionMembershipChange = false;
288 {
289 auto it = find_if(begin(mPeers), end(mPeers), [&](const Peer& peer) {
290 return peer.first.ident() == nodeId && peer.second == gatewayAddr;
291 });
292
293 if (it != end(mPeers))
294 {
295 mPeers.erase(move(it));
296 didSessionMembershipChange = true;
297 }
298 } // end lock
299
300 if (didSessionMembershipChange)
301 {
302 mSessionMembershipCallback();
303 }
304 }
305
306 void gatewayClosed(const asio::ip::address& gatewayAddr)
307 {
308 using namespace std;
309
310 {
311 mPeers.erase(
312 remove_if(begin(mPeers), end(mPeers),
313 [&gatewayAddr](const Peer& peer) { return peer.second == gatewayAddr; }),
314 end(mPeers));
315 } // end lock
316
317 mSessionMembershipCallback();
318 }

320 template <typename Predicate>
321 bool hasPeerWith(const SessionId& sessionId, Predicate predicate)
322 {
323 using namespace std;
324 return find_if(begin(mPeers), end(mPeers), [&](const Peer& peer) {
325 return peer.first.sessionId() == sessionId && predicate(peer.first);
326 }) != end(mPeers);
327 }
328
329 bool sessionTimelineExists(const SessionId& session, const Timeline& timeline)
330 {
331 return hasPeerWith(session,
332 [&](const PeerState& peerState) { return peerState.timeline() == timeline; });
333 }
334
335 bool sessionStartStopStateExists(
336 const SessionId& sessionId, const StartStopState& startStopState)
337 {
338 return hasPeerWith(sessionId, [&](const PeerState& peerState) {
339 return peerState.startStopState() == startStopState;
340 });
341 }

343 struct PeerIdComp
344 {
345 bool operator()(const Peer& lhs, const Peer& rhs) const
346 {
347 return lhs.first.ident() < rhs.first.ident();
348 }
349 };
350
351 struct AddrComp
352 {
353 bool operator()(const Peer& lhs, const Peer& rhs) const
354 {
355 return lhs.second < rhs.second;
356 }
357 };
