59static
60std::array<int8_t, 256> createCharIndexTable()
61{
62 std::array<int8_t, 256> table;
63 table.fill(-1);
64 int8_t index = 0;
65 for (uint8_t c = 'A'; c <= 'Z'; ++c) table[c] = index++;
66 for (uint8_t c = 'a'; c <= 'z'; ++c) table[c] = index++;
67 for (uint8_t c = '0'; c <= '9'; ++c) table[c] = index++;
68 table['+'] = index++;
69 table['/'] = index++;
70 return table;
71}

73static const std::array<int8_t, 256> kCharIndexTable = createCharIndexTable();

80static inline
81std::vector<uint8_t> d_getChunkFromBase64String(const char* const base64string, std::size_t base64stringLen = ~std::size_t(0))
82{
83 std::vector<uint8_t> ret;
84
85 if (! base64string)
86 return ret;
87
88 uint32_t i=0, j=0;
89 uint32_t charArray3[3], charArray4[4];
90
91 if (base64stringLen == ~std::size_t(0))
92 base64stringLen = std::strlen(base64string);
93
94 ret.reserve(base64stringLen*3/4 + 4);
95
96 for (std::size_t l=0; l<base64stringLen; ++l)
97 {
98 const unsigned char c = base64string[l];
99
100 if (c == '\0' || c == '=')
101 break;
102 if (DistrhoBase64Helpers::kCharIndexTable[c] == -1)
103 continue;
104
105 charArray4[i++] = static_cast<uint32_t>(c);
106
107 if (i == 4)
108 {
109 for (i=0; i<4; ++i)
110 charArray4[i] = static_cast<uint8_t>(DistrhoBase64Helpers::kCharIndexTable[charArray4[i]]);
111
112 charArray3[0] = (charArray4[0] << 2) + ((charArray4[1] & 0x30) >> 4);
113 charArray3[1] = ((charArray4[1] & 0xf) << 4) + ((charArray4[2] & 0x3c) >> 2);
114 charArray3[2] = ((charArray4[2] & 0x3) << 6) + charArray4[3];
115
116 for (i=0; i<3; ++i)
117 ret.push_back(static_cast<uint8_t>(charArray3[i]));
118
119 i = 0;
120 }
121 }
122
123 if (i != 0)
124 {
125 for (j=0; j<i && j<4; ++j)
126 charArray4[j] = static_cast<uint8_t>(DistrhoBase64Helpers::kCharIndexTable[charArray4[j]]);
127
128 for (j=i; j<4; ++j)
129 charArray4[j] = 0;
130
131 charArray3[0] = (charArray4[0] << 2) + ((charArray4[1] & 0x30) >> 4);
132 charArray3[1] = ((charArray4[1] & 0xf) << 4) + ((charArray4[2] & 0x3c) >> 2);
133 charArray3[2] = ((charArray4[2] & 0x3) << 6) + charArray4[3];
134
135 for (j=0; i>0 && j<i-1; j++)
136 ret.push_back(static_cast<uint8_t>(charArray3[j]));
137 }
138
139 return ret;
140}

142static inline
143std::string d_getBase64StringFromChunk(const void* const data, std::size_t dataSize)
144{
145 static const char* const kBase64Chars =
146 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
147 "abcdefghijklmnopqrstuvwxyz"
148 "0123456789+/";
149
150 const uint8_t* bytesToEncode((const uint8_t*)data);
151
152 uint32_t i=0, j=0;
153 uint32_t charArray3[3], charArray4[4];
154
155 std::string ret;
156 ret.reserve(((dataSize + 2) / 3) * 4);
157
158 for (std::size_t s=0; s<dataSize; ++s)
159 {
160 charArray3[i++] = *(bytesToEncode++);
161
162 if (i == 3)
163 {
164 charArray4[0] = (charArray3[0] & 0xfc) >> 2;
165 charArray4[1] = ((charArray3[0] & 0x03) << 4) + ((charArray3[1] & 0xf0) >> 4);
166 charArray4[2] = ((charArray3[1] & 0x0f) << 2) + ((charArray3[2] & 0xc0) >> 6);
167 charArray4[3] = charArray3[2] & 0x3f;
168
169 for (i=0; i<4; ++i)
170 ret.push_back(kBase64Chars[charArray4[i]]);
171
172 i = 0;
173 }
174 }
175
176 if (i != 0)
177 {
178 for (j=i; j<3; ++j)
179 charArray3[j] = '\0';
180
181 charArray4[0] = (charArray3[0] & 0xfc) >> 2;
182 charArray4[1] = ((charArray3[0] & 0x03) << 4) + ((charArray3[1] & 0xf0) >> 4);
183 charArray4[2] = ((charArray3[1] & 0x0f) << 2) + ((charArray3[2] & 0xc0) >> 6);
184 charArray4[3] = charArray3[2] & 0x3f;
185
186 for (j=0; j<4 && i<3 && j<i+1; ++j)
187 ret.push_back(kBase64Chars[charArray4[j]]);
188
189 for (; i++ < 3;)
190 ret.push_back('=');
191 }
192
193 return ret;
194}
