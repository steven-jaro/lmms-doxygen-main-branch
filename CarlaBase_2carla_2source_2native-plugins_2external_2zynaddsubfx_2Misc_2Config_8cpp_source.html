302{

304 if(xmlcfg.loadXMLfile(filename) < 0)

324 cfg.BankUIAutoClose,
325 0,

327

329 cfg.GzipCompression,

331 9);
332

334 cfg.Interpolation = xmlcfg.getpar("interpolation",
335 cfg.Interpolation,

338

340 cfg.CheckPADsynth,
341 0,

345 cfg.IgnoreProgramChange,
346 0,

351 cfg.UserInterfaceMode,

356 0,
357 10);

360 for(int i = 0; i < MAX_BANK_ROOT_DIRS; ++i)

362 cfg.bankRootDirList[i] = xmlcfg.getparstr("bank_root", "");

365
366 //get preset root dirs

370 xmlcfg.exitbranch();
371 }
372
373 //Get favs
374 for(int i = 0; i < MAX_BANK_ROOT_DIRS; ++i)
375 if(xmlcfg.enterbranch("FAVSROOT", i)) {
376 cfg.favoriteList[i] = xmlcfg.getparstr("favoirtes_root", "");
377 xmlcfg.exitbranch();
378 }
379
380 //linux stuff
381 xmlcfg.getparstr("linux_oss_wave_out_dev",
382 cfg.oss_devs.linux_wave_out,
383 MAX_STRING_SIZE);
384 xmlcfg.getparstr("linux_oss_seq_in_dev",
385 cfg.oss_devs.linux_seq_in,
386 MAX_STRING_SIZE);
387
388 //windows stuff
389 cfg.WindowsWaveOutId = xmlcfg.getpar("windows_wave_out_id",
390 cfg.WindowsWaveOutId,
391 0,
392 winwavemax);
393 cfg.WindowsMidiInId = xmlcfg.getpar("windows_midi_in_id",
394 cfg.WindowsMidiInId,
395 0,
396 winmidimax);
397
398 xmlcfg.exitbranch();
399 }
400
401 cfg.OscilSize = (int) powf(2, ceil(logf(cfg.OscilSize - 1.0f) / logf(2.0f)));

404void Config::saveConfig(const char *filename) const
405{
406 XMLwrapper *xmlcfg = new XMLwrapper();
407
408 xmlcfg->beginbranch("CONFIGURATION");
409
410 xmlcfg->addpar("sample_rate", cfg.SampleRate);
411 xmlcfg->addpar("sound_buffer_size", cfg.SoundBufferSize);
412 xmlcfg->addpar("oscil_size", cfg.OscilSize);
413 xmlcfg->addpar("swap_stereo", cfg.SwapStereo);
414 xmlcfg->addpar("bank_window_auto_close", cfg.BankUIAutoClose);
415
416 xmlcfg->addpar("gzip_compression", cfg.GzipCompression);
417
418 xmlcfg->addpar("check_pad_synth", cfg.CheckPADsynth);
419 xmlcfg->addpar("ignore_program_change", cfg.IgnoreProgramChange);
420
421 xmlcfg->addparstr("bank_current", cfg.currentBankDir);
422
423 xmlcfg->addpar("user_interface_mode", cfg.UserInterfaceMode);
424 xmlcfg->addpar("virtual_keyboard_layout", cfg.VirKeybLayout);
425
426
427 for(int i = 0; i < MAX_BANK_ROOT_DIRS; ++i)
428 if(!cfg.bankRootDirList[i].empty()) {
429 xmlcfg->beginbranch("BANKROOT", i);
430 xmlcfg->addparstr("bank_root", cfg.bankRootDirList[i]);
431 xmlcfg->endbranch();
432 }
433
434 for(int i = 0; i < MAX_BANK_ROOT_DIRS; ++i)
435 if(!cfg.presetsDirList[i].empty()) {
436 xmlcfg->beginbranch("PRESETSROOT", i);
437 xmlcfg->addparstr("presets_root", cfg.presetsDirList[i]);
438 xmlcfg->endbranch();
439 }
440
441 for(int i = 0; i < MAX_BANK_ROOT_DIRS; ++i)
442 if(!cfg.favoriteList[i].empty()) {
443 xmlcfg->beginbranch("FAVSROOT", i);
444 xmlcfg->addparstr("favoirtes_root", cfg.favoriteList[i]);
445 xmlcfg->endbranch();
446 }
447
448 xmlcfg->addpar("interpolation", cfg.Interpolation);
449
450 //linux stuff
451 xmlcfg->addparstr("linux_oss_wave_out_dev", cfg.oss_devs.linux_wave_out);
452 xmlcfg->addparstr("linux_oss_seq_in_dev", cfg.oss_devs.linux_seq_in);
453
454 //windows stuff
455 xmlcfg->addpar("windows_wave_out_id", cfg.WindowsWaveOutId);
456 xmlcfg->addpar("windows_midi_in_id", cfg.WindowsMidiInId);
457
458 xmlcfg->endbranch();
459
460 // for some reason (which one?), the gzip compression is bashed to 0
461 xmlcfg->saveXMLfile(filename, 0);
462
463 delete (xmlcfg);
464}
