47void Effect::crossover(float &a, float &b, float crossover)
48{
49 float tmpa = a;
50 float tmpb = b;
51 a = tmpa * (1.0f - crossover) + tmpb * crossover;
52 b = tmpb * (1.0f - crossover) + tmpa * crossover;
53}

55void Effect::setpanning(char Ppanning_)
56{
57 Ppanning = Ppanning_;

59 pangainL = cosf(t * PI / 2.0f);
60 pangainR = cosf((1.0f - t) * PI / 2.0f);
