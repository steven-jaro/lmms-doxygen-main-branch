53 struct on_invoker_exit
54 {
55 invoker* this_;
56
57 ~on_invoker_exit()
58 {
59 this_->impl_->mutex_->lock();
60 this_->impl_->ready_queue_.push(this_->impl_->waiting_queue_);
61 bool more_handlers = this_->impl_->locked_ =
62 !this_->impl_->ready_queue_.empty();
63 this_->impl_->mutex_->unlock();
64
65 if (more_handlers)
66 {
67 Executor ex(this_->work_.get_executor());
68 recycling_allocator<void> allocator;
69 ex.post(ASIO_MOVE_CAST(invoker)(*this_), allocator);
70 }
71 }
72 };
73
74 void operator()()
75 {
76 // Indicate that this strand is executing on the current thread.
77 call_stack<strand_impl>::context ctx(impl_.get());
78
79 // Ensure the next handler, if any, is scheduled on block exit.
80 on_invoker_exit on_exit = { this };
81 (void)on_exit;
82
83 // Run all ready handlers. No lock is required since the ready queue is
84 // accessed only within the strand.
85 asio::error_code ec;
86 while (scheduler_operation* o = impl_->ready_queue_.front())
87 {
88 impl_->ready_queue_.pop();
89 o->complete(impl_.get(), ec, 0);
90 }
91 }
92
93private:
94 implementation_type impl_;
95 executor_work_guard<Executor> work_;
96};

98template <typename Executor, typename Function, typename Allocator>
99void strand_executor_service::dispatch(const implementation_type& impl,
100 Executor& ex, ASIO_MOVE_ARG(Function) function, const Allocator& a)
101{
102 typedef typename decay<Function>::type function_type;
103
104 // If we are already in the strand then the function can run immediately.
105 if (call_stack<strand_impl>::contains(impl.get()))
106 {
107 // Make a local, non-const copy of the function.
108 function_type tmp(ASIO_MOVE_CAST(Function)(function));
109
110 fenced_block b(fenced_block::full);
111 asio_handler_invoke_helpers::invoke(tmp, tmp);
112 return;
113 }
114
115 // Allocate and construct an operation to wrap the function.
116 typedef executor_op<function_type, Allocator> op;
117 typename op::ptr p = { detail::addressof(a), op::ptr::allocate(a), 0 };
118 p.p = new (p.v) op(ASIO_MOVE_CAST(Function)(function), a);
119
120 ASIO_HANDLER_CREATION((impl->service_->context(), *p.p,
121 "strand_executor", impl.get(), 0, "dispatch"));
122
123 // Add the function to the strand and schedule the strand if required.
124 bool first = enqueue(impl, p.p);
125 p.v = p.p = 0;
126 if (first)
127 ex.dispatch(invoker<Executor>(impl, ex), a);
128}
