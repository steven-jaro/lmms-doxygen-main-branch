79void PresetsStore::scanforpresets()
80{
81 clearpresets();
82 string ftype = ".xpz";
83
84 for(int i = 0; i < MAX_BANK_ROOT_DIRS; ++i) {
85 if(config.cfg.presetsDirList[i].empty())
86 continue;
87
88 //open directory
89 string dirname = config.cfg.presetsDirList[i];
90 DIR *dir = opendir(dirname.c_str());
91 if(dir == NULL)
92 continue;
93 struct dirent *fn;
94
95 //check all files in directory
96 while((fn = readdir(dir))) {
97 string filename = fn->d_name;
98 if(filename.find(ftype) == string::npos)
99 continue;
100
101 //ensure proper path is formed
102 char tmpc = dirname[dirname.size() - 1];
103 const char *tmps;
104 if((tmpc == '/') || (tmpc == '\\'))
105 tmps = "";
106 else
107 tmps = "/";
108
109 string location = "" + dirname + tmps + filename;
110
111 //trim file type off of name
112 string name_type = filename.substr(0, filename.find(ftype));
113
114 size_t tmp = name_type.find_last_of(".");
115 if(tmp == string::npos)
116 continue;
117 string type = name_type.substr(tmp+1);
118 string name = name_type.substr(0, tmp);
119
120 //put on list
121 presets.push_back(presetstruct{location, name, type});
122 }
123
124 closedir(dir);
125 }
126
127 //sort the presets
128 sort(presets.begin(), presets.end());
129}
