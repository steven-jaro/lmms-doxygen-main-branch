32struct Clock
33{
34 using Ticks = std::uint64_t;
35 using Micros = std::chrono::microseconds;
36
37 Clock()
38 {
39 mach_timebase_info_data_t timeInfo;
40 mach_timebase_info(&timeInfo);
41 // numer / denom gives nanoseconds, we want microseconds
42 mTicksToMicros = timeInfo.numer / (timeInfo.denom * 1000.);
43 }
44
45 Micros ticksToMicros(const Ticks ticks) const
46 {
47 return Micros{llround(mTicksToMicros * ticks)};
48 }
49
50 Ticks microsToTicks(const Micros micros) const
51 {
52 return static_cast<Ticks>(micros.count() / mTicksToMicros);
53 }
54
55 Ticks ticks() const
56 {
57 return mach_absolute_time();
58 }
59
60 std::chrono::microseconds micros() const
61 {
62 return ticksToMicros(ticks());
63 }
64
65 double mTicksToMicros;
66};
