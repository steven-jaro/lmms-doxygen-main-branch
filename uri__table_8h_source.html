44static LV2_URID
45uri_table_map(LV2_URID_Map_Handle handle,
46 const char* uri)
47{
48 URITable* table = (URITable*)handle;
49 for (size_t i = 0; i < table->n_uris; ++i) {
50 if (!strcmp(table->uris[i], uri)) {
51 return i + 1;
52 }
53 }
54
55 const size_t len = strlen(uri);
56 table->uris = (char**)realloc(table->uris, ++table->n_uris * sizeof(char*));
57 table->uris[table->n_uris - 1] = (char*)malloc(len + 1);
58 memcpy(table->uris[table->n_uris - 1], uri, len + 1);
59 return table->n_uris;
60}

62static const char*
63uri_table_unmap(LV2_URID_Map_Handle handle,
64 LV2_URID urid)
65{
66 URITable* table = (URITable*)handle;
67 if (urid > 0 && urid <= table->n_uris) {
68 return table->uris[urid - 1];
69 }
70 return NULL;
71}
