55template <typename IoObjectService,
56 typename Executor = io_context::executor_type>
57class io_object_impl
58{
59public:
60 // The type of the service that will be used to provide I/O operations.
61 typedef IoObjectService service_type;

63 // The underlying implementation type of I/O object.

65
66 // The type of the executor associated with the object.

69 // The type of executor to be used when implementing asynchronous operations.

72 // Construct an I/O object using an executor.
73 explicit io_object_impl(const executor_type& ex)
74 : service_(&asio::use_service<IoObjectService>(ex.context())),
75 implementation_executor_(ex, (is_native_io_executor)(ex))

77 service_->construct(implementation_);
78 }

87 is_same<ExecutionContext, io_context>::value)

119#if defined(ASIO_HAS_MOVE)
120 // Move-assign an I/O object.
