24FFTwrapper::FFTwrapper(int fftsize_)
25{
26 //first one will spawn the mutex (yeah this may be a race itself)
27 if(!mutex) {
28 mutex = new pthread_mutex_t;
29 pthread_mutex_init(mutex, NULL);
30 }
31
32
33 fftsize = fftsize_;
34 time = new fftw_real[fftsize];
35 fft = new fftw_complex[fftsize + 1];
36 pthread_mutex_lock(mutex);
37 planfftw = fftw_plan_dft_r2c_1d(fftsize,
38 time,
39 fft,
40 FFTW_ESTIMATE);
41 planfftw_inv = fftw_plan_dft_c2r_1d(fftsize,
42 fft,
43 time,
44 FFTW_ESTIMATE);
45 pthread_mutex_unlock(mutex);
46}
47
48FFTwrapper::~FFTwrapper()
49{
50 pthread_mutex_lock(mutex);
51 fftw_destroy_plan(planfftw);
52 fftw_destroy_plan(planfftw_inv);
53 pthread_mutex_unlock(mutex);
54
55 delete [] time;
56 delete [] fft;
57}

59void FFTwrapper::smps2freqs(const float *smps, fft_t *freqs)
60{
61 //Load data
62 for(int i = 0; i < fftsize; ++i)
63 time[i] = static_cast<double>(smps[i]);
64
65 //DFT
66 fftw_execute(planfftw);
67
68 //Grab data
69 memcpy((void *)freqs, (const void *)fft, fftsize * sizeof(double));
70}

72void FFTwrapper::freqs2smps(const fft_t *freqs, float *smps)
73{
74 //Load data
75 memcpy((void *)fft, (const void *)freqs, fftsize * sizeof(double));
76
77 //clear unused freq channel
78 fft[fftsize / 2][0] = 0.0f;
79 fft[fftsize / 2][1] = 0.0f;
80
81 //IDFT
82 fftw_execute(planfftw_inv);
83
84 //Grab data
85 for(int i = 0; i < fftsize; ++i)
86 smps[i] = static_cast<float>(time[i]);
87}
