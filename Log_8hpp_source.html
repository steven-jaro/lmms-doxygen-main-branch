163 template <typename Stream>
164 Stream logTimestamp(Stream&& streamRef) const
165 {
166 using namespace std::chrono;
167 Stream stream = std::forward<Stream>(streamRef);
168 stream << "|"
169 << duration_cast<milliseconds>(system_clock::now().time_since_epoch()).count()
170 << "ms| ";
171 return stream;
