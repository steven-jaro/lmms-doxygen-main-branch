57template <typename Handler, typename IoExecutor>
58class reactive_socket_connect_op : public reactive_socket_connect_op_base
59{
60public:
61 ASIO_DEFINE_HANDLER_PTR(reactive_socket_connect_op);
62
63 reactive_socket_connect_op(socket_type socket,
64 Handler& handler, const IoExecutor& io_ex)
65 : reactive_socket_connect_op_base(socket,
66 &reactive_socket_connect_op::do_complete),
67 handler_(ASIO_MOVE_CAST(Handler)(handler)),
68 io_executor_(io_ex)
69 {
70 handler_work<Handler, IoExecutor>::start(handler_, io_executor_);
71 }
72
73 static void do_complete(void* owner, operation* base,
74 const asio::error_code& /*ec*/,
75 std::size_t /*bytes_transferred*/)
76 {
77 // Take ownership of the handler object.
78 reactive_socket_connect_op* o
79 (static_cast<reactive_socket_connect_op*>(base));
80 ptr p = { asio::detail::addressof(o->handler_), o, o };
81 handler_work<Handler, IoExecutor> w(o->handler_, o->io_executor_);
82
83 ASIO_HANDLER_COMPLETION((*o));
84
85 // Make a copy of the handler so that the memory can be deallocated before
86 // the upcall is made. Even if we're not about to make an upcall, a
87 // sub-object of the handler may be the true owner of the memory associated
88 // with the handler. Consequently, a local copy of the handler is required
89 // to ensure that any owning sub-object remains valid until after we have
90 // deallocated the memory here.
91 detail::binder1<Handler, asio::error_code>
92 handler(o->handler_, o->ec_);
93 p.h = asio::detail::addressof(handler.handler_);
94 p.reset();
95
96 // Make the upcall if required.
97 if (owner)
98 {
99 fenced_block b(fenced_block::half);
100 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
101 w.complete(handler, handler.handler_);
102 ASIO_HANDLER_INVOCATION_END;
103 }
104 }
105
106private:
107 Handler handler_;
108 IoExecutor io_executor_;
109};
