134{

136 lpcomb[i] = 0.0f;

154//Process one channel; 0=left, 1=right
155void Reverb::processmono(int ch, float *output, float *inputbuf)
156{
157 //todo: implement the high part from lohidamp
158
159 for(int j = REV_COMBS * ch; j < REV_COMBS * (ch + 1); ++j) {
160 int &ck = combk[j];
161 const int comblength = comblen[j];
162 float &lpcombj = lpcomb[j];
163
164 for(int i = 0; i < buffersize; ++i) {
165 float fbout = comb[j][ck] * combfb[j];
166 fbout = fbout * (1.0f - lohifb) + lpcombj * lohifb;
167 lpcombj = fbout;
168
169 comb[j][ck] = inputbuf[i] + fbout;
170 output[i] += fbout;
171
172 if((++ck) >= comblength)
173 ck = 0;
174 }
175 }
176
177 for(int j = REV_APS * ch; j < REV_APS * (1 + ch); ++j) {
178 int &ak = apk[j];
179 const int aplength = aplen[j];
180 for(int i = 0; i < buffersize; ++i) {
181 float tmp = ap[j][ak];
182 ap[j][ak] = 0.7f * tmp + output[i];
183 output[i] = tmp - 0.7f * ap[j][ak];
184 if((++ak) >= aplength)
185 ak = 0;
186 }
187 }
188}

190//Effect output
191void Reverb::out(const Stereo<float *> &smp)
192{
193 if(!Pvolume && insertion)
194 return;
195
196 float inputbuf[buffersize];
197 for(int i = 0; i < buffersize; ++i)
198 inputbuf[i] = (smp.l[i] + smp.r[i]) / 2.0f;
199
200 if(idelay)
201 for(int i = 0; i < buffersize; ++i) {
202 //Initial delay r
203 float tmp = inputbuf[i] + idelay[idelayk] * idelayfb;
204 inputbuf[i] = idelay[idelayk];
205 idelay[idelayk] = tmp;
206 idelayk++;
207 if(idelayk >= idelaylen)
208 idelayk = 0;
209 }
210
211 if(bandwidth)
212 bandwidth->process(buffersize, inputbuf);
213
214 if(lpf)
215 lpf->filterout(inputbuf);
216 if(hpf)
217 hpf->filterout(inputbuf);
218
219 processmono(0, efxoutl, inputbuf); //left
220 processmono(1, efxoutr, inputbuf); //right
221
222 float lvol = rs / REV_COMBS * pangainL;
223 float rvol = rs / REV_COMBS * pangainR;
224 if(insertion != 0) {
225 lvol *= 2.0f;
226 rvol *= 2.0f;
227 }
228 for(int i = 0; i < buffersize; ++i) {
229 efxoutl[i] *= lvol;
230 efxoutr[i] *= rvol;
231 }
232}

236void Reverb::setvolume(unsigned char _Pvolume)
237{
238 Pvolume = _Pvolume;
239 if(!insertion) {
240 if (Pvolume == 0) {
241 outvolume = 0.0f;
242 } else {
243 outvolume = powf(0.01f, (1.0f - Pvolume / 127.0f)) * 4.0f;
244 }
245 volume = 1.0f;
246 }
247 else {
248 volume = outvolume = Pvolume / 127.0f;
249 if(Pvolume == 0)
250 cleanup();
251 }
252}

254void Reverb::settime(unsigned char _Ptime)
255{
256 Ptime = _Ptime;
257 float t = powf(60.0f, Ptime / 127.0f) - 0.97f;
258
259 for(int i = 0; i < REV_COMBS * 2; ++i)
260 combfb[i] =
261 -expf((float)comblen[i] / samplerate_f * logf(0.001f) / t);
262 //the feedback is negative because it removes the DC

265void Reverb::setlohidamp(unsigned char _Plohidamp)
266{
267 Plohidamp = (_Plohidamp < 64) ? 64 : _Plohidamp;
268 //remove this when the high part from lohidamp is added
269 if(Plohidamp == 64) {
270 lohidamptype = 0;
271 lohifb = 0.0f;
272 }
273 else {
274 if(Plohidamp < 64)
275 lohidamptype = 1;
276 if(Plohidamp > 64)
277 lohidamptype = 2;
278 float x = fabsf((float)(Plohidamp - 64) / 64.1f);
279 lohifb = x * x;
280 }

283void Reverb::setidelay(unsigned char _Pidelay)
284{
285 Pidelay = _Pidelay;
286 float delay = powf(50.0f * Pidelay / 127.0f, 2.0f) - 1.0f;
287 int newDelayLen = (int) (samplerate_f * delay / 1000);
288 if(newDelayLen == idelaylen)
289 return;
290
291 memory.devalloc(idelay);
292
293 idelaylen = newDelayLen;
294 if(idelaylen > 1) {
295 idelayk = 0;
296 idelay = memory.valloc<float>(idelaylen);
297 memset(idelay, 0, idelaylen * sizeof(float));
298 }
299}

307void Reverb::sethpf(unsigned char _Phpf)
308{
309 Phpf = _Phpf;
310 if(Phpf == 0) { //No HighPass
311 memory.dealloc(hpf);
312 } else {
313 float fr = expf(sqrtf(Phpf / 127.0f) * logf(10000.0f)) + 20.0f;
314 if(hpf == NULL)
315 hpf = memory.alloc<AnalogFilter>(3, fr, 1, 0, samplerate, buffersize);
316 else
317 hpf->setfreq(fr);
318 }
319}

321void Reverb::setlpf(unsigned char _Plpf)
322{
323 Plpf = _Plpf;
324 if(Plpf == 127) { //No LowPass
325 memory.dealloc(lpf);
326 } else {
327 float fr = expf(sqrtf(Plpf / 127.0f) * logf(25000.0f)) + 40.0f;
328 if(!lpf)
329 lpf = memory.alloc<AnalogFilter>(2, fr, 1, 0, samplerate, buffersize);
330 else
331 lpf->setfreq(fr);
332 }

335void Reverb::settype(unsigned char _Ptype)
336{
337 Ptype = _Ptype;
338 const int NUM_TYPES = 3;
339 const int combtunings[NUM_TYPES][REV_COMBS] = {
340 //this is unused (for random)
341 {0, 0, 0, 0, 0, 0, 0, 0 },
342 //Freeverb by Jezar at Dreampoint
343 {1116, 1188, 1277, 1356, 1422, 1491, 1557, 1617 },
344 //duplicate of Freeverb by Jezar at Dreampoint
345 {1116, 1188, 1277, 1356, 1422, 1491, 1557, 1617 }
346 };
347
348 const int aptunings[NUM_TYPES][REV_APS] = {
349 //this is unused (for random)
350 {0, 0, 0, 0 },
351 //Freeverb by Jezar at Dreampoint
352 {225, 341, 441, 556 },
353 //duplicate of Freeverb by Jezar at Dreampoint
354 {225, 341, 441, 556 }
355 };
356
357 if(Ptype >= NUM_TYPES)
358 Ptype = NUM_TYPES - 1;
359
360 // adjust the combs according to the samplerate
361 float samplerate_adjust = samplerate_f / 44100.0f;
362 float tmp;
363 for(int i = 0; i < REV_COMBS * 2; ++i) {
364 if(Ptype == 0)
365 tmp = 800.0f + (int)(RND * 1400.0f);
366 else
367 tmp = combtunings[Ptype][i % REV_COMBS];
368 tmp *= roomsize;
369 if(i > REV_COMBS)
370 tmp += 23.0f;
371 tmp *= samplerate_adjust; //adjust the combs according to the samplerate
372 if(tmp < 10.0f)
373 tmp = 10.0f;
374 combk[i] = 0;
375 lpcomb[i] = 0;
376 if(comblen[i] != (int)tmp || comb[i] == NULL) {
377 comblen[i] = (int) tmp;
378 memory.devalloc(comb[i]);
379 comb[i] = memory.valloc<float>(comblen[i]);
380 }
381 }
382
383 for(int i = 0; i < REV_APS * 2; ++i) {
384 if(Ptype == 0)
385 tmp = 500 + (int)(RND * 500.0f);
386 else
387 tmp = aptunings[Ptype][i % REV_APS];
388 tmp *= roomsize;
389 if(i > REV_APS)
390 tmp += 23.0f;
391 tmp *= samplerate_adjust; //adjust the combs according to the samplerate
392 if(tmp < 10)
393 tmp = 10;
394 apk[i] = 0;
395 if(aplen[i] != (int)tmp || ap[i] == NULL) {
396 aplen[i] = (int) tmp;
397 memory.devalloc(ap[i]);
398 ap[i] = memory.valloc<float>(aplen[i]);
399 }
400 }
401 memory.dealloc(bandwidth);
402 if(Ptype == 2) { //bandwidth
403 //TODO the size of the unison buffer may be too small, though this has
404 //not been verified yet.
405 //As this cannot be resized in a RT context, a good upper bound should
406 //be found
407 bandwidth = memory.alloc<Unison>(&memory, buffersize / 4 + 1, 2.0f, samplerate_f);
408 bandwidth->setSize(50);
409 bandwidth->setBaseFrequency(1.0f);
410 }
411 settime(Ptime);
412 cleanup();
413}

415void Reverb::setroomsize(unsigned char _Proomsize)
416{
417 Proomsize = _Proomsize;
418 if(!Proomsize)
419 this->Proomsize = 64; //this is because the older versions consider roomsize=0
420 roomsize = (this->Proomsize - 64.0f) / 64.0f;
421 if(roomsize > 0.0f)
422 roomsize *= 2.0f;
423 roomsize = powf(10.0f, roomsize);
424 rs = sqrtf(roomsize);
425 settype(Ptype);
426}
