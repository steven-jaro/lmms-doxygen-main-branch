76template <typename NumericType>
77std::array<NumericType, 6> ArrayCoefficients<NumericType>::makeLowPass (double sampleRate,
78 NumericType frequency,
79 NumericType Q)
80{
81 jassert (sampleRate > 0.0);
82 jassert (frequency > 0 && frequency <= static_cast<float> (sampleRate * 0.5));
83 jassert (Q > 0.0);
84
85 auto n = 1 / std::tan (MathConstants<NumericType>::pi * frequency / static_cast<NumericType> (sampleRate));
86 auto nSquared = n * n;
87 auto invQ = 1 / Q;
88 auto c1 = 1 / (1 + invQ * n + nSquared);
89
90 return { { c1, c1 * 2, c1,
91 1, c1 * 2 * (1 - nSquared),
92 c1 * (1 - invQ * n + nSquared) } };
93}

200template <typename NumericType>
201std::array<NumericType, 6> ArrayCoefficients<NumericType>::makeLowShelf (double sampleRate,
202 NumericType cutOffFrequency,
203 NumericType Q,
204 NumericType gainFactor)
205{
206 jassert (sampleRate > 0.0);
207 jassert (cutOffFrequency > 0.0 && cutOffFrequency <= sampleRate * 0.5);
208 jassert (Q > 0.0);
209
210 auto A = jmax (static_cast<NumericType> (0.0), std::sqrt (gainFactor));
211 auto aminus1 = A - 1;
212 auto aplus1 = A + 1;
213 auto omega = (2 * MathConstants<NumericType>::pi * jmax (cutOffFrequency, static_cast<NumericType> (2.0))) / static_cast<NumericType> (sampleRate);
214 auto coso = std::cos (omega);
215 auto beta = std::sin (omega) * std::sqrt (A) / Q;
216 auto aminus1TimesCoso = aminus1 * coso;
217
218 return { { A * (aplus1 - aminus1TimesCoso + beta),
219 A * 2 * (aminus1 - aplus1 * coso),
220 A * (aplus1 - aminus1TimesCoso - beta),
221 aplus1 + aminus1TimesCoso + beta,
222 -2 * (aminus1 + aplus1 * coso),
223 aplus1 + aminus1TimesCoso - beta } };
224}

227std::array<NumericType, 6> ArrayCoefficients<NumericType>::makeHighShelf (double sampleRate,
228 NumericType cutOffFrequency,
229 NumericType Q,
230 NumericType gainFactor)
231{
232 jassert (sampleRate > 0);
233 jassert (cutOffFrequency > 0 && cutOffFrequency <= static_cast<NumericType> (sampleRate * 0.5));
234 jassert (Q > 0);
235
236 auto A = jmax (static_cast<NumericType> (0.0), std::sqrt (gainFactor));
237 auto aminus1 = A - 1;
238 auto aplus1 = A + 1;
239 auto omega = (2 * MathConstants<NumericType>::pi * jmax (cutOffFrequency, static_cast<NumericType> (2.0))) / static_cast<NumericType> (sampleRate);
240 auto coso = std::cos (omega);
241 auto beta = std::sin (omega) * std::sqrt (A) / Q;
242 auto aminus1TimesCoso = aminus1 * coso;
243
244 return { { A * (aplus1 + aminus1TimesCoso + beta),
245 A * -2 * (aminus1 + aplus1 * coso),
246 A * (aplus1 + aminus1TimesCoso - beta),
247 aplus1 - aminus1TimesCoso + beta,

252template <typename NumericType>
253std::array<NumericType, 6> ArrayCoefficients<NumericType>::makePeakFilter (double sampleRate,
254 NumericType frequency,
255 NumericType Q,
256 NumericType gainFactor)
257{
258 jassert (sampleRate > 0);
259 jassert (frequency > 0 && frequency <= static_cast<NumericType> (sampleRate * 0.5));
260 jassert (Q > 0);
261 jassert (gainFactor > 0);
262
263 auto A = jmax (static_cast<NumericType> (0.0), std::sqrt (gainFactor));
264 auto omega = (2 * MathConstants<NumericType>::pi * jmax (frequency, static_cast<NumericType> (2.0))) / static_cast<NumericType> (sampleRate);
265 auto alpha = std::sin (omega) / (Q * 2);
266 auto c2 = -2 * std::cos (omega);
267 auto alphaTimesA = alpha * A;
268 auto alphaOverA = alpha / A;
269
270 return { { 1 + alphaTimesA, c2, 1 - alphaTimesA, 1 + alphaOverA, c2, 1 - alphaOverA } };
271}

437template <typename NumericType>
438double Coefficients<NumericType>::getMagnitudeForFrequency (double frequency, double sampleRate) const noexcept
439{
440 constexpr Complex<double> j (0, 1);
441 const auto order = getFilterOrder();
442 const auto* coefs = coefficients.begin();
443
444 jassert (frequency >= 0 && frequency <= sampleRate * 0.5);
445
446 Complex<double> numerator = 0.0, denominator = 0.0, factor = 1.0;
447 Complex<double> jw = std::exp (-MathConstants<double>::twoPi * frequency * j / sampleRate);
448
449 for (size_t n = 0; n <= order; ++n)
450 {
451 numerator += static_cast<double> (coefs[n]) * factor;
452 factor *= jw;
453 }
454
455 denominator = 1.0;
456 factor = jw;
457
458 for (size_t n = order + 1; n <= 2 * order; ++n)
459 {
460 denominator += static_cast<double> (coefs[n]) * factor;
461 factor *= jw;
462 }
463
464 return std::abs (numerator / denominator);
465}
