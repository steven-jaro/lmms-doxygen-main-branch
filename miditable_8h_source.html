172 class PendingQueue
173 {
174 public:
175 PendingQueue()
176 :pos_r(0), pos_w(0), size(0)
177 {
178 for(int i=0; i<32; ++i)
179 vals[i] = -1;
180 }
181 void insert(int x)
182 {
183 if(has(x) || size > 31)
184 return;
185 vals[pos_w] = x;
186 size++;
187 pos_w = (pos_w+1)%32;
188 }
189 void pop(void)
190 {
191 if(size == 0)
192 return;
193 size--;
194 vals[pos_r] = -1;
195 pos_r = (1+pos_r)%32;
196 }
197 bool has(int x)
198 {
199 for(int i=0; i<32; ++i)
200 if(vals[i] == x)
201 return true;
202 return false;
203 }
204 int vals[32];
205 int pos_r;
206 int pos_w;
207 int size;
208
209 };
