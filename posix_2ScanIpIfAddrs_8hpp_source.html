38// RAII type to make [get,free]ifaddrs function pairs exception safe
39class GetIfAddrs
40{
41public:
42 GetIfAddrs()
43 {
44 if (getifaddrs(&interfaces)) // returns 0 on success
45 {
46 interfaces = NULL;
47 }
48 }
49 ~GetIfAddrs()
50 {
51 if (interfaces)
52 freeifaddrs(interfaces);
53 }
54
55 // RAII must not copy
56 GetIfAddrs(GetIfAddrs&) = delete;
57 GetIfAddrs& operator=(GetIfAddrs&) = delete;
58
59 template <typename Function>
60 void withIfAddrs(Function f)
61 {
62 if (interfaces)
63 f(*interfaces);
64 }
65
66private:
67 struct ifaddrs* interfaces = NULL;
68};
69
70} // namespace detail

73// Posix implementation of ip interface address scanner
74struct ScanIpIfAddrs
75{
76 // Scan active network interfaces and return corresponding addresses
77 // for all ip-based interfaces.
78 std::vector<::asio::ip::address> operator()()
79 {
80 std::vector<::asio::ip::address> addrs;
81
82 detail::GetIfAddrs getIfAddrs;
83 getIfAddrs.withIfAddrs([&addrs](const struct ifaddrs& interfaces) {
84 const struct ifaddrs* interface;
85 for (interface = &interfaces; interface; interface = interface->ifa_next)
86 {
87 auto addr = reinterpret_cast<const struct sockaddr_in*>(interface->ifa_addr);
88 if (addr && interface->ifa_flags & IFF_UP)
89 {
90 if (addr->sin_family == AF_INET)
91 {
92 auto bytes = reinterpret_cast<const char*>(&addr->sin_addr);
93 addrs.emplace_back(asio::makeAddress<::asio::ip::address_v4>(bytes));
94 }
95 else if (addr->sin_family == AF_INET6)
96 {
97 auto addr6 = reinterpret_cast<const struct sockaddr_in6*>(addr);
98 auto bytes = reinterpret_cast<const char*>(&addr6->sin6_addr);
99 addrs.emplace_back(asio::makeAddress<::asio::ip::address_v6>(bytes));
100 }
101 }
102 }
103 });
104 return addrs;
105 }
106};
