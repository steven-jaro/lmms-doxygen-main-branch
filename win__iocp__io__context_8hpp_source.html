86 // Determine whether the io_context is stopped.
87 bool stopped() const
88 {
89 return ::InterlockedExchangeAdd(&stopped_, 0) != 0;
90 }
91
92 // Restart in preparation for a subsequent run invocation.
93 void restart()
94 {
95 ::InterlockedExchange(&stopped_, 0);
96 }

98 // Notify that some work has started.
99 void work_started()
100 {
101 ::InterlockedIncrement(&outstanding_work_);
102 }
103
104 // Notify that some work has finished.
105 void work_finished()
106 {
107 if (::InterlockedDecrement(&outstanding_work_) == 0)
108 stop();
109 }
110
111 // Return whether a handler can be dispatched immediately.
112 bool can_dispatch()
113 {
114 return thread_call_stack::contains(this) != 0;
115 }

117 // Request invocation of the given operation and return immediately. Assumes
118 // that work_started() has not yet been called for the operation.
119 void post_immediate_completion(win_iocp_operation* op, bool)
120 {

122 post_deferred_completion(op);
123 }

125 // Request invocation of the given operation and return immediately. Assumes
