97/*
98==============================================================================
99Name: process_sample
100Description:
101 Upsamples (x2) the input vector, generating two output vectors.
102Input parameters:
103 - input: The input vector.
104Output parameters:
105 - out_0: First output vector.
106 - out_1: Second output vector.
107Throws: Nothing
108==============================================================================
109*/
110
111template <int NC>
112void Upsampler2x16Avx512 <NC>::process_sample (__m512 &out_0, __m512 &out_1, __m512 input) noexcept
113{
114 __m512 even = input;
115 __m512 odd = input;
116 StageProc16Avx512 <NBR_COEFS>::process_sample_pos (
117 NBR_COEFS,
118 even,
119 odd,
120 &_filter [0]
121 );
122 out_0 = even;
123 out_1 = odd;
124}

128/*
129==============================================================================
130Name: process_block
131Description:
132 Upsamples (x2) the input vector block.
133 Input and output blocks may overlap, see assert() for details.
134Input parameters:
135 - in_ptr: Input array, containing nbr_spl vector.
136 No alignment constraint.
137 - nbr_spl: Number of input vectors to process, > 0
138Output parameters:
139 - out_0_ptr: Output vector array, capacity: nbr_spl * 2 vectors.
140 No alignment constraint.
141Throws: Nothing
142==============================================================================
143*/
144
145template <int NC>
146void Upsampler2x16Avx512 <NC>::process_block (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
147{
148 assert (out_ptr != nullptr);
149 assert (in_ptr != nullptr);
150 assert ( out_ptr >= in_ptr + nbr_spl * _nbr_chn
151 || in_ptr >= out_ptr + nbr_spl * _nbr_chn);
152 assert (nbr_spl > 0);
153
154 long pos = 0;
155 do
156 {
157 __m512 dst_0;
158 __m512 dst_1;
159 const __m512 src = _mm512_loadu_ps (in_ptr + pos * _nbr_chn);
160 process_sample (dst_0, dst_1, src);
161 _mm512_storeu_ps (out_ptr + pos * (_nbr_chn * 2) , dst_0);
162 _mm512_storeu_ps (out_ptr + pos * (_nbr_chn * 2) + _nbr_chn, dst_1);
163 ++ pos;
164 }
165 while (pos < nbr_spl);
166}
