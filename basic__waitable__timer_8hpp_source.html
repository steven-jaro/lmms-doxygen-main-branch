709 struct initiate_async_wait
710 {
711 template <typename WaitHandler>
712 void operator()(ASIO_MOVE_ARG(WaitHandler) handler,
713 basic_waitable_timer* self) const
714 {
715 // If you get an error on the following line it means that your handler
716 // does not meet the documented type requirements for a WaitHandler.
717 ASIO_WAIT_HANDLER_CHECK(WaitHandler, handler) type_check;
718
719 detail::non_const_lvalue<WaitHandler> handler2(handler);
720 self->impl_.get_service().async_wait(
721 self->impl_.get_implementation(), handler2.value,
722 self->impl_.get_implementation_executor());
723 }
724 };
