55 void init (double _h = .001, double seed = .0)
56 {
57 h = _h;
58
59 I = 0;
60
61 x[0] = .0001 + .0001 * seed;
62 y[0] = .0001;
63 z[0] = .0001;
64
65 for (int i = 0; i < 5000; ++i)
66 get();
67 }
68
69 sample_t get()
70 {
71 int J = I ^ 1;
72
73 x[J] = x[I] + h * (- y[I] - z[I]);
74 y[J] = y[I] + h * (x[I] + a * y[I]);
75 z[J] = z[I] + h * (b + z[I] * (x[I] - c));
76
77 I = J;
78
79 return x[I] * .01725 + z[I] * .015;
80 }
81
82 double get_x()
83 {
84 return x[I];
85 }
86
87 double get_y()
88 {
89 return y[I];
90 }
91
92 double get_z()
93 {
94 return z[I];
95 }
96};
