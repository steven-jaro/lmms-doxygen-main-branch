80 template <typename ConnectCondition, typename Iterator>
81 inline Iterator call_connect_condition(ConnectCondition& connect_condition,
82 const asio::error_code& ec, Iterator next, Iterator end,
83 typename enable_if<is_legacy_connect_condition<
84 ConnectCondition, Iterator>::value>::type* = 0)
85 {
86 if (next != end)
87 return connect_condition(ec, next);
88 return end;

91 template <typename ConnectCondition, typename Iterator>
92 inline Iterator call_connect_condition(ConnectCondition& connect_condition,
93 const asio::error_code& ec, Iterator next, Iterator end,
94 typename enable_if<!is_legacy_connect_condition<
95 ConnectCondition, Iterator>::value>::type* = 0)
96 {
97 for (;next != end; ++next)
98 if (connect_condition(ec, *next))
99 return next;
100 return end;
101 }

226template <typename Protocol, typename Executor,
227 typename Iterator, typename ConnectCondition>
228Iterator connect(basic_socket<Protocol, Executor>& s, Iterator begin,
229 Iterator end, ConnectCondition connect_condition,
230 asio::error_code& ec)
231{
232 ec = asio::error_code();
233
234 for (Iterator iter = begin; iter != end; ++iter)
235 {
236 iter = (detail::call_connect_condition(connect_condition, ec, iter, end));
237 if (iter != end)
238 {
239 s.close(ec);
240 s.connect(*iter, ec);
241 if (!ec)
242 return iter;
243 }
244 else
245 break;
246 }
247
248 if (!ec)
249 ec = asio::error::not_found;
250
251 return end;
252}

296 class range_connect_op : base_from_connect_condition<ConnectCondition>
297 {

309 {
310 }
311

314 : base_from_connect_condition<ConnectCondition>(other),
315 socket_(other.socket_),
316 endpoints_(other.endpoints_),
317 index_(other.index_),
318 start_(other.start_),
319 handler_(other.handler_)
320 {
321 }
322

324 : base_from_connect_condition<ConnectCondition>(other),
325 socket_(other.socket_),
326 endpoints_(other.endpoints_),
327 index_(other.index_),
328 start_(other.start_),
329 handler_(ASIO_MOVE_CAST(RangeConnectHandler)(other.handler_))

332#endif // defined(ASIO_HAS_MOVE)
333
334 void operator()(asio::error_code ec, int start = 0)
335 {
336 this->process(ec, start,
337 const_cast<const EndpointSequence&>(endpoints_).begin(),
338 const_cast<const EndpointSequence&>(endpoints_).end());

341 //private:
342 template <typename Iterator>
343 void process(asio::error_code ec,
344 int start, Iterator begin, Iterator end)
345 {
346 Iterator iter = begin;
347 std::advance(iter, index_);
348
349 switch (start_ = start)
350 {
351 case 1:
352 for (;;)
353 {
354 this->check_condition(ec, iter, end);
355 index_ = std::distance(begin, iter);
356
357 if (iter != end)
358 {
359 socket_.close(ec);
360 socket_.async_connect(*iter,
361 ASIO_MOVE_CAST(range_connect_op)(*this));
362 return;
363 }
364
365 if (start)
366 {
367 ec = asio::error::not_found;
368 asio::post(socket_.get_executor(),
369 detail::bind_handler(
370 ASIO_MOVE_CAST(range_connect_op)(*this), ec));
371 return;
372 }
373
374 default:
375
376 if (iter == end)
377 break;
378
379 if (!socket_.is_open())
380 {
381 ec = asio::error::operation_aborted;
382 break;
383 }
384
385 if (!ec)
386 break;
387
388 ++iter;
389 ++index_;
390 }
391
392 handler_(static_cast<const asio::error_code&>(ec),
393 static_cast<const typename Protocol::endpoint&>(
394 ec || iter == end ? typename Protocol::endpoint() : *iter));
395 }
396 }
397
398 basic_socket<Protocol, Executor>& socket_;
399 EndpointSequence endpoints_;
400 std::size_t index_;
401 int start_;
402 RangeConnectHandler handler_;
403 };

480 class iterator_connect_op : base_from_connect_condition<ConnectCondition>
481 {
482 public:
483 iterator_connect_op(basic_socket<Protocol, Executor>& sock,
484 const Iterator& begin, const Iterator& end,

493 {

497 iterator_connect_op(const iterator_connect_op& other)
498 : base_from_connect_condition<ConnectCondition>(other),
499 socket_(other.socket_),
500 iter_(other.iter_),
501 end_(other.end_),

507 iterator_connect_op(iterator_connect_op&& other)
508 : base_from_connect_condition<ConnectCondition>(other),
509 socket_(other.socket_),
510 iter_(other.iter_),
511 end_(other.end_),
512 start_(other.start_),
513 handler_(ASIO_MOVE_CAST(IteratorConnectHandler)(other.handler_))
514 {
515 }
516#endif // defined(ASIO_HAS_MOVE)
517
518 void operator()(asio::error_code ec, int start = 0)
519 {
520 switch (start_ = start)
521 {
522 case 1:
523 for (;;)
524 {
525 this->check_condition(ec, iter_, end_);
526
527 if (iter_ != end_)
528 {
529 socket_.close(ec);
530 socket_.async_connect(*iter_,
531 ASIO_MOVE_CAST(iterator_connect_op)(*this));
532 return;
533 }
534
535 if (start)
536 {
537 ec = asio::error::not_found;
538 asio::post(socket_.get_executor(),
539 detail::bind_handler(
540 ASIO_MOVE_CAST(iterator_connect_op)(*this), ec));
541 return;
542 }
543
544 default:
545
546 if (iter_ == end_)
547 break;
548
549 if (!socket_.is_open())
550 {
551 ec = asio::error::operation_aborted;
552 break;
553 }
554
555 if (!ec)
556 break;
557
558 ++iter_;
559 }
560
561 handler_(static_cast<const asio::error_code&>(ec),
562 static_cast<const Iterator&>(iter_));
563 }
564 }
