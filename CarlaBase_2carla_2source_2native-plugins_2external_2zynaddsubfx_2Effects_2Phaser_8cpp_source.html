121void Phaser::analog_setup()
122{
123 //model mismatch between JFET devices
124 offset[0] = -0.2509303f;
125 offset[1] = 0.9408924f;
126 offset[2] = 0.998f;
127 offset[3] = -0.3486182f;
128 offset[4] = -0.2762545f;
129 offset[5] = -0.5215785f;
130 offset[6] = 0.2509303f;
131 offset[7] = -0.9408924f;
132 offset[8] = -0.998f;
133 offset[9] = 0.3486182f;
134 offset[10] = 0.2762545f;
135 offset[11] = 0.5215785f;
136
137 barber = 0; //Deactivate barber pole phasing by default
138
139 mis = 1.0f;
140 Rmin = 625.0f; // 2N5457 typical on resistance at Vgs = 0
141 Rmax = 22000.0f; // Resistor parallel to FET
142 Rmx = Rmin / Rmax;
143 Rconst = 1.0f + Rmx; // Handle parallel resistor relationship
144 C = 0.00000005f; // 50 nF
145 CFs = 2.0f * samplerate_f * C;
146 invperiod = 1.0f / buffersize_f;
147}

170void Phaser::AnalogPhase(const Stereo<float *> &input)
171{
172 Stereo<float> gain(0.0f), lfoVal(0.0f), mod(0.0f), g(0.0f), b(0.0f), hpf(
173 0.0f);
174
175 lfo.effectlfoout(&lfoVal.l, &lfoVal.r);
176 mod.l = lfoVal.l * width + (depth - 0.5f);
177 mod.r = lfoVal.r * width + (depth - 0.5f);
178
179 mod.l = limit(mod.l, ZERO_, ONE_);
180 mod.r = limit(mod.r, ZERO_, ONE_);
181
182 if(Phyper) {
183 //Triangle wave squared is approximately sin on bottom, tri on top
184 //Result is exponential sweep more akin to filter in synth with
185 //exponential generator circuitry.
186 mod.l *= mod.l;
187 mod.r *= mod.r;
188 }
189
190 //g.l,g.r is Vp - Vgs. Typical FET drain-source resistance follows constant/[1-sqrt(Vp - Vgs)]
191 mod.l = sqrtf(1.0f - mod.l);
192 mod.r = sqrtf(1.0f - mod.r);
193
194 diff.r = (mod.r - oldgain.r) * invperiod;
195 diff.l = (mod.l - oldgain.l) * invperiod;
196
197 g = oldgain;
198 oldgain = mod;
199
200 for(int i = 0; i < buffersize; ++i) {
201 g.l += diff.l; // Linear interpolation between LFO samples
202 g.r += diff.r;
203
204 Stereo<float> xn(input.l[i] * pangainL, input.r[i] * pangainR);
205
206 if(barber) {
207 g.l += 0.25;
208 g.l -= floorf(g.l);
209 g.r += 0.25;
210 g.r -= floorf(g.r);
211 }
212
213 xn.l = applyPhase(xn.l, g.l, fb.l, hpf.l, yn1.l, xn1.l);
214 xn.r = applyPhase(xn.r, g.r, fb.r, hpf.r, yn1.r, xn1.r);
215
216
217 fb.l = xn.l * feedback;
218 fb.r = xn.r * feedback;
219 efxoutl[i] = xn.l;
220 efxoutr[i] = xn.r;
221 }
222
223 if(Poutsub) {
224 invSignal(efxoutl, buffersize);
225 invSignal(efxoutr, buffersize);
226 }
227}

229float Phaser::applyPhase(float x, float g, float fb,
230 float &hpf, float *yn1, float *xn1)
231{
232 for(int j = 0; j < Pstages; ++j) { //Phasing routine
233 mis = 1.0f + offsetpct * offset[j];
234
235 //This is symmetrical.
236 //FET is not, so this deviates slightly, however sym dist. is
237 //better sounding than a real FET.
238 float d = (1.0f + 2.0f * (0.25f + g) * hpf * hpf * distortion) * mis;
239 Rconst = 1.0f + mis * Rmx;
240
241 // This is 1/R. R is being modulated to control filter fc.
242 float b = (Rconst - g) / (d * Rmin);
243 float gain = (CFs - b) / (CFs + b);
244 yn1[j] = gain * (x + yn1[j]) - xn1[j];
245
246 //high pass filter:
247 //Distortion depends on the high-pass part of the AP stage.
248 hpf = yn1[j] + (1.0f - gain) * xn1[j];
249
250 xn1[j] = x;
251 x = yn1[j];
252 if(j == 1)
253 x += fb; //Insert feedback after first phase stage
254 }
255 return x;
256}
257void Phaser::normalPhase(const Stereo<float *> &input)
258{
259 Stereo<float> gain(0.0f), lfoVal(0.0f);
260
261 lfo.effectlfoout(&lfoVal.l, &lfoVal.r);
262 gain.l =
263 (expf(lfoVal.l
264 * PHASER_LFO_SHAPE) - 1) / (expf(PHASER_LFO_SHAPE) - 1.0f);
265 gain.r =
266 (expf(lfoVal.r
267 * PHASER_LFO_SHAPE) - 1) / (expf(PHASER_LFO_SHAPE) - 1.0f);
268
269 gain.l = 1.0f - phase * (1.0f - depth) - (1.0f - phase) * gain.l * depth;
270 gain.r = 1.0f - phase * (1.0f - depth) - (1.0f - phase) * gain.r * depth;
271
272 gain.l = limit(gain.l, ZERO_, ONE_);
273 gain.r = limit(gain.r, ZERO_, ONE_);
274
275 for(int i = 0; i < buffersize; ++i) {
276 float x = (float) i / buffersize_f;
277 float x1 = 1.0f - x;
278 //TODO think about making panning an external feature
279 Stereo<float> xn(input.l[i] * pangainL + fb.l,
280 input.r[i] * pangainR + fb.r);
281
282 Stereo<float> g(gain.l * x + oldgain.l * x1,
283 gain.r * x + oldgain.r * x1);
284
285 xn.l = applyPhase(xn.l, g.l, old.l);
286 xn.r = applyPhase(xn.r, g.r, old.r);
287
288 //Left/Right crossing
289 crossover(xn.l, xn.r, lrcross);
290
291 fb.l = xn.l * feedback;
292 fb.r = xn.r * feedback;
293 efxoutl[i] = xn.l;
294 efxoutr[i] = xn.r;
295 }
296
297 oldgain = gain;
298
299 if(Poutsub) {
300 invSignal(efxoutl, buffersize);
301 invSignal(efxoutr, buffersize);
302 }
303}

305float Phaser::applyPhase(float x, float g, float *old)
306{
307 for(int j = 0; j < Pstages * 2; ++j) { //Phasing routine
308 float tmp = old[j];
309 old[j] = g * tmp + x;
310 x = tmp - g * old[j];
311 }
312 return x;
313}

321 for(int i = 0; i < Pstages * 2; ++i) {
322 old.l[i] = 0.0f;
