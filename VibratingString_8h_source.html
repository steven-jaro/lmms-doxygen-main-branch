50 sample_t nextSample()
51 {
52 for (int i = 0; i < m_oversample; ++i)
53 {
54 // Output at pickup position
55 m_outsamp[i] = fromBridgeAccess(m_fromBridge.get(), m_pickupLoc);
56 m_outsamp[i] += toBridgeAccess(m_toBridge.get(), m_pickupLoc);
57
58 // Sample traveling into "bridge"
59 sample_t ym0 = toBridgeAccess(m_toBridge.get(), 1);
60 // Sample to "nut"
61 sample_t ypM = fromBridgeAccess(m_fromBridge.get(), m_fromBridge->length - 2);
62
63 // String state update
64
65 // Decrement pointer and then update
66 fromBridgeUpdate(m_fromBridge.get(), -bridgeReflection(ym0));
67 // Update and then increment pointer
68 toBridgeUpdate(m_toBridge.get(), -ypM);
69 }
70
71 return m_outsamp[m_choice];
72 }

104 void setDelayLine(DelayLine* dl, int pick, const float* values, int len, float scale, bool state)
105 {
106 if (!state)
107 {
108 for (int i = 0; i < pick; ++i)
109 {
110 float r = static_cast<float>(std::rand()) / static_cast<float>(RAND_MAX);
111 float offset = (m_randomize / 2.0f - m_randomize) * r;
112 dl->data[i] = scale * values[dl->length - i - 1] + offset;
113 }
114 for (int i = pick; i < dl->length; ++i)
115 {
116 float r = static_cast<float>(std::rand()) / static_cast<float>(RAND_MAX);
117 float offset = (m_randomize / 2.0f - m_randomize) * r;
118 dl->data[i] = scale * values[i - pick] + offset;
119 }
120 }
121 else
122 {
123 if (len + pick > dl->length)
124 {
125 for (int i = pick; i < dl->length; ++i)
126 {
127 float r = static_cast<float>(std::rand()) / static_cast<float>(RAND_MAX);
128 float offset = (m_randomize / 2.0f - m_randomize) * r;
129 dl->data[i] = scale * values[i - pick] + offset;
130 }
131 }
132 else
133 {
134 for (int i = 0; i < len; ++i)
135 {
136 float r = static_cast<float>(std::rand()) / static_cast<float>(RAND_MAX);
137 float offset = (m_randomize / 2.0f - m_randomize) * r;
138 dl->data[i+pick] = scale * values[i] + offset;
139 }
140 }
141 }

153 void toBridgeUpdate(DelayLine* dl, sample_t insamp)
154 {
155 sample_t* ptr = dl->pointer;
156 *ptr = insamp * m_stringLoss;
157 ++ptr;
158 if (ptr > dl->end)
159 {
160 ptr = dl->data.get();
161 }
162 dl->pointer = ptr;
163 }
164
173 void fromBridgeUpdate(DelayLine* dl, sample_t insamp)
174 {
175 sample_t* ptr = dl->pointer;
176 --ptr;
177 if (ptr < dl->data.get())
178 {
179 ptr = dl->end;
180 }
181 *ptr = insamp * m_stringLoss;
182 dl->pointer = ptr;
183 }
184
190 static sample_t dlAccess(DelayLine* dl, int position)
191 {
192 sample_t* outpos = dl->pointer + position;
193 while (outpos < dl->data.get())
194 {
195 outpos += dl->length;
196 }
197 while (outpos > dl->end)
198 {
199 outpos -= dl->length;
200 }
201 return *outpos;
202 }
203
204 /*
205 * Right-going delay line:
206 * -->---->---->---
207 * x=0
208 * (pointer)
209 * Left-going delay line:
210 * --<----<----<---
211 * x=0
212 * (pointer)
213 */
214
222 static sample_t fromBridgeAccess(DelayLine* dl, int position)
223 {
224 return dlAccess(dl, position);
225 }

234 static sample_t toBridgeAccess(DelayLine* dl, int position)
