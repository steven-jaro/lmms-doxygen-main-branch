32template <typename Messenger, typename PeerObserver, typename IoContext>
33class PeerGateway
34{
35public:
36 // The peer types are defined by the observer but must match with those
37 // used by the Messenger
38 using ObserverT = typename util::Injected<PeerObserver>::type;
39 using NodeState = typename ObserverT::GatewayObserverNodeState;
40 using NodeId = typename ObserverT::GatewayObserverNodeId;
41 using Timer = typename util::Injected<IoContext>::type::Timer;
42 using TimerError = typename Timer::ErrorCode;

45 util::Injected<PeerObserver> observer,

50 }

66 using PeerTimeout = std::pair<std::chrono::system_clock::time_point, NodeId>;
67 using PeerTimeouts = std::vector<PeerTimeout>;

69 struct Impl : std::enable_shared_from_this<Impl>
70 {
71 Impl(util::Injected<Messenger> messenger,
72 util::Injected<PeerObserver> observer,
73 util::Injected<IoContext> io)
74 : mMessenger(std::move(messenger))
75 , mObserver(std::move(observer))
76 , mIo(std::move(io))
77 , mPruneTimer(mIo->makeTimer())
78 {
79 }
80
81 void updateState(NodeState state)
82 {
83 mMessenger->updateState(std::move(state));
84 try
85 {
86 mMessenger->broadcastState();
87 }
88 catch (const std::runtime_error& err)
89 {
90 info(mIo->log()) << "State broadcast failed on gateway: " << err.what();
91 }
92 }
93
94 void listen()
95 {
96 mMessenger->receive(util::makeAsyncSafe(this->shared_from_this()));
97 }
98
99 // Operators for handling incoming messages
100 void operator()(const PeerState<NodeState>& msg)
101 {
102 onPeerState(msg.peerState, msg.ttl);
103 listen();
104 }
105
106 void operator()(const ByeBye<NodeId>& msg)
107 {
108 onByeBye(msg.peerId);
109 listen();
110 }
111
112 void onPeerState(const NodeState& nodeState, const int ttl)
113 {
114 using namespace std;
115 const auto peerId = nodeState.ident();
116 const auto existing = findPeer(peerId);
117 if (existing != end(mPeerTimeouts))
118 {
119 // If the peer is already present in our timeout list, remove it
120 // as it will be re-inserted below.
121 mPeerTimeouts.erase(existing);
122 }
123
124 auto newTo = make_pair(mPruneTimer.now() + std::chrono::seconds(ttl), peerId);
125 mPeerTimeouts.insert(
126 upper_bound(begin(mPeerTimeouts), end(mPeerTimeouts), newTo, TimeoutCompare{}),
127 move(newTo));
128
129 sawPeer(*mObserver, nodeState);
130 scheduleNextPruning();
131 }
132
133 void onByeBye(const NodeId& peerId)
134 {
135 const auto it = findPeer(peerId);
136 if (it != mPeerTimeouts.end())
137 {
138 peerLeft(*mObserver, it->second);
139 mPeerTimeouts.erase(it);
140 }
141 }
142
143 void pruneExpiredPeers()
144 {
145 using namespace std;
146
147 const auto test = make_pair(mPruneTimer.now(), NodeId{});
148 debug(mIo->log()) << "pruning peers @ " << test.first.time_since_epoch().count();
149
150 const auto endExpired =
151 lower_bound(begin(mPeerTimeouts), end(mPeerTimeouts), test, TimeoutCompare{});
152
153 for_each(begin(mPeerTimeouts), endExpired, [this](const PeerTimeout& pto) {
154 info(mIo->log()) << "pruning peer " << pto.second;
155 peerTimedOut(*mObserver, pto.second);
156 });
157 mPeerTimeouts.erase(begin(mPeerTimeouts), endExpired);
158 scheduleNextPruning();
159 }
160
161 void scheduleNextPruning()
162 {
163 // Find the next peer to expire and set the timer based on it
164 if (!mPeerTimeouts.empty())
165 {
166 // Add a second of padding to the timer to avoid over-eager timeouts
167 const auto t = mPeerTimeouts.front().first + std::chrono::seconds(1);
168
169 debug(mIo->log()) << "scheduling next pruning for "
170 << t.time_since_epoch().count() << " because of peer "
171 << mPeerTimeouts.front().second;
172
173 mPruneTimer.expires_at(t);
174 mPruneTimer.async_wait([this](const TimerError e) {
175 if (!e)
176 {
177 pruneExpiredPeers();
178 }
179 });
180 }
181 }
182
183 struct TimeoutCompare
184 {
185 bool operator()(const PeerTimeout& lhs, const PeerTimeout& rhs) const
186 {
187 return lhs.first < rhs.first;
188 }
189 };
190
191 typename PeerTimeouts::iterator findPeer(const NodeId& peerId)
192 {
193 return std::find_if(begin(mPeerTimeouts), end(mPeerTimeouts),
194 [&peerId](const PeerTimeout& pto) { return pto.second == peerId; });
195 }
196
197 util::Injected<Messenger> mMessenger;
198 util::Injected<PeerObserver> mObserver;
199 util::Injected<IoContext> mIo;
200 Timer mPruneTimer;
201 PeerTimeouts mPeerTimeouts; // Invariant: sorted by time_point
202 };
203
204 std::shared_ptr<Impl> mpImpl;
205};

229// Factory function to bind a PeerGateway to an IpV4Interface with the given address.
230template <typename PeerObserver, typename NodeState, typename IoContext>
231IpV4Gateway<PeerObserver, NodeState, IoContext> makeIpV4Gateway(
232 util::Injected<IoContext> io,
233 const asio::ip::address_v4& addr,
234 util::Injected<PeerObserver> observer,
235 NodeState state)
236{
237 using namespace std;
238 using namespace util;
239
240 const uint8_t ttl = 5;
241 const uint8_t ttlRatio = 20;
242
243 auto iface = makeIpV4Interface<v1::kMaxMessageSize>(injectRef(*io), addr);
244
245 auto messenger =
246 makeUdpMessenger(injectVal(move(iface)), move(state), injectRef(*io), ttl, ttlRatio);
247 return {injectVal(move(messenger)), move(observer), move(io)};
248}
