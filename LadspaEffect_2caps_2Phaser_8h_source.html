35/* all-pass as used by the phaser. */
36class PhaserAP
37{
38 public:
39 sample_t a, m;
40
41 PhaserAP()
42 {
43 a = m = 0.;
44 }
45
46 void set (double delay)
47 {
48 a = (1 - delay) / (1 + delay);
49 }
50
51 sample_t process (sample_t x)
52 {
53 sample_t y = -a * x + m;
54 m = a * y + x;
55
56 return y;
57 }
58};
