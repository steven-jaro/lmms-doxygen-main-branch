47 // Create new threads in the group.
48 template <typename Function>

55 // Wait for all threads in the group to exit.
56 void join()
57 {
58 while (first_)
59 {
60 first_->thread_.join();
61 item* tmp = first_;
62 first_ = first_->next_;
63 delete tmp;
64 }
65 }
66
67 // Test whether the group is empty.

69 {
70 return first_ == 0;
71 }
