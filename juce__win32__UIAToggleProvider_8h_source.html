30class UIAToggleProvider : public UIAProviderBase,
31 public ComBaseClassHelper<ComTypes::IToggleProvider>
32{
33public:
34 using UIAProviderBase::UIAProviderBase;
35
36 //==============================================================================
37 JUCE_COMRESULT Toggle() override
38 {
39 if (! isElementValid())
40 return (HRESULT) UIA_E_ELEMENTNOTAVAILABLE;
41
42 const auto& handler = getHandler();
43
44 if (handler.getActions().invoke (AccessibilityActionType::toggle)
45 || handler.getActions().invoke (AccessibilityActionType::press))
46 {
47 VARIANT newValue;
48 VariantHelpers::setInt (getCurrentToggleState(), &newValue);
49
50 sendAccessibilityPropertyChangedEvent (handler, UIA_ToggleToggleStatePropertyId, newValue);
51
52 return S_OK;
53 }
54
55 return (HRESULT) UIA_E_NOTSUPPORTED;
56 }
57
58 JUCE_COMRESULT get_ToggleState (ComTypes::ToggleState* pRetVal) override
59 {
60 return withCheckedComArgs (pRetVal, *this, [&]
61 {
62 *pRetVal = getCurrentToggleState();
63 return S_OK;
64 });
65 }
66
67private:
68 ComTypes::ToggleState getCurrentToggleState() const
69 {
70 return getHandler().getCurrentState().isChecked() ? ComTypes::ToggleState_On
71 : ComTypes::ToggleState_Off;
72 }
73
74 //==============================================================================
