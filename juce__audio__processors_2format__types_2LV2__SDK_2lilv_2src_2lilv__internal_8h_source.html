365static inline int
366lilv_version_cmp(const LilvVersion* a, const LilvVersion* b)
367{
368 if (a->minor == b->minor && a->micro == b->micro) {
369 return 0;
370 }
371
372 if ((a->minor < b->minor) || (a->minor == b->minor && a->micro < b->micro)) {
373 return -1;
374 }
375
376 return 1;
377}
