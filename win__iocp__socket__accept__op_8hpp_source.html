77 static void do_complete(void* owner, operation* base,
78 const asio::error_code& result_ec,
79 std::size_t /*bytes_transferred*/)
80 {
81 asio::error_code ec(result_ec);
82
83 // Take ownership of the operation object.
84 win_iocp_socket_accept_op* o(static_cast<win_iocp_socket_accept_op*>(base));
85 ptr p = { asio::detail::addressof(o->handler_), o, o };
86 handler_work<Handler, IoExecutor> w(o->handler_, o->io_executor_);
87
88 if (owner)
89 {
90 typename Protocol::endpoint peer_endpoint;
91 std::size_t addr_len = peer_endpoint.capacity();
92 socket_ops::complete_iocp_accept(o->socket_,
93 o->output_buffer(), o->address_length(),
94 peer_endpoint.data(), &addr_len,
95 o->new_socket_.get(), ec);
96
97 // Restart the accept operation if we got the connection_aborted error
98 // and the enable_connection_aborted socket option is not set.
99 if (ec == asio::error::connection_aborted
100 && !o->enable_connection_aborted_)
101 {
102 o->reset();
103 o->socket_service_.restart_accept_op(o->socket_,
104 o->new_socket_, o->protocol_.family(),
105 o->protocol_.type(), o->protocol_.protocol(),
106 o->output_buffer(), o->address_length(), o);
107 p.v = p.p = 0;
108 return;
109 }
110
111 // If the socket was successfully accepted, transfer ownership of the
112 // socket to the peer object.
113 if (!ec)
114 {
115 o->peer_.assign(o->protocol_,
116 typename Socket::native_handle_type(
117 o->new_socket_.get(), peer_endpoint), ec);
118 if (!ec)
119 o->new_socket_.release();
120 }
121
122 // Pass endpoint back to caller.
123 if (o->peer_endpoint_)
124 *o->peer_endpoint_ = peer_endpoint;
125 }
126
127 ASIO_HANDLER_COMPLETION((*o));
128
129 // Make a copy of the handler so that the memory can be deallocated before
130 // the upcall is made. Even if we're not about to make an upcall, a
131 // sub-object of the handler may be the true owner of the memory associated
132 // with the handler. Consequently, a local copy of the handler is required
133 // to ensure that any owning sub-object remains valid until after we have
134 // deallocated the memory here.
135 detail::binder1<Handler, asio::error_code>
136 handler(o->handler_, ec);
137 p.h = asio::detail::addressof(handler.handler_);
138 p.reset();
139
140 // Make the upcall if required.
141 if (owner)
142 {
143 fenced_block b(fenced_block::half);
144 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
145 w.complete(handler, handler.handler_);
146 ASIO_HANDLER_INVOCATION_END;
147 }
148 }
149
150private:
151 win_iocp_socket_service_base& socket_service_;
152 socket_type socket_;
153 socket_holder new_socket_;
154 Socket& peer_;
155 Protocol protocol_;
156 typename Protocol::endpoint* peer_endpoint_;
157 unsigned char output_buffer_[(sizeof(sockaddr_storage_type) + 16) * 2];
158 bool enable_connection_aborted_;
159 Handler handler_;
160 IoExecutor io_executor_;
161};
