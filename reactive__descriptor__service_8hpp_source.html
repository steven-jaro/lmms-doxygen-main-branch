166 // Wait for the descriptor to become ready to read, ready to write, or to have
167 // pending error conditions.
168 asio::error_code wait(implementation_type& impl,
169 posix::descriptor_base::wait_type w, asio::error_code& ec)
170 {
171 switch (w)
172 {
173 case posix::descriptor_base::wait_read:
174 descriptor_ops::poll_read(impl.descriptor_, impl.state_, ec);
175 break;
176 case posix::descriptor_base::wait_write:
177 descriptor_ops::poll_write(impl.descriptor_, impl.state_, ec);
178 break;
179 case posix::descriptor_base::wait_error:
180 descriptor_ops::poll_error(impl.descriptor_, impl.state_, ec);
181 break;
182 default:
183 ec = asio::error::invalid_argument;
184 break;
185 }
186
187 return ec;
188 }

190 // Asynchronously wait for the descriptor to become ready to read, ready to
191 // write, or to have pending error conditions.
192 template <typename Handler, typename IoExecutor>
193 void async_wait(implementation_type& impl,
194 posix::descriptor_base::wait_type w,
195 Handler& handler, const IoExecutor& io_ex)
196 {
197 bool is_continuation =
198 asio_handler_cont_helpers::is_continuation(handler);
199
200 // Allocate and construct an operation to wrap the handler.
201 typedef reactive_wait_op<Handler, IoExecutor> op;
202 typename op::ptr p = { asio::detail::addressof(handler),
203 op::ptr::allocate(handler), 0 };
204 p.p = new (p.v) op(handler, io_ex);
205
206 ASIO_HANDLER_CREATION((reactor_.context(), *p.p, "descriptor",
207 &impl, impl.descriptor_, "async_wait"));
208
209 int op_type;
210 switch (w)
211 {
212 case posix::descriptor_base::wait_read:
213 op_type = reactor::read_op;
214 break;
215 case posix::descriptor_base::wait_write:
216 op_type = reactor::write_op;
217 break;
218 case posix::descriptor_base::wait_error:
219 op_type = reactor::except_op;
220 break;
221 default:
222 p.p->ec_ = asio::error::invalid_argument;
223 reactor_.post_immediate_completion(p.p, is_continuation);
224 p.v = p.p = 0;
225 return;
226 }
227
228 start_op(impl, op_type, p.p, is_continuation, false, false);
229 p.v = p.p = 0;
230 }
