51float SUBnote::setupFilters(int *pos, bool automation)
52{
53 //how much the amplitude is normalised (because the harmonics)
54 float reduceamp = 0.0f;
55
56 for(int n = 0; n < numharmonics; ++n) {
57 const float freq = basefreq * pars.POvertoneFreqMult[pos[n]];
58 overtone_freq[n] = freq;
59 overtone_rolloff[n] = computerolloff(freq);
60
61 //the bandwidth is not absolute(Hz); it is relative to frequency
62 const float bw = SUBnoteParameters::convertBandwidth(pars.Pbandwidth,
63 numstages, freq, pars.Pbwscale, pars.Phrelbw[pos[n]]);
64
65 //try to keep same amplitude on all freqs and bw. (empirically)
66 const float hgain = SUBnoteParameters::convertHarmonicMag(pars.Phmag[pos[n]],
67 pars.Phmagtype);
68 const float gain = hgain * sqrt(1500.0f / (bw * freq));
69
70 reduceamp += hgain;
71
72 for(int nph = 0; nph < numstages; ++nph) {
73 float amp = 1.0f;
74 if(nph == 0)
75 amp = gain;
76 initfilter(lfilter[nph + n * numstages], freq + OffsetHz, bw,
77 amp, hgain, automation);
78 if(stereo)
79 initfilter(rfilter[nph + n * numstages], freq + OffsetHz, bw,
80 amp, hgain, automation);
81 }
82 }
83
84 if(reduceamp < 0.001f)
85 reduceamp = 1.0f;
86
87 return reduceamp;
88}

90void SUBnote::setup(float freq,
91 float velocity,
92 int portamento_,
93 int midinote,
94 bool legato)
95{
96 this->velocity = velocity;
97 portamento = portamento_;
98 NoteEnabled = ON;
99 volume = powf(0.1f, 3.0f * (1.0f - pars.PVolume / 96.0f)); //-60 dB .. 0 dB
100 volume *= VelF(velocity, pars.PAmpVelocityScaleFunction);
101 if(pars.PPanning != 0)
102 panning = pars.PPanning / 127.0f;
103 else
104 panning = RND;
105
106 if(!legato) { //normal note
107 numstages = pars.Pnumstages;
108 stereo = pars.Pstereo;
109 start = pars.Pstart;
110 firsttick = 1;
111 }
112
113
114 if(pars.Pfixedfreq == 0)
115 basefreq = freq;
116 else {
117 basefreq = 440.0f;
118 int fixedfreqET = pars.PfixedfreqET;
119 if(fixedfreqET) { //if the frequency varies according the keyboard note
120 float tmp = (midinote - 69.0f) / 12.0f
121 * (powf(2.0f, (fixedfreqET - 1) / 63.0f) - 1.0f);
122 if(fixedfreqET <= 64)
123 basefreq *= powf(2.0f, tmp);
124 else
125 basefreq *= powf(3.0f, tmp);
126 }
127 }
128 int BendAdj = pars.PBendAdjust - 64;
129 if (BendAdj % 24 == 0)
130 BendAdjust = BendAdj / 24;
131 else
132 BendAdjust = BendAdj / 24.0f;
133 float offset_val = (pars.POffsetHz - 64)/64.0f;
134 OffsetHz = 15.0f*(offset_val * sqrtf(fabsf(offset_val)));
135 float detune = getdetune(pars.PDetuneType,
136 pars.PCoarseDetune,
137 pars.PDetune);
138 basefreq *= powf(2.0f, detune / 1200.0f); //detune
139// basefreq*=ctl.pitchwheel.relfreq;//pitch wheel
140
141 int pos[MAX_SUB_HARMONICS];
142 int harmonics;
143
144 pars.activeHarmonics(pos, harmonics);
145
146 if(!legato) //normal note
147 firstnumharmonics = numharmonics = harmonics;
148 else {
149 if(harmonics > firstnumharmonics)
150 numharmonics = firstnumharmonics;
151 else
152 numharmonics = harmonics;
153 }
154
155
156 if(numharmonics == 0) {
157 NoteEnabled = false;
158 return;
159 }
160
161
162 if(!legato) { //normal note
163 lfilter = memory.valloc<bpfilter>(numstages * numharmonics);
164 if(stereo)
165 rfilter = memory.valloc<bpfilter>(numstages * numharmonics);
166 }
167
168 //how much the amplitude is normalised (because the harmonics)
169 float reduceamp = setupFilters(pos, false);
170 oldreduceamp = reduceamp;
171
172 volume /= reduceamp;
173
174 oldpitchwheel = 0;
175 oldbandwidth = 64;
176 if(!legato) { //normal note
177 if(pars.Pfixedfreq == 0)
178 initparameters(basefreq, wm);
179 else
180 initparameters(basefreq / 440.0f * freq, wm);
181 }
182 else {
183 if(pars.Pfixedfreq == 0)
184 freq = basefreq;
185 else
186 freq *= basefreq / 440.0f;
187
188 if(GlobalFilter)
189 GlobalFilter->updateNoteFreq(basefreq);
190 }
191
192 oldamplitude = newamplitude;
193}

241/*
242 * Compute the filters coefficients
243 */
244void SUBnote::computefiltercoefs(bpfilter &filter,
245 float freq,
246 float bw,
247 float gain)
248{
249 if(freq > synth.samplerate_f / 2.0f - 200.0f)
250 freq = synth.samplerate_f / 2.0f - 200.0f;
251
252
253 float omega = 2.0f * PI * freq / synth.samplerate_f;
254 float sn = sinf(omega);
255 float cs = cosf(omega);
256 float alpha = sn * sinh(LOG_2 / 2.0f * bw * omega / sn);
257
258 if(alpha > 1)
259 alpha = 1;
260 if(alpha > bw)
261 alpha = bw;
262
263 filter.b0 = alpha / (1.0f + alpha) * filter.amp * gain;
264 filter.b2 = -alpha / (1.0f + alpha) * filter.amp * gain;
265 filter.a1 = -2.0f * cs / (1.0f + alpha);
266 filter.a2 = (1.0f - alpha) / (1.0f + alpha);
267}

270/*
271 * Initialise the filters
272 */
273void SUBnote::initfilter(bpfilter &filter,
274 float freq,
275 float bw,
276 float amp,
277 float mag,
278 bool automation)
279{
280 if(!automation) {
281 filter.xn1 = 0.0f;
282 filter.xn2 = 0.0f;
283
284 if(start == 0) {
285 filter.yn1 = 0.0f;
286 filter.yn2 = 0.0f;
287 }
288 else {
289 float a = 0.1f * mag; //empirically
290 float p = RND * 2.0f * PI;
291 if(start == 1)
292 a *= RND;
293 filter.yn1 = a * cosf(p);
294 filter.yn2 = a * cosf(p + freq * 2.0f * PI / synth.samplerate_f);
295
296 //correct the error of computation the start amplitude
297 //at very high frequencies
298 if(freq > synth.samplerate_f * 0.96f) {
299 filter.yn1 = 0.0f;
300 filter.yn2 = 0.0f;
301 }
302 }
303 }
304
305 filter.amp = amp;
306 filter.freq = freq;
307 filter.bw = bw;
308 computefiltercoefs(filter, freq, bw, 1.0f);
309}

315inline void SubFilterA(const float coeff[4], float &src, float work[4])
316{
317 work[3] = src*coeff[0]+work[1]*coeff[1]+work[2]*coeff[2]+work[3]*coeff[3];
318 work[1] = src;
319 src = work[3];
320}
321
322inline void SubFilterB(const float coeff[4], float &src, float work[4])
323{
324 work[2] = src*coeff[0]+work[0]*coeff[1]+work[3]*coeff[2]+work[2]*coeff[3];
325 work[0] = src;
326 src = work[2];
327}

329//This dance is designed to minimize unneeded memory operations which can result
330//in quite a bit of wasted time
331void SUBnote::filter(bpfilter &filter, float *smps)
332{
333 assert(synth.buffersize % 8 == 0);
334 float coeff[4] = {filter.b0, filter.b2, -filter.a1, -filter.a2};
335 float work[4] = {filter.xn1, filter.xn2, filter.yn1, filter.yn2};
336
337 for(int i = 0; i < synth.buffersize; i += 8) {
338 SubFilterA(coeff, smps[i + 0], work);
339 SubFilterB(coeff, smps[i + 1], work);
340 SubFilterA(coeff, smps[i + 2], work);
341 SubFilterB(coeff, smps[i + 3], work);
342 SubFilterA(coeff, smps[i + 4], work);
343 SubFilterB(coeff, smps[i + 5], work);
344 SubFilterA(coeff, smps[i + 6], work);
345 SubFilterB(coeff, smps[i + 7], work);
346 }
347 filter.xn1 = work[0];
348 filter.xn2 = work[1];
349 filter.yn1 = work[2];
350 filter.yn2 = work[3];
351}

386/*
387 * Compute how much to reduce amplitude near nyquist or subaudible frequencies.
388 */
389float SUBnote::computerolloff(float freq)
390{
391 const float lower_limit = 10.0f;
392 const float lower_width = 10.0f;
393 const float upper_width = 200.0f;
394 float upper_limit = synth.samplerate / 2.0f;
395
396 if (freq > lower_limit + lower_width &&
397 freq < upper_limit - upper_width)
398 return 1.0f;
399 if (freq <= lower_limit || freq >= upper_limit)
400 return 0.0f;
401 if (freq <= lower_limit + lower_width)
402 return (1.0f - cosf(M_PI * (freq - lower_limit) / lower_width)) / 2.0f;
403 return (1.0f - cosf(M_PI * (freq - upper_limit) / upper_width)) / 2.0f;
404}

406/*
407 * Compute Parameters of SUBnote for each tick
408 */
409void SUBnote::computecurrentparameters()
410{
411 //Recompute parameters for realtime automation
412 if(pars.time && pars.last_update_timestamp == pars.time->time()) {
413 //A little bit of copy/paste for now
414
415 int pos[MAX_SUB_HARMONICS];
416 int harmonics;
417
418 pars.activeHarmonics(pos, harmonics);
419
420 bool delta_harmonics = (harmonics != numharmonics);
421 if(delta_harmonics) {
422 memory.devalloc(lfilter);
423 memory.devalloc(rfilter);
424
425 firstnumharmonics = numharmonics = harmonics;
426 lfilter = memory.valloc<bpfilter>(numstages * numharmonics);
427 if(stereo)
428 rfilter = memory.valloc<bpfilter>(numstages * numharmonics);
429 }
430
431 float reduceamp = setupFilters(pos, !delta_harmonics);
432 volume = volume*oldreduceamp/reduceamp;
433 oldreduceamp = reduceamp;
434 }
435
436 if(FreqEnvelope || BandWidthEnvelope
437 || (oldpitchwheel != ctl.pitchwheel.data)
438 || (oldbandwidth != ctl.bandwidth.data)
439 || portamento) {
440 float envfreq = 1.0f;
441 float envbw = 1.0f;
442
443 if(FreqEnvelope) {
444 envfreq = FreqEnvelope->envout() / 1200;
445 envfreq = powf(2.0f, envfreq);
446 }
447
448 envfreq *=
449 powf(ctl.pitchwheel.relfreq, BendAdjust); //pitch wheel
450
451 //Update frequency while portamento is converging
452 if(portamento) {
453 envfreq *= ctl.portamento.freqrap;
454 if(!ctl.portamento.used) //the portamento has finished
455 portamento = false;
456 }
457
458 if(BandWidthEnvelope) {
459 envbw = BandWidthEnvelope->envout();
460 envbw = powf(2, envbw);
461 }
462
463 envbw *= ctl.bandwidth.relbw; //bandwidth controller
464
465
466 //Recompute High Frequency Dampening Terms
467 for(int n = 0; n < numharmonics; ++n)
468 overtone_rolloff[n] = computerolloff(overtone_freq[n] * envfreq);
469
470
471 //Recompute Filter Coefficients
472 float tmpgain = 1.0f / sqrt(envbw * envfreq);
473 computeallfiltercoefs(lfilter, envfreq, envbw, tmpgain);
474 if(stereo)
475 computeallfiltercoefs(rfilter, envfreq, envbw, tmpgain);
476
477
478 oldbandwidth = ctl.bandwidth.data;
479 oldpitchwheel = ctl.pitchwheel.data;
480 }
481 newamplitude = volume * AmpEnvelope->envout_dB() * 2.0f;
482
483 //Filter
484 if(GlobalFilter)
485 GlobalFilter->update(ctl.filtercutoff.relfreq,
486 ctl.filterq.relq);
487}

489void SUBnote::computeallfiltercoefs(bpfilter *filters, float envfreq,
490 float envbw, float gain)
491{
492 for(int n = 0; n < numharmonics; ++n)
493 for(int nph = 0; nph < numstages; ++nph)
494 computefiltercoefs(filters[nph + n * numstages],
495 filters[nph + n * numstages].freq * envfreq,
496 filters[nph + n * numstages].bw * envbw,
497 nph == 0 ? gain : 1.0);
498}

500void SUBnote::chanOutput(float *out, bpfilter *bp, int buffer_size)
501{
502 float tmprnd[buffer_size];
503 float tmpsmp[buffer_size];
504
505 //Initialize Random Input
506 for(int i = 0; i < buffer_size; ++i)
507 tmprnd[i] = RND * 2.0f - 1.0f;
508
509 //For each harmonic apply the filter on the random input stream
510 //Sum the filter outputs to obtain the output signal
511 for(int n = 0; n < numharmonics; ++n) {
512 float rolloff = overtone_rolloff[n];
513 memcpy(tmpsmp, tmprnd, synth.bufferbytes);
514
515 for(int nph = 0; nph < numstages; ++nph)
516 filter(bp[nph + n * numstages], tmpsmp);
517
518 for(int i = 0; i < synth.buffersize; ++i)
519 out[i] += tmpsmp[i] * rolloff;
520 }
521}

523/*
524 * Note Output
525 */
526int SUBnote::noteout(float *outl, float *outr)
527{
528 memcpy(outl, synth.denormalkillbuf, synth.bufferbytes);
529 memcpy(outr, synth.denormalkillbuf, synth.bufferbytes);
530
531 if(!NoteEnabled)
532 return 0;
533
534 if(stereo) {
535 chanOutput(outl, lfilter, synth.buffersize);
536 chanOutput(outr, rfilter, synth.buffersize);
537
538 if(GlobalFilter)
539 GlobalFilter->filter(outl, outr);
540
541 } else {
542 chanOutput(outl, lfilter, synth.buffersize);
543
544 if(GlobalFilter)
545 GlobalFilter->filter(outl, 0);
546
547 memcpy(outr, outl, synth.bufferbytes);
548 }
549
550 if(firsttick) {
551 int n = 10;
552 if(n > synth.buffersize)
553 n = synth.buffersize;
554 for(int i = 0; i < n; ++i) {
555 float ampfadein = 0.5f - 0.5f * cosf(
556 (float) i / (float) n * PI);
557 outl[i] *= ampfadein;
558 outr[i] *= ampfadein;
559 }
560 firsttick = false;
561 }
562
563 if(ABOVE_AMPLITUDE_THRESHOLD(oldamplitude, newamplitude))
564 // Amplitude interpolation
565 for(int i = 0; i < synth.buffersize; ++i) {
566 float tmpvol = INTERPOLATE_AMPLITUDE(oldamplitude,
567 newamplitude,
568 i,
569 synth.buffersize);
570 outl[i] *= tmpvol * panning;
571 outr[i] *= tmpvol * (1.0f - panning);
572 }
573 else
574 for(int i = 0; i < synth.buffersize; ++i) {
575 outl[i] *= newamplitude * panning;
576 outr[i] *= newamplitude * (1.0f - panning);
577 }
578
579 oldamplitude = newamplitude;
580 computecurrentparameters();
581
582 // Apply legato-specific sound signal modifications
583 legato.apply(*this, outl, outr);
584
585 // Check if the note needs to be computed more
586 if(AmpEnvelope->finished() != 0) {
587 for(int i = 0; i < synth.buffersize; ++i) { //fade-out
588 float tmp = 1.0f - (float)i / synth.buffersize_f;
589 outl[i] *= tmp;
590 outr[i] *= tmp;
591 }
592 KillNote();
593 }
594 return 1;
595}
