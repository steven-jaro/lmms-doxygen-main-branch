164void Controller::setpitchwheel(int value)
165{
166 pitchwheel.data = value;
167 float cents = value / 8192.0f;
168 if(pitchwheel.is_split && cents < 0)
169 cents *= pitchwheel.bendrange_down;
170 else
171 cents *= pitchwheel.bendrange;
172 pitchwheel.relfreq = powf(2, cents / 1200.0f);

204void Controller::setbandwidth(int value)
205{
206 bandwidth.data = value;
207 if(bandwidth.exponential == 0) {
208 float tmp = powf(25.0f, powf(bandwidth.depth / 127.0f, 1.5f)) - 1.0f;
209 if((value < 64) && (bandwidth.depth >= 64))
210 tmp = 1.0f;
211 bandwidth.relbw = (value / 64.0f - 1.0f) * tmp + 1.0f;
212 if(bandwidth.relbw < 0.01f)
213 bandwidth.relbw = 0.01f;
214 }
215 else
216 bandwidth.relbw =
217 powf(25.0f, (value - 64.0f) / 64.0f * (bandwidth.depth / 64.0f));
218 ;
219}

221void Controller::setmodwheel(int value)
222{
223 modwheel.data = value;
224 if(modwheel.exponential == 0) {
225 float tmp =
226 powf(25.0f, powf(modwheel.depth / 127.0f, 1.5f) * 2.0f) / 25.0f;
227 if((value < 64) && (modwheel.depth >= 64))
228 tmp = 1.0f;
229 modwheel.relmod = (value / 64.0f - 1.0f) * tmp + 1.0f;
230 if(modwheel.relmod < 0.0f)
231 modwheel.relmod = 0.0f;
232 }
233 else
234 modwheel.relmod =
235 powf(25.0f, (value - 64.0f) / 64.0f * (modwheel.depth / 80.0f));
236}

273int Controller::initportamento(float oldfreq,
274 float newfreq,
275 bool legatoflag)
276{
277 portamento.x = 0.0f;
278
279 if(legatoflag) { // Legato in progress
280 if(portamento.portamento == 0)
281 return 0;
282 }
283 else // No legato, do the original if...return
284 if((portamento.used != 0) || (portamento.portamento == 0))
285 return 0;
286 ;
287
288 float portamentotime = powf(100.0f, portamento.time / 127.0f) / 50.0f; //portamento time in seconds
289
290 if(portamento.proportional) {
291 //If there is a min(float,float) and a max(float,float) then they
292 //could be used here
293 //Linear functors could also make this nicer
294 if(oldfreq > newfreq) //2 is the center of propRate
295 portamentotime *=
296 powf(oldfreq / newfreq
297 / (portamento.propRate / 127.0f * 3 + .05),
298 (portamento.propDepth / 127.0f * 1.6f + .2));
299 else //1 is the center of propDepth
300 portamentotime *=
301 powf(newfreq / oldfreq
302 / (portamento.propRate / 127.0f * 3 + .05),
303 (portamento.propDepth / 127.0f * 1.6f + .2));
304 }
305
306 if((portamento.updowntimestretch >= 64) && (newfreq < oldfreq)) {
307 if(portamento.updowntimestretch == 127)
308 return 0;
309 portamentotime *= powf(0.1f,
310 (portamento.updowntimestretch - 64) / 63.0f);
311 }
312 if((portamento.updowntimestretch < 64) && (newfreq > oldfreq)) {
313 if(portamento.updowntimestretch == 0)
314 return 0;
315 portamentotime *= powf(0.1f,
316 (64.0f - portamento.updowntimestretch) / 64.0f);
317 }
318
319 //printf("%f->%f : Time %f\n",oldfreq,newfreq,portamentotime);
320
321 portamento.dx = synth.buffersize_f / (portamentotime * synth.samplerate_f);
322 portamento.origfreqrap = oldfreq / newfreq;
323
324 float tmprap = ((portamento.origfreqrap > 1.0f) ?
325 (portamento.origfreqrap) :
326 (1.0f / portamento.origfreqrap));
327
328 float thresholdrap = powf(2.0f, portamento.pitchthresh / 12.0f);
329 if((portamento.pitchthreshtype == 0) && (tmprap - 0.00001f > thresholdrap))
330 return 0;
331 if((portamento.pitchthreshtype == 1) && (tmprap + 0.00001f < thresholdrap))
332 return 0;
333
334 portamento.used = 1;
335 portamento.freqrap = portamento.origfreqrap;
336 return 1;
337}
338
339void Controller::updateportamento()
340{
341 if(portamento.used == 0)
342 return;
343
344 portamento.x += portamento.dx;
345 if(portamento.x > 1.0f) {
346 portamento.x = 1.0f;
347 portamento.used = 0;
348 }
349 portamento.freqrap =
350 (1.0f - portamento.x) * portamento.origfreqrap + portamento.x;
351}
