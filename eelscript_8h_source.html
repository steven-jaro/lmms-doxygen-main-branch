371int eelScriptInst::runcode(const char *codeptr, int showerr, const char *showerrfn, bool canfree, bool ignoreEndOfInputChk, bool doExec)
372{
373 if (m_vm)
374 {
375 NSEEL_CODEHANDLE code = NSEEL_code_compile_ex(m_vm,codeptr,0,canfree ? 0 : NSEEL_CODE_COMPILE_FLAG_COMMONFUNCS);
376 if (code) m_string_context->update_named_vars(m_vm);
377
378 char *err;
379 if (!code && (err=NSEEL_code_getcodeerror(m_vm)))
380 {
381 if (!ignoreEndOfInputChk && (NSEEL_code_geterror_flag(m_vm)&1)) return 1;
382 if (showerr)
383 {
384#ifdef EEL_STRING_DEBUGOUT
385 if (showerr==2)
386 {
387 EEL_STRING_DEBUGOUT("Warning: %s:%s",WDL_get_filepart(showerrfn),err);
388 }
389 else
390 {
391 EEL_STRING_DEBUGOUT("%s:%s",WDL_get_filepart(showerrfn),err);
392 }
393#endif
394 }
395 return -1;
396 }
397 else
398 {
399 if (code)
400 {
401#ifdef EELSCRIPT_DO_DISASSEMBLE
402 codeHandleType *p = (codeHandleType*)code;
403
404 char buf[512];
405 buf[0]=0;
406#ifdef _WIN32
407 GetTempPath(sizeof(buf)-64,buf);
408 lstrcatn(buf,"jsfx-out",sizeof(buf));
409#else
410 lstrcpyn_safe(buf,"/tmp/jsfx-out",sizeof(buf));
411#endif
412 FILE *fp = fopenUTF8(buf,"wb");
413 if (fp)
414 {
415 fwrite(p->code,1,p->code_size,fp);
416 fclose(fp);
417 char buf2[2048];
418#ifdef _WIN32
419 snprintf(buf2,sizeof(buf2),"disasm \"%s\"",buf);
420#else
421 #ifdef __aarch64__
422 snprintf(buf2,sizeof(buf2), "objdump -D -b binary -maarch64 \"%s\"",buf);
423 #elif defined(__arm__)
424 snprintf(buf2,sizeof(buf2), "objdump -D -b binary -m arm \"%s\"",buf);
425 #elif defined(__LP64__)
426 #ifdef __APPLE__
427 snprintf(buf2,sizeof(buf2),"distorm3 --b64 \"%s\"",buf);
428 #else
429 snprintf(buf2,sizeof(buf2),"objdump -D -b binary -m i386:x86-64 \"%s\"",buf);
430 #endif
431 #else
432 snprintf(buf2,sizeof(buf2),"distorm3 --b32 \"%s\"",buf);
433 #endif
434#endif
435 system(buf2);
436 }
437#endif
438
439 if (doExec) NSEEL_code_execute(code);
440 if (canfree) NSEEL_code_free(code);
441 else m_code_freelist.Add((void*)code);
442 }
443 return 0;
444 }
445 }
446 return -1;
447}

450FILE *eelscript_resolvePath(WDL_FastString &usefn, const char *fn, const char *callerfn)
451{
452 // resolve path relative to current
453 int x;
454 bool had_abs=false;
455 for (x=0;x<2; x ++)
456 {
457#ifdef _WIN32
458 if (!x && ((fn[0] == '\\' && fn[1] == '\\') || (fn[0] && fn[1] == ':')))
459#else
460 if (!x && fn[0] == '/')
461#endif
462 {
463 usefn.Set(fn);
464 had_abs=true;
465 }
466 else
467 {
468 const char *fnu = fn;
469 if (x)
470 {
471 while (*fnu) fnu++;
472 while (fnu >= fn && *fnu != '\\' && *fnu != '/') fnu--;
473 if (fnu < fn) break;
474 fnu++;
475 }
476
477 usefn.Set(callerfn);
478 int l=usefn.GetLength();
479 while (l > 0 && usefn.Get()[l-1] != '\\' && usefn.Get()[l-1] != '/') l--;
480 if (l > 0)
481 {
482 usefn.SetLen(l);
483 usefn.Append(fnu);
484 }
485 else
486 {
487 usefn.Set(fnu);
488 }
489 int last_slash_pos=-1;
490 for (l = 0; l < usefn.GetLength(); l ++)
491 {
492 if (usefn.Get()[l] == '/' || usefn.Get()[l] == '\\')
493 {
494 if (usefn.Get()[l+1] == '.' && usefn.Get()[l+2] == '.' &&
495 (usefn.Get()[l+3] == '/' || usefn.Get()[l+3] == '\\'))
496 {
497 if (last_slash_pos >= 0)
498 usefn.DeleteSub(last_slash_pos, l+3-last_slash_pos);
499 else
500 usefn.DeleteSub(0,l+3+1);
501 }
502 else
503 {
504 last_slash_pos=l;
505 }
506 }
507 // take currentfn, remove filename part, add fnu
508 }
509 }
510
511 FILE *fp = fopenUTF8(usefn.Get(),"r");
512 if (fp) return fp;
513 }
514 if (had_abs) usefn.Set(fn);
515 return NULL;
516}

518int eelScriptInst::loadfile(const char *fn, const char *callerfn, bool allowstdin)
519{
520 WDL_FastString usefn;
521 FILE *fp = NULL;
522 if (!strcmp(fn,"-"))
523 {
524 if (callerfn)
525 {
526#ifdef EEL_STRING_DEBUGOUT
527 EEL_STRING_DEBUGOUT("@import: can't import \"-\" (stdin)");
528#endif
529 return -1;
530 }
531 if (allowstdin)
532 {
533 fp = stdin;
534 fn = "(stdin)";
535 }
536 }
537 else if (!callerfn)
538 {
539 fp = fopenUTF8(fn,"r");
540 if (fp) m_loaded_fnlist.Insert(fn,true);
541 }
542 else
543 {
544 fp = eelscript_resolvePath(usefn,fn,callerfn);
545 if (fp)
546 {
547 if (m_loaded_fnlist.Get(usefn.Get()))
548 {
549 fclose(fp);
550 return 0;
551 }
552 m_loaded_fnlist.Insert(usefn.Get(),true);
553 fn = usefn.Get();
554 }
555 }
556
557 if (!fp)
558 {
559#ifdef EEL_STRING_DEBUGOUT
560 if (callerfn)
561 EEL_STRING_DEBUGOUT("Warning: @import could not open '%s'",fn);
562 else
563 EEL_STRING_DEBUGOUT("Error opening %s",fn);
564#endif
565 return -1;
566 }
567
568 WDL_FastString code;
569 char line[4096];
570 for (;;)
571 {
572 line[0]=0;
573 fgets(line,sizeof(line),fp);
574 if (!line[0]) break;
575 if (!strnicmp(line,"@import",7) && isspace((unsigned char)line[7]))
576 {
577 char *p=line+7;
578 while (isspace((unsigned char)*p)) p++;
579
580 char *ep=p;
581 while (*ep) ep++;
582 while (ep>p && isspace((unsigned char)ep[-1])) ep--;
583 *ep=0;
584
585 if (*p) loadfile(p,fn,false);
586 }
587 else
588 {
589 code.Append(line);
590 }
591 }
592 if (fp != stdin) fclose(fp);
593
594 return runcode(code.Get(),callerfn ? 2 : 1, fn,false,true,!callerfn);
595}

597char *eelScriptInst::evalCacheGet(const char *str, NSEEL_CODEHANDLE *ch)
598{
599 // should mutex protect if multiple threads access this eelScriptInst context
600 int x=m_eval_cache.GetSize();
601 while (--x >= 0)
602 {
603 char *ret;
604 if (!strcmp(ret=m_eval_cache.Get()[x].str, str))
605 {
606 *ch = m_eval_cache.Get()[x].ch;
607 m_eval_cache.Delete(x);
608 return ret;
609 }
610 }
611 return NULL;
612}

615{

622 NSEEL_code_free(m_eval_cache.Get()->ch);
623 free(m_eval_cache.Get()->str);
624 m_eval_cache.Delete(0);
625 }
626 m_eval_cache.Add(ecc);

672void eelScriptInst::runCodeQ(WDL_Queue *q, const char *callername)
673{
674 const int endptr = q->Available();
675 int offs = 0;
676 while (offs < endptr)
677 {
678 if (q->Available() < endptr) break; // should never happen, but safety first!
679
680 const char *ptr = (const char *)q->Get() + offs;
681 offs += strlen(ptr)+1;
682
683 NSEEL_CODEHANDLE ch=NULL;
684 char *sv=evalCacheGet(ptr,&ch);
685
686 if (!sv) sv=strdup(ptr);
687 if (!ch) ch=NSEEL_code_compile(m_vm,sv,0);
688 if (!ch)
689 {
