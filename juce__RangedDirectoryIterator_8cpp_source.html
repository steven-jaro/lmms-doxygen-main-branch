54bool RangedDirectoryIterator::next()
55{
56 const auto result = iterator->next (&entry.directory,
57 &entry.hidden,
58 &entry.fileSize,
59 &entry.modTime,
60 &entry.creationTime,
61 &entry.readOnly);
62 if (result)
63 entry.file = iterator->getFile();
64 else
65 entry = {};
