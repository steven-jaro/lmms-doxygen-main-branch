69void LcdWidget::setValue(int value)
70{
71 QString s = m_textForValue[value];
72 if (s.isEmpty())
73 {
74 s = QString::number(value);
75 if (m_leadingZero)
76 {
77 s = s.rightJustified(m_numDigits, '0');
78 }
79 }
80
81 if (m_display != s)
82 {
83 m_display = s;

85 update();

137void LcdWidget::paintEvent( QPaintEvent* )
138{
139 QPainter p( this );
140
141 QSize cellSize( m_cellWidth, m_cellHeight );
142
143 QRect cellRect( 0, 0, m_cellWidth, m_cellHeight );
144
145 int margin = 1; // QStyle::PM_DefaultFrameWidth;
146 //int lcdWidth = m_cellWidth * m_numDigits + (margin*m_marginWidth)*2;
147
148// p.translate( width() / 2 - lcdWidth / 2, 0 );
149 p.save();
150
151 // Don't skip any space and don't draw margin on the left side in seamless mode
152 if (m_seamlessLeft)
153 {
154 p.translate(0, margin);
155 }
156 else
157 {
158 p.translate(margin, margin);
159 // Left Margin
160 p.drawPixmap(cellRect, m_lcdPixmap,
161 QRect(QPoint(charsPerPixmap * m_cellWidth, isEnabled() ? 0 : m_cellHeight), cellSize));
162
163 p.translate(m_marginWidth, 0);
164 }
165
166 // Padding
167 for( int i=0; i < m_numDigits - m_display.length(); i++ )
168 {
169 p.drawPixmap(cellRect, m_lcdPixmap, QRect(QPoint(10 * m_cellWidth, isEnabled() ? 0 : m_cellHeight), cellSize));
170 p.translate( m_cellWidth, 0 );
171 }
172
173 // Digits
174 for (const auto& digit : m_display)
175 {
176 int val = digit.digitValue();
177 if( val < 0 )
178 {
179 if (digit == '-') val = 11;
180 else
181 val = 10;
182 }
183 p.drawPixmap(cellRect, m_lcdPixmap, QRect(QPoint(val * m_cellWidth, isEnabled() ? 0 : m_cellHeight), cellSize));
184 p.translate( m_cellWidth, 0 );
185 }
186
187 // Right Margin
188 p.drawPixmap(QRect(0, 0, m_seamlessRight ? 0 : m_marginWidth - 1, m_cellHeight), m_lcdPixmap,
189 QRect(charsPerPixmap * m_cellWidth, isEnabled() ? 0 : m_cellHeight, m_cellWidth / 2, m_cellHeight));
190
191 p.restore();
192
193 // Border
194 // When either the left or right edge is seamless, the border drawing must be done
195 // by the encapsulating class (usually LcdFloatSpinBox).
196 if (!m_seamlessLeft && !m_seamlessRight)
197 {
198 QStyleOptionFrame opt;
199 opt.initFrom(this);
200 opt.state = QStyle::State_Sunken;
201 opt.rect = QRect(0, 0, m_cellWidth * m_numDigits + (margin + m_marginWidth) * 2 - 1,
202 m_cellHeight + (margin * 2));
203
204 style()->drawPrimitive(QStyle::PE_Frame, &opt, &p, this);
205 }
206
207 p.resetTransform();
208
209 // Label
210 if( !m_label.isEmpty() )
211 {
212 p.setFont(adjustedToPixelSize(p.font(), DEFAULT_FONT_SIZE));
213 p.setPen( textShadowColor() );
214 p.drawText(width() / 2 -
215 horizontalAdvance(p.fontMetrics(), m_label) / 2 + 1,
216 height(), m_label);
217 p.setPen( textColor() );
218 p.drawText(width() / 2 -
219 horizontalAdvance(p.fontMetrics(), m_label) / 2,
220 height() - 1, m_label);
221 }
222
223}

247void LcdWidget::updateSize()
248{
249 const int marginX1 = m_seamlessLeft ? 0 : 1 + m_marginWidth;
250 const int marginX2 = m_seamlessRight ? 0 : 1 + m_marginWidth;
251 const int marginY = 1;
252 if (m_label.isEmpty())
253 {
254 setFixedSize(
255 m_cellWidth * m_numDigits + marginX1 + marginX2,
256 m_cellHeight + (2 * marginY)
257 );
258 }
259 else
260 {
261 setFixedSize(
262 qMax<int>(
263 m_cellWidth * m_numDigits + marginX1 + marginX2,
264 horizontalAdvance(QFontMetrics(adjustedToPixelSize(font(), DEFAULT_FONT_SIZE)), m_label)
265 ),
266 m_cellHeight + (2 * marginY) + 9
267 );
268 }
269
270 update();
271}
