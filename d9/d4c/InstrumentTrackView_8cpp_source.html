185void InstrumentTrackView::toggleMidiCCRack()
186{
187 // Lazy creation: midiCCRackView is only created when accessed the first time.
188 // this->model() returns pointer to the InstrumentTrack who owns this InstrumentTrackView.
189 if (!m_midiCCRackView)
190 {
191 m_midiCCRackView = std::unique_ptr<MidiCCRackView>(new MidiCCRackView(this->model()));
192 }
193
194 if (m_midiCCRackView->parentWidget()->isVisible())
195 {
196 m_midiCCRackView->parentWidget()->hide();
197 }
198 else
199 {
200 m_midiCCRackView->parentWidget()->show();
201 m_midiCCRackView->show();
202 }
203}

364//FIXME: This is identical to SampleTrackView::createMixerMenu
365QMenu * InstrumentTrackView::createMixerMenu(QString title, QString newMixerLabel)
366{
367 int channelIndex = model()->mixerChannelModel()->value();
368
369 MixerChannel *mixerChannel = Engine::mixer()->mixerChannel( channelIndex );
370
371 // If title allows interpolation, pass channel index and name
372 if ( title.contains( "%2" ) )
373 {
374 title = title.arg( channelIndex ).arg( mixerChannel->m_name );
375 }
376
377 auto mixerMenu = new QMenu(title);
378
379 mixerMenu->addAction( newMixerLabel, this, SLOT(createMixerLine()));
380 mixerMenu->addSeparator();
381
382 for (int i = 0; i < Engine::mixer()->numChannels(); ++i)
383 {
384 MixerChannel * currentChannel = Engine::mixer()->mixerChannel( i );
385
386 if ( currentChannel != mixerChannel )
387 {
388 auto index = currentChannel->index();
389 QString label = tr( "%1: %2" ).arg(index).arg(currentChannel->m_name);
390 mixerMenu->addAction(label, [this, index](){
391 assignMixerLine(index);
392 });
393 }
394 }
395
396 return mixerMenu;
397}

399QPixmap InstrumentTrackView::determinePixmap(InstrumentTrack* instrumentTrack)
400{
401 if (instrumentTrack)
402 {
403 Instrument* instrument = instrumentTrack->instrument();
404
405 if (instrument && instrument->descriptor())
406 {
407 const PixmapLoader* pl = instrument->key().isValid()
408 ? instrument->key().logo()
409 : instrument->descriptor()->logo;
410
411 if (pl)
412 {
413 return pl->pixmap();
414 }
415 }
416 }
417
418 return embed::getIconPixmap("instrument_track");
419}
