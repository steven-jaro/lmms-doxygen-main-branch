36GroupBox::GroupBox( const QString & _caption, QWidget * _parent ) :
37 QWidget( _parent ),
38 BoolModelView( nullptr, this ),
39 m_caption( _caption ),
40 m_titleBarHeight( 11 )
41{
42 m_led = new PixmapButton( this, _caption );
43 m_led->setCheckable( true );
44 m_led->move( 3, 0 );
45 m_led->setActiveGraphic( embed::getIconPixmap( "led_green" ) );
46 m_led->setInactiveGraphic( embed::getIconPixmap( "led_off" ) );
47
48 setModel( new BoolModel( false, nullptr, _caption, true ) );
49 setAutoFillBackground( true );
50 unsetCursor();
51}

93void GroupBox::paintEvent( QPaintEvent * pe )
94{
95 QPainter p( this );
96
97 // Draw background
98 p.fillRect( 0, 0, width() - 1, height() - 1, p.background() );
99
100 // outer rect
101 p.setPen( p.background().color().darker( 150 ) );
102 p.drawRect( 0, 0, width() - 1, height() - 1 );
103
104 // draw line below titlebar
105 p.fillRect( 1, 1, width() - 2, m_titleBarHeight + 1, p.background().color().darker( 150 ) );
106
107 // draw text
108 p.setPen( palette().color( QPalette::Active, QPalette::Text ) );
109 p.setFont(adjustedToPixelSize(font(), DEFAULT_FONT_SIZE));
110
111 int const captionX = ledButtonShown() ? 22 : 6;
112 p.drawText(captionX, m_titleBarHeight, m_caption);
113}
