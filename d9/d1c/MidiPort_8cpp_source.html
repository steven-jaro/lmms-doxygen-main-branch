45MidiPort::MidiPort( const QString& name,
46 MidiClient* client,
47 MidiEventProcessor* eventProcessor,
48 Model* parent,
49 Mode mode ) :
50 Model( parent ),
51 m_readablePortsMenu( nullptr ),
52 m_writablePortsMenu( nullptr ),
53 m_midiClient( client ),
54 m_midiEventProcessor( eventProcessor ),
55 m_mode( mode ),
56 m_inputChannelModel( 0, 0, MidiChannelCount, this, tr( "Input channel" ) ),
57 m_outputChannelModel( 1, 0, MidiChannelCount, this, tr( "Output channel" ) ),
58 m_inputControllerModel(MidiController::NONE, MidiController::NONE, MidiControllerCount - 1, this, tr( "Input controller" )),
59 m_outputControllerModel(MidiController::NONE, MidiController::NONE, MidiControllerCount - 1, this, tr( "Output controller" )),
60 m_fixedInputVelocityModel( -1, -1, MidiMaxVelocity, this, tr( "Fixed input velocity" ) ),
61 m_fixedOutputVelocityModel( -1, -1, MidiMaxVelocity, this, tr( "Fixed output velocity" ) ),
62 m_fixedOutputNoteModel( -1, -1, MidiMaxKey, this, tr( "Fixed output note" ) ),
63 m_outputProgramModel( 1, 1, MidiProgramCount, this, tr( "Output MIDI program" ) ),
64 m_baseVelocityModel( MidiMaxVelocity/2, 1, MidiMaxVelocity, this, tr( "Base velocity" ) ),
65 m_readableModel( false, this, tr( "Receive MIDI-events" ) ),
66 m_writableModel( false, this, tr( "Send MIDI-events" ) )
67{
68 m_midiClient->addPort( this );
69
70 m_readableModel.setValue( m_mode == Mode::Input || m_mode == Mode::Duplex );
71 m_writableModel.setValue( m_mode == Mode::Output || m_mode == Mode::Duplex );
72
73 connect( &m_readableModel, SIGNAL(dataChanged()),
74 this, SLOT(updateMidiPortMode()), Qt::DirectConnection );
75 connect( &m_writableModel, SIGNAL(dataChanged()),
76 this, SLOT(updateMidiPortMode()), Qt::DirectConnection );
77 connect( &m_outputProgramModel, SIGNAL(dataChanged()),
78 this, SLOT(updateOutputProgram()), Qt::DirectConnection );
79
80
81 // when using with non-raw-clients we can provide buttons showing
82 // our port-menus when being clicked
83 if( m_midiClient->isRaw() == false )
84 {
85 updateReadablePorts();
86 updateWritablePorts();
87
88 // we want to get informed about port-changes!
89 m_midiClient->connectRPChanged( this, SLOT(updateReadablePorts()));
90 m_midiClient->connectWPChanged( this, SLOT(updateWritablePorts()));
91 }
92
93 updateMidiPortMode();
94}

130void MidiPort::processInEvent( const MidiEvent& event, const TimePos& time )
131{
132 // mask event
133 if( isInputEnabled() &&
134 ( inputChannel() == 0 || inputChannel()-1 == event.channel() ) )
135 {
136 MidiEvent inEvent = event;
137 if( event.type() == MidiNoteOn ||
138 event.type() == MidiNoteOff ||
139 event.type() == MidiKeyPressure )
140 {
141 if( inEvent.key() < 0 || inEvent.key() >= NumKeys )
142 {
143 return;
144 }
145
146 if( fixedInputVelocity() >= 0 && inEvent.velocity() > 0 )
147 {
148 inEvent.setVelocity( fixedInputVelocity() );
149 }
150 }
151
152 m_midiEventProcessor->processInEvent( inEvent, time );
153 }
154}

159void MidiPort::processOutEvent( const MidiEvent& event, const TimePos& time )
160{
161 // When output is enabled, route midi events if the selected channel matches
162 // the event channel or if there's no selected channel (value 0, represented by "--")
163 if( isOutputEnabled() && ( outputChannel() == 0 || realOutputChannel() == event.channel() ) )
164 {
165 MidiEvent outEvent = event;
166
167 if( fixedOutputVelocity() >= 0 && event.velocity() > 0 &&
168 ( event.type() == MidiNoteOn || event.type() == MidiKeyPressure ) )
169 {
170 outEvent.setVelocity( fixedOutputVelocity() );
171 }
172
173 if( fixedOutputNote() >= 0 &&
174 ( event.type() == MidiNoteOn || event.type() == MidiNoteOff || event.type() == MidiKeyPressure ) )
175 {
176 outEvent.setKey( fixedOutputNote() );
177 }
178
179 m_midiClient->processOutEvent( outEvent, time, this );
180 }
181}

186void MidiPort::saveSettings( QDomDocument& doc, QDomElement& thisElement )
187{
188 m_inputChannelModel.saveSettings( doc, thisElement, "inputchannel" );
189 m_outputChannelModel.saveSettings( doc, thisElement, "outputchannel" );
190 m_inputControllerModel.saveSettings( doc, thisElement, "inputcontroller" );
191 m_outputControllerModel.saveSettings( doc, thisElement, "outputcontroller" );
192 m_fixedInputVelocityModel.saveSettings( doc, thisElement, "fixedinputvelocity" );
193 m_fixedOutputVelocityModel.saveSettings( doc, thisElement, "fixedoutputvelocity" );
194 m_fixedOutputNoteModel.saveSettings( doc, thisElement, "fixedoutputnote" );
195 m_outputProgramModel.saveSettings( doc, thisElement, "outputprogram" );
196 m_baseVelocityModel.saveSettings( doc, thisElement, "basevelocity" );
197 m_readableModel.saveSettings( doc, thisElement, "readable" );
198 m_writableModel.saveSettings( doc, thisElement, "writable" );
199
200 if( isInputEnabled() )
201 {
202 QString rp;
203 for( Map::ConstIterator it = m_readablePorts.begin(); it != m_readablePorts.end(); ++it )
204 {
205 if( it.value() )
206 {
207 rp += it.key() + ",";
208 }
209 }
210 // cut off comma
211 if( rp.length() > 0 )
212 {
213 rp.truncate( rp.length() - 1 );
214 }
215 thisElement.setAttribute( "inports", rp );
216 }
217
218 if( isOutputEnabled() )
219 {
220 QString wp;
221 for( Map::ConstIterator it = m_writablePorts.begin(); it != m_writablePorts.end(); ++it )
222 {
223 if( it.value() )
224 {
225 wp += it.key() + ",";
226 }
227 }
228 // cut off comma
229 if( wp.length() > 0 )
230 {
231 wp.truncate( wp.length() - 1 );
232 }
233 thisElement.setAttribute( "outports", wp );
234 }
235}

240void MidiPort::loadSettings( const QDomElement& thisElement )
241{
242 m_inputChannelModel.loadSettings( thisElement, "inputchannel" );
243 m_outputChannelModel.loadSettings( thisElement, "outputchannel" );
244 m_inputControllerModel.loadSettings( thisElement, "inputcontroller" );
245 m_outputControllerModel.loadSettings( thisElement, "outputcontroller" );
246 m_fixedInputVelocityModel.loadSettings( thisElement, "fixedinputvelocity" );
247 m_fixedOutputVelocityModel.loadSettings( thisElement, "fixedoutputvelocity" );
248 m_fixedOutputNoteModel.loadSettings( thisElement, "fixedoutputnote" );
249 m_outputProgramModel.loadSettings( thisElement, "outputprogram" );
250 m_baseVelocityModel.loadSettings( thisElement, "basevelocity" );
251 m_readableModel.loadSettings( thisElement, "readable" );
252 m_writableModel.loadSettings( thisElement, "writable" );
253
254 // restore connections
255
256 if( isInputEnabled() )
257 {
258 QStringList rp = thisElement.attribute( "inports" ).split( ',' );
259 for( Map::ConstIterator it = m_readablePorts.begin(); it != m_readablePorts.end(); ++it )
260 {
261 if( it.value() != ( rp.indexOf( it.key() ) != -1 ) )
262 {
263 subscribeReadablePort( it.key() );
264 }
265 }
266 emit readablePortsChanged();
267 }
268
269 if( isOutputEnabled() )
270 {
271 QStringList wp = thisElement.attribute( "outports" ).split( ',' );
272 for( Map::ConstIterator it = m_writablePorts.begin(); it != m_writablePorts.end(); ++it )
273 {
274 if( it.value() != ( wp.indexOf( it.key() ) != -1 ) )
275 {
276 subscribeWritablePort( it.key() );
277 }
278 }
279 emit writablePortsChanged();
280 }
281
282 if( thisElement.hasAttribute( "basevelocity" ) == false )
283 {
284 // for projects created by LMMS < 0.9.92 there's no value for the base
285 // velocity and for compat reasons we have to stick with maximum velocity
286 // which did not allow note volumes > 100%
287 m_baseVelocityModel.setValue( MidiMaxVelocity );
288 }

326void MidiPort::updateMidiPortMode()
327{
328 // this small lookup-table makes everything easier
329 static const Mode modeTable[2][2] =
330 {
331 { Mode::Disabled, Mode::Output },
332 { Mode::Input, Mode::Duplex }
333 } ;
334 setMode( modeTable[m_readableModel.value()][m_writableModel.value()] );
335
336 // check whether we have to dis-check items in connection-menu
337 if( !isInputEnabled() )
338 {
339 for( Map::ConstIterator it = m_readablePorts.begin(); it != m_readablePorts.end(); ++it )
340 {
341 // subscribed?
342 if( it.value() )
343 {
344 subscribeReadablePort( it.key(), false );
345 }
346 }
347 }
348
349 if( !isOutputEnabled() )
350 {
351 for( Map::ConstIterator it = m_writablePorts.begin(); it != m_writablePorts.end(); ++it )
352 {
353 // subscribed?
354 if( it.value() )
355 {
356 subscribeWritablePort( it.key(), false );
357 }
358 }
359 }
360
361 emit readablePortsChanged();
362 emit writablePortsChanged();
363 emit modeChanged();
364
365 if( Engine::getSong() )
366 {
367 Engine::getSong()->setModified();
368 }
369}

374void MidiPort::updateReadablePorts()
375{
376 // first save all selected ports
377 QStringList selectedPorts;
378 for( Map::ConstIterator it = m_readablePorts.begin(); it != m_readablePorts.end(); ++it )
379 {
380 if( it.value() )
381 {
382 selectedPorts.push_back( it.key() );
383 }
384 }
385
386 m_readablePorts.clear();
387 const QStringList& wp = m_midiClient->readablePorts();
388 // now insert new ports and restore selections
389 for (const auto& port : wp)
390 {
391 m_readablePorts[port] = (selectedPorts.indexOf(port) != -1);
392 }
393
394 emit readablePortsChanged();
395}
