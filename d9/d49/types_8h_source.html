461/**
462 * A port has a set of flags that are formed by AND-ing together the
463 * desired values from the list below. The flags "JackPortIsInput" and
464 * "JackPortIsOutput" are mutually exclusive and it is an error to use
465 * them both.
466 */
467enum JackPortFlags {
468
469 /**
470 * if JackPortIsInput is set, then the port can receive
471 * data.
472 */
473 JackPortIsInput = 0x1,
474
475 /**
476 * if JackPortIsOutput is set, then data can be read from
477 * the port.
478 */
479 JackPortIsOutput = 0x2,
480
481 /**
482 * if JackPortIsPhysical is set, then the port corresponds
483 * to some kind of physical I/O connector.
484 */
485 JackPortIsPhysical = 0x4,
486
487 /**
488 * if JackPortCanMonitor is set, then a call to
489 * jack_port_request_monitor() makes sense.
490 *
491 * Precisely what this means is dependent on the client. A typical
492 * result of it being called with TRUE as the second argument is
493 * that data that would be available from an output port (with
494 * JackPortIsPhysical set) is sent to a physical output connector
495 * as well, so that it can be heard/seen/whatever.
496 *
497 * Clients that do not control physical interfaces
498 * should never create ports with this bit set.
499 */
500 JackPortCanMonitor = 0x8,
501
502 /**
503 * JackPortIsTerminal means:
504 *
505 * for an input port: the data received by the port
506 * will not be passed on or made
507 * available at any other port
508 *
509 * for an output port: the data available at the port
510 * does not originate from any other port
511 *
512 * Audio synthesizers, I/O hardware interface clients, HDR
513 * systems are examples of clients that would set this flag for
514 * their ports.
515 */
516 JackPortIsTerminal = 0x10,

636/**
637 * Prototype for the @a sync_callback defined by slow-sync clients.
638 * When the client is active, this callback is invoked just before
639 * process() in the same thread. This occurs once after registration,
640 * then subsequently whenever some client requests a new position, or
641 * the transport enters the ::JackTransportStarting state. This
642 * realtime function must not wait.
643 *
644 * The transport @a state will be:
645 *
646 * - ::JackTransportStopped when a new position is requested;
647 * - ::JackTransportStarting when the transport is waiting to start;
648 * - ::JackTransportRolling when the timeout has expired, and the
649 * position is now a moving target.
650 *
651 * @param state current transport state.
652 * @param pos new transport position.
653 * @param arg the argument supplied by jack_set_sync_callback().
654 *
655 * @return TRUE (non-zero) when ready to roll.
656 */

662/**
663 * Prototype for the @a timebase_callback used to provide extended
664 * position information. Its output affects all of the following
665 * process cycle. This realtime function must not wait.
666 *
667 * This function is called immediately after process() in the same
668 * thread whenever the transport is rolling, or when any client has
669 * requested a new position in the previous cycle. The first cycle
670 * after jack_set_timebase_callback() is also treated as a new
671 * position, or the first cycle after jack_activate() if the client
672 * had been inactive.
673 *
674 * The timebase master may not use its @a pos argument to set @a
675 * pos->frame. To change position, use jack_transport_reposition() or

685 * The @a timebase_callback's task is to update the extended
686 * information here.

688 * the first cycle after the @a timebase_callback is defined.
689 * @param arg the argument supplied by jack_set_timebase_callback().
