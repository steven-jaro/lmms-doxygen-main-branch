70void Draggable::paintEvent(QPaintEvent* event)
71{

85void Draggable::mouseMoveEvent(QMouseEvent* me)
86{
87 QPoint pPos = mapToParent(me->pos());
88
89 if (m_buttonPressed && pPos != m_lastMousePos)
90 {
91 float point = (m_directionOfManipulation == DirectionOfManipulation::Vertical) ? pPos.y() : pPos.x();
92 float progress = (point - m_pointA) / (m_pointB - m_pointA);
93
94 if (progress >= 0 && progress <= 1)
95 {
96 float newVal = progress * (model()->maxValue() - model()->minValue()) + model()->minValue();
97 model()->setValue(newVal);
98 }
99 else if (progress < 0)
100 {
101 model()->setValue(model()->minValue());
102 }
103 else
104 {
105 model()->setValue(model()->maxValue());
106 }
107
108 emit sliderMoved(model()->value());
109 m_lastMousePos = pPos;
110 s_textFloat->setText(displayValue());
111 s_textFloat->moveGlobal(this, QPoint(width() + 2, 0));
112 }
113}

115void Draggable::handleMovement()
116{
117 float newCoord = std::lerp(m_pointA, m_pointB, (model()->value() - model()->minValue()) / (model()->maxValue() - model()->minValue()));
118 if (m_directionOfManipulation == DirectionOfManipulation::Vertical)
119 {
120 move(x(), newCoord - m_pixmap.height() / 2.f);
121 }
122 else if (m_directionOfManipulation == DirectionOfManipulation::Horizontal)
123 {
124 move(newCoord - m_pixmap.width() / 2.f, y());
125 }
126}
