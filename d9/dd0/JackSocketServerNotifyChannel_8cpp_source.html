29int JackSocketServerNotifyChannel::Open(const char* server_name)
30{
31 if (fRequestSocket.Connect(jack_server_dir, server_name, 0) < 0) {
32 jack_error("Cannot connect to server socket");

51void JackSocketServerNotifyChannel::Notify(int refnum, int notify, int value)
52{
53 JackClientNotificationRequest req(refnum, notify, value);
54 if (req.Write(&fRequestSocket) < 0) {
55 jack_error("Could not write notification ref = %d notify = %d", refnum, notify);
