117void PluginBrowser::onFilterChanged( const QString & filter )
118{
119 int rootCount = m_descTree->topLevelItemCount();
120 for (int rootIndex = 0; rootIndex < rootCount; ++rootIndex)
121 {
122 QTreeWidgetItem * root = m_descTree->topLevelItem( rootIndex );
123
124 int itemCount = root->childCount();
125 for (int itemIndex = 0; itemIndex < itemCount; ++itemIndex)
126 {
127 QTreeWidgetItem * item = root->child( itemIndex );
128 auto descWidget = static_cast<PluginDescWidget*>(m_descTree->itemWidget(item, 0));
129 if (descWidget->name().contains(filter, Qt::CaseInsensitive))
130 {
131 item->setHidden( false );
132 }
133 else
134 {
135 item->setHidden( true );
136 }
137 }
138 }

142void PluginBrowser::addPlugins()
143{
144 // Add a root node to the plugin tree with the specified `label` and return it
145 const auto addRoot = [this](auto label)
146 {
147 const auto root = new QTreeWidgetItem();
148 root->setText(0, label);
149 m_descTree->addTopLevelItem(root);
150 return root;
151 };
152
153 // Add the plugin identified by `key` to the tree under the root node `root`
154 const auto addPlugin = [this](const auto& key, auto root)
155 {
156 const auto item = new QTreeWidgetItem();
157 root->addChild(item);
158 m_descTree->setItemWidget(item, 0, new PluginDescWidget(key, m_descTree));
159 };
160
161 // Remove any existing plugins from the tree
162 m_descTree->clear();
163
164 // Fetch and sort all instrument plugin descriptors
165 auto descs = getPluginFactory()->descriptors(Plugin::Type::Instrument);
166 std::sort(descs.begin(), descs.end(),
167 [](auto d1, auto d2)
168 {
169 return qstricmp(d1->displayName, d2->displayName) < 0;
170 }
171 );
172
173 // Add a root node to the tree for native LMMS plugins
174 const auto lmmsRoot = addRoot("LMMS");
175 lmmsRoot->setExpanded(true);
176
177 // Add all of the descriptors to the tree
178 for (const auto desc : descs)
179 {
180 if (desc->subPluginFeatures)
181 {
182 // Fetch and sort all subplugins for this plugin descriptor
183 auto subPluginKeys = Plugin::Descriptor::SubPluginFeatures::KeyList{};
184 desc->subPluginFeatures->listSubPluginKeys(desc, subPluginKeys);
185 std::sort(subPluginKeys.begin(), subPluginKeys.end(),
186 [](const auto& l, const auto& r)
187 {
188 return QString::compare(l.displayName(), r.displayName(), Qt::CaseInsensitive) < 0;
189 }
190 );
191
192 // Create a root node for this plugin and add the subplugins under it
193 const auto root = addRoot(desc->displayName);
194 for (const auto& key : subPluginKeys) { addPlugin(key, root); }
195 }
196 else
197 {
198 addPlugin(Plugin::Descriptor::SubPluginFeatures::Key(desc, desc->name), lmmsRoot);
199 }
200 }
201}

232void PluginDescWidget::paintEvent( QPaintEvent * )
233{
234
235 QPainter p( this );
236
237 // Paint everything according to the style sheet
238 QStyleOption o;
239 o.initFrom( this );
240 style()->drawPrimitive( QStyle::PE_Widget, &o, &p, this );
241
242 // Draw the rest
243 const int s = 16 + ( 32 * ( qBound( 24, height(), 60 ) - 24 ) ) /
244 ( 60 - 24 );
245 const QSize logo_size( s, s );
246 QPixmap logo = m_logo.scaled( logo_size, Qt::KeepAspectRatio,
247 Qt::SmoothTransformation );
248 p.drawPixmap( 4, 4, logo );
249
250 QFont f = p.font();
251 if ( m_mouseOver )
252 {
253 f.setBold( true );
254 }
255
256 p.setFont( f );
257 p.drawText( 10 + logo_size.width(), 15, m_pluginKey.displayName());
258}
