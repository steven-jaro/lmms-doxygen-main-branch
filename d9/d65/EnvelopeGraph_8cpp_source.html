65void EnvelopeGraph::contextMenuEvent(QContextMenuEvent* event)
66{
67 QMenu menu(this);
68 QMenu* scalingMenu = menu.addMenu(tr("Scaling"));
69 scalingMenu->setToolTipsVisible(true);
70
71 auto switchTo = [&](ScalingMode scaling)
72 {
73 if (m_scaling != scaling)
74 {
75 m_scaling = scaling;
76 update();
77 }
78 };
79
80 auto addScalingEntry = [scalingMenu, &switchTo, this](const QString& text, const QString& toolTip, ScalingMode scaling)
81 {
82 QAction* action = scalingMenu->addAction(text, [&switchTo, scaling]() { switchTo(scaling); });
83 action->setCheckable(true);
84 action->setChecked(m_scaling == scaling);
85 action->setToolTip(toolTip);
86 };
87
88 addScalingEntry(
89 tr("Dynamic"),
90 tr("Uses absolute spacings but switches to relative spacing if it's running out of space"),
91 ScalingMode::Dynamic);
92 addScalingEntry(
93 tr("Absolute"),
94 tr("Provides enough potential space for each segment but does not scale"),
95 ScalingMode::Absolute);
96 addScalingEntry(
97 tr("Relative"),
98 tr("Always uses all of the available space to display the envelope graph"),
99 ScalingMode::Relative);
100
101 menu.exec(event->globalPos());
102}

104void EnvelopeGraph::paintEvent(QPaintEvent*)
105{
106 QPainter p{this};
107 p.setRenderHint(QPainter::Antialiasing);
108
109 // Draw the graph background
110 p.drawPixmap(rect(), m_envGraph);
111
112 const auto* params = castModel<EnvelopeAndLfoParameters>();
113 if (!params) { return; }
114
115 // For the calculation of the percentages we will for now make use of the knowledge
116 // that the range goes from 0 to a positive max value, i.e. that it is in [0, max].
117 const float amount = params->getAmountModel().value();
118
119 const float predelay = params->getPredelayModel().value();
120 const float predelayPercentage = predelay / params->getPredelayModel().maxValue();
121
122 const float attack = params->getAttackModel().value();
123 const float attackPercentage = attack / params->getAttackModel().maxValue();
124
125 const float hold = params->getHoldModel().value();
126 const float holdPercentage = hold / params->getHoldModel().maxValue();
127
128 const float decay = params->getDecayModel().value();
129 const float decayPercentage = decay / params->getDecayModel().maxValue();
130
131 const float sustain = params->getSustainModel().value();
132
133 const float release = params->getReleaseModel().value();
134 const float releasePercentage = release / params->getReleaseModel().maxValue();
135
136 // The margin to the left and right so that we do not clip too much of the lines and markers
137 const float margin = 2.0;
138 const float availableWidth = width() - margin * 2;
139
140 // Now determine the maximum width for one segment according to the scaling setting.
141 // The different scalings use different means to compute the maximum available width per segment.
142 const auto computeMaximumSegmentWidthAbsolute = [&]() -> float
143 {
144 return availableWidth / 5.;
145 };
146
147 const auto computeMaximumSegmentWidthRelative = [&]() -> float
148 {
149 const float sumOfSegments = predelayPercentage + attackPercentage + holdPercentage + decayPercentage + releasePercentage;
150
151 return sumOfSegments != 0.
152 ? availableWidth / sumOfSegments
153 : computeMaximumSegmentWidthAbsolute();
154 };
155
156 const auto computeMaximumSegmentWidthDynamic = [&]() -> float
157 {
158 const float sumOfSegments = predelayPercentage + attackPercentage + holdPercentage + decayPercentage + releasePercentage;
159
160 float preliminarySegmentWidth = 80. / 182. * availableWidth;
161
162 const float neededWidth = sumOfSegments * preliminarySegmentWidth;
163
164 if (neededWidth > availableWidth && sumOfSegments != 0.)
165 {
166 return computeMaximumSegmentWidthRelative();
167 }
168
169 return preliminarySegmentWidth;
170 };
171
172 // This is the maximum width that each of the five segments (DAHDR) can occupy.
173 float maximumSegmentWidth;
174
175 switch (m_scaling)
176 {
177 case ScalingMode::Absolute:
178 maximumSegmentWidth = computeMaximumSegmentWidthAbsolute();
179 break;
180 case ScalingMode::Relative:
181 maximumSegmentWidth = computeMaximumSegmentWidthRelative();
182 break;
183 case ScalingMode::Dynamic:
184 default:
185 maximumSegmentWidth = computeMaximumSegmentWidthDynamic();
186 break;
187 }
188
189 // Compute the actual widths that the segments occupy and add them to the
190 // previous x coordinates starting at the margin.
191 const float predelayX = margin + predelayPercentage * maximumSegmentWidth;
192 const float attackX = predelayX + attackPercentage * maximumSegmentWidth;
193 const float holdX = attackX + holdPercentage * maximumSegmentWidth;
194 const float decayX = holdX + (decayPercentage * (1 - sustain)) * maximumSegmentWidth;
195 const float releaseX = decayX + releasePercentage * maximumSegmentWidth;
196
197 // Now compute the "full" points including y coordinates
198 const int yTop = 3;
199 const qreal yBase = height() - 3;
200 const int availableHeight = yBase - yTop;
201
202 const QPointF predelayPoint{predelayX, yBase};
203 const QPointF attackPoint{attackX, yTop};
204 const QPointF holdPoint{holdX, yTop};
205 const QPointF decayPoint{decayX, yTop + (1 - sustain) * availableHeight};
206 const QPointF releasePoint{releaseX, yBase};
207
208 // Now that we have all points we can draw the lines
209
210 // Compute the color of the lines based on the amount of the envelope
211 const float absAmount = std::abs(amount);
212 const QColor lineColor{ColorHelper::interpolateInRgb(m_noAmountColor, m_fullAmountColor, absAmount)};
213
214 // Determine the line width so that it scales with the widget
215 // Use the minimum value of the current width and height to compute it.
216 const qreal lineWidth = std::min(width(), height()) / 20.;
217 const QPen linePen{lineColor, lineWidth};
218 p.setPen(linePen);
219
220 QPolygonF linePoly;
221 linePoly << predelayPoint << attackPoint << holdPoint << decayPoint << releasePoint;
222 p.drawPolyline(linePoly);
223
224 // Now draw all marker on top of the lines
225 QPen pen;
226 pen.setWidthF(lineWidth * 0.75);
227 pen.setBrush(m_markerOutlineColor);
228 p.setPen(pen);
229 p.setBrush(m_markerFillColor);
230
231 // Compute the size of the circle we will draw based on the line width
232 const qreal baseRectSize = lineWidth * 3;
233 const QSizeF rectSize{baseRectSize, baseRectSize};
234
235 auto drawMarker = [&](const QPointF& point)
236 {
237 // Create a rectangle that has the given point at its center
238 QRectF bgRect{point + QPointF(-baseRectSize / 2, -baseRectSize / 2), rectSize};
239 p.drawEllipse(bgRect);
240 };
241
242 drawMarker(predelayPoint);
243 drawMarker(attackPoint);
244 drawMarker(holdPoint);
245 drawMarker(decayPoint);
246 drawMarker(releasePoint);
247}
