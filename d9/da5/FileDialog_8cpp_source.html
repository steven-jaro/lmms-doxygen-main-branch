39FileDialog::FileDialog( QWidget *parent, const QString &caption,
40 const QString &directory, const QString &filter ) :
41 QFileDialog( parent, caption, directory, filter )
42{
43#if QT_VERSION > 0x050200
44 setOption( QFileDialog::DontUseCustomDirectoryIcons );
45#endif
46
47 setOption( QFileDialog::DontUseNativeDialog );
48
49#ifdef LMMS_BUILD_LINUX
50 QList<QUrl> urls;
51#else
52 QList<QUrl> urls = sidebarUrls();
53#endif
54
55 QDir desktopDir;
56 desktopDir.setPath(QStandardPaths::writableLocation(QStandardPaths::DesktopLocation));
57 if (desktopDir.exists())
58 {
59 urls << QUrl::fromLocalFile(desktopDir.absolutePath());
60 }
61
62 QDir downloadDir(QDir::homePath() + "/Downloads");
63 if (!downloadDir.exists())
64 {
65 downloadDir.setPath(QStandardPaths::writableLocation(QStandardPaths::DownloadLocation));
66 }
67 if (downloadDir.exists())
68 {
69 urls << QUrl::fromLocalFile(downloadDir.absolutePath());
70 }
71
72 QDir musicDir;
73 musicDir.setPath(QStandardPaths::writableLocation(QStandardPaths::MusicLocation));
74 if (musicDir.exists())
75 {
76 urls << QUrl::fromLocalFile(musicDir.absolutePath());
77 }
78
79 urls << QUrl::fromLocalFile(ConfigManager::inst()->workingDir());
80
81 // Add `/Volumes` directory on OS X systems, this allows the user to browse
82 // external disk drives.
83#ifdef LMMS_BUILD_APPLE
84 QDir volumesDir( QDir("/Volumes") );
85 if ( volumesDir.exists() )
86 urls << QUrl::fromLocalFile( volumesDir.absolutePath() );
87#endif
88
89#ifdef LMMS_BUILD_LINUX
90
91 // FileSystem types : https://www.javatpoint.com/linux-file-system
92 QStringList usableFileSystems = {"ext", "ext2", "ext3", "ext4", "jfs", "reiserfs", "ntfs3", "fuse.sshfs", "fuseblk"};
93
94 for(QStorageInfo storage : QStorageInfo::mountedVolumes())
95 {
96 storage.refresh();
97
98 if (usableFileSystems.contains(QString(storage.fileSystemType()), Qt::CaseInsensitive) && storage.isValid() && storage.isReady())
99 {
100 urls << QUrl::fromLocalFile(storage.rootPath());
101 }
102 }
103#endif
104
105 setSidebarUrls(urls);
106}

109QString FileDialog::getExistingDirectory(QWidget *parent,
110 const QString &caption,
111 const QString &directory,
112 QFileDialog::Options options)
113{
114 FileDialog dialog(parent, caption, directory, QString());
115 dialog.setFileMode(QFileDialog::Directory);
116 dialog.setOptions(dialog.options() | options);
117 if (dialog.exec() == QDialog::Accepted) {
118 return dialog.selectedFiles().value(0);
119 }
120 return QString();
121}

123QString FileDialog::getOpenFileName(QWidget *parent,
124 const QString &caption,
125 const QString &directory,
126 const QString &filter,
127 QString *selectedFilter)
128{
129 FileDialog dialog(parent, caption, directory, filter);
130 if (selectedFilter && !selectedFilter->isEmpty())
131 dialog.selectNameFilter(*selectedFilter);
132 if (dialog.exec() == QDialog::Accepted) {
133 if (selectedFilter)
134 *selectedFilter = dialog.selectedNameFilter();
135 return dialog.selectedFiles().value(0);
136 }
137 return QString();
138}
