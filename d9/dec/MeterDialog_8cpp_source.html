39MeterDialog::MeterDialog( QWidget * _parent, bool _simple ) :
40 QWidget( _parent ),
41 ModelView( nullptr, this )
42{
43 auto vlayout = new QVBoxLayout(this);
44 vlayout->setSpacing( 0 );
45 vlayout->setContentsMargins(0, 0, 0, 0);
46
47 auto num = new QWidget(this);
48 auto num_layout = new QHBoxLayout(num);
49 num_layout->setSpacing( 0 );
50 num_layout->setContentsMargins(0, 0, 0, 0);
51
52
53 m_numerator = new LcdSpinBox( 2, num, tr( "Meter Numerator" ) );
54 m_numerator->setToolTip(tr("Meter numerator"));
55
56 num_layout->addWidget( m_numerator );
57
58 if( !_simple )
59 {
60 auto num_label = new QLabel(tr("Meter Numerator"), num);
61 QFont f = num_label->font();
62 num_layout->addSpacing( 5 );
63 num_layout->addWidget( num_label );
64 }
65 num_layout->addStretch();
66
67 auto den = new QWidget(this);
68 auto den_layout = new QHBoxLayout(den);
69 den_layout->setSpacing( 0 );
70 den_layout->setContentsMargins(0, 0, 0, 0);
71
72 m_denominator = new LcdSpinBox( 2, den, tr( "Meter Denominator" ) );
73 m_denominator->setToolTip(tr("Meter denominator"));
74 if( _simple )
75 {
76 m_denominator->setLabel( tr( "TIME SIG" ) );
77 }
78
79 den_layout->addWidget( m_denominator );
80
81 if( !_simple )
82 {
83 auto den_label = new QLabel(tr("Meter Denominator"), den);
84 QFont f = den_label->font();
85 den_layout->addSpacing( 5 );
86 den_layout->addWidget( den_label );

88 den_layout->addStretch();

90
91 vlayout->addSpacing( _simple ? 1 : 3 );
92 vlayout->addWidget( num );

94 vlayout->addWidget( den );
95 vlayout->addStretch();
96}
