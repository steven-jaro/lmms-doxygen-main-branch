44InstrumentSoundShapingView::InstrumentSoundShapingView(QWidget* parent) :
45 QWidget(parent),
46 ModelView(nullptr, this)
47{
48 QVBoxLayout* mainLayout = new QVBoxLayout(this);
49 mainLayout->setContentsMargins(5, 5, 5, 5);
50
51 m_targetsTabWidget = new TabWidget(tr("TARGET"), this);
52
53 m_volumeView = new EnvelopeAndLfoView(m_targetsTabWidget);
54 m_cutoffView = new EnvelopeAndLfoView(m_targetsTabWidget);
55 m_resonanceView = new EnvelopeAndLfoView(m_targetsTabWidget);
56
57 m_targetsTabWidget->addTab(m_volumeView, tr("VOLUME"), nullptr);
58 m_targetsTabWidget->addTab(m_cutoffView, tr("CUTOFF"), nullptr);
59 m_targetsTabWidget->addTab(m_resonanceView, tr("RESO"), nullptr);
60
61 mainLayout->addWidget(m_targetsTabWidget, 1);
62
63
64 m_filterGroupBox = new GroupBox(tr("FILTER"), this);
65 QHBoxLayout* filterLayout = new QHBoxLayout(m_filterGroupBox);
66 QMargins filterMargins = filterLayout->contentsMargins();
67 filterMargins.setTop(18);
68 filterLayout->setContentsMargins(filterMargins);
69
70 m_filterComboBox = new ComboBox(m_filterGroupBox);
71 filterLayout->addWidget(m_filterComboBox);
72
73 m_filterCutKnob = new Knob(KnobType::Bright26, tr("FREQ"), m_filterGroupBox, Knob::LabelRendering::LegacyFixedFontSize);
74 m_filterCutKnob->setHintText(tr("Cutoff frequency:"), " " + tr("Hz"));
75 filterLayout->addWidget(m_filterCutKnob);
76
77 m_filterResKnob = new Knob(KnobType::Bright26, tr("Q/RESO"), m_filterGroupBox, Knob::LabelRendering::LegacyFixedFontSize);
78 m_filterResKnob->setHintText(tr("Q/Resonance:"), "");
79 filterLayout->addWidget(m_filterResKnob);
80
81 mainLayout->addWidget(m_filterGroupBox);
