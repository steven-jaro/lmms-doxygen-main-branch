165void InstrumentMidiIOView::modelChanged()
166{
167 auto mp = castModel<MidiPort>();
168
169 m_midiInputGroupBox->setModel( &mp->m_readableModel );
170 m_inputChannelSpinBox->setModel( &mp->m_inputChannelModel );
171 m_fixedInputVelocitySpinBox->setModel( &mp->m_fixedInputVelocityModel );
172
173 m_midiOutputGroupBox->setModel( &mp->m_writableModel );
174 m_outputChannelSpinBox->setModel( &mp->m_outputChannelModel );
175 m_fixedOutputVelocitySpinBox->setModel( &mp->m_fixedOutputVelocityModel );
176 m_fixedOutputNoteSpinBox->setModel( &mp->m_fixedOutputNoteModel );
177 m_outputProgramSpinBox->setModel( &mp->m_outputProgramModel );
178
179 m_baseVelocitySpinBox->setModel( &mp->m_baseVelocityModel );
180
181 if( m_rpBtn )
182 {
183 m_rpBtn->setMenu( mp->m_readablePortsMenu );
184 }
185 if( m_wpBtn )
186 {
187 m_wpBtn->setMenu( mp->m_writablePortsMenu );
188 }
189}
