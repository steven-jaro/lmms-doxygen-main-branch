248{
249 using namespace lmms;
250
251 bool coreOnly = false;
252 bool fullscreen = true;
253 bool exitAfterImport = false;
254 bool allowRoot = false;
255 bool renderLoop = false;
256 bool renderTracks = false;
257 QString fileToLoad, fileToImport, renderOut, profilerOutputFile, configFile;
258
259 // first of two command-line parsing stages
260 for (int i = 1; i < argc; ++i)
261 {
262 QString arg = argv[i];
263
264 if (arg == "--help" || arg == "-h")
265 {
266 printHelp();
267 return EXIT_SUCCESS;
268 }
269 else if (arg == "--version" || arg == "-v")
270 {
271 printVersion(argv[0]);
272 return EXIT_SUCCESS;
273 }
274 else if (arg == "render" || arg == "--render" || arg == "-r" )
275 {
276 coreOnly = true;
277 }
278 else if (arg == "rendertracks" || arg == "--rendertracks")
279 {
280 coreOnly = true;
281 renderTracks = true;
282 }
283 else if (arg == "--allowroot")
284 {
285 allowRoot = true;
286 }
287 else if (arg == "--geometry" || arg == "-geometry")
288 {
289 if (arg == "--geometry") { argv[i]++; } // Delete the first "-" so Qt recognize the option
290 // option -geometry is filtered by Qt later,
291 // so we need to check its presence now to
292 // determine, if the application should run in
293 // fullscreen mode (default, no -geometry given).
294 fullscreen = false;
295 }
296 }
297
298#ifdef LMMS_DEBUG_FPE
299 // Enable exceptions for certain floating point results
300 // FE_UNDERFLOW is disabled for the time being
301 feenableexcept( FE_INVALID |
302 FE_DIVBYZERO |
303 FE_OVERFLOW /*|
304 FE_UNDERFLOW*/);
305
306 // Install the trap handler
307 // register signal SIGFPE and signal handler
308 signal(SIGFPE, sigfpeHandler);
309#endif
310 signal(SIGINT, gui::GuiApplication::sigintHandler);
311
312#ifdef LMMS_BUILD_WIN32
313 // Don't touch redirected streams here
314 // GetStdHandle should be called before AttachConsole
315 HANDLE hStdIn = GetStdHandle(STD_INPUT_HANDLE);
316 HANDLE hStdOut = GetStdHandle(STD_OUTPUT_HANDLE);
317 HANDLE hStdErr = GetStdHandle(STD_ERROR_HANDLE);
318 FILE *fIn, *fOut, *fErr;
319 // Enable console output if available
320 if (AttachConsole(ATTACH_PARENT_PROCESS))
321 {
322 if (!hStdIn)
323 {
324 freopen_s(&fIn, "CONIN$", "r", stdin);
325 }
326 if (!hStdOut)
327 {
328 freopen_s(&fOut, "CONOUT$", "w", stdout);
329 }
330 if (!hStdErr)
331 {
332 freopen_s(&fErr, "CONOUT$", "w", stderr);
333 }
334 }
335 // Make Qt's debug message handlers work
336 qInstallMessageHandler(consoleMessageHandler);
337#endif
338
339#if defined(LMMS_HAVE_SYS_PRCTL_H) && defined(PR_SET_CHILD_SUBREAPER)
340 // Set the "child subreaper" attribute so that plugin child processes remain as lmms'
341 // children even when some wrapper process exits, as it may happen with wine
342 if (prctl(PR_SET_CHILD_SUBREAPER, 1))
343 {
344 perror("prctl(PR_SET_CHILD_SUBREAPER)");
345 }
346#endif
347
348 // initialize memory managers
349 NotePlayHandleManager::init();
350
351 // initialize RNG
352 srand( getpid() + time( 0 ) );
353
354 disable_denormals();
355
356#if !defined(LMMS_BUILD_WIN32) && !defined(LMMS_BUILD_HAIKU)
357 if ( ( getuid() == 0 || geteuid() == 0 ) && !allowRoot )
358 {
359 printf( "LMMS cannot be run as root.\nUse \"--allowroot\" to override.\n\n" );
360 return EXIT_FAILURE;
361 }
362#endif
363 QCoreApplication::setAttribute(Qt::AA_EnableHighDpiScaling);
364 QCoreApplication * app = coreOnly ?
365 new QCoreApplication( argc, argv ) :
366 new gui::MainApplication(argc, argv);
367
368 OutputSettings os(44100, 160, OutputSettings::BitDepth::Depth16Bit, OutputSettings::StereoMode::JointStereo);
369 ProjectRenderer::ExportFileFormat eff = ProjectRenderer::ExportFileFormat::Wave;
370
371 // second of two command-line parsing stages
372 for( int i = 1; i < argc; ++i )
373 {
374 QString arg = argv[i];
375
376 if (arg == "upgrade" || arg == "--upgrade" || arg == "-u")
377 {
378 ++i;
379
380 if( i == argc )
381 {
382 return noInputFileError();
383 }
384
385
386 DataFile dataFile( QString::fromLocal8Bit( argv[i] ) );
387
388 if( argc > i+1 ) // output file specified
389 {
390 dataFile.writeFile( QString::fromLocal8Bit( argv[i+1] ) );
391 }
392 else // no output file specified; use stdout
393 {
394 QTextStream ts( stdout );
395 dataFile.write( ts );
396 fflush( stdout );
397 }
398
399 return EXIT_SUCCESS;
400 }
401 else if (arg == "makebundle")
402 {
403 ++i;
404
405 if (i == argc)
406 {
407 return noInputFileError();
408 }
409
410 DataFile dataFile(QString::fromLocal8Bit(argv[i]));
411
412 if (argc > i+1) // Project bundle file name given
413 {
414 printf("Making bundle\n");
415 dataFile.writeFile(QString::fromLocal8Bit(argv[i+1]), true);
416 return EXIT_SUCCESS;
417 }
418 else
419 {
420 return usageError("No project bundle name given");
421 }
422 }
423 else if( arg == "--allowroot" )
424 {
425 // Ignore, processed earlier
426#ifdef LMMS_BUILD_WIN32
427 if( allowRoot )
428 {
429 printf( "\nOption \"--allowroot\" will be ignored on this platform.\n\n" );
430 }
431#endif
432
433 }
434 else if( arg == "dump" || arg == "--dump" || arg == "-d" )
435 {
436 ++i;
437
438 if( i == argc )
439 {
440 return noInputFileError();
441 }
442
443
444 QFile f( QString::fromLocal8Bit( argv[i] ) );
445 f.open( QIODevice::ReadOnly );
446 QString d = qUncompress( f.readAll() );
447 printf( "%s\n", d.toUtf8().constData() );
448
449 return EXIT_SUCCESS;
450 }
451 else if( arg == "compress" || arg == "--compress" )
452 {
453 ++i;
454
455 if( i == argc )
456 {
457 return noInputFileError();
458 }
459
460 QFile f( QString::fromLocal8Bit( argv[i] ) );
461 f.open( QIODevice::ReadOnly );
462 QByteArray d = qCompress( f.readAll() ) ;
463 fwrite( d.constData(), sizeof(char), d.size(), stdout );
464
465 return EXIT_SUCCESS;
466 }
467 else if( arg == "render" || arg == "--render" || arg == "-r" ||
468 arg == "rendertracks" || arg == "--rendertracks" )
469 {
470 ++i;
471
472 if( i == argc )
473 {
474 return noInputFileError();
475 }
476
477
478 fileToLoad = QString::fromLocal8Bit( argv[i] );
479 renderOut = fileToLoad;
480 }
481 else if( arg == "--loop" || arg == "-l" )
482 {
483 renderLoop = true;
484 }
485 else if( arg == "--output" || arg == "-o" )
486 {
487 ++i;
488
489 if( i == argc )
490 {
491 return usageError( "No output file specified" );
492 }
493
494
495 renderOut = QString::fromLocal8Bit( argv[i] );
496 }
497 else if( arg == "--format" || arg == "-f" )
498 {
499 ++i;
500
501 if( i == argc )
502 {
503 return usageError( "No output format specified" );
504 }
505
506
507 const QString ext = QString( argv[i] );
508
509 if( ext == "wav" )
510 {
511 eff = ProjectRenderer::ExportFileFormat::Wave;
512 }
513#ifdef LMMS_HAVE_OGGVORBIS
514 else if( ext == "ogg" )
515 {
516 eff = ProjectRenderer::ExportFileFormat::Ogg;
517 }
518#endif
519#ifdef LMMS_HAVE_MP3LAME
520 else if( ext == "mp3" )
521 {
522 eff = ProjectRenderer::ExportFileFormat::MP3;
523 }
524#endif
525 else if (ext == "flac")
526 {
527 eff = ProjectRenderer::ExportFileFormat::Flac;
528 }
529 else
530 {
531 return usageError( QString( "Invalid output format %1" ).arg( argv[i] ) );
532 }
533 }
534 else if( arg == "--samplerate" || arg == "-s" )
535 {
536 ++i;
537
538 if( i == argc )
539 {
540 return usageError( "No samplerate specified" );
541 }
542
543
544 sample_rate_t sr = QString( argv[i] ).toUInt();
545 if( sr >= 44100 && sr <= 192000 )
546 {
547 os.setSampleRate(sr);
548 }
549 else
550 {
551 return usageError( QString( "Invalid samplerate %1" ).arg( argv[i] ) );
552 }
553 }
554 else if( arg == "--bitrate" || arg == "-b" )
555 {
556 ++i;
557
558 if( i == argc )
559 {
560 return usageError( "No bitrate specified" );
561 }
562
563
564 int br = QString( argv[i] ).toUInt();
565
566 if( br >= 64 && br <= 384 )
567 {
568 os.setBitrate(br);
569 }
570 else
571 {
572 return usageError( QString( "Invalid bitrate %1" ).arg( argv[i] ) );
573 }
574 }
575 else if( arg == "--mode" || arg == "-m" )
576 {
577 ++i;
578
579 if( i == argc )
580 {
581 return usageError( "No stereo mode specified" );
582 }
583
584 QString const mode( argv[i] );
585
586 if( mode == "s" )
587 {
588 os.setStereoMode(OutputSettings::StereoMode::Stereo);
589 }
590 else if( mode == "j" )
591 {
592 os.setStereoMode(OutputSettings::StereoMode::JointStereo);
593 }
594 else if( mode == "m" )
595 {
596 os.setStereoMode(OutputSettings::StereoMode::Mono);
597 }
598 else
599 {
600 return usageError( QString( "Invalid stereo mode %1" ).arg( argv[i] ) );
601 }
602 }
603 else if( arg =="--float" || arg == "-a" )
604 {
605 os.setBitDepth(OutputSettings::BitDepth::Depth32Bit);
606 }
607 else if( arg == "--import" )
608 {
609 ++i;
610
611 if( i == argc )
612 {
613 return usageError( "No file specified for importing" );
614 }
615
616 fileToImport = QString::fromLocal8Bit( argv[i] );
617
618 // exit after import? (only for debugging)
619 if( QString( argv[i + 1] ) == "-e" )
620 {
621 exitAfterImport = true;
622 ++i;
623 }
624 }
625 else if( arg == "--profile" || arg == "-p" )
626 {
627 ++i;
628
629 if( i == argc )
630 {
631 return usageError( "No profile specified" );
632 }
633
634
635 profilerOutputFile = QString::fromLocal8Bit( argv[i] );
636 }
637 else if( arg == "--config" || arg == "-c" )
638 {
639 ++i;
640
641 if( i == argc )
642 {
643 return usageError( "No configuration file specified" );
644 }
645
646 configFile = QString::fromLocal8Bit( argv[i] );
647 }
648 else
649 {
650 if( argv[i][0] == '-' )
651 {
652 return usageError( QString( "Invalid option %1" ).arg( argv[i] ) );
653 }
654 fileToLoad = QString::fromLocal8Bit( argv[i] );
655 }
656 }
657
658 // Test file argument before continuing
659 if( !fileToLoad.isEmpty() )
660 {
661 fileCheck( fileToLoad );
662 }
663 else if( !fileToImport.isEmpty() )
664 {
665 fileCheck( fileToImport );
666 }
667
668 ConfigManager::inst()->loadConfigFile(configFile);
669
670 // Hidden settings
671 MixHelpers::setNaNHandler( ConfigManager::inst()->value( "app",
672 "nanhandler", "1" ).toInt() );
673
674 // set language
675 QString pos = ConfigManager::inst()->value( "app", "language" );
676 if( pos.isEmpty() )
677 {
678 pos = QLocale::system().name().left( 2 );
679 }
680
681 // load actual translation for LMMS
682 loadTranslation( pos );
683
684 // load translation for Qt-widgets/-dialogs
685#ifdef QT_TRANSLATIONS_DIR
686 // load from the original path first
687 loadTranslation(QString("qt_") + pos, QT_TRANSLATIONS_DIR);
688#endif
689 // override it with bundled/custom one, if exists
690 loadTranslation(QString("qt_") + pos, ConfigManager::inst()->localeDir());
691
692#if _POSIX_C_SOURCE >= 1 || _XOPEN_SOURCE || _POSIX_SOURCE
693 struct sigaction sa;
694 sa.sa_handler = SIG_IGN;
695 sa.sa_flags = SA_SIGINFO;
696 if ( sigemptyset( &sa.sa_mask ) )
697 {
698 fprintf( stderr, "Signal initialization failed.\n" );
699 }
700 if ( sigaction( SIGPIPE, &sa, nullptr ) )
701 {
702 fprintf( stderr, "Signal initialization failed.\n" );
703 }
704#endif
705
706 bool destroyEngine = false;
707
708 // if we have an output file for rendering, just render the song
709 // without starting the GUI
710 if( !renderOut.isEmpty() )
711 {
712 Engine::init( true );
713 destroyEngine = true;
714
715 printf( "Loading project...\n" );
716 Engine::getSong()->loadProject( fileToLoad );
717 if( Engine::getSong()->isEmpty() )
718 {
719 printf("The project %s is empty, aborting!\n", fileToLoad.toUtf8().constData() );
720 exit( EXIT_FAILURE );
721 }
722 printf( "Done\n" );
723
724 Engine::getSong()->setExportLoop( renderLoop );
725
726 // when rendering multiple tracks, renderOut is a directory
727 // otherwise, it is a file, so we need to append the file extension
728 if ( !renderTracks )
729 {
730 renderOut = baseName( renderOut ) +
731 ProjectRenderer::getFileExtensionFromFormat(eff);
732 }
733
734 // create renderer
735 auto r = new RenderManager(os, eff, renderOut);
736 QCoreApplication::instance()->connect( r,
737 SIGNAL(finished()), SLOT(quit()));
738
739 // timer for progress-updates
740 auto t = new QTimer(r);
741 r->connect( t, SIGNAL(timeout()),
742 SLOT(updateConsoleProgress()));
743 t->start( 200 );
744
745 if( profilerOutputFile.isEmpty() == false )
746 {
747 Engine::audioEngine()->profiler().setOutputFile( profilerOutputFile );
748 }
749
750 // start now!
751 if ( renderTracks )
752 {
753 r->renderTracks();
754 }
755 else
756 {
757 r->renderProject();
758 }
759 }
760 else // otherwise, start the GUI
761 {
762 using namespace lmms::gui;
763
764 new GuiApplication();
765
766 // re-intialize RNG - shared libraries might have srand() or
767 // srandom() calls in their init procedure
768 srand( getpid() + time( 0 ) );
769
770 // recover a file?
771 QString recoveryFile = ConfigManager::inst()->recoveryFile();
772
773 bool recoveryFilePresent = QFileInfo( recoveryFile ).exists() &&
774 QFileInfo( recoveryFile ).isFile();
775 bool autoSaveEnabled =
776 ConfigManager::inst()->value( "ui", "enableautosave" ).toInt();
777 if( recoveryFilePresent )
778 {
779 QMessageBox mb;
780 mb.setWindowTitle( MainWindow::tr( "Project recovery" ) );
781 mb.setText( QString(
782 "<html>"
783 "<p style=\"margin-left:6\">%1</p>"
784 "<table cellpadding=\"3\">"
785 " <tr>"
786 " <td><b>%2</b></td>"
787 " <td>%3</td>"
788 " </tr>"
789 " <tr>"
790 " <td><b>%4</b></td>"
791 " <td>%5</td>"
792 " </tr>"
793 "</table>"
794 "</html>" ).arg(
795 MainWindow::tr( "There is a recovery file present. "
796 "It looks like the last session did not end "
797 "properly or another instance of LMMS is "
798 "already running. Do you want to recover the "
799 "project of this session?" ),
800 MainWindow::tr( "Recover" ),
801 MainWindow::tr( "Recover the file. Please don't run "
802 "multiple instances of LMMS when you do this." ),
803 MainWindow::tr( "Discard" ),
804 MainWindow::tr( "Launch a default session and delete "
805 "the restored files. This is not reversible." )
806 ) );
807
808 mb.setIcon( QMessageBox::Warning );
809 mb.setWindowIcon( embed::getIconPixmap( "icon_small" ) );
810 mb.setWindowFlags( Qt::WindowCloseButtonHint );
811
812 // setting all buttons to the same roles allows us
813 // to have a custom layout
814 auto discard = mb.addButton(MainWindow::tr("Discard"), QMessageBox::AcceptRole);
815 auto recover = mb.addButton(MainWindow::tr("Recover"), QMessageBox::AcceptRole);
816
817 // have a hidden exit button

819 exit->setVisible(false);

822 recover->setIcon( embed::getIconPixmap( "recover" ) );

826 mb.setEscapeButton( exit );
827

831 getGUI()->mainWindow()->sessionCleanup();
832 }
833 else if( mb.clickedButton() == recover ) // Recover
834 {
835 fileToLoad = recoveryFile;
836 getGUI()->mainWindow()->setSession( MainWindow::SessionState::Recover );
837 }
838 else // Exit
839 {
840 return EXIT_SUCCESS;
841 }
842 }
843
844 // first show the Main Window and then try to load given file
845
846 // [Settel] workaround: showMaximized() doesn't work with
847 // FVWM2 unless the window is already visible -> show() first
848 getGUI()->mainWindow()->show();
849 if( fullscreen )

851 getGUI()->mainWindow()->showMaximized();
852 }
853
854 // Handle macOS-style FileOpen QEvents
855 QString queuedFile = static_cast<MainApplication *>( app )->queuedFile();
856 if ( !queuedFile.isEmpty() ) {
857 fileToLoad = queuedFile;
858 }
859
860 if( !fileToLoad.isEmpty() )
861 {

863 {
864 Engine::getSong()->createNewProjectFromTemplate( fileToLoad );
865 }
866 else
867 {
868 Engine::getSong()->loadProject( fileToLoad );
869 }
870 }
871 else if( !fileToImport.isEmpty() )
872 {
873 ImportFilter::import( fileToImport, Engine::getSong() );
874 if( exitAfterImport )
875 {
876 return EXIT_SUCCESS;
877 }
878 }
879 // If enabled, open last project if there is one. Else, create
880 // a new one.
881 else if( ConfigManager::inst()->
882 value( "app", "openlastproject" ).toInt() &&
883 !ConfigManager::inst()->
884 recentlyOpenedProjects().isEmpty() &&
885 !recoveryFilePresent )
886 {
887 QString f = ConfigManager::inst()->
888 recentlyOpenedProjects().first();
889 QFileInfo recentFile( f );
890
891 if ( recentFile.exists() &&
892 recentFile.suffix().toLower() != "mpt" )
893 {
894 Engine::getSong()->loadProject( f );
895 }
896 else
897 {
898 Engine::getSong()->createNewProject();
899 }
900 }
901 else
902 {
903 Engine::getSong()->createNewProject();
904 }
905
906 // Finally we start the auto save timer and also trigger the
907 // autosave one time as recover.mmp is a signal to possible other
908 // instances of LMMS.
909 if( autoSaveEnabled )
910 {
911 gui::getGUI()->mainWindow()->autoSaveTimerReset();
912 }
913 }
914
915 const int ret = app->exec();
916 delete app;
917
918 if( destroyEngine )
919 {
920 Engine::destroy();
921 }
922
923 // ProjectRenderer::updateConsoleProgress() doesn't return line after render
924 if( coreOnly )
925 {
926 printf( "\n" );
927 }
928
929#ifdef LMMS_BUILD_WIN32
930 // Cleanup console
931 HWND hConsole = GetConsoleWindow();

933 {
934 SendMessage(hConsole, WM_CHAR, (WPARAM)VK_RETURN, (LPARAM)0);
935 FreeConsole();
936 }
937#endif
938
939
940 NotePlayHandleManager::free();
941
942 return ret;
