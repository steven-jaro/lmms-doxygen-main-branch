104void LedCheckBox::onTextUpdated()
105{
106 QFontMetrics const fm = fontMetrics();
107
108 int const width = m_ledOffPixmap.width() + 5 + horizontalAdvance(fm, text());
109 int const height = m_legacyMode ? m_ledOffPixmap.height() : qMax(m_ledOffPixmap.height(), fm.height());
110
111 setFixedSize(width, height);
112}

114void LedCheckBox::paintLegacy(QPaintEvent * pe)
115{
116 QPainter p( this );
117 p.setFont(adjustedToPixelSize(font(), DEFAULT_FONT_SIZE));
118
119 p.drawPixmap(0, 0, model()->value() ? m_ledOnPixmap : m_ledOffPixmap);
120
121 p.setPen( QColor( 64, 64, 64 ) );
122 p.drawText(m_ledOffPixmap.width() + 4, 11, text());
123 p.setPen( QColor( 255, 255, 255 ) );
124 p.drawText(m_ledOffPixmap.width() + 3, 10, text());
125}
126
127void LedCheckBox::paintNonLegacy(QPaintEvent * pe)
128{
129 QPainter p(this);
130
131 auto drawnPixmap = model()->value() ? m_ledOnPixmap : m_ledOffPixmap;
132
133 p.drawPixmap(0, rect().height() / 2 - drawnPixmap.height() / 2, drawnPixmap);
134
135 QRect r = rect();
136 r -= QMargins(m_ledOffPixmap.width() + 5, 0, 0, 0);
137 p.drawText(r, text());
138}
