49JackWinMMEInputPort::JackWinMMEInputPort(const char *alias_name,
50 const char *client_name,
51 const char *driver_name, UINT index,
52 size_t max_bytes, size_t max_messages)
53{
54 thread_queue = new JackMidiAsyncQueue(max_bytes, max_messages);
55 std::unique_ptr<JackMidiAsyncQueue> thread_queue_ptr(thread_queue);
56 write_queue = new JackMidiBufferWriteQueue();
57 std::unique_ptr<JackMidiBufferWriteQueue> write_queue_ptr(write_queue);
58 sysex_buffer = new jack_midi_data_t[max_bytes];
59 char error_message[MAXERRORLENGTH];
60 MMRESULT result = midiInOpen(&handle, index,
61 (DWORD_PTR) HandleMidiInputEvent,
62 (DWORD_PTR)this,
63 CALLBACK_FUNCTION | MIDI_IO_STATUS);
64 if (result != MMSYSERR_NOERROR) {
65 GetInErrorString(result, error_message);
66 goto delete_sysex_buffer;
67 }
68 sysex_header.dwBufferLength = max_bytes;
69 sysex_header.dwFlags = 0;
70 sysex_header.lpData = (LPSTR)sysex_buffer;
71 result = midiInPrepareHeader(handle, &sysex_header, sizeof(MIDIHDR));
72 if (result != MMSYSERR_NOERROR) {
73 GetInErrorString(result, error_message);
74 goto close_handle;
75 }
76 result = midiInAddBuffer(handle, &sysex_header, sizeof(MIDIHDR));
77 if (result != MMSYSERR_NOERROR) {
78 GetInErrorString(result, error_message);
79 goto unprepare_header;
80 }
81
82 MIDIINCAPS capabilities;
83 char *name_tmp;
84 result = midiInGetDevCaps(index, &capabilities, sizeof(capabilities));
85 if (result != MMSYSERR_NOERROR) {
86 WriteInError("JackWinMMEInputPort [constructor]", "midiInGetDevCaps",
87 result);
88 name_tmp = (char*) driver_name;
89 } else {
90 name_tmp = capabilities.szPname;
91 }
92
93 snprintf(alias, sizeof(alias) - 1, "%s:%s:in%d", alias_name, name_tmp,
94 index + 1);
95 snprintf(name, sizeof(name) - 1, "%s:capture_%d", client_name, index + 1);
96 strncpy(device_name, name_tmp, sizeof(device_name) - 1);
97 jack_event = 0;
98 started = false;
99 write_queue_ptr.release();
100 thread_queue_ptr.release();
101 return;
102
103 unprepare_header:
104 result = midiInUnprepareHeader(handle, &sysex_header, sizeof(MIDIHDR));
105 if (result != MMSYSERR_NOERROR) {
106 WriteInError("JackWinMMEInputPort [constructor]",
107 "midiInUnprepareHeader", result);
108 }
109 close_handle:
110 result = midiInClose(handle);
111 if (result != MMSYSERR_NOERROR) {
112 WriteInError("JackWinMMEInputPort [constructor]", "midiInClose",
113 result);
114 }
115 delete_sysex_buffer:
116 delete[] sysex_buffer;
117 throw std::runtime_error(error_message);
118}

140void
141JackWinMMEInputPort::EnqueueMessage(DWORD timestamp, size_t length,
142 jack_midi_data_t *data)
143{
144 jack_nframes_t frame =
145 GetFramesFromTime(start_time + (((jack_time_t) timestamp) * 1000));
146
147 // Debugging code
148 jack_time_t current_time = GetMicroSeconds();
149 jack_log("JackWinMMEInputPort::EnqueueMessage - enqueueing event at %f "
150 "(frame: %d) with start offset '%d' scheduled for frame '%d'",
151 ((double) current_time) / 1000.0, GetFramesFromTime(current_time),
152 timestamp, frame);
153 // End debugging code
154
155 switch (thread_queue->EnqueueEvent(frame, length, data)) {
156 case JackMidiWriteQueue::BUFFER_FULL:
157 jack_error("JackWinMMEInputPort::EnqueueMessage - The thread queue "
158 "cannot currently accept a %d-byte event. Dropping event.",
159 length);
160 break;
161 case JackMidiWriteQueue::BUFFER_TOO_SMALL:
162 jack_error("JackWinMMEInputPort::EnqueueMessage - The thread queue "
163 "buffer is too small to enqueue a %d-byte event. Dropping "
164 "event.", length);
165 break;
166 default:
167 ;
168 }
169}

180void
181JackWinMMEInputPort::ProcessJack(JackMidiBuffer *port_buffer,
182 jack_nframes_t frames)
183{
184 write_queue->ResetMidiBuffer(port_buffer, frames);
185 if (! jack_event) {
186 jack_event = thread_queue->DequeueEvent();
187 }
188 for (; jack_event; jack_event = thread_queue->DequeueEvent()) {
189 switch (write_queue->EnqueueEvent(jack_event, frames)) {
190 case JackMidiWriteQueue::BUFFER_TOO_SMALL:
191 jack_error("JackWinMMEMidiInputPort::Process - The buffer write "
192 "queue couldn't enqueue a %d-byte event. Dropping "
193 "event.", jack_event->size);
194 // Fallthrough on purpose
195 case JackMidiWriteQueue::OK:
196 continue;
197 default:
198 break;
199 }
200 break;
201 }
202}

204void
205JackWinMMEInputPort::ProcessWinMME(UINT message, DWORD_PTR param1, DWORD_PTR param2)
206{
207 set_threaded_log_function();
208 switch (message) {
209 case MIM_CLOSE:
210 jack_log("JackWinMMEInputPort::ProcessWinMME - MIDI device closed.");
211 break;
212 case MIM_MOREDATA:
213 jack_log("JackWinMMEInputPort::ProcessWinMME - The MIDI input device "
214 "driver thinks that JACK is not processing messages fast "
215 "enough.");
216 // Fallthrough on purpose.
217 case MIM_DATA: {
218 jack_midi_data_t message_buffer[3];
219 jack_midi_data_t status = param1 & 0xff;
220 int length = GetMessageLength(status);
221
222 switch (length) {
223 case 3:
224 message_buffer[2] = (param1 >> 16) & 0xff;
225 // Fallthrough on purpose.
226 case 2:
227 message_buffer[1] = (param1 >> 8) & 0xff;
228 // Fallthrough on purpose.
229 case 1:
230 message_buffer[0] = status;
231 break;
232 case 0:
233 jack_error("JackWinMMEInputPort::ProcessWinMME - **BUG** MIDI "
234 "input driver sent an MIM_DATA message with a sysex "
235 "status byte.");
236 return;
237 case -1:
238 jack_error("JackWinMMEInputPort::ProcessWinMME - **BUG** MIDI "
239 "input driver sent an MIM_DATA message with an invalid "
240 "status byte.");
241 return;
242 }
243 EnqueueMessage(param2, (size_t) length, message_buffer);
244 break;
245 }
246 case MIM_LONGDATA: {
247 LPMIDIHDR header = (LPMIDIHDR) param1;
248 size_t byte_count = header->dwBytesRecorded;
249 if (! byte_count) {
250 jack_log("JackWinMMEInputPort::ProcessWinMME - WinMME driver has "
251 "returned sysex header to us with no bytes. The JACK "
252 "driver is probably being stopped.");
253 break;
254 }
255 jack_midi_data_t *data = (jack_midi_data_t *) header->lpData;
256 if ((data[0] != 0xf0) || (data[byte_count - 1] != 0xf7)) {
257 jack_error("JackWinMMEInputPort::ProcessWinMME - Discarding "
258 "%d-byte sysex chunk.", byte_count);
259 } else {
260 EnqueueMessage(param2, byte_count, data);
261 }
262 // Is this realtime-safe? This function isn't run in the JACK thread,
263 // but we still want it to perform as quickly as possible. Even if
264 // this isn't realtime safe, it may not be avoidable.
265 MMRESULT result = midiInAddBuffer(handle, &sysex_header,
266 sizeof(MIDIHDR));
267 if (result != MMSYSERR_NOERROR) {
268 WriteInError("JackWinMMEInputPort::ProcessWinMME",
269 "midiInAddBuffer", result);
270 }
271 break;
272 }
273 case MIM_LONGERROR:
274 jack_error("JackWinMMEInputPort::ProcessWinMME - Invalid or "
275 "incomplete sysex message received.");
276 break;
277 case MIM_OPEN:
278 jack_log("JackWinMMEInputPort::ProcessWinMME - MIDI device opened.");
279 }
280}

282bool
283JackWinMMEInputPort::Start()
284{
285 if (! started) {
286 start_time = GetMicroSeconds();
287 MMRESULT result = midiInStart(handle);
288 started = result == MMSYSERR_NOERROR;
289 if (! started) {
290 WriteInError("JackWinMMEInputPort::Start", "midiInStart", result);
291 }
292 }
293 return started;
294}

304 }
