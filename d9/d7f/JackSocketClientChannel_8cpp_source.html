42int JackSocketClientChannel::Open(const char* server_name, const char* name, jack_uuid_t uuid, char* name_res, JackClient* client, jack_options_t options, jack_status_t* status)
43{
44 int result = 0;
45 jack_log("JackSocketClientChannel::Open name = %s", name);
46
47 // Before any server/client call
48 fClient = client;
49
50 if (fRequest->Connect(jack_server_dir, server_name, 0) < 0) {
51 jack_error("Cannot connect to server socket");
52 goto error;
53 }
54
55 // OK so server is there...
56 JackGlobals::fServerRunning = true;
57
58 // Check name in server
59 ClientCheck(name, uuid, name_res, JACK_PROTOCOL_VERSION, (int)options, (int*)status, &result, true);
60 if (result < 0) {
61 int status1 = *status;
62 if (status1 & JackVersionError) {
63 jack_error("JACK protocol mismatch %d", JACK_PROTOCOL_VERSION);
64 } else {
65 jack_error("Client name = %s conflits with another running client", name);
66 }
67 goto error;
68 }
69
70 if (fNotificationListenSocket.Bind(jack_client_dir, name_res, 0) < 0) {
71 jack_error("Cannot bind socket");
72 goto error;
73 }
74
75 return 0;
76
77error:
78 fRequest->Close();
79 fNotificationListenSocket.Close();
80 return -1;
81}

112bool JackSocketClientChannel::Init()
113{
114 jack_log("JackSocketClientChannel::Init");
115 fNotificationSocket = fNotificationListenSocket.Accept();
116
117 // No more needed
118 fNotificationListenSocket.Close();
119
120 // Setup context
121 if (!jack_tls_set(JackGlobals::fNotificationThread, this)) {
122 jack_error("Failed to set thread notification key");
123 }
124
125 if (!fNotificationSocket) {
126 jack_error("JackSocketClientChannel: cannot establish notification socket");
127 return false;
128 } else {
129 return true;
130 }
131}

133bool JackSocketClientChannel::Execute()
134{
135 JackClientNotification event;
136 JackResult res;
137
138 int err = event.Read(fNotificationSocket);
139 if (err < 0) {
140 // aborted reading due to shutdown
141 if (err != JACK_REQUEST_ERR_ABORTED) {
142 jack_error("JackSocketClientChannel read fail");
143 }
144 goto error;
145 }
146
147 res.fResult = fClient->ClientNotify(event.fRefNum, event.fName, event.fNotify, event.fSync, event.fMessage, event.fValue1, event.fValue2);
148
149 if (event.fSync) {
150 if (res.Write(fNotificationSocket) < 0) {
151 jack_error("JackSocketClientChannel write fail");
152 goto error;
153 }
154 }
155 return true;
156
157error:
158 fNotificationSocket->Close();
159 fClient->ShutDown(jack_status_t(JackFailure | JackServerError), JACK_SERVER_FAILURE);
160 return false;
161}
