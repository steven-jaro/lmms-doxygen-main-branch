45MidiCCRackView::MidiCCRackView(InstrumentTrack * track) :
46 QWidget(),
47 m_track(track)
48{
49 setWindowIcon(embed::getIconPixmap("midi_cc_rack"));
50 setWindowTitle(tr("MIDI CC Rack - %1").arg(m_track->name()));
51
52 QMdiSubWindow * subWin = getGUI()->mainWindow()->addWindowedWidget(this);
53
54 // Remove maximize button
55 Qt::WindowFlags flags = subWin->windowFlags();
56 flags &= ~Qt::WindowMaximizeButtonHint;
57 subWin->setWindowFlags(flags);
58
59 // Adjust window attributes, sizing and position
60 subWin->setAttribute(Qt::WA_DeleteOnClose, false);
61 subWin->resize(350, 300);
62 subWin->setFixedWidth(350);
63 subWin->setMinimumHeight(300);
64 subWin->hide();
65
66 // Main window layout
67 auto mainLayout = new QVBoxLayout(this);
68
69 // Knobs GroupBox - Here we have the MIDI CC controller knobs for the selected track
70 m_midiCCGroupBox = new GroupBox(tr("MIDI CC Knobs:"));
71
72 // Layout to keep scrollable area under the GroupBox header
73 auto knobsGroupBoxLayout = new QVBoxLayout();
74 knobsGroupBoxLayout->setContentsMargins(5, 16, 5, 5);
75
76 m_midiCCGroupBox->setLayout(knobsGroupBoxLayout);
77
78 // Scrollable area + widget + its layout that will have all the knobs
79 auto knobsScrollArea = new QScrollArea();
80 auto knobsArea = new QWidget();
81 auto knobsAreaLayout = new QGridLayout();
82 knobsAreaLayout->setVerticalSpacing(10);
83
84 knobsArea->setLayout(knobsAreaLayout);
85 knobsScrollArea->setWidget(knobsArea);
86 knobsScrollArea->setWidgetResizable(true);
87
88 knobsGroupBoxLayout->addWidget(knobsScrollArea);
89
90 // Adds the controller knobs and sets their models
91 for (int i = 0; i < MidiControllerCount; ++i)
92 {
93 auto knob = new Knob(KnobType::Bright26, tr("CC %1").arg(i), this);
94 knob->setModel(m_track->m_midiCCModel[i].get());
95 knobsAreaLayout->addWidget(knob, i/4, i%4, Qt::AlignHCenter);
96
97 // TODO It seems that this is not really used/needed?
98 m_controllerKnob[i] = knob;
99 }
100
101 // Set all the models
102 // Set the LED button to enable/disable the track midi cc
103 m_midiCCGroupBox->setModel(m_track->m_midiCCEnable.get());
104
105 // Connection to update the name of the track on the label
106 connect(m_track, SIGNAL(nameChanged()),
107 this, SLOT(renameWindow()));
108
109 // Adding everything to the main layout
110 mainLayout->addWidget(m_midiCCGroupBox);
111}
