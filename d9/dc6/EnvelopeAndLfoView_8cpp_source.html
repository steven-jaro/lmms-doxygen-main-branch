239void EnvelopeAndLfoView::dropEvent( QDropEvent * _de )
240{
241 QString type = StringPairDrag::decodeKey( _de );
242 QString value = StringPairDrag::decodeValue( _de );
243 if( type == "samplefile" )
244 {
245 m_params->m_userWave = SampleLoader::createBufferFromFile(value);
246 m_userLfoBtn->model()->setValue( true );
247 m_params->m_lfoWaveModel.setValue(static_cast<int>(EnvelopeAndLfoParameters::LfoShape::UserDefinedWave));
248 _de->accept();
249 update();
250 }
251 else if( type == QString( "clip_%1" ).arg( static_cast<int>(Track::Type::Sample) ) )
252 {
253 DataFile dataFile( value.toUtf8() );
254 auto file = dataFile.content().
255 firstChildElement().firstChildElement().
256 firstChildElement().attribute("src");
257 m_params->m_userWave = SampleLoader::createBufferFromFile(file);
258 m_userLfoBtn->model()->setValue( true );
259 m_params->m_lfoWaveModel.setValue(static_cast<int>(EnvelopeAndLfoParameters::LfoShape::UserDefinedWave));
260 _de->accept();
261 update();
262 }
263}

268void EnvelopeAndLfoView::lfoUserWaveChanged()
269{
270 if( static_cast<EnvelopeAndLfoParameters::LfoShape>(m_params->m_lfoWaveModel.value()) ==
271 EnvelopeAndLfoParameters::LfoShape::UserDefinedWave )
272 {
273 if (m_params->m_userWave->size() <= 1)
274 {
275 TextFloat::displayMessage( tr( "Hint" ),
276 tr( "Drag and drop a sample into this window." ),
277 embed::getIconPixmap( "hint" ), 3000 );
278 }
279 }
