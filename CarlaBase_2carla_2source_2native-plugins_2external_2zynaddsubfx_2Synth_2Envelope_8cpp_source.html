20Envelope::Envelope(EnvelopeParams &pars, float basefreq, float bufferdt,
21 WatchManager *m, const char *watch_prefix)
22 :watchOut(m, watch_prefix, "out")
23{
24 envpoints = pars.Penvpoints;
25 if(envpoints > MAX_ENVELOPE_POINTS)
26 envpoints = MAX_ENVELOPE_POINTS;
27 envsustain = (pars.Penvsustain == 0) ? -1 : pars.Penvsustain;
28 forcedrelease = pars.Pforcedrelease;
29 envstretch = powf(440.0f / basefreq, pars.Penvstretch / 64.0f);
30 linearenvelope = pars.Plinearenvelope;
31
32 if(!pars.Pfreemode)
33 pars.converttofree();
34
35 int mode = pars.Envmode;
36
37 //for amplitude envelopes
38 if((mode == 1) && !linearenvelope)
39 mode = 2; //change to log envelope
40 if((mode == 2) && linearenvelope)
41 mode = 1; //change to linear
42
43 for(int i = 0; i < MAX_ENVELOPE_POINTS; ++i) {
44 const float tmp = pars.getdt(i) / 1000.0f * envstretch;
45 if(tmp > bufferdt)
46 envdt[i] = bufferdt / tmp;
47 else
48 envdt[i] = 2.0f; //any value larger than 1
49
50 switch(mode) {
51 case 2:
52 envval[i] = (1.0f - pars.Penvval[i] / 127.0f) * -40;
53 break;
54 case 3:
55 envval[i] =
56 (powf(2, 6.0f
57 * fabs(pars.Penvval[i]
58 - 64.0f) / 64.0f) - 1.0f) * 100.0f;
59 if(pars.Penvval[i] < 64)
60 envval[i] = -envval[i];
61 break;
62 case 4:
63 envval[i] = (pars.Penvval[i] - 64.0f) / 64.0f * 6.0f; //6 octaves (filtru)
64 break;
65 case 5:
66 envval[i] = (pars.Penvval[i] - 64.0f) / 64.0f * 10;
67 break;
68 default:
69 envval[i] = pars.Penvval[i] / 127.0f;
70 }
71 }
72
73 envdt[0] = 1.0f;
74
75 currentpoint = 1; //the envelope starts from 1
76 keyreleased = false;
77 t = 0.0f;
78 envfinish = false;
79 inct = envdt[1];
80 envoutval = 0.0f;
81}

104/*
105 * Envelope Output
106 */
107float Envelope::envout(bool doWatch)
108{
109 float out;
110
111 if(envfinish) { //if the envelope is finished
112 envoutval = envval[envpoints - 1];
113 if(doWatch) {
114 float pos[2] = {(float)envpoints - 1, envoutval};
115 watchOut(pos, 2);
116 }
117 return envoutval;
118 }
119 if((currentpoint == envsustain + 1) && !keyreleased) { //if it is sustaining now
120 envoutval = envval[envsustain];
121 if(doWatch) {
122 float pos[2] = {(float)envsustain, envoutval};
123 watchOut(pos, 2);
124 }
125 return envoutval;
126 }
127
128 if(keyreleased && forcedrelease) { //do the forced release
129 int tmp = (envsustain < 0) ? (envpoints - 1) : (envsustain + 1); //if there is no sustain point, use the last point for release
130
131 if(envdt[tmp] < 0.00000001f)
132 out = envval[tmp];
133 else
134 out = envoutval + (envval[tmp] - envoutval) * t;
135 t += envdt[tmp] * envstretch;
136
137 if(t >= 1.0f) {
138 currentpoint = envsustain + 2;
139 forcedrelease = 0;
140 t = 0.0f;
141 inct = envdt[currentpoint];
142 if((currentpoint >= envpoints) || (envsustain < 0))
143 envfinish = true;
144 }
145
146 if(doWatch) {
147 float pos[2] = {(float)tmp + t, envoutval};
148 watchOut(pos, 2);
149 }
150
151 return out;
152 }
153 if(inct >= 1.0f)
154 out = envval[currentpoint];
155 else
156 out = envval[currentpoint - 1]
157 + (envval[currentpoint] - envval[currentpoint - 1]) * t;
158
159 t += inct;
160 if(t >= 1.0f) {
161 if(currentpoint >= envpoints - 1)
162 envfinish = true;
163 else
164 currentpoint++;
165 t = 0.0f;
166 inct = envdt[currentpoint];
167 }
168
169 envoutval = out;
170
171 if(doWatch) {
172 float pos[2] = {(float)currentpoint + t, envoutval};
173 watchOut(pos, 2);
174 }
175 return out;
176}

178/*
179 * Envelope Output (dB)
180 */
181float Envelope::envout_dB()
182{
183 float out;
184 if(linearenvelope)
185 return envout(true);
186
187 if((currentpoint == 1) && (!keyreleased || !forcedrelease)) { //first point is always lineary interpolated
188 float v1 = EnvelopeParams::env_dB2rap(envval[0]);
189 float v2 = EnvelopeParams::env_dB2rap(envval[1]);
190 out = v1 + (v2 - v1) * t;
191
192 t += inct;
193 if(t >= 1.0f) {
194 t = 0.0f;
195 inct = envdt[2];
196 currentpoint++;
197 out = v2;
198 }
199
200 if(out > 0.001f)
201 envoutval = EnvelopeParams::env_rap2dB(out);
202 else
203 envoutval = MIN_ENVELOPE_DB;
204 } else
205 out = EnvelopeParams::env_dB2rap(envout(false));
206
207 float pos[2] = {(float)currentpoint + t, out};
208 watchOut(pos, 2);
209
210 return out;
211}
