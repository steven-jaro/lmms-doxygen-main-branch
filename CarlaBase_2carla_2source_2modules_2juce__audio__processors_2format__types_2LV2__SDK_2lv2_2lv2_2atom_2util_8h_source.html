166static inline LV2_Atom_Event*
167lv2_atom_sequence_append_event(LV2_Atom_Sequence* seq,
168 uint32_t capacity,
169 const LV2_Atom_Event* event)
170{
171 const uint32_t total_size = (uint32_t)sizeof(*event) + event->body.size;
172 if (capacity - seq->atom.size < total_size) {
173 return NULL;
174 }
175
176 LV2_Atom_Event* e = lv2_atom_sequence_end(&seq->body, seq->atom.size);
177 memcpy(e, event, total_size);
178
179 seq->atom.size += lv2_atom_pad_size(total_size);
180
181 return e;
182}

337static inline int
338lv2_atom_object_query(const LV2_Atom_Object* object,
339 LV2_Atom_Object_Query* query)
340{
341 int matches = 0;
342 int n_queries = 0;
343
344 /* Count number of query keys so we can short-circuit when done */
345 for (LV2_Atom_Object_Query* q = query; q->key; ++q) {
346 ++n_queries;
347 }
348
349 LV2_ATOM_OBJECT_FOREACH (object, prop) {
350 for (LV2_Atom_Object_Query* q = query; q->key; ++q) {
351 if (q->key == prop->key && !*q->value) {
352 *q->value = &prop->value;
353 if (++matches == n_queries) {
354 return matches;
355 }
356 break;
357 }
358 }
359 }
360 return matches;
361}
