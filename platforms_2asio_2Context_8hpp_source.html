54 template <typename ExceptionHandler>
55 explicit Context(ExceptionHandler exceptHandler)
56 : mpService(new ::asio::io_service())
57 , mpWork(new ::asio::io_service::work(*mpService))
58 {
59 mThread =
60 std::thread{[](::asio::io_service& service, ExceptionHandler handler) {
61 for (;;)
62 {
63 try
64 {
65 service.run();
66 break;
67 }
68 catch (const typename ExceptionHandler::Exception& exception)
69 {
70 handler(exception);
71 }
72 }
73 },
74 std::ref(*mpService), std::move(exceptHandler)};
75 }

97 template <std::size_t BufferSize>
98 Socket<BufferSize> openUnicastSocket(const ::asio::ip::address_v4& addr)
99 {
100 auto socket = Socket<BufferSize>{*mpService};
101 socket.mpImpl->mSocket.set_option(
102 ::asio::ip::multicast::enable_loopback(addr.is_loopback()));
103 socket.mpImpl->mSocket.set_option(::asio::ip::multicast::outbound_interface(addr));
104 socket.mpImpl->mSocket.bind(::asio::ip::udp::endpoint{addr, 0});
105 return socket;
106 }
107
108 template <std::size_t BufferSize>
109 Socket<BufferSize> openMulticastSocket(const ::asio::ip::address_v4& addr)
110 {
111 auto socket = Socket<BufferSize>{*mpService};
112 socket.mpImpl->mSocket.set_option(::asio::ip::udp::socket::reuse_address(true));
113 socket.mpImpl->mSocket.set_option(
114 ::asio::socket_base::broadcast(!addr.is_loopback()));
115 socket.mpImpl->mSocket.set_option(
116 ::asio::ip::multicast::enable_loopback(addr.is_loopback()));
117 socket.mpImpl->mSocket.set_option(::asio::ip::multicast::outbound_interface(addr));
118 socket.mpImpl->mSocket.bind({::asio::ip::address::from_string("0.0.0.0"),
119 discovery::multicastEndpoint().port()});
120 socket.mpImpl->mSocket.set_option(::asio::ip::multicast::join_group(
121 discovery::multicastEndpoint().address().to_v4(), addr));
122 return socket;
123 }
