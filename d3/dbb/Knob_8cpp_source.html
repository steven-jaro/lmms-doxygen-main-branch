76void Knob::initUi( const QString & _name )
77{
78 onKnobNumUpdated();
79 setTotalAngle( 270.0f );
80 setInnerRadius( 1.0f );
81 setOuterRadius( 10.0f );
82
83 // This is a workaround to enable style sheets for knobs which are not styled knobs.
84 //
85 // It works as follows: the palette colors that are assigned as the line color previously
86 // had been hard coded in the drawKnob method for the different knob types. Now the
87 // drawKnob method uses the line color to draw the lines. By assigning the palette colors
88 // as the line colors here the knob lines will be drawn in this color unless the stylesheet
89 // overrides that color.
90 switch (knobNum())
91 {
92 case KnobType::Small17:
93 case KnobType::Bright26:
94 case KnobType::Dark28:
95 m_lineActiveColor = QApplication::palette().color(QPalette::Active, QPalette::WindowText);
96 m_arcActiveColor = QColor(QApplication::palette().color(
97 QPalette::Active, QPalette::WindowText));
98 m_arcActiveColor.setAlpha(70);
99 break;
100 case KnobType::Vintage32:
101 m_lineActiveColor = QApplication::palette().color(QPalette::Active, QPalette::Shadow);
102 m_arcActiveColor = QColor(QApplication::palette().color(
103 QPalette::Active, QPalette::Shadow));
104 m_arcActiveColor.setAlpha(70);
105 break;
106 default:
107 break;
108 }
109}

112void Knob::onKnobNumUpdated()
113{
114 if( m_knobNum != KnobType::Styled )
115 {
116 QString knobFilename;
117 switch (m_knobNum)
118 {
119 case KnobType::Dark28:
120 knobFilename = "knob01";
121 break;
122 case KnobType::Bright26:
123 knobFilename = "knob02";
124 break;
125 case KnobType::Small17:
126 knobFilename = "knob03";
127 break;
128 case KnobType::Vintage32:
129 knobFilename = "knob05";
130 break;
131 case KnobType::Styled: // only here to stop the compiler from complaining
132 break;
133 }
134
135 // If knobFilename is still empty here we should get the fallback pixmap of size 1x1
136 m_knobPixmap = std::make_unique<QPixmap>(QPixmap(embed::getIconPixmap(knobFilename.toUtf8().constData())));
137 if (!this->isEnabled())
138 {
139 convertPixmapToGrayScale(*m_knobPixmap.get());
140 }
141 setFixedSize( m_knobPixmap->width(), m_knobPixmap->height() );
142 }
143}

171void Knob::updateFixedSize()
172{
173 if (fixedFontSizeLabelRendering())
174 {
175 if (m_knobPixmap)
176 {
177 // In legacy mode only the width of the label is taken into account while the height is not
178 const int labelWidth = horizontalAdvance(QFontMetrics(adjustedToPixelSize(font(), SMALL_FONT_SIZE)), m_label);
179 const int width = std::max(m_knobPixmap->width(), labelWidth);
180
181 // Legacy mode assumes that the label will fit into 10 pixels plus some of the pixmap area
182 setFixedSize(width, m_knobPixmap->height() + 10);
183 }
184 }
185 else
186 {
187 // Styled knobs do not use pixmaps and have no labels. Their size is set from the outside and
188 // they are painted within these limits. Hence we should not compute a new size from a pixmap
189 // and/or label the case of styled knobs.
190 if (knobNum() == KnobType::Styled)
191 {
192 return;
193 }
194
195 QSize pixmapSize = m_knobPixmap ? m_knobPixmap->size() : QSize(0, 0);
196
197 auto fm = QFontMetrics(font());
198
199 const int width = std::max(pixmapSize.width(), horizontalAdvance(fm, m_label));
200 const int height = pixmapSize.height() + fm.height();
201
202 setFixedSize(width, height);
203 }

349QLineF Knob::calculateLine( const QPointF & _mid, float _radius, float _innerRadius ) const
350{
351 const float rarc = m_angle * std::numbers::pi_v<float> / 180.0;
352 const float ca = std::cos(rarc);
353 const float sa = -std::sin(rarc);
354
355 return QLineF( _mid.x() - sa*_innerRadius, _mid.y() - ca*_innerRadius,
356 _mid.x() - sa*_radius, _mid.y() - ca*_radius );
357}

361bool Knob::updateAngle()
362{
363 int angle = 0;
364 if( model() && model()->maxValue() != model()->minValue() )
365 {
366 angle = angleFromValue( model()->inverseScaledValue( model()->value() ), model()->minValue(), model()->maxValue(), m_totalAngle );
367 }
368 if( qAbs( angle - m_angle ) > 0 )
369 {
370 m_angle = angle;
371 return true;
372 }
373 return false;
374}

379void Knob::drawKnob( QPainter * _p )
380{
381 bool enabled = this->isEnabled();
382 QColor currentArcColor = enabled ? m_arcActiveColor : m_arcInactiveColor;
383 QColor currentLineColor = enabled ? m_lineActiveColor : m_lineInactiveColor;
384
385 if( updateAngle() == false && !m_cache.isNull() )
386 {
387 _p->drawImage( 0, 0, m_cache );
388 return;
389 }
390
391 m_cache = QImage( size(), QImage::Format_ARGB32 );
392 m_cache.fill( qRgba( 0, 0, 0, 0 ) );
393
394 QPainter p( &m_cache );
395
396 QPoint mid;
397
398 if( m_knobNum == KnobType::Styled )
399 {
400 p.setRenderHint( QPainter::Antialiasing );
401
402 // Perhaps this can move to setOuterRadius()
403 if( m_outerColor.isValid() )
404 {
405 QRadialGradient gradient( centerPoint(), outerRadius() );
406 gradient.setColorAt( 0.4, _p->pen().brush().color() );
407 gradient.setColorAt( 1, m_outerColor );
408
409 p.setPen( QPen( gradient, lineWidth(),
410 Qt::SolidLine, Qt::RoundCap ) );
411 }
412 else {
413 QPen pen = p.pen();
414 pen.setWidth( (int) lineWidth() );
415 pen.setCapStyle( Qt::RoundCap );
416
417 p.setPen( pen );
418 }
419
420 p.drawLine( calculateLine( centerPoint(), outerRadius(),
421 innerRadius() ) );
422 p.end();
423 _p->drawImage( 0, 0, m_cache );
424 return;
425 }
426
427
428 // Old-skool knobs
429 const float radius = m_knobPixmap->width() / 2.0f - 1;
430 mid = QPoint( width() / 2, m_knobPixmap->height() / 2 );
431
432 p.drawPixmap( static_cast<int>(
433 width() / 2 - m_knobPixmap->width() / 2 ), 0,
434 *m_knobPixmap );
435
436 p.setRenderHint( QPainter::Antialiasing );
437
438 const int centerAngle = angleFromValue( model()->inverseScaledValue( model()->centerValue() ), model()->minValue(), model()->maxValue(), m_totalAngle );
439
440 const int arcLineWidth = 2;
441 const int arcRectSize = m_knobPixmap->width() - arcLineWidth;
442
443 p.setPen(QPen(currentArcColor, 2));
444 p.drawArc( mid.x() - arcRectSize/2, 1, arcRectSize, arcRectSize, 315*16, 16*m_totalAngle );
445
446 p.setPen(QPen(currentLineColor, 2));
447 switch( m_knobNum )
448 {
449 case KnobType::Small17:
450 {
451 p.drawLine( calculateLine( mid, radius-2 ) );
452 break;
453 }
454 case KnobType::Bright26:
455 {
456 p.drawLine( calculateLine( mid, radius-5 ) );
457 break;
458 }
459 case KnobType::Dark28:
460 {
461 const float rb = qMax<float>( ( radius - 10 ) / 3.0,
462 0.0 );
463 const float re = qMax<float>( ( radius - 4 ), 0.0 );
464 QLineF ln = calculateLine( mid, re, rb );
465 ln.translate( 1, 1 );
466 p.drawLine( ln );
467 break;
468 }
469 case KnobType::Vintage32:
470 {
471 p.drawLine( calculateLine( mid, radius-2, 2 ) );
472 break;
473 }
474 case KnobType::Styled:
475 break;
476 }
477
478 p.drawArc( mid.x() - arcRectSize/2, 1, arcRectSize, arcRectSize, (90-centerAngle)*16, -16*(m_angle-centerAngle) );
479
480 p.end();
481
482 _p->drawImage( 0, 0, m_cache );
483}

485void Knob::drawLabel(QPainter& p)
486{
487 if( !m_label.isEmpty() )
488 {
489 if (fixedFontSizeLabelRendering())
490 {
491 p.setFont(adjustedToPixelSize(p.font(), SMALL_FONT_SIZE));
492 }
493 auto fm = p.fontMetrics();
494 const auto x = (width() - horizontalAdvance(fm, m_label)) / 2;
495 const auto descent = fixedFontSizeLabelRendering() ? 2 : fm.descent();
496 const auto y = height() - descent;
497
498 p.setPen(textColor());
499 p.drawText(x, y, m_label);
500 }

511void Knob::changeEvent(QEvent * ev)
512{
513 if (ev->type() == QEvent::EnabledChange)
514 {
515 onKnobNumUpdated();
516 if (!m_label.isEmpty())
517 {
518 setLabel(m_label);
519 }
520 m_cache = QImage();
521 update();
522 }
523 else if (ev->type() == QEvent::FontChange)
524 {
525 // The size of the label might have changed so update
526 // the size of this widget.
527 updateFixedSize();
528
529 update();
530 }

534void convertPixmapToGrayScale(QPixmap& pixMap)
535{
536 QImage temp = pixMap.toImage().convertToFormat(QImage::Format_ARGB32);
537 for (int i = 0; i < temp.height(); ++i)
538 {
539 for (int j = 0; j < temp.width(); ++j)
540 {
541 const auto pix = temp.pixelColor(i, j);
542 const auto gscale = 0.2126 * pix.redF() + 0.7152 * pix.greenF() + 0.0722 * pix.blueF();
543 const auto pixGray = QColor::fromRgbF(gscale, gscale, gscale, pix.alphaF());
544 temp.setPixelColor(i, j, pixGray);
545 }
546 }
547 pixMap.convertFromImage(temp);
548}
