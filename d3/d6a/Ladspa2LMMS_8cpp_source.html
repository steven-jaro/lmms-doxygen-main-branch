34Ladspa2LMMS::Ladspa2LMMS()
35{
36 l_sortable_plugin_t plugins = getSortedPlugins();
37
38 for (const auto& plugin : plugins)
39 {
40 ladspa_key_t key = plugin.second;
41 LadspaManagerDescription * desc = getDescription( key );
42
43 if( desc->type == LadspaPluginType::Source )
44 {
45 m_instruments.append( qMakePair( getName( key ),
46 key ) );
47 }
48 else if( desc->type == LadspaPluginType::Transfer &&
49 ( desc->inputChannels == desc->outputChannels &&
50 ( desc->inputChannels == 1 ||
51 desc->inputChannels == 2 ||
52 desc->inputChannels == 4 )/* &&
53 isRealTimeCapable( key )*/ ) )
54 {
55 m_validEffects.append( qMakePair( getName( key ),
56 key ) );
57 }
58 else if( desc->type == LadspaPluginType::Transfer &&
59 ( desc->inputChannels != desc->outputChannels ||
60 ( desc->inputChannels != 1 &&
61 desc->inputChannels != 2 &&
62 desc->inputChannels != 4 ) ||
63 !isRealTimeCapable( key ) ) )
64 {
65 m_invalidEffects.append( qMakePair( getName( key ),
66 key ) );
67 }
68 else if( desc->type == LadspaPluginType::Sink )
69 {
70 m_analysisTools.append( qMakePair( getName( key ),
71 key ) );
72 }
73 else if( desc->type == LadspaPluginType::Other )
74 {
75 m_otherPlugins.append( qMakePair( getName( key ),
76 key ) );
77 }
78 }
79}

84QString Ladspa2LMMS::getShortName( const ladspa_key_t & _key )
85{
86 QString name = getName( _key );
87
88 if( name.indexOf( "(" ) > 0 )
89 {
90 name = name.left( name.indexOf( "(" ) );
91 }
92 if( name.indexOf( " - " ) > 0 )
93 {
94 name = name.left( name.indexOf( " - " ) );
95 }
96 if( name.indexOf( " " ) > 0 )
97 {
98 name = name.left( name.indexOf( " " ) );
99 }
100 Qt::CaseSensitivity cs = Qt::CaseInsensitive;
101 if( name.indexOf( " with ", 0, cs ) > 0 )
102 {
103 name = name.left( name.indexOf( " with ", 0, cs ) );
104 }
105 if( name.indexOf( ",", 0, cs ) > 0 )
106 {
107 name = name.left( name.indexOf( ",", 0, cs ) );
108 }
109 if( name.length() > 40 )
110 {
111 int i = 40;
112 while( name[i] != ' ' && i > 0 )
113 {
114 i--;
115 }
116 name = name.left( i );
117 }
118 if( name.length() == 0 )
119 {
120 name = "LADSPA Plugin";
121 }
122
123 return name;
124}
