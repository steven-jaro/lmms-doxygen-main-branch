59template <int NC>
60Downsampler2xNeon <NC>::Downsampler2xNeon () noexcept
61: _filter ()
62{
63 for (int i = 0; i < _nbr_stages + 1; ++i)
64 {
65 storea (_filter [i]._mem, vdup_n_f32 (0));
66 }
67 if ((NBR_COEFS & 1) != 0)
68 {
69 const int pos = (NBR_COEFS ^ 1) & (_stage_width - 1);
70 _filter [_nbr_stages]._coef [pos] = 1;
71 }
72
73 clear_buffers ();
74}

78/*
79==============================================================================
80Name: set_coefs
81Description:
82 Sets filter coefficients. Generate them with the PolyphaseIir2Designer
83 class.
84 Call this function before doing any processing.
85Input parameters:
86 - coef_arr: Array of coefficients. There should be as many coefficients as
87 mentioned in the class template parameter.
88Throws: Nothing
89==============================================================================
90*/
91
92template <int NC>
93void Downsampler2xNeon <NC>::set_coefs (const double coef_arr []) noexcept
94{
95 assert (coef_arr != nullptr);
96
97 for (int i = 0; i < NBR_COEFS; ++i)
98 {
99 const int stage = (i / _stage_width) + 1;
100 const int pos = (i ^ 1) & (_stage_width - 1);
101 _filter [stage]._coef [pos] = float (coef_arr [i]);
102 }
103}

107/*
108==============================================================================
109Name: process_sample
110Description:
111 Downsamples (x2) one pair of samples, to generate one output sample.
112Input parameters:
113 - in_ptr: pointer on the two samples to decimate
114Returns: Samplerate-reduced sample.
115Throws: Nothing
116==============================================================================
117*/
118
119template <int NC>
120float Downsampler2xNeon <NC>::process_sample (const float in_ptr [2]) noexcept
121{
122 assert (in_ptr != nullptr);
123
124 auto x = load2u (in_ptr);
125 StageProcNeonV2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
126#if 0
127 const float out = vget_lane_f32 (vpadd_f32 (x, x), 0);
128#else
129 const float out = vget_lane_f32 (x, 1) + vget_lane_f32 (x, 0);
130#endif
131
132 return out * 0.5f;
133}

137/*
138==============================================================================
139Name: process_block
140Description:
141 Downsamples (x2) a block of samples.
142 Input and output blocks may overlap, see assert() for details.
143Input parameters:
144 - in_ptr: Input array, containing nbr_spl * 2 samples.
145 - nbr_spl: Number of samples to output, > 0
146Output parameters:
147 - out_ptr: Array for the output samples, capacity: nbr_spl samples.
148Throws: Nothing
149==============================================================================
150*/
151
152template <int NC>
153void Downsampler2xNeon <NC>::process_block (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
154{
155 assert (in_ptr != nullptr);
156 assert (out_ptr != nullptr);
157 assert (out_ptr <= in_ptr || out_ptr >= in_ptr + nbr_spl * 2);
158 assert (nbr_spl > 0);
159
160 const long n4 = process_block_quad (
161 out_ptr, nullptr, in_ptr, nbr_spl, store_low, bypass
162 );
163
164 for (long pos = n4; pos < nbr_spl; ++pos)
165 {
166 out_ptr [pos] = process_sample (in_ptr + pos * 2);
167 }
168}

172/*
173==============================================================================
174Name: process_sample_split
175Description:
176 Split (spectrum-wise) in half a pair of samples. The lower part of the
177 spectrum is a classic downsampling, equivalent to the output of
178 process_sample().
179 The higher part is the complementary signal: original filter response
180 is flipped from left to right, becoming a high-pass filter with the same
181 cutoff frequency. This signal is then critically sampled (decimation by 2),
182 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
183Input parameters:
184 - in_ptr: pointer on the pair of input samples
185Output parameters:
186 - low: output sample, lower part of the spectrum (downsampling)
187 - high: output sample, higher part of the spectrum.
188Throws: Nothing
189==============================================================================
190*/
191
192template <int NC>
193void Downsampler2xNeon <NC>::process_sample_split (float &low, float &high, const float in_ptr [2]) noexcept
194{
195 assert (in_ptr != nullptr);
196
197 auto x = load2u (in_ptr);
198 StageProcNeonV2 <_nbr_stages>::process_sample_pos (x, _filter.data ());
199 x *= vdup_n_f32 (0.5f);
200 const float x0 = vget_lane_f32 (x, 0);
201 const float x1 = vget_lane_f32 (x, 1);
202 low = x1 + x0;
203 high = x1 - x0;
204}

208/*
209==============================================================================
210Name: process_block_split
211Description:
212 Split (spectrum-wise) in half a block of samples. The lower part of the
213 spectrum is a classic downsampling, equivalent to the output of
214 process_block().
215 The higher part is the complementary signal: original filter response
216 is flipped from left to right, becoming a high-pass filter with the same
217 cutoff frequency. This signal is then critically sampled (decimation by 2),
218 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
219 Input and output blocks may overlap, see assert() for details.
220Input parameters:
221 - in_ptr: Input array, containing nbr_spl * 2 samples.
222 - nbr_spl: Number of samples for each output, > 0
223Output parameters:
224 - out_l_ptr: Array for the output samples, lower part of the spectrum
225 (downsampling). Capacity: nbr_spl samples.
226 - out_h_ptr: Array for the output samples, higher part of the spectrum.
227 Capacity: nbr_spl samples.
228Throws: Nothing
229==============================================================================
230*/
231
232template <int NC>
233void Downsampler2xNeon <NC>::process_block_split (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl) noexcept
234{
235 assert (in_ptr != nullptr);
236 assert (out_l_ptr != nullptr);
237 assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl * 2);
238 assert (out_h_ptr != nullptr);
239 assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl * 2);
240 assert (out_h_ptr != out_l_ptr);
241 assert (nbr_spl > 0);
242
243 const long n4 = process_block_quad (
244 out_l_ptr, out_h_ptr, in_ptr, nbr_spl, store_low, store_high
245 );
246
247 for (long pos = n4; pos < nbr_spl; ++pos)
248 {
249 process_sample_split (out_l_ptr [pos], out_h_ptr [pos], in_ptr + pos * 2);
250 }
251}

291template <int NC>
292template <typename FL, typename FH>
293long Downsampler2xNeon <NC>::process_block_quad (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl, FL fnc_l, FH fnc_h) noexcept
294{
295 assert (_phase == 0);
296
297 const auto half = vdupq_n_f32 (0.5f);
298
299 const long n4 = nbr_spl & ~(4-1);
300 for (long pos = 0; pos < n4; pos += 4)
301 {
302 const auto x_0 = load4u (in_ptr + pos * 2 );
303 const auto x_2 = load4u (in_ptr + pos * 2 + 4);
304
305 auto y_0 = vget_low_f32 (x_0);
306 auto y_1 = vget_high_f32 (x_0);
307 auto y_2 = vget_low_f32 (x_2);
308 auto y_3 = vget_high_f32 (x_2);
309 StageProcNeonV2 <_nbr_stages>::process_sample_pos (y_0, _filter.data ());
310 StageProcNeonV2 <_nbr_stages>::process_sample_pos (y_1, _filter.data ());
311 StageProcNeonV2 <_nbr_stages>::process_sample_pos (y_2, _filter.data ());
312 StageProcNeonV2 <_nbr_stages>::process_sample_pos (y_3, _filter.data ());
313
314 const auto u_02 = vcombine_f32 (y_0, y_2);
315 const auto u_13 = vcombine_f32 (y_1, y_3);
316 const auto both = vtrnq_f32 (u_02, u_13);
317 const auto odd = both.val [0];
318 const auto even = both.val [1];
319 fnc_l (out_l_ptr + pos, even, odd, half);
320 fnc_h (out_h_ptr + pos, even, odd, half);
321 }
322
323 return n4;
324}
