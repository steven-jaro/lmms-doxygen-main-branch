83void FadeButton::noteEnd()
84{
85 if (activeNotes <= 0)
86 {
87 qWarning("noteEnd() triggered without a corresponding activate()!");
88 activeNotes = 0;
89 }
90 else
91 {
92 activeNotes--;
93 }
94
95 if (activeNotes == 0)
96 {
97 m_releaseTimer.restart();
98 }

100 update();

106void FadeButton::paintEvent(QPaintEvent * _pe)
107{
108 QColor col = m_normalColor;
109
110 if(m_stateTimer.isValid() && m_stateTimer.elapsed() < FadeDuration)
111 {
112 // The first part of the fade, when a note is triggered.
113 col = fadeToColor(m_activatedColor, m_holdColor, m_stateTimer, FadeDuration);
114 QTimer::singleShot(20, this, SLOT(update()));
115 }
116 else if (m_stateTimer.isValid()
117 && m_stateTimer.elapsed() >= FadeDuration
118 && activeNotes > 0)
119 {
120 // The fade is done, but at least one note is still held.
121 col = m_holdColor;
122 }
123 else if (m_releaseTimer.isValid() && m_releaseTimer.elapsed() < FadeDuration)
124 {
125 // Last note just ended. Fade to default color.
126 col = fadeToColor(m_holdColor, m_normalColor, m_releaseTimer, FadeDuration);
127 QTimer::singleShot(20, this, SLOT(update()));
128 }
129 else
130 {
131 // No fade, no notes. Set to default color.
132 col = m_normalColor;
133 }
134
135 QPainter p(this);
136 p.fillRect(rect(), col);
137
138 int w = rect().right();
139 int h = rect().bottom();
140 p.setPen(m_normalColor.darker(130));
141 p.drawLine(w, 1, w, h);
142 p.drawLine(1, h, w, h);
143 p.setPen(m_normalColor.lighter(130));
144 p.drawLine(0, 0, 0, h-1);
145 p.drawLine(0, 0, w, 0);
146}

149QColor FadeButton::fadeToColor(QColor startCol, QColor endCol, QElapsedTimer timer, float duration)
150{
151 QColor col;
152
153 const float state = 1 - timer.elapsed() / duration;
154 const int r = (int)(endCol.red() * (1.0f - state)
155 + startCol.red() * state);
156 const int g = (int)(endCol.green() * (1.0f - state)
157 + startCol.green() * state);
158 const int b = (int)(endCol.blue() * (1.0f - state)
159 + startCol.blue() * state);
160 col.setRgb(r, g, b);
161
162 return col;
163}
