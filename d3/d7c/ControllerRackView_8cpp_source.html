116void ControllerRackView::deleteController( ControllerView * _view )
117{
118 Controller * c = _view->getController();
119
120 if( c->connectionCount() > 0 )
121 {
122 QMessageBox msgBox;
123 msgBox.setIcon( QMessageBox::Question );
124 msgBox.setWindowTitle( tr("Confirm Delete") );
125 msgBox.setText( tr("Confirm delete? There are existing connection(s) "
126 "associated with this controller. There is no way to undo.") );
127 msgBox.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
128 if( msgBox.exec() != QMessageBox::Ok )
129 {
130 return;
131 }
132 }
133
134 Song * song = Engine::getSong();
135 song->removeController( c );
136}

138void ControllerRackView::moveUp(ControllerView* view)
139{
140 if (view == m_controllerViews.first()) { return; }
141
142 const auto storedView = std::find(m_controllerViews.begin(), m_controllerViews.end(), view);
143 assert(storedView != m_controllerViews.end());
144
145 const auto index = std::distance(m_controllerViews.begin(), storedView);
146
147 std::swap(m_controllerViews[index - 1], m_controllerViews[index]);
148 m_scrollAreaLayout->removeWidget(view);
149 m_scrollAreaLayout->insertWidget(index - 1, view);
150}

152void ControllerRackView::moveDown(ControllerView* view)
153{
154 if (view == m_controllerViews.last()) { return; }
155
156 const auto storedView = std::find(m_controllerViews.begin(), m_controllerViews.end(), view);
157 assert(storedView != m_controllerViews.end());
158 moveUp(*std::next(storedView));
159}

161void ControllerRackView::addController(Controller* controller)
162{
163 QWidget * scrollAreaWidget = m_scrollArea->widget();
164
165 auto controllerView = new ControllerView(controller, scrollAreaWidget);
166
167 connect(controllerView, &ControllerView::movedUp, this, &ControllerRackView::moveUp);
168 connect(controllerView, &ControllerView::movedDown, this, &ControllerRackView::moveDown);
169 connect(controllerView, &ControllerView::removedController, this, &ControllerRackView::deleteController, Qt::QueuedConnection);
170
171 auto moveUpAction = new QAction(controllerView);
172 moveUpAction->setShortcut(combine(Qt::Key_Up, Qt::AltModifier));
173 moveUpAction->setShortcutContext(Qt::WidgetShortcut);
174 connect(moveUpAction, &QAction::triggered, controllerView, &ControllerView::moveUp);
175 controllerView->addAction(moveUpAction);
176
177 auto moveDownAction = new QAction(controllerView);
178 moveDownAction->setShortcut(combine(Qt::Key_Down, Qt::AltModifier));
179 moveDownAction->setShortcutContext(Qt::WidgetShortcut);
180 connect(moveDownAction, &QAction::triggered, controllerView, &ControllerView::moveDown);
181 controllerView->addAction(moveDownAction);
182
183
184 m_controllerViews.append( controllerView );
185 m_scrollAreaLayout->insertWidget( m_nextIndex, controllerView );
186
187 ++m_nextIndex;
188}

193void ControllerRackView::removeController(Controller* removedController)
194{
195 ControllerView * viewOfRemovedController = 0;
196
197 QVector<ControllerView *>::const_iterator end = m_controllerViews.end();
198 for ( QVector<ControllerView *>::const_iterator it = m_controllerViews.begin(); it != end; ++it)
199 {
200 ControllerView *currentControllerView = *it;
201 if ( currentControllerView->getController() == removedController )
202 {
203 viewOfRemovedController = currentControllerView;
204 break;
205 }
206 }
207
208 if (viewOfRemovedController )
209 {
210 m_controllerViews.erase( std::find( m_controllerViews.begin(),
211 m_controllerViews.end(), viewOfRemovedController ) );
212
213 delete viewOfRemovedController;
214 --m_nextIndex;
215 }
216}
