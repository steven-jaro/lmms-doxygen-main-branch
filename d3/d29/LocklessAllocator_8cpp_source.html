55LocklessAllocator::LocklessAllocator( size_t nmemb, size_t size )
56{
57 m_capacity = align( nmemb, SIZEOF_SET );
58 m_elementSize = align( size, sizeof( void * ) );
59 m_pool = new char[m_capacity * m_elementSize];
60
61 m_freeStateSets = m_capacity / SIZEOF_SET;
62 m_freeState = new std::atomic_int[m_freeStateSets];
63 std::fill(m_freeState, m_freeState + m_freeStateSets, 0);
64
65 m_available = m_capacity;
66 m_startIndex = 0;
67}

107void * LocklessAllocator::alloc()
108{
109 // Some of these CAS loops could probably use relaxed atomics, as discussed
110 // in http://en.cppreference.com/w/cpp/atomic/atomic/compare_exchange.
111 // Let's use sequentially-consistent ops to be safe for now.
112 auto available = m_available.load();
113 do
114 {
115 if( !available )
116 {
117 fprintf( stderr, "LocklessAllocator: No free space\n" );
118 return nullptr;
119 }
120 }
121 while (!m_available.compare_exchange_weak(available, available - 1));
122
123 const size_t startIndex = m_startIndex++ % m_freeStateSets;
124 for (size_t set = startIndex;; set = ( set + 1 ) % m_freeStateSets)
125 {
126 for (int freeState = m_freeState[set]; freeState != -1;)
127 {
128 int bit = ffs( ~freeState ) - 1;
129 if (m_freeState[set].compare_exchange_weak(freeState,
130 freeState | 1 << bit ) )
131 {
132 return m_pool + ( SIZEOF_SET * set + bit )
133 * m_elementSize;
134 }
135 }
136 }
137}

142void LocklessAllocator::free( void * ptr )
143{
144 ptrdiff_t diff = (char *)ptr - m_pool;
145 if( diff < 0 || diff % m_elementSize )
146 {
147invalid:
148 fprintf( stderr, "LocklessAllocator: Invalid pointer\n" );
149 return;
150 }
151 size_t offset = diff / m_elementSize;
152 if( offset >= m_capacity )
153 {
154 goto invalid;
155 }
156 size_t set = offset / SIZEOF_SET;
157 int bit = offset % SIZEOF_SET;
158 int mask = 1 << bit;
159 int prevState = m_freeState[set].fetch_and(~mask);
160 if ( !( prevState & mask ) )
161 {
162 fprintf( stderr, "LocklessAllocator: Block not in use\n" );
163 return;
164 }
165 ++m_available;
166}
