61bool JackMessageBuffer::Stop()
62{
63 if (fOverruns > 0) {
64 jack_error("WARNING: %d message buffer overruns!", fOverruns);
65 } else {
66 jack_log("no message buffer overruns");
67 }
68
69 if (fGuard.Lock()) {
70 fRunning = false;
71 fGuard.Signal();
72 fGuard.Unlock();
73 fThread.Stop();
74 } else {
75 fThread.Kill();
76 }
77
78 Flush();
79 return true;
80}

90void JackMessageBuffer::AddMessage(int level, const char *message)
91{
92 if (fGuard.Trylock()) {
93 fBuffers[fInBuffer].level = level;
94 strncpy(fBuffers[fInBuffer].message, message, MB_BUFFERSIZE);
95 fInBuffer = MB_NEXT(fInBuffer);
96 fGuard.Signal();
97 fGuard.Unlock();
98 } else { /* lock collision */
99 INC_ATOMIC(&fOverruns);

103bool JackMessageBuffer::Execute()
104{
105 if (fGuard.Lock()) {
106 while (fRunning) {
107 fGuard.Wait();
108 /* the client asked for all threads to run a thread
109 initialization callback, which includes us.
110 */
111 if (fInit) {
112 fInit(fInitArg);
113 fInit = NULL;
114 /* and we're done */
115 fGuard.Signal();
116 }
117
118 /* releasing the mutex reduces contention */
119 fGuard.Unlock();
120 Flush();
121 fGuard.Lock();
122 }
123 fGuard.Unlock();
124 } else {
125 jack_error("JackMessageBuffer::Execute lock cannot be taken");
126 }
127
128 return false;
129}

168int JackMessageBuffer::SetInitCallback(JackThreadInitCallback callback, void *arg)
169{
170 if (fInstance && callback && fRunning && fGuard.Lock()) {
171 /* set up the callback */
172 fInitArg = arg;
173 fInit = callback;
174
175 #ifndef WIN32
176 // wake msg buffer thread
177 fGuard.Signal();
178 // wait for it to be done
179 fGuard.Wait();
180 // and we're done
181 fGuard.Unlock();
182 #else
183 /*
184 The condition variable emulation code does not work reliably on Windows (lost signal).
185 So use a "hackish" way to signal/wait for the result.
186 Probably better in the long term : use pthread-win32 (http://sourceware.org/pthreads-win32/`
187 */
188 fGuard.Unlock();
189 int count = 0;
190 while (fInit && ++count < 1000) {
191 /* wake msg buffer thread */
192 fGuard.Signal();
193 JackSleep(1000);
194 }
195 if (count == 1000) {
196 jack_error("JackMessageBuffer::SetInitCallback : signal lost");
197 return -1;
198 }
199 #endif
200 return 0;
201 }
202 jack_error("JackMessageBuffer::SetInitCallback : callback could not be executed");
203 return -1;
204}
