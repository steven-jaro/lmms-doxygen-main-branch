599bool MainWindow::mayChangeProject(bool stopPlayback)
600{
601 if( stopPlayback )
602 {
603 Engine::getSong()->stop();
604 }
605
606 if( !Engine::getSong()->isModified() && getSession() != SessionState::Recover )
607 {
608 return( true );
609 }
610
611 // Separate message strings for modified and recovered files
612 QString messageTitleRecovered = tr( "Recovered project not saved" );
613 QString messageRecovered = tr( "This project was recovered from the "
614 "previous session. It is currently "
615 "unsaved and will be lost if you don't "
616 "save it. Do you want to save it now?" );
617
618 QString messageTitleUnsaved = tr( "Project not saved" );
619 QString messageUnsaved = tr( "The current project was modified since "
620 "last saving. Do you want to save it "
621 "now?" );
622
623 QMessageBox mb( ( getSession() == SessionState::Recover ?
624 messageTitleRecovered : messageTitleUnsaved ),
625 ( getSession() == SessionState::Recover ?
626 messageRecovered : messageUnsaved ),
627 QMessageBox::Question,
628 QMessageBox::Save,
629 QMessageBox::Discard,
630 QMessageBox::Cancel,
631 this );
632 int answer = mb.exec();
633
634 if( answer == QMessageBox::Save )
635 {
636 return( saveProject() );
637 }
638 else if( answer == QMessageBox::Discard )
639 {
640 if( getSession() == SessionState::Recover )
641 {
642 sessionCleanup();
643 }
644 return( true );
645 }
646
647 return( false );
648}

663void MainWindow::saveWidgetState( QWidget * _w, QDomElement & _de )
664{
665 // If our widget is the main content of a window (e.g. piano roll, Mixer, etc),
666 // we really care about the position of the *window* - not the position of the widget within its window
667 if( _w->parentWidget() != nullptr &&
668 _w->parentWidget()->inherits( "QMdiSubWindow" ) )
669 {
670 _w = _w->parentWidget();
671 }
672
673 // If the widget is a SubWindow, then we can make use of the getTrueNormalGeometry() method that
674 // performs the same as normalGeometry, but isn't broken on X11 ( see https://bugreports.qt.io/browse/QTBUG-256 )
675 auto asSubWindow = qobject_cast<SubWindow*>(_w);
676 QRect normalGeom = asSubWindow != nullptr ? asSubWindow->getTrueNormalGeometry() : _w->normalGeometry();
677
678 bool visible = _w->isVisible();
679 _de.setAttribute( "visible", visible );
680 _de.setAttribute( "minimized", _w->isMinimized() );
681 _de.setAttribute( "maximized", _w->isMaximized() );
682
683 _de.setAttribute( "x", normalGeom.x() );
684 _de.setAttribute( "y", normalGeom.y() );
685
686 QSize sizeToStore = normalGeom.size();
687 _de.setAttribute( "width", sizeToStore.width() );
688 _de.setAttribute( "height", sizeToStore.height() );
689}

694void MainWindow::restoreWidgetState( QWidget * _w, const QDomElement & _de )
695{
696 QRect r( qMax( 1, _de.attribute( "x" ).toInt() ),
697 qMax( 1, _de.attribute( "y" ).toInt() ),
698 qMax( _w->sizeHint().width(), _de.attribute( "width" ).toInt() ),
699 qMax( _w->minimumHeight(), _de.attribute( "height" ).toInt() ) );
700 if( _de.hasAttribute( "visible" ) && !r.isNull() )
701 {
702 // If our widget is the main content of a window (e.g. piano roll, Mixer, etc),
703 // we really care about the position of the *window* - not the position of the widget within its window
704 if ( _w->parentWidget() != nullptr &&
705 _w->parentWidget()->inherits( "QMdiSubWindow" ) )
706 {
707 _w = _w->parentWidget();
708 }
709 // first restore the window, as attempting to resize a maximized window causes graphics glitching
710 _w->setWindowState( _w->windowState() & ~(Qt::WindowMaximized | Qt::WindowMinimized) );
711
712 // Check isEmpty() to work around corrupt project files with empty size
713 if ( ! r.size().isEmpty() ) {
714 _w->resize( r.size() );
715 }
716 _w->move( r.topLeft() );
717
718 // set the window to its correct minimized/maximized/restored state
719 Qt::WindowStates flags = _w->windowState();
720 flags = _de.attribute( "minimized" ).toInt() ?
721 ( flags | Qt::WindowMinimized ) :
722 ( flags & ~Qt::WindowMinimized );
723 flags = _de.attribute( "maximized" ).toInt() ?
724 ( flags | Qt::WindowMaximized ) :
725 ( flags & ~Qt::WindowMaximized );
726 _w->setWindowState( flags );
727
728 _w->setVisible( _de.attribute( "visible" ).toInt() );
729 }

1616void MainWindow::MovableQMdiArea::mouseMoveEvent(QMouseEvent* event)
1617{
1618 if (m_isBeingMoved == false) { return; }
1619
1620 int minXBoundary = window()->width() - 100;
1621 int maxXBoundary = 100;
1622 int minYBoundary = window()->height() - 100;
1623 int maxYBoundary = 100;
1624
1625 int minX = minXBoundary;
1626 int maxX = maxXBoundary;
1627 int minY = minYBoundary;
1628 int maxY = maxYBoundary;
1629
1630 auto subWindows = subWindowList();
1631 for (auto* curWindow : subWindows)
1632 {
1633 if (curWindow->isVisible())
1634 {
1635 minX = std::min(minX, curWindow->x());
1636 maxX = std::max(maxX, curWindow->x() + curWindow->width());
1637 minY = std::min(minY, curWindow->y());
1638 maxY = std::max(maxY, curWindow->y() + curWindow->height());
1639 }
1640 }
1641
1642 int scrollX = m_lastX - event->x();
1643 int scrollY = m_lastY - event->y();
1644
1645 scrollX = scrollX < 0 && minX >= minXBoundary ? 0 : scrollX;
1646 scrollX = scrollX > 0 && maxX <= maxXBoundary ? 0 : scrollX;
1647 scrollY = scrollY < 0 && minY >= minYBoundary ? 0 : scrollY;
1648 scrollY = scrollY > 0 && maxY <= maxYBoundary ? 0 : scrollY;
1649
1650 for (auto* curWindow : subWindows)
1651 {
1652 // if widgets are maximized, then they shouldn't be moved
1653 // moving a maximized window's normalGeometry is not implemented because of difficulties
1654 if (curWindow->isMaximized() == false)
1655 {
1656 curWindow->move(curWindow->x() - scrollX, curWindow->y() - scrollY);
1657 }
1658 }
1659
1660 m_lastX = event->x();
1661 m_lastY = event->y();
1662}
