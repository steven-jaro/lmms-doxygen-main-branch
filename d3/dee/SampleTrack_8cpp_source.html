73bool SampleTrack::play( const TimePos & _start, const fpp_t _frames,
74 const f_cnt_t _offset, int _clip_num )
75{
76 m_audioBusHandle.effects()->startRunning();
77 bool played_a_note = false; // will be return variable
78
79
80 clipVector clips;
81 class PatternTrack * pattern_track = nullptr;
82 if( _clip_num >= 0 )
83 {
84 if (_start > getClip(_clip_num)->length())
85 {
86 setPlaying(false);
87 }
88 if( _start != 0 )
89 {
90 return false;
91 }
92 clips.push_back( getClip( _clip_num ) );
93 if (trackContainer() == Engine::patternStore())
94 {
95 pattern_track = PatternTrack::findPatternTrack(_clip_num);
96 setPlaying(true);
97 }
98 }
99 else
100 {
101 bool nowPlaying = false;
102 for( int i = 0; i < numOfClips(); ++i )
103 {
104 Clip * clip = getClip( i );
105 auto sClip = dynamic_cast<SampleClip*>(clip);
106
107 if( _start >= sClip->startPosition() && _start < sClip->endPosition() )
108 {
109 if( sClip->isPlaying() == false && _start >= (sClip->startPosition() + sClip->startTimeOffset()) )
110 {
111 auto bufferFramesPerTick = Engine::framesPerTick(sClip->sample().sampleRate());
112 f_cnt_t sampleStart = bufferFramesPerTick * ( _start - sClip->startPosition() - sClip->startTimeOffset() );
113 f_cnt_t clipFrameLength = bufferFramesPerTick * ( sClip->endPosition() - sClip->startPosition() - sClip->startTimeOffset() );
114 f_cnt_t sampleBufferLength = sClip->sample().sampleSize();
115 //if the Clip smaller than the sample length we play only until Clip end
116 //else we play the sample to the end but nothing more
117 f_cnt_t samplePlayLength = clipFrameLength > sampleBufferLength ? sampleBufferLength : clipFrameLength;
118 //we only play within the sampleBuffer limits
119 if( sampleStart < sampleBufferLength )
120 {
121 sClip->setSampleStartFrame( sampleStart );
122 sClip->setSamplePlayLength( samplePlayLength );
123 clips.push_back( sClip );
124 sClip->setIsPlaying( true );
125 nowPlaying = true;
126 }
127 }
128 }
129 else
130 {
131 sClip->setIsPlaying( false );
132 }
133 nowPlaying = nowPlaying || sClip->isPlaying();
134 }
135 setPlaying(nowPlaying);
136 }
137
138 for (const auto& clip : clips)
139 {
140 auto st = dynamic_cast<SampleClip*>(clip);
141 if( !st->isMuted() )
142 {
143 PlayHandle* handle;
144 if( st->isRecord() )
145 {
146 if( !Engine::getSong()->isRecording() )
147 {
148 return played_a_note;
149 }
150 auto smpHandle = new SampleRecordHandle(st);
151 handle = smpHandle;
152 }
153 else
154 {
155 auto smpHandle = new SamplePlayHandle(st);
156 smpHandle->setVolumeModel( &m_volumeModel );
157 smpHandle->setPatternTrack(pattern_track);
158 handle = smpHandle;
159 }
160 handle->setOffset( _offset );
161 // send it to the audio engine
162 Engine::audioEngine()->addPlayHandle( handle );
163 played_a_note = true;
164 }
165 }
166
167 return played_a_note;
168}

202void SampleTrack::loadTrackSpecificSettings(const QDomElement & thisElem)
203{
204 QDomNode node = thisElem.firstChild();
205 m_audioBusHandle.effects()->clear();
206 while(!node.isNull())
207 {
208 if (node.isElement())
209 {
210 if (m_audioBusHandle.effects()->nodeName() == node.nodeName())
211 {
212 m_audioBusHandle.effects()->restoreState(node.toElement());
213 }
214 }
215 node = node.nextSibling();
216 }
217 m_volumeModel.loadSettings(thisElem, "vol");
218 m_panningModel.loadSettings(thisElem, "pan");
219 m_mixerChannelModel.setRange(0, Engine::mixer()->numChannels() - 1);
220 m_mixerChannelModel.loadSettings(thisElem, "mixch");
221}
