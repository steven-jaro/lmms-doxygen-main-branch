50SampleTrackWindow::SampleTrackWindow(SampleTrackView * tv) :
51 QWidget(),
52 ModelView(nullptr, this),
53 m_track(tv->model()),
54 m_stv(tv)
55{
56#if QT_VERSION < 0x50C00
57 // Workaround for a bug in Qt versions below 5.12,
58 // where argument-dependent-lookup fails for QFlags operators
59 // declared inside a namespace.
60 // This affects the Q_DECLARE_OPERATORS_FOR_FLAGS macro in Instrument.h
61 // See also: https://codereview.qt-project.org/c/qt/qtbase/+/225348
62
63 using ::operator|;
64#endif
65
66 // init own layout + widgets
67 setFocusPolicy(Qt::StrongFocus);
68 auto vlayout = new QVBoxLayout(this);
69 vlayout->setContentsMargins(0, 0, 0, 0);
70 vlayout->setSpacing(0);
71
72 auto generalSettingsWidget = new QWidget(this);
73 auto generalSettingsLayout = new QVBoxLayout(generalSettingsWidget);
74
75 auto nameWidget = new QWidget(generalSettingsWidget);
76 auto nameLayout = new QHBoxLayout(nameWidget);
77 nameLayout->setContentsMargins(0, 0, 0, 0);
78 nameLayout->setSpacing(2);
79
80 // setup line edit for changing sample track name
81 m_nameLineEdit = new QLineEdit;
82 connect(m_nameLineEdit, SIGNAL(textChanged(const QString&)),
83 this, SLOT(textChanged(const QString&)));
84
85 m_nameLineEdit->setSizePolicy(QSizePolicy(QSizePolicy::Expanding, QSizePolicy::Preferred));
86 nameLayout->addWidget(m_nameLineEdit);
87
88
89 generalSettingsLayout->addWidget(nameWidget);
90
91 auto basicControlsLayout = new QGridLayout;
92 basicControlsLayout->setHorizontalSpacing(3);
93 basicControlsLayout->setVerticalSpacing(0);
94 basicControlsLayout->setContentsMargins(0, 0, 0, 0);
95
96 QString labelStyleSheet = "font-size: 10px;";
97 Qt::Alignment labelAlignment = Qt::AlignHCenter | Qt::AlignTop;
98 Qt::Alignment widgetAlignment = Qt::AlignHCenter | Qt::AlignCenter;
99
100 auto soloMuteLayout = new QVBoxLayout();
101 soloMuteLayout->setContentsMargins(0, 0, 2, 0);
102 soloMuteLayout->setSpacing(2);
103
104 m_muteBtn = new AutomatableButton(this, tr("Mute"));
105 m_muteBtn->setModel(&m_track->m_mutedModel);
106 m_muteBtn->setObjectName("btn-mute");
107 m_muteBtn->setCheckable(true);
108 m_muteBtn->setToolTip(tr("Mute this sample track"));
109 soloMuteLayout->addWidget(m_muteBtn, 0, widgetAlignment);
110
111 m_soloBtn = new AutomatableButton(this, tr("Solo"));
112 m_soloBtn->setModel(&m_track->m_soloModel);
113 m_soloBtn->setObjectName("btn-solo");
114 m_soloBtn->setCheckable(true);
115 m_soloBtn->setToolTip(tr("Solo this sample track"));
116 soloMuteLayout->addWidget(m_soloBtn, 0, widgetAlignment);
117
118 basicControlsLayout->addLayout(soloMuteLayout, 0, 0, 2, 1, widgetAlignment);
119
120 // set up volume knob
121 m_volumeKnob = new Knob(KnobType::Bright26, nullptr, tr("Sample volume"));
122 m_volumeKnob->setVolumeKnob(true);
123 m_volumeKnob->setHintText(tr("Volume:"), "%");
124
125 basicControlsLayout->addWidget(m_volumeKnob, 0, 1);
126 basicControlsLayout->setAlignment(m_volumeKnob, widgetAlignment);
127
128 auto label = new QLabel(tr("VOL"), this);
129 label->setStyleSheet(labelStyleSheet);
130 basicControlsLayout->addWidget(label, 1, 1);
131 basicControlsLayout->setAlignment(label, labelAlignment);

134 m_panningKnob = new Knob(KnobType::Bright26, nullptr, tr("Panning"));
135 m_panningKnob->setHintText(tr("Panning:"), "");
136
137 basicControlsLayout->addWidget(m_panningKnob, 0, 2);
138 basicControlsLayout->setAlignment(m_panningKnob, widgetAlignment);
139
140 label = new QLabel(tr("PAN"),this);
141 label->setStyleSheet(labelStyleSheet);
142 basicControlsLayout->addWidget(label, 1, 2);
143 basicControlsLayout->setAlignment(label, labelAlignment);
144
145 basicControlsLayout->setColumnStretch(3, 1);
146
147
148 // setup spinbox for selecting Mixer-channel
149 m_mixerChannelNumber = new MixerChannelLcdSpinBox(2, nullptr, tr("Mixer channel"), m_stv);
150
151 basicControlsLayout->addWidget(m_mixerChannelNumber, 0, 4);
152 basicControlsLayout->setAlignment(m_mixerChannelNumber, widgetAlignment);
153
154 label = new QLabel(tr("CHAN"), this);
155 label->setStyleSheet(labelStyleSheet);
156 basicControlsLayout->addWidget(label, 1, 4);
157 basicControlsLayout->setAlignment(label, labelAlignment);
158
159 generalSettingsLayout->addLayout(basicControlsLayout);
160
161 m_effectRack = new EffectRackView(tv->model()->audioBusHandle()->effects());
162 m_effectRack->setFixedSize(EffectRackView::DEFAULT_WIDTH, 242);
163
164 vlayout->addWidget(generalSettingsWidget);
165 vlayout->addWidget(m_effectRack);
166
167
168 setModel(tv->model());
169
170 QMdiSubWindow * subWin = getGUI()->mainWindow()->addWindowedWidget(this);
171 Qt::WindowFlags flags = subWin->windowFlags();
172 flags |= Qt::MSWindowsFixedSizeDialogHint;
173 flags &= ~Qt::WindowMaximizeButtonHint;
174 subWin->setWindowFlags(flags);
175
176 // Hide the Size and Maximize options from the system menu
177 // since the dialog size is fixed.
178 QMenu * systemMenu = subWin->systemMenu();
179 systemMenu->actions().at(2)->setVisible(false); // Size
180 systemMenu->actions().at(4)->setVisible(false); // Maximize
181
182 subWin->setWindowIcon(embed::getIconPixmap("sample_track"));
183 subWin->setFixedSize(subWin->size());
184 subWin->hide();
185}

200void SampleTrackWindow::modelChanged()
201{
202 m_track = castModel<SampleTrack>();
203
204 m_nameLineEdit->setText(m_track->name());
205
206 m_track->disconnect(SIGNAL(nameChanged()), this);
207
208 connect(m_track, SIGNAL(nameChanged()),
209 this, SLOT(updateName()));
210
211 m_volumeKnob->setModel(&m_track->m_volumeModel);
212 m_panningKnob->setModel(&m_track->m_panningModel);
213 m_mixerChannelNumber->setModel(&m_track->m_mixerChannelModel);
214
215 updateName();
216}

221void SampleTrackWindow::updateName()
222{
223 setWindowTitle(m_track->name().length() > 25 ? (m_track->name().left(24) + "...") : m_track->name());
224
225 if(m_nameLineEdit->text() != m_track->name())
226 {
227 m_nameLineEdit->setText(m_track->name());
228 }
229}
