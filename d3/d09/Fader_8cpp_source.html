127void Fader::adjustByDialog()
128{
129 bool ok;
130
131 if (modelIsLinear())
132 {
133 auto maxDB = ampToDbfs(model()->maxValue());
134 const auto currentValue = model()->value() <= 0. ? c_faderMinDb : ampToDbfs(model()->value());
135
136 float enteredValue = QInputDialog::getDouble(this, tr("Set value"),
137 tr("Please enter a new value between %1 and %2:").arg(c_faderMinDb).arg(maxDB),
138 currentValue, c_faderMinDb, maxDB, model()->getDigitCount(), &ok);
139
140 if (ok)
141 {
142 model()->setValue(dbfsToAmp(enteredValue));
143 }
144 return;
145 }
146 else
147 {
148 // The model already is in dB
149 auto minv = model()->minValue() * m_conversionFactor;
150 auto maxv = model()->maxValue() * m_conversionFactor;
151 float enteredValue = QInputDialog::getDouble(this, tr("Set value"),
152 tr("Please enter a new value between %1 and %2:").arg(minv).arg(maxv),
153 model()->getRoundedValue() * m_conversionFactor, minv, maxv, model()->getDigitCount(), &ok);
154
155 if (ok)
156 {
157 model()->setValue(enteredValue / m_conversionFactor);
158 }
159 }
160}

187void Fader::mousePressEvent(QMouseEvent* mouseEvent)
188{
189 if (mouseEvent->button() == Qt::LeftButton &&
190 !(mouseEvent->modifiers() & KBD_COPY_MODIFIER))
191 {
192 AutomatableModel* thisModel = model();
193 if (thisModel)
194 {
195 thisModel->addJournalCheckPoint();
196 thisModel->saveJournallingState(false);
197 }
198
199 const int localY = mouseEvent->y();
200 const auto knobLowerPosY = calculateKnobPosYFromModel();
201 const auto knobUpperPosY = knobLowerPosY - m_knobSize.height();
202
203 const auto clickedOnKnob = localY >= knobUpperPosY && localY <= knobLowerPosY;
204
205 if (clickedOnKnob)
206 {
207 // If the users clicked on the knob we want to compensate for the offset to the center line
208 // of the knob when dealing with mouse move events.
209 // This will make it feel like the users have grabbed the knob where they clicked.
210 const auto knobCenterPos = knobLowerPosY - (m_knobSize.height() / 2);
211 m_knobCenterOffset = localY - knobCenterPos;
212
213 // In this case we also will not call setVolumeByLocalPixelValue, i.e. we do not make any immediate
214 // changes. This should only happen if the users actually move the mouse while grabbing the knob.
215 // This makes the knobs less "jumpy".
216 }
217 else
218 {
219 // If the users did not click on the knob then we assume that the fader knob's center should move to
220 // the position of the click. We do not compensate for any offset.
221 m_knobCenterOffset = 0;
222
223 setVolumeByLocalPixelValue(localY);
224 }
225
226 updateTextFloat();
227 s_textFloat->show();
228
229 mouseEvent->accept();
230 }
231 else
232 {
233 AutomatableModelView::mousePressEvent(mouseEvent);
234 }
235}

277float Fader::determineAdjustmentDelta(const Qt::KeyboardModifiers & modifiers) const
278{
279 if (modifiers == Qt::ShiftModifier)
280 {
281 // The shift is intended to go through the values in very coarse steps as in:
282 // "Shift into overdrive"
283 return 3.f;
284 }
285 else if (modifiers == Qt::ControlModifier)
286 {
287 // The control key gives more control, i.e. it enables more fine-grained adjustments
288 return 0.1f;
289 }
290 else if (modifiers & Qt::AltModifier)
291 {
292 // Work around a Qt bug in conjunction with the scroll wheel and the Alt key
293 return 0.f;
294 }
295
296 return 1.f;
297}

299void Fader::adjustModelByDBDelta(float value)
300{
301 if (modelIsLinear())
302 {
303 const auto modelValue = model()->value();
304
305 if (modelValue <= 0.)
306 {
307 // We are at -inf dB. Do nothing if we user wishes to decrease.
308 if (value > 0)
309 {
310 // Otherwise set the model to the minimum value supported by the fader.
311 model()->setValue(dbfsToAmp(c_faderMinDb));
312 }
313 }
314 else
315 {
316 // We can safely compute the dB value as the value is greater than 0
317 const auto valueInDB = ampToDbfs(modelValue);
318
319 const auto adjustedValue = valueInDB + value;
320
321 model()->setValue(adjustedValue < c_faderMinDb ? 0. : dbfsToAmp(adjustedValue));
322 }
323 }
324 else
325 {
326 const auto adjustedValue = std::clamp(model()->value() + value, model()->minValue(), model()->maxValue());
327
328 model()->setValue(adjustedValue);
329 }
330}

332int Fader::calculateKnobPosYFromModel() const
333{
334 auto* m = model();
335
336 auto const minV = m->minValue();
337 auto const maxV = m->maxValue();
338 auto const value = m->value();
339
340 if (modelIsLinear())
341 {
342 // This method calculates the pixel position where the lower end of
343 // the fader knob should be for the amplification value in the model.
344 //
345 // The following assumes that the model describes an amplification,
346 // i.e. that values are in [0, max] and that 1 is unity, i.e. 0 dbFS.
347
348 auto const distanceToMin = value - minV;
349
350 // Prevent dbFS calculations with zero or negative values
351 if (distanceToMin <= 0)
352 {
353 return height();
354 }
355 else
356 {
357 // Make sure that we do not get values less that the minimum fader dbFS
358 // for the calculations that will follow.
359 auto const actualDb = std::max(c_faderMinDb, ampToDbfs(value));
360
361 const auto scaledRatio = computeScaledRatio(actualDb);
362
363 // This returns results between:
364 // * m_knobSize.height() for a ratio of 1
365 // * height() for a ratio of 0
366 return height() - (height() - m_knobSize.height()) * scaledRatio;
367 }
368 }
369 else
370 {
371 // The model is in dB so we just show that in a linear fashion
372
373 auto const clampedValue = std::clamp(value, minV, maxV);
374
375 auto const ratio = (clampedValue - minV) / (maxV - minV);
376
377 // This returns results between:
378 // * m_knobSize.height() for a ratio of 1
379 // * height() for a ratio of 0
380 return height() - (height() - m_knobSize.height()) * ratio;
381 }
382}

385void Fader::setVolumeByLocalPixelValue(int y)
386{
387 auto* m = model();
388
389 // Compensate the offset where users have actually clicked
390 y -= m_knobCenterOffset;
391
392 // The y parameter gives us where the mouse click went.
393 // Assume that the middle of the fader should go there.
394 int const lowerFaderKnob = y + (m_knobSize.height() / 2);
395
396 // In some cases we need the clamped lower position of the fader knob so we can ensure
397 // that we only set allowed values in the model range.
398 int const clampedLowerFaderKnob = std::clamp(lowerFaderKnob, m_knobSize.height(), height());
399
400 if (modelIsLinear())
401 {
402 if (lowerFaderKnob >= height())
403 {
404 // Check the non-clamped value because otherwise we wouldn't be able to set -inf dB!
405 model()->setValue(0);
406 }
407 else
408 {
409 // We are in the case where we set a value that's different from -inf dB so we use the clamped value
410 // of the lower knob position so that we only set allowed values in the model range.
411
412 // First map the lower knob position to [0, 1] so that we can apply some curve mapping, e.g.
413 // square, cube, etc.
414 LinearMap<float> knobMap(float(m_knobSize.height()), 1., float(height()), 0.);
415
416 // Apply the inverse of what is done in calculateKnobPosYFromModel
417 auto const knobPos = std::pow(knobMap.map(clampedLowerFaderKnob), 1./c_dBScalingExponent);
418
419 float const maxDb = ampToDbfs(m->maxValue());
420
421 LinearMap<float> dbMap(1., maxDb, 0., c_faderMinDb);
422
423 float const dbValue = dbMap.map(knobPos);
424
425 // Pull everything that's quieter than the minimum fader dbFS value down to 0 amplification.
426 // This should not happen due to the steps above but let's be sure.
427 // Otherwise compute the amplification value from the mapped dbFS value but make sure that we
428 // do not exceed the maximum dbValue of the model
429 float ampValue = dbValue < c_faderMinDb ? 0. : dbfsToAmp(std::min(maxDb, dbValue));
430
431 model()->setValue(ampValue);
432 }
433 }
434 else
435 {
436 LinearMap<float> valueMap(float(m_knobSize.height()), model()->maxValue(), float(height()), model()->minValue());
437
438 model()->setValue(valueMap.map(clampedLowerFaderKnob));
439 }
440}
441
442float Fader::computeScaledRatio(float dBValue) const
443{
444 const auto maxDb = ampToDbfs(model()->maxValue());
445
446 const auto ratio = (dBValue - c_faderMinDb) / (maxDb - c_faderMinDb);
447
448 return std::pow(ratio, c_dBScalingExponent);
449}

453/// Set peak value (0.0 .. 1.0)
454///
455void Fader::setPeak(float fPeak, float& targetPeak, float& persistentPeak, QElapsedTimer& lastPeakTimer)
456{
457 if (targetPeak != fPeak)
458 {
459 targetPeak = fPeak;
460 if (targetPeak >= persistentPeak)
461 {
462 persistentPeak = targetPeak;
463 lastPeakTimer.restart();
464 emit peakChanged(persistentPeak);
465 }
466 update();
467 }
468
469 if (persistentPeak > 0 && lastPeakTimer.elapsed() > 1500)
470 {
471 persistentPeak = qMax<float>(0, persistentPeak-0.05);
472 emit peakChanged(persistentPeak);
473 update();
474 }
475}

513QString Fader::getModelValueAsDbString() const
514{
515 QString label(tr("Volume: %1 dB"));
516 // Check that the pointer isn't dangling, which can happen if the
517 // model was dropped in order to load a new project from an existing one.
518 auto* newModel = model();
519 if (!newModel)
520 {
521 // model() was a dangling pointer, so return a sane default value.
522 return label.arg(tr("-inf"));
523 }
524 const auto value = newModel->value();
525
526 if (modelIsLinear())
527 {
528 if (value <= 0.)
529 {
530 return label.arg(tr("-inf"));
531 }
532 else
533 {
534 return label.arg(ampToDbfs(value), 3, 'f', 2);
535 }
536 }
537 else
538 {
539 return label.arg(value, 3, 'f', 2);
540 }

559void Fader::paintLevels(QPaintEvent* ev, QPainter& painter, bool linear)
560{
561 const auto mapper = linear
562 ? +[](float value) -> float { return value; }
563 : +[](float value) -> float { return ampToDbfs(qMax(0.0001f, value)); };
564
565 const float mappedMinPeak = mapper(m_fMinPeak);
566 const float mappedMaxPeak = mapper(m_fMaxPeak);
567 const float mappedPeakL = mapper(m_fPeakValue_L);
568 const float mappedPeakR = mapper(m_fPeakValue_R);
569 const float mappedPersistentPeakL = mapper(m_persistentPeak_L);
570 const float mappedPersistentPeakR = mapper(m_persistentPeak_R);
571 const float mappedUnity = mapper(1.f);
572
573 painter.save();
574
575 const QRect baseRect = rect();
576
577 const int height = baseRect.height();
578
579 const int margin = 1;
580 const int distanceBetweenMeters = 2;
581
582 const int numberOfMeters = 2;
583
584 // Compute the width of a single meter by removing the margins and the space between meters
585 const int leftAndRightMargin = 2 * margin;
586 const int pixelsBetweenAllMeters = distanceBetweenMeters * (numberOfMeters - 1);
587 const int remainingPixelsForMeters = baseRect.width() - leftAndRightMargin - pixelsBetweenAllMeters;
588 const int meterWidth = remainingPixelsForMeters / numberOfMeters;
589
590 QRect leftMeterOutlineRect(margin, margin, meterWidth, height - 2 * margin);
591 QRect rightMeterOutlineRect(baseRect.width() - margin - meterWidth, margin, meterWidth, height - 2 * margin);
592
593 QMargins removedMargins(1, 1, 1, 1);
594 QRect leftMeterRect = leftMeterOutlineRect.marginsRemoved(removedMargins);
595 QRect rightMeterRect = rightMeterOutlineRect.marginsRemoved(removedMargins);
596
597 QPainterPath path;
598 qreal radius = 2;
599 path.addRoundedRect(leftMeterOutlineRect, radius, radius);
600 path.addRoundedRect(rightMeterOutlineRect, radius, radius);
601 painter.fillPath(path, Qt::black);
602
603 // Now clip everything to the paths of the meters
604 painter.setClipPath(path);
605
606 // This linear map performs the following mapping:
607 // Value (dbFS or linear) -> window coordinates of the widget
608 // It is for example used to determine the height of peaks, markers and to define the gradient for the levels
609 const LinearMap<float> valuesToWindowCoordinates(mappedMaxPeak, leftMeterRect.y(), mappedMinPeak, leftMeterRect.y() + leftMeterRect.height());
610
611 // This lambda takes a value (in dbFS or linear) and a rectangle and computes a rectangle
612 // that represent the value within the rectangle. It is for example used to compute the unity indicators.
613 const auto computeLevelMarkerRect = [&valuesToWindowCoordinates](const QRect& rect, float peak) -> QRect
614 {
615 return QRect(rect.x(), valuesToWindowCoordinates.map(peak), rect.width(), 1);
616 };
617
618 // This lambda takes a peak value (in dbFS or linear) and a rectangle and computes a rectangle
619 // that represent the peak value within the rectangle. It's used to compute the peak indicators
620 // which "dance" on top of the level meters.
621 const auto computePeakRect = [&valuesToWindowCoordinates](const QRect& rect, float peak) -> QRect
622 {
623 return QRect(rect.x(), valuesToWindowCoordinates.map(peak), rect.width(), 1);
624 };
625
626 // This lambda takes a peak value (in dbFS or linear) and a rectangle and returns an adjusted copy of the
627 // rectangle that represents the peak value. It is used to compute the level meters themselves.
628 const auto computeLevelRect = [&valuesToWindowCoordinates](const QRect& rect, float peak) -> QRect
629 {
630 QRect result(rect);
631 result.setTop(valuesToWindowCoordinates.map(peak));
632
633 return result;
634 };
635
636 // Draw left and right level markers for the unity lines (0 dbFS, 1.0 amplitude)
637 if (getRenderUnityLine())
638 {
639 const auto unityRectL = computeLevelMarkerRect(leftMeterRect, mappedUnity);
640 painter.fillRect(unityRectL, m_unityMarker);
641
642 const auto unityRectR = computeLevelMarkerRect(rightMeterRect, mappedUnity);
643 painter.fillRect(unityRectR, m_unityMarker);
644 }
645
646 // These values define where the gradient changes values, i.e. the ranges
647 // for clipping, warning and ok.
648 // Please ensure that "clip starts" is the maximum value and that "ok ends"
649 // is the minimum value and that all other values lie inbetween. Otherwise
650 // there will be warnings when the gradient is defined.
651 const float mappedClipStarts = mapper(dbfsToAmp(0.f));
652 const float mappedWarnEnd = mapper(dbfsToAmp(-0.01f));
653 const float mappedWarnStart = mapper(dbfsToAmp(-6.f));
654 const float mappedOkEnd = mapper(dbfsToAmp(-12.f));
655
656 // Prepare the gradient for the meters
657 //
658 // The idea is the following. We want to be able to render arbitrary ranges of min and max values.
659 // Therefore we first compute the start and end point of the gradient in window coordinates.
660 // The gradient is assumed to start with the clip color and to end with the ok color with warning values in between.
661 // We know the min and max peaks that map to a rectangle where we draw the levels. We can use the values of the min and max peaks
662 // as well as the Y-coordinates of the rectangle to compute a map which will give us the coordinates of the value where the clipping
663 // starts and where the ok area end. These coordinates are used to initialize the gradient. Please note that the gradient might thus
664 // extend the rectangle into which we paint.
665 float clipStartYCoord = valuesToWindowCoordinates.map(mappedClipStarts);
666 float okEndYCoord = valuesToWindowCoordinates.map(mappedOkEnd);
667
668 QLinearGradient linearGrad(0, clipStartYCoord, 0, okEndYCoord);
669
670 // We already know for the gradient that the clip color will be at 0 and that the ok color is at 1.
671 // What's left to do is to map the inbetween values into the interval [0,1].
672 const LinearMap<float> mapBetweenClipAndOk(mappedClipStarts, 0.f, mappedOkEnd, 1.f);
673
674 linearGrad.setColorAt(0, m_peakClip);
675 linearGrad.setColorAt(mapBetweenClipAndOk.map(mappedWarnEnd), m_peakWarn);
676 linearGrad.setColorAt(mapBetweenClipAndOk.map(mappedWarnStart), m_peakWarn);
677 linearGrad.setColorAt(1, m_peakOk);
678
679 // Draw left levels
680 if (mappedPeakL > mappedMinPeak)
681 {
682 QPainterPath leftMeterPath;
683 leftMeterPath.addRoundedRect(computeLevelRect(leftMeterRect, mappedPeakL), radius, radius);
684 painter.fillPath(leftMeterPath, linearGrad);
685 }
686
687 // Draw left peaks
688 if (mappedPersistentPeakL > mappedMinPeak)
689 {
690 const auto peakRectL = computePeakRect(leftMeterRect, mappedPersistentPeakL);
691 painter.fillRect(peakRectL, linearGrad);
692 }
693
694 // Draw right levels
695 if (mappedPeakR > mappedMinPeak)
696 {
697 QPainterPath rightMeterPath;
698 rightMeterPath.addRoundedRect(computeLevelRect(rightMeterRect, mappedPeakR), radius, radius);
699 painter.fillPath(rightMeterPath, linearGrad);
700 }
701
702 // Draw right peaks
703 if (mappedPersistentPeakR > mappedMinPeak)
704 {
705 const auto peakRectR = computePeakRect(rightMeterRect, mappedPersistentPeakR);
706 painter.fillRect(peakRectR, linearGrad);
707 }
708
709 painter.restore();
710}

712void Fader::paintFaderTicks(QPainter& painter)
713{
714 painter.save();
715
716 const QPen zeroPen(QColor(255, 255, 255, 216), 2.5);
717 const QPen nonZeroPen(QColor(255, 255, 255, 128), 1.);
718
719 // We use the maximum dB value of the model to calculate the nearest multiple
720 // of the step size that we use to paint the ticks so that we know the start point.
721 // This code will paint ticks with steps that are defined by the step size around
722 // the 0 dB marker.
723 const auto maxDB = ampToDbfs(model()->maxValue());
724 const auto stepSize = 6.f;
725 const auto startValue = std::floor(maxDB / stepSize) * stepSize;
726
727 for (float i = startValue; i >= c_faderMinDb; i-= stepSize)
728 {
729 const auto scaledRatio = computeScaledRatio(i);
730 const auto maxHeight = height() - (height() - m_knobSize.height()) * scaledRatio - (m_knobSize.height() / 2);
731
732 if (approximatelyEqual(i, 0.))
733 {
734 painter.setPen(zeroPen);
735 }
736 else
737 {
738 painter.setPen(nonZeroPen);
739 }
740
741 painter.drawLine(QPointF(0, maxHeight), QPointF(1, maxHeight));
742 painter.drawLine(QPointF(width() - 1, maxHeight), QPointF(width(), maxHeight));
743 }
744
745 painter.restore();
746}
