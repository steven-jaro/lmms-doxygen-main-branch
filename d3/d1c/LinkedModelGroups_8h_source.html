140/**
141 Container for a group of linked models
142
143 Each group contains the same models and model types. The models are
144 numbered, and equal numbered models are associated and always linked.
145
146 A typical application are two mono plugins making a stereo plugin.
147
148 @note Though this class can contain multiple model groups, a corresponding
149 view ("LinkedModelGroupViews") will only display one group, as they all have
150 the same values
151
152 @note Though called "container", this class does not contain, but only
153 know the single groups. The inheriting classes are responsible for storage.
154*/
155class LinkedModelGroups
156{
157public:
158 virtual ~LinkedModelGroups() = default;
159
160 void linkAllModels();
161
162 /*
163 Load/Save
164 */
165 void saveSettings(class QDomDocument& doc, class QDomElement& that);
166 void loadSettings(const class QDomElement& that);
167
168 /*
169 General
170 */
171 //! Derived classes must return the group with index @p idx,
172 //! or nullptr if @p is out of range
173 virtual LinkedModelGroup* getGroup(std::size_t idx) = 0;
174 //! @see getGroup
175 virtual const LinkedModelGroup* getGroup(std::size_t idx) const = 0;
176};
