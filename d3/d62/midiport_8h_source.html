60/** Get a MIDI event from an event port buffer.
61 *
62 * Jack MIDI is normalised, the MIDI event returned by this function is
63 * guaranteed to be a complete MIDI event (the status byte will always be
64 * present, and no realtime events will interspered with the event).
65 *
66 * This rule does not apply to System Exclusive MIDI messages
67 * since they can be of arbitrary length.
68 * To maintain smooth realtime operation such events CAN be deliverd
69 * as multiple, non-normalised events.
70 * The maximum size of one event "chunk" depends on the MIDI backend in use.
71 * For example the midiseq driver will create chunks of 256 bytes.
72 * The first SysEx "chunked" event starts with 0xF0 and the last
73 * delivered chunk ends with 0xF7.
74 * To receive the full SysEx message, a caller of jack_midi_event_get()
75 * must concatenate chunks until a chunk ends with 0xF7.
76 *
77 * @param event Event structure to store retrieved event in.
78 * @param port_buffer Port buffer from which to retrieve event.
79 * @param event_index Index of event to retrieve.
80 * @return 0 on success, ENODATA if buffer is empty.
81 */
82int
83jack_midi_event_get(jack_midi_event_t *event,
84 void *port_buffer,
85 uint32_t event_index) JACK_OPTIONAL_WEAK_EXPORT;

124/** Allocate space for an event to be written to an event port buffer.
125 *
126 * Clients are to write the actual event data to be written starting at the
127 * pointer returned by this function. Clients must not write more than
128 * @a data_size bytes into this buffer. Clients must write normalised
129 * MIDI data to the port - no running status and no (1-byte) realtime
130 * messages interspersed with other messages (realtime messages are fine
131 * when they occur on their own, like other messages).
132 *
133 * Events must be written in order, sorted by their sample offsets.
134 * JACK will not sort the events for you, and will refuse to store
135 * out-of-order events.
136 *
137 * @param port_buffer Buffer to write event to.
138 * @param time Sample offset of event.
139 * @param data_size Length of event's raw data in bytes.
140 * @return Pointer to the beginning of the reserved event's data buffer, or
141 * NULL on error (ie not enough space).
142 */
143jack_midi_data_t*
144jack_midi_event_reserve(void *port_buffer,
145 jack_nframes_t time,
146 size_t data_size) JACK_OPTIONAL_WEAK_EXPORT;

149/** Write an event into an event port buffer.
150 *
151 * This function is simply a wrapper for @ref jack_midi_event_reserve
152 * which writes the event data into the space reserved in the buffer.
153 *
154 * Clients must not write more than
155 * @a data_size bytes into this buffer. Clients must write normalised
156 * MIDI data to the port - no running status and no (1-byte) realtime
157 * messages interspersed with other messages (realtime messages are fine
158 * when they occur on their own, like other messages).
159 *
160 * Events must be written in order, sorted by their sample offsets.
161 * JACK will not sort the events for you, and will refuse to store
162 * out-of-order events.
163 *
164 * @param port_buffer Buffer to write event to.
165 * @param time Sample offset of event.
166 * @param data Message data to be written.
167 * @param data_size Length of @a data in bytes.
168 * @return 0 on success, ENOBUFS if there's not enough space in buffer for event.
169 */
170int
171jack_midi_event_write(void *port_buffer,
172 jack_nframes_t time,
173 const jack_midi_data_t *data,
174 size_t data_size) JACK_OPTIONAL_WEAK_EXPORT;
