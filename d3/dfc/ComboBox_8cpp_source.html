101void ComboBox::mousePressEvent( QMouseEvent* event )
102{
103 if( model() == nullptr )
104 {
105 return;
106 }
107
108 if( event->button() == Qt::LeftButton && ! ( event->modifiers() & Qt::ControlModifier ) )
109 {
110 if( event->x() > width() - CB_ARROW_BTN_WIDTH )
111 {
112 m_pressed = true;
113 update();
114
115 m_menu.clear();
116 for( int i = 0; i < model()->size(); ++i )
117 {
118 QAction * a = m_menu.addAction( model()->itemPixmap( i ) ? model()->itemPixmap( i )->pixmap() : QPixmap(),
119 model()->itemText( i ) );
120 a->setData( i );
121 }
122
123 QPoint gpos = mapToGlobal(QPoint(0, height()));
124
125 #if (QT_SUPPORTS_WIDGET_SCREEN)
126 bool const menuCanBeFullyShown = screen()->geometry().contains(QRect(gpos, m_menu.sizeHint()));
127 #else
128 bool const menuCanBeFullyShown = gpos.y() + m_menu.sizeHint().height() < qApp->desktop()->height();
129 #endif
130
131 if (menuCanBeFullyShown)
132 {
133 m_menu.exec(gpos);
134 }
135 else
136 {
137 m_menu.exec(mapToGlobal(QPoint(width(), 0)));
138 }
139
140 m_pressed = false;
141 update();
142 }
143 else if( event->button() == Qt::LeftButton )
144 {
145 selectNext();
146 update();
147 }
148 }
149 else if( event->button() == Qt::RightButton )
150 {
151 selectPrevious();
152 update();
153 }
154 else
155 {
156 IntModelView::mousePressEvent( event );
157 }
158}

163void ComboBox::paintEvent( QPaintEvent * _pe )
164{
165 QPainter p( this );
166
167 p.fillRect(2, 2, width() - 2, height() - 4, m_background);
168
169 QColor shadow = palette().shadow().color();
170 QColor highlight = palette().highlight().color();
171
172 shadow.setAlpha( 124 );
173 highlight.setAlpha( 124 );
174
175 // button-separator
176 p.setPen( shadow );
177 p.drawLine( width() - CB_ARROW_BTN_WIDTH - 1, 1, width() - CB_ARROW_BTN_WIDTH - 1, height() - 3 );
178
179 p.setPen( highlight );
180 p.drawLine( width() - CB_ARROW_BTN_WIDTH, 1, width() - CB_ARROW_BTN_WIDTH, height() - 3 );
181
182 // Border
183 QStyleOptionFrame opt;
184 opt.initFrom( this );
185 opt.state = QStyle::StateFlag::State_None;
186
187 style()->drawPrimitive( QStyle::PE_Frame, &opt, &p, this );
188
189 auto arrow = m_pressed ? m_arrowSelected : m_arrow;
190
191 p.drawPixmap(width() - CB_ARROW_BTN_WIDTH + 3, 4, arrow);
192
193 if( model() && model()->size() > 0 )
194 {
195 p.setFont( font() );
196 p.setClipRect( QRect( 4, 2, width() - CB_ARROW_BTN_WIDTH - 8, height() - 2 ) );
197 QPixmap pm = model()->currentData() ? model()->currentData()->pixmap() : QPixmap();
198 int tx = 5;
199 if( !pm.isNull() )
200 {
201 if( pm.height() > 16 )
202 {
203 pm = pm.scaledToHeight( 16, Qt::SmoothTransformation );
204 }
205 p.drawPixmap( tx, 3, pm );
206 tx += pm.width() + 3;
207 }
208 const int y = ( height()+p.fontMetrics().height() ) /2;
209 p.setPen( QColor( 64, 64, 64 ) );
210 p.drawText( tx+1, y-3, model()->currentText() );
211 p.setPen( QColor( 224, 224, 224 ) );
212 p.drawText( tx, y-4, model()->currentText() );
213 }
214}

219void ComboBox::wheelEvent( QWheelEvent* event )
220{
221 if( model() )
222 {
223 const int direction = (event->angleDelta().y() < 0 ? 1 : -1) * (event->inverted() ? -1 : 1);
224 model()->setValue(model()->value() + direction);
225 update();
226 event->accept();
227 }
228}
