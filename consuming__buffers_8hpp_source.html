52// A buffer sequence used to represent a subsequence of the buffers.
53template <typename Buffer, std::size_t MaxBuffers>
54struct prepared_buffers
55{
56 typedef Buffer value_type;
57 typedef const Buffer* const_iterator;
58
59 enum { max_buffers = MaxBuffers < 16 ? MaxBuffers : 16 };
60
61 prepared_buffers() : count(0) {}
62 const_iterator begin() const { return elems; }
63 const_iterator end() const { return elems + count; }
64
65 Buffer elems[max_buffers];
66 std::size_t count;
67};

69// A proxy for a sub-range in a list of buffers.
70template <typename Buffer, typename Buffers, typename Buffer_Iterator>
71class consuming_buffers
72{
73public:
74 typedef prepared_buffers<Buffer, prepared_buffers_max<Buffers>::value>
75 prepared_buffers_type;
76
77 // Construct to represent the entire list of buffers.
78 explicit consuming_buffers(const Buffers& buffers)
79 : buffers_(buffers),
80 total_consumed_(0),
81 next_elem_(0),
82 next_elem_offset_(0)
83 {
84 using asio::buffer_size;
85 total_size_ = buffer_size(buffers);
86 }
87
88 // Determine if we are at the end of the buffers.
89 bool empty() const
90 {
91 return total_consumed_ >= total_size_;
92 }
93
94 // Get the buffer for a single transfer, with a size.
95 prepared_buffers_type prepare(std::size_t max_size)
96 {
97 prepared_buffers_type result;
98
99 Buffer_Iterator next = asio::buffer_sequence_begin(buffers_);
100 Buffer_Iterator end = asio::buffer_sequence_end(buffers_);
101
102 std::advance(next, next_elem_);
103 std::size_t elem_offset = next_elem_offset_;
104 while (next != end && max_size > 0 && (result.count) < result.max_buffers)
105 {
106 Buffer next_buf = Buffer(*next) + elem_offset;
107 result.elems[result.count] = asio::buffer(next_buf, max_size);
108 max_size -= result.elems[result.count].size();
109 elem_offset = 0;
110 if (result.elems[result.count].size() > 0)
111 ++result.count;
112 ++next;
113 }
114
115 return result;
116 }
117
118 // Consume the specified number of bytes from the buffers.
119 void consume(std::size_t size)
120 {
121 total_consumed_ += size;
122
123 Buffer_Iterator next = asio::buffer_sequence_begin(buffers_);
124 Buffer_Iterator end = asio::buffer_sequence_end(buffers_);
125
126 std::advance(next, next_elem_);
127 while (next != end && size > 0)
128 {
129 Buffer next_buf = Buffer(*next) + next_elem_offset_;
130 if (size < next_buf.size())
131 {
132 next_elem_offset_ += size;
133 size = 0;
134 }
135 else
136 {
137 size -= next_buf.size();
138 next_elem_offset_ = 0;
139 ++next_elem_;
140 ++next;
141 }
142 }
143 }
144
145 // Get the total number of bytes consumed from the buffers.
146 std::size_t total_consumed() const
147 {
148 return total_consumed_;
149 }
150
151private:
152 Buffers buffers_;
153 std::size_t total_size_;
154 std::size_t total_consumed_;
155 std::size_t next_elem_;
156 std::size_t next_elem_offset_;
157};

272template <typename Buffer, typename Elem>
273class consuming_buffers<Buffer, boost::array<Elem, 2>,
274 typename boost::array<Elem, 2>::const_iterator>
275{
276public:
277 // Construct to represent the entire list of buffers.
278 explicit consuming_buffers(const boost::array<Elem, 2>& buffers)
279 : buffers_(buffers),
280 total_consumed_(0)
281 {
282 }
283
284 // Determine if we are at the end of the buffers.
285 bool empty() const
286 {
287 return total_consumed_ >=
288 Buffer(buffers_[0]).size() + Buffer(buffers_[1]).size();
289 }
290
291 // Get the buffer for a single transfer, with a size.
292 boost::array<Buffer, 2> prepare(std::size_t max_size)
293 {
294 boost::array<Buffer, 2> result = {{
295 Buffer(buffers_[0]), Buffer(buffers_[1]) }};
296 std::size_t buffer0_size = result[0].size();
297 result[0] = asio::buffer(result[0] + total_consumed_, max_size);
298 result[1] = asio::buffer(
299 result[1] + (total_consumed_ < buffer0_size
300 ? 0 : total_consumed_ - buffer0_size),
301 max_size - result[0].size());
302 return result;
303 }
304
305 // Consume the specified number of bytes from the buffers.
306 void consume(std::size_t size)
307 {
308 total_consumed_ += size;
309 }
310
311 // Get the total number of bytes consumed from the buffers.
312 std::size_t total_consumed() const
313 {
314 return total_consumed_;
315 }
316
317private:
318 boost::array<Elem, 2> buffers_;
319 std::size_t total_consumed_;
320};
