124inline float positiveFraction(float x)
125{
126 if (std::isnan(x) || std::isinf(x))
127 return 0;
128 if (x<0)
129 {
130 x+=static_cast<int>(1-x);
131 }
132 return x-static_cast<int>(x);
133}
