52 float smoothParameter(float in, int axis) {
53 sZ[axis] = (in * sB[axis]) + (sZ[axis] * sA[axis]);
54 return sZ[axis];
55 }

57 float getSinePhase(float x) {
58 return (-std::sin(x));
59 }
60 float getSawPhase(float x) {
61 return (-(2/M_PI *std::atan(1/std::tan(x/2))));
62 }
63 float getRevSawPhase(float x) {
64 return ((2/M_PI *std::atan(1/std::tan(x/2))));
65 }
66 float getSquarePhase(float x) {
67 return (std::round((std::sin(x)+1)/2)-0.5)*2;
68 }
69
70 //saw, sqr, sin, revSaw
71 float getBlendedPhase(float x, float wave)
72 {
73 //wave = 2;
74 if (wave>=1 && wave<2) {
75 /* saw vs sqr */
76 waveBlend = wave-1;
77 return (getSawPhase(x)*(1-waveBlend) + getSquarePhase(x)*waveBlend);
78 } else if (wave>=2 && wave<3) {
79 /* sqr vs sin */
80 waveBlend = wave-2;
81 return (getSquarePhase(x)*(1-waveBlend) + getSinePhase(x)*waveBlend);
82 } else if (wave>=3 && wave<=4) {
83 /* sin vs revSaw */
84 waveBlend = wave-3;
85 return (getSinePhase(x)*(1-waveBlend) + getRevSawPhase(x)*waveBlend);
86 } else {
87 return 0.0f;
88 }
89 }

91 float tN(float x)
92 {
93 if (x>0) return x;
94 else return 0;
95 }

97 void animate()
98 {
99 //sync orbit with frame, bpm
100 const TimePosition& time(getTimePosition());
101 bar = ((120.0/(time.bbt.valid ? time.bbt.beatsPerMinute : 120.0))*(getSampleRate()));
102
103 int multiplier = 16;//2000*4;
104 tickX = bar/(std::round(orbitSpeedX))*multiplier;
105 tickY = bar/(std::round(orbitSpeedY))*multiplier;
106 tick = bar/(std::round(subOrbitSpeed))*multiplier;
107
108 if (time.playing)
109 {
110 /* if rolling then sync to timepos */
111 tickOffsetX = time.frame-std::floor(time.frame/tickX)*tickX;
112 tickOffsetY = time.frame-std::floor(time.frame/tickY)*tickY;
113 tickOffset = time.frame-std::floor(time.frame/tick)*tick;
114 percentageX = tickOffsetX/tickX;
115 percentageY = tickOffsetY/tickY;
116 percentage = tickOffset/tick;
117 sinePosX = (M_PI*2)*percentageX;
118 sinePosY = (M_PI*2)*percentageY;
119 sinePos = (M_PI*2)*percentage;
120 } else {
121 /* else just keep on wobblin' */
122 sinePosX += (2*M_PI)/(tickX);
123 sinePosY += (2*M_PI)/(tickY);
124 sinePos += (M_PI)/(tick);
125 if (sinePosX>2*M_PI) {
126 sinePosX = 0;
127 }
128 if (sinePosY>2*M_PI) {
129 sinePosY = 0;
130 }
131 if (sinePos>2*M_PI) {
132 sinePos = 0;
133 }
134 }
135
136 //0..1
137 //0..3
138 //0, 1, 2, 3
139 //* 0.25
140 //0, 0.25, 0.5, 0.75
141
142 float tempPhaseX = std::round(orbitPhaseX)*0.25-0.25;
143 float tempPhaseY = std::round(orbitPhaseY)*0.25-0.25;
144
145 orbitX = x+getBlendedPhase(sinePosX + tempPhaseX*(2*M_PI), std::round(orbitWaveX))*(orbitSizeX/2);
146 orbitY = y+getBlendedPhase(sinePosY+M_PI/2 + tempPhaseY*(2*M_PI), std::round(orbitWaveY))*(orbitSizeY/2);
147 subOrbitX = smoothParameter(orbitX+getBlendedPhase(sinePos, 3)*(subOrbitSize/3), 0);
148 subOrbitY = smoothParameter(orbitY+getBlendedPhase(sinePos+M_PI/2, 3)*(subOrbitSize/3), 1);
149 if (orbitX<0) orbitX=0;
150 if (orbitX>1) orbitX=1;
151 if (orbitY<0) orbitY=0;
152 if (orbitY>1) orbitY=1;
153
154 if (subOrbitX<0) subOrbitX=0;
155 if (subOrbitX>1) subOrbitX=1;
156 if (subOrbitY<0) subOrbitY=0;
157 if (subOrbitY>1) subOrbitY=1;
158 }
