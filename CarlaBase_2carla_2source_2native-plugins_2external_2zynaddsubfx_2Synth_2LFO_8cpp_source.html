24LFO::LFO(const LFOParams &lfopars, float basefreq, const AbsTime &t, WatchManager *m,
25 const char *watch_prefix)
26 :first_half(-1),
27 delayTime(t, lfopars.Pdelay / 127.0f * 4.0f), //0..4 sec
28 waveShape(lfopars.PLFOtype),
29 deterministic(!lfopars.Pfreqrand),
30 dt_(t.dt()),
31 lfopars_(lfopars), basefreq_(basefreq),
32 watchOut(m, watch_prefix, "out")
33{
34 int stretch = lfopars.Pstretch;
35 if(stretch == 0)
36 stretch = 1;
37
38 //max 2x/octave
39 const float lfostretch = powf(basefreq / 440.0f, (stretch - 64.0f) / 63.0f);
40
41 const float lfofreq =
42 (powf(2, lfopars.Pfreq * 10.0f) - 1.0f) / 12.0f * lfostretch;
43 phaseInc = fabs(lfofreq) * t.dt();
44
45 if(!lfopars.Pcontinous) {
46 if(lfopars.Pstartphase == 0)
47 phase = RND;
48 else
49 phase = fmod((lfopars.Pstartphase - 64.0f) / 127.0f + 1.0f, 1.0f);
50 }
51 else {
52 const float tmp = fmod(t.time() * phaseInc, 1.0f);
53 phase = fmod((lfopars.Pstartphase - 64.0f) / 127.0f + 1.0f + tmp, 1.0f);
54 }
55
56 //Limit the Frequency(or else...)
57 if(phaseInc > 0.49999999f)
58 phaseInc = 0.499999999f;
59
60
61 lfornd = limit(lfopars.Prandomness / 127.0f, 0.0f, 1.0f);
62 lfofreqrnd = powf(lfopars.Pfreqrand / 127.0f, 2.0f) * 4.0f;
63
64 switch(lfopars.fel) {
65 case 1:
66 lfointensity = lfopars.Pintensity / 127.0f;
67 break;
68 case 2:
69 lfointensity = lfopars.Pintensity / 127.0f * 4.0f;
70 break; //in octave
71 default:
72 lfointensity = powf(2, lfopars.Pintensity / 127.0f * 11.0f) - 1.0f; //in centi
73 phase -= 0.25f; //chance the starting phase
74 break;
75 }
76
77 amp1 = (1 - lfornd) + lfornd * RND;
78 amp2 = (1 - lfornd) + lfornd * RND;
79 incrnd = nextincrnd = 1.0f;
80 computeNextFreqRnd();
81 computeNextFreqRnd(); //twice because I want incrnd & nextincrnd to be random
82}

87float LFO::baseOut(const char waveShape, const float phase)
88{
89 switch(waveShape) {
90 case LFO_TRIANGLE:
91 if(phase >= 0.0f && phase < 0.25f)
92 return 4.0f * phase;
93 else if(phase > 0.25f && phase < 0.75f)
94 return 2 - 4 * phase;
95 else
96 return 4.0f * phase - 4.0f;
97 break;
98 case LFO_SQUARE:
99 if(phase < 0.5f)
100 return -1;
101 else
102 return 1;
103 break;
104 case LFO_RAMPUP: return (phase - 0.5f) * 2.0f;
105 case LFO_RAMPDOWN: return (0.5f - phase) * 2.0f;
106 case LFO_EXP_DOWN1: return powf(0.05f, phase) * 2.0f - 1.0f;
107 case LFO_EXP_DOWN2: return powf(0.001f, phase) * 2.0f - 1.0f;
108 case LFO_RANDOM:
109 if ((phase < 0.5) != first_half) {
110 first_half = phase < 0.5;
111 last_random = 2*RND-1;
112 }
113 return last_random;
114 default: return cosf(phase * 2.0f * PI); //LFO_SINE
115 }
116}

119float LFO::lfoout()
120{
121 //update internals XXX TODO cleanup
122 if ( ! lfopars_.time || lfopars_.last_update_timestamp == lfopars_.time->time())
123 {
124 waveShape = lfopars_.PLFOtype;
125 int stretch = lfopars_.Pstretch;
126 if(stretch == 0)
127 stretch = 1;
128 const float lfostretch = powf(basefreq_ / 440.0f, (stretch - 64.0f) / 63.0f);
129
130 float lfofreq =
131 (powf(2, lfopars_.Pfreq * 10.0f) - 1.0f) / 12.0f * lfostretch;
132
133 phaseInc = fabs(lfofreq) * dt_;
134
135 switch(lfopars_.fel) {
136 case 1:
137 lfointensity = lfopars_.Pintensity / 127.0f;
138 break;
139 case 2:
140 lfointensity = lfopars_.Pintensity / 127.0f * 4.0f;
141 break; //in octave
142 default:
143 lfointensity = powf(2, lfopars_.Pintensity / 127.0f * 11.0f) - 1.0f; //in centi
144 //x -= 0.25f; //chance the starting phase
145 break;
146 }
147 }
148
149 float out = baseOut(waveShape, phase);
150
151 if(waveShape == LFO_SINE || waveShape == LFO_TRIANGLE)
152 out *= lfointensity * (amp1 + phase * (amp2 - amp1));
153 else
154 out *= lfointensity * amp2;
155
156 if(delayTime.inFuture())
157 return out;
158
159 //Start oscillating
160 if(deterministic)
161 phase += phaseInc;
162 else {
163 const float tmp = (incrnd * (1.0f - phase) + nextincrnd * phase);
164 phase += phaseInc * limit(tmp, 0.0f, 1.0f);
165 }
166 if(phase >= 1) {
167 phase = fmod(phase, 1.0f);
168 amp1 = amp2;
169 amp2 = (1 - lfornd) + lfornd * RND;
170
171 computeNextFreqRnd();
172 }
173
174 float watch_data[2] = {phase, out};
175 watchOut(watch_data, 2);
176
177 return out;
178}

189void LFO::computeNextFreqRnd()
190{
191 if(deterministic)
192 return;
193 incrnd = nextincrnd;
194 nextincrnd = powf(0.5f, lfofreqrnd) + RND * (powf(2.0f, lfofreqrnd) - 1.0f);
195}
