38PatternTrackView::PatternTrackView(PatternTrack* pt, TrackContainerView* tcv) :
39 TrackView(pt, tcv),
40 m_patternTrack(pt)
41{
42 setFixedHeight( 32 );
43 // drag'n'drop with pattern tracks only causes troubles (and makes no sense too), so disable it
44 setAcceptDrops( false );
45
46 m_trackLabel = new TrackLabelButton( this, getTrackSettingsWidget() );
47 m_trackLabel->setIcon( embed::getIconPixmap("pattern_track"));
48 m_trackLabel->move( 3, 1 );
49 m_trackLabel->show();
50 connect( m_trackLabel, SIGNAL(clicked(bool)),
51 this, SLOT(clickedTrackLabel()));
52 setModel(pt);
53}
