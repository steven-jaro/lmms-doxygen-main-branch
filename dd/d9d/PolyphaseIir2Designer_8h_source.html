1/*****************************************************************************
2
3 PolyphaseIir2Designer.h
4 Author: Laurent de Soras, 2005
5
6Compute coefficients for 2-path polyphase IIR filter, half-band filter or
7Pi/2 phaser.
8
9 -2
10 a + z
11 N/2-1 2k
12A0 (z) = Prod ----------
13 k = 0 -2
14 1 + a z
15 2k
16
17 -2
18 a + z
19 -1 (N-1)/2 2k+1
20A1 (z) = z . Prod ------------
21 k = 0 -2
22 1 + a z
23 2k+1
24
25 1
26H (z) = - (A0 (z) + A1 (z))
27 2
28
29Sum of A0 and A1 gives a low-pass filter.
30Difference of A0 and A1 gives the complementary high-pass filter.
31
32For the Pi/2 phaser, product form is (a - z^-2) / (1 - az^-2)
33Sum and difference of A0 and A1 have a Pi/2 phase difference.
34

216==============================================================================
217Name: compute_nbr_coefs_from_proto
218Description:
219 Finds the minimum number of coefficients for a given filter specification
220Input parameters:
221 - attenuation: stopband attenuation, dB. > 0.
222
223 - transition: normalized transition bandwith (% relative to Fs).
224 Range ]0 ; 1/2[
225Returns: Number of coefficients, > 0
226Throws: Nothing
227==============================================================================
228*/
229
230inline int PolyphaseIir2Designer::compute_nbr_coefs_from_proto (double attenuation, double transition) noexcept
231{
232 assert (attenuation > 0);
233 assert (transition > 0);
234 assert (transition < 0.5);
235
236 double k;
237 double q;
238 compute_transition_param (k, q, transition);

240 const int nbr_coefs = (order - 1) / 2;
241
242 return nbr_coefs;
243}

279/*
280==============================================================================
281Name: compute_coefs
282Description:
283 Computes coefficients for a half-band polyphase IIR filter, function of a
284 given stopband gain / transition bandwidth specification.
285 Order is automatically calculated.
286Input parameters:
287 - attenuation: stopband attenuation, dB. > 0.
288 - transition: normalized transition bandwith (% relative to Fs).
289 Range ]0 ; 1/2[
290Output parameters:
291 - coef_arr: Coefficient list, must be large enough to store all the
292 coefficients. Filter order = nbr_coefs * 2 + 1
293Returns: number of coefficients
294Throws: Nothing
295==============================================================================
296*/
297
298inline int PolyphaseIir2Designer::compute_coefs (double coef_arr [], double attenuation, double transition) noexcept
299{
300 assert (attenuation > 0);
301 assert (transition > 0);
302 assert (transition < 0.5);
303
304 double k;
305 double q;
306 compute_transition_param (k, q, transition);
307
308 // Computes number of required coefficients
309 const int order = compute_order (attenuation, q);
310 const int nbr_coefs = (order - 1) / 2;
311
312 // Coefficient calculation
313 for (int index = 0; index < nbr_coefs; ++index)
314 {
315 coef_arr [index] = compute_coef (index, k, q, order);
316 }
317
318 return nbr_coefs;
319}

364/*
365==============================================================================
366Name: compute_coefs_spec_order_gdly
367Description:
368 Computes the coefficients for a half-band polyphase IIR filter by
369 specifying a phase delay and the desired filter order.
370
371 It is also possible to specify the allowed ranges for attenuation and
372 transition bandwidth.
373 Phase delay range is constrained by the filter order and the measurement
374 frequency, so the function may fail if the requirement cannot be met.
375 Important note: the downsampler classes introduce an *advance* of 1/2
376 output sample (half rate, so that's 1 input sample full rate). You have to
377 take that into account if you're targetting the downsamplers by adding 1.0
378 to the phase_delay value.
379Input parameters:
380 - nbr_coefs: Number of desired coefficients. > 0.
381 - phase_delay: required phase delay in samples. The phase delay is measured
382 at f_rel, and is reached within +/-prec samples. > 0.
383 - f_rel: Normalised relative frequency for the phase delay measurement.
384 Range is [0 ; 1[, where 1/2 is the Nyquist frequency for a full-rate
385 (not decimated) filter.
386 - prec: Precision of the phase delay requirement, in samples. > 0.
387 - atten_lb: Lower bound for the stopband attenuation, in dB.
388 Range ]0 ; atten_ub[
389 - atten_ub: Upper bound for the stopband attenuation, in dB. > atten_lb.
390 - trans_lb: Lower bound for the normalized transition bandwidth (relative
391 to the sampling rate). Range ]0 ; trans_ub[
392 - trans_ub: Upper bound for the normalized transition bandwidth (relative
393 to the sampling rate). Range ]trans_lb ; 1/2[
394Output parameters:
395 - coef_arr: Coefficient list, must be large enough to store all the
396 coefficients.
397Input/output parameters:
398 - attenuation_ptr: if the function succeeds and attenuation_ptr is not 0,
399
400 the attenuation in dB for the designed filter is written on pointed
401 location.
402 - transition_ptr: if the function succeeds and transition_ptr is not 0,
403 the normalized transition bandwidth is written there.
404Returns:
405 - 0 on success,
406 - Negative number on failure, mainly because the constraints are too tight
407 or the requirements unreachable. In this case, the output parameters
408 should be considered as invalid.
409Throws: Nothing
410==============================================================================
411*/

503/*
504==============================================================================
505Name: compute_phase_delay
506Description:
507 Computes the phase delay introduced by a complete filter at a specified
508 frequency.
509 The delay is given for a constant sampling rate between input and output.
510Input parameters:
511 - coef_arr: filter coefficient, as given by the designing functions.
512
513 - nbr_coefs: Number of filter coefficients. > 0.
514 - f_fs: frequency relative to the sampling rate, [0 ; 0.5].
515 - ph_flag: set if filter is used in pi/2-phaser mode, in the form
516 (a - z^-2) / (1 - az^-2)
517Returns:
518 The phase delay in samples, >= 0. In the phaser mode, the delay is
519 negative and corresponds to the path for even coefficients (A0).
520Throws: Nothing
521==============================================================================
522*/
523
524inline double PolyphaseIir2Designer::compute_phase_delay (const double coef_arr [], int nbr_coefs, double f_fs, bool ph_flag) noexcept
525{
526 assert (coef_arr != nullptr);
527 assert (nbr_coefs > 0);
528 assert (f_fs >= 0);
529 assert (f_fs < 0.5);
530
531 auto dly_arr = std::array <double, 2> { 0, 1 };
532 for (int k = 0; k < nbr_coefs; ++k)
533 {
534 const auto dly =
535 compute_unit_phase_delay (coef_arr [k], f_fs, ph_flag);
536 dly_arr [k & 1] += dly;
537 }
538
539 const auto com = dly_arr [0]; // Common delay (the even path)
540 auto dly_final = com;
541
542 if (! ph_flag && f_fs > 0)
543 {
544 const auto w = 2 * hiir::PI * f_fs;
545 const auto phi = (dly_arr [1] - com) * w;
546 assert (std::abs (phi) < hiir::PI);
547 const auto re = cos (phi) + 1; // Complex sum of both paths
548 const auto im = sin (phi) + 0;
549 const auto dif = atan2 (im, re) / w; // Odd path relative to the even
550 dly_final += dif;
551 }
552
553 return dly_final;
554}

562
563 Computes the group delay introduced by a complete filter at a specified

566 The delay is given for a constant sampling rate between input and output.

572 - nbr_coefs: Number of filter coefficients. > 0.
573 - f_fs: frequency relative to the sampling rate, [0 ; 0.5].
574 - ph_flag: set if filter is used in pi/2-phaser mode, in the form

578Throws: Nothing
579==============================================================================
580*/
581
582inline double PolyphaseIir2Designer::compute_group_delay (const double coef_arr [], int nbr_coefs, double f_fs, bool ph_flag) noexcept
583{
584 assert (coef_arr != nullptr);
585 assert (nbr_coefs > 0);
586 assert (f_fs >= 0);
587 assert (f_fs < 0.5);
588
589 double dly_total = 0;
590 for (int k = 0; k < nbr_coefs; k += 2)

602/*
603==============================================================================
604Name: compute_unit_phase_delay
605Description:
606 Computes the phase delay introduced by a single filtering unit at a
607 specified frequency.
608 The delay is given for a constant sampling rate between input and output.
609Input parameters:
610 - a: coefficient for the cell, [0 ; 1[
611 - f_fs: frequency relative to the sampling rate, [0 ; 0.5].
612 - ph_flag: set if filtering unit is used in pi/2-phaser mode, in the form
613 (a - z^-2) / (1 - az^-2)
614Returns:
615 The phase delay in samples, positive when ph_flag is false, negative in
616 phaser mode.
617
618Throws: Nothing
619==============================================================================
620*/
621
622inline double PolyphaseIir2Designer::compute_unit_phase_delay (double a, double f_fs, bool ph_flag) noexcept
623{
624 assert (a >= 0);
625 assert (a < 1);
626 assert (f_fs >= 0);
627 assert (f_fs < 0.5);
628
629 const auto w = 2 * hiir::PI * f_fs;
630 const auto c = cos (w);
631 const auto s = sin (w);
632 const auto u = (ph_flag) ? -1 : 1;
633 const auto ac = (a + u) * c;
634 const auto as = (a - u) * s;
635 const auto x = u * (ac * ac - as * as);
636 const auto y = u * 2 * ac * as;
637 auto ph = -atan2 (y, x);
638 if (ph * u < 0)
639 {
640 ph += u * 2 * hiir::PI;
641 }
642 const auto dly = (w > 0) ? ph / w : -2 * u * (a - u) / (a + u);
643
644 return dly;
645}

649/*
650==============================================================================
651Name: compute_unit_group_delay
652Description:
653 Computes the group delay introduced by a single filtering unit at a
654 specified frequency.
655 The delay is given for a constant sampling rate between input and output.
656 To compute the group delay of a complete filter, add the group delays
657 of all the units in A0 (z).
658
659Input parameters:
660 - a: coefficient for the cell, [0 ; 1[
661 - f_fs: frequency relative to the sampling rate, [0 ; 0.5].
662 - ph_flag: set if filtering unit is used in pi/2-phaser mode, in the form
663 (a - z^-2) / (1 - az^-2)
664Returns:
665 The group delay in samples, >= 0.
666Throws: Nothing
667==============================================================================
668*/
669
670inline double PolyphaseIir2Designer::compute_unit_group_delay (double a, double f_fs, bool ph_flag) noexcept
671{
672 assert (a >= 0);
673 assert (a < 1);
674 assert (f_fs >= 0);
675 assert (f_fs < 0.5);
676
677 const auto w = 2 * hiir::PI * f_fs;
678 const auto a2 = a * a;
679 const auto sig = (ph_flag) ? -2 : 2;
680 const auto dly = 2 * (1 - a2) / (a2 + sig * a * cos (2 * w) + 1);
681
682 return dly;
683}

694
695inline void PolyphaseIir2Designer::compute_transition_param (double &k, double &q, double transition) noexcept
696{
697 assert (transition > 0);
698 assert (transition < 0.5);
699
700 k = tan ((1 - transition * 2) * hiir::PI / 4);
701 k *= k;
702 assert (k < 1);
703 assert (k > 0);
704 double kksqrt = pow (1 - k * k, 0.25);
705 const double e = 0.5 * (1 - kksqrt) / (1 + kksqrt);
706 const double e2 = e * e;
707 const double e4 = e2 * e2;
708 q = e * (1 + e4 * (2 + e4 * (15 + 150 * e4)));
709 assert (q > 0);
710}

713
714inline int PolyphaseIir2Designer::compute_order (double attenuation, double q) noexcept
715{
716 assert (attenuation > 0);
717 assert (q > 0);
718
719 const double attn_p2 = pow (10.0, -attenuation / 10);
720 const double a = attn_p2 / (1 - attn_p2);
721 int order = hiir::ceil_int (log (a * a / 16) / log (q));
722 if ((order & 1) == 0)

739 assert (order > 0);

753inline double PolyphaseIir2Designer::compute_coef (int index, double k, double q, int order) noexcept
754{
755 assert (index >= 0);
756 assert (index * 2 < order);
757
758 const int c = index + 1;
759 const double num = compute_acc_num (q, order, c) * pow (q, 0.25);
760 const double den = compute_acc_den (q, order, c) + 0.5;
761 const double ww = num / den;
762 const double wwsq = ww * ww;
763
764 const double x = sqrt ((1 - wwsq * k) * (1 - wwsq / k)) / (1 + wwsq);
765 const double coef = (1 - x) / (1 + x);
766
767 return coef;
768}

772inline double PolyphaseIir2Designer::compute_acc_num (double q, int order, int c) noexcept
773{
774 assert (c >= 1);
775 assert (c < order * 2);
776
777 int i = 0;
778 int j = 1;
779 double acc = 0;
780 double q_ii1;
781 do
782 {
783 q_ii1 = hiir::ipowp (q, i * (i + 1));
784 q_ii1 *= sin ((i * 2 + 1) * c * hiir::PI / order) * j;
785 acc += q_ii1;
786
787 j = -j;
788 ++i;
789 }
790 while (fabs (q_ii1) > 1e-100);
791
792 return acc;

823PolyphaseIir2Designer::ResCode PolyphaseIir2Designer::compute_coefs_spec_order_delay (double coef_arr [], double *attenuation_ptr, double *transition_ptr, int nbr_coefs, double delay, double f_rel, double prec, double atten_lb, double atten_ub, double trans_lb, double trans_ub, F compute_delay) noexcept
824{
825 assert (nbr_coefs > 0);
826 assert (nbr_coefs <= _max_order);
827 assert (delay > 0);
828 assert (f_rel >= 0);
829 assert (f_rel < 1);
830 assert (prec > 0);
831 assert (atten_lb > 0);
832 assert (atten_lb < atten_ub);
833 assert (trans_lb > 0);
834 assert (trans_lb < trans_ub);
835 assert (trans_ub < 0.5);
836
837 ResCode ret_val = ResCode_OK;
838
839 double lb_tbw = trans_lb;
840 double ub_tbw = trans_ub;
841 std::array <double, _max_order> lb_coef_arr;
842 std::array <double, _max_order> ub_coef_arr;
843
844 compute_coefs_spec_order_tbw (ub_coef_arr.data (), nbr_coefs, ub_tbw);
845 compute_coefs_spec_order_tbw (lb_coef_arr.data (), nbr_coefs, lb_tbw);
846
847 double ub_gdly = compute_delay (ub_coef_arr.data (), nbr_coefs, f_rel);
848 double lb_gdly = compute_delay (lb_coef_arr.data (), nbr_coefs, f_rel);
849
850 // Checks if the group delay and transition bandwidth requirements
851 // could be met
852 if ((ub_gdly - delay) * (delay - lb_gdly) <= 0)
853 {
854 ret_val = ResCode_FAIL_GD_TBW;
855 }
856
857 double rs_attn = 0;
858 double rs_tbw = 0;
859
860 if (ret_val == ResCode_OK)
861 {
862 // Simple bisection method
863 const int max_it = 1000;
864 int nbr_it = 0;
865 double rs_gdly = 0;
866 bool conv_flag = false;
867 do
868 {
869 rs_tbw = (ub_tbw + lb_tbw) * 0.5;
870 rs_attn = compute_atten_from_order_tbw (nbr_coefs, rs_tbw);
871 compute_coefs_spec_order_tbw (coef_arr, nbr_coefs, rs_tbw);
872 rs_gdly = compute_delay (coef_arr, nbr_coefs, f_rel);
873
874 if ((delay - lb_gdly) * (delay - rs_gdly) < 0)
875 {
876 ub_tbw = rs_tbw;
877// ub_gdly = rs_gdly; // ub_gdly not used in the loop actually
878 }
879 else
880 {
881 lb_tbw = rs_tbw;
882 lb_gdly = rs_gdly;
883 }
884
885 ++ nbr_it;
886 conv_flag = (fabs (rs_gdly - delay) > prec);
887 }
888 while (conv_flag && nbr_it < max_it);
889
890 // Checks convergence
891 if (nbr_it >= max_it && ! conv_flag)
892 {
893 ret_val = ResCode_FAIL_CONV;
894 }
895
896 // Checks the attenuation requirement
897 else if ( rs_attn < atten_lb
898 || rs_attn > atten_ub)
899 {
900 ret_val = ResCode_FAIL_ATTEN;
901 }
902 }
903
904 if (ret_val == ResCode_OK)
905 {
906 if (attenuation_ptr != nullptr)
907 {
908 *attenuation_ptr = rs_attn;
909 }
910 if (transition_ptr != nullptr)
911 {
912 *transition_ptr = rs_tbw;
913 }
914 }
915
916 return ret_val;
917}
