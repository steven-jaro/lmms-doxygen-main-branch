49void MidiController::updateValueBuffer()
50{
51 if( m_previousValue != m_lastValue )
52 {
53 m_valueBuffer.interpolate( m_previousValue, m_lastValue );
54 m_previousValue = m_lastValue;
55 }
56 else
57 {
58 m_valueBuffer.fill( m_lastValue );
59 }
60 m_bufferLastUpdated = s_periods;
61}

74void MidiController::processInEvent(const MidiEvent& event, const TimePos& time, f_cnt_t offset)
75{
76 switch(event.type())
77 {
78 case MidiControlChange:
79 {
80 unsigned char controllerNum = event.controllerNumber();
81
82 if (m_midiPort.inputController() == controllerNum &&
83 (m_midiPort.inputChannel() == event.channel() + 1 || m_midiPort.inputChannel() == 0))
84 {
85 unsigned char val = event.controllerValue();
86 m_previousValue = m_lastValue;
87 m_lastValue = static_cast<float>(val) / 127.0f;
88 emit valueChanged();
89 }
90 break;
91 }
92 default:
93 // Don't care - maybe add special cases for pitch and mod later
94 break;
95 }
96}
