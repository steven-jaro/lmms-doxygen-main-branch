55void AutomatableModelView::addDefaultActions( QMenu* menu )
56{
57 AutomatableModel* model = modelUntyped();
58
59 auto amvSlots = new AutomatableModelViewSlots(this, menu);
60
61 menu->addAction( embed::getIconPixmap( "reload" ),
62 AutomatableModel::tr( "&Reset (%1%2)" ).
63 arg( model->initValue<float>() * m_conversionFactor ).
64 arg( m_unit ),
65 model, SLOT(reset()));
66
67 menu->addSeparator();
68 menu->addAction( embed::getIconPixmap( "edit_copy" ),
69 AutomatableModel::tr( "&Copy value (%1%2)" ).
70 arg( model->value<float>() * m_conversionFactor ).
71 arg( m_unit ),
72 amvSlots, SLOT(copyToClipboard()));
73
74 bool canPaste = true;
75 const float valueToPaste = floatFromClipboard(&canPaste);
76 const QString pasteDesc = canPaste ?
77 AutomatableModel::tr( "&Paste value (%1%2)").
78 arg( valueToPaste ).
79 arg( m_unit )
80 : AutomatableModel::tr( "&Paste value");
81 QAction* pasteAction = menu->addAction( embed::getIconPixmap( "edit_paste" ),
82 pasteDesc, amvSlots, SLOT(pasteFromClipboard()));
83 pasteAction->setEnabled(canPaste);
84
85 menu->addSeparator();
86
87 menu->addAction( embed::getIconPixmap( "automation" ),
88 AutomatableModel::tr( "Edit song-global automation" ),
89 amvSlots,
90 SLOT(editSongGlobalAutomation()));
91
92 menu->addAction( QPixmap(),
93 AutomatableModel::tr( "Remove song-global automation" ),
94 amvSlots,
95 SLOT(removeSongGlobalAutomation()));
96
97 menu->addSeparator();
98
99 if( model->hasLinkedModels() )
100 {
101 menu->addAction( embed::getIconPixmap( "edit-delete" ),
102 AutomatableModel::tr( "Remove all linked controls" ),
103 amvSlots, SLOT(unlinkAllModels()));
104 menu->addSeparator();
105 }
106
107 QString controllerTxt;
108 if( model->controllerConnection() )
109 {
110 Controller* cont = model->controllerConnection()->getController();
111 if( cont )
112 {
113 controllerTxt = AutomatableModel::tr( "Connected to %1" ).arg( cont->name() );
114 }
115 else
116 {
117 controllerTxt = AutomatableModel::tr( "Connected to controller" );
118 }
119
120 QMenu* contMenu = menu->addMenu( embed::getIconPixmap( "controller" ), controllerTxt );
121
122 contMenu->addAction( embed::getIconPixmap( "controller" ),
123 AutomatableModel::tr("Edit connection..."),
124 amvSlots, SLOT(execConnectionDialog()));
125 contMenu->addAction( embed::getIconPixmap( "cancel" ),
126 AutomatableModel::tr("Remove connection"),
127 amvSlots, SLOT(removeConnection()));
128 }
129 else
130 {
131 menu->addAction( embed::getIconPixmap( "controller" ),
132 AutomatableModel::tr("Connect to controller..."),
133 amvSlots, SLOT(execConnectionDialog()));
134 }
135}

148// Unsets the current model by setting a dummy empty model. The dummy model is marked as
149// "defaultConstructed", so the next call to setModel will delete it.
150void AutomatableModelView::unsetModel()
151{
152 if (dynamic_cast<FloatModelView*>(this))
153 {
154 setModel(new FloatModel(0, 0, 0, 1, nullptr, QString(), true));
155 }
156 else if (dynamic_cast<IntModelView*>(this))
157 {
158 setModel(new IntModel(0, 0, 0, nullptr, QString(), true));
159 }
160 else if (dynamic_cast<BoolModelView*>(this))
161 {
162 setModel(new BoolModel(false, nullptr, QString(), true));
163 }
164 else
165 {
166 ModelView::unsetModel();
167 }
168}

173void AutomatableModelView::mousePressEvent( QMouseEvent* event )
174{
175 if (event->button() == Qt::LeftButton && event->modifiers() & KBD_COPY_MODIFIER)
176 {
177 new gui::StringPairDrag( "automatable_model", QString::number( modelUntyped()->id() ), QPixmap(), widget() );
178 event->accept();
179 }
180 else if( event->button() == Qt::MiddleButton )
181 {
182 modelUntyped()->reset();
183 }

212void AutomatableModelViewSlots::execConnectionDialog()
213{
214 // TODO[pg]: Display a dialog with list of controllers currently in the song
215 // in addition to any system MIDI controllers
216 AutomatableModel* m = m_amv->modelUntyped();
217
218 m->displayName();
219 gui::ControllerConnectionDialog d( getGUI()->mainWindow(), m );
220
221 if( d.exec() == 1 )
222 {
223 // Actually chose something
224 if( d.chosenController() )
225 {
226 // Update
227 if( m->controllerConnection() )
228 {
229 m->controllerConnection()->setController( d.chosenController() );
230 }
231 // New
232 else
233 {
234 auto cc = new ControllerConnection(d.chosenController());
235 m->setControllerConnection( cc );
236 //cc->setTargetName( m->displayName() );
237 }
238 }
239 // no controller, so delete existing connection
240 else
241 {
242 removeConnection();
243 }
244 }
245}

292void AutomatableModelViewSlots::pasteFromClipboard()
293{
294 bool isNumber = false;
295 const float number = floatFromClipboard(&isNumber);
296 if (isNumber) {
297 m_amv->modelUntyped()->setValue(number / m_amv->getConversionFactor());
298 }
299}

301/// Attempt to parse a float from the clipboard
302static float floatFromClipboard(bool* ok)
303{
304 // For getString() and MimeType enum class
305 using namespace Clipboard;
306
307 return getString( MimeType::Default ).toFloat(ok);
308}
