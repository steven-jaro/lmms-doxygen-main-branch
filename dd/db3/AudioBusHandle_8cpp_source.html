109void AudioBusHandle::doProcessing()
110{
111 if (m_mutedModel && m_mutedModel->value())
112 {
113 return;
114 }
115
116 const fpp_t fpp = Engine::audioEngine()->framesPerPeriod();
117
118 // clear the buffer
119 zeroSampleFrames(m_buffer, fpp);
120
121 //qDebug( "Playhandles: %d", m_playHandles.size() );
122 for (PlayHandle* ph : m_playHandles) // now we mix all playhandle buffers into our internal buffer
123 {
124 if (ph->buffer())
125 {
126 if (ph->usesBuffer()
127 && (ph->type() == PlayHandle::Type::NotePlayHandle
128 || !MixHelpers::isSilent(ph->buffer(), fpp)))
129 {
130 m_bufferUsage = true;
131 MixHelpers::add(m_buffer, ph->buffer(), fpp);
132 }
133 ph->releaseBuffer(); // gets rid of playhandle's buffer and sets
134 // pointer to null, so if it doesn't get re-acquired we know to skip it next time
135 }
136 }
137
138 if (m_bufferUsage)
139 {
140 // handle volume and panning
141 // has both vol and pan models
142 if (m_volumeModel && m_panningModel)
143 {
144 ValueBuffer* volBuf = m_volumeModel->valueBuffer();
145 ValueBuffer* panBuf = m_panningModel->valueBuffer();
146
147 // both vol and pan have s.ex.data:
148 if (volBuf && panBuf)
149 {
150 for (f_cnt_t f = 0; f < fpp; ++f)
151 {
152 float v = volBuf->values()[f] * 0.01f;
153 float p = panBuf->values()[f] * 0.01f;
154 m_buffer[f][0] *= (p <= 0 ? 1.0f : 1.0f - p) * v;
155 m_buffer[f][1] *= (p >= 0 ? 1.0f : 1.0f + p) * v;
156 }
157 }
158
159 // only vol has s.ex.data:
160 else if (volBuf)
161 {
162 float p = m_panningModel->value() * 0.01f;
163 float l = (p <= 0 ? 1.0f : 1.0f - p);
164 float r = (p >= 0 ? 1.0f : 1.0f + p);
165 for (f_cnt_t f = 0; f < fpp; ++f)
166 {
167 float v = volBuf->values()[f] * 0.01f;
168 m_buffer[f][0] *= v * l;
169 m_buffer[f][1] *= v * r;
170 }
171 }
172
173 // only pan has s.ex.data:
174 else if (panBuf)
175 {
176 float v = m_volumeModel->value() * 0.01f;
177 for (f_cnt_t f = 0; f < fpp; ++f)
178 {
179 float p = panBuf->values()[f] * 0.01f;
180 m_buffer[f][0] *= (p <= 0 ? 1.0f : 1.0f - p) * v;
181 m_buffer[f][1] *= (p >= 0 ? 1.0f : 1.0f + p) * v;
182 }
183 }
184
185 // neither has s.ex.data:
186 else
187 {
188 float p = m_panningModel->value() * 0.01f;
189 float v = m_volumeModel->value() * 0.01f;
190 for (f_cnt_t f = 0; f < fpp; ++f)
191 {
192 m_buffer[f][0] *= (p <= 0 ? 1.0f : 1.0f - p) * v;
193 m_buffer[f][1] *= (p >= 0 ? 1.0f : 1.0f + p) * v;
194 }
195 }
196 }
197
198 // has vol model only
199 else if (m_volumeModel)
200 {
201 ValueBuffer* volBuf = m_volumeModel->valueBuffer();
202
203 if (volBuf)
204 {
205 for (f_cnt_t f = 0; f < fpp; ++f)
206 {
207 float v = volBuf->values()[f] * 0.01f;
208 m_buffer[f][0] *= v;
209 m_buffer[f][1] *= v;
210 }
211 }
212 else
213 {
214 float v = m_volumeModel->value() * 0.01f;
215 for (f_cnt_t f = 0; f < fpp; ++f)
216 {
217 m_buffer[f][0] *= v;
218 m_buffer[f][1] *= v;
219 }
220 }
221 }
222 }
223 // as of now there's no situation where we only have panning model but no volume model
224 // if we have neither, we don't have to do anything here - just pass the audio as is
225
226 // handle effects
227 const bool anyOutputAfterEffects = processEffects();
228 if (anyOutputAfterEffects || m_bufferUsage)
229 {
230 Engine::mixer()->mixToChannel(m_buffer, m_nextMixerChannel); // send output to mixer
231 // TODO: improve the flow here - convert to pull model
232 m_bufferUsage = false;
233 }
234}

244void AudioBusHandle::removePlayHandle(PlayHandle* handle)
245{
246 QMutexLocker lockGuard(&m_playHandleLock);
247 PlayHandleList::Iterator it = std::find(m_playHandles.begin(), m_playHandles.end(), handle);
248 if (it != m_playHandles.end())
249 {
250 m_playHandles.erase(it);
251 }
