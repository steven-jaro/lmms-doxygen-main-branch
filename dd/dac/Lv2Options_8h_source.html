47/**
48 Option container
49
50 References all available options for a plugin and maps them to their URIDs.
51 This class is used per Lv2 processor (justification in Lv2Proc::initMOptions())
52
53 The public member functions should be called in descending order:
54
55 1. supportOption: set all supported option URIDs
56 2. initOption: initialize options with values
57 3. createOptionVectors: create the option vectors required for
58 the feature
59 4. access the latter using feature()
60*/
61class Lv2Options
62{
63public:
64 //! Return if an option is supported by LMMS
65 static bool isOptionSupported(LV2_URID key);
66 //! Mark option as supported
67 static void supportOption(LV2_URID key);
68
69 //! Initialize an option
70 template<typename Opt, typename Arg>
71 void initOption(Lv2UridCache::Id key, Arg&& value,
72 LV2_Options_Context context = LV2_OPTIONS_INSTANCE,
73 std::uint32_t subject = 0)
74 {
75 const Lv2UridCache& cache = Engine::getLv2Manager()->uridCache();
76 initOption(cache[key], sizeof(Opt), cache[Lv2UridCache::IdForType<Opt>::value],
77 std::make_shared<Opt>(std::forward<Arg>(value)), context, subject);
78 }
79 //! Fill m_options and m_optionPointers with all options
80 void createOptionVectors();
81 //! Return the feature
82 const LV2_Options_Option* feature() const
83 {
84 return m_options.data();
85 }
86
87 void clear();
88
89private:
90 //! Initialize an option internally
91 void initOption(LV2_URID key,
92 uint32_t size,
93 LV2_URID type,
94 std::shared_ptr<void> value,
95 LV2_Options_Context context = LV2_OPTIONS_INSTANCE,
96 uint32_t subject = 0);
97 //! options that are supported by every processor
98 static std::set<LV2_URID> s_supportedOptions;
99 //! options + data, ordered by URID
100 std::map<LV2_URID, LV2_Options_Option> m_optionByUrid;
101 //! option storage
102 std::vector<LV2_Options_Option> m_options;
103 //! option value storage
104 std::map<LV2_URID, std::shared_ptr<void>> m_optionValues;
105};
