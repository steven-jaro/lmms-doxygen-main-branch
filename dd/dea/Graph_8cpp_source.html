181void Graph::drawLineAt( int _x, int _y, int _lastx )
182{
183 float minVal = model()->minValue();
184 float maxVal = model()->maxValue();
185 if ( width() <= 4 )
186 {
187 return;
188 }
189
190 float xscale = static_cast<float>( model()->length() ) /
191 ( width()-4 );
192
193 //consider border
194 _x -= 2;
195 _y -= 2;
196 _lastx -= 2;
197
198 _lastx = qMax( 0, qMin( _lastx, width()-5 ) );
199
200 float range = minVal - maxVal;
201 float val = ( _y*range/( height()-5 ) ) + maxVal;
202
203 int sample_begin, sample_end;
204 float lastval;
205 float val_begin, val_end;
206
207 if (_lastx > _x)
208 {
209 sample_begin = (int)((_x) * xscale);
210 sample_end = (int)ceil((_lastx+1) * xscale);
211 lastval = model() -> m_samples[ (int)( sample_end - 1 ) ];
212 val_begin = val;
213 val_end = lastval;
214
215 }
216 else
217 {
218 sample_begin = (int)(_lastx * xscale);
219 sample_end = (int)ceil((_x+1) * xscale);
220 lastval = model() -> m_samples[ (int)( sample_begin ) ];
221 val_begin = lastval;
222 val_end = val;
223
224 }
225
226 // calculate line drawing variables
227 int linelen = sample_end - sample_begin;
228 if (linelen == 1)
229 {
230 val_begin = val;
231 }
232 //int xstep = _x > _lastx ? -1 : 1;
233 float ystep = ( val_end - val_begin ) / linelen;
234
235 // draw a line
236 for ( int i = 0 ; i < linelen; i++ )
237 {
238 model()->drawSampleAt( sample_begin + i , val_begin + ((i ) * ystep));
239 }
240
241 // We've changed [sample_end, sample_begin)
242 // However, samplesChanged expects two end points
243 model()->samplesChanged(sample_begin, sample_end - 1);
244}

287void Graph::paintEvent( QPaintEvent * )
288{
289 QPainter p( this );
290
291 p.setPen( QPen( m_graphColor, 1 ) );
292 QColor gcol = QColor( m_graphColor.red(), m_graphColor.green(), m_graphColor.blue(), 100 );
293
294 QVector<float> * samps = &(model()->m_samples);
295 int length = model()->length();
296 const float maxVal = model()->maxValue();
297 const float minVal = model()->minValue();
298
299 float xscale = (float)( width()-4 ) / length;
300 float yscale = (float)( height()-4 ) / ( minVal - maxVal );
301
302 // Max index, more useful below
303 length--;
304
305
306 switch( m_graphStyle )
307 {
308 case Style::Linear:
309 p.setRenderHints( QPainter::Antialiasing, true );
310
311 for( int i=0; i < length; i++ )
312 {
313 // Needs to be rewritten
314 p.drawLine(
315 2+static_cast<int>(i*xscale),
316 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale ),
317 2+static_cast<int>((i+1)*xscale),
318 2+static_cast<int>( ( (*samps)[i+1] - maxVal ) * yscale )
319 );
320 }
321
322 // Draw last segment wrapped around
323 p.drawLine(2+static_cast<int>(length*xscale),
324 2+static_cast<int>( ( (*samps)[length] - maxVal ) * yscale ),
325 width()-3,
326 2+static_cast<int>( ( (*samps)[0] - maxVal ) * yscale ) );
327
328 p.setRenderHints( QPainter::Antialiasing, false );
329 break;
330
331
332 case Style::Nearest:
333 for( int i=0; i < length; i++ )
334 {
335 p.drawLine(2+static_cast<int>(i*xscale),
336 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale ),
337 2+static_cast<int>((i+1)*xscale),
338 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale )
339 );
340 p.drawLine(2+static_cast<int>((i+1)*xscale),
341 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale ),
342 2+static_cast<int>((i+1)*xscale),
343 2+static_cast<int>( ( (*samps)[i+1] - maxVal ) * yscale )
344 );
345 }
346
347 p.drawLine(2+static_cast<int>(length*xscale),
348 2+static_cast<int>( ( (*samps)[length] - maxVal ) * yscale ),
349 width()-3,
350 2+static_cast<int>( ( (*samps)[length] - maxVal ) * yscale ) );
351 break;
352
353 case Style::LinearNonCyclic:
354 p.setRenderHints( QPainter::Antialiasing, true );
355
356 for( int i=0; i < length; i++ )
357 {
358 // Needs to be rewritten
359 p.drawLine(
360 2+static_cast<int>(i*xscale),
361 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale ),
362 2+static_cast<int>((i+1)*xscale),
363 2+static_cast<int>( ( (*samps)[i+1] - maxVal ) * yscale )
364 );
365 }
366
367 // Do not draw last segment wrapped around - hence, "non-cyclic"
368
369 p.setRenderHints( QPainter::Antialiasing, false );
370 break;
371
372 case Style::Bar:
373 for( int i=0; i <= length; i++ )
374 {
375 p.fillRect( 2+static_cast<int>( i*xscale ),
376 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale ),
377 qMax( static_cast<int>(xscale) - 1, 1 ),
378 qMax( static_cast<int>( ( minVal - maxVal ) * yscale ) - static_cast<int>( ( (*samps)[i] - maxVal ) * yscale ), 1 ),
379 gcol );
380
381 p.setPen( QPen( m_graphColor, 1.0 ) );
382
383 p.drawLine( 2+static_cast<int>(i*xscale),
384 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale ),
385 qMax( static_cast<int>(i*xscale) + static_cast<int>(xscale), 2+static_cast<int>(i*xscale) ),
386 2+static_cast<int>( ( (*samps)[i] - maxVal ) * yscale )
387 );
388 }
389
390 break;
391
392 default:
393 break;
394 }
395
396
397 // draw Pointer
398 if( m_mouseDown )
399 {
400 QPoint cursor = mapFromGlobal( QCursor::pos() );
401 p.setPen( QColor( 0x70, 0x7C, 0x91 ) );
402 p.drawLine( 2, cursor.y(), width()-2, cursor.y() );
403 p.drawLine( cursor.x(), 2, cursor.x(), height()-2 );
404 }
405 p.drawPixmap( 0, 0, m_foreground );
406}

471void graphModel::setRange(float ymin, float ymax)
472{
473 if (ymin == m_minValue && ymax == m_maxValue) { return; }
474 // Step sizes less than zero or larger than the entire range of
475 // values are nonsense, clamp those
476 m_step = std::clamp(m_step, 0.f, std::abs(ymax - ymin));
477 m_minValue = ymin;
478 m_maxValue = ymax;
479 // Trim existing values
480 for (float& sample : m_samples) {
481 sample = std::clamp(sample, ymin, ymax);
482 }
483 emit rangeChanged();
484}

599void graphModel::smooth()
600{
601 // store values in temporary array
602 QVector<float> temp = m_samples;
603
604 // Smoothing
605 m_samples[0] = ( temp[length()-1] + ( temp[0] * 2 ) + temp[1] ) * 0.25f;
606 for ( int i=1; i < ( length()-1 ); i++ )
607 {
608 m_samples[i] = ( temp[i-1] + ( temp[i] * 2 ) + temp[i+1] ) * 0.25f;
609 }
610 m_samples[length()-1] = ( temp[length()-2] + ( temp[length()-1] * 2 ) + temp[0] ) * 0.25f;
611
612 emit samplesChanged(0, length()-1);
613}

631void graphModel::convolve(const float *convolution,
632 const int convolutionLength, const int centerOffset)
633{
634 // store values in temporary array
635 QVector<float> temp = m_samples;
636 const int graphLength = length();
637 // make a cyclic convolution
638 for ( int i = 0; i < graphLength; i++ )
639 {
640 float sum = 0.0f;
641 for ( int j = 0; j < convolutionLength; j++ )
642 {
643 sum += convolution[j] * temp[( i + j ) % graphLength];
644 }
645 m_samples[( i + centerOffset ) % graphLength] = sum;
646 }
647 emit samplesChanged(0, graphLength - 1);
648}

650void graphModel::normalize()
651{
652 float max = 0.0001f;
653 float avg = 0.0f;
654
655 // first correct dc offset by normalizing to average
656 for( int i = 0; i < length(); i++ )
657 avg += m_samples[i];
658 avg /= length();
659 for( int i = 0; i < length(); i++ )
660 m_samples[i] -= avg;
661
662 // then maximize
663 for( int i = 0; i < length(); i++ )
664 max = qMax( max, qAbs( m_samples[i] ) );
665
666 for( int i = 0; i < length(); i++ )
667 m_samples[i] = qBound( m_minValue, m_samples[i] / max, m_maxValue );
668
669 // signal changes if any
670 if( max != 1.0f || avg != 0.0f )
671 emit samplesChanged( 0, length()-1 );
672}

688void graphModel::shiftPhase( int _deg )
689{
690 // calculate offset in samples
691 const int offset = ( _deg * length() ) / 360; //multiply first because integers
692
693 // store values in temporary array
694 QVector<float> temp = m_samples;
695
696 // shift phase
697 for( int i = 0; i < length(); i++ )
698 {
699 int o = ( i + offset ) % length();
700 while( o < 0 ) o += length();
701 m_samples[i] = temp[o];
702 }
