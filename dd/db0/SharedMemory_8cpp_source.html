58template<typename F>
59int retryWhileInterrupted(F&& function) noexcept(std::is_nothrow_invocable_v<F>)
60{
61 while (true) {
62 const auto result = function();
63 if (result != -1 || errno != EINTR) { return result; }
64 }
65}

75class SharedMemoryImpl
76{
77public:
78 SharedMemoryImpl(const std::string& key, bool readOnly) :
79 m_key{'/' + key}
80 {
81 const auto openFlags = readOnly ? O_RDONLY : O_RDWR;
82 const auto fd = FileDescriptor{
83 retryWhileInterrupted([&]() noexcept { return shm_open(m_key.c_str(), openFlags, 0); })
84 };
85 if (!fd) { throwSystemError("SharedMemoryImpl: shm_open() failed"); }
86
87 auto stat = (struct stat){};
88 if (fstat(fd.get(), &stat) == -1) { throwSystemError("SharedMemoryImpl: fstat() failed"); }
89 m_size = stat.st_size;
90
91 const auto mappingProtection = readOnly ? PROT_READ : PROT_READ | PROT_WRITE;
92 m_mapping = mmap(nullptr, m_size, mappingProtection, MAP_SHARED, fd.get(), 0);
93 if (m_mapping == MAP_FAILED) { throwSystemError("SharedMemoryImpl: mmap() failed"); }
94 }
95
96 SharedMemoryImpl(const std::string& key, std::size_t size, bool readOnly) :
97 m_key{'/' + key},
98 m_size{size}
99 {
100 const auto fd = FileDescriptor{
101 retryWhileInterrupted([&]() noexcept { return shm_open(m_key.c_str(), O_RDWR | O_CREAT | O_EXCL, 0600); })
102 };
103 if (fd.get() == -1) { throwSystemError("SharedMemoryImpl: shm_open() failed"); }
104 m_object.reset(m_key.c_str());
105
106 if (retryWhileInterrupted([&]() noexcept { return ftruncate(fd.get(), m_size); }) == -1) {
107 throwSystemError("SharedMemoryImpl: ftruncate() failed");
108 }
109
110 const auto mappingProtection = readOnly ? PROT_READ : PROT_READ | PROT_WRITE;
111 m_mapping = mmap(nullptr, m_size, mappingProtection, MAP_SHARED, fd.get(), 0);
112 if (m_mapping == MAP_FAILED) { throwSystemError("SharedMemoryImpl: mmap() failed"); }
113 }
114
115 SharedMemoryImpl(const SharedMemoryImpl&) = delete;
116 SharedMemoryImpl& operator=(const SharedMemoryImpl&) = delete;
117
118 ~SharedMemoryImpl()
119 {
120 munmap(m_mapping, m_size);
121 }
122
123 auto get() const noexcept -> void* { return m_mapping; }
124 auto size_bytes() const noexcept -> std::size_t { return m_size; }
125
126private:
127 std::string m_key;
128 std::size_t m_size = 0;
129 void* m_mapping = nullptr;
130 ShmObject m_object;
131};

137template<typename T>
138auto sizeToHighAndLow(T size) -> std::pair<DWORD, DWORD>
139{
140 if constexpr (sizeof(T) <= sizeof(DWORD)) {
141 return {0, size};
142 } else {
143 return {static_cast<DWORD>(size >> 32), static_cast<DWORD>(size)};
144 }
145}

157class SharedMemoryImpl
158{
159public:
160 SharedMemoryImpl(const std::string& key, bool readOnly)
161 {
162 const auto access = readOnly ? FILE_MAP_READ : FILE_MAP_WRITE;
163 m_mapping.reset(OpenFileMappingA(access, false, key.c_str()));
164 if (!m_mapping) { throwLastError("SharedMemoryImpl: OpenFileMappingA() failed"); }
165
166 m_view.reset(MapViewOfFile(m_mapping.get(), access, 0, 0, 0));
167 if (!m_view) { throwLastError("SharedMemoryImpl: MapViewOfFile() failed"); }
168
169 MEMORY_BASIC_INFORMATION mbi;
170 if (VirtualQuery(m_view.get(), &mbi, sizeof(mbi)) == 0)
171 {
172 throwLastError("SharedMemoryImpl: VirtualQuery() failed");
173 }
174
175 m_size = static_cast<std::size_t>(mbi.RegionSize);
176 }
177
178 SharedMemoryImpl(const std::string& key, std::size_t size, bool readOnly) :
179 m_size{size}
180 {
181 const auto [high, low] = sizeToHighAndLow(size);
182 m_mapping.reset(CreateFileMappingA(INVALID_HANDLE_VALUE, nullptr, PAGE_READWRITE, high, low, key.c_str()));
183 // This constructor is supposed to create a new shared memory object,
184 // but passing the name of an existing object causes CreateFileMappingA
185 // to succeed and return a handle to that object. Thus we have to check
186 // GetLastError() too.
187 if (!m_mapping || GetLastError() == ERROR_ALREADY_EXISTS) {
188 throwLastError("SharedMemoryImpl: CreateFileMappingA() failed");
189 }
190
191 const auto access = readOnly ? FILE_MAP_READ : FILE_MAP_WRITE;
192 m_view.reset(MapViewOfFile(m_mapping.get(), access, 0, 0, 0));
193 if (!m_view) { throwLastError("SharedMemoryImpl: MapViewOfFile() failed"); }
194 }
195
196 SharedMemoryImpl(const SharedMemoryImpl&) = delete;
197 auto operator=(const SharedMemoryImpl&) -> SharedMemoryImpl& = delete;
198
199 auto get() const noexcept -> void* { return m_view.get(); }
200 auto size_bytes() const noexcept -> std::size_t { return m_size; }
201
202private:
203 UniqueHandle m_mapping;
204 FileView m_view;
205 std::size_t m_size = 0;
206};

212auto createKey() -> std::string
213{
214 // Max length (minus prepended '/') on macOS (PSHMNAMLEN=31)
215 constexpr int length = 30;
216
217 std::string key;
218 std::random_device rd;
219 auto gen = std::mt19937{rd()}; // mersenne twister, seeded
220 auto distrib = std::uniform_int_distribution{0, 15}; // hex range (0-15)
221
222 key.reserve(length + 1);
223 for (int i = 0; i < length; ++i)
224 {
225 key += "0123456789ABCDEF"[distrib(gen)];
226 }
227
228 return key;
229}
