47void
48JackMidiBufferReadQueue::ResetMidiBuffer(JackMidiBuffer *buffer)
49{
50 event_count = 0;
51 index = 0;
52 if (! buffer) {
53 jack_error("JackMidiBufferReadQueue::ResetMidiBuffer - buffer reset "
54 "to NULL");
55 } else if (! buffer->IsValid()) {
56 jack_error("JackMidiBufferReadQueue::ResetMidiBuffer - buffer reset "
57 "to invalid buffer");
58 } else {
59 uint32_t lost_events = buffer->lost_events;
60 if (lost_events) {
61 jack_error("JackMidiBufferReadQueue::ResetMidiBuffer - %d events "
62 "lost during mixdown", lost_events);
63 }
64 this->buffer = buffer;
65 event_count = buffer->event_count;
66 last_frame_time = GetLastFrame();
67 }
