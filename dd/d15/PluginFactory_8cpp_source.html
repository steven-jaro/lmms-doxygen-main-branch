62void PluginFactory::setupSearchPaths()
63{
64 // Adds a search path relative to the main executable if the path exists.
65 auto addRelativeIfExists = [](const QString & path) {
66 QDir dir(qApp->applicationDirPath());
67 if (!path.isEmpty() && dir.cd(path)) {
68 QDir::addSearchPath("plugins", dir.absolutePath());
69 }
70 };
71
72 // We're either running LMMS installed on an Unixoid or we're running a
73 // portable version like we do on Windows.
74 // We want to find our plugins in both cases:
75 // (a) Installed (Unix):
76 // e.g. binary at /usr/bin/lmms - plugin dir at /usr/lib/lmms/
77 // (b) Portable:
78 // e.g. binary at "C:/Program Files/LMMS/lmms.exe"
79 // plugins at "C:/Program Files/LMMS/plugins/"
80
81#ifndef LMMS_BUILD_WIN32
82 addRelativeIfExists("../lib/lmms"); // Installed
83#endif
84 addRelativeIfExists("plugins"); // Portable
85#ifdef PLUGIN_DIR // We may also have received a relative directory via a define
86 addRelativeIfExists(PLUGIN_DIR);
87#endif
88 // Or via an environment variable:
89 if (const char* env_path = std::getenv("LMMS_PLUGIN_DIR"))
90 QDir::addSearchPath("plugins", env_path);
91
92 QDir::addSearchPath("plugins", ConfigManager::inst()->workingDir() + "plugins");
93}

144void PluginFactory::discoverPlugins()
145{
146 DescriptorMap descriptors;
147 PluginInfoList pluginInfos;
148 m_pluginByExt.clear();
149
150 QSet<QFileInfo> files;
151 for (const QString& searchPath : QDir::searchPaths("plugins"))
152 {
153#if (QT_VERSION >= QT_VERSION_CHECK(5,14,0))
154 auto discoveredPluginList = QDir(searchPath).entryInfoList(nameFilters);
155 files.unite(QSet<QFileInfo>(discoveredPluginList.begin(), discoveredPluginList.end()));
156#else
157 files.unite(QDir(searchPath).entryInfoList(nameFilters).toSet());
158#endif
159 }
160
161 // Apply any plugin filters from environment LMMS_EXCLUDE_PLUGINS
162 filterPlugins(files);
163
164 // Cheap dependency handling: zynaddsubfx needs ZynAddSubFxCore. By loading
165 // all libraries twice we ensure that libZynAddSubFxCore is found.
166 for (const QFileInfo& file : files)
167 {
168 QLibrary(file.absoluteFilePath()).load();
169 }
170
171 for (const QFileInfo& file : files)
172 {
173 auto library = std::make_shared<QLibrary>(file.absoluteFilePath());
174 if (! library->load()) {
175 m_errors[file.baseName()] = library->errorString();
176 qWarning("%s", library->errorString().toLocal8Bit().data());
177 continue;
178 }
179
180 Plugin::Descriptor* pluginDescriptor = nullptr;
181 if (library->resolve("lmms_plugin_main"))
182 {
183 QString descriptorName = file.baseName() + "_plugin_descriptor";
184 if( descriptorName.left(3) == "lib" )
185 {
186 descriptorName = descriptorName.mid(3);
187 }
188
189 pluginDescriptor = reinterpret_cast<Plugin::Descriptor*>(library->resolve(descriptorName.toUtf8().constData()));
190 if(pluginDescriptor == nullptr)
191 {
192 qWarning() << qApp->translate("PluginFactory", "LMMS plugin %1 does not have a plugin descriptor named %2!").
193 arg(file.absoluteFilePath()).arg(descriptorName);
194 continue;
195 }
196 }
197
198 if(pluginDescriptor)
199 {
200 PluginInfo info;
201 info.file = file;
202 info.library = library;
203 info.descriptor = pluginDescriptor;
204 pluginInfos << info;
205
206 auto addSupportedFileTypes =
207 [this](QString supportedFileTypes,
208 const PluginInfo& info,
209 const Plugin::Descriptor::SubPluginFeatures::Key* key = nullptr)
210 {
211 if(!supportedFileTypes.isNull())
212 {
213 for (const QString& ext : supportedFileTypes.split(','))
214 {
215 //qDebug() << "Plugin " << info.name()
216 // << "supports" << ext;
217 PluginInfoAndKey infoAndKey;
218 infoAndKey.info = info;
219 infoAndKey.key = key
220 ? *key
221 : Plugin::Descriptor::SubPluginFeatures::Key();
222 m_pluginByExt.insert(ext, infoAndKey);
223 }
224 }
225 };
226
227 if (info.descriptor->supportedFileTypes)
228 addSupportedFileTypes(QString(info.descriptor->supportedFileTypes), info);
229
230 if (info.descriptor->subPluginFeatures)
231 {
232 Plugin::Descriptor::SubPluginFeatures::KeyList
233 subPluginKeys;
234 info.descriptor->subPluginFeatures->listSubPluginKeys(
235 info.descriptor,
236 subPluginKeys);
237 for(const Plugin::Descriptor::SubPluginFeatures::Key& key
238 : subPluginKeys)
239 {
240 addSupportedFileTypes(key.additionalFileExtensions(), info, &key);
241 }
242 }
243
244 descriptors.insert(info.descriptor->type, info.descriptor);
245 }
246 }
247
248 m_pluginInfos = pluginInfos;
249 m_descriptors = descriptors;
250}

252// Builds QList<QRegularExpression> based on environment variable envVar
253QList<QRegularExpression> PluginFactory::getExcludePatterns(const char* envVar) {
254 QList<QRegularExpression> excludePatterns;
255 QString excludePatternString = std::getenv(envVar);
256
257 if (!excludePatternString.isEmpty()) {
258 QStringList patterns = excludePatternString.split(',');
259 for (const QString& pattern : patterns) {
260 if (pattern.trimmed().isEmpty()) {
261 continue;
262 }
263 QRegularExpression regex(pattern.trimmed());
264 if (regex.isValid()) {
265 excludePatterns << regex;
266 } else {
267 qWarning() << "Invalid regular expression:" << pattern;
268 }
269 }
270 }
271 return excludePatterns;
272}

274// Filter plugins based on environment variable, e.g. export LMMS_EXCLUDE_PLUGINS="libcarla"
275void PluginFactory::filterPlugins(QSet<QFileInfo>& files) {
276 // Get filter
277 QList<QRegularExpression> excludePatterns = getExcludePatterns("LMMS_EXCLUDE_PLUGINS");
278 if (excludePatterns.isEmpty()) {
279 return;
280 }
281
282 // Get files to remove
283 QSet<QFileInfo> filesToRemove;
284 for (const QFileInfo& fileInfo : files) {
285 bool exclude = false;
286 QString filePath = fileInfo.filePath();
287
288 for (const QRegularExpression& pattern : excludePatterns) {
289 if (pattern.match(filePath).hasMatch()) {
290 exclude = true;
291 break;
292 }
293 }
294
295 if (exclude) {
296 filesToRemove.insert(fileInfo);
297 }
298 }
299
300 // Remove them
301 for (const QFileInfo& fileInfo : filesToRemove) {
302 files.remove(fileInfo);
303 }
304}
