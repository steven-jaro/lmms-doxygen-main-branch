47Controller::Controller( ControllerType _type, Model * _parent,
48 const QString & _display_name ) :
49 Model( _parent, _display_name ),
50 JournallingObject(),
51 m_valueBuffer( Engine::audioEngine()->framesPerPeriod() ),
52 m_bufferLastUpdated( -1 ),
53 m_connectionCount( 0 ),
54 m_type( _type )
55{
56 if( _type != ControllerType::Dummy && _type != ControllerType::Midi )
57 {
58 s_controllers.push_back(this);
59 // Determine which name to use
60 for ( uint i=s_controllers.size(); ; i++ )
61 {
62 QString new_name = QString( tr( "Controller %1" ) )
63 .arg( i );
64
65 // Check if name is already in use
66 bool name_used = false;
67 for (Controller * controller : s_controllers)
68 {
69 if ( controller->name() == new_name )
70 {
71 name_used = true;
72 break;
73 }
74 }
75 if ( ! name_used )
76 {
77 m_name = new_name;
78 break;
79 }
80 }
81 }
82 updateValueBuffer();
83}

185Controller * Controller::create( ControllerType _ct, Model * _parent )
186{
187 static Controller * dummy = nullptr;
188 Controller * c = nullptr;
189
190 switch( _ct )
191 {
192 case ControllerType::Dummy:
193 if (!dummy)
194 dummy = new Controller( ControllerType::Dummy, nullptr,
195 QString() );
196 c = dummy;
197 break;
198
199 case ControllerType::Lfo:
200 c = new class LfoController( _parent );
201 break;
202
203 case ControllerType::Peak:
204 //Already instantiated in EffectChain::loadSettings()
205 Q_ASSERT( false );
206 break;
207
208 case ControllerType::Midi:
209 c = new class MidiController( _parent );
210 break;
211
212 default:
213 break;
214 }
215
216 return( c );
217}

221Controller * Controller::create( const QDomElement & _this, Model * _parent )
222{
223 const auto controllerType = static_cast<ControllerType>(_this.attribute("type").toInt());
224 auto controller = controllerType == ControllerType::Peak
225 ? PeakController::getControllerBySetting(_this)
226 : create(controllerType, _parent);
227 if (controller) { controller->restoreState(_this); }
228 return controller;
229}

233bool Controller::hasModel( const Model * m ) const
234{
235 for (QObject * c : children())
236 {
237 auto am = qobject_cast<AutomatableModel*>(c);
238 if( am != nullptr )
239 {
240 if( am == m )
241 {
242 return true;
243 }
244
245 ControllerConnection * cc = am->controllerConnection();
246 if( cc != nullptr && cc->getController()->hasModel( m ) )
247 {
248 return true;
249 }
250 }
251 }
252
253 return false;
254}
