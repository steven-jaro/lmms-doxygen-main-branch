39SampleClip::SampleClip(Track* _track, Sample sample, bool isPlaying)
40 : Clip(_track)
41 , m_sample(std::move(sample))
42 , m_isPlaying(false)
43{
44 saveJournallingState( false );
45 setSampleFile( "" );
46 restoreJournallingState();
47
48 // we need to receive bpm-change-events, because then we have to
49 // change length of this Clip
50 connect( Engine::getSong(), SIGNAL(tempoChanged(lmms::bpm_t)),
51 this, SLOT(updateLength()), Qt::DirectConnection );
52 connect( Engine::getSong(), SIGNAL(timeSignatureChanged(int,int)),
53 this, SLOT(updateLength()));
54
55 //playbutton clicked or space key / on Export Song set isPlaying to false
56 connect( Engine::getSong(), SIGNAL(playbackStateChanged()),
57 this, SLOT(playbackPositionChanged()), Qt::DirectConnection );
58 //care about loops and jumps
59 connect( Engine::getSong(), SIGNAL(updateSampleTracks()),
60 this, SLOT(playbackPositionChanged()), Qt::DirectConnection );
61 //care about mute Clips
62 connect( this, SIGNAL(dataChanged()), this, SLOT(playbackPositionChanged()));
63 //care about mute track
64 connect( getTrack()->getMutedModel(), SIGNAL(dataChanged()),
65 this, SLOT(playbackPositionChanged()), Qt::DirectConnection );
66 //care about Clip position
67 connect( this, SIGNAL(positionChanged()), this, SLOT(updateTrackClips()));
68
69 updateTrackClips();
70}

155void SampleClip::setSampleFile(const QString& sf)
156{
157 // Remove any prior offset in the clip
158 setStartTimeOffset(0);
159 if (!sf.isEmpty())
160 {
161 m_sample = Sample(gui::SampleLoader::createBufferFromFile(sf));
162 updateLength();
163 }
164 else
165 {
166 // If there is no sample, make the clip a bar long
167 float nom = Engine::getSong()->getTimeSigModel().getNumerator();
168 float den = Engine::getSong()->getTimeSigModel().getDenominator();
169 changeLength(DefaultTicksPerBar * (nom / den));
170 }
171
172 emit sampleChanged();
173 emit playbackPositionChanged();
174}

267void SampleClip::saveSettings( QDomDocument & _doc, QDomElement & _this )
268{
269 if( _this.parentNode().nodeName() == "clipboard" )
270 {
271 _this.setAttribute( "pos", -1 );
272 }
273 else
274 {
275 _this.setAttribute( "pos", startPosition() );
276 }
277 _this.setAttribute( "len", length() );
278 _this.setAttribute( "muted", isMuted() );
279 _this.setAttribute( "src", sampleFile() );
280 _this.setAttribute( "off", startTimeOffset() );
281 _this.setAttribute("autoresize", QString::number(getAutoResize()));
282 if( sampleFile() == "" )
283 {
284 QString s;
285 _this.setAttribute("data", m_sample.toBase64());
286 }
287
288 _this.setAttribute( "sample_rate", m_sample.sampleRate());
289 if (const auto& c = color())
290 {
291 _this.setAttribute("color", c->name());
292 }
293 if (m_sample.reversed())
294 {
295 _this.setAttribute("reversed", "true");
296 }
297 // TODO: start- and end-frame
298}

303void SampleClip::loadSettings( const QDomElement & _this )
304{
305 if( _this.attribute( "pos" ).toInt() >= 0 )
306 {
307 movePosition( _this.attribute( "pos" ).toInt() );
308 }
309
310 if (const auto srcFile = _this.attribute("src"); !srcFile.isEmpty())
311 {
312 if (QFileInfo(PathUtil::toAbsolute(srcFile)).exists())
313 {
314 setSampleFile(srcFile);
315 }
316 else { Engine::getSong()->collectError(QString("%1: %2").arg(tr("Sample not found"), srcFile)); }
317 }
318
319 if( sampleFile().isEmpty() && _this.hasAttribute( "data" ) )
320 {
321 auto sampleRate = _this.hasAttribute("sample_rate") ? _this.attribute("sample_rate").toInt() :
322 Engine::audioEngine()->outputSampleRate();
323
324 auto buffer = gui::SampleLoader::createBufferFromBase64(_this.attribute("data"), sampleRate);
325 m_sample = Sample(std::move(buffer));
326 }
327 changeLength( _this.attribute( "len" ).toInt() );
328 setMuted( _this.attribute( "muted" ).toInt() );
329 setStartTimeOffset( _this.attribute( "off" ).toInt() );
330 setAutoResize(_this.attribute("autoresize", "1").toInt());
331
332 if (_this.hasAttribute("color"))
333 {
334 setColor(QColor{_this.attribute("color")});
335 }
336
337 if(_this.hasAttribute("reversed"))
338 {
339 m_sample.setReversed(true);
340 emit wasReversed(); // tell SampleClipView to update the view
341 }
342}
