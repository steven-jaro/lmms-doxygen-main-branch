179void jackbridge_discovery_pipe_destroy(void* pipe)
180{
181#if defined(JACKBRIDGE_DUMMY) || !defined(__WINE__)
182 // unused
183 (void)pipe;
184#else
185 carla_pipe_client_lock(pipe);
186 carla_pipe_client_write_msg(pipe, "exiting\n");
187 carla_pipe_client_flush_and_unlock(pipe);
188
189 // NOTE: no more messages are handled after this point
190 // pData->clientClosingDown = true;
191
192 for (int i=0; i < 100 && carla_pipe_client_is_running(pipe); ++i)
193 {
194 carla_msleep(50);
195 carla_pipe_client_idle(pipe);
196 }
197
198 if (carla_pipe_client_is_running(pipe))
199 carla_stderr2("jackbridge_discovery_pipe_destroy: pipe is still running!");
200
201 carla_pipe_client_destroy(pipe);
202#endif
203}
