106/*
107==============================================================================
108Name: process_sample
109Description:
110 Upsamples (x2) the input sample, generating two output samples.
111Input parameters:
112 - input: The input sample.
113Output parameters:
114 - out_0: First output sample.
115 - out_1: Second output sample.
116Throws: Nothing
117==============================================================================
118*/
119
120template <int NC>
121void Upsampler2xSseOld <NC>::process_sample (float &out_0, float &out_1, float input) noexcept
122{
123 const auto spl_in = _mm_set_ss (input);
124 const auto spl_mid = _mm_load_ps (_filter [_nbr_stages]._mem);
125 constexpr auto shuf = (2 << 0) | (3 << 2) | (0 << 4) | (0 << 6);
126 auto y = _mm_shuffle_ps (spl_mid, spl_in, shuf);
127
128 auto mem = _mm_load_ps (_filter [0]._mem);
129
130 StageProcSseV4 <_nbr_stages>::process_sample_pos (&_filter [0], y, mem);
131
132 _mm_store_ps (_filter [_nbr_stages]._mem, y);
133
134 out_0 = _mm_cvtss_f32 (_mm_shuffle_ps (y, y, 1));
135 out_1 = _mm_cvtss_f32 (y);
136}

140/*
141==============================================================================
142Name: process_block
143Description:
144 Upsamples (x2) the input sample block.
145 Input and output blocks may overlap, see assert() for details.
146Input parameters:
147 - in_ptr: Input array, containing nbr_spl samples.
148 - nbr_spl: Number of input samples to process, > 0
149Output parameters:
150 - out_0_ptr: Output sample array, capacity: nbr_spl * 2 samples.
151Throws: Nothing
152==============================================================================
153*/
154
155template <int NC>
156void Upsampler2xSseOld <NC>::process_block (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
157{
158 assert (out_ptr != nullptr);
159 assert (in_ptr != nullptr);
160 assert (out_ptr >= in_ptr + nbr_spl || in_ptr >= out_ptr + nbr_spl);
161 assert (nbr_spl > 0);
162
163 const long n4 = process_block_quad (out_ptr, in_ptr, nbr_spl);
164
165 for (long pos = n4; pos < nbr_spl; ++pos)
166 {
167 process_sample (out_ptr [pos * 2], out_ptr [pos * 2 + 1], in_ptr [pos]);
168 }
169}

211template <int NC>
212void Upsampler2xSseOld <NC>::set_single_coef (int index, double coef) noexcept
213{
214 assert (index >= 0);
215 assert (index < _nbr_stages * _stage_width);
216
217 const int stage = (index / _stage_width) + 1;
218 const int pos = (index ^ _coef_shift) & (_stage_width - 1);
219 _filter [stage]._coef [pos] = DataType (coef);
220}

224template <int NC>
225long Upsampler2xSseOld <NC>::process_block_quad (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
226{
227 constexpr auto shuf_y = (2 << 0) | (3 << 2);
228 constexpr auto shuf_0 = shuf_y | (0 << 4) | (0 << 6);
229 constexpr auto shuf_1 = shuf_y | (1 << 4) | (1 << 6);
230 constexpr auto shuf_2 = shuf_y | (2 << 4) | (2 << 6);
231 constexpr auto shuf_3 = shuf_y | (3 << 4) | (3 << 6);
232
233 const long n4 = nbr_spl & ~(4-1);
234 auto y_3 = _mm_load_ps (_filter [_nbr_stages]._mem);
235 for (long pos = 0; pos < n4; pos += 4)
236 {
237 const auto x = _mm_loadu_ps (in_ptr + pos);
238
239 auto y_0 = _mm_shuffle_ps (y_3, x, shuf_0);
240 auto mem_0 = _mm_load_ps (_filter [0]._mem);
241 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_0, mem_0);
242 _mm_store_ps (_filter [_nbr_stages]._mem, y_0);
243
244 auto y_1 = _mm_shuffle_ps (y_0, x, shuf_1);
245 auto mem_1 = _mm_load_ps (_filter [0]._mem);
246 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_1, mem_1);
247 _mm_store_ps (_filter [_nbr_stages]._mem, y_1);
248
249 auto y_2 = _mm_shuffle_ps (y_1, x, shuf_2);
250 auto mem_2 = _mm_load_ps (_filter [0]._mem);
251 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_2, mem_2);
252 _mm_store_ps (_filter [_nbr_stages]._mem, y_2);
253
254 y_3 = _mm_shuffle_ps (y_2, x, shuf_3);
255 auto mem_3 = _mm_load_ps (_filter [0]._mem);
256 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_3, mem_3);
257 _mm_store_ps (_filter [_nbr_stages]._mem, y_3);
258
259 constexpr auto shuf = (1 << 0) | (0 << 2) | (1 << 4) | (0 << 6);
260 const auto y_01 = _mm_shuffle_ps (y_0, y_1, shuf);
261 const auto y_23 = _mm_shuffle_ps (y_2, y_3, shuf);
262 _mm_storeu_ps (out_ptr + pos * 2 , y_01);
263 _mm_storeu_ps (out_ptr + pos * 2 + 4, y_23);
264 }
265
266 return n4;
267}
