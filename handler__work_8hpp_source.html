27// A helper class template to allow completion handlers to be dispatched
28// through either the new executors framework or the old invocaton hook. The
29// primary template uses the new executors framework.
30template <typename Handler,
31 typename IoExecutor = system_executor, typename HandlerExecutor
32 = typename associated_executor<Handler, IoExecutor>::type>
33class handler_work
34{
35public:
36 explicit handler_work(Handler& handler) ASIO_NOEXCEPT
37 : io_executor_(),
38 executor_(asio::get_associated_executor(handler, io_executor_))
39 {
40 }
41
42 handler_work(Handler& handler, const IoExecutor& io_ex) ASIO_NOEXCEPT
43 : io_executor_(io_ex),
44 executor_(asio::get_associated_executor(handler, io_executor_))
45 {
46 }
47
48 static void start(Handler& handler) ASIO_NOEXCEPT
49 {
50 HandlerExecutor ex(asio::get_associated_executor(handler));
51 ex.on_work_started();
52 }
53
54 static void start(Handler& handler,
55 const IoExecutor& io_ex) ASIO_NOEXCEPT
56 {
57 HandlerExecutor ex(asio::get_associated_executor(handler, io_ex));
58 ex.on_work_started();
59 io_ex.on_work_started();
60 }
61
62 ~handler_work()
63 {
64 io_executor_.on_work_finished();
65 executor_.on_work_finished();
66 }
67
68 template <typename Function>
69 void complete(Function& function, Handler& handler)
70 {
71 executor_.dispatch(ASIO_MOVE_CAST(Function)(function),
72 asio::get_associated_allocator(handler));
73 }
74
75private:
76 // Disallow copying and assignment.
77 handler_work(const handler_work&);
78 handler_work& operator=(const handler_work&);
79
80 IoExecutor io_executor_;
81 HandlerExecutor executor_;
82};

84// This specialisation dispatches a handler through the old invocation hook.
85// The specialisation is not strictly required for correctness, as the
86// system_executor will dispatch through the hook anyway. However, by doing
87// this we avoid an extra copy of the handler.
88template <typename Handler>
89class handler_work<Handler, system_executor, system_executor>
90{
91public:
92 explicit handler_work(Handler&) ASIO_NOEXCEPT {}
93 static void start(Handler&) ASIO_NOEXCEPT {}
94 ~handler_work() {}
95
96 template <typename Function>
97 void complete(Function& function, Handler& handler)
98 {
99 asio_handler_invoke_helpers::invoke(function, handler);
100 }
101
102private:
103 // Disallow copying and assignment.
104 handler_work(const handler_work&);
105 handler_work& operator=(const handler_work&);
106};
