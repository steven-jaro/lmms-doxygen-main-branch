57template <int NC>
58HalfBandNeon <NC>::HalfBandNeon () noexcept
59: _filter ()
60, _prev (0)
61, _phase (0)
62{
63 for (int phase = 0; phase < _nbr_phases; ++phase)
64 {
65 for (int i = 0; i < _nbr_stages + 1; ++i)
66 {
67 storea (_filter [phase] [i]._coef, vdupq_n_f32 (0));
68 }
69 }
70 for (int i = NBR_COEFS; i < _nbr_stages * _stage_width; ++i)
71 {
72 set_single_coef (i, 1);
73 }
74
75 clear_buffers ();
76}

107/*
108==============================================================================
109Name: process_sample
110Description:
111 Performs a half-band low-pass filtering on a sample.
112Input parameters:
113 - in: sample from the stream to be filtered.
114Returns: filtered sample.
115Throws: Nothing
116==============================================================================
117*/
118
119template <int NC>
120float HalfBandNeon <NC>::process_sample (float input) noexcept
121{
122 auto y = process_2_paths (input);
123 const auto even = vget_lane_f32 (y, 1);
124 const auto odd = vget_lane_f32 (y, 0);
125 const auto low = even + odd;
126
127 return low;
128}

132/*
133==============================================================================
134Name: process_block
135Description:
136 Performs a half-band low-pass filtering on a block of samples.
137 Input and output blocks may overlap, see assert() for details.
138Input parameters:
139 - in_ptr: Input array, containing nbr_spl samples.
140 - nbr_spl: Number of samples to output, > 0
141Output parameters:
142 - out_ptr: Array for the output samples, capacity: nbr_spl samples.
143Throws: Nothing
144==============================================================================
145*/
146
147template <int NC>
148void HalfBandNeon <NC>::process_block (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
149{
150 assert (in_ptr != nullptr);
151 assert (out_ptr != nullptr);
152 assert (out_ptr <= in_ptr || out_ptr >= in_ptr + nbr_spl);
153 assert (nbr_spl > 0);
154
155 if (_phase != 0)
156 {
157 out_ptr [0] = process_sample (in_ptr [0]);
158 ++ out_ptr;
159 ++ in_ptr;
160 -- nbr_spl;
161 }
162
163 if (nbr_spl > 0)
164 {
165 const long n4 = process_block_quad (
166 out_ptr, nullptr, in_ptr, nbr_spl, store_low, bypass
167 );
168
169 for (long pos = n4; pos < nbr_spl; ++pos)
170 {
171 out_ptr [pos] = process_sample (in_ptr [pos]);
172 }
173 }
174}

249/*
250==============================================================================
251Name: process_sample_split
252Description:
253 Splits (spectrum-wise) in half a sample from a stream. Both part are
254 results of a low-pass and a high-pass filtering, equivalent to the output
255 of process_sample() and process_sample_hpf().
256Input parameters:
257 - in: sample from the stream to be filtered.
258Output parameters:
259 - low: output sample, lower part of the spectrum.
260 - high: output sample, higher part of the spectrum.
261Throws: Nothing
262==============================================================================
263*/
264
265template <int NC>
266void HalfBandNeon <NC>::process_sample_split (float &low, float &hi, float input) noexcept
267{
268 auto y = process_2_paths (input);
269 const auto even = vget_lane_f32 (y, 1);
270 const auto odd = vget_lane_f32 (y, 0);
271 low = even + odd;
272 hi = even - odd;
273}

277/*
278==============================================================================
279Name: process_block_split
280Description:
281 Splits (spectrum-wise) in half a block of samples. Both part are
282 results of a low-pass and a high-pass filtering, equivalent to the output
283 of process_sample() and process_sample_hpf().
284 Input and output blocks may overlap, see assert() for details.
285Input parameters:
286 - in_ptr: Input array, containing nbr_spl samples.
287 - nbr_spl: Number of samples to output, > 0
288Output parameters:
289 - out_l_ptr: Array for the output samples, lower part of the spectrum
290 Capacity: nbr_spl samples.
291 - out_h_ptr: Array for the output samples, higher part of the spectrum.
292 Capacity: nbr_spl samples.
293Throws: Nothing
294==============================================================================
295*/
296
297template <int NC>
298void HalfBandNeon <NC>::process_block_split (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl) noexcept
299{
300 assert (out_l_ptr != nullptr);
301 assert (out_h_ptr != nullptr);
302 assert (in_ptr != nullptr);
303 assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl);
304 assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl);
305 assert (out_l_ptr + nbr_spl <= out_h_ptr || out_h_ptr + nbr_spl <= out_l_ptr);
306 assert (nbr_spl > 0);
307
308 if (_phase != 0)
309 {
310 process_sample_split (out_l_ptr [0], out_h_ptr [0], in_ptr [0]);
311 ++ out_l_ptr;
312 ++ out_h_ptr;
313 ++ in_ptr;
314 -- nbr_spl;
315 }
316
317 if (nbr_spl > 0)
318 {
319 const long n4 = process_block_quad (
320 out_l_ptr, out_h_ptr, in_ptr, nbr_spl, store_low, store_high
321 );
322
323 for (long pos = n4; pos < nbr_spl; ++pos)
324 {
325 process_sample_split (out_l_ptr [pos], out_h_ptr [pos], in_ptr [pos]);
326 }
327 }
328}

375template <int NC>
376void HalfBandNeon <NC>::set_single_coef (int index, double coef) noexcept
377{
378 assert (index >= 0);
379 assert (index < _nbr_stages * _stage_width);
380
381 const int stage = (index / _stage_width) + 1;
382 const int pos = (index ^ _coef_shift) & (_stage_width - 1);
383 for (int phase = 0; phase < _nbr_phases; ++phase)
384 {
385 _filter [phase] [stage]._coef [pos] = DataType (coef);
386 }
387}

391// Shared processing function, outputs both paths of the all-pass filter pair.
392// y [0] = Odd coefs
393// y [1] = Even coefs
394template <int NC>
395float32x2_t HalfBandNeon <NC>::process_2_paths (float input) noexcept
396{
397 const auto filter_ptr = _filter [_phase].data ();
398
399 const float32x2_t comb = vset_lane_f32 (input, vdup_n_f32 (_prev), 1);
400 const float32x2_t mid =
401#if ! defined (__BYTE_ORDER__) || (__BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__)
402 // Requires a little-endian architecture, which is generally the case
403 load2a (&filter_ptr [_nbr_stages]._mem [2]);
404#else
405 // Safe on any platform, but possibly slower.
406 vget_high_f32 (load4a (filter_ptr [_nbr_stages]._mem));
407#endif
408 float32x4_t y = vcombine_f32 (mid, comb);
409 float32x4_t mem = load4a (filter_ptr [0]._mem);
410
411 StageProcNeonV4 <_nbr_stages>::process_sample_pos (filter_ptr, y, mem);
412
413 _prev = input;
414 _phase = 1 - _phase;
415
416 return vget_low_f32 (y) * vdup_n_f32 (0.5f);
417}

421template <int NC>
422template <typename FL, typename FH>
423long HalfBandNeon <NC>::process_block_quad (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl, FL fnc_l, FH fnc_h) noexcept
424{
425 assert (_phase == 0);
426
427 const auto half = vdupq_n_f32 (0.5f);
428
429 const long n4 = nbr_spl & ~(4-1);
430 auto prev = _prev;
431 auto y_2 = load4a (_filter [0] [_nbr_stages]._mem);
432 auto y_3 = load4a (_filter [1] [_nbr_stages]._mem);
433 for (long pos = 0; pos < n4; pos += 4)
434 {
435 const auto x = load4u (in_ptr + pos);
436
437 const auto comb_0 = float32x2_t { prev, vgetq_lane_f32 (x, 0) };
438 auto y_0 = vcombine_f32 (vget_high_f32 (y_2), comb_0);
439 auto mem_0 = load4a (_filter [0] [0]._mem);
440 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter [0].data (), y_0, mem_0);
441 storea (_filter [0] [_nbr_stages]._mem, y_0);
442
443 auto y_1 = vcombine_f32 (vget_high_f32 (y_3), vget_low_f32 (x));
444 auto mem_1 = load4a (_filter [1] [0]._mem);
445 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter [1].data (), y_1, mem_1);
446 storea (_filter [1] [_nbr_stages]._mem, y_1);
447
448 const auto comb_2 = float32x2_t { vgetq_lane_f32 (x, 1), vgetq_lane_f32 (x, 2) };
449 y_2 = vcombine_f32 (vget_high_f32 (y_0), comb_2);
450 auto mem_2 = load4a (_filter [0] [0]._mem);
451 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter [0].data (), y_2, mem_2);
452 storea (_filter [0] [_nbr_stages]._mem, y_2);
453
454 y_3 = vcombine_f32 (vget_high_f32 (y_1), vget_high_f32 (x));
455 auto mem_3 = load4a (_filter [1] [0]._mem);
456 StageProcNeonV4 <_nbr_stages>::process_sample_pos (_filter [1].data (), y_3, mem_3);
457 storea (_filter [1] [_nbr_stages]._mem, y_3);
458
459 prev = vgetq_lane_f32 (x, 3);
460
461 const auto u_02 = vcombine_f32 (vget_low_f32 (y_0), vget_low_f32 (y_2));
462 const auto u_13 = vcombine_f32 (vget_low_f32 (y_1), vget_low_f32 (y_3));
463 const auto both = vtrnq_f32 (u_02, u_13);
464 const auto odd = both.val [0];
465 const auto even = both.val [1];
466 fnc_l (out_l_ptr + pos, even, odd, half);
467 fnc_h (out_h_ptr + pos, even, odd, half);
468 }
469 _prev = prev;
470
471 return n4;
472}
