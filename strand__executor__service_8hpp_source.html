33// Default service implementation for a strand.
34class strand_executor_service
35 : public execution_context_service_base<strand_executor_service>
36{
37public:
38 // The underlying implementation of a strand.
39 class strand_impl
40 {
41 public:
42 ASIO_DECL ~strand_impl();
43
44 private:
45 friend class strand_executor_service;
46
47 // Mutex to protect access to internal data.
48 mutex* mutex_;
49
50 // Indicates whether the strand is currently "locked" by a handler. This
51 // means that there is a handler upcall in progress, or that the strand
52 // itself has been scheduled in order to invoke some pending handlers.
53 bool locked_;
54
55 // Indicates that the strand has been shut down and will accept no further
56 // handlers.
57 bool shutdown_;
58
59 // The handlers that are waiting on the strand but should not be run until
60 // after the next time the strand is scheduled. This queue must only be
61 // modified while the mutex is locked.
62 op_queue<scheduler_operation> waiting_queue_;
63
64 // The handlers that are ready to be run. Logically speaking, these are the
65 // handlers that hold the strand's lock. The ready queue is only modified
66 // from within the strand and so may be accessed without locking the mutex.
67 op_queue<scheduler_operation> ready_queue_;
68
69 // Pointers to adjacent handle implementations in linked list.
70 strand_impl* next_;
71 strand_impl* prev_;
72
73 // The strand service in where the implementation is held.
74 strand_executor_service* service_;
75 };
