35static LICE_IBitmap *LICE_LoadSTB(const char *filename, LICE_IBitmap *bmp)
36{
37 LICE_IBitmap *delbmp = nullptr;
38 stbi_uc *srcpx = nullptr;
39 LICE_pixel *dstpx = nullptr;
40 bool dstflip = false;
41 unsigned dstspan = 0;
42 unsigned w = 0;
43 unsigned h = 0;
44 unsigned ch = 0;
45
46 srcpx = stbi_load(filename, (int *)&w, (int *)&h, (int *)&ch, 4);
47 if (!srcpx)
48 goto fail;
49
50 if (bmp)
51 bmp->resize(w, h);
52 else
53 bmp = delbmp = new WDL_NEW LICE_MemBitmap(w, h);
54
55 if (!bmp || (unsigned)bmp->getWidth() != w || (unsigned)bmp->getHeight() != h)
56 goto fail;
57
58 dstpx = bmp->getBits();
59 dstflip = bmp->isFlipped();
60 dstspan = bmp->getRowSpan();
61
62 for (unsigned row = 0; row < h; ++row) {
63 const stbi_uc *src = srcpx + row * (4 * w);
64 LICE_pixel *dst = dstpx + dstspan * (dstflip ? (h - 1 - row) : row);
65 for (unsigned col = 0; col < w; ++col, src += 4, ++dst)
66 *dst = LICE_RGBA(src[0], src[1], src[2], src[3]);
67 }
68
69 stbi_image_free(srcpx);
70 return bmp;
71
72fail:
73 delete delbmp;
74 stbi_image_free(srcpx);
75 return nullptr;
76}
