50 inline void gain( SampleFrame* buf, const fpp_t frames, float scale, SampleFrame* peak )
51 {
52 peak[0][0] = 0.0f; peak[0][1] = 0.0f;
53 for( fpp_t f = 0; f < frames; ++f )
54 {
55 auto & sf = buf[f];
56
57 // Apply gain to sample frame
58 sf[0] *= scale;
59 sf[1] *= scale;
60
61 // Update peaks
62 peak[0][0] = std::max(peak[0][0], (float)fabs(sf[0]));
63 peak[0][1] = std::max(peak[0][1], (float)fabs(sf[1]));
64 }
65 }
