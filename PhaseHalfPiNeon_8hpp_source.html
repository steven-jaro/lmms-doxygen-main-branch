57template <int NC>
58PhaseHalfPiNeon <NC>::PhaseHalfPiNeon () noexcept
59: _filter ()
60, _prev (0)
61, _phase (0)
62{
63 for (int phase = 0; phase < _nbr_phases; ++phase)
64 {
65 for (int i = 0; i < _nbr_stages + 1; ++i)
66 {
67 storea (_filter [phase] [i]._coef, vdupq_n_f32 (0));
68 }
69 }

71 {
72 set_single_coef (i, 1);
73 }
74
75 clear_buffers ();
76}

107/*
108==============================================================================
109Name: process_sample
110Description:
111 From one input sample, generates two samples with a pi/2 phase shift.
112Input parameters:
113 - input: The input sample.
114Output parameters:
115 - out_0: Output sample, ahead.
116 - out_1: Output sample, late.
117Throws: Nothing
118==============================================================================
119*/
120
121template <int NC>
122hiir_FORCEINLINE void PhaseHalfPiNeon <NC>::process_sample (float &out_0, float &out_1, float input) noexcept
123{
124 StageDataNeonV4 * filter_ptr = &_filter [_phase] [0];
125
126 const auto comb = vset_lane_f32 (input, vdup_n_f32 (_prev), 1);
127 const auto spl_mid =
128#if ! defined (__BYTE_ORDER__) || (__BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__)
129 // Requires a little-endian architecture, which is generally the case
130 load2a (filter_ptr [_nbr_stages]._mem + 2);
131#else
132 // Safe on any platform, but possibly slower.
133 vget_high_f32 (load4a (filter_ptr [_nbr_stages]._mem));
134#endif
135 auto y = vcombine_f32 (spl_mid, comb);
136 auto mem = load4a (filter_ptr [0]._mem);
137
138 StageProcNeonV4 <_nbr_stages>::process_sample_neg (&filter_ptr [0], y, mem);
139
140 out_0 = vgetq_lane_f32 (y, 1);
141 out_1 = vgetq_lane_f32 (y, 0);
142
143 _prev = input;
144 _phase = 1 - _phase;
145}

149/*
150==============================================================================
151Name: process_block
152Description:
153 From a block of samples, generates two blocks of samples, with a pi/2
154 phase shift between these signals.
155 Input and output blocks may overlap, see assert() for details.
156Input parameters:
157 - in_ptr: Input array, containing nbr_spl samples.
158 - nbr_spl: Number of input samples to process, > 0
159Output parameters:
160 - out_0_ptr: Output sample array (ahead), capacity: nbr_spl samples.
161 - out_1_ptr: Output sample array (late), capacity: nbr_spl samples.
162Throws: Nothing
163==============================================================================
164*/
165
166template <int NC>
167void PhaseHalfPiNeon <NC>::process_block (float out_0_ptr [], float out_1_ptr [], const float in_ptr [], long nbr_spl) noexcept
168{
169 assert (out_0_ptr != nullptr);
170 assert (out_1_ptr != nullptr);
171 assert (in_ptr != nullptr);
172 assert (out_0_ptr <= in_ptr || out_0_ptr >= in_ptr + nbr_spl);
173 assert (out_1_ptr <= in_ptr || out_1_ptr >= in_ptr + nbr_spl);
174 assert (out_0_ptr + nbr_spl <= out_1_ptr || out_1_ptr + nbr_spl <= out_0_ptr);
175 assert (nbr_spl > 0);
176
177 if (_phase != 0)
178 {
179 process_sample (out_0_ptr [0], out_1_ptr [0], in_ptr [0]);
180 ++ out_0_ptr;
181 ++ out_1_ptr;
182 ++ in_ptr;
183 -- nbr_spl;
184 }
185
186 if (nbr_spl > 0)
187 {
188 const long n4 =
189 process_block_quad (out_0_ptr, out_1_ptr, in_ptr, nbr_spl);
190
191 for (long pos = n4; pos < nbr_spl; ++pos)
192 {
193 process_sample (out_0_ptr [pos], out_1_ptr [pos], in_ptr [pos]);
194 }
195 }
196}

243template <int NC>
244void PhaseHalfPiNeon <NC>::set_single_coef (int index, double coef) noexcept
245{
246 assert (index >= 0);
247 assert (index < _nbr_stages * _stage_width);
248
249 const int stage = (index / _stage_width) + 1;
250 const int pos = (index ^ _coef_shift) & (_stage_width - 1);
251 for (int phase = 0; phase < _nbr_phases; ++phase)
252 {
253 _filter [phase] [stage]._coef [pos] = DataType (coef);
254 }
255}

259template <int NC>
260long PhaseHalfPiNeon <NC>::process_block_quad (float out_0_ptr [], float out_1_ptr [], const float in_ptr [], long nbr_spl) noexcept
261{
262 assert (_phase == 0);
263
264 const long n4 = nbr_spl & ~(4-1);
265 auto prev = _prev;
266 auto y_2 = load4a (_filter [0] [_nbr_stages]._mem);
267 auto y_3 = load4a (_filter [1] [_nbr_stages]._mem);
268 for (long pos = 0; pos < n4; pos += 4)
269 {
270 const auto x = load4u (in_ptr + pos);
271
272 const auto comb_0 = float32x2_t { prev, vgetq_lane_f32 (x, 0) };
273 auto y_0 = vcombine_f32 (vget_high_f32 (y_2), comb_0);
274 auto mem_0 = load4a (_filter [0] [0]._mem);
275 StageProcNeonV4 <_nbr_stages>::process_sample_neg (_filter [0].data (), y_0, mem_0);
276 storea (_filter [0] [_nbr_stages]._mem, y_0);
277
278 auto y_1 = vcombine_f32 (vget_high_f32 (y_3), vget_low_f32 (x));
279 auto mem_1 = load4a (_filter [1] [0]._mem);
280 StageProcNeonV4 <_nbr_stages>::process_sample_neg (_filter [1].data (), y_1, mem_1);
281 storea (_filter [1] [_nbr_stages]._mem, y_1);
282
283 const auto comb_2 = float32x2_t { vgetq_lane_f32 (x, 1), vgetq_lane_f32 (x, 2) };
284 y_2 = vcombine_f32 (vget_high_f32 (y_0), comb_2);
285 auto mem_2 = load4a (_filter [0] [0]._mem);
286 StageProcNeonV4 <_nbr_stages>::process_sample_neg (_filter [0].data (), y_2, mem_2);
287 storea (_filter [0] [_nbr_stages]._mem, y_2);
288
289 y_3 = vcombine_f32 (vget_high_f32 (y_1), vget_high_f32 (x));
290 auto mem_3 = load4a (_filter [1] [0]._mem);
291 StageProcNeonV4 <_nbr_stages>::process_sample_neg (_filter [1].data (), y_3, mem_3);
292 storea (_filter [1] [_nbr_stages]._mem, y_3);
293
294 prev = vgetq_lane_f32 (x, 3);
295
296 const auto u_02 = vcombine_f32 (vget_low_f32 (y_0), vget_low_f32 (y_2));
297 const auto u_13 = vcombine_f32 (vget_low_f32 (y_1), vget_low_f32 (y_3));
298 const auto both = vtrnq_f32 (u_02, u_13);
299 const auto odd = both.val [0];
300 const auto even = both.val [1];
301 storeu (out_0_ptr + pos, even);
302 storeu (out_1_ptr + pos, odd);
303 }
304 _prev = prev;
305
306 return n4;
307}
