108/*
109==============================================================================
110Name: process_sample
111Description:
112 Downsamples (x2) one pair of samples, to generate one output sample.
113Input parameters:
114 - in_ptr: pointer on the two samples to decimate
115Returns: Samplerate-reduced sample.
116Throws: Nothing
117==============================================================================
118*/
119
120template <int NC>
121float Downsampler2xSseOld <NC>::process_sample (const float in_ptr [2]) noexcept
122{
123 assert (in_ptr != nullptr);
124
125 // Combines two input samples and two mid-processing data
126 const auto spl_in = _mm_loadu_ps (in_ptr);
127 const auto spl_mid = _mm_load_ps (_filter [_nbr_stages]._mem);
128 constexpr auto shuf = (2 << 0) | (3 << 2) | (0 << 4) | (1 << 6);
129 auto y = _mm_shuffle_ps (spl_mid, spl_in, shuf);
130
131 auto mem = _mm_load_ps (_filter [0]._mem);
132
133 // Processes each stage
134 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y, mem);
135
136 _mm_store_ps (_filter [_nbr_stages]._mem, y);
137
138 // Averages both paths and outputs the result
139 // Outputs the result
140 y = _mm_mul_ps (y, _mm_set1_ps (0.5f));
141 const auto even = _mm_cvtss_f32 (_mm_shuffle_ps (y, y, 1));
142 const auto odd = _mm_cvtss_f32 (y);
143 const auto low = even + odd;
144
145 return low;
146}

165template <int NC>
166void Downsampler2xSseOld <NC>::process_block (float out_ptr [], const float in_ptr [], long nbr_spl) noexcept
167{
168 assert (in_ptr != nullptr);
169 assert (out_ptr != nullptr);
170 assert (out_ptr <= in_ptr || out_ptr >= in_ptr + nbr_spl * 2);
171 assert (nbr_spl > 0);
172
173 const long n4 = process_block_quad (
174 out_ptr, nullptr, in_ptr, nbr_spl, store_low, bypass
175 );
176
177 for (long pos = n4; pos < nbr_spl; ++pos)
178 {
179 out_ptr [pos] = process_sample (in_ptr + pos * 2);
180 }
181}

185/*
186==============================================================================
187Name: process_sample_split
188Description:
189 Split (spectrum-wise) in half a pair of samples. The lower part of the
190 spectrum is a classic downsampling, equivalent to the output of
191 process_sample().
192 The higher part is the complementary signal: original filter response
193 is flipped from left to right, becoming a high-pass filter with the same
194 cutoff frequency. This signal is then critically sampled (decimation by 2),
195 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
196Input parameters:
197 - in_ptr: pointer on the pair of input samples.
198Output parameters:
199 - low: output sample, lower part of the spectrum (downsampling)
200 - high: output sample, higher part of the spectrum.
201Throws: Nothing
202==============================================================================
203*/
204
205template <int NC>
206void Downsampler2xSseOld <NC>::process_sample_split (float &low, float &high, const float in_ptr [2]) noexcept
207{
208 assert (in_ptr != nullptr);
209
210 // Combines two input samples and two mid-processing data
211 const auto spl_in = _mm_loadu_ps (in_ptr);
212 const auto spl_mid = _mm_load_ps (_filter [_nbr_stages]._mem);
213 constexpr auto shuf = (2 << 0) | (3 << 2) | (0 << 4) | (1 << 6);
214 auto y = _mm_shuffle_ps (spl_mid, spl_in, shuf);
215
216 auto mem = _mm_load_ps (_filter [0]._mem);
217
218 // Processes each stage
219 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y, mem);
220
221 _mm_store_ps (_filter [_nbr_stages]._mem, y);
222
223 // Outputs the result
224 y = _mm_mul_ps (y, _mm_set1_ps (0.5f));
225 const auto even = _mm_cvtss_f32 (_mm_shuffle_ps (y, y, 1));
226 const auto odd = _mm_cvtss_f32 (y);
227 low = even + odd;
228 high = even - odd;
229}

233/*
234==============================================================================
235Name: process_block_split
236Description:
237 Split (spectrum-wise) in half a block of samples. The lower part of the
238 spectrum is a classic downsampling, equivalent to the output of
239 process_block().
240 The higher part is the complementary signal: original filter response
241 is flipped from left to right, becoming a high-pass filter with the same
242 cutoff frequency. This signal is then critically sampled (decimation by 2),
243 flipping the spectrum: Fs/4...Fs/2 becomes Fs/4...0.
244 Input and output blocks may overlap, see assert() for details.
245Input parameters:
246 - in_ptr: Input array, containing nbr_spl * 2 samples.
247 - nbr_spl: Number of samples for each output, > 0
248Output parameters:
249 - out_l_ptr: Array for the output samples, lower part of the spectrum
250 (downsampling). Capacity: nbr_spl samples.
251 - out_h_ptr: Array for the output samples, higher part of the spectrum.
252 Capacity: nbr_spl samples.
253Throws: Nothing
254==============================================================================
255*/
256
257template <int NC>
258void Downsampler2xSseOld <NC>::process_block_split (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl) noexcept
259{
260 assert (in_ptr != nullptr);
261 assert (out_l_ptr != nullptr);
262 assert (out_l_ptr <= in_ptr || out_l_ptr >= in_ptr + nbr_spl * 2);
263 assert (out_h_ptr != nullptr);
264 assert (out_h_ptr <= in_ptr || out_h_ptr >= in_ptr + nbr_spl * 2);
265 assert (out_h_ptr != out_l_ptr);
266 assert (nbr_spl > 0);
267
268 const long n4 = process_block_quad (
269 out_l_ptr, out_h_ptr, in_ptr, nbr_spl, store_low, store_high
270 );
271
272 for (long pos = n4; pos < nbr_spl; ++pos)
273 {
274 process_sample_split (out_l_ptr [pos], out_h_ptr [pos], in_ptr + pos * 2);
275 }
276}

320{
321 assert (index >= 0);
322 assert (index < _nbr_stages * _stage_width);
323

331template <int NC>
332template <typename FL, typename FH>
333long Downsampler2xSseOld <NC>::process_block_quad (float out_l_ptr [], float out_h_ptr [], const float in_ptr [], long nbr_spl, FL fnc_l, FH fnc_h) noexcept
334{
335 const auto half = _mm_set1_ps (0.5f);
336
337 constexpr auto shuf_y = (2 << 0) | (3 << 2);
338 constexpr auto shuf_0 = shuf_y | (0 << 4) | (1 << 6);
339 constexpr auto shuf_1 = shuf_y | (2 << 4) | (3 << 6);
340
341 const long n4 = nbr_spl & ~(4-1);
342 auto y_3 = _mm_load_ps (_filter [_nbr_stages]._mem);
343 for (long pos = 0; pos < n4; pos += 4)
344 {
345 const auto x_0 = _mm_loadu_ps (in_ptr + pos * 2 );
346 const auto x_2 = _mm_loadu_ps (in_ptr + pos * 2 + 4);
347
348 auto y_0 = _mm_shuffle_ps (y_3, x_0, shuf_0);
349 auto mem_0 = _mm_load_ps (_filter [0]._mem);
350 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_0, mem_0);
351 _mm_store_ps (_filter [_nbr_stages]._mem, y_0);
352
353 auto y_1 = _mm_shuffle_ps (y_0, x_0, shuf_1);
354 auto mem_1 = _mm_load_ps (_filter [0]._mem);
355 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_1, mem_1);
356 _mm_store_ps (_filter [_nbr_stages]._mem, y_1);
357
358 auto y_2 = _mm_shuffle_ps (y_1, x_2, shuf_0);
359 auto mem_2 = _mm_load_ps (_filter [0]._mem);
360 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_2, mem_2);
361 _mm_store_ps (_filter [_nbr_stages]._mem, y_2);
362
363 y_3 = _mm_shuffle_ps (y_2, x_2, shuf_1);
364 auto mem_3 = _mm_load_ps (_filter [0]._mem);
365 StageProcSseV4 <_nbr_stages>::process_sample_pos (_filter.data (), y_3, mem_3);
366 _mm_store_ps (_filter [_nbr_stages]._mem, y_3);
367
368 const auto u_01 = _mm_unpacklo_ps (y_0, y_1); // o0, o1, e0, e1
369 const auto u_23 = _mm_unpacklo_ps (y_2, y_3); // o2, o3, e2, e3
370 const auto odd = _mm_movelh_ps (u_01, u_23);
371 const auto even = _mm_movehl_ps (u_23, u_01);
372 fnc_l (out_l_ptr + pos, even, odd, half);
373 fnc_h (out_h_ptr + pos, even, odd, half);
374 }
375
376 return n4;
377}
