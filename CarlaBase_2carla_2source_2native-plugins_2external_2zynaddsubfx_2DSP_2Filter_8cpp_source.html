35Filter *Filter::generate(Allocator &memory, const FilterParams *pars,
36 unsigned int srate, int bufsize)
37{
38 assert(srate != 0);
39 assert(bufsize != 0);
40
41 unsigned char Ftype = pars->Ptype;
42 unsigned char Fstages = pars->Pstages;
43
44 Filter *filter;
45 switch(pars->Pcategory) {
46 case 1:
47 filter = memory.alloc<FormantFilter>(pars, &memory, srate, bufsize);
48 break;
49 case 2:
50 filter = memory.alloc<SVFilter>(Ftype, 1000.0f, pars->getq(), Fstages, srate, bufsize);
51 filter->outgain = dB2rap(pars->getgain());
52 if(filter->outgain > 1.0f)
53 filter->outgain = sqrt(filter->outgain);
54 break;
55 default:
56 filter = memory.alloc<AnalogFilter>(Ftype, 1000.0f, pars->getq(), Fstages, srate, bufsize);
57 if((Ftype >= 6) && (Ftype <= 8))
58 filter->setgain(pars->getgain());
59 else
60 filter->outgain = dB2rap(pars->getgain());
61 break;
62 }
63 return filter;
64}
