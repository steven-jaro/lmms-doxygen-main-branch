92struct initiate_co_spawn
93{
94 template <typename Handler, typename Executor, typename F>
95 void operator()(Handler&& handler, const Executor& ex, F&& f) const
96 {
97 typedef typename result_of<F()>::type awaitable_type;
98 typedef typename awaitable_type::executor_type executor_type;
99
100 executor_type ex2(ex);

102 ex2, std::forward<F>(f), std::forward<Handler>(handler));
103 awaitable_handler<executor_type, void>(std::move(a), ex2).launch();
