40bool DialogWindow::escapeKeyPressed()
41{

51bool DialogWindow::keyPressed (const KeyPress& key)

59void DialogWindow::resized()
60{
61 DocumentWindow::resized();
62
63 if (escapeKeyTriggersCloseButton)
64 {
65 if (auto* close = getCloseButton())
66 {
67 const KeyPress esc (KeyPress::escapeKey, 0, 0);
68
69 if (! close->isRegisteredForShortcut (esc))
70 close->addShortcut (esc);
71 }
72 }
73}

93

95 setResizable (options.resizable, options.useBottomRightCornerResizer);
96 }

104 JUCE_DECLARE_NON_COPYABLE (DefaultDialogWindow)
