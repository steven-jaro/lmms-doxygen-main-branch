69SVFilter::response SVFilter::computeResponse(int type,
70 float freq, float pq, int stages, float gain, float fs)
71{
72 typedef SVFilter::response res;
73 float f = freq / fs * 4.0;
74 if(f > 0.99999f)
75 f = 0.99999f;
76 float q = 1.0f - atanf(sqrtf(pq)) * 2.0f / PI;
77 q = powf(q, 1.0f / (stages + 1));
78 float qrt = sqrtf(q);
79 float g = powf(gain, 1.0 / (stages + 1));
80 if(type == 0) { //Low
81 return res{0, g*f*f*qrt, 0,
82 1, (q*f+f*f-2), (1-q*f)};
83 }
84 if(type == 1) {//High
85 //g *= qrt/(1+f*q);
86 g *= qrt;
87 return res{g, -2*g, g,
88 //1, (f*f-2*f*q-2)/(1+f*q), 1};
89 1, (q*f+f*f-2), (1-q*f)};
90 }
91 if(type == 2) {//Band
92 g *= f*qrt;
93 return res{g, -g, 0,
94 1, (q*f+f*f-2), (1-q*f)};
95 }
96 if(type == 3 || true) {//Notch
97 g *= qrt;
98 return res{g, -2*g+g*f*f, g,
99 1, (q*f+f*f-2), (1-q*f)};
100 }
101}

103void SVFilter::computefiltercoefs(void)
104{
105 par.f = freq / samplerate_f * 4.0f;
106 if(par.f > 0.99999f)
107 par.f = 0.99999f;
108 par.q = 1.0f - atanf(sqrtf(q)) * 2.0f / PI;
109 par.q = powf(par.q, 1.0f / (stages + 1));
110 par.q_sqrt = sqrtf(par.q);
111}

114void SVFilter::setfreq(float frequency)
115{
116 if(frequency < 0.1f)
117 frequency = 0.1f;
118 float rap = freq / frequency;
119 if(rap < 1.0f)
120 rap = 1.0f / rap;
121
122 oldabovenq = abovenq;
123 abovenq = frequency > (samplerate_f / 2 - 500.0f);
124
125 bool nyquistthresh = (abovenq ^ oldabovenq);
126
127 //if the frequency is changed fast, it needs interpolation
128 if((rap > 3.0f) || nyquistthresh) { //(now, filter and coefficients backup)
129 if(!firsttime)
130 needsinterpolation = true;
131 ipar = par;
132 }
133 freq = frequency;
134 computefiltercoefs();
135 firsttime = false;
136}

171void SVFilter::singlefilterout(float *smp, fstage &x, parameters &par)
172{
173 float *out = NULL;
174 switch(type) {
175 case 0:
176 out = &x.low;
177 break;
178 case 1:
179 out = &x.high;
180 break;
181 case 2:
182 out = &x.band;
183 break;
184 case 3:
185 out = &x.notch;
186 break;
187 default:
188 out = &x.low;
189 warnx("Impossible SVFilter type encountered [%d]", type);
190 }
191
192 for(int i = 0; i < buffersize; ++i) {
193 x.low = x.low + par.f * x.band;
194 x.high = par.q_sqrt * smp[i] - x.low - par.q * x.band;
195 x.band = par.f * x.high + x.band;
196 x.notch = x.high + x.low;
197 smp[i] = *out;
198 }
199}
200
201// simplifying the responses
202// xl = xl*z(-1) + pf*xb*z(-1)
203// xh = pq1*x - xl - pq*xb*z(-1)
204// xb = pf*xh + xb*z(-1)
205// xn = xh + xl
206//
207// xl = pf*xb*z(-1)/(1-z(-1))
208// xb = pf*xh/(1-z(-1))
209// xl = pf*pfxh*z(-1)/(1-z(-1))^2
210
211
212void SVFilter::filterout(float *smp)
213{
214 for(int i = 0; i < stages + 1; ++i)
215 singlefilterout(smp, st[i], par);
216
217 if(needsinterpolation) {
218 float ismp[buffersize];
219 memcpy(ismp, smp, bufferbytes);
220
221 for(int i = 0; i < stages + 1; ++i)
222 singlefilterout(ismp, st[i], ipar);
223
224 for(int i = 0; i < buffersize; ++i) {
225 float x = i / buffersize_f;
226 smp[i] = ismp[i] * (1.0f - x) + smp[i] * x;
227 }
228 needsinterpolation = false;
229 }
230
231 for(int i = 0; i < buffersize; ++i)
232 smp[i] *= outgain;
233}
