30template <typename Stream, typename Operation>
31std::size_t io(Stream& next_layer, stream_core& core,
32 const Operation& op, asio::error_code& ec)
33{
34 asio::error_code io_ec;
35 std::size_t bytes_transferred = 0;
36 do switch (op(core.engine_, ec, bytes_transferred))
37 {
38 case engine::want_input_and_retry:
39
40 // If the input buffer is empty then we need to read some more data from
41 // the underlying transport.
42 if (core.input_.size() == 0)
43 {
44 core.input_ = asio::buffer(core.input_buffer_,
45 next_layer.read_some(core.input_buffer_, io_ec));
46 if (!ec)
47 ec = io_ec;
48 }
49
50 // Pass the new input data to the engine.
51 core.input_ = core.engine_.put_input(core.input_);
52
53 // Try the operation again.
54 continue;
55
56 case engine::want_output_and_retry:
57
58 // Get output data from the engine and write it to the underlying
59 // transport.
60 asio::write(next_layer,
61 core.engine_.get_output(core.output_buffer_), io_ec);
62 if (!ec)
63 ec = io_ec;
64
65 // Try the operation again.
66 continue;
67
68 case engine::want_output:
69
70 // Get output data from the engine and write it to the underlying
71 // transport.
72 asio::write(next_layer,
73 core.engine_.get_output(core.output_buffer_), io_ec);
74 if (!ec)
75 ec = io_ec;
76
77 // Operation is complete. Return result to caller.
78 core.engine_.map_error_code(ec);
79 return bytes_transferred;
80
81 default:
82
83 // Operation is complete. Return result to caller.
84 core.engine_.map_error_code(ec);
85 return bytes_transferred;
86
87 } while (!ec);
88
89 // Operation failed. Return result to caller.
90 core.engine_.map_error_code(ec);
91 return 0;
92}

94template <typename Stream, typename Operation, typename Handler>
95class io_op
96{
97public:
98 io_op(Stream& next_layer, stream_core& core,
99 const Operation& op, Handler& handler)
100 : next_layer_(next_layer),
101 core_(core),
102 op_(op),
103 start_(0),
104 want_(engine::want_nothing),
105 bytes_transferred_(0),
106 handler_(ASIO_MOVE_CAST(Handler)(handler))
107 {
108 }
109
110#if defined(ASIO_HAS_MOVE)
111 io_op(const io_op& other)
112 : next_layer_(other.next_layer_),
113 core_(other.core_),
114 op_(other.op_),
115 start_(other.start_),
116 want_(other.want_),
117 ec_(other.ec_),
118 bytes_transferred_(other.bytes_transferred_),
119 handler_(other.handler_)
120 {
121 }
122
123 io_op(io_op&& other)
124 : next_layer_(other.next_layer_),
125 core_(other.core_),
126 op_(ASIO_MOVE_CAST(Operation)(other.op_)),
127 start_(other.start_),
128 want_(other.want_),
129 ec_(other.ec_),
130 bytes_transferred_(other.bytes_transferred_),
131 handler_(ASIO_MOVE_CAST(Handler)(other.handler_))
132 {
133 }
134#endif // defined(ASIO_HAS_MOVE)
135
136 void operator()(asio::error_code ec,
137 std::size_t bytes_transferred = ~std::size_t(0), int start = 0)
138 {
139 switch (start_ = start)
140 {
141 case 1: // Called after at least one async operation.
142 do
143 {
144 switch (want_ = op_(core_.engine_, ec_, bytes_transferred_))
145 {
146 case engine::want_input_and_retry:
147
148 // If the input buffer already has data in it we can pass it to the
149 // engine and then retry the operation immediately.
150 if (core_.input_.size() != 0)
151 {
152 core_.input_ = core_.engine_.put_input(core_.input_);
153 continue;
154 }
155
156 // The engine wants more data to be read from input. However, we
157 // cannot allow more than one read operation at a time on the
158 // underlying transport. The pending_read_ timer's expiry is set to
159 // pos_infin if a read is in progress, and neg_infin otherwise.
160 if (core_.expiry(core_.pending_read_) == core_.neg_infin())
161 {
162 // Prevent other read operations from being started.
163 core_.pending_read_.expires_at(core_.pos_infin());
164
165 // Start reading some data from the underlying transport.
166 next_layer_.async_read_some(
167 asio::buffer(core_.input_buffer_),
168 ASIO_MOVE_CAST(io_op)(*this));
169 }
170 else
171 {
172 // Wait until the current read operation completes.
173 core_.pending_read_.async_wait(ASIO_MOVE_CAST(io_op)(*this));
174 }
175
176 // Yield control until asynchronous operation completes. Control
177 // resumes at the "default:" label below.
178 return;
179
180 case engine::want_output_and_retry:
181 case engine::want_output:
182
183 // The engine wants some data to be written to the output. However, we
184 // cannot allow more than one write operation at a time on the
185 // underlying transport. The pending_write_ timer's expiry is set to
186 // pos_infin if a write is in progress, and neg_infin otherwise.
187 if (core_.expiry(core_.pending_write_) == core_.neg_infin())
188 {
189 // Prevent other write operations from being started.
190 core_.pending_write_.expires_at(core_.pos_infin());
191
192 // Start writing all the data to the underlying transport.
193 asio::async_write(next_layer_,
194 core_.engine_.get_output(core_.output_buffer_),
195 ASIO_MOVE_CAST(io_op)(*this));
196 }
197 else
198 {
199 // Wait until the current write operation completes.
200 core_.pending_write_.async_wait(ASIO_MOVE_CAST(io_op)(*this));
201 }
202
203 // Yield control until asynchronous operation completes. Control
204 // resumes at the "default:" label below.
205 return;
206
207 default:
208
209 // The SSL operation is done and we can invoke the handler, but we
210 // have to keep in mind that this function might be being called from
211 // the async operation's initiating function. In this case we're not
212 // allowed to call the handler directly. Instead, issue a zero-sized
213 // read so the handler runs "as-if" posted using io_context::post().
214 if (start)
215 {
216 next_layer_.async_read_some(
217 asio::buffer(core_.input_buffer_, 0),
218 ASIO_MOVE_CAST(io_op)(*this));
219
220 // Yield control until asynchronous operation completes. Control
221 // resumes at the "default:" label below.
222 return;
223 }
224 else
225 {
226 // Continue on to run handler directly.
227 break;
228 }
229 }
230
231 default:
232 if (bytes_transferred == ~std::size_t(0))
233 bytes_transferred = 0; // Timer cancellation, no data transferred.
234 else if (!ec_)
235 ec_ = ec;
236
237 switch (want_)
238 {
239 case engine::want_input_and_retry:
240
241 // Add received data to the engine's input.
242 core_.input_ = asio::buffer(
243 core_.input_buffer_, bytes_transferred);
244 core_.input_ = core_.engine_.put_input(core_.input_);
245
246 // Release any waiting read operations.
247 core_.pending_read_.expires_at(core_.neg_infin());
248
249 // Try the operation again.
250 continue;
251
252 case engine::want_output_and_retry:
253
254 // Release any waiting write operations.
255 core_.pending_write_.expires_at(core_.neg_infin());
256
257 // Try the operation again.
258 continue;
259
260 case engine::want_output:
261
262 // Release any waiting write operations.
263 core_.pending_write_.expires_at(core_.neg_infin());
264
265 // Fall through to call handler.
266
267 default:
268
269 // Pass the result to the handler.
270 op_.call_handler(handler_,
271 core_.engine_.map_error_code(ec_),
272 ec_ ? 0 : bytes_transferred_);
273
274 // Our work here is done.
275 return;
276 }
277 } while (!ec_);
278
279 // Operation failed. Pass the result to the handler.
280 op_.call_handler(handler_, core_.engine_.map_error_code(ec_), 0);
281 }
282 }

286 stream_core& core_;
287 Operation op_;
288 int start_;
289 engine::want want_;
290 asio::error_code ec_;
291 std::size_t bytes_transferred_;
292 Handler handler_;
293};
