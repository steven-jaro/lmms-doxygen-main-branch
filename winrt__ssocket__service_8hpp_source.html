110 // Open a new socket implementation.
111 asio::error_code open(implementation_type& impl,
112 const protocol_type& protocol, asio::error_code& ec)
113 {
114 if (is_open(impl))
115 {
116 ec = asio::error::already_open;
117 return ec;
118 }
119
120 try
121 {
122 impl.socket_ = ref new Windows::Networking::Sockets::StreamSocket;
123 impl.protocol_ = protocol;
124 ec = asio::error_code();
125 }
126 catch (Platform::Exception^ e)
127 {
128 ec = asio::error_code(e->HResult,
129 asio::system_category());
130 }
131
132 return ec;
133 }

212 // Start an asynchronous connect.
213 template <typename Handler, typename IoExecutor>
214 void async_connect(implementation_type& impl,
215 const endpoint_type& peer_endpoint,
216 Handler& handler, const IoExecutor& io_ex)
217 {
218 bool is_continuation =
219 asio_handler_cont_helpers::is_continuation(handler);
220
221 // Allocate and construct an operation to wrap the handler.
222 typedef winrt_socket_connect_op<Handler, IoExecutor> op;
223 typename op::ptr p = { asio::detail::addressof(handler),
224 op::ptr::allocate(handler), 0 };
225 p.p = new (p.v) op(handler, io_ex);
226
227 ASIO_HANDLER_CREATION((scheduler_.context(),
228 *p.p, "socket", &impl, 0, "async_connect"));
229
230 start_connect_op(impl, peer_endpoint.data(), p.p, is_continuation);
231 p.v = p.p = 0;
232 }
