124static inline
125void fillEngineEventsFromWaterMidiBuffer(EngineEvent engineEvents[kMaxEngineEventInternalCount], const water::MidiBuffer& midiBuffer)
126{
127 const uint8_t* midiData;
128 int numBytes, sampleNumber;
129 ushort engineEventIndex = 0;
130
131 for (ushort i=0; i < kMaxEngineEventInternalCount; ++i)
132 {
133 const EngineEvent& engineEvent(engineEvents[i]);
134
135 if (engineEvent.type != kEngineEventTypeNull)
136 continue;
137
138 engineEventIndex = i;
139 break;
140 }
141
142 for (water::MidiBuffer::Iterator midiBufferIterator(midiBuffer); midiBufferIterator.getNextEvent(midiData, numBytes, sampleNumber) && engineEventIndex < kMaxEngineEventInternalCount;)
143 {
144 CARLA_SAFE_ASSERT_CONTINUE(numBytes > 0);
145 CARLA_SAFE_ASSERT_CONTINUE(sampleNumber >= 0);
146 CARLA_SAFE_ASSERT_CONTINUE(numBytes < 0xFF /* uint8_t max */);
147
148 EngineEvent& engineEvent(engineEvents[engineEventIndex++]);
149
150 engineEvent.time = static_cast<uint32_t>(sampleNumber);
151 engineEvent.fillFromMidiData(static_cast<uint8_t>(numBytes), midiData, 0);
152 }
153}

157static inline
158void fillWaterMidiBufferFromEngineEvents(water::MidiBuffer& midiBuffer, const EngineEvent engineEvents[kMaxEngineEventInternalCount])
159{
160 uint8_t size = 0;
161 uint8_t mdata[3] = { 0, 0, 0 };
162 uint8_t mdataTmp[EngineMidiEvent::kDataSize];
163 const uint8_t* mdataPtr;
164
165 for (ushort i=0; i < kMaxEngineEventInternalCount; ++i)
166 {
167 const EngineEvent& engineEvent(engineEvents[i]);
168
169 /**/ if (engineEvent.type == kEngineEventTypeNull)
170 {
171 break;
172 }
173 else if (engineEvent.type == kEngineEventTypeControl)
174 {
175 const EngineControlEvent& ctrlEvent(engineEvent.ctrl);
176
177 size = ctrlEvent.convertToMidiData(engineEvent.channel, mdata);
178 mdataPtr = mdata;
179 }
180 else if (engineEvent.type == kEngineEventTypeMidi)
181 {
182 const EngineMidiEvent& midiEvent(engineEvent.midi);
183
184 size = midiEvent.size;
185 CARLA_SAFE_ASSERT_CONTINUE(size > 0);
186
187 if (size > EngineMidiEvent::kDataSize)
188 {
189 CARLA_SAFE_ASSERT_CONTINUE(midiEvent.dataExt != nullptr);
190 mdataPtr = midiEvent.dataExt;
191 }
192 else
193 {
194 // set first byte
195 mdataTmp[0] = static_cast<uint8_t>(midiEvent.data[0] | (engineEvent.channel & MIDI_CHANNEL_BIT));
196
197 if (size > 1)
198 {
199 // copy rest
200 carla_copy<uint8_t>(mdataTmp+1, midiEvent.data+1, size-1U);
201 }
202
203 // done
204 mdataPtr = mdataTmp;
205 }
206 }
207 else
208 {
209 continue;
210 }
211
212 if (size > 0)
213 midiBuffer.addEvent(mdataPtr, static_cast<int>(size), static_cast<int>(engineEvent.time));
214 }
215}
