80template <typename ConstBufferSequence>
81inline Windows::Storage::Streams::IBuffer^ buffer_dup(
82 const ConstBufferSequence& buffers)
83{
84 using Microsoft::WRL::ComPtr;
85 using asio::buffer_size;
86 std::size_t size = buffer_size(buffers);
87 auto b = ref new Windows::Storage::Streams::Buffer(size);
88 ComPtr<IInspectable> insp = reinterpret_cast<IInspectable*>(b);
89 ComPtr<Windows::Storage::Streams::IBufferByteAccess> bacc;
90 insp.As(&bacc);
91 byte* bytes = nullptr;
92 bacc->Buffer(&bytes);
93 asio::buffer_copy(asio::buffer(bytes, size), buffers);
94 b->Length = size;
95 return b;
96}
