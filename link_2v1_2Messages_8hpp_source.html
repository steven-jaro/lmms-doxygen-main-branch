33const std::size_t kMaxMessageSize = 512;

73// Types that are only used in the sending/parsing of messages, not
74// publicly exposed.
75using ProtocolHeader = std::array<char, 8>;
76const ProtocolHeader kProtocolHeader = {{'_', 'l', 'i', 'n', 'k', '_', 'v', 1}};

78// Must have at least kMaxMessageSize bytes available in the output stream
79template <typename Payload, typename It>
80It encodeMessage(const MessageType messageType, const Payload& payload, It out)
81{
82 using namespace std;
83 const MessageHeader header = {messageType};
84 const auto messageSize =
85 kProtocolHeader.size() + sizeInByteStream(header) + sizeInByteStream(payload);
86
87 if (messageSize < kMaxMessageSize)
88 {
89 return toNetworkByteStream(
90 payload, toNetworkByteStream(
91 header, copy(begin(kProtocolHeader), end(kProtocolHeader), move(out))));
92 }
93 else
94 {
95 throw range_error("Exceeded maximum message size");
96 }
97}
98
99} // namespace detail

113template <typename It>
114std::pair<MessageHeader, It> parseMessageHeader(It bytesBegin, const It bytesEnd)
115{
116 using ItDiff = typename std::iterator_traits<It>::difference_type;
117
118 MessageHeader header = {};
119 const auto protocolHeaderSize = discovery::sizeInByteStream(detail::kProtocolHeader);
120 const auto minMessageSize =
121 static_cast<ItDiff>(protocolHeaderSize + sizeInByteStream(header));
122
123 // If there are enough bytes in the stream to make a header and if
124 // the first bytes in the stream are the protocol header, then
125 // proceed to parse the stream.
126 if (std::distance(bytesBegin, bytesEnd) >= minMessageSize
127 && std::equal(
128 begin(detail::kProtocolHeader), end(detail::kProtocolHeader), bytesBegin))
129 {
130 std::tie(header, bytesBegin) =
131 MessageHeader::fromNetworkByteStream(bytesBegin + protocolHeaderSize, bytesEnd);
132 }
133 return std::make_pair(std::move(header), std::move(bytesBegin));
134}
