42//Update the changed parameters
43void EffectLFO::updateparams(void)
44{
45 float lfofreq = (powf(2.0f, Pfreq / 127.0f * 10.0f) - 1.0f) * 0.03f;
46 incx = fabsf(lfofreq) * buffersize_f / samplerate_f;
47 if(incx > 0.49999999f)
48 incx = 0.499999999f; //Limit the Frequency
49
50 lfornd = Prandomness / 127.0f;
51 lfornd = (lfornd > 1.0f) ? 1.0f : lfornd;
52
53 if(PLFOtype > 1)
54 PLFOtype = 1; //this has to be updated if more lfo's are added
55 lfotype = PLFOtype;
56 xr = xl + (Pstereo - 64.0f) / 127.0f + 1.0f;
57 xr -= floorf(xr);
58}

61//Compute the shape of the LFO
62float EffectLFO::getlfoshape(float x)
63{
64 float out;
65 switch(lfotype) {
66 case 1: //EffectLFO_TRIANGLE
67 if((x > 0.0f) && (x < 0.25f))
68 out = 4.0f * x;
69 else
70 if((x > 0.25f) && (x < 0.75f))
71 out = 2.0f - 4.0f * x;
72 else
73 out = 4.0f * x - 4.0f;
74 break;
75 //when adding more, ensure ::updateparams() gets updated
76 default:
77 out = cosf(x * 2.0f * PI); //EffectLFO_SINE // TODO: use M_PI ?
78 }
79 return out;
80}

82//LFO output
83void EffectLFO::effectlfoout(float *outl, float *outr)
84{
85 float out;
86
87 out = getlfoshape(xl);
88 if((lfotype == 0) || (lfotype == 1))
89 out *= (ampl1 + xl * (ampl2 - ampl1));
90 xl += incx;
91 if(xl > 1.0f) {
92 xl -= 1.0f;
93 ampl1 = ampl2;
94 ampl2 = (1.0f - lfornd) + lfornd * RND;
95 }
96 *outl = (out + 1.0f) * 0.5f;
97
98 out = getlfoshape(xr);
99 if((lfotype == 0) || (lfotype == 1))
100 out *= (ampr1 + xr * (ampr2 - ampr1));
101 xr += incx;
102 if(xr > 1.0f) {
103 xr -= 1.0f;
104 ampr1 = ampr2;
105 ampr2 = (1.0f - lfornd) + lfornd * RND;
106 }
107 *outr = (out + 1.0f) * 0.5f;
108}
