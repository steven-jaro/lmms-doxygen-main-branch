57 static void do_complete(void* owner, operation* base,
58 const asio::error_code& result_ec,
59 std::size_t bytes_transferred)
60 {
61 asio::error_code ec(result_ec);
62
63 // Take ownership of the operation object.
64 win_iocp_socket_recv_op* o(static_cast<win_iocp_socket_recv_op*>(base));
65 ptr p = { asio::detail::addressof(o->handler_), o, o };
66 handler_work<Handler, IoExecutor> w(o->handler_, o->io_executor_);
67
68 ASIO_HANDLER_COMPLETION((*o));
69
70#if defined(ASIO_ENABLE_BUFFER_DEBUGGING)
71 // Check whether buffers are still valid.
72 if (owner)
73 {
74 buffer_sequence_adapter<asio::mutable_buffer,
75 MutableBufferSequence>::validate(o->buffers_);
76 }
77#endif // defined(ASIO_ENABLE_BUFFER_DEBUGGING)
78
79 socket_ops::complete_iocp_recv(o->state_, o->cancel_token_,
80 buffer_sequence_adapter<asio::mutable_buffer,
81 MutableBufferSequence>::all_empty(o->buffers_),
82 ec, bytes_transferred);
83
84 // Make a copy of the handler so that the memory can be deallocated before
85 // the upcall is made. Even if we're not about to make an upcall, a
86 // sub-object of the handler may be the true owner of the memory associated
87 // with the handler. Consequently, a local copy of the handler is required
88 // to ensure that any owning sub-object remains valid until after we have
89 // deallocated the memory here.
90 detail::binder2<Handler, asio::error_code, std::size_t>
91 handler(o->handler_, ec, bytes_transferred);
92 p.h = asio::detail::addressof(handler.handler_);
93 p.reset();
94
95 // Make the upcall if required.
96 if (owner)
97 {
98 fenced_block b(fenced_block::half);
99 ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
100 w.complete(handler, handler.handler_);
101 ASIO_HANDLER_INVOCATION_END;
102 }
103 }
104
105private:
106 socket_ops::state_type state_;
107 socket_ops::weak_cancel_token_type cancel_token_;
108 MutableBufferSequence buffers_;
109 Handler handler_;
110 IoExecutor io_executor_;
111};
