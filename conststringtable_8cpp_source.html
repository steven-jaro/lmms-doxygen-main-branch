88char16* generateUTF16 (const char8* str)
89{
90 int32 len = (int32)strlen (str);
91 char16* result = new char16[len + 1];
92 for (int32 i = 0; i < len; i++)
93 {
94#if BYTEORDER == kBigEndian
95 char8* pChr = (char8*)&result[i];
96 pChr[0] = 0;
97 pChr[1] = str[i];
98#else
99 result[i] = str[i];
100#endif
101 }
102 result[len] = 0;
103 return result;
104}
