26template <typename Executor, typename T>
27class awaitable_handler_base
28 : public awaitable_thread<Executor>
29{
30public:
31 typedef void result_type;
32 typedef awaitable<T, Executor> awaitable_type;
33
34 // Construct from the entry point of a new thread of execution.
35 awaitable_handler_base(awaitable<void, Executor> a, const Executor& ex)
36 : awaitable_thread<Executor>(std::move(a), ex)
37 {
38 }
39
40 // Transfer ownership from another awaitable_thread.
41 explicit awaitable_handler_base(awaitable_thread<Executor>* h)
42 : awaitable_thread<Executor>(std::move(*h))
43 {
44 }
45
46protected:
47 awaitable_frame<T, Executor>* frame() noexcept
48 {
49 return static_cast<awaitable_frame<T, Executor>*>(this->top_of_stack_);
50 }
51};
