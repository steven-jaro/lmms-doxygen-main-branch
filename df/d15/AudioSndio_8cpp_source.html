44AudioSndio::AudioSndio(bool & _success_ful, AudioEngine * _audioEngine) :
45 AudioDevice(std::clamp<ch_cnt_t>(
46 ConfigManager::inst()->value("audiosndio", "channels").toInt(),
47 DEFAULT_CHANNELS,
48 DEFAULT_CHANNELS), _audioEngine),
49 m_convertEndian ( false )
50{
51 _success_ful = false;
52
53 QString dev = ConfigManager::inst()->value( "audiosndio", "device" );
54
55 if (dev == "")
56 {
57 m_hdl = sio_open( nullptr, SIO_PLAY, 0 );
58 }
59 else
60 {
61 m_hdl = sio_open( dev.toLatin1().constData(), SIO_PLAY, 0 );
62 }
63
64 if( m_hdl == nullptr )
65 {
66 printf( "sndio: failed opening audio-device\n" );
67 return;
68 }
69
70 sio_initpar(&m_par);
71
72 m_par.pchan = channels();
73 m_par.bits = 16;
74 m_par.le = SIO_LE_NATIVE;
75 m_par.rate = sampleRate();
76 m_par.round = audioEngine()->framesPerPeriod();
77 m_par.appbufsz = m_par.round * 2;
78
79 if ( (isLittleEndian() && (m_par.le == 0)) ||
80 (!isLittleEndian() && (m_par.le == 1))) {
81 m_convertEndian = true;
82 }
83
84 struct sio_par reqpar = m_par;
85
86 if (!sio_setpar(m_hdl, &m_par))
87 {
88 printf( "sndio: sio_setpar failed\n" );
89 return;
90 }
91 if (!sio_getpar(m_hdl, &m_par))
92 {
93 printf( "sndio: sio_getpar failed\n" );
94 return;
95 }
96
97 if (reqpar.pchan != m_par.pchan ||
98 reqpar.bits != m_par.bits ||
99 reqpar.le != m_par.le ||
100 (::abs(static_cast<int>(reqpar.rate) - static_cast<int>(m_par.rate)) * 100)/reqpar.rate > 2)
101 {
102 printf( "sndio: returned params not as requested\n" );
103 return;
104 }
105
106 if (!sio_start(m_hdl))
107 {
108 printf( "sndio: sio_start failed\n" );
109 return;
110 }
111
112 _success_ful = true;
113}

141void AudioSndio::run()
142{
143 SampleFrame* temp = new SampleFrame[audioEngine()->framesPerPeriod()];
144 int_sample_t * outbuf = new int_sample_t[audioEngine()->framesPerPeriod() * channels()];
145
146 while( true )
147 {
148 const fpp_t frames = getNextBuffer( temp );
149 if( !frames )
150 {
151 break;
152 }
153
154 uint bytes = convertToS16(temp, frames, outbuf, m_convertEndian);
155 if( sio_write( m_hdl, outbuf, bytes ) != bytes )
156 {
157 break;
158 }
159 }
160
161 delete[] temp;
162 delete[] outbuf;
163}
