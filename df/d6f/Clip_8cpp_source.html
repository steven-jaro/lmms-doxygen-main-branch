108/*! \brief Move this Clip's position in time
109 *
110 * If the clip has moved, update its position. We
111 * also add a journal entry for undo and update the display.
112 *
113 * \param _pos The new position of the clip.
114 */
115void Clip::movePosition( const TimePos & pos )
116{
117 TimePos newPos = std::max(0, pos.getTicks());
118 if (m_startPosition != newPos)
119 {
120 Engine::audioEngine()->requestChangeInModel();
121 m_startPosition = newPos;
122 Engine::audioEngine()->doneChangeInModel();
123 Engine::getSong()->updateLength();
124 emit positionChanged();
125 }
126}

156/*! \brief Copies the state of a Clip to another Clip
157 *
158 * This method copies the state of a Clip to another Clip
159 */
160void Clip::copyStateTo( Clip *src, Clip *dst )
161{
162 // If the node names match we copy the state
163 if( src->nodeName() == dst->nodeName() ){
164 QDomDocument doc;
165 QDomElement parent = doc.createElement( "StateCopy" );
166 src->saveState( doc, parent );
167
168 const TimePos pos = dst->startPosition();
169 dst->restoreState( parent.firstChild().toElement() );
170 dst->movePosition( pos );
171
172 AutomationClip::resolveAllIDs();
173 gui::getGUI()->automationEditor()->m_editor->updateAfterClipChange();
174 }
175}
