55void AudioDevice::processNextBuffer()
56{
57 const fpp_t frames = getNextBuffer( m_buffer );
58 if (frames) { writeBuffer(m_buffer, frames); }
59 else
60 {
61 m_inProcess = false;
62 }

65fpp_t AudioDevice::getNextBuffer(SampleFrame* _ab)
66{
67 fpp_t frames = audioEngine()->framesPerPeriod();
68 const SampleFrame* b = audioEngine()->nextBuffer();
69
70 if (!b) { return 0; }
71
72 memcpy(_ab, b, frames * sizeof(SampleFrame));
73
74 if (audioEngine()->hasFifoWriter()) { delete[] b; }
75 return frames;
76}

128int AudioDevice::convertToS16(const SampleFrame* _ab,
129 const fpp_t _frames,
130 int_sample_t * _output_buffer,
131 const bool _convert_endian )
132{
133 if( _convert_endian )
134 {
135 for( fpp_t frame = 0; frame < _frames; ++frame )
136 {
137 for( ch_cnt_t chnl = 0; chnl < channels(); ++chnl )
138 {
139 auto temp = static_cast<int_sample_t>(AudioEngine::clip(_ab[frame][chnl]) * OUTPUT_SAMPLE_MULTIPLIER);
140
141 ( _output_buffer + frame * channels() )[chnl] =
142 ( temp & 0x00ff ) << 8 |
143 ( temp & 0xff00 ) >> 8;
144 }
145 }
146 }
147 else
148 {
149 for( fpp_t frame = 0; frame < _frames; ++frame )
150 {
151 for( ch_cnt_t chnl = 0; chnl < channels(); ++chnl )
152 {
153 (_output_buffer + frame * channels())[chnl]
154 = static_cast<int_sample_t>(AudioEngine::clip(_ab[frame][chnl]) * OUTPUT_SAMPLE_MULTIPLIER);
155 }
156 }
157 }
158
159 return _frames * channels() * BYTES_PER_INT_SAMPLE;
160}
