50EffectSelectDialog::EffectSelectDialog(QWidget* parent) :
51 QDialog(parent),
52 m_effectKeys(),
53 m_currentSelection(),
54 m_sourceModel(),
55 m_model(),
56 m_descriptionWidget(nullptr),
57 m_pluginList(new QTableView(this)),
58 m_scrollArea(new QScrollArea(this))
59{
60 setWindowTitle(tr("Add effect"));
61 resize(640, 480);
62
63 setWindowIcon(embed::getIconPixmap("setup_audio"));
64
65 // Query effects
66 EffectKeyList subPluginEffectKeys;
67 for (const auto desc : getPluginFactory()->descriptors(Plugin::Type::Effect))
68 {
69 if (desc->subPluginFeatures)
70 {
71 desc->subPluginFeatures->listSubPluginKeys(desc, subPluginEffectKeys);
72 }
73 else
74 {
75 m_effectKeys << EffectKey(desc, desc->name);
76 }
77 }
78 m_effectKeys += subPluginEffectKeys;
79
80 // Fill the source model
81 m_sourceModel.setHorizontalHeaderItem(0, new QStandardItem(tr("Name")));
82 m_sourceModel.setHorizontalHeaderItem(1, new QStandardItem(tr("Type")));
83 int row = 0;
84 for (EffectKeyList::ConstIterator it = m_effectKeys.begin(); it != m_effectKeys.end(); ++it)
85 {
86 QString name;
87 QString type;
88 if (it->desc->subPluginFeatures)
89 {
90 name = it->displayName();
91 type = it->desc->displayName;
92 }
93 else
94 {
95 name = it->desc->displayName;
96 type = "LMMS";
97 }
98 m_sourceModel.setItem(row, 0, new QStandardItem(name));
99 m_sourceModel.setItem(row, 1, new QStandardItem(type));
100 ++row;
101 }
102
103 // Setup filtering
104 m_model.setSourceModel(&m_sourceModel);
105 m_model.setFilterCaseSensitivity(Qt::CaseInsensitive);
106
107 QHBoxLayout* mainLayout = new QHBoxLayout(this);
108
109 QVBoxLayout* leftSectionLayout = new QVBoxLayout();
110
111 QStringList buttonLabels = { tr("All"), "LMMS", "LADSPA", "LV2", "VST" };
112 QStringList buttonSearchString = { "", "LMMS", "LADSPA", "LV2", "VST" };
113
114 for (int i = 0; i < buttonLabels.size(); ++i)
115 {
116 const QString& label = buttonLabels[i];
117 const QString& searchString = buttonSearchString[i];
118
119 QPushButton* button = new QPushButton(label, this);
120 button->setFixedSize(50, 50);
121 button->setFocusPolicy(Qt::NoFocus);
122 leftSectionLayout->addWidget(button);
123
124 connect(button, &QPushButton::clicked, this, [this, searchString] {
125 m_model.setEffectTypeFilter(searchString);
126 updateSelection();
127 });
128 }
129
130 leftSectionLayout->addStretch();// Add stretch to the button layout to push buttons to the top
131 mainLayout->addLayout(leftSectionLayout);
132
133 m_filterEdit = new QLineEdit(this);
134 connect(m_filterEdit, &QLineEdit::textChanged, this, [this](const QString &text) {
135// TODO: Cleanup when we don't support Qt5 anymore
136#if (QT_VERSION >= QT_VERSION_CHECK(5,12,0))
137 m_model.setFilterRegularExpression(QRegularExpression(text, QRegularExpression::CaseInsensitiveOption));
138#else
139 m_model.setFilterRegExp(QRegExp(text, Qt::CaseInsensitive));
140#endif
141 });
142 connect(m_filterEdit, &QLineEdit::textChanged, this, &EffectSelectDialog::updateSelection);
143 m_filterEdit->setFocus();
144 m_filterEdit->setFocusPolicy(Qt::StrongFocus);
145 m_filterEdit->setPlaceholderText(tr("Search"));
146 m_filterEdit->setClearButtonEnabled(true);
147 m_filterEdit->addAction(embed::getIconPixmap("zoom"), QLineEdit::LeadingPosition);
148
149 m_pluginList->setModel(&m_model);
150 m_pluginList->setEditTriggers(QAbstractItemView::NoEditTriggers);
151 m_pluginList->setSelectionBehavior(QAbstractItemView::SelectRows);
152 m_pluginList->setSelectionMode(QAbstractItemView::SingleSelection);
153 m_pluginList->setSortingEnabled(true);
154 m_pluginList->sortByColumn(0, Qt::AscendingOrder); // Initial sort by column 0 (Name)
155 m_pluginList->verticalHeader()->setSectionResizeMode(QHeaderView::ResizeToContents);
156 m_pluginList->verticalHeader()->hide();
157 m_pluginList->horizontalHeader()->setSectionResizeMode(0, QHeaderView::Stretch);
158 m_pluginList->horizontalHeader()->setSectionResizeMode(1, QHeaderView::ResizeToContents);
159 m_pluginList->setFocusPolicy(Qt::NoFocus);
160
161 // Scroll Area
162 m_scrollArea->setWidgetResizable(true);
163 QWidget* scrollAreaWidgetContents = new QWidget(m_scrollArea);
164 scrollAreaWidgetContents->setObjectName("scrollAreaWidgetContents");
165 m_scrollArea->setWidget(scrollAreaWidgetContents);
166 m_scrollArea->setMaximumHeight(180);
167 m_scrollArea->setFocusPolicy(Qt::NoFocus);
168
169 // Button Box
170 QDialogButtonBox* buttonBox = new QDialogButtonBox(Qt::Horizontal, this);
171 buttonBox->setStandardButtons(QDialogButtonBox::Cancel | QDialogButtonBox::Ok);
172 buttonBox->setFocusPolicy(Qt::NoFocus);
173 connect(buttonBox, &QDialogButtonBox::accepted, this, &EffectSelectDialog::acceptSelection);
174 connect(buttonBox, &QDialogButtonBox::rejected, this, &EffectSelectDialog::reject);
175
176 QVBoxLayout* rightSectionLayout = new QVBoxLayout();
177 rightSectionLayout->addWidget(m_filterEdit);
178 rightSectionLayout->addWidget(m_pluginList);
179 rightSectionLayout->addWidget(m_scrollArea);
180 rightSectionLayout->addWidget(buttonBox);
181 mainLayout->addLayout(rightSectionLayout);
182
183 setLayout(mainLayout);
184
185 auto selectionModel = new QItemSelectionModel(&m_model);
186 m_pluginList->setSelectionModel(selectionModel);
187 connect(selectionModel, &QItemSelectionModel::currentRowChanged,
188 this, &EffectSelectDialog::rowChanged);
189
190 connect(m_pluginList, &QTableView::doubleClicked,
191 this, &EffectSelectDialog::acceptSelection);
192
193 setModal(true);
194 installEventFilter(this);
195
196 updateSelection();
197 show();
198}

223void EffectSelectDialog::rowChanged(const QModelIndex& idx, const QModelIndex&)
224{
225 delete m_descriptionWidget;
226 m_descriptionWidget = nullptr;
227
228 if (m_model.mapToSource(idx).row() < 0)
229 {
230 // Invalidate current selection
231 m_currentSelection = Plugin::Descriptor::SubPluginFeatures::Key();
232 }
233 else
234 {
235 m_currentSelection = m_effectKeys[m_model.mapToSource(idx).row()];
236 }
237 if (m_currentSelection.desc)
238 {
239 m_descriptionWidget = new QWidget;
240
241 auto hbox = new QHBoxLayout(m_descriptionWidget);
242 hbox->setAlignment(Qt::AlignTop);
243
244 Plugin::Descriptor const& descriptor = *(m_currentSelection.desc);
245
246 const PixmapLoader* pixLoa = m_currentSelection.logo();
247 if (pixLoa)
248 {
249 auto logoLabel = new QLabel(m_descriptionWidget);
250 logoLabel->setPixmap(pixLoa->pixmap());
251 logoLabel->setSizePolicy(QSizePolicy::Minimum, QSizePolicy::Minimum);
252 logoLabel->setMaximumSize(64, 64);
253
254 hbox->addWidget(logoLabel);
255 }
256
257 auto textualInfoWidget = new QWidget(m_descriptionWidget);
258 hbox->addWidget(textualInfoWidget);
259
260 auto textWidgetLayout = new QVBoxLayout(textualInfoWidget);
261 textWidgetLayout->setContentsMargins(4, 4, 4, 4);
262 textWidgetLayout->setSpacing(8);
263
264 if (m_currentSelection.desc->subPluginFeatures)
265 {
266 auto subWidget = new QWidget(textualInfoWidget);
267 auto subLayout = new QVBoxLayout(subWidget);
268 subLayout->setContentsMargins(0, 0, 0, 0);
269 subLayout->setSpacing(8);
270
271 m_currentSelection.desc->subPluginFeatures->
272 fillDescriptionWidget(subWidget, &m_currentSelection);
273 for (QWidget* w : subWidget->findChildren<QWidget*>())
274 {
275 if (w->parent() == subWidget)
276 {
277 subLayout->addWidget(w);
278 subLayout->setAlignment(w, QFlags<Qt::AlignmentFlag>(Qt::AlignTop | Qt::AlignLeft));
279 }
280 }
281 textWidgetLayout->addWidget(subWidget);
282 }
283 else
284 {
285 auto label = new QLabel(m_descriptionWidget);
286 QString labelText = "<p><b>" + tr("Name") + ":</b> " + QString::fromUtf8(descriptor.displayName) + "</p>";
287 labelText += "<p><b>" + tr("Description") + ":</b> " + qApp->translate("PluginBrowser", descriptor.description) + "</p>";
288 labelText += "<p><b>" + tr("Author") + ":</b> " + QString::fromUtf8(descriptor.author) + "</p>";
289
290 label->setText(labelText);
291 label->setWordWrap(true);
292
293 textWidgetLayout->addWidget(label);
294 }
295
296 m_scrollArea->setWidget(m_descriptionWidget);
297 m_descriptionWidget->show();
298 }

313bool EffectSelectDialog::eventFilter(QObject *obj, QEvent *event)
314{
315 if (obj == this && event->type() == QEvent::KeyPress)
316 {
317 QKeyEvent *keyEvent = static_cast<QKeyEvent*>(event);
318 if (keyEvent->key() == Qt::Key_Up || keyEvent->key() == Qt::Key_Down)
319 {
320 QItemSelectionModel* selectionModel = m_pluginList->selectionModel();
321 int currentRow = selectionModel->currentIndex().row();
322 int newRow = (keyEvent->key() == Qt::Key_Up) ? currentRow - 1 : currentRow + 1;
323 int rowCount = m_pluginList->model()->rowCount();
324 newRow = qBound(0, newRow, rowCount - 1);
325
326 selectionModel->setCurrentIndex(m_pluginList->model()->index(newRow, 0),
327 QItemSelectionModel::ClearAndSelect | QItemSelectionModel::Rows);
328 m_pluginList->scrollTo(m_pluginList->model()->index(newRow, 0));
329 return true;
330 }
331 }
332
333 return QDialog::eventFilter(obj, event);
334}
