210void FileBrowser::onSearch(const QString& filter)
211{
212 if (filter.isEmpty())
213 {
214 m_searchTreeWidget->hide();
215 m_fileBrowserTreeWidget->show();
216 return;
217 }
218
219 auto directories = m_directories.split('*');
220 if (m_showUserContent && !m_showUserContent->isChecked()) { directories.removeAll(m_userDir); }
221 if (m_showFactoryContent && !m_showFactoryContent->isChecked()) { directories.removeAll(m_factoryDir); }
222 if (directories.isEmpty()) { return; }
223
224 auto directoryFilters = QDir::AllEntries | QDir::NoDotAndDotDot;
225 if (m_showHiddenContent) { directoryFilters |= QDir::Hidden; }
226
227 const auto searchTask = FileSearchJob::Task{.filter = filter,
228 .paths = directories,
229 .extensions = FileItem::defaultFilters().split(" "),
230 .dirFilters = directoryFilters};
231
232 m_searchTreeWidget->clear();
233 m_searchTreeWidget->show();
234 m_fileBrowserTreeWidget->hide();
235 m_searchJob.search(searchTask);
236}

258void FileBrowser::reloadTree()
259{
260 if (m_filterEdit->text().isEmpty())
261 {
262 saveDirectoriesStates();
263 }
264
265 m_fileBrowserTreeWidget->clear();
266
267 auto paths = m_directories.isEmpty() ? QStringList{} : m_directories.split('*');
268
269 if (m_showUserContent && !m_showUserContent->isChecked())
270 {
271 paths.removeAll(m_userDir);
272 }
273
274 if (m_showFactoryContent && !m_showFactoryContent->isChecked())
275 {
276 paths.removeAll(m_factoryDir);
277 }
278
279 switch (m_type)
280 {
281 case Type::Favorites:
282 for (auto& path : paths)
283 {
284 while (path.endsWith('/') || path.endsWith('\\') || path.endsWith("."))
285 {
286 path.chop(1);
287 }
288
289 auto info = QFileInfo{PathUtil::toAbsolute(path)};
290
291 if (info.isDir())
292 {
293 auto dir = new Directory(info.fileName(), info.absolutePath(), m_filter);
294 dir->update();
295 m_fileBrowserTreeWidget->addTopLevelItem(dir);
296 }
297 else if (info.isFile())
298 {
299 auto file = new FileItem(info.fileName(), info.path());
300 m_fileBrowserTreeWidget->addTopLevelItem(file);
301 }
302 }
303 break;
304 case Type::Normal:
305 for (const auto& path : paths)
306 {
307 addItems(path);
308 }
309 break;
310 }
311
312 if (m_filterEdit->text().isEmpty())
313 {
314 restoreDirectoriesStates();
315 }
316 else
317 {
318 onSearch(m_filterEdit->text());
319 }
320}

324void FileBrowser::expandItems(const QList<QString>& expandedDirs, QTreeWidgetItem* item)
325{
326 if (expandedDirs.isEmpty()) { return; }
327
328 int numChildren = item ? item->childCount() : m_fileBrowserTreeWidget->topLevelItemCount();
329 for (int i = 0; i < numChildren; ++i)
330 {
331 auto it = item ? item->child(i) : m_fileBrowserTreeWidget->topLevelItem(i);
332 auto d = dynamic_cast<Directory*>(it);
333 if (d)
334 {
335 d->setExpanded(expandedDirs.contains(d->fullName()));
336 if (it->childCount() > 0)
337 {
338 expandItems(expandedDirs, it);
339 }
340 }
341
342 it->setHidden(false);
343 }
344}

360void FileBrowser::addItems(const QString & path )
361{
362 if( m_dirsAsItems )
363 {
364 m_fileBrowserTreeWidget->addTopLevelItem( new Directory( path, QString(), m_filter ) );
365 return;
366 }
367
368 // try to add all directories from file system alphabetically into the tree
369 QDir cdir(path);
370 if (!cdir.isReadable()) { return; }
371 QFileInfoList entries = cdir.entryInfoList(
372 m_filter.split(' '),
373 dirFilters(),
374 QDir::LocaleAware | QDir::DirsFirst | QDir::Name | QDir::IgnoreCase);
375 for (const auto& entry : entries)
376 {
377 QString fileName = entry.fileName();
378 if (entry.isHidden() && m_showHiddenContent && !m_showHiddenContent->isChecked()) continue;
379 if (entry.isDir())
380 {
381 // Merge dir's together
382 bool orphan = true;
383 for (int i = 0; i < m_fileBrowserTreeWidget->topLevelItemCount(); ++i)
384 {
385 auto d = dynamic_cast<Directory*>(m_fileBrowserTreeWidget->topLevelItem(i));
386 if (d == nullptr || fileName < d->text(0))
387 {
388 // insert before item, we're done
389 auto dd = new Directory(fileName, path, m_filter);
390 m_fileBrowserTreeWidget->insertTopLevelItem(i,dd);
391 dd->update(); // add files to the directory
392 orphan = false;
393 break;
394 }
395 else if (fileName == d->text(0))
396 {
397 // imagine we have subdirs named "TripleOscillator/xyz" in
398 // two directories from m_directories
399 // then only add one tree widget for both
400 // so we don't add a new Directory - we just
401 // add the path to the current directory
402 d->addDirectory(path);
403 d->update();
404 orphan = false;
405 break;
406 }
407 }
408 if (orphan)
409 {
410 // it has not yet been added yet, so it's (lexically)
411 // larger than all other dirs => append it at the bottom
412 auto d = new Directory(fileName, path, m_filter);
413 d->update();
414 m_fileBrowserTreeWidget->addTopLevelItem(d);
415 }
416 }
417 else if (entry.isFile())
418 {
419 // TODO: don't insert instead of removing, order changed
420 // remove existing file-items
421 QList<QTreeWidgetItem *> existing = m_fileBrowserTreeWidget->findItems(fileName, Qt::MatchFixedString);
422 if (!existing.empty())
423 {
424 delete existing.front();
425 }
426 (void) new FileItem(m_fileBrowserTreeWidget, fileName, path);
427 }
428 }
429}

484QList<QString> FileBrowserTreeWidget::expandedDirs( QTreeWidgetItem * item ) const
485{
486 int numChildren = item ? item->childCount() : topLevelItemCount();
487 QList<QString> dirs;
488 for (int i = 0; i < numChildren; ++i)
489 {
490 QTreeWidgetItem * it = item ? item->child(i) : topLevelItem(i);
491
492 // Add expanded top level directories.
493 if (it->isExpanded() && (it->type() == TypeDirectoryItem))
494 {
495 auto d = static_cast<Directory*>(it);
496 dirs.append( d->fullName() );
497 }
498
499 // Add expanded child directories (recurse).
500 if (it->childCount())
501 {
502 dirs.append( expandedDirs( it ) );
503 }
504 }
505 return dirs;
506}

509void FileBrowserTreeWidget::keyPressEvent(QKeyEvent * ke )
510{
511 // Shorter names for some commonly used properties of the event
512 const auto key = ke->key();
513 const bool vertical = (key == Qt::Key_Up || key == Qt::Key_Down);
514 const bool horizontal = (key == Qt::Key_Left || key == Qt::Key_Right);
515 const bool insert = (key == Qt::Key_Enter || key == Qt::Key_Return);
516 const bool preview = (key == Qt::Key_Space);
517
518 // First of all, forward all keypresses
519 QTreeWidget::keyPressEvent(ke);
520 // Then, ignore all autorepeats (they would spam new tracks or previews)
521 if (ke->isAutoRepeat()) { return; }
522 // We should stop any running previews before we do anything new
523 else if (vertical || horizontal || preview || insert) { stopPreview(); }
524
525 // Try to get the currently selected item as a FileItem
526 auto file = dynamic_cast<FileItem*>(currentItem());
527 // If it's null (folder, separator, etc.), there's nothing left for us to do
528 if (file == nullptr) { return; }
529
530 // When moving to a new sound, preview it. Skip presets, they can play forever
531 if (vertical && file->type() == FileItem::FileType::Sample)
532 {
533 previewFileItem(file);
534 }
535
536 // When enter is pressed, add the selected item...
537 if (insert)
538 {
539 // ...to the song editor by default, or to the pattern editor if ctrl is held
540 bool songEditor = !(ke->modifiers() & Qt::ControlModifier);
541 // If shift is held, we send the item to a new sample track...
542 bool sampleTrack = ke->modifiers() & Qt::ShiftModifier;
543 // ...but only in the song editor. So, ctrl+shift enter does nothing
544 if (sampleTrack && songEditor){ openInNewSampleTrack(file); }
545 // Otherwise we send the item as a new instrument track
546 else if (!sampleTrack){ openInNewInstrumentTrack(file, songEditor); }
547 }
548
549 // When space is pressed, start a preview of the selected item
550 if (preview) { previewFileItem(file); }
551}

582void FileBrowserTreeWidget::contextMenuEvent(QContextMenuEvent* e)
583{
584#ifdef LMMS_BUILD_APPLE
585 QString fileManager = tr("Finder");
586#elif defined(LMMS_BUILD_WIN32)
587 QString fileManager = tr("Explorer");
588#else
589 QString fileManager = tr("file manager");
590#endif
591
592 auto item = dynamic_cast<FileBrowserWidgetItem*>(itemAt(e->pos()));
593 if (item == nullptr) { return; }
594
595 auto contextMenu = QMenu{this};
596
597 const auto fontMetrics = QFontMetrics{qApp->font()};
598 const auto maxHeaderWidth = 50 * fontMetrics.averageCharWidth();
599 const auto elidedPath = fontMetrics.elidedText(PathUtil::toShortestRelative(item->fullName()), Qt::TextElideMode::ElideMiddle, maxHeaderWidth);
600
601 auto header = new QAction{elidedPath};
602 header->setDisabled(true);
603
604 contextMenu.addAction(header);
605 contextMenu.addSeparator();
606
607 switch (item->type())
608 {
609 case TypeFileItem: {
610 auto file = dynamic_cast<FileItem*>(item);
611
612 contextMenu.addAction(QIcon(embed::getIconPixmap("folder")), tr("Show in %1").arg(fileManager),
613 [file] { FileRevealer::reveal(file->fullName()); });
614
615 if (file->isTrack())
616 {
617 contextMenu.addAction(
618 tr("Send to active instrument-track"), [file, this] { sendToActiveInstrumentTrack(file); });
619 }
620
621 const auto path = QFileInfo{file->fullName()}.absoluteFilePath();
622
623 if (ConfigManager::inst()->isFavoriteItem(file->fullName()))
624 {
625 contextMenu.addAction(
626 QIcon(embed::getIconPixmap("star")), tr("Remove favorite file"), [path] { ConfigManager::inst()->removeFavoriteItem(path); });
627 }
628 else
629 {
630 contextMenu.addAction(
631 QIcon(embed::getIconPixmap("star")), tr("Add favorite file"), [path] { ConfigManager::inst()->addFavoriteItem(path); });
632 }
633
634 if (file->isTrack())
635 {
636 contextMenu.addSeparator();
637 contextMenu.addAction(tr("Send to active instrument-track"), [&] { sendToActiveInstrumentTrack(file); });
638 }
639
640 auto songEditorHeader = new QAction(tr("Song Editor"), nullptr);
641 songEditorHeader->setDisabled(true);
642 contextMenu.addAction(songEditorHeader);
643 contextMenu.addActions(getContextActions(file, true));
644
645 auto patternEditorHeader = new QAction(tr("Pattern Editor"), nullptr);
646 patternEditorHeader->setDisabled(true);
647 contextMenu.addAction(patternEditorHeader);
648 contextMenu.addActions(getContextActions(file, false));
649 break;
650 }
651 case TypeDirectoryItem: {
652 auto dir = dynamic_cast<Directory*>(item);
653 const auto path = QFileInfo{dir->fullName()}.absoluteFilePath();
654
655 contextMenu.addAction(QIcon(embed::getIconPixmap("folder")), tr("Open in %1").arg(fileManager), [=] {
656 FileRevealer::openDir(dir->fullName());
657 });
658
659 if (ConfigManager::inst()->isFavoriteItem(dir->fullName()))
660 {
661 contextMenu.addAction(QIcon(embed::getIconPixmap("star")), tr("Remove favorite folder"), [path] { ConfigManager::inst()->removeFavoriteItem(path); });
662 }
663 else
664 {
665 contextMenu.addAction(QIcon(embed::getIconPixmap("star")), tr("Add favorite folder"), [path] { ConfigManager::inst()->addFavoriteItem(path); });
666 }
667 break;
668 }
669 }
670
671 // Only show the menu if it contains items
672 if (!contextMenu.isEmpty()) { contextMenu.exec(e->globalPos()); }
673}

685QList<QAction*> FileBrowserTreeWidget::getContextActions(FileItem* file, bool songEditor)
686{
687 QList<QAction*> result = QList<QAction*>();
688 const bool fileIsSample = file->type() == FileItem::FileType::Sample;
689
690 QString instrumentAction = fileIsSample ?
691 tr("Send to new AudioFileProcessor instance") :
692 tr("Send to new instrument track");
693 QString shortcutMod = songEditor ? "" : UI_CTRL_KEY + QString(" + ");
694
695 auto toInstrument = new QAction(instrumentAction + tr(" (%2Enter)").arg(shortcutMod));
696 connect(toInstrument, &QAction::triggered,
697 [=, this]{ openInNewInstrumentTrack(file, songEditor); });
698 result.append(toInstrument);
699
700 if (songEditor && fileIsSample)
701 {
702 auto toSampleTrack = new QAction(tr("Send to new sample track (Shift + Enter)"));
703 connect(toSampleTrack, &QAction::triggered,
704 [=, this]{ openInNewSampleTrack(file); });
705 result.append(toSampleTrack);
706 }
707
708 if (fileIsSample && !PluginFactory::instance()->pluginInfo("slicert").isNull())
709 {
710 auto openInSlicer = new QAction(tr("Send to new SlicerT instance"));
711 connect(openInSlicer, &QAction::triggered,
712 [=, this]{ openInSlicerT(file); });
713 result.append(openInSlicer);
714 }
715
716 return result;
717}

733// if ( _me->x() > header()->cellPos( header()->mapToActual( 0 ) )
734// + treeStepSize() * ( i->depth() + ( rootIsDecorated() ?
735// 1 : 0 ) ) + itemMargin() ||
736// _me->x() < header()->cellPos(
737// header()->mapToActual( 0 ) ) )
738// {
739 m_pressPos = me->pos();
740 m_mousePressed = true;
741// }
742 }

751void FileBrowserTreeWidget::previewFileItem(FileItem* file)
752{ // TODO: We should do this work outside the event thread
753 // Lock the preview mutex
754 QMutexLocker previewLocker(&m_pphMutex);
755 // If something is already playing, stop it before we continue
756 stopPreview();
757
758 PlayHandle* newPPH = nullptr;
759 const QString fileName = file->fullName();
760 const QString ext = file->extension();
761
762 // In special case of sample-files we do not care about
763 // handling() rather than directly creating a SamplePlayHandle
764 if (file->type() == FileItem::FileType::Sample)
765 {
766 TextFloat * tf = TextFloat::displayMessage(
767 tr("Loading sample"),
768 tr("Please wait, loading sample for preview..."),
769 embed::getIconPixmap("sample_file", 24, 24), 0);
770 // TODO: this can be removed once we do this outside the event thread
771 qApp->processEvents(QEventLoop::ExcludeUserInputEvents);
772 if (auto buffer = SampleLoader::createBufferFromFile(fileName))
773 {
774 auto s = new SamplePlayHandle(new lmms::Sample{std::move(buffer)});
775 s->setDoneMayReturnTrue(false);
776 newPPH = s;
777 }
778 delete tf;
779 }
780 else if (
781 (ext == "xiz" || ext == "sf2" || ext == "sf3" ||
782 ext == "gig" || ext == "pat")
783 && !getPluginFactory()->pluginSupportingExtension(ext).isNull())
784 {
785 const bool isPlugin = file->handling() == FileItem::FileHandling::LoadByPlugin;
786 newPPH = new PresetPreviewPlayHandle(fileName, isPlugin);
787 }
788 else if (file->type() != FileItem::FileType::VstPlugin && file->isTrack())
789 {
790 DataFile dataFile(fileName);
791 if (dataFile.validate(ext))
792 {
793 const bool isPlugin = file->handling() == FileItem::FileHandling::LoadByPlugin;
794 newPPH = new PresetPreviewPlayHandle(fileName, isPlugin, &dataFile);
795 }
796 else
797 {
798 QMessageBox::warning(0, tr ("Error"),
799 tr("%1 does not appear to be a valid %2 file")
800 .arg(fileName, ext),
801 QMessageBox::Ok, QMessageBox::NoButton);
802 }
803 }
804
805 if (newPPH != nullptr)
806 {
807 if (Engine::audioEngine()->addPlayHandle(newPPH))
808 {
809 m_previewPlayHandle = newPPH;
810 }
811 else { m_previewPlayHandle = nullptr; }
812 }
813}

906void FileBrowserTreeWidget::handleFile(FileItem * f, InstrumentTrack * it)
907{
908 Engine::audioEngine()->requestChangeInModel();
909 switch( f->handling() )
910 {
911 case FileItem::FileHandling::LoadAsProject:
912 if( getGUI()->mainWindow()->mayChangeProject(true) )
913 {
914 Engine::getSong()->loadProject( f->fullName() );
915 }
916 break;
917
918 case FileItem::FileHandling::LoadByPlugin:
919 {
920 const QString e = f->extension();
921 Instrument * i = it->instrument();
922 if( i == nullptr ||
923 !i->descriptor()->supportsFileType( e ) )
924 {
925 PluginFactory::PluginInfoAndKey piakn =
926 getPluginFactory()->pluginSupportingExtension(e);
927 i = it->loadInstrument(piakn.info.name(), &piakn.key);
928 }
929 i->loadFile( f->fullName() );
930 break;
931 }
932
933 case FileItem::FileHandling::LoadAsPreset: {
934 DataFile dataFile(f->fullName());
935 it->replaceInstrument(dataFile);
936 break;
937 }
938 case FileItem::FileHandling::ImportAsProject:
939 ImportFilter::import( f->fullName(),
940 Engine::getSong() );
941 break;
942
943 case FileItem::FileHandling::NotSupported:
944 default:
945 break;
946 }
947 Engine::audioEngine()->doneChangeInModel();
948}

1106bool Directory::addItems(const QString& path)
1107{
1108 QDir thisDir(path);
1109 if (!thisDir.isReadable()) { return false; }
1110
1111 treeWidget()->setUpdatesEnabled(false);
1112
1113 QFileInfoList entries
1114 = thisDir.entryInfoList(m_filter.split(' '), FileBrowser::dirFilters(), FileBrowser::sortFlags());
1115 for (const auto& entry : entries)
1116 {
1117 QString fileName = entry.fileName();
1118 if (entry.isDir())
1119 {
1120 auto dir = new Directory(fileName, path, m_filter);
1121 addChild(dir);
1122 m_dirCount++;
1123 }
1124 else if (entry.isFile())
1125 {
1126 auto fileItem = new FileItem(fileName, path);
1127 addChild(fileItem);
1128 }
1129 }
1130
1131 treeWidget()->setUpdatesEnabled(true);
1132
1133 // return true if we added any child items
1134 return childCount() > 0;

1196
1197void FileItem::determineFileType()
1198{
1199 m_handling = FileHandling::NotSupported;
1200
1201 const QString ext = extension();
1202 if( ext == "mmp" || ext == "mpt" || ext == "mmpz" )
1203 {
1204 m_type = FileType::Project;
1205 m_handling = FileHandling::LoadAsProject;
1206 }
1207 else if( ext == "xpf" || ext == "xml" )
1208 {
1209 m_type = FileType::Preset;
1210 m_handling = FileHandling::LoadAsPreset;
1211 }
1212 else if( ext == "xiz" && ! getPluginFactory()->pluginSupportingExtension(ext).isNull() )
1213 {
1214 m_type = FileType::Preset;
1215 m_handling = FileHandling::LoadByPlugin;
1216 }
1217 else if( ext == "sf2" || ext == "sf3" )
1218 {
1219 m_type = FileType::SoundFont;
1220 }
1221 else if( ext == "pat" )
1222 {
1223 m_type = FileType::Patch;
1224 }
1225 else if( ext == "mid" || ext == "midi" || ext == "rmi" )
1226 {
1227 m_type = FileType::Midi;
1228 m_handling = FileHandling::ImportAsProject;
1229 }
1230#ifdef LMMS_HAVE_VST
1231 else if (
1232# if defined(LMMS_BUILD_LINUX)
1233 ext == "so" ||
1234# endif
1235# if defined(LMMS_HAVE_VST_32) || defined(LMMS_HAVE_VST_64)
1236 ext == "dll" ||
1237# endif
1238 false
1239 )
1240 {
1241 m_type = FileType::VstPlugin;
1242 m_handling = FileHandling::LoadByPlugin;
1243 }
1244#endif
1245 else if ( ext == "lv2" )
1246 {
1247 m_type = FileType::Preset;
1248 m_handling = FileHandling::LoadByPlugin;
1249 }
1250 else
1251 {
1252 m_type = FileType::Unknown;
1253 }
1254
1255 if( m_handling == FileHandling::NotSupported &&
1256 !ext.isEmpty() && ! getPluginFactory()->pluginSupportingExtension(ext).isNull() )
1257 {
1258 m_handling = FileHandling::LoadByPlugin;
1259 // classify as sample if not classified by anything yet but can
1260 // be handled by a certain plugin
1261 if( m_type == FileType::Unknown )
1262 {
1263 m_type = FileType::Sample;
1264 }
1265 }
1266}
