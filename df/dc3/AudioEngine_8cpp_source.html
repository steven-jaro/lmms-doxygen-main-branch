73AudioEngine::AudioEngine( bool renderOnly ) :
74 m_renderOnly( renderOnly ),
75 m_framesPerPeriod( DEFAULT_BUFFER_SIZE ),
76 m_baseSampleRate(std::max(ConfigManager::inst()->value("audioengine", "samplerate").toInt(), SUPPORTED_SAMPLERATES.front())),
77 m_inputBufferRead( 0 ),
78 m_inputBufferWrite( 1 ),
79 m_outputBufferRead(nullptr),
80 m_outputBufferWrite(nullptr),
81 m_workers(),
82 m_numWorkers( QThread::idealThreadCount()-1 ),
83 m_newPlayHandles( PlayHandle::MaxNumber ),
84 m_masterGain( 1.0f ),
85 m_audioDev( nullptr ),
86 m_oldAudioDev( nullptr ),
87 m_audioDevStartFailed( false ),
88 m_profiler(),
89 m_clearSignal(false)
90{
91 for( int i = 0; i < 2; ++i )
92 {
93 m_inputBufferFrames[i] = 0;
94 m_inputBufferSize[i] = DEFAULT_BUFFER_SIZE * 100;
95 m_inputBuffer[i] = new SampleFrame[ DEFAULT_BUFFER_SIZE * 100 ];
96 zeroSampleFrames(m_inputBuffer[i], m_inputBufferSize[i]);
97 }
98
99 // determine FIFO size and number of frames per period
100 int fifoSize = 1;
101
102 // if not only rendering (that is, using the GUI), load the buffer
103 // size from user configuration
104 if( renderOnly == false )
105 {
106 m_framesPerPeriod =
107 ( fpp_t ) ConfigManager::inst()->value( "audioengine", "framesperaudiobuffer" ).toInt();
108
109 // if the value read from user configuration is not set or
110 // lower than the minimum allowed, use the default value and
111 // save it to the configuration
112 if( m_framesPerPeriod < MINIMUM_BUFFER_SIZE )
113 {
114 ConfigManager::inst()->setValue( "audioengine",
115 "framesperaudiobuffer",
116 QString::number( DEFAULT_BUFFER_SIZE ) );
117
118 m_framesPerPeriod = DEFAULT_BUFFER_SIZE;
119 }
120 // lmms works with chunks of size DEFAULT_BUFFER_SIZE (256) and only the final mix will use the actual
121 // buffer size. Plugins don't see a larger buffer size than 256. If m_framesPerPeriod is larger than
122 // DEFAULT_BUFFER_SIZE, it's set to DEFAULT_BUFFER_SIZE and the rest is handled by an increased fifoSize.
123 else if( m_framesPerPeriod > DEFAULT_BUFFER_SIZE )
124 {
125 fifoSize = m_framesPerPeriod / DEFAULT_BUFFER_SIZE;
126 m_framesPerPeriod = DEFAULT_BUFFER_SIZE;
127 }
128 }
129
130 // allocate the FIFO from the determined size
131 m_fifo = new Fifo( fifoSize );
132
133 // now that framesPerPeriod is fixed initialize global BufferManager
134 BufferManager::init( m_framesPerPeriod );
135
136 m_outputBufferRead = std::make_unique<SampleFrame[]>(m_framesPerPeriod);
137 m_outputBufferWrite = std::make_unique<SampleFrame[]>(m_framesPerPeriod);
138
139
140 for( int i = 0; i < m_numWorkers+1; ++i )
141 {
142 auto wt = new AudioEngineWorkerThread(this);
143 if( i < m_numWorkers )
144 {
145 wt->start( QThread::TimeCriticalPriority );
146 }
147 m_workers.push_back( wt );
148 }
149}

154AudioEngine::~AudioEngine()
155{
156 for( int w = 0; w < m_numWorkers; ++w )
157 {
158 m_workers[w]->quit();
159 }
160
161 AudioEngineWorkerThread::startAndWaitForJobs();
162
163 for( int w = 0; w < m_numWorkers; ++w )
164 {
165 m_workers[w]->wait( 500 );
166 }
167
168 while( m_fifo->available() )
169 {
170 delete[] m_fifo->read();
171 }
172 delete m_fifo;
173
174 delete m_midiClient;
175 delete m_audioDev;
176
177
178 for (const auto& input : m_inputBuffer)
179 {
180 delete[] input;
181 }
182}

251void AudioEngine::pushInputFrames( SampleFrame* _ab, const f_cnt_t _frames )
252{
253 requestChangeInModel();
254
255 f_cnt_t frames = m_inputBufferFrames[ m_inputBufferWrite ];
256 auto size = m_inputBufferSize[m_inputBufferWrite];
257 SampleFrame* buf = m_inputBuffer[ m_inputBufferWrite ];
258
259 if( frames + _frames > size )
260 {
261 size = std::max(size * 2, frames + _frames);
262 auto ab = new SampleFrame[size];
263 memcpy( ab, buf, frames * sizeof( SampleFrame ) );
264 delete [] buf;
265
266 m_inputBufferSize[ m_inputBufferWrite ] = size;
267 m_inputBuffer[ m_inputBufferWrite ] = ab;
268
269 buf = ab;
270 }
271
272 memcpy( &buf[ frames ], _ab, _frames * sizeof( SampleFrame ) );
273 m_inputBufferFrames[ m_inputBufferWrite ] += _frames;
274
275 doneChangeInModel();
276}

280void AudioEngine::renderStageNoteSetup()
281{
282 AudioEngineProfiler::Probe profilerProbe(m_profiler, AudioEngineProfiler::DetailType::NoteSetup);
283
284 if( m_clearSignal )
285 {
286 m_clearSignal = false;
287 clearInternal();
288 }
289
290 // remove all play-handles that have to be deleted and delete
291 // them if they still exist...
292 // maybe this algorithm could be optimized...
293 ConstPlayHandleList::Iterator it_rem = m_playHandlesToRemove.begin();
294 while( it_rem != m_playHandlesToRemove.end() )
295 {
296 PlayHandleList::Iterator it = std::find( m_playHandles.begin(), m_playHandles.end(), *it_rem );
297
298 if( it != m_playHandles.end() )
299 {
300 (*it)->audioBusHandle()->removePlayHandle(*it);
301 if((*it)->type() == PlayHandle::Type::NotePlayHandle)
302 {
303 NotePlayHandleManager::release((NotePlayHandle*)*it);
304 }
305 else delete *it;
306 m_playHandles.erase(it);
307 }
308
309 it_rem = m_playHandlesToRemove.erase( it_rem );
310 }
311
312 swapBuffers();
313
314 // prepare master mix (clear internal buffers etc.)
315 Mixer * mixer = Engine::mixer();
316 mixer->prepareMasterMix();
317
318 // create play-handles for new notes, samples etc.
319 Engine::getSong()->processNextBuffer();
320
321 // add all play-handles that have to be added
322 for( LocklessListElement * e = m_newPlayHandles.popList(); e; )
323 {
324 m_playHandles += e->value;
325 LocklessListElement * next = e->next;
326 m_newPlayHandles.free( e );
327 e = next;
328 }
329}

343void AudioEngine::renderStageEffects()
344{
345 AudioEngineProfiler::Probe profilerProbe(m_profiler, AudioEngineProfiler::DetailType::Effects);
346
347 // STAGE 2: process effects of all instrument- and sampletracks
348 AudioEngineWorkerThread::fillJobQueue(m_audioBusHandles);
349 AudioEngineWorkerThread::startAndWaitForJobs();
350
351 // removed all play handles which are done
352 for( PlayHandleList::Iterator it = m_playHandles.begin();
353 it != m_playHandles.end(); )
354 {
355 if( ( *it )->affinityMatters() &&
356 ( *it )->affinity() != QThread::currentThread() )
357 {
358 ++it;
359 continue;
360 }
361 if( ( *it )->isFinished() )
362 {
363 (*it)->audioBusHandle()->removePlayHandle(*it);
364 if((*it)->type() == PlayHandle::Type::NotePlayHandle)
365 {
366 NotePlayHandleManager::release((NotePlayHandle*)*it);
367 }
368 else delete *it;
369 it = m_playHandles.erase(it);
370 }
371 else
372 {
373 ++it;
374 }
375 }
376}

380void AudioEngine::renderStageMix()
381{
382 AudioEngineProfiler::Probe profilerProbe(m_profiler, AudioEngineProfiler::DetailType::Mixing);
383
384 Mixer *mixer = Engine::mixer();
385 mixer->masterMix(m_outputBufferWrite.get());
386
387 MixHelpers::multiply(m_outputBufferWrite.get(), m_masterGain, m_framesPerPeriod);
388
389 emit nextAudioBuffer(m_outputBufferRead.get());
390
391 // and trigger LFOs
392 EnvelopeAndLfoParameters::instances()->trigger();
393 Controller::triggerFrameCounter();
394 AutomatableModel::incrementPeriodCounter();
395}

399const SampleFrame* AudioEngine::renderNextBuffer()
400{
401 const auto lock = std::lock_guard{m_changeMutex};
402
403 m_profiler.startPeriod();
404 s_renderingThread = true;
405
406 renderStageNoteSetup(); // STAGE 0: clear old play handles and buffers, setup new play handles
407 renderStageInstruments(); // STAGE 1: run and render all play handles
408 renderStageEffects(); // STAGE 2: process effects of all instrument- and sampletracks
409 renderStageMix(); // STAGE 3: do master mix in mixer
410
411 s_renderingThread = false;
412 m_profiler.finishPeriod(outputSampleRate(), m_framesPerPeriod);
413
414 return m_outputBufferRead.get();
415}

420void AudioEngine::swapBuffers()
421{
422 m_inputBufferWrite = (m_inputBufferWrite + 1) % 2;
423 m_inputBufferRead = (m_inputBufferRead + 1) % 2;
424 m_inputBufferFrames[m_inputBufferWrite] = 0;
425
426 std::swap(m_outputBufferRead, m_outputBufferWrite);
427 zeroSampleFrames(m_outputBufferWrite.get(), m_framesPerPeriod);
428}

543bool AudioEngine::addPlayHandle( PlayHandle* handle )
544{
545 // Only add play handles if we have the CPU capacity to process them.
546 // Instrument play handles are not added during playback, but when the
547 // associated instrument is created, so add those unconditionally.
548 if (handle->type() == PlayHandle::Type::InstrumentPlayHandle || !criticalXRuns())
549 {
550 m_newPlayHandles.push( handle );
551 handle->audioBusHandle()->addPlayHandle(handle);
552 return true;
553 }
554
555 if( handle->type() == PlayHandle::Type::NotePlayHandle )
556 {
557 NotePlayHandleManager::release( (NotePlayHandle*)handle );
558 }
559 else delete handle;
560
561 return false;
562}

565void AudioEngine::removePlayHandle(PlayHandle * ph)
566{
567 requestChangeInModel();
568 // check thread affinity as we must not delete play-handles
569 // which were created in a thread different than the audio engine thread
570 if (ph->affinityMatters() && ph->affinity() == QThread::currentThread())
571 {
572 ph->audioBusHandle()->removePlayHandle(ph);
573 bool removedFromList = false;
574 // Check m_newPlayHandles first because doing it the other way around
575 // creates a race condition
576 for( LocklessListElement * e = m_newPlayHandles.first(),
577 * ePrev = nullptr; e; ePrev = e, e = e->next )
578 {
579 if (e->value == ph)
580 {
581 if( ePrev )
582 {
583 ePrev->next = e->next;
584 }
585 else
586 {
587 m_newPlayHandles.setFirst( e->next );
588 }
589 m_newPlayHandles.free( e );
590 removedFromList = true;
591 break;
592 }
593 }
594 // Now check m_playHandles
595 PlayHandleList::Iterator it = std::find(m_playHandles.begin(), m_playHandles.end(), ph);
596 if (it != m_playHandles.end())
597 {
598 m_playHandles.erase(it);
599 removedFromList = true;
600 }
601 // Only deleting PlayHandles that were actually found in the list
602 // "fixes crash when previewing a preset under high load"
603 // (See tobydox's 2008 commit 4583e48)
604 if ( removedFromList )
605 {
606 if (ph->type() == PlayHandle::Type::NotePlayHandle)
607 {
608 NotePlayHandleManager::release(dynamic_cast<NotePlayHandle*>(ph));
609 }
610 else { delete ph; }
611 }
612 }
613 else
614 {
615 m_playHandlesToRemove.push_back(ph);
616 }
617 doneChangeInModel();
618}

1074void AudioEngine::fifoWriter::run()
1075{
1076 disable_denormals();
1077
1078 const fpp_t frames = m_audioEngine->framesPerPeriod();
1079 while( m_writing )
1080 {
1081 auto buffer = new SampleFrame[frames];
1082 const SampleFrame* b = m_audioEngine->renderNextBuffer();
1083 memcpy(buffer, b, frames * sizeof(SampleFrame));
1084 m_fifo->write(buffer);
1085 }
1086
1087 // Let audio backend stop processing
1088 m_fifo->write(nullptr);
1089 m_fifo->waitUntilRead();
1090}
