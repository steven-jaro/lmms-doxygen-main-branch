67AutomationClip::AutomationClip( const AutomationClip & _clip_to_copy ) :
68 Clip(_clip_to_copy),
69#if (QT_VERSION < QT_VERSION_CHECK(5,14,0))
70 m_clipMutex(QMutex::Recursive),
71#endif
72 m_autoTrack( _clip_to_copy.m_autoTrack ),
73 m_objects( _clip_to_copy.m_objects ),
74 m_tension( _clip_to_copy.m_tension ),
75 m_progressionType(_clip_to_copy.m_progressionType),
76 m_dragging(false),
77 m_isRecording(_clip_to_copy.m_isRecording),
78 m_lastRecordedValue(0)
79{
80 // Locks the mutex of the copied AutomationClip to make sure it
81 // doesn't change while it's being copied
82 QMutexLocker m(&_clip_to_copy.m_clipMutex);
83
84 for( timeMap::const_iterator it = _clip_to_copy.m_timeMap.begin();
85 it != _clip_to_copy.m_timeMap.end(); ++it )
86 {
87 // Copies the automation node (in/out values and in/out tangents)
88 m_timeMap[POS(it)] = it.value();
89 // Sets the node's clip to this one
90 m_timeMap[POS(it)].setClip(this);
91 }
92}

94bool AutomationClip::addObject( AutomatableModel * _obj, bool _search_dup )
95{
96 QMutexLocker m(&m_clipMutex);
97
98 if (_search_dup && std::find(m_objects.begin(), m_objects.end(), _obj) != m_objects.end())
99 {
100 return false;
101 }
102
103 // the automation track is unconnected and there is nothing in the track
104 if (m_objects.empty() && hasAutomation() == false)
105 {
106 // then initialize first value
107 putValue( TimePos(0), _obj->inverseScaledValue( _obj->value<float>() ), false );
108 }
109
110 m_objects.push_back(_obj);
111
112 connect( _obj, SIGNAL(destroyed(lmms::jo_id_t)),
113 this, SLOT(objectDestroyed(lmms::jo_id_t)),
114 Qt::DirectConnection );
115
116 emit dataChanged();
117
118 return true;
119}

181TimePos AutomationClip::timeMapLength() const
182{
183 QMutexLocker m(&m_clipMutex);
184
185 TimePos one_bar = TimePos(1, 0);
186 if (m_timeMap.isEmpty()) { return one_bar; }
187
188 timeMap::const_iterator it = m_timeMap.end();
189 auto last_tick = static_cast<tick_t>(POS(it - 1));
190 // if last_tick is 0 (single item at tick 0)
191 // return length as a whole bar to prevent disappearing Clip
192 if (last_tick == 0) { return one_bar; }
193
194 return TimePos(last_tick);
195}

200void AutomationClip::updateLength()
201{
202 // Technically it only matters if the clip has been resized from the right, but this
203 // checks if it has been resized from either direction.
204 if (getAutoResize())
205 {
206 // Using 1 bar as the min length for an un-resized clip.
207 // This does not prevent the user from resizing the clip to be less than a bar later on.
208 changeLength(std::max(TimePos::ticksPerBar(), static_cast<tick_t>(timeMapLength())));
209 setStartTimeOffset(TimePos(0));
210 }

216/**
217 * @brief Puts an automation node on the timeMap with the given value.
218 * The inValue and outValue of the created node will be the same.
219 * @param TimePos time to add the node to
220 * @param Float inValue and outValue of the node
221 * @param Boolean True to quantize the position (defaults to true)
222 * @param Boolean True to ignore unquantized surrounding nodes (defaults to true)
223 * @return TimePos of the recently added automation node
224 */
225TimePos AutomationClip::putValue(
226 const TimePos & time,
227 const float value,
228 const bool quantPos,
229 const bool ignoreSurroundingPoints
230)
231{
232 QMutexLocker m(&m_clipMutex);
233
234 cleanObjects();
235
236 TimePos newTime = quantPos ? Note::quantized(time, quantization()) : time;
237 newTime = std::max(TimePos(0), newTime);
238
239 // Create a node or replace the existing one on newTime
240 m_timeMap[newTime] = AutomationNode(this, value, newTime);
241
242 timeMap::iterator it = m_timeMap.find(newTime);
243
244 // Remove control points that are covered by the new points
245 // quantization value. Control Key to override
246 if (!ignoreSurroundingPoints)
247 {
248 // We need to check that to avoid removing nodes from
249 // newTime + 1 to newTime (removing the node we are adding)
250 if (quantization() > 1)
251 {
252 // Remove nodes between the quantization points, them not
253 // being included
254 removeNodes(newTime + 1, newTime + quantization() - 1);
255 }
256 }
257 if (it != m_timeMap.begin()) { --it; }
258 generateTangents(it, 3);
259
260 updateLength();
261
262 emit dataChanged();
263
264 return newTime;
265}

326void AutomationClip::removeNode(const TimePos & time)
327{
328 QMutexLocker m(&m_clipMutex);
329
330 cleanObjects();
331
332 m_timeMap.remove( time );
333 timeMap::iterator it = m_timeMap.lowerBound(time);
334 if( it != m_timeMap.begin() )
335 {
336 --it;
337 }
338 generateTangents(it, 3);
339
340 updateLength();
341
342 emit dataChanged();
343}

348/**
349 * @brief Removes all automation nodes between the given ticks
350 * @param Int first tick of the range
351 * @param Int second tick of the range
352 */
353void AutomationClip::removeNodes(const int tick0, const int tick1)
354{
355 if (tick0 == tick1)
356 {
357 removeNode(TimePos(tick0));
358 return;
359 }
360
361 auto start = TimePos(std::min(tick0, tick1));
362 auto end = TimePos(std::max(tick0, tick1));
363
364 // Make a list of TimePos with nodes to be removed
365 // because we can't simply remove the nodes from
366 // the timeMap while we are iterating it.
367 std::vector<TimePos> nodesToRemove;
368
369 for (auto it = m_timeMap.lowerBound(start), endIt = m_timeMap.upperBound(end); it != endIt; ++it)
370 {
371 nodesToRemove.push_back(POS(it));
372 }
373
374 for (auto node: nodesToRemove)
375 {
376 removeNode(node);
377 }
378}

383/**
384 * @brief Resets the outValues of all automation nodes between the given ticks
385 * @param Int first tick of the range
386 * @param Int second tick of the range
387 */
388void AutomationClip::resetNodes(const int tick0, const int tick1)
389{
390 if (tick0 == tick1)
391 {
392 auto it = m_timeMap.find(TimePos(tick0));
393 if (it != m_timeMap.end()) { it.value().resetOutValue(); }
394 return;
395 }
396
397 auto start = TimePos(std::min(tick0, tick1));
398 auto end = TimePos(std::max(tick0, tick1));
399
400 for (auto it = m_timeMap.lowerBound(start), endIt = m_timeMap.upperBound(end); it != endIt; ++it)
401 {
402 it.value().resetOutValue();
403 }
404}

409void AutomationClip::resetTangents(const int tick0, const int tick1)
410{
411 if (tick0 == tick1)
412 {
413 auto it = m_timeMap.find(TimePos(tick0));
414 if (it != m_timeMap.end())
415 {
416 it.value().setLockedTangents(false);
417 generateTangents(it, 1);
418 }
419 return;
420 }
421
422 TimePos start = TimePos(std::min(tick0, tick1));
423 TimePos end = TimePos(std::max(tick0, tick1));
424
425 for (auto it = m_timeMap.lowerBound(start), endIt = m_timeMap.upperBound(end); it != endIt; ++it)
426 {
427 it.value().setLockedTangents(false);
428 generateTangents(it, 1);
429 }

436{

453/**
454 * @brief Set the position of the point that is being dragged.
455 * Calling this function will also automatically set m_dragging to true.
456 * When applyDragValue() is called, m_dragging is set back to false.
457 * @param TimePos of the node being dragged
458 * @param Float with the value to assign to the point being dragged
459 * @param Boolean. True to snip x position
460 * @param Boolean. True to ignore unquantized surrounding nodes
461 * @return TimePos with current time of the dragged value
462 */
463TimePos AutomationClip::setDragValue(
464 const TimePos & time,
465 const float value,
466 const bool quantPos,
467 const bool controlKey
468)
469{
470 QMutexLocker m(&m_clipMutex);
471
472 if (m_dragging == false)
473 {
474 TimePos newTime = quantPos ? Note::quantized(time, quantization()) : time;
475
476 // We will keep the same outValue only if it's different from the
477 // inValue
478 m_dragKeepOutValue = false;
479
480 // We will set the tangents back to what they were if the node had
481 // its tangents locked
482 m_dragLockedTan = false;
483
484 // Check if we already have a node on the position we are dragging
485 // and if we do, store the outValue so the discrete jump can be kept
486 // and information about the tangents
487 timeMap::iterator it = m_timeMap.find(newTime);
488 if (it != m_timeMap.end())
489 {
490 // If we don't have a discrete jump, the outValue will be the
491 // same as the inValue
492 if (OFFSET(it) != 0)
493 {
494 m_dragKeepOutValue = true;
495 m_dragOutValue = OUTVAL(it);
496 }
497 // For the tangents, we will only keep them if the tangents were
498 // locked
499 if (LOCKEDTAN(it))
500 {
501 m_dragLockedTan = true;
502 m_dragInTan = INTAN(it);
503 m_dragOutTan = OUTTAN(it);
504 }
505 }
506
507 this->removeNode(newTime);
508 m_oldTimeMap = m_timeMap;
509 m_dragging = true;
510 }
511
512 //Restore to the state before it the point were being dragged
513 m_timeMap = m_oldTimeMap;
514
515 generateTangents();
516
517 TimePos returnedPos;
518
519 if (m_dragKeepOutValue)
520 {
521 returnedPos = this->putValues(time, value, m_dragOutValue, quantPos, controlKey);
522 }
523 else
524 {
525 returnedPos = this->putValue(time, value, quantPos, controlKey);
526 }
527
528 // Set the tangents on the newly created node if they were locked
529 // before dragging
530 if (m_dragLockedTan)
531 {
532 timeMap::iterator it = m_timeMap.find(returnedPos);
533 if (it != m_timeMap.end())
534 {
535 it.value().setInTangent(m_dragInTan);
536 it.value().setOutTangent(m_dragOutTan);
537 it.value().setLockedTangents(true);
538 }
539 }
540
541 return returnedPos;
542}

560float AutomationClip::valueAt( const TimePos & _time ) const
561{
562 QMutexLocker m(&m_clipMutex);
563
564 if( m_timeMap.isEmpty() )
565 {
566 return 0;
567 }
568
569 // If we have a node at that time, just return its value
570 if (m_timeMap.contains(_time))
571 {
572 // When the time is exactly the node's time, we want the inValue
573 return m_timeMap[_time].getInValue();
574 }
575
576 // lowerBound returns next value with equal or greater key. Since we already
577 // checked if the key contains a node, we know the returned node has a greater
578 // key than _time. Therefore we take the previous element to calculate the current value
579 timeMap::const_iterator v = m_timeMap.lowerBound(_time);
580
581 if( v == m_timeMap.begin() )
582 {
583 return 0;
584 }
585 if( v == m_timeMap.end() )
586 {
587 // When the time is after the last node, we want the outValue of it
588 return OUTVAL(v - 1);
589 }
590
591 return valueAt(v - 1, _time - POS(v - 1));
592}

597// This method will get the value at an offset from a node, so we use the outValue of
598// that node and the inValue of the next node for the calculations.
599float AutomationClip::valueAt( timeMap::const_iterator v, int offset ) const
600{
601 QMutexLocker m(&m_clipMutex);
602
603 // We never use it with offset 0, but doesn't hurt to return a correct
604 // value if we do
605 if (offset == 0) { return INVAL(v); }
606
607 if (m_progressionType == ProgressionType::Discrete)
608 {
609 return OUTVAL(v);
610 }
611 else if( m_progressionType == ProgressionType::Linear )
612 {
613 float slope =
614 (INVAL(v + 1) - OUTVAL(v))
615 / (POS(v + 1) - POS(v));
616
617 return OUTVAL(v) + offset * slope;
618 }
619 else /* ProgressionType::CubicHermite */
620 {
621 // Implements a Cubic Hermite spline as explained at:
622 // http://en.wikipedia.org/wiki/Cubic_Hermite_spline#Unit_interval_.280.2C_1.29
623 //
624 // Note that we are not interpolating a 2 dimensional point over
625 // time as the article describes. We are interpolating a single
626 // value: y. To make this work we map the values of x that this
627 // segment spans to values of t for t = 0.0 -> 1.0 and scale the
628 // tangents _m1 and _m2
629 int numValues = (POS(v + 1) - POS(v));
630 float t = (float) offset / (float) numValues;
631 float m1 = OUTTAN(v) * numValues * m_tension;
632 float m2 = INTAN(v + 1) * numValues * m_tension;
633
634 auto t2 = t * t, t3 = t2 * t;
635 return (2 * t3 - 3 * t2 + 1) * OUTVAL(v)
636 + (t3 - 2 * t2 + t) * m1
637 + (-2 * t3 + 3 * t2) * INVAL(v + 1)
638 + (t3 - t2) * m2;
639 }
640}

645float *AutomationClip::valuesAfter( const TimePos & _time ) const
646{
647 QMutexLocker m(&m_clipMutex);
648
649 timeMap::const_iterator v = m_timeMap.lowerBound(_time);
650 if( v == m_timeMap.end() || (v+1) == m_timeMap.end() )
651 {
652 return nullptr;
653 }
654
655 int numValues = POS(v + 1) - POS(v);
656 auto ret = new float[numValues];
657
658 for( int i = 0; i < numValues; i++ )
659 {
660 ret[i] = valueAt( v, i );
661 }
662
663 return ret;
664}

669void AutomationClip::flipY(int min, int max)
670{
671 QMutexLocker m(&m_clipMutex);
672
673 bool changedTimeMap = false;
674
675 for (auto it = m_timeMap.begin(); it != m_timeMap.end(); ++it)
676 {
677 // Get distance from IN/OUT values to max value
678 float inValDist = max - INVAL(it);
679 float outValDist = max - OUTVAL(it);
680
681 // To flip, that will be the new distance between
682 // the IN/OUT values and the min value
683 it.value().setInValue(min + inValDist);
684 it.value().setOutValue(min + outValDist);
685
686 changedTimeMap = true;
687 }
688
689 if (changedTimeMap)
690 {
691 generateTangents();
692 emit dataChanged();
693 }
694}

706
707void AutomationClip::flipX(int start, int end)
708{
709 QMutexLocker m(&m_clipMutex);
710
711 timeMap::const_iterator firstIterator = m_timeMap.lowerBound(0);
712
713 if (firstIterator == m_timeMap.end()) { return; }
714
715 if (start == -1 && end == -1) { start = 0; end = length() - startTimeOffset(); }
716 else if (!(end >= 0 && start >= 0 && end > start)) { return; }
717
718 // Temporary map where we will store the flipped version
719 // of our clip
720 timeMap tempMap;
721
722 for (auto it = m_timeMap.begin(); it != m_timeMap.end(); ++it)
723 {
724 if (POS(it) < start || POS(it) > end)
725 {
726 tempMap[POS(it)] = *it;
727 }
728 else
729 {
730 // If the first node in the clip is not at the start, it can break things when clipping, so
731 // we have to set its in value to 0.
732 if (it == firstIterator && POS(firstIterator) > 0)
733 {
734 tempMap[end - (POS(it) - start)] = AutomationNode(this, OUTVAL(it), 0, end - (POS(it) - start));
735 }
736 else
737 {
738 tempMap[end - (POS(it) - start)] = AutomationNode(this, OUTVAL(it), INVAL(it), end - (POS(it) - start));
739 }
740 }
741 }
742
743 if (m_timeMap.contains(start) && m_timeMap.contains(end))
744 {
745 tempMap[start] = AutomationNode(this, m_timeMap[start].getInValue(), m_timeMap[end].getInValue(), start);
746 tempMap[end] = AutomationNode(this, m_timeMap[start].getOutValue(), m_timeMap[end].getOutValue(), end);
747 }
748 else if (m_timeMap.contains(start))
749 {
750 tempMap[start] = AutomationNode(this, m_timeMap[start].getInValue(), valueAt(end), start);
751 tempMap[end] = AutomationNode(this, m_timeMap[start].getOutValue(), valueAt(end), end);
752 }
753 else if (m_timeMap.contains(end))
754 {
755 tempMap[start] = AutomationNode(this, valueAt(start), m_timeMap[end].getInValue(), start);
756 tempMap[end] = AutomationNode(this, valueAt(start), m_timeMap[end].getOutValue(), end);
757 }
758 else
759 {
760 tempMap[start] = AutomationNode(this, valueAt(start), valueAt(end), start);
761 tempMap[end] = AutomationNode(this, valueAt(start), valueAt(end), end);
762 }
763
764 m_timeMap.clear();
765
766 m_timeMap = tempMap;
767
768 cleanObjects();
769
770 generateTangents();
771 emit dataChanged();
772}

777void AutomationClip::saveSettings( QDomDocument & _doc, QDomElement & _this )
778{
779 QMutexLocker m(&m_clipMutex);
780
781 _this.setAttribute( "pos", startPosition() );
782 _this.setAttribute( "len", length() );
783 _this.setAttribute( "name", name() );
784 _this.setAttribute( "prog", QString::number( static_cast<int>(progressionType()) ) );
785 _this.setAttribute( "tens", QString::number( getTension() ) );
786 _this.setAttribute( "mute", QString::number( isMuted() ) );
787 _this.setAttribute("off", startTimeOffset());
788 _this.setAttribute("autoresize", QString::number(getAutoResize()));
789
790 if (const auto& c = color())
791 {
792 _this.setAttribute("color", c->name());
793 }
794
795 for( timeMap::const_iterator it = m_timeMap.begin();
796 it != m_timeMap.end(); ++it )

823{

825

827 // the project file. So if any node doesn't have tangent information

871 shouldGenerateTangents = true;
872 }
873 }

881 {

883 }
884

886 if( len <= 0 )
887 {
888 // TODO: Handle with an upgrade method
889 updateLength();

1134
1135void AutomationClip::generateTangents()
1136{
1137 generateTangents(m_timeMap.begin(), m_timeMap.size());
1138}
1139
1140
1141
1142
1143// We have two tangents, one for the left side of the node and one for the right side
1144// of the node (in case we have discrete value jumps in the middle of a curve).
1145// If the inValue and outValue of a node are the same, consequently the inTangent and
1146// outTangent values of the node will be the same too.
1147void AutomationClip::generateTangents(timeMap::iterator it, int numToGenerate)
1148{
1149 QMutexLocker m(&m_clipMutex);
1150
1151 for (int i = 0; i < numToGenerate && it != m_timeMap.end(); ++i, ++it)
1152 {
1153 // Skip the node if it has locked tangents (were manually edited)
1154 if (LOCKEDTAN(it))
1155 {
1156 continue;
1157 }
1158
1159 if (it + 1 == m_timeMap.end())
1160 {
1161 // Previously, the last value's tangent was always set to 0. That logic was kept for both tangents
1162 // of the last node
1163 it.value().setInTangent(0);
1164 it.value().setOutTangent(0);
1165 }
1166 else if (it == m_timeMap.begin())
1167 {
1168 // On the first node there's no curve behind it, so we will only calculate the outTangent
1169 // and inTangent will be set to 0.
1170 float tangent = (INVAL(it + 1) - OUTVAL(it)) / (POS(it + 1) - POS(it));
1171 it.value().setInTangent(0);
1172 it.value().setOutTangent(tangent);
1173 }
1174 else
1175 {
1176 // When we are in a node that is in the middle of two other nodes, we need to check if we
1177 // have a discrete jump at this node. If we do not, then we can calculate the tangents normally.
1178 // If we do have a discrete jump, then we have to calculate the tangents differently for each side
1179 // of the curve.
1180 // TODO: This behavior means that a very small difference between the inValue and outValue can
1181 // result in a big change in the curve. In the future, allowing the user to manually adjust
1182 // the tangents would be better.
1183 if (OFFSET(it) == 0)
1184 {
1185 float inTangent = (INVAL(it + 1) - OUTVAL(it - 1)) / (POS(it + 1) - POS(it - 1));
1186 it.value().setInTangent(inTangent);
1187 // inTangent == outTangent in this case
1188 it.value().setOutTangent(inTangent);
1189 }
1190 else
1191 {
1192 // Calculate the left side of the curve
1193 float inTangent = (INVAL(it) - OUTVAL(it - 1)) / (POS(it) - POS(it - 1));
1194 // Calculate the right side of the curve
1195 float outTangent = (INVAL(it + 1) - OUTVAL(it)) / (POS(it + 1) - POS(it));
1196 it.value().setInTangent(inTangent);
1197 it.value().setOutTangent(outTangent);
1198 }
1199 }
1200 }
