60template<typename F>
61auto retryWhileInterrupted(F&& function, std::invoke_result_t<F> error = -1)
62 noexcept(std::is_nothrow_invocable_v<F>) -> auto
63{
64 while (true)
65 {
66 const auto result = function();
67 if (result != error || errno != EINTR) { return result; }
68 }
69}

75class SystemSemaphoreImpl
76{
77public:
78 SystemSemaphoreImpl(const std::string& key, unsigned int value) :
79 m_key{"/" + key},
80 m_sem{retryWhileInterrupted([&]() noexcept {
81 return sem_open(m_key.c_str(), O_CREAT | O_EXCL, 0600, value);
82 }, SEM_FAILED)}
83 {
84 if (m_sem == SEM_FAILED) { throwSystemError("SystemSemaphoreImpl: sem_open() failed"); }
85 m_ownedSemaphore.reset(m_key.c_str());
86 }
87
88 explicit SystemSemaphoreImpl(const std::string& key) :
89 m_key{"/" + key},
90 m_sem{retryWhileInterrupted([&]() noexcept {
91 return sem_open(m_key.c_str(), 0);
92 }, SEM_FAILED)}
93 {
94 if (m_sem == SEM_FAILED) { throwSystemError("SystemSemaphoreImpl: sem_open() failed"); }
95 }
96
97 ~SystemSemaphoreImpl()
98 {
99 // We can't use `UniqueNullableResource` for `m_sem`, as the null value
100 // (`SEM_FAILED`) is not a constant expression on macOS (it's defined as
101 // `(sem_t*) -1`), so can't be used as a template parameter.
102 sem_close(m_sem);
103 }
104
105 auto acquire() noexcept -> bool
106 {
107 return retryWhileInterrupted([&]() noexcept {
108 return sem_wait(m_sem);
109 }) == 0;
110 }
111
112 auto release() noexcept -> bool { return sem_post(m_sem) == 0; }
113
114private:
115 std::string m_key;
116 sem_t* m_sem;
117 UniqueSemaphore m_ownedSemaphore;
118};

145 m_sem{OpenSemaphoreA(SEMAPHORE_MODIFY_STATE | SYNCHRONIZE, false, key.c_str())}
146 {
147 if (!m_sem) { throwSystemError("SystemSemaphoreImpl: OpenSemaphoreA failed"); }
148 }

150 auto acquire() noexcept -> bool { return WaitForSingleObject(m_sem.get(), INFINITE) == WAIT_OBJECT_0; }
151 auto release() noexcept -> bool { return ReleaseSemaphore(m_sem.get(), 1, nullptr); }
