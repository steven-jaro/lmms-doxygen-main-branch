103void MidiClientRaw::parseData( const unsigned char c )
104{
105 /*********************************************************************/
106 /* 'Process' system real-time messages */
107 /*********************************************************************/
108 /* There are not too many real-time messages that are of interest here.
109 * They can occur anywhere, even in the middle of a noteon message!
110 * Real-time range: 0xF8 .. 0xFF
111 * Note: Real-time does not affect (running) status.
112 */
113 if( c >= 0xF8 )
114 {
115 if( c == MidiSystemReset )
116 {
117 m_midiParseData.m_midiEvent.setType( MidiSystemReset );
118 m_midiParseData.m_status = 0;
119 processParsedEvent();
120 }
121 return;
122 }
123
124 /*********************************************************************/
125 /* 'Process' system common messages (again, just skip them) */
126 /*********************************************************************/
127 /* There are no system common messages that are of interest here.
128 * System common range: 0xF0 .. 0xF7
129 */
130 if( c > 0xF0 )
131 {
132 /* MIDI spec say: To ignore a non-real-time message, just discard all
133 * data up to the next status byte. And our parser will ignore data
134 * that is received without a valid status.
135 * Note: system common cancels running status. */
136 m_midiParseData.m_status = 0;
137 return;
138 }
139
140 /*********************************************************************/
141 /* Process voice category messages: */
142 /*********************************************************************/
143 /* Now that we have handled realtime and system common messages, only
144 * voice messages are left.
145 * Only a status byte has bit # 7 set.
146 * So no matter the status of the parser (in case we have lost sync),
147 * as soon as a byte >= 0x80 comes in, we are dealing with a status byte
148 * and start a new event.
149 */
150 if( c & 0x80 )
151 {
152 m_midiParseData.m_channel = c & 0x0F;
153 m_midiParseData.m_status = c & 0xF0;
154 /* The event consumes x bytes of data...
155 (subtract 1 for the status byte) */
156 m_midiParseData.m_bytesTotal = eventLength( m_midiParseData.m_status ) - 1;
157 /* of which we have read 0 at this time. */
158 m_midiParseData.m_bytes = 0;
159 return;
160 }
161
162 /*********************************************************************/
163 /* Process data */
164 /*********************************************************************/
165 /* If we made it this far, then the received char belongs to the data
166 * of the last event. */
167 if( m_midiParseData.m_status == 0 )
168 {
169 /* We are not interested in the event currently received.
170 Discard the data. */
171 return;
172 }
173
174 /* Store the first couple of bytes */
175 if( m_midiParseData.m_bytes < RAW_MIDI_PARSE_BUF_SIZE )
176 {
177 m_midiParseData.m_buffer[m_midiParseData.m_bytes] = c;
178 }
179 ++m_midiParseData.m_bytes;
180
181 /* Do we still need more data to get this event complete? */
182 if( m_midiParseData.m_bytes < m_midiParseData.m_bytesTotal )
183 {
184 return;
185 }
186
187 /*********************************************************************/
188 /* Send the event */
189 /*********************************************************************/
190 /* The event is ready-to-go. About 'running status':
191 *
192 * The MIDI protocol has a built-in compression mechanism. If several
193 * similar events are sent in-a-row, for example note-ons, then the
194 * event type is only sent once. For this case, the last event type
195 * (status) is remembered.
196 * We simply keep the status as it is, just reset the parameter counter.
197 * If another status byte comes in, it will overwrite the status.
198 */
199 m_midiParseData.m_midiEvent.setType(static_cast<MidiEventTypes>(m_midiParseData.m_status));
200 m_midiParseData.m_midiEvent.setChannel(m_midiParseData.m_channel);
201 m_midiParseData.m_bytes = 0; /* Related to running status! */
202 switch (m_midiParseData.m_midiEvent.type())
203 {
204 case MidiNoteOff:
205 case MidiNoteOn:
206 case MidiKeyPressure:
207 m_midiParseData.m_midiEvent.setKey(m_midiParseData.m_buffer[0]);
208 m_midiParseData.m_midiEvent.setVelocity(m_midiParseData.m_buffer[1]);
209 break;
210
211 case MidiChannelPressure:
212 case MidiProgramChange:
213 m_midiParseData.m_midiEvent.setParam(0, m_midiParseData.m_buffer[0]);
214 break;
215
216 case MidiControlChange:
217 m_midiParseData.m_midiEvent.setControllerNumber(m_midiParseData.m_buffer[0]);
218 m_midiParseData.m_midiEvent.setControllerValue( m_midiParseData.m_buffer[1]);
219 break;
220
221 case MidiPitchBend:
222 // Pitch-bend is transmitted with 14-bit precision.
223 // Note: '|' does here the same as '+' (no common bits),
224 // but might be faster
225 m_midiParseData.m_midiEvent.setPitchBend((m_midiParseData.m_buffer[1] * 128) | m_midiParseData.m_buffer[0]);
226 break;
227
228 default:
229 // Unlikely
230 return;
231 }
232
233 processParsedEvent();
234}

250void MidiClientRaw::processOutEvent(const MidiEvent& event, const TimePos&, const MidiPort* port)
251{
252 // TODO: also evaluate _time and queue event if necessary
253 switch (event.type())
254 {
255 case MidiNoteOn:
256 case MidiNoteOff:
257 case MidiKeyPressure:
258 sendByte(event.type() | event.channel());
259 sendByte(event.key());
260 sendByte(event.velocity());
261 break;
262
263 default:
264 qWarning("MidiClientRaw: unhandled MIDI-event %d\n", (int)event.type());
265 break;
266 }
267}

274// Taken from Nagano Daisuke's USB-MIDI driver
275static const auto REMAINS_F0F6 = std::array<unsigned char, 7>
276{
277 0, /* 0xF0 */
278 2, /* 0XF1 */
279 3, /* 0XF2 */
280 2, /* 0XF3 */
281 2, /* 0XF4 (Undefined by MIDI Spec, and subject to change) */
282 2, /* 0XF5 (Undefined by MIDI Spec, and subject to change) */
283 1 /* 0XF6 */
284} ;
285
286static const auto REMAINS_80E0 = std::array<unsigned char, 7>
287{
288 3, /* 0x8X Note Off */
289 3, /* 0x9X Note On */
290 3, /* 0xAX Poly-key pressure */
291 3, /* 0xBX Control Change */
292 2, /* 0xCX Program Change */
293 2, /* 0xDX Channel pressure */
294 3 /* 0xEX PitchBend Change */
295} ;

299// Returns the length of the MIDI message starting with _event.
300// Taken from Nagano Daisuke's USB-MIDI driver
301int MidiClientRaw::eventLength( const unsigned char event )
302{
303 if ( event < 0xF0 )
304 {
305 return REMAINS_80E0[( ( event - 0x80 ) >> 4 ) & 0x0F];
306 }
307 else if ( event < 0xF7 )
308 {
309 return REMAINS_F0F6[event - 0xF0];
310 }
311 return 1;
312}
