117bool Sample::play(SampleFrame* dst, PlaybackState* state, size_t numFrames, Loop loop, double ratio) const
118{
119 state->m_frameIndex = std::max<int>(m_startFrame, state->m_frameIndex);
120
121 const auto sampleRateRatio = static_cast<double>(Engine::audioEngine()->outputSampleRate()) / m_buffer->sampleRate();
122 const auto freqRatio = frequency() / DefaultBaseFreq;
123 state->m_resampler.setRatio(sampleRateRatio * freqRatio * ratio);
124
125 // TODO: These kind of playback pipelines/graphs are repeated within other parts of the codebase that work with
126 // audio samples. We should find a way to unify this but the right abstraction is not so clear yet.
127 while (numFrames > 0)
128 {
129 if (state->m_bufferView.empty())
130 {
131 const auto rendered = render(state->m_buffer.data(), state->m_buffer.size(), state, loop);
132 state->m_bufferView = {state->m_buffer.data(), rendered};
133 }
134
135 const auto [inputFramesUsed, outputFramesGenerated] = state->m_resampler.process(
136 {&state->m_bufferView.data()[0][0], 2, state->m_bufferView.size()}, {&dst[0][0], 2, numFrames});
137
138 if (inputFramesUsed == 0 && outputFramesGenerated == 0)
139 {
140 std::fill_n(dst, numFrames, SampleFrame{});
141 break;
142 }
143
144 state->m_bufferView = state->m_bufferView.subspan(inputFramesUsed);
145 dst += outputFramesGenerated;
146 numFrames -= outputFramesGenerated;
147 }
148
149 return numFrames < Engine::audioEngine()->framesPerPeriod();
150}

152f_cnt_t Sample::render(SampleFrame* dst, f_cnt_t size, PlaybackState* state, Loop loop) const
153{
154 for (f_cnt_t frame = 0; frame < size; ++frame)
155 {
156 switch (loop)
157 {
158 case Loop::Off:
159 if (state->m_frameIndex < 0 || state->m_frameIndex >= m_endFrame) { return frame; }
160 break;
161 case Loop::On:
162 if (state->m_frameIndex < m_loopStartFrame && state->m_backwards)
163 {
164 state->m_frameIndex = m_loopEndFrame - 1;
165 }
166 else if (state->m_frameIndex >= m_loopEndFrame) { state->m_frameIndex = m_loopStartFrame; }
167 break;
168 case Loop::PingPong:
169 if (state->m_frameIndex < m_loopStartFrame && state->m_backwards)
170 {
171 state->m_frameIndex = m_loopStartFrame;
172 state->m_backwards = false;
173 }
174 else if (state->m_frameIndex >= m_loopEndFrame)
175 {
176 state->m_frameIndex = m_loopEndFrame - 1;
177 state->m_backwards = true;
178 }
179 break;
180 default:
181 break;
182 }
183
184 const auto value
185 = m_buffer->data()[m_reversed ? m_buffer->size() - state->m_frameIndex - 1 : state->m_frameIndex]
186 * m_amplification;
187 dst[frame] = value;
188 state->m_backwards ? --state->m_frameIndex : ++state->m_frameIndex;
189 }
190
191 return size;
192}

194auto Sample::sampleDuration() const -> std::chrono::milliseconds
195{
196 const auto numFrames = endFrame() - startFrame();
197 const auto duration = numFrames / static_cast<float>(m_buffer->sampleRate()) * 1000;
198 return std::chrono::milliseconds{static_cast<int>(duration)};
