31std::size_t ringbuffer_common_t::calc_size(std::size_t sz)
32{
33 std::size_t power_of_two;
34 for (power_of_two = 1;
35 (static_cast<std::size_t>(1) << power_of_two) < sz; power_of_two++) ;
36 return 1 << power_of_two;
37}

47bool ringbuffer_base::munlock(const void* const buf, std::size_t each)
48{
49 // we return true iff the buffer is unlocked at return time
50#ifdef USE_MLOCK
51 if (!mlocked) {
52 return true;
53 }
54 else {
55 // mlocked is true now
56 if (buf && !::munlock(buf, size * each)) {
57 mlocked = false;
58 }
59 return !mlocked;
60 }
61#else
62 (void)buf;
63 (void)each;
64 return !mlocked;
65#endif
66}

111void ringbuffer_base::init_variables_for_write(std::size_t cnt,
112 std::size_t& w, std::size_t& to_write,
113 std::size_t& n1, std::size_t& n2)
114{
115 w = w_ptr.load(); // TODO: relaxed?
116 std::size_t rl = readers_left.load(); // TODO: consume?
117
118 // size calculations
119 std::size_t free_cnt = write_space_preloaded(w, rl);
120
121 to_write = cnt > free_cnt ? free_cnt : cnt;
122 const std::size_t cnt2 = w + to_write;
123
124 if (cnt2 > size) {
125 n1 = size - w;
126 n2 = cnt2 & size_mask;
127 } else {
128 n1 = to_write;
129 n2 = 0;
130 }
131
132 // reset reader_left
133 // TODO: inefficient xor:
134 if((w ^ ((w + to_write) & size_mask)) & (size >> 1)) // msb flipped
135 {
136 if(rl)
137 throw "impossible";
138 readers_left.store(num_readers);
139 }
140
141}
