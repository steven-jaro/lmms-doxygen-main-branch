198 // Start an asynchronous send. The data being sent must be valid for the
199 // lifetime of the asynchronous operation.
200 template <typename ConstBufferSequence, typename Handler, typename IoExecutor>
201 void async_send(base_implementation_type& impl,
202 const ConstBufferSequence& buffers, socket_base::message_flags flags,
203 Handler& handler, const IoExecutor& io_ex)
204 {
205 bool is_continuation =
206 asio_handler_cont_helpers::is_continuation(handler);
207
208 // Allocate and construct an operation to wrap the handler.
209 typedef winrt_socket_send_op<ConstBufferSequence, Handler, IoExecutor> op;
210 typename op::ptr p = { asio::detail::addressof(handler),
211 op::ptr::allocate(handler), 0 };
212 p.p = new (p.v) op(buffers, handler, io_ex);
213
214 ASIO_HANDLER_CREATION((scheduler_.context(),
215 *p.p, "socket", &impl, 0, "async_send"));
216
217 start_send_op(impl,
218 buffer_sequence_adapter<asio::const_buffer,
219 ConstBufferSequence>::first(buffers),
220 flags, p.p, is_continuation);
221 p.v = p.p = 0;
222 }
