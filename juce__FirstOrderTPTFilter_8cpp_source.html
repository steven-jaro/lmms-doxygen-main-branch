81template <typename SampleType>
82SampleType FirstOrderTPTFilter<SampleType>::processSample (int channel, SampleType inputValue)
83{
84 auto& s = s1[(size_t) channel];
85
86 auto v = G * (inputValue - s);
87 auto y = v + s;
88 s = y + v;
89
90 switch (filterType)
91 {
92 case Type::lowpass: return y;
93 case Type::highpass: return inputValue - y;
94 case Type::allpass: return 2 * y - inputValue;
95 default: break;
96 }
97
98 jassertfalse;
99 return y;
100}

110template <typename SampleType>
111void FirstOrderTPTFilter<SampleType>::update()
112{
113 auto g = SampleType (std::tan (juce::MathConstants<double>::pi * cutoffFrequency / sampleRate));
114 G = g / (1 + g);
115}
