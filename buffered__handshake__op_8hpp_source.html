57private:
58 template <typename Iterator>
59 engine::want process(engine& eng,
60 asio::error_code& ec,
61 std::size_t& bytes_transferred,
62 Iterator begin, Iterator end) const
63 {
64 Iterator iter = begin;
65 std::size_t accumulated_size = 0;
66
67 for (;;)
68 {
69 engine::want want = eng.handshake(type_, ec);
70 if (want != engine::want_input_and_retry
71 || bytes_transferred == total_buffer_size_)
72 return want;
73
74 // Find the next buffer piece to be fed to the engine.
75 while (iter != end)
76 {
77 const_buffer buffer(*iter);
78
79 // Skip over any buffers which have already been consumed by the engine.
80 if (bytes_transferred >= accumulated_size + buffer.size())
81 {
82 accumulated_size += buffer.size();
83 ++iter;
84 continue;
85 }
86
87 // The current buffer may have been partially consumed by the engine on
88 // a previous iteration. If so, adjust the buffer to point to the
89 // unused portion.
90 if (bytes_transferred > accumulated_size)
91 buffer = buffer + (bytes_transferred - accumulated_size);
92
93 // Pass the buffer to the engine, and update the bytes transferred to
94 // reflect the total number of bytes consumed so far.
95 bytes_transferred += buffer.size();
96 buffer = eng.put_input(buffer);
97 bytes_transferred -= buffer.size();
98 break;
99 }
100 }
101 }
102
103 stream_base::handshake_type type_;
104 ConstBufferSequence buffers_;
105 std::size_t total_buffer_size_;
106};
