40int writeVarLength(uint32_t val, uint8_t *buffer)
41{
42 /*
43 Accept an input, and write a MIDI-compatible variable length stream
44
45 The MIDI format is a little strange, and makes use of so-called variable
46 length quantities. These quantities are a stream of bytes. If the most
47 significant bit is 1, then more bytes follow. If it is zero, then the
48 byte in question is the last in the stream
49 */
50 int size = 0;
51 uint8_t little_endian[4];
52 uint8_t result = val & 0x7F;
53 little_endian[size++] = result;
54 val = val >> 7;
55 while (val > 0)
56 {
57 result = val & 0x7F;
58 result = result | 0x80;
59 little_endian[size++] = result;
60 val = val >> 7;
61 }
62 for (int i=0; i<size; i++)
63 {
64 buffer[i] = little_endian[size-i-1];
65 }
66
67 return size;
68}

70int writeBigEndian4(uint32_t val, uint8_t *buf)
71{
72 buf[0] = val >> 24;
73 buf[1] = val >> 16 & 0xff;
74 buf[2] = val >> 8 & 0xff;
75 buf[3] = val & 0xff;
76 return 4;
