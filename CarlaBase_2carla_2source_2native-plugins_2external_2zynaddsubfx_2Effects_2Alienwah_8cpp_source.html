83//Apply the effect
84void Alienwah::out(const Stereo<float *> &smp)
85{
86 float lfol, lfor; //Left/Right LFOs
87 complex<float> clfol, clfor;
92 lfo.effectlfoout(&lfol, &lfor);
93 lfol *= depth * PI * 2.0f;
94 lfor *= depth * PI * 2.0f;
95 clfol = complex<float>(cosf(lfol + phase) * fb, sinf(lfol + phase) * fb); //rework
96 clfor = complex<float>(cosf(lfor + phase) * fb, sinf(lfor + phase) * fb); //rework
97
98 for(int i = 0; i < buffersize; ++i) {
99 float x = ((float) i) / buffersize_f;
100 float x1 = 1.0f - x;
101 //left
102 complex<float> tmp = clfol * x + oldclfol * x1;
103
104 complex<float> out = tmp * oldl[oldk];
105 out += (1 - fabs(fb)) * smp.l[i] * pangainL;
106
107 oldl[oldk] = out;
108 float l = out.real() * 10.0f * (fb + 0.1f);
109
110 //right
111 tmp = clfor * x + oldclfor * x1;
112
113 out = tmp * oldr[oldk];
114 out += (1 - fabs(fb)) * smp.r[i] * pangainR;
115
116 oldr[oldk] = out;
117 float r = out.real() * 10.0f * (fb + 0.1f);
118
119
120 if(++oldk >= Pdelay)
121 oldk = 0;
122 //LRcross
123 efxoutl[i] = l * (1.0f - lrcross) + r * lrcross;
124 efxoutr[i] = r * (1.0f - lrcross) + l * lrcross;
125 }
126
127 oldclfol = clfol;
128 oldclfor = clfor;
129}

149void Alienwah::setfb(unsigned char _Pfb)
150{
151 Pfb = _Pfb;
152 fb = fabs((Pfb - 64.0f) / 64.1f);
153 fb = sqrtf(fb);
154 if(fb < 0.4f)
155 fb = 0.4f;
156 if(Pfb < 64)
157 fb = -fb;
158}

186void Alienwah::setpreset(unsigned char npreset)
187{
188 const int PRESET_SIZE = 11;
189 const int NUM_PRESETS = 4;
190 unsigned char presets[NUM_PRESETS][PRESET_SIZE] = {
191 //AlienWah1
192 {127, 64, 70, 0, 0, 62, 60, 105, 25, 0, 64},
193 //AlienWah2
194 {127, 64, 73, 106, 0, 101, 60, 105, 17, 0, 64},
195 //AlienWah3
196 {127, 64, 63, 0, 1, 100, 112, 105, 31, 0, 42},
197 //AlienWah4
198 {93, 64, 25, 0, 1, 66, 101, 11, 47, 0, 86}
199 };
200
201 if(npreset >= NUM_PRESETS)
202 npreset = NUM_PRESETS - 1;
203 for(int n = 0; n < PRESET_SIZE; ++n)
204 changepar(n, presets[npreset][n]);
205 if(insertion == 0)
206 changepar(0, presets[npreset][0] / 2); //lower the volume if this is system effect
207 Ppreset = npreset;
208}
