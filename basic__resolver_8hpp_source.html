951 void operator()(ASIO_MOVE_ARG(ResolveHandler) handler,

953 {

956 ASIO_RESOLVE_HANDLER_CHECK(
957 ResolveHandler, handler, results_type) type_check;
958
959 asio::detail::non_const_lvalue<ResolveHandler> handler2(handler);
960 self->impl_.get_service().async_resolve(
961 self->impl_.get_implementation(), q, handler2.value,
962 self->impl_.get_implementation_executor());
