257float EnvelopeParams::dt(char val)
258{
259 return (powf(2.0f, val / 127.0f * 12.0f) - 1.0f) * 10.0f; //miliseconds
260}
261
262char EnvelopeParams::inv_dt(float val)
263{
264 int ival = roundf(logf(val/10.0f + 1.0f)/logf(2.0f) * 127.0f/12.0f);
265 return limit(ival, 0, 127);
266}

432struct version_fixer_t
433{
434 const bool mismatch;
435public:
436 int operator()(int input) const
437 {
438 return (mismatch)
439 // The errors occured when calling env_dB2rap. Let f be the
440 // conversion function for mode 2 (see Envelope.cpp), then we
441 // load values with (let "o" be the function composition symbol):
442 // f^{-1} o (env_dB2rap^{-1}) o dB2rap o f
443 // from the xml file. This results in the following formula:
444 ? roundf(127.0f * (0.5f *
445 log10f( 0.01f + 0.99f *
446 powf(100, input/127.0f - 1))
447 + 1))
448 : input;
449 }
450 version_fixer_t(const version_type& fileversion, int env_mode) :
451 mismatch(fileversion < version_type(2,4,4) &&
452 (env_mode == 2)) {}
453};

455void EnvelopeParams::getfromXML(XMLwrapper& xml)
456{
457 Pfreemode = xml.getparbool("free_mode", Pfreemode);
458 Penvpoints = xml.getpar127("env_points", Penvpoints);
459 Penvsustain = xml.getpar127("env_sustain", Penvsustain);
460 Penvstretch = xml.getpar127("env_stretch", Penvstretch);
461 Pforcedrelease = xml.getparbool("forced_release", Pforcedrelease);
462 Plinearenvelope = xml.getparbool("linear_envelope", Plinearenvelope);
463
464 version_fixer_t version_fix(xml.fileversion(), Envmode);
465
466 PA_dt = xml.getpar127("A_dt", PA_dt);
467 PD_dt = xml.getpar127("D_dt", PD_dt);
468 PR_dt = xml.getpar127("R_dt", PR_dt);
469 PA_val = version_fix(xml.getpar127("A_val", PA_val));
470 PD_val = version_fix(xml.getpar127("D_val", PD_val));
471 PS_val = version_fix(xml.getpar127("S_val", PS_val));
472 PR_val = version_fix(xml.getpar127("R_val", PR_val));
473
474 for(int i = 0; i < Penvpoints; ++i) {
475 if(xml.enterbranch("POINT", i) == 0)
476 continue;
477 if(i != 0)
478 Penvdt[i] = xml.getpar127("dt", Penvdt[i]);
479 Penvval[i] = version_fix(xml.getpar127("val", Penvval[i]));
480 xml.exitbranch();
481 }
482
483 if(!Pfreemode)
484 converttofree();
485}
